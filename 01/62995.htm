'Code starts below this line

'-----------------------------------------------
'Find out if an executable is 32 bits or 64 bits.
'18.11.2015 FNL
'-----------------------------------------------
If WScript.Arguments.Count = 0 Then
    MsgBox "Missing file name"
    WScript.Quit
End If

Set oFSO = CreateObject("Scripting.Filesystemobject")
Set oFile = ofso.OpenTextFile(WScript.Arguments(0))
sData = oFile.Read(10000)
p = InStr(sData, "PE")
sMode = "Cannot tell"
If p > 0 Then
    sData = Mid(sData, p)
    q = InStr(sData, "d†")
    r = InStr(sData, "L")
    Do
        If q + r = 0 Then Exit Do
        If q > 0 And r = 0 Then sMode = "64 bits": Exit Do
        If r > 0 And q = 0 Then sMode = "32 bits": Exit Do
        If q < r Then sMode = "64 bits" Else sMode = "32 bits"
        Exit Do
