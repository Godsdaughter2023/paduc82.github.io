    Function StopService(Computer, ServiceName, Wait)
    On Error Resume Next
    Dim cimv2, oService, Result
    'Get the WMI administration object
    Set cimv2 = GetObject("winmgmts:{impersonationLevel=impersonate}!\\" & _
    Computer & "\root\cimv2")
    'Get the service object
    Set oService = cimv2.Get("Win32_Service.Name='" & ServiceName & "'")
    'Check base properties
    If Not oService.Started Then
    ' the service is Not started
    Progress ServiceName & " service is not running"
    exit Function
    End If

    If Not oService.AcceptStop Then
    ' the service does Not accept stop command
    Progress ServiceName & " does not accept STOP command"
    exit Function
    End If

    'Stop the service
    Result = oService.StopService
    If Result <> 0 Then
    Progress "Stop: " & ServiceName & " Error: " & Result
    exit function
    End If

    Do While oService.Started And Wait
    'get the current service state
    Set oService = cimv2.Get("Win32_Service.Name='" & ServiceName & "'")
    Wscript.Sleep 200
    Loop
    Progress "Service Stopped: " & ServiceName
    Wscript.Sleep 2000
    End Function
