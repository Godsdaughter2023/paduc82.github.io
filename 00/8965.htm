Hi Pronobis, you can do that with a bit of VBA - I have assumed you want the users to save the file to its original location with the same name - is that correct?

In Excel open that document

Press Alt + F11 to open the VB Editor

At the top left, double click ThisWordbook

Paste the following code into the window on the right

You will need to change the file name in that macro

Private Sub Workbook_BeforeSave(ByVal SaveAsUI As Boolean, Cancel As Boolean)
If SaveAsUI = True Then
Application.EnableEvents = False
Cancel = True
Me.SaveAs "C:\aFilderName\aFile.xlsx"
End If
End Sub

bill mosca say:

Based on the patterns you gave me, this should work.


Public Function GetAcctNum(strFile As String) As Variant
    Dim varAccountNum As Variant
    Dim bolLenOk As Boolean
    
    'Initialize Boolean
    bolLenOk = False
    
    If InStr(strFile, "_") > 0 Then
        'Get characters after underscore.
        varAccountNum = Mid(strFile, InStr(strFile, "_") + 1)
    ElseIf InStr(strFile, "[") > 0 Then
        'Get characters after left bracket.
        varAccountNum = Mid(strFile, InStr(strFile, "[") + 1)
        'trim trailing bracket
        If Right(varAccountNum, 1) = "]" Then
            varAccountNum = Left(varAccountNum, Len(varAccountNum) - 1)
        End If
    End If
    
    'Check initial length of string
    If Len(varAccountNum) >= 10 Then
        'Remove characters up and including any hyphens.
        If InStr(varAccountNum, "-") Then
            varAccountNum = Mid(varAccountNum, InStr(varAccountNum, "-") + 1)
        End If
    
    End If
    
    'Check if account is still at least 10 chars long.
    bolLenOk = Len(varAccountNum) >= 10
    
    If bolLenOk = True Then
        GetAcctNum = varAccountNum
    End If
    
End Function
