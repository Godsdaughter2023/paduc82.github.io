doug robbins - mvp office apps & services (word) say:

The only way that you could do it would be by running a macro.

The following code will apply that formatting to the percentages in the column in which the selection is located.

Dim crange As Range
With Selection.Columns(1)
    For i = 1 To .Cells.Count
        Set crange = .Cells(i).Range
        crange.End = crange.End - 2
        Select Case Val(crange.Text)
            Case Is < 100
                crange.End = crange.End + 1
                crange.Font.ColorIndex = wdRed
            Case Is > 104.94
                crange.End = crange.End + 1
                crange.Font.ColorIndex = wdBlue
            Case Else
                crange.End = crange.End + 1
                crange.Font.ColorIndex = wdGreen
        End Select
        crange.Font.Bold = True
    Next i
End With

If your table contains a header row, change the i = 1 to i = 2 so that the first row is excluded from the operation.

-- Hope this helps.

Private Sub Worksheet_BeforeDoubleClick(ByVal Target As Range, _
  Cancel As Boolean)
   
    '******************************************************************************
    'Could use Alternative to code between Asterisk lines is to simply test _
     Target for Intersect in a specified range or test the column of Target
   
    'Test that Double Clicked cell is valid cell for this procedure
    'before going any further.
    'Test if Target contains validation
    On Error GoTo errHandler
    Application.EnableEvents = False    'Required before test for validation cell
   
    If Intersect(Target, Me.UsedRange.SpecialCells(xlCellTypeAllValidation)) Is Nothing Then
        'If nothing then not a validation cell so exit the sub
        GoTo errHandler
    Else
        'If IS validation cell then test for Type 3 validation
        If Target.Validation.Type <> 3 Then GoTo errHandler    'Not Type 3 then Exit sub
    End If
    '*****************************************************************************
   
    Cancel = True   'Only cancel Edit Mode after decision to continue process
   
    Dim str As String
    Dim cboTemp As OLEObject
    Dim ws As Worksheet
    Set ws = ActiveSheet
   
    Set cboTemp = ws.OLEObjects("RateCombo")
   
    With cboTemp
        'clear and hide the combo box
        .ListFillRange = ""
        .LinkedCell = ""
        .Visible = False
    End With
   
    'get the data validation formula
    str = Target.Validation.Formula1
    'Remove the INDIRECT portion of the formula
    str = Replace(str, "INDIRECT(", "")     'Remove INDIRECT and opening parenthesis
    str = Left(str, Len(str) - 1)           'Remove last closing parenthesis
    str = Evaluate(str)                     'Evaluate the formula to return named range
   
    With cboTemp
      'show the combobox with the list
      .Visible = True
      .Left = Target.Left
      .Top = Target.Top
      .Width = Target.Width + 15
      .Height = Target.Height
      .ListFillRange = str
      .LinkedCell = Target.Address
      .Activate
      .Object.DropDown
    End With
   
errHandler:
    If Err.Number <> 0 Then
        MsgBox "Error occurred in Module Sheet1 Private Sub Worksheet_BeforeDoubleClick." _
            & vbCrLf & "Contact the Administrator of this application."
    End If
    On Error GoTo 0
    Application.EnableEvents = True
   
End Sub

Private Sub Worksheet_SelectionChange(ByVal Target As Range)
    'Test for decision whether or not to continue process as early as possible in the sub
