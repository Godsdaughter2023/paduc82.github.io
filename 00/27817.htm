Help prevent changes to a final version of a file

The Mark as Final command helps you communicate that you are sharing a completed version of a file. It also helps prevent reviewers or readers from making

Before you share an electronic copy of a Microsoft Office document with other people, you can use the Mark as Final command to make the file read-only and help prevent changes to the file. Ouch...Access 97.

The biggest reason to use the FileDialog object is that it works in a 32bit and 64bit environment. I have tried to pointed out many times, rewriting all your apps to be 64bit aware now before it blows up with the next customer.

 

Here is my code for selecting a file.

 

Public Function SelectFileDialog() As Variant
    Dim oFileDialog As Object               'Late binding
    Dim varFile As Variant
    Dim aryFileList As Variant ' Used as an Array
    Dim iPtr As Integer
   
    'Constances for type of File Dialog.
    'Const msoFileDialogOpen = 1
    'Const msoFileDialogSaveAs = 2
    Const msoFileDialogFilePicker = 3
    'Const msoFileDialogFolderPicker = 4
   
    Set oFileDialog = Application.FileDialog(msoFileDialogFilePicker)
   
    With oFileDialog

        ' Allow user to make multiple selections in dialog box '
        .AllowMultiSelect = True
       
        ' Set the title of the dialog box. '
        .Title = "Please select one or more files"
               
        ' Set initial directory
        'If not set uses users last directory
        .InitialFileName = Application.CurrentProject.Path & "\"
       
        ' Clear out the current filters, and add our own.'
        .Filters.Clear
        .Filters.Add "All Access Databases", "*.accdb;*.accde;*.mdb;*.mde"
        .Filters.Add "Access 2007 Databases", "*.accdb;*.accde"
        .Filters.Add "Access 97-2003 Databases", "*.mdb;*.mde"
        .Filters.Add "All Files", "*.*"
       
       
        ' Set default value in list
        '.FilterIndex = 3 'Would make 3 item the default, but listed as added if use drop-down
        '.FilterIndex = 1 'Default value if not coded
       
        ' Show the dialog box. If the .Show method returns True, the '
        ' user picked at least one file. If the .Show method returns '
        ' False, the user clicked Cancel. '
        If .Show = True Then
            'MsgBox "Selected file count: " & .SelectedItems.Count

            'Loop through each file selected
            ReDim aryFileList(.SelectedItems.Count)
            iPtr = 0
            For Each varFile In .SelectedItems
                aryFileList(iPtr) = varFile
                iPtr = iPtr + 1
                'MsgBox "Filename returned: " & varFile
            Next
            SelectFileDialog = aryFileList
        Else
            'MsgBox "You clicked Cancel in the file dialog box."
            SelectFileDialog = "Canceled"
        End If
    End With
End Function
