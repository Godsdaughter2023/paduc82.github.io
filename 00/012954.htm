So, the first new sheet would have the name of Sun, Dec 30, 2012 and in cell K3 would be 12/30/2012 and then the next new sheet would have the name Sun, Jan 06, 2013 and in cell K3 of this sheet would be 1/6/2013 and so on.

    Sub CreateNameSheets()
       ' by Dave Peterson
       ' List sheetnames required in col E in a sheet: List
       ' Sub will copy sheets based on the sheet named as: Template
       ' and name the sheets accordingly

        Dim TemplateWks As Worksheet
        Dim ListWks As Worksheet
        Dim ListRng As Range
        Dim mycell As Range

        Set TemplateWks = Worksheets("Sep Master")
        Set ListWks = Worksheets("list")
            With ListWks
            Set ListRng = .Range("E4", .Cells(.Rows.Count, "E").End(xlUp))
        End With

        For Each mycell In ListRng.Cells
            TemplateWks.Copy after:=Worksheets(Worksheets.Count)
            On Error Resume Next
            ActiveSheet.Name = Format(mycell.Value, "ddd, mmm dd, yyyy")
            If Err.Number <> 0 Then
                MsgBox "Please fix: " & ActiveSheet.Name
                Err.Clear
            End If
            On Error GoTo 0
        Next mycell

    End Sub

     

    Any help is appreciated,

    Dale
