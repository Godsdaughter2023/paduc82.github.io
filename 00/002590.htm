Sub ImportData2()
    Dim wbData As Workbook
    Dim wsSym As Worksheet
    Dim rng As Range, c As Range, rngToCopy As Range
    Dim lnLastCol As Long, lnLastRow As Long
   
    On Error Resume Next
    Set wbData = Workbooks("data.xlsm") 'other workbook must be open!
    On Error GoTo 0
    If wbData Is Nothing Then MsgBox "'data.xlsm' must be open", vbCritical: Exit Sub
       
     'Import the data:
    With ThisWorkbook.Worksheets("DATA")
       
         'Delete the old data:
        lnLastCol = .Cells(9, .Columns.Count).End(xlToLeft).Column
        lnLastRow = .Cells(.Rows.Count, "A").End(xlUp).Row
        If lnLastRow > 10 Then .Range(.Cells(11, "A"), .Cells(lnLastRow, lnLastCol)).ClearContents
       
         'Copy the dates from the 1st sheet in DATA.xlsm:
        Set wsSym = wbData.Worksheets(1)
        lnLastRow = wsSym.Cells(wsSym.Rows.Count, "A").End(xlUp).Row
        wsSym.Range("A11:A" & lnLastRow).Copy Destination:=.Range("A11")
       
        Set rng = .Range(.Cells(9, "C"), .Cells(9, lnLastCol))
        For Each c In rng                       'Step across each cell in row 9
            If c.Value <> vbNullString Then
                Set wsSym = Nothing
                On Error Resume Next
                Set wsSym = wbData.Worksheets(c.Value)  'Get the right sheet
                On Error GoTo 0
                If Not wsSym Is Nothing Then
                    lnLastRow = wsSym.Cells(wsSym.Rows.Count, "F").End(xlUp).Row
                    Set rngToCopy = wsSym.Range("F11:F" & lnLastRow)
                    rngToCopy.Copy Destination:=c.Offset(2, 0)
                Else
                    MsgBox "Could not find a sheet called '" & c.Value & "'!", vbExclamation
                End If
            End If
        Next c
       
         'Get range for re-formatting:
        lnLastRow = .Cells(.Rows.Count, "A").End(xlUp).Row
        Set rng = .Range(.Cells(11, "A"), .Cells(lnLastRow, lnLastCol))
    End With
   
     'Re-format the pasted cells:
    With rng
        .HorizontalAlignment = xlCenter
        .VerticalAlignment = xlBottom
        .WrapText = False
        .Orientation = 0
        .AddIndent = False
        .IndentLevel = 0
        .ShrinkToFit = False
        .ReadingOrder = xlContext
        .MergeCells = False
       
        .Borders(xlDiagonalDown).LineStyle = xlNone
        .Borders(xlDiagonalUp).LineStyle = xlNone
        With .Borders
            .LineStyle = xlContinuous
            .ColorIndex = 0
            .TintAndShade = 0
            .Weight = xlThin
        End With
    End With
End Sub
