Option Explicit
Function FindSerialNums(s As String) As String
    Dim rSrc As Range, c As Range
    Dim FirstAddress As String
    Dim sTemp As String
With Worksheets("Sheet2")
    Set rSrc = .Range("A1", .Cells(.Rows.Count, "A").End(xlUp)).Resize(columnsize:=2)
End With
    With rSrc.Columns(2)
        Set c = .Find(what:=s, after:=.Cells(1), LookIn:=xlValues, _
                    lookat:=xlPart, searchdirection:=xlNext, MatchCase:=False)
        If Not c Is Nothing Then
            FirstAddress = c.Address
            sTemp = c.Offset(columnoffset:=-1)
            Do
                Set c = .Find(what:=s, after:=c)
                If Not c.Address = FirstAddress Then
                    sTemp = sTemp & ", " & c.Offset(columnoffset:=-1)
                End If
            Loop Until c.Address = FirstAddress
        End If
    End With
FindSerialNums = sTemp
