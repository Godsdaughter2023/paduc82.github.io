steve rindsberg say:

Give this a try. 

ON COPIES OF YOUR ORIGINAL PRESENTATIONS.

ONLY!

PLEASE!

There. You've been warned. 

Sub MoveTheText()
' Open this presentation and the other presentation
' you want to move this presentation's text into
' No other presentations should be open.
' This should be the ACTIVE presentation when you start the macro!

    Dim oPresA As Presentation
    Dim oPresB As Presentation
    Dim oSl As Slide
    Dim oSh As Shape
    Dim x As Long
    
    Set oPresA = ActivePresentation
    
    ' Get a reference to the other presentation
    If oPresA.Name = Presentations(1).Name Then
        Set oPresB = Presentations(2)
    Else
        Set oPresB = Presentations(1)
    End If

    ' Delete the shapes with text in presentation B
    For Each oSl In oPresB.Slides
        For x = oSl.Shapes.Count To 1 Step -1
            With oSl.Shapes(x)
                If .HasTextFrame Then
                    If .TextFrame.HasText Then
                        .Delete
                    End If
                End If
            End With
        Next
    Next

    ' And because some of the text boxes might be text placeholders,
    ' you might want to do this to:
    For Each oSl In oPresB.Slides
        For x = oSl.Shapes.Count To 1 Step -1
            With oSl.Shapes(x)
                If .Type = msoPlaceholder Then
                    Select Case .PlaceholderFormat.Type
                        Case ppPlaceholderCenterTitle
                            .Delete
                        Case ppPlaceholderBody
                            .Delete
                        Case ppplaceholdertite
                            .Delete
                        Case ppPlaceholderObject
                            .Delete
                        ' add other types here if desired
                    End Select
                End If
            End With
        Next
    Next
    

    ' now copy the shapes over from Presentation A
    For Each oSl In oPresA.Slides
        For Each oSh In oSl.Shapes
            If oSh.HasTextFrame Then
                If oSh.TextFrame.HasText Then
                    oSh.Copy
                    oPresB.Slides(oSl.SlideIndex).Shapes.Paste
                End If
            End If
        Next
    Next
End Sub
