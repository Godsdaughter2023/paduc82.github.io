teylyn say:

Hello,

you need four conditional formats, one for each color.  Select the cell. In the screenshot I selected cell A2. The formula will use the currently selected cell. 

Create a conditional format with a formula, using

=A2-TODAY()>=30

Format to blue. Note that there are no $ signs in the cell reference. A2 is the current cell and using no $ signs in the formula will look at the current cell when you apply the format to other cells.

Create three more formats with these formulas:

=A2-TODAY()<=29  -- format to yellow

=A2-TODAY()<=14  -- format to orange

=A2-TODAY()<=5 -- format to red.

In the Rules Manager the rules should be in this order:

You can use the "Applies To" fields in this dialog to select the range for which the format should be applied, or you can copy the formatted cell and use Paste Special > Formats to copy the conditional format to other cells.

' Throwaway Macro created by Doug Robbins to "preserve" bookmarks during a MailMerge

'

Dim abm As Bookmark, bmrange As Range, i As Long, Result As Document, j As Long, k As Long, linkrange As Range, linktarget As String

Dim Source As Document

Set Source = ActiveDocument

i = 1

For j = 1 To Source.MailMerge.DataSource.RecordCount

    For Each abm In ActiveDocument.Range.Bookmarks

        System.PrivateProfileString("c:\bookmarks.txt", "bookmarkNames", "bookmark" & i) = abm.Name & Format(j)

        i = i + 1

    Next

Next j

For Each abm In ActiveDocument.Range.Bookmarks

    abm.Range.InsertBefore "#"

    abm.Range.InsertAfter "#"

Next

With ActiveDocument.MailMerge

    .Destination = wdSendToNewDocument

    .Execute

End With

Set Result = ActiveDocument

k = 1

Selection.HomeKey wdStory

Selection.Find.ClearFormatting

With Selection.Find

    Do While .Execute(FindText:="#*#", MatchWildcards:=True, Wrap:=wdFindContinue, Forward:=True) = True

        Set bmrange = Selection.Range

        bmrange.Characters(bmrange.Characters.Count).Delete

        bmrange.Characters(1).Delete

        Result.Bookmarks.Add System.PrivateProfileString("c:\bookmarks.txt", "bookmarkNames", "bookmark" & k), bmrange

        k = k + 1

    Loop

End With

For i = 1 To Result.Hyperlinks.Count

    linktarget = Result.Hyperlinks(i).SubAddress

    Set linkrange = Result.Hyperlinks(i).Range

    linkrange.Select

    linktarget = linktarget & Format(Selection.Information(wdActiveEndSectionNumber))

    Result.Hyperlinks.Add Result.Hyperlinks(i).Range, "", linktarget

Next i

   

Source.Activate

Selection.Find.ClearFormatting

With Selection.Find

    Do While .Execute(FindText:="#*#", MatchWildcards:=True, Wrap:=wdFindContinue, Forward:=True) = True

        Set bmrange = Selection.Range

        bmrange.Characters(bmrange.Characters.Count).Delete

        bmrange.Characters(1).Delete

    Loop

End With

macropod say:
Bookmarks always get deleted from a mailmerge output document. Even if that were not so, given that a document can only contain a single instance of a given bookmark, only a single instance of a given bookmark would survive a merge. Instead of using bookmarks, you may find it better to insert plain text or rich text content controls at the relevant points in the mailmerge main document; those do survive a merge.

Sub Test()
  Dim S As String
  Dim Lines() As String
  Dim i As Long, j As Long, f As Long, t As Long, m As Long
  Dim InComment As Boolean
 
  S = _
    "'***********************************" & vbCrLf & _
    "'*  This is a test comment block  *" & vbCrLf & _
    "'*  that I might put in my code    *" & vbCrLf & _
    "'***********************************" & vbCrLf & _
    vbCrLf & _
    "Sub Test()" & vbCrLf & _
    "'*  This is a test " & vbCrLf & _
    "End Sub" & vbCrLf

  Debug.Print "Before:"
  Debug.Print
  Debug.Print S
  Debug.Print

  Lines = Split(S, vbCrLf)
  For i = 0 To UBound(Lines)
    S = Trim$(Lines(i))
    If Left$(S, 1) = "'" Then
      'We found a comment
      If Not InComment Then f = i
      InComment = True
      t = i
    Else
      If InComment Then
        'We have a comment from line f to t

        'Step 1: Remove the existing formatting
        For j = f To t
          'Remove the leading and trailing signs
          Lines(j) = Trim$(Lines(j))
          If Left$(Lines(j), 2) = "'*" Then
            Lines(j) = Mid$(Lines(j), 3)
          End If
          If Right$(Lines(j), 1) = "*" Then
            Lines(j) = Left$(Lines(j), Len(Lines(j)) - 1)
          End If
          'Remove leading and trailing star lines
          If j = f Or j = t Then
            If Lines(j) = String(Len(Lines(j)), "*") Then
              Lines(j) = ""
            End If
          End If
        Next

        'Step 2: Calculate the length
        For j = f To t
          Lines(j) = Trim$(Lines(j))
          If Len(Lines(j)) > m Then m = Len(Lines(j))
        Next

        'Step 3: Add the formatting
        For j = f To t
          'Format this line
          S = "'*  " & Space(m) & "  *"
          Mid(S, 5, Len(Lines(j))) = Lines(j)
          'Special case for first and last line
          If j = f Or j = t Then
            'Is there a place for a star line?
            If Len(Lines(j)) = 0 Then
              Mid(S, 2, Len(S)) = String(Len(S) - 1, "*")
            Else
              'Add it to this line
              If j = f Then S = "'" & String(m + 6, "*") & vbCrLf & S
              If j = f Then S = S & vbCrLf & "'" & String(m + 6, "*")
            End If
          End If
          Lines(j) = S
        Next
      End If
      InComment = False
    End If
  Next

  S = Join(Lines, vbCrLf)
  Debug.Print "After:"
  Debug.Print
  Debug.Print S
  Debug.Print
End Sub
