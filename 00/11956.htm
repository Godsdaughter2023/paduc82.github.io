Dim lngLastRow As Long

Dim bNonZeroFound As Boolean

tom ogilvy say:

the minimum needed as arguments for filecopy is the src file name and the destination file name.  You column B doesn't have filenames.  So if I assume you want it to have the same name, then I have modified the code to extract the filename from the source string. 

Sub CopyFiles()

'On Error Resume Next

For Each Cell In Range("A2:A7000")
  sSrc = Cell.Text
  sDest = Cell.Offset(0, 1).Text
  If Right(sDest, 1) <> "\" Then sDest = sDest & "\"
  iloc = InStrRev(sSrc, "\")
  fName = Trim(Mid(sSrc, iloc + 1, 255))
  FileCopy sSrc, sDest & fName
  'Kill cell.text

Next

End Sub
