petrib say: The 6070 needs a CA-42 cable according to this table (and presumably does not work with a CA-53 cable at all):
http://europe.nokia.com/A4144914

The 6280 works with these cables DKU-2, CA-53, CA-70.

That explains why your 6280 works and your 6070 doesn't.

The difference is that the CA-42 does USB to serial conversions for the phone, when the phone doesn't have real USB support at all.

DKU-2, CA-53 and CA-70 are "straight" USB cables for phones with real USB support built-on.

bill manville say:

Not clear if there is a header row on sheet1 row 1, I am assuming so
And I am assuming that all rows relevant to the cut/paste have "NTS" in column A, not just the first [ambiguity in your description]
And I am assuming no blank rows within the table
And I am assuming that both sheets are in the same workbook and that the workbook is active when you run the macro
If any of the above are not the case, come back and say so and the solution can probably be adapted.

I would filter the table on sheet1 for "NTS" in column A
Then I would select the visible rows and copy the information from them to sheet "NTS new"
And finally delete the rows that were copied and remove the filter.
Thus:

Sub MoveNTSRows()
  MoveRows "NTS"
End Sub

Sub MoveRows(Unit As String)
  With Sheets("Sheet1").Range("A1").CurrentRegion
    .AutoFilter  ' switch filtering on or clear previous filter
    .AutoFilter 1, Unit  ' filter so that only rows with "NTS" or whichever unit is passed in, are visible
    If .Columns(1).SpecialCells(xlVisible).Count=1 Then
      ' none found
      MsgBox "No " & Unit & " rows found"
    Else
      With .Offset(1).Resize(.Rows.Count-1).SpecialCells(xlVisible)
        ' now dealing with the data only, not the headings
        .Copy
        ' paste the data to the row after the last used row in new sheet for the unit
        Sheets(Unit & " New").Cells(Rows.Count,1).End(xlUp).Offset(1).PasteSpecial xlAll
        .EntireRow.Delete   ' remove the copied rows
      End With
    End If
    .AutoFilter ' remove the filter
  End With
End Sub

Option Explicit

Sub Test()
  Dim R As Range, C As Range, All As Range
  Dim Value
  'Find all occurences of NTS in column A
  Set R = FindAll(Columns(1), "NTS")
  'Visit each found cell
  For Each C In R
    'Get the value from column B
    Value = C.Offset(0, 1).Value
    'Is the value in our range?
    If Value >= 20 And Value <= 200 Then
      'Yes, remember this cell
      If All Is Nothing Then Set All = C Else Set All = Union(All, C)
    End If
  Next
  'Something found?
  If Not All Is Nothing Then
    'Yes, find the last row and return next row (which is empty)
    Set R = Range("A" & Rows.Count).End(xlUp).Offset(1, 0)
    'Copy the values to this position
    All.EntireRow.Copy R
    'Find the new end
    Set C = Range("A" & Rows.Count).End(xlUp)
    'Fill all this cells with the new string
    Range(R, C) = "NTS new"
    'Now it's save to delete the old one's
    All.EntireRow.Delete
  End If
End Sub

Function FindAll(ByVal Where As Range, ByVal What, _
    Optional ByVal After As Variant, _
    Optional ByVal LookIn As XlFindLookIn = xlValues, _
    Optional ByVal LookAt As XlLookAt = xlWhole, _
    Optional ByVal SearchOrder As XlSearchOrder = xlByRows, _
    Optional ByVal SearchDirection As XlSearchDirection = xlNext, _
    Optional MatchCase As Boolean = False) As Range
  'Find all occurances of What in Where
  Const Max = 375 'FastUnion-Bound
  Dim FirstAddress As String
  Dim C As Range, R As Range, Temp As Range
  Dim TempIsNothing As Boolean
  If Where Is Nothing Then Exit Function
  If SearchDirection = xlNext And IsMissing(After) Then
    'Set After to the last cell in Where to return the first cell in Where in front if it match What
    Set C = Where.Areas(Where.Areas.Count)
    Set After = C.Cells(C.Cells.Count)
  End If
  Set C = Where.Find(What, After, LookIn, LookAt, SearchOrder, SearchDirection, MatchCase)
  If Not C Is Nothing Then
    FirstAddress = C.Address
    TempIsNothing = True
    Do
      If TempIsNothing Then
        Set Temp = C
        TempIsNothing = False
      Else
        If Temp.Count < Max Then
          Set Temp = Union(Temp, C)
        Else
          If FindAll Is Nothing Then
            Set FindAll = Temp
          Else
            Set FindAll = Union(FindAll, Temp)
          End If
          Set Temp = C
        End If
      End If
      If SearchDirection = xlNext Then
        Set C = Where.FindNext(C)
      Else
        Set C = Where.FindPrevious(C)
      End If
    Loop Until FirstAddress = C.Address
    If FindAll Is Nothing Then
      Set FindAll = Temp
    Else
      Set FindAll = Union(FindAll, Temp)
    End If
  End If
End Function
