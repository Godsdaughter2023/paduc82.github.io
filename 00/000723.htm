Sub ChooseFilter()
'
' ChooseFilter Macro
'
' Keyboard Shortcut: Ctrl+f
Dim wks As Worksheet
Dim pvt As PivotTable

'On Error Resume Next
For Each wks In Worksheets
For Each pvt In wks.PivotTables

  ' even though your are looping through you sheets, you don't change which sheet

    ActiveSheet.PivotTables("PivotTable1").PivotFields("ship_code").ClearAllFilters
ActiveSheet.PivotTables("PivotTable1").PivotFields("ship_code").CurrentPage = _
"LI"

Sub ChooseFilter()
'
' ChooseFilter Macro
'
' Keyboard Shortcut: Ctrl+f
Dim wks As Worksheet
Dim pvt As PivotTable

Dim i as long, v as variant

'On Error Resume Next

v = Array("LI", "AA", "AB", "CD", "EF")

 

i = lbound(v)
For Each wks In Worksheets
For Each pvt In wks.PivotTables
    pvt.PivotFields("ship_code").ClearAllFilters
    pvt.PivotFields("ship_code").CurrentPage = _
       v(i)

        i = i + 1
Next pvt
Next wks

End Sub

 

Untested, but this is the general direction.  the V array would have the list of ship codes and you would need to have the quantity of items match the number of pivot tables.
