You can setup a condition for what range of numbers to attach to a data validation list. If L1:L11 has the integers 1 through 11 (e.g. months) and M1:M5 had 1 through 5 (e.g. years) the cell that is intended to contain either months or years should have a data validation list with.

=IF(loan_option=2,$M$1:$M$5,IF(loan_option=1,$L$1:$L$11,""))

Andreas.

Sub Example_FindFolders()
  Dim F As Folder
  For Each F In FindSessionFolders(olMailItem)
    Debug.Print F.Name; ": "; F.FolderPath
    'F.Display
  Next
End Sub

Function FindSessionFolders(ByVal DefaultItemType As OlItemType) As Collection
  Dim F As Folder, SubF As Folder
  Set FindSessionFolders = New Collection
  For Each F In Session.Folders
    If F.DefaultItemType = DefaultItemType Then FindSessionFolders.Add F
    If F.Folders.Count > 0 Then
      For Each SubF In FindFolders(F, DefaultItemType)
        FindSessionFolders.Add SubF
      Next
    End If
  Next
End Function

Function FindFolders(ByVal MainFolder As Folder, _
    ByVal DefaultItemType As OlItemType) As Collection
  Dim F As Folder, SubF As Folder
  Set FindFolders = New Collection
  For Each F In MainFolder.Folders
    If F.DefaultItemType = DefaultItemType Then FindFolders.Add F
    If F.Folders.Count > 0 Then
      For Each SubF In FindFolders(F, DefaultItemType)
        FindFolders.Add SubF
      Next
    End If
  Next
End Function

Sub Test_FindFolderByName()
  Dim ChosenFolder As Folder
  Set ChosenFolder = FindFolderByName("my tickets")
  If ChosenFolder Is Nothing Then
    MsgBox "Folder not found"
  Else
    Debug.Print ChosenFolder.FolderPath
  End If

Function FindFolderByName(ByVal FolderName As String, _
    Optional ByVal DefaultItemType As OlItemType = olMailItem) As Folder
  For Each FindFolderByName In FindSessionFolders(DefaultItemType)
    If StrComp(FindFolderByName.Name, FolderName, vbTextCompare) = 0 Then Exit For
  Next
End Function
