Sub macro_02()
'01 Nov 2012
Dim ws1 As Worksheet, ws2 As Worksheet
Dim r1 As Long, r2 As Long, c1 As Long, X As Long
Set ws1 = Sheets("Log") ' <<< source sheet name
Application.ScreenUpdating = False
ws1.Columns(1).Insert
r1 = ws1.Cells(Rows.Count, "B").End(xlUp).Row
c1 = ws1.Cells(1, Columns.Count).End(xlToLeft).Column
For X = 1 To r1
ws1.Cells(X, 2).Value = Trim(ws1.Cells(X, 2).Value)
Next X
ws1.Range("$B$1:B" & r1).AdvancedFilter Action:=xlFilterInPlace, Unique:=True
ws1.Range("B1:B" & r1).Copy
ws1.Range("A1").PasteSpecial Paste:=xlPasteValues
Application.CutCopyMode = False
ws1.Range("B1:B" & r1).AdvancedFilter Action:=xlFilterInPlace, Unique:=False
r2 = ws1.Cells(Rows.Count, "A").End(xlUp).Row

For X = 2 To r2
For Each ws In Sheets
Application.DisplayAlerts = False
If ws.Name = ws1.Cells(X, 1) Then ws.Delete
Application.DisplayAlerts = True
Next
Next

For X = 2 To r2
Set ws2 = Worksheets.Add(after:=Sheets(Sheets.Count))
ws2.Name = ws1.Cells(X, 1)
ws1.Range("A1").AutoFilter Field:=2, Criteria1:=ws1.Cells(X, 1)
ws1.Rows("1:" & r1).Copy
ws2.Range("A1").PasteSpecial Paste:=xlPasteValues
Application.CutCopyMode = False
Columns(1).Delete
ws2.Range(Cells(1, 1), Cells(1, c1)).EntireColumn.AutoFit
Next X
With ws1
.Range("A1").AutoFilter
.Columns(1).Delete
.Activate
.Range("A1").Select
End With
Application.ScreenUpdating = True
End Sub
