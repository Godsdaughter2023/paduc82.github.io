ray cacciatore say:

Hi kitpi,

The solution I had found was to create a new small Access application and adding a new function in it. Use the following VBA code to create a linked ODBC table that does not use a local DSN associated with it. This saves the connection information within Access at the table level. 

Function AttachDSNLessTable(stLocalTableName As String, stRemoteTableName As String, stServer As String, stDatabase As String, Optional stUsername As String, Optional stPassword As String)

On Error GoTo AttachDSNLessTable_Err

   

    Dim td As TableDef

    Dim stConnect As String

    Dim objAccess As Access.Application

    Dim dbExternal As DAO.Database

    Dim tbl As DAO.TableDef

   

    Set objAccess = New Access.Application

   

    '/////////////////////////////////////////////////////////////////

    '/// CAUTION

    '/////////////////////////////////////////////////////////////////

    Set dbExternal = DAO.OpenDatabase("C:\Apps\Metro\StatisPgm02\StatisPgm02.MDB")

   

    On Error Resume Next

    dbExternal.TableDefs.Delete "dbo_vwZnCorcCli"

    On Error GoTo AttachDSNLessTable_Err

    dbExternal.TableDefs.Refresh

   

    '//WARNING: This will save the username and the password with the linked table information.

    stConnect = "ODBC;DRIVER=SQL Server;SERVER=" & stServer & ";DATABASE=" & stDatabase & ";UID=" & stUsername & ";PWD=" & stPassword

   

    Set td = dbExternal.CreateTableDef(stLocalTableName, dbAttachSavePWD, stRemoteTableName, stConnect)

    dbExternal.TableDefs.Append td

    AttachDSNLessTable = True

    dbExternal.TableDefs.Refresh

    dbExternal.Close

    MsgBox "Update completed with success."

    Application.Quit

   

    Exit Function

 

AttachDSNLessTable_Err:

    MsgBox Err.Description

    AttachDSNLessTable = False

 

End Function

 

Public Function ChangeODBCTable()

       

    Call AttachDSNLessTable("dbo_vwZnCorcCli", "dbo.vwZnCorcCli", "SQLServerName", "DatabaseName", "username", "password")

 

End Function
