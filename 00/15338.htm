public string Send(string sendTo, string sendFrom, string subject, string message, string replyTo, string attachement, bool isHTML)

     {

         try

         {

             System.Net.Mail.SmtpClient client = new System.Net.Mail.SmtpClient();

             System.Net.Mail.MailMessage mail = new System.Net.Mail.MailMessage();

             //Create the Html part

             System.Net.Mail.AlternateView htmlView = System.Net.Mail.AlternateView.CreateAlternateViewFromString(message, Encoding.Default, "text/html");

             htmlView.TransferEncoding = System.Net.Mime.TransferEncoding.QuotedPrintable;

             mail.AlternateViews.Add(htmlView);

             if (attachement != string.Empty)

             {

                 System.Net.Mail.Attachment mAttachement = new System.Net.Mail.Attachment(attachement);

                 mail.Attachments.Add(mAttachement);

             }

             if (replyTo != string.Empty)

             {

                 mail.ReplyToList.Add(replyTo);

             }

             mail.From = new System.Net.Mail.MailAddress(sendFrom);

             mail.To.Add(sendTo);

             mail.Subject = subject;

             mail.IsBodyHtml = isHTML;

             mail.BodyEncoding = Encoding.Default;

             //****** DO NOT Change Order.   Causes process to fail!

             client.UseDefaultCredentials = false;

             client.Credentials = new System.Net.NetworkCredential(zbdConstants.SystemEmail, zbdConstants.SystemEmailPassword);

             client.DeliveryMethod = System.Net.Mail.SmtpDeliveryMethod.Network;

             client.EnableSsl = true;

             client.Port = 587;

             client.Host = "smtp.office365.com";

             client.Send(mail);

             return "success";

         }

         catch (Exception ex)

         {

             return ex.Message.ToString();

         }

     }
