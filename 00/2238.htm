mike h.. say:

Hi,

 

Yiou can use the code below. Note I have put no error checking in to ensure andy sheet named in the range actually exists, if a sheet named in the range doesn't exist then the code will fail.

 

 

Sub GroupSheets()
Dim ShtGroup() As String
Dim Lr As Long
Dim ShtName As String
Dim n As Long
Lr = Sheets("Index").Range("A" & Rows.Count).End(xlUp).Row
For Each c In Sheets("Index").Range("A7:A" & Lr)
    If UCase(c.Offset(, 1)) = "Y" Then
        ShtName = c.Value
        n = n + 1
        ReDim Preserve ShtGroup(1 To n)
        ShtGroup(n) = ShtName
    End If
Next
    Sheets(ShtGroup).Select
End Sub

EDIT....Same code with error checking

 

Sub GroupSheets()
Dim ShtGroup() As String
Dim Lr As Long, ws As Worksheet
Dim ShtName As String, c As Range
Dim n As Long
Lr = Sheets("Index").Range("A" & Rows.Count).End(xlUp).Row
For Each c In Sheets("Index").Range("A7:A" & Lr)
    If UCase(c.Offset(, 1)) = "Y" Then
        ShtName = c.Value
        On Error Resume Next
        Set ws = Sheets(ShtName)
        On Error GoTo 0
        If Not ws Is Nothing Then
        n = n + 1
        ReDim Preserve ShtGroup(1 To n)
        ShtGroup(n) = ShtName
        End If
        Set ws = Nothing
    End If
Next
    Sheets(ShtGroup).Select
End Sub
