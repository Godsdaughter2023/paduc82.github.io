Option Explicit

Sub Send_Extract_As_EMail()
Dim olApp As Outlook.Application
Dim olItem As Outlook.MailItem
Dim objDoc As Word.Document
Dim objXL As Excel.Worksheet
Dim objSel As Word.Selection
Dim strEmail As String
Dim dFname As msforms.DataObject
Dim sClipboard As String
Dim strSigPath As String
Dim strSigName As String
Dim strSignature As String
Dim strSubject As String
Dim strEaddress As String

'Set required defaults
strSigName = "Graham" 'the Outlook signature name
strSubject = "The Subject of the message"
strEaddress = "*** Email address is removed for privacy ***" 'the recipient
strSigPath = "C:\Users\" & Environ("username") & _
         "\AppData\Roaming\Microsoft\Signatures\" & strSigName & ".txt"
If Dir(strSigPath) = "" Then
    strSigPath = "C:\Documents and Settings\" & Environ("username") & _
         "\Application Data\Microsoft\Signatures\" & strSigName & ".txt"
End If
If Dir(strSigPath) <> "" Then
    strSignature = GetText(strSigPath)
Else
    strSignature = ""
End If
'Copy the selected text to the clipboard
'On Error Resume Next
Set objXL = ActiveWorkbook.ActiveSheet
objXL.Range("A1", "E32").Copy
'Assign the clipboard content to a text string

Set dFname = New msforms.DataObject
dFname.GetFromClipboard
sClipboard = dFname.GetText

'Get Outlook if it's running
Set olApp = GetObject(, "Outlook.Application")
'Outlook wasn't running, start it from code
If Err <> 0 Then
    Set olApp = CreateObject("Outlook.Application")
End If

Set olItem = olApp.CreateItem(0)
Set objDoc = olItem.GetInspector.WordEditor
Set objSel = objDoc.Windows(1).Selection
With olItem
    .To = strEaddress
    .BodyFormat = olFormatHTML
    .Subject = strSubject
    .Body = sClipboard & vbCr & vbCr & strSignature
    .Display
End With
'Clean up
Set olItem = Nothing
Set objDoc = Nothing
Set objSel = Nothing
Set dFname = Nothing
Set olApp = Nothing
End Sub

Function GetText(ByVal sFile As String) As String
'by Dick Kusleika
    Dim fso As Object
    Dim ts As Object
    Set fso = CreateObject("Scripting.FileSystemObject")
    Set ts = fso.GetFile(sFile).OpenAsTextStream(1, -2)
    GetText = ts.ReadAll
    ts.Close
End Function

suzanne s. barnhill say:
Yes, it can be docked. This can be very frustrating, I know, but the solution is really simple. Just grab the pane and drag it to the right beyond the edge of the window. I've seen suggestions that it is necessary to "slam" the pane against the right side, and I guess if you slam hard enough that it actually goes off the screen, then that would work, but in fact violence is not necessary--just drag it all the way off, and it will pop back, docked.
