barb reinhardt say:

This is not going to fix the existing links, but should prevent the issue in the future (on your machine).  

 

You can either fix them all manually (not my choice) or use VBA to do it.   Here is a short procedure to show you how to list them in the immediate window.

 

To add this to your workbook, use ALT F11,  insert a module in your active workbook and put this in it.  To see the immediate window, enter CTRL G.    Once you know what you have for the hyperlink address, it can be fixed programmatically as well. 

 

 

Option Explicit

Sub FixHyperlinks()
Dim myWB As Excel.Workbook
Dim myWS As Excel.Worksheet
Dim myHyp As Excel.Hyperlink
Dim myAddress As String

Set myWB = ActiveWorkbook
For Each myWS In myWB.Worksheets
    For Each myHyp In myWS.Hyperlinks
        myAddress = ""
        On Error Resume Next
        myAddress = myHyp.Address
        On Error GoTo 0
        If myAddress <> "" Then
             Debug.Print myWS.Name, myHyp.TextToDisplay, myHyp.Address
             myAddress = Replace(myAddress, "FIND AS STRING", "REPLACE AS STRING") 'Change as needed
             myHyp.Address = myAddress
        End If
    Next myHyp
Next myWS
End Sub

david marcovitz say: Something like this might work:

Sub CopyNotes()
    Dim oSld1 As Slide
    Dim oSld2 As Slide
    Dim oShp1 As Shape
    Dim oShp2 As Shape
    Dim i As Long
    Dim j As Long
    For i = 1 To Presentations(1).Slides.Count
        Set oSld1 = Presentations(1).Slides(i)
        Set oSld2 = Presentations(2).Slides(i)
        For j = 1 To oSld1.Shapes.Count
            Set oShp1 = oSld1.NotesPage.Shapes(j)
            Set oShp2 = oSld2.NotesPage.Shapes(j)
            If oShp1.HasTextFrame = msoTrue And oShp2.HasTextFrame = msoTrue Then
                oShp2.TextFrame.TextRange.Text = _
                    oShp1.TextFrame.TextRange.Text
            End If
        Next j
    Next i
End Sub
