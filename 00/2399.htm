myall_blues say:

To address the second question first, presumably your "Old Batteries" sheet currently has 12 used rows, so the line:

J = Worksheets("Old Batteries").UsedRange.Rows.Count

sets J=12 and then

xCell.EntireRow.Copy Destination:=Worksheets("Old Batteries").Range("A" & J + 1)

copies to J+1 (13)

See if the code below does what you want. I have made a couple of assumptions:

1. That you want the copied cells to start at row 1 (based on the comment about starting in row 13 being undesirable).

2. That you want columns A,C,D,F,G,I,R,T to be copied to the same columns on the second sheet.

Private Sub copy_sub()
    Dim xRg As Range, xCell As Range
    Dim I As Long, J As Long
    I = Worksheets("Scadamate RC").UsedRange.Rows.Count
    Set xRg = Worksheets("Scadamate RC").Range("AA1:AA" & I)
    Application.ScreenUpdating = False
    '
    J = 1
    For Each xCell In xRg
        If xCell.Value2 = "yes" Then    ' with column name
            Worksheets("Scadamate RC").Range("A" & xCell.Row).Copy Worksheets("Old Batteries").Range("A" & J)
            Worksheets("Scadamate RC").Range("C" & xCell.Row).Copy Worksheets("Old Batteries").Range("C" & J)
            Worksheets("Scadamate RC").Range("D" & xCell.Row).Copy Worksheets("Old Batteries").Range("D" & J)
            Worksheets("Scadamate RC").Range("F" & xCell.Row).Copy Worksheets("Old Batteries").Range("F" & J)
            Worksheets("Scadamate RC").Range("G" & xCell.Row).Copy Worksheets("Old Batteries").Range("G" & J)
            Worksheets("Scadamate RC").Range("I" & xCell.Row).Copy Worksheets("Old Batteries").Range("I" & J)
            Worksheets("Scadamate RC").Range("R" & xCell.Row).Copy Worksheets("Old Batteries").Range("R" & J)
            Worksheets("Scadamate RC").Range("T" & xCell.Row).Copy Worksheets("Old Batteries").Range("T" & J)
            J = J + 1
        End If
    Next
    Application.ScreenUpdating = True
End Sub
