potvinsucks24 say:

Filename of the photo is what I refer to as the photo name, yes...so the photo is called "Upstream Elevation" in the file explorer, the caption should simply be "Upstream Elevation". NOT the full name with the file path.

The selection process is by selecting photos in the file explorer, yes. Should have just done this to start but the current state of the macro is below.

Sub AddPics()

Application.ScreenUpdating = False

Dim i As Long, j As Long, c As Long, r As Long, NumCols As Long, iShp As InlineShape

Dim oTbl As Table, TblWdth As Single, StrTxt As String, RwHght As Single, ColWdth As Single

On Error GoTo ErrExit

NumCols = 1

RwHght = 9.525

On Error GoTo 0

'Select and insert the Pics

With Application.FileDialog(msoFileDialogFilePicker)

  .Title = "Select image files and click OK"

  .Filters.Add "Images", "*.gif; *.jpg; *.jpeg; *.bmp; *.tif; *.png"

  .FilterIndex = 2

  If .Show = -1 Then

    'Create a paragraph Style with 0 space before/after & centre-aligned

    On Error Resume Next

    With ActiveDocument

      .Styles.Add Name:="TblPic", Type:=wdStyleTypeParagraph

      On Error GoTo 0

      With .Styles("TblPic").ParagraphFormat

        .Alignment = wdAlignParagraphCenter

        .KeepWithNext = True

        .SpaceAfter = 0

        .SpaceBefore = 0

      End With

    End With

    'Add a 2-row by NumCols-column table to take the images

    Set oTbl = Selection.Tables.Add(Range:=Selection.Range, NumRows:=2, NumColumns:=NumCols)

    With ActiveDocument.PageSetup

      TblWdth = .PageWidth - .LeftMargin - .RightMargin - .Gutter

      ColWdth = TblWdth / NumCols

    End With

    With oTbl

      .AutoFitBehavior (wdAutoFitFixed)

      .Columns.Width = ColWdth

    End With

    CaptionLabels.Add Name:="Picture"

    For i = 1 To .SelectedItems.Count Step NumCols

      r = ((i - 1) / NumCols + 1) * 2 - 1

      'Format the rows

      Call FormatRows(oTbl, r, RwHght)

      For c = 1 To NumCols

        j = j + 1

        'Insert the Picture

        Set iShp = ActiveDocument.InlineShapes.AddPicture( _

          FileName:=.SelectedItems(j), LinkToFile:=False, _

          SaveWithDocument:=True, Range:=oTbl.Cell(r, c).Range)

        With iShp

          .LockAspectRatio = True

          If (.Width < ColWdth) And (.Height < RwHght) Then

            .Width = ColWdth

            If .Height > RwHght Then .Height = RwHght

          End If

        End With

        'Get the Image name for the Caption

        StrTxt = Split(.SelectedItems(j), "\")(UBound(Split(.SelectedItems(j), "\")))

        StrTxt = ": " & Split(StrTxt, ".")(0)

        'Insert the Caption on the row below the picture

        With oTbl.Cell(r + 1, c).Range

          .InsertBefore vbCr

          .Characters.First.InsertCaption _

          Label:="Picture", Title:=StrTxt, _

          Position:=wdCaptionPositionBelow, ExcludeLabel:=True

          .Characters.First = vbNullString

          .Characters.Last.Previous = vbNullString

        

        End With

        'Exit when we're done

        If j = .SelectedItems.Count Then Exit For

      Next

      'Add extra rows as needed

      If j < .SelectedItems.Count Then

        oTbl.Rows.Add

        oTbl.Rows.Add

      End If

    Next

  Else

   End If

End With

ErrExit:

Application.ScreenUpdating = True

End Sub


Sub FormatRows(oTbl As Table, x As Long, Hght As Single)

With oTbl

  With .Rows(x)

    .Height = CentimetersToPoints(Hght)

    .HeightRule = wdRowHeightExactly

    .Range.Style = "TblPic"

    .Cells.VerticalAlignment = wdCellAlignVerticalCenter

  End With

  With .Rows(x + 1)

    .Height = CentimetersToPoints(0.5)

    .HeightRule = wdRowHeightExactly

    .Range.Style = "Caption"

  End With

End With

End Sub

ossiemac say:

Try the code below. I have used 2 events; the Double click to insert the Check Mark and the Change event. The reason for not putting it all in the double click event is if the user simply deletes the check mark instead of double click or enters the letter "a" instead of double click then the code will not run. By having a separate change event then it does not matter if the change is made via the double click event or simply by the user editing he cell.

Note the comments and where you need to edit the code both for the data validation and vlookup.

To get the formula for the Vlookup you need to record the code to create the vlookup and then copy the formula part only of the result of the recorded code into the line in lieu of my vlookup formula.

The formula needs to be in the R1C1 style as returned by recording the code.

To record the code for the formula you only have to turn on the recorder and then select a cell that already has the formula and then click in the formula bar where formulas are edited and press return and then turn off the recorder. the formula will be in a new standard module that gets automatically created.

Private Sub Worksheet_BeforeDoubleClick(ByVal Target As Range, Cancel As Boolean)

    If Not Intersect(Target, Columns(3)) Is Nothing Then    'Not nothing then is something
        Cancel = True   'Cancel edit mode that is normally invoked with double click
        If Target.Value = "" Then
            'No current check mark so insert check mark
            Target.Font.Name = "Webdings"   'Not required if column font already set
            'Target.Font.Name = "Marlett"   'Alternate to previous line for Check Mark Font
            Target.Value = "a"
        Else
            Target.ClearContents
        End If
    End If
   
End Sub

Private Sub Worksheet_Change(ByVal Target As Range)

    If Not Intersect(Target, Columns(3)) Is Nothing Then    'Not nothing then is something
        'Next If / End If prevents any change other than "a" or clear
        If Target.Value <> "a" And Target.Value <> "" Then
            Application.Undo    'Revert back to precious value
            MsgBox "Only double click cell to toggle between Checked and Un-Checked"
            Exit Sub
        End If
        If Target.Value = "a" Then
            'Has Check Mark so insert validation in column to right.
            Target.Offset(0, 1).ClearContents
            With Target.Offset(0, 1).Validation
                .Delete
                .Add Type:=xlValidateList, _
                    AlertStyle:=xlValidAlertStop, _
                    Operator:=xlBetween, _
                    Formula1:="=Sheet2!$D$2:$D$4"   'Edit "=Sheet2!$D$2:$D$4" to your validation list
                'Following lines for Data Validation are optional depending on your requirements
                .IgnoreBlank = True
                .InCellDropdown = True
                .InputTitle = ""
                .ErrorTitle = ""
                .InputMessage = ""
                .ErrorMessage = ""
                .ShowInput = True
                .ShowError = True
            End With
        Else    'No check mark so clear validation and value and insert formula
            Target.Offset(0, 1).Validation.Delete
            Target.ClearContents
            'Record your Vlookup formula and edit following line with result
            Target.Offset(0, 1).Formula = "=VLOOKUP(RC[-2],Sheet2!R2C1:R9C2,2,FALSE)"
        End If
    End If

End Sub
