graham mayor say:
Does the field ordinarily have numeric data? If so you can use a formatting switch e.g.

{Mergefield Fieldname \# ",0.00;;")

For more about formatting switches see http://www.gmayor.com/formatting_word_fields.htm

If the data is not numeric then you can conditionally insert the field e.g.

{IF {Mergefield Fieldname} <> "0" {Mergefield Fieldname}}

Sub ConvertData()

'29 Oct 2012
Dim ws1 As Worksheet, ws2 As Worksheet
Dim r1 As Long, r As Long, c As Long
Set ws1 = Sheets("Sheet1") '<<< Source Sheet  change
Set ws2 = Sheets.Add
r1 = ws1.Cells(Rows.Count, 1).End(xlUp).Row

ws1.Range("A1:" & r1).AdvancedFilter _
Action:=xlFilterInPlace, Unique:=True
ws1.Range("A1:A" & r1).Copy
ws2.Range("A1").PasteSpecial (xlPasteValues)
Application.CutCopyMode = False

ws1.Range("B1:B" & r1).AdvancedFilter _
Action:=xlFilterInPlace, Unique:=True
ws1.Range("B2:B" & r1).Copy
ws2.Range("B1").PasteSpecial xlPasteValues, Transpose:=True
Application.CutCopyMode = False
ws1.Range("B1:B" & r1).AdvancedFilter _
Action:=xlFilterInPlace, Unique:=False

r = Cells(Rows.Count, 1).End(xlUp).Row
c = Cells(1, Columns.Count).End(xlToLeft).Column
For i = 2 To r
For j = 2 To c
For Each rr In ws1.Range("A2:A" & r1)
If Cells(i, 1) & Cells(1, j) = rr & rr.Offset(, 1) Then
If Cells(i, j) = "" Then Cells(i, j) = rr.Offset(, 2) Else Cells(i, j) = Cells(i, j) & "," & rr.Offset(, 2)
End If
Next
Next
Next
Range(Cells(1, 1), Cells(1, c)).EntireColumn.AutoFit
Range("A1").Select
End Sub
