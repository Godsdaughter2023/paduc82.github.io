Hi,

 

This will leave the cells selected

 

Sub Exclamation_Mark()
 Dim MyRange As Range, c As Range
 Dim CopyRange
 On Error Resume Next
 Set MyRange = ActiveSheet.UsedRange.SpecialCells(xlCellTypeFormulas)
 For Each c In MyRange
     If InStr(1, c.Formula, "!", vbTextCompare) > 0 Then
        If CopyRange Is Nothing Then
                Set CopyRange = c
        Else
                Set CopyRange = Union(CopyRange, c)
        End If
     End If
 Next
 If Not CopyRange Is Nothing Then
    CopyRange.Select
End If
 End Sub
