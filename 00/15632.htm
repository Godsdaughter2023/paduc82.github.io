    There may be a better way but the only thing I can think of now is to use a temporary table. 

    Create a table that will have the fields from the crosstab plus the average and a field for Ranking. Use code like the following to clear the table, append the current report card info, then add the rankings.

    Dim db As Database, rs As Recordset

    Dim strSQL As String

    Dim dblCurrAvg As Double. dblPrevAvg as Double

    Dim intRank As Integer

    'Clear records from temp table

    strSQL = "DELETE * FROM tmpReportCard;"

    CurrentDB.Execute strSQL, dbFailOnError

    'Append records from Crosstab query

    strSQL = "INSERT INTO tmpReportCard (StudentID, subject1, subject2, subject3, AverageScore) " & _

                   "SELECT StudentID, subject1, subject2, subject3, AverageScore FROM qryScoresXT;"

    CurrentDB.Execute strSQL, dbFailOnError

    'create Recordset and initialize variables

    Set db = CurrentDB()

    Set rs = db.OpenRecordset("SELECT * FROM tmpReportCard ORDER BY AverageScore;")

    intRank = 0

    dblPrevAvg=0

    'Loop through Recordset to assign rankings

    With rs

        .MoveFirst

        Do until .EOF

            dblCurrAvg = .Fields("AverageScore")

            If dblCurrAvg <> dblPrevAvg

                 intRank = intRank+1

            End If

            .Edit

            .Fields("Rank") = intRank

            .Update

            .MoveNext

        Loop

    End With

    Set rs = Nothing

    Set db=Nothing

    That should assign a ranking to each student with ties getting the same rank.
