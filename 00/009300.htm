Maybe something like the following will help. This assumes the cursor is in the message body and that the first image is to be cropped to the size quoted. The commands are essentially similar to those used in Word.

Sub CropImage()
Dim oShape As Object
Dim oDoc As Object
    On Error GoTo ErrHandler
    If TypeName(ActiveWindow) = "Inspector" Then
        If ActiveInspector.IsWordMail And ActiveInspector.EditorType = olEditorWord Then
            Set oDoc = ActiveInspector.WordEditor

        End If
    End If
lbl_Exit:
    Set oShape = Nothing
    Set oDoc = Nothing
    Exit Sub
ErrHandler:
    Beep
    Err.Clear
    GoTo lbl_Exit
End Sub
