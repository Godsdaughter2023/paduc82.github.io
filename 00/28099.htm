' Scan column E starting with row 16 till end. If column A starts with P 
' and columns E,F,G,H, J,K,L,M,N,P, and Q are all zeros then store the
' line number in array and delete.

Sub DeleteRowsWithZero()

Dim MyRange As Range, c As Range

Dim CopyRange As Range, LastRow As Long

LastRow = Cells(Cells.Rows.Count, "E").End(xlUp).Row

Application.Calculation = xlCalculationManual
Application.ScreenUpdating = False
Application.DisplayStatusBar = True
Set MyRange = Range("e16:e" & LastRow)
'Set MyRange = Range("e18:e18")
Application.StatusBar = "Please wait while performing task 1..."
For Each c In MyRange

If Left(c.Offset(0, -4), 1) <> "P" _
    And c.Offset(0, 0) = 0 And c.Offset(0, 1) = 0 And c.Offset(0, 2) = 0 And c.Offset(0, 3) = 0 _
    And c.Offset(0, 6) = 0 And c.Offset(0, 7) = 0 And c.Offset(0, 8) = 0 And c.Offset(0, 9) = 0 And c.Offset(0, 10) = 0 _
    And c.Offset(0, 12) = 0 And c.Offset(0, 13) = 0 _
Then
   
        If CopyRange Is Nothing Then
                Set CopyRange = c.EntireRow
        Else
                Set CopyRange = Union(CopyRange, c.EntireRow)
        End If
End If
Application.StatusBar = "Working On Line #: " & c.Offset(0, -4)


Next

If Not CopyRange Is Nothing Then

CopyRange.Delete

End If

Application.Calculation = xlCalculationAutomatic
Application.ScreenUpdating = True
Application.StatusBar = "Task 1 Completed"
End Sub

per jessen say:

Turning off screenupdating and calculation has a big effort here.

 

I have modified the code a bit using an autofilter to filter out entries in column A starting with 'P' reducing the number of rows to evaluate.

 

See if this speed up things further..

 

Sub DeleteRowsWithZero()
Dim MyRange As Range, c As Range
Dim CopyRange As Range, LastRow As Long

LastRow = Cells(Cells.Rows.Count, "A").End(xlUp).Row

Application.Calculation = xlCalculationManual
Application.ScreenUpdating = False
Application.DisplayStatusBar = True

Set MyRange = Range("A1:A" & LastRow)

Application.StatusBar = "Please wait while performing task 1..."
MyRange.AutoFilter Field:=1, Criteria1:="<>P*", Operator:=xlAnd
For Each cell In MyRange.SpecialCells(xlCellTypeVisible)
    If Application.WorksheetFunction.Sum(cell.Offset(0, 4).Resize(1, 10), cell.Offset(0, 10).Resize(1, 2)) = 0 Then
     
            If CopyRange Is Nothing Then
                    Set CopyRange = cell.EntireRow
            Else
                    Set CopyRange = Union(CopyRange, cell.EntireRow)
            End If
    End If
    Application.StatusBar = "Working On Line #: " & cell
Next

If Not CopyRange Is Nothing Then
    CopyRange.Delete
End If
MyRange.AutoFilter
Application.Calculation = xlCalculationAutomatic
Application.ScreenUpdating = True
Application.StatusBar = "Task 1 Completed"
End Sub
