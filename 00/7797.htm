See if this gets you started:

Sub foo()
    Dim lr As Long
    Dim ws As Worksheet
    Dim RS As Worksheet
    Set RS = Sheets("Recovery Sheet")
   
    For Each ws In ActiveWorkbook.Worksheets
        If ws.Name <> "Recovery Sheet" Then
            lr = RS.Cells(Rows.Count, 2).End(xlUp).Offset(1).Row
           
            RS.Cells(lr, 1).Value = ws.Range("A8").Value
            RS.Cells(lr, 2).Value = ws.Range("A9").Value
            RS.Cells(lr, 3).Value = ws.Range("F12").Value
            RS.Cells(lr, 4).Value = ws.Range("F15").Value
            RS.Cells(lr, 5).Value = ws.Range("F16").Value
        End If
    Next ws
   
End Sub
