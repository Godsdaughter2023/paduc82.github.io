Use what you need from Tom Ogilvy's code.

Sub Make_XLSX()
'tom ogilvy convert xlsm to xlsx
    Dim sPath As String, sName As String
    Dim wkbk As Workbook, sName1 As String
    sPath = "C:\Data\Test\"     '<==  change to your path
    On Error GoTo ErrHandler
    If Right(sPath, 1) <> "\" Then sPath = sPath & "\"
    sName = Dir(sPath & "*.xlsm")
    Application.EnableEvents = False
    Do While sName <> ""
        sName1 = Replace(LCase(sName), ".xlsm", "")
        Application.DisplayAlerts = False
        Set wkbk = Workbooks.Open(sPath & sName)
        wkbk.SaveAs FileName:=sPath & sName1 & ".xlsx", FileFormat:=xlOpenXMLWorkbook
        wkbk.Close SaveChanges:=False
        Application.DisplayAlerts = True
        sName = Dir()
    Loop
    Kill sPath & "*.xlsm"
ErrHandler:
    Application.EnableEvents = True
End Sub

joebox2673 say:

How do i make each bar in an excel graph automatically change colors based on the data?

I want my excel bar graph colors to change based on the cell data. For example cell A1 in my graph may be over goal. If that is the case I want the bar graph representing that cell to be the color red.
rich007a say:

Where did you paste the macro?

Did you "Right-click on the sheet's tab and click View Code" then paste the macro there?

Since we are using the worksheet change event, the macro has to go in the Sheet module for the sheet which you want it to run on.

Sorry, there was a slight error in the code I posted before.  Please try this (it has an error catch so it wont run if there is not a chart in the worksheet):

 

'======================================================
Private Sub Worksheet_Change(ByVal Target As Range)
    Dim ch As Chart, ser As Series, pt As Point
    Dim i As Long, rngData As Range, rngGoal As Range
   
    Set rngData = Range("E4:E7") '#### The plotted range of values ####
    Set rngGoal = Range("E9")    '#### Set the address of the Goal here ####
    If Intersect(Union(rngData, rngGoal), Target) Is Nothing Then Exit Sub
   
    Application.ScreenUpdating = False
   
    On Error Resume Next
    Set ch = ActiveSheet.ChartObjects(1).Chart
    If Err.Number <> 0 Then
        'MsgBox "Chart object can not be found in the active sheet", vbCritical
        Application.ScreenUpdating = True
        Exit Sub
    End If
    On Error GoTo 0
   
    Set ser = ch.SeriesCollection(1)
   
    i = 1
    For Each pt In ser.Points
        If rngData(i).Value > rngGoal.Value Then
            With pt.Interior
                .ColorIndex = 3 'Red
                .Pattern = xlSolid
            End With
        Else
            With pt.Interior
                .ColorIndex = 4 'Green
                .Pattern = xlSolid
            End With
        End If
        i = i + 1
    Next pt
    Application.ScreenUpdating = True
End Sub
'======================================================

Note: the above will not fire if the values change during recalculation, for example if you enter =A1  in cell E7, when you change the value of A1, the above code does not run.  To do that, you would need to use the worksheet calculate event like this:

'======================================================
Private Sub Worksheet_Calculate()
    Dim ch As Chart, ser As Series, pt As Point
    Dim i As Long, rngData As Range, rngGoal As Range
   
    Set rngData = Range("E4:E7") '#### The plotted range of values ####
    Set rngGoal = Range("E9")    '#### Set the address of the Goal here ####
   
    Application.ScreenUpdating = False
   
    On Error Resume Next
    Set ch = ActiveSheet.ChartObjects(1).Chart
    If Err.Number <> 0 Then
        'MsgBox "Chart object can not be found in the active sheet", vbCritical
        Application.ScreenUpdating = True
        Exit Sub
    End If
    On Error GoTo 0
   
    Set ser = ch.SeriesCollection(1)
   
    i = 1
    For Each pt In ser.Points
        If rngData(i).Value > rngGoal.Value Then
            With pt.Interior
                .ColorIndex = 3 'Red
                .Pattern = xlSolid
            End With
        Else
            With pt.Interior
                .ColorIndex = 4 'Green
                .Pattern = xlSolid
            End With
        End If
        i = i + 1
    Next pt
    Application.ScreenUpdating = True
End Sub
'======================================================
