To batch process do something like:

Sub BatchMapPoint()
' You must set a reference to the MapPoint Object Library
' via VBE > Tools > References...
'
'This code has not been tested!
'
    Dim c As Range, rng As Range
    Dim MP As MapPoint.Application
    Dim mpMap As MapPoint.Map        '?
    Dim mpRoute As MapPoint.Route '?
    Dim strZip1 As String, strZip2 As String
    Dim bWeStartedMapPoint As Boolean
   
     'Get MapPoint if it's running:
    On Error Resume Next
    Set MP = GetObject(, "MapPoint.Application.NA.11") 'Don't know what the NA and 11 is for...
    If Err <> 0 Then
        On Error GoTo ExitPoint
         'MapPoint wasn't running, start it from code:
        Set MP = CreateObject("MapPoint.Application.NA.11")
        bWeStartedMapPoint = True
    End If
    On Error GoTo ExitPoint
   
    MP.Visible = True
    Set mpMap = MP.NewMap
    Set mpRoute = mpMap.ActiveRoute
   
     'Set range where data is stored:
    Set rng = Worksheets("Sheet1").Range("A1:A100")

     'Loop through each cell in the range:
    For Each c In rng
        strZip1 = c.Value                          'col A
        strZip2 = c.Offset(0, 1).Value     'col B
       
         'Add route stops and calculate the route
        mpRoute.Waypoints.Add objMap.FindResults(strZip1).Item(1)
        mpRoute.Waypoints.Add objMap.FindResults(strZip1).Item(1)
        mpRoute.Calculate
       
        c.Offset(0, 2).Value = objRoute.Distance 'Output to col C
    Next c

ExitPoint:
    On Error Resume Next
    If bWeStartedMapPoint Then MP.Quit
    Set mpRoute = Nothing
    Set mpMap = Nothing
    Set MP = Nothing
End Sub

 

Note: I do not have MapPoint, I've written the above using what I found in that link, it may not be correct!!!!

My comment about the "NA" and "11" I think are answered in part 1 of the above link.  NA=North America, EU = Europe.  11 is the version I think.  It will be the same as number in the available Reference ("Microsoft MapPoint 11.0 Object Library" (either North America or Europe)).

The above assumes you have zip/postal codes in columns A and B.  Adjust the address of rng accordingly.
