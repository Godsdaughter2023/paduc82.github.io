    tasos_k say:

    Okay,

    if z text file already exists

    try this code,

     

    Sub macro_001()

    'file z.txt exists
    Dim myPath As String, myFile As String
    Dim L As Long
    myPath = ThisWorkbook.Path & "\"
    myFile = "z.txt"
    Dim WB As Workbook, txtWB As Workbook
    Set WB = ThisWorkbook
    Application.ScreenUpdating = False
    Set txtWB = Workbooks.Open(myPath & myFile)
    L = txtWB.ActiveSheet.UsedRange.Rows.Count + 1
    WB.Sheets(1).UsedRange.Copy txtWB.Sheets(1).Cells(L, 1)
    With txtWB
    Application.DisplayAlerts = False
    .SaveAs Filename:=myPath & myFile, FileFormat:=xlText

    Application.DisplayAlerts = True
    .Close True
    End With
    WB.Save
    Application.ScreenUpdating = True
    End Sub
