frank arendt-theilen say:

Hi,

ok, here is a differnt solution without the Split function:

 

Dim SDP_file_Name As String
Dim IsOpen As Boolean
Dim TargetWB As Workbook
Dim wb As Workbook
Dim lngI As Long
Dim lngBackslashPosition As Long

  SDP_file_Name = Application.GetOpenFilename(, , "Open")
 
  For lngI = 1 To Len(SDP_file_Name)
    If InStr(lngI, SDP_file_Name, "\") <> 0 Then
      lngBackslashPosition = InStr(lngI, SDP_file_Name, "\")
    End If
  Next
 
  SDP_file_Name = Right(SDP_file_Name, Len(SDP_file_Name) - lngBackslashPosition)

  For Each wb In Application.Workbooks
    If wb.Name = SDP_file_Name Then
      IsOpen = True
      Exit For
    End If
  Next
 
  If Not IsOpen Then
    Set TargetWB = Workbooks.Open(SDP_file_Name)
  Else
    MsgBox "Workbook [" & SDP_file_Name & "] is already open!", vbCritical
  End If

Dim FileNameArr As Variant
Dim SDP_file_Name As String
Dim IsOpen As Boolean
Dim TargetWB As Workbook

SDP_file_Name = Application.GetOpenFilename(, , "Open")
FileNameArr = Split(SDP_file_Name, "\")

For Each wb In Application.Workbooks
    If wb.Name = FileNameArr(UBound(FileNameArr)) Then
        IsOpen = True
        Exit For
    End If
Next
If Not IsOpen Then
   Set TargetWB = Workbooks.Open(SDP_file_Name)
Else
    Set TargetWB = Workbooks(FileNameArr(UBound(FileNameArr)))
End If

Remove extra spaces from whole worksheet in one step - Microsoft Community

caroline 09876 say:

Remove extra spaces from whole worksheet in one step

Is it possible to create a macro which I can apply to a whole worksheet to remove all extra spaces? 
I currently just use =trim formula but if i have many columns it takes ages to keep doing each column one at a time
I have also used 'replace' to replace a double space with one space but often have a random number of spaces or returns from a clients excel file so the replace doesn't always work. 


chip pearson say:

Try code like the following:

Sub RemoveSpaces()
    Dim R As Range
    Dim WS As Worksheet
    Set WS = ActiveSheet
    With Application
        .EnableEvents = False
        .Calculation = xlCalculationManual
        .ScreenUpdating = False
    End With
    For Each R In WS.UsedRange.SpecialCells(xlCellTypeConstants, xlTextValues)
            R.Value = Application.WorksheetFunction.Trim(R.Text)
    Next R
    With Application
        .EnableEvents = True
        .Calculation = xlCalculationAutomatic
        .ScreenUpdating = True
    End With
End Sub
