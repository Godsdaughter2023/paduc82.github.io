hansv mvp say:

You could perform a Replace to change the date in the formulas. This could be automated by using a macro, for example:

Sub ReplaceDates()
    Dim strOld As String
    Dim strNew As String
    Dim wsh As Worksheet
    strOld = InputBox("Enter the old date exactly as used in the filename")
    If strOld = "" Then Exit Sub
    strNew = InputBox("Enter the new date exactly as used in the filename")
    If strNew = "" Then Exit Sub
    Application.ScreenUpdating = False
    Application.Calculation = xlCalculationManual
    For Each wsh In Worksheets
        wsh.Cells.Replace What:=strOld, Replacement:=strNew, LookAt:=xlPart
    Next wsh
    Application.Calculation = xlCalculationAutomatic
    Application.ScreenUpdating = True
End Sub

Sub x()
    Dim objCht As Chart

    objCht.Parent.Activate
    objCht.ExportAsFixedFormat xlTypePDF, "C:\temp\xx.pdf"
   
End Sub
