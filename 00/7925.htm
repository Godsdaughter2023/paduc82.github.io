john korchok say:

This is possible if you use Word's Workgroup Templates feature. This requires that both computers have access to a shared network drive:

  1. On the shared network drive, create a folder that will house the templates.
  2. In Word on both computers, open Word>Preferences>File Locations, then repeat the next two steps for both machines.
  3. Select the Workgroup templates option, then click on the Modify button.
  4. Navigate to the network share, find the template folder and click on the Choose button. The path to the folder should show (at least in part) in the Prefs panel. OK out.
  5. Copy the templates to the folder on the network share.

Now, when you choose File>New from Template..., you will see the network folder name just below My Templates. Click on it and select the template to use.

graham mayor say:
I am not sure of the point of the bookmark here. You simply need to set the range to the point where you want the insertion i.e. after the space following the colon (Chr(58)) and set the end of the range to the start.

Then insert the text in the range as follows.

Then set the bookmark to encompass the range, should you need the bookmark for some other purpose.

Dim wrdApp As Object
Dim wrdDoc As Object
Dim oRng As Object

On Error Resume Next
Set wrdApp = GetObject(, "Word.Application")
If Err Then
    Set wrdApp = CreateObject("Word.Application")
End If
On Error GoTo 0

wrdApp.Visible = True

Set wrdDoc = wrdApp.Documents.Add(Template:="C:\Users\DD.dotx")
Set oRng = wrdDoc.Paragraphs(1).Range
oRng.movestartuntil Chr(58)
oRng.Start = oRng.Start + 2
oRng.End = oRng.Start
oRng.Text = "ABC "
wrdDoc.Bookmarks.Add Name:="colName", Range:=oRng

Alternatively you could replace
oRng.movestartuntil Chr(58)
oRng.Start = oRng.Start + 2
with
oRng.Start = oRng.Start + 8

vincentchoy say:

DMARC checks two things

1. Whether the domain names match between the message-envelope FROM and the message FROM

2. Whether the domain names match between the DKIM d= and the message FROM

As such, you have to ensure you put in the required DKIM entries for Office365 into your DNS, and turn on DKIM or else DMARC will fail.

Your screenshot didn't appear so I can only guess you might have forgotten to set DKIM.

Other possible reasons are 

1. You have apps that break dkim (eg disclaimer or signature software that modify the email while being sent)

2. You have legitimate emails that are sent on your behalf (eg mailchimp or other bulk email services) that don't originate from Office365 (will fail envelope FROM and DKIM d=) , but impersonates one of your corporate email address (message FROM)

3. Your dmarc reports found malicious impersonation mails that didn't originate from office365 (real spoof mails)

The problem I'm having is that as soon as I input data into that blank cell and press enter, all of the cells in the range lose their fill or highlight. How can I keep this from happening? I want the cells to remain orange. The code is posted below:

Worksheet Code:

Option Explicit
Dim bCellCheck As Boolean
Dim bBlink As Boolean

Private Sub Worksheet_Change(ByVal Target As Excel.Range)
Dim rColumn As Range
Dim rCell As Range
Dim sAdress As String

On Error GoTo ErrorHandle

Set rColumn = Range("C1", "C30")

bCellCheck = False

For Each rCell In rColumn
   If rCell.Value = "" Then
      bCellCheck = True
      If Len(sAdress) > 0 Then
         sAdress = sAdress & "," & rCell.Address
      Else
         sAdress = sAdress & rCell.Address
      End If
   End If
Next

If bCellCheck = True And bBlink = False Then
   Set rRange = Range(sAdress)
   bBlink = True
   StartBlink
ElseIf bCellCheck = True And bBlink = True Then
   Set rRange = rColumn
   StopBlink
   Set rRange = Range(sAdress)
   StartBlink
ElseIf bCellCheck = False And bBlink = True Then
   Set rRange = rColumn
   StopBlink
   bBlink = False
End If

Exit Sub
ErrorHandle:
MsgBox Err.Description & " Procedure Worksheet_Change."
Set rRange = Nothing
Set rColumn = Nothing
Set rCell = Nothing
bCellCheck = False
End Sub

Module Code:

Option Explicit
Public rRange As Range
Dim dNextTime As Double

Sub StartBlink()

On Error GoTo ErrorHandle

With rRange.Interior
   If .ColorIndex = 3 Then   
      .ColorIndex = 40   
   Else
      .ColorIndex = 3        
   End If
End With

dNextTime = Now + TimeSerial(0, 0, 1)

Application.OnTime dNextTime, "StartBlink", , True

Exit Sub
ErrorHandle:
MsgBox Err.Description & " Procedure StartBlink."
Set rRange = Nothing
End Sub

Sub StopBlink()

On Error GoTo ErrorHandle

rRange.Interior.ColorIndex = xlNone

Application.OnTime dNextTime, "StartBlink", , False

BeforeExit:
Set rRange = Nothing

Exit Sub
ErrorHandle:
MsgBox Err.Description & " Procedure StopBlink."
Resume BeforeExit
End Sub

Thanks in advance,

James

   '
   '    nDuplexSetting - One of the following standard settings:
   '       1 = None
   '       2 = Duplex on long edge (book)
   '       3 = Duplex on short edge (legal)
   '
   ' ==================================================================
   Public Function SetPrinterDuplex(ByVal sPrinterName As String, _
      ByVal nDuplexSetting As Long) As Boolean
      Dim hPrinter As Long
      Dim pd As PRINTER_DEFAULTS
      Dim pinfo As PRINTER_INFO_2
      Dim dm As DEVMODE
      Dim yDevModeData() As Byte
      Dim yPInfoMemory() As Byte
      Dim nBytesNeeded As Long
      Dim nRet As Long, nJunk As Long

      On Error GoTo cleanup

      If (nDuplexSetting < 1) Or (nDuplexSetting > 3) Then
         MsgBox "Error: dwDuplexSetting is incorrect."
         Exit Function
      End If

      pd.DesiredAccess = PRINTER_ALL_ACCESS
      nRet = OpenPrinter(sPrinterName, hPrinter, pd)
      If (nRet = 0) Or (hPrinter = 0) Then
         If Err.LastDllError = 5 Then
'            MsgBox "Access denied -- See the article for more info.",
            Debug.Print "Access denied -- Check your printer's permissions.", _
            vbInformation + vbOKOnly
         Else
            Debug.Print "Cannot open the printer specified " & _
              "(make sure the printer name is correct).", _
              vbInformation + vbOKOnly
         End If
         Exit Function
      End If

      nRet = DocumentProperties(0, hPrinter, sPrinterName, 0, 0, 0)
      If (nRet < 0) Then
         Debug.Print "Cannot get the size of the DEVMODE structure.", _
         vbInformation + vbOKOnly
         GoTo cleanup
      End If

      ReDim yDevModeData(nRet + 100) As Byte
      nRet = DocumentProperties(0, hPrinter, sPrinterName, _
                  VarPtr(yDevModeData(0)), 0, DM_OUT_BUFFER)
      If (nRet < 0) Then
         Debug.Print "Cannot get the DEVMODE structure.", _
         vbInformation + vbOKOnly
         GoTo cleanup
      End If

      Call CopyMemory(dm, yDevModeData(0), Len(dm))

      If Not CBool(dm.dmFields And DM_DUPLEX) Then
        Debug.Print "You cannot modify the duplex flag for this printer " & _

      End If

        VarPtr(yDevModeData(0)), VarPtr(yDevModeData(0)), _

You have to decide what you want to occur here?

 

If the user does not want to save the data and not finish saving the required fields, then do you want to exit?

 

However, you certainly could skip the prompt we just added. Our wee bit of new code expects err.number = 0. That’s how it knows the save went well. And with you “undo”, then error code is also zero.

 

You can simple set the error number to some “junk” value, so it not zero.

 

I would say try this:

 

      If MsgBox("Do you want to complete the fields", vbYesNo) <> vbYes Then
