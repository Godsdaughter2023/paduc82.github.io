ronrosenfeld say:

It is not clear to me exactly how you want your output structured, but his macro will give you a starting point.

It assumes

  • your "Narrations" are in Column A, on Sheet1
  • All bill numbers except possibly the last are followed by commas
  • the narration ends with the phrase "for the month of ..."
  • your output is similar to what you have above, with the bill numbers in the left hand column; and the narration in the right hand column adjacent.

To enter this Macro (Sub), <alt-F11> opens the Visual Basic Editor.
Ensure your project is highlighted in the Project Explorer window.
Then, from the top menu, select Insert/Module and
paste the code below into the window that opens.

To use this Macro (Sub), <alt-F8> opens the macro dialog box. Select the macro by name, and <RUN>.

===========================

Option Base 0
Sub ExtractBills()
    Dim wsSrc As Worksheet, wsRes As Worksheet
    Dim vSrc As Variant, rRes As Range, vRes() As Variant
    Dim S As String
    Dim COL As Collection
    Dim V(1) As Variant
    Dim RE As Object, MC As Object
    Dim I As Long, J As Long
   
'Set worksheets for Source and Results
Set wsSrc = Worksheets("sheet1")
Set wsRes = Worksheets("sheet1")
    Set rRes = wsRes.Range("D1")
   
'Collect source data
With wsSrc
    vSrc = .Range("a1", .Cells(.Rows.Count, "A").End(xlUp))
End With

Set RE = CreateObject("vbscript.regexp")
With RE
    .Global = True
    .ignorecase = True
    .Pattern = "(\d+(?=,|\s*for the month of))+|(?:for the month of\s+(.*))"
End With

For I = 1 To UBound(vSrc)
    S = vSrc(I, 1)
    Do Until InStr(S, "month of") > 0
        I = I + 1
        S = S & vSrc(I, 1)
    Loop
    If RE.test(S) = True Then
        V(0) = ""
        V(1) = S
        COL.Add V
        Set MC = RE.Execute(S)
        For J = 1 To MC.Count - 1
            'V(0) = CStr(MC(MC.Count - 1).submatches(1)) 'Bill month, year
            V(0) = MC(J - 1).submatches(0) 'Bill number
            COL.Add V
        Next J
    End If
Next I

    vRes(I, 1) = COL(I)(0)
    If vRes(I, 1) = "" Then vRes(I + 1, 2) = COL(I)(1)
Next I
       
Set rRes = rRes.Resize(UBound(vRes, 1) + 1, UBound(vRes, 2))
With rRes
    .EntireColumn.Clear
    .EntireColumn.NumberFormat = "@"
    .Value = vRes
    With .Rows(1)
        .Font.Bold = True
        .HorizontalAlignment = xlCenter
    End With
    .EntireColumn.AutoFit
    With .Columns(2)
        .WrapText = True
        .ColumnWidth = 35
    End With
    .EntireRow.AutoFit
End With
       
End Sub

rick rothstein, mvp - excel say:

This more compact code seems to do what your code does, but more compactly. Note, though, that I did not do anything with headers or cell formatting on the output (I wasn't sure about their need)...

Sub ExtractBills()
  Dim R As Long, RR As Long, X As Long, Temp As Variant, Data As Variant, Record As Variant
  Dim wsSrc As Worksheet, wsRes As Worksheet, rRes As Range
 
  '  Set worksheets for Source and Results
  Set wsSrc = Worksheets("sheet1")
  Set wsRes = Worksheets("sheet1")
  Set rRes = wsRes.Range("D2")
 
  Data = wsSrc.Range("A1", wsSrc.Cells(Rows.Count, "A").End(xlUp))
  ReDim Record(1 To UBound(Data), 1 To 2)
  For R = 1 To UBound(Data)
    If Len(Data(R, 1)) Then
      Temp = Temp & " " & Data(R, 1)
      If Temp Like "*for the month of*" Then
        X = X + 1
        Record(X, 2) = Trim(Temp)
        Temp = ""
      End If
    End If
  Next
  For R = 1 To X
    rRes.Offset(RR, 1).Value = Record(R, 2)
    Temp = Split(Trim(Split(Split(Record(R, 2), "for the month of")(0), "bill no.")(1)), ", ")
    rRes.Offset(RR, 0).Resize(UBound(Temp) + 1) = Application.Transpose(Temp)
    RR = RR + UBound(Temp) + 2
  Next
End Sub

ronrosenfeld say:

I used to worry about compact code when writing in machine language with limited memory resources.  Not so much anymore.

However, I do find a significant speed advantage in the writing of everything to a VBA array, and then writing to the worksheet in a single step, rather than writing line by line and also having to deal with the Transpose function.  Hence the collection and the rest of the code I used which you did not.

Of course, whether the speed advantage is meaningful (i.e. worth the extra complexity) would have to do with the size of the data to be processed. And whether the compactness is an issue would depend on the overall size of the procedure.
