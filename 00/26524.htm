cristin yan say:

As you know, the main issue is that how to calculate the intersection of working time period and allowance time period. It looks very interesting.

 

Here are some codes you could refer.

 

In a Module:

 

Option Explicit

 

'Special hour point from.

Public Const intHSF As Integer = 22

'Special hour point to.

Public Const intHST As Integer = 6

'Allowance per minute.

Public Const sngAPM As Single = 1 / 30

 

'######################################################

'【Function Name】

' CalcAllowance

'【Parameter Meaning】

' [1]. tmWorkingFrom    -->>    Time working from;

' [2]. tmWorkingTo      -->>    Time working to;

' [3]. sngAPM           -->>    Allowance per minute;

'【Return Type】

' Single    -->>    Result of Allowance.

'######################################################

Public Function CalcAllowance(ByVal intHSF As Integer, _

                              ByVal intHST As Integer, _

                              ByVal tmWorkingFrom As String, _

                              ByVal tmWorkingTo As String, _

                              ByVal sngAPM As Single) As Single

   

    'Time difference by minute.

    Dim tmMinuteDiff As Integer

    'Time difference by hour.

    Dim tmHourDiff As Integer

   

    'An array of wroking hour periods.

    Dim arrWH(1) As Integer

   

    '╱＊Get the hour part of working time.＊╱

    arrWH(0) = Hour(tmWorkingFrom)

    arrWH(1) = Hour(tmWorkingTo)

   

    'An array of wroking minute periods.

    Dim arrWM(1) As Integer

   

    '╱＊Get the minute part of working time.＊╱

    arrWM(0) = Minute(tmWorkingFrom)

    arrWM(1) = Minute(tmWorkingTo)

   

    tmHourDiff = GetHourDiff(intHSF, intHST, arrWH(0), arrWH(1))

   

    '╱＊Calculate time difference by minute.＊╱

    If arrWM(1) >= arrWM(0) Then

        tmMinuteDiff = ((tmHourDiff - 1) * 60 + arrWM(1)) - arrWM(0)

    Else

        tmMinuteDiff = ((tmHourDiff - 2) * 60 + 60 + arrWM(1)) - arrWM(0)

    End If

   

    If tmHourDiff <> 0 Then CalcAllowance = tmMinuteDiff * sngAPM

End Function

 

'###################################################################

'【Function Name】

' GetHourDiff

'【Parameter Meaning】

' [1]. intHourSpecialFrom   -->>    Special hour point from;

' [2]. intHourSpecialTo     -->>    Special hour point to;

' [3]. intHourWorkFrom      -->>    Working hour point from;

' [4]. intHourWorkTo        -->>    Working hour point to.

'【Return Type】

' Integer   -->>    Hour difference.

'【Notification】

' All parameters must be set between 0 and 23 because the time is

' always represented in the 24-hour system.

'###################################################################

Private Function GetHourDiff(ByVal intHourSpecialFrom As Integer, _

                             ByVal intHourSpecialTo As Integer, _

                             ByVal intHourWorkFrom As Integer, _

                             ByVal intHourWorkTo As Integer) As Integer

   

    Dim arrHS   'An array containing special hour periods.

    Dim arrHW   'An array containing work hour periods.

   

    '╱＊Check if the time period within 24 hours.＊╱

    If (intHourSpecialFrom < 0) Or (intHourSpecialFrom > 23) Or _

       (intHourSpecialTo < 0) Or (intHourSpecialTo > 23) Or _

       (intHourWorkFrom < 0) Or (intHourWorkFrom > 23) Or _

       (intHourWorkTo < 0) Or (intHourWorkTo > 23) Then Exit Function

   

    '╱＊Generate two arrays.＊╱

    arrHS = GenerateHourPeriodToArray(intHourSpecialFrom, intHourSpecialTo)

    arrHW = GenerateHourPeriodToArray(intHourWorkFrom, intHourWorkTo)

   

    Dim m%, n%

    Dim intCountHour%

   

    intCountHour = 0

   

    '╱＊Compare two arrays to count the special hour number.＊╱

    For m = 0 To UBound(arrHW)

        For n = 0 To UBound(arrHS)

            If arrHW(m) = arrHS(n) Then

                intCountHour = intCountHour + 1

                Exit For

            End If

        Next

    Next

   

    GetHourDiff = intCountHour

End Function

 

'##################################################################

'【Function Name】

' GenerateHourPeriodToArray

'【Parameter Meaning】

' [1]. intWorkingHourFrom   -->>    Working hour point from;

' [2]. intWorkingHourTo     -->>    Working hour point to.

'【Return Type】

' Variant   -->>    If success, it will return an Integer array.

'【Notification】

' All parameters must be set between 0 and 23 because the time is

' always represented in the 24-hour system.

'##################################################################

Private Function GenerateHourPeriodToArray(ByVal intWorkingHourFrom As Integer, ByVal intWorkingHourTo As Integer) As Variant

    Dim i%, k%

   

    k = 0

    i = intWorkingHourFrom

   

    '╱＊Check if the wroking hour period within 24 hours.＊╱

    If (intWorkingHourFrom < 0) Or (intWorkingHourFrom > 23) Or _

       (intWorkingHourTo < 0) Or (intWorkingHourTo > 23) Then Exit Function

   

    Dim arrTemp%()

   

    '╱＊Do a loop to generate 'arrTemp'.＊╱

    Do

        ReDim Preserve arrTemp(k)

        arrTemp(k) = i

        i = i + 1

        k = k + 1

       

        '╱＊Add the last hour number to 'arrTemp'.＊╱

        If i = intWorkingHourTo Then

            ReDim Preserve arrTemp(k)

            arrTemp(k) = i

            Exit Do

        End If

       

        '╱＊Reset the value of 'i' to 0 if it is up to 24.＊╱

        If i = 24 Then i = 0

    Loop

   

    GenerateHourPeriodToArray = arrTemp

End Function

 

In a Worksheet(default name is Sheet1):

 

Option Explicit

 

Private Sub Worksheet_Change(ByVal Target As Range)

    Application.EnableEvents = False

   

    Dim iRow&, iColumn&

   

    iRow = Target.Row

    iColumn = Target.Column

   

    Dim tmWorkingFrom$, tmWorkingTo$

   

    On Error GoTo errHandler

   

    '╱＊If column number is 2 or 3.＊╱

    If iColumn = 2 Or iColumn = 3 Then

        '╱＊If column number is 2.＊╱

        If iColumn = 2 Then

            tmWorkingFrom = Target.Text

            tmWorkingTo = Target.Offset(0, 1).Text

        End If

       

        '╱＊If column number is 3.＊╱

        If iColumn = 3 Then

            tmWorkingFrom = Target.Offset(0, -1).Text

            tmWorkingTo = Target.Text

        End If

    End If

   

    '╱＊If 'tmWorkingFrom' and 'tmWorkingTo' is a date format.＊╱

    If IsDate(tmWorkingFrom) And IsDate(tmWorkingTo) Then

        Range("G" & iRow).Value2 = CalcAllowance(intHSF, intHST, tmWorkingFrom, tmWorkingTo, sngAPM)

    Else

        Range("G" & iRow).Value2 = ""

    End If

   

    Application.EnableEvents = True

    Exit Sub

   

errHandler:

    Range("G" & iRow).Value2 = ""

    Application.EnableEvents = True

End Sub

[2]. Forget that the constant ‘sngAPM’ is set by minute.
