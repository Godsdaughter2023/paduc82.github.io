Sub ControlSub()

    Dim wsSumList As Worksheet
    On Error Resume Next    'Next line errors if worksheet does not exist
    Set wsSumList = Worksheets("SumList")
    On Error GoTo 0     'Resume error trapping as soon as possible
   
    If wsSumList Is Nothing Then 'If Nothing then worksheet does not already exist
        'Create a temporary worksheet for Summary column of data
        Set wsSumList = Sheets.Add(After:=Sheets(Sheets.Count))
        wsSumList.Name = "SumList"
    Else
        wsSumList.Cells.Clear    'If SumList already exists then clear all previous data
    End If
   
    wsSumList.Cells(1, "B") = "Link to Output Sheet"
    wsSumList.Cells(1, "D") = "Link to Output Sheet"
    wsSumList.Rows(1).Font.Bold = True
   
    'R is the column with filter values, A is destination for the SumList list
    Call ExtractData(Worksheets("Working File"), "R", "A")
    Worksheets("Working File").AutoFilterMode = False
   
    'U is the column with filter values, C is destination for the SumList list
    Call ExtractData(Worksheets("Scanned file"), "U", "C")
    Worksheets("Scanned file").AutoFilterMode = False
   
    wsSumList.Columns.AutoFit
    Application.Goto wsSumList.Cells(1, "A")
End Sub

Sub ExtractData(ws As Worksheet, strColSource As String, strColDest As String)
    Dim wsFiltList As Worksheet
    Dim rngUnique As Range
    Dim cel As Range
    Dim wsOutput As Worksheet
    Dim lngLastRow As Long
    Dim lngLastCol As Long
    Dim rngFiltVisible As Range
    Dim strWsName As String
    Dim lngFiltCol As Long
   
    Set wsFiltList = Worksheets("SumList")
   
    ws.AutoFilterMode = False   'Start by turning off AutoFilter to cancel any previous filter settings
    ws.Columns(strColSource).Copy   'Copy the Summary column
    wsFiltList.Cells(1, strColDest).PasteSpecial Paste:=xlPasteValues    'Paste values only to temp worksheet
   
    wsFiltList.Columns(strColDest).RemoveDuplicates Columns:=1, Header:=xlYes  'Remove any duplicates from list
    With wsFiltList
        'Assign the unique list from 2nd row to end of data to a range variable
        Set rngUnique = .Range(.Cells(2, strColDest), .Cells(.Rows.Count, strColDest).End(xlUp))
    End With
   
    lngLastRow = LastRowOrCol(ws, True)  'True finds last used row on sheet
    lngLastCol = LastRowOrCol(ws, False) 'False finds last used column on sheet
    lngFiltCol = ws.Columns(strColSource).Column   'Convert the alpha column to numeric for the filter Field
   
    For Each cel In rngUnique
        If Trim(cel.Value) <> "" Then    'Don't attempt to process blanks
            With ws
                .AutoFilterMode = False     'Start by turning off AutoFilter to cancel any previous filter settings
                .Range(.Cells(1, "A"), .Cells(lngLastRow, lngLastCol)).AutoFilter
                .AutoFilter.Range.AutoFilter Field:=lngFiltCol, Criteria1:=cel.Value
            End With
       
            With ws.AutoFilter.Range
                Set rngFiltVisible = .SpecialCells(xlCellTypeVisible)
            End With
           
            Set wsOutput = Sheets.Add(After:=Sheets(Sheets.Count))
            strWsName = cel.Value
            strWsName = Replace(strWsName, "/", "-") 'Worksheet names cannot contain slashes
            strWsName = Left(strWsName, 31) 'Can only have 31 characters in a worksheet name
           
            Err.Number = 0
            On Error Resume Next
            wsOutput.Name = strWsName
            On Error GoTo 0
           
            'Insert Hyperlink to the worksheet
            cel.Offset(0, 1).FormulaR1C1 = "=HYPERLINK(" & Chr(34) & "[" & ThisWorkbook.Name & "]'" _
                            & wsOutput.Name & "'!A1" & Chr(34) & "," & Chr(34) & wsOutput.Name & Chr(34) & ")"
           
            rngFiltVisible.Copy Destination:=wsOutput.Cells(1, 1)
        End If
    Next cel
   
End Sub


Function LastRowOrCol(ws As Worksheet, bolRowCol As Boolean, Optional rng As Range) As Long
    'Finds the last used row or column in a worksheet
    'First parameter is Worksheet
    'Second parameter is True for Last Row or False for last Column
    'Third parameter is optional. Use to find the last row or column in a specific range
   
    Dim lngRowCol As Long
    Dim rngToFind As Range
   
    ws.AutoFilterMode = False   'Does not work if filters applied
   
    If rng Is Nothing Then
        Set rng = ws.Cells
    End If
   
    If bolRowCol Then
        lngRowCol = xlByRows
    Else
        lngRowCol = xlByColumns
    End If
   
    With ws
        Set rngToFind = rng.Find(What:="*", _
                After:=.Cells(1, "A"), _
                LookIn:=xlFormulas, _
                LookAt:=xlPart, _
                SearchOrder:=lngRowCol, _
                SearchDirection:=xlPrevious, _
                MatchCase:=False)
    End With
   
    If Not rngToFind Is Nothing Then
        If bolRowCol Then
            LastRowOrCol = rngToFind.Row
        Else
            LastRowOrCol = rngToFind.Column
        End If
    End If
   
End Function


Sub DeleteWorksheets()
    'Used during testing to delete the created worksheets

    Dim ws As Worksheet
   
    Application.DisplayAlerts = False
    For Each ws In Worksheets
        'The Case text comparison is case sensitive so convert all to Upper Case
        'to avoid problems with incorrect case being used in the code or worksheet names
        Select Case UCase(ws.Name)  'Converts worksheet name to upper case for comparison
            Case "DIAGNOSTICS", "WORKING FILE", "SCANNED FILE"
                'DO NOTHING
            Case Else
                ws.Delete
        End Select
    Next ws
    Application.DisplayAlerts = True

End Sub

ken sheridan say:

In QBdata, the amount field is also defined as a Number, field size Double, format Currency, decimal places 2 (just as in the officeData table).

Why are you not using columns of Currency data type?  The discrepancies which you are experiencing are almost certainly due to your use of double precision floating point number data types.  The currency data type is specifically designed to handle currency values, storing such values to a precision of four decimal places to prevent cumulative rounding errors in normal financial transactions.  The reason you are finding that apparently equal values do not match even at a precision of four decimal places is probably due to their not matching at the higher precision in which the data is currently stored.  The format at which you see the values is irrelevant; it is merely how the values are presented, not the values themselves.

This is why Tom is recommending that you test the miscreant values in columns of currency data type in two new tables.

Hi,

 

This extracts unuqie items from Sheet1 A1:A20. It's an ARRAY formula, see below on how to enter it.

 

=IFERROR(INDEX(Sheet1!$A$1:$A$20,MATCH(0,IF(ISBLANK(Sheet1!$A$1:$A$20),"",COUNTIF(A$1:$A1,Sheet1!$A$1:$A$20)),0)),"")

jay freedman say:
I suspect the Styles pane has moved off-screen.

The following procedure isn't guaranteed to bring it back where it belongs. Because you can't see it, you won't know when it has the focus, or which direction to move it, so it will take some experimentation if it works at all.

Start with a new blank document, and close all other panes (Navigation, Research, etc.).

Click File > Options > Quick Access Toolbar. Set the category above the left-hand list to Commands Not In the Ribbon. Scroll down until you can click on the command named Style (it has a blue square at the right side of the list), and then click the Add button between the lists. That adds a Style box to the Quick Access Toolbar that shows the name of the style at the current cursor location. You need this for the next step.

Click in the blank document and press F6. This might move the focus to the Styles pane, if it's open but off-screen. To find out, press the down arrow key once or twice and then press Enter. If the Style box on the Quick Access Toolbar changes from Normal to something else, you know that the Style pane had the focus. If nothing happened, click the button for Show the Styles Window (or press Ctrl+Alt+Shift+S), click in the blank document, and try F6/arrow down/Enter again.

When you know that the Style pane is open, click in the document and press F6 one more time. The Style pane now has focus, and it's time to try to move it inside the Word window. Press Ctrl+Alt+spacebar, release it, and press M. That sequence opens a small menu (usually available by clicking a triangular arrow in the title bar of the pane) and selects the Move command.

Now press and hold one of the arrow keys. If the pane is off-screen, it's probably out to the right (the nearest side of the Word window to where it usually is), so try using the left arrow key to move it to the left. If that's correct, at some point the pane will suddenly pop in and dock at the right side of Word. If nothing happens within about 5 to 10 seconds, try moving it with the other arrow keys.

If none of this works, you can try deleting and rebuilding the Data key in the registry as described in http://word.mvps.org/FAQs/AppErrors/MissingMenusEtc.htm.  For Word 2010, the path of the key is HKEY_CURRENT_USER\Software\Microsoft\Office\14.0\Word\Data.

petersmerdon say: OK - no further response ... deadline staring me in the face, so I plugged away and solved it myself (sort of).
Here's what I found.
1. The audio is a string of objects inserted in the individual slides during the slide "Show Record" process (and each can be replayed with a click on the speaker icon on the slide).

2. The audio transition at slide changes involves a slow audio fade down, a brief silence, and a slow fade up on the audio from the following slide. 
Two consequences: 
(a) there must be no narration through the slide transitions, 
and 
(b) for pre-recorded narration the background noise during this pause must be very low - even with typical lecture room noise (mainly air-con) fade-down/fade-up can be easily perceived.

3. The recorded timings on the first couple of slides were always short by up to 2 seconds resulting in premature slide change & clipped-off audio.

My workarounds, which achieved a (largely) acceptable result were to:
(i) Run a noise removal tool in my audio editor on the audio file (Audacity - "Noise Removal" in Effects) to get the background noise low enough to make the transition's total silence period less obvious
(ii) Edit the audio to add an extra second or so of room noise at each slide-change point - making the manual "next slide" click timing more forgiving.
(iii) When the recording has finished, check the transition timings against each slide's audio length, and change the timing entry if different.

This gives an accurate result with correct audio - but to the critical listener it still sounds like a bunch of slides with the audio crudely "stitched together" at each transition.
