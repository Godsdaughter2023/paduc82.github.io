ken sheridan say: Add the following procedure to a standard module in the database:

Public Sub DeleteTables(strExcludeTable As String)

    Dim tdf As DAO.TableDef
    Dim strSQL As String
    Dim strTable As String
    
    
    For Each tdf In CurrentDb.TableDefs
        strTable = tdf.Name
        If Left(strTable, 4) <> "MSys" Then
            If strTable <> strExcludeTable Then
                strSQL = "DROP TABLE " & strTable
                CurrentDb.Execute strSQL, dbFailOnError
            End If
        End If
    Next tdf
    
End Sub

Call it with:

DeleteTables "TableMaster"

ken sheridan say: Change the function so that it accepts a parameter array as its argument:

Public Sub DeleteTables(ParamArray ExcludeTables())

    Dim tdf As DAO.TableDef
    Dim strSQL As String
    Dim strTable As String
    Dim varItem As Variant
    Dim blnDeleteTable As Boolean

    
    For Each tdf In CurrentDb.TableDefs
        blnDeleteTable = True
        strTable = tdf.Name
        If Left(strTable, 4) <> "MSys" Then
            For Each varItem In ExcludeTables()
                If strTable = varItem Then
                    blnDeleteTable = False
                    Exit For
                End If
            Next varItem
            If blnDeleteTable Then
                strSQL = "DROP TABLE " & strTable
                CurrentDb.Execute strSQL, dbFailOnError
            End If
        End If
    Next tdf
    
End Sub
