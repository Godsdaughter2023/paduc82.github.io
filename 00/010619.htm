paul domaskis say:

I actually use Find (code below).  Function caller provides the table as a range, specifies the text for the column heading containing the row names, then specifies the row name and the column name of the data of interest to extract the cell.  The problem is that a particular table of mine has headings consisting of multi-line equations using greek letters and such.  Providing this as a label makes for cryptic code.  I really need a version of this function that allows the invoker to specify the column of interest as a range.  (It doesn't make sense to specify the row as a range because that's what's going to be looked up repetitively with different values).

 

Function tblLook_rowColNames( _
   rgTbl As Range, _
   stRowNameHed As String, _
   stRowName As String, _
   stColName As String, _
   Optional wutIfEmpty As Variant _
) As Variant

    Dim rgRowNameHed As Range
   
    'Determine column to search for stRowName
    Set rgRowNameHed = rgTbl.Rows(1).Find( _
        What:=stRowNameHed, After:=rgTbl(rgTbl.Columns.Count), _
        LookIn:=xlValues, LookAt:=xlWhole, _
        SearchDirection:=xlNext, MatchCase:=True _
    )

    Set tblLook_rowColNames = rgTbl.Parent.Cells( _
        rgTbl.Columns(rgRowNameHed.Column - rgTbl.Column + 1).Find( _
            What:=stRowName, After:=rgRowNameHed, _
            LookIn:=xlValues, LookAt:=xlWhole, _
            SearchDirection:=xlNext, MatchCase:=True _
        ).Row, _
        rgTbl.Rows(1).Find( _
            What:=stColName, After:=rgTbl(rgTbl.Columns.Count), _
            LookIn:=xlValues, LookAt:=xlWhole, _
            SearchDirection:=xlNext, MatchCase:=True _
        ).Column _
    )

    If IsEmpty(tblLook_rowColNames) And Not IsMissing(wutIfEmpty) Then
        tblLook_rowColNames = wutIfEmpty
    End If

End Function

   Option Explicit

   Function tblLook_rowColNames( _
       rgTbl As Range, _
       stRowNameHed As String, _
       stRowName As String, _
       stColName As String, _
       Optional wutIfEmpty As Variant, _
       Optional wutIfNotFound As Variant _
   ) As Variant

       'rgTbl = The table to look in
       'stRowNameHed = Column heading for column of row headings
       'stRowName = Row heading of row being sought
       'stColName = column heading of column being sought
       'wutIfEmpty = What to return if looked up cell is empty
       'wutIfNotFound =
       '               What to return if stRowName or stColName not found

       Dim rgRowNameHed As Range
       Dim celInRow As Range
       Dim celInCol As Range

       'Determine column to search for stRowName
       Set rgRowNameHed = rgTbl.Rows(1).Find( _
           What:=stRowNameHed, After:=rgTbl(rgTbl.Columns.Count), _
           LookIn:=xlValues, LookAt:=xlWhole, _
           SearchDirection:=xlNext, MatchCase:=True _
       )

       Set celInRow = rgTbl.Columns( _
           rgRowNameHed.Column - rgTbl.Column + 1 _
       ).Find( _
           What:=stRowName, After:=rgRowNameHed, _
           LookIn:=xlValues, LookAt:=xlWhole, _
           SearchDirection:=xlNext, MatchCase:=True _
       )
       Set celInCol = rgTbl.Rows(1).Find( _
           What:=stColName, After:=rgTbl(rgTbl.Columns.Count), _
           LookIn:=xlValues, LookAt:=xlWhole, _
           SearchDirection:=xlNext, MatchCase:=True _
       )
       If _
           ((celInRow Is Nothing) Or (celInCol Is Nothing)) _
           And Not IsMissing(wutIfNotFound) _
       Then
           tblLook_rowColNames = wutIfNotFound
           Exit Function
       End If

       Set tblLook_rowColNames = _
           rgTbl.Parent.Cells(celInRow.Row, celInCol.Column)

       If IsEmpty(tblLook_rowColNames) And Not IsMissing(wutIfEmpty) Then
           tblLook_rowColNames = wutIfEmpty
       End If

   End Function

tony chen chn say:

Hi,

You get this error probably because the database that keeps track of the pictures and videos in your collection becomes corrupted. For the fix, rebuild your database as follows:

1. Close Windows Live Photo Gallery
2. Click Start
3. Click My Computer
4. Paste this address in the Address field %userprofile%\Local Settings\Application Data\Microsoft\Windows Live Photo Gallery
5. Right click Pictures.pd5
6. Click Rename
7. Change the file name to OLD_Pictures.pd5
8. Open Windows Live Photo Gallery

Now Windows Live Photo Gallery will rebuild its database, which should stop the error you are receiving from occurring.
