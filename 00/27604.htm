bernie deitrick, excel mvp 2000-2010 say:

To get the unique values into once string is tricky, because you need to use an array of arrays, not always possible with a single formula.

You could use a UDF, code below, used like

=AppleList(Apple!A2:A100,Apple!C2:C100,"Apple")

Copy this code into a codemodule in your workbook:


Function AppleList(T As Range, R As Range, strM As String) As String
    Dim i As Long
    If T.Cells.Count <> R.Cells.Count Then
        AppleList = "Unbalanced entry"
        Exit Function
    End If
    AppleList = ""
    For i = 1 To T.Cells.Count
        If T.Cells(i).Value = strM Then
            If AppleList = "" Then AppleList = R.Cells(i).Value
       
            If InStr(1, AppleList, R.Cells(i).Value) = 0 Then
                AppleList = AppleList & ", " & R.Cells(i).Value
            End If
            End If
       
    Next i
End Function

HTH, Bernie
