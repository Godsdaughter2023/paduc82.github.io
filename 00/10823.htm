            ActiveSheet.Hyperlinks.Add Cell, Trim(Cell.Value)
        End If
    Next
End Sub

Sub HyperAdd()
' Converts each text hyperlink selected into a working hyperlink
    For Each xCell In Selection
        ActiveSheet.Hyperlinks.Add Anchor:=xCell, Address:=xCell.Formula
    Next xCell
End Sub
