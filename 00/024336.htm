a. user say:

I found root cause and solution

Issues of workflow tasks not starting and error while creating new 'Team Site' were happening because of site columns of Tasks content type were somehow deleted. So Task content types had not columns associated with them.

Solution: Deactivate/Activate Site columns and Site content types feature using powershell

Adding below powershell script code for others reference



function Change-Feature
{
  
   param (
   [Parameter(Mandatory=$true,Position=1)]
[string]$Username,
[Parameter(Mandatory=$true,Position=2)]
$AdminPassword,
        [Parameter(Mandatory=$true,Position=3)]
[string]$Url
)
  
  $ctx=New-Object Microsoft.SharePoint.Client.ClientContext($Url)
  $ctx.Credentials = New-Object Microsoft.SharePoint.Client.SharePointOnlineCredentials($Username, $AdminPassword)
  $ctx.ExecuteQuery() 

        
  $ctx.Load($ctx.Site);
  $ctx.ExecuteQuery();

  #deactivate site columns feature
  $guid = [System.Guid]"ca7bd552-10b1-4563-85b9-5ed1d39c962a"
  $ctx.Site.Features.Remove($guid,$true)
  $ctx.ExecuteQuery();                
  
  #deactivate site content type feature
  $guid = [System.Guid]"695b6570-a48b-4a8e-8ea5-26ea7fc1d162"
  $ctx.Site.Features.Remove($guid,$true)
  $ctx.ExecuteQuery();                
  
  #activate site columns feature
  $guid = [System.Guid]"ca7bd552-10b1-4563-85b9-5ed1d39c962a"
  $ctx.Site.Features.Add($guid,$true,[Microsoft.SharePoint.Client.FeatureDefinitionScope]::None)
  $ctx.ExecuteQuery();                

  #activate site content type feature
  $guid = [System.Guid]"695b6570-a48b-4a8e-8ea5-26ea7fc1d162"
  $ctx.Site.Features.Add($guid,$true,[Microsoft.SharePoint.Client.FeatureDefinitionScope]::None)
  $ctx.ExecuteQuery();                
        
}



# Insert the credentials and the name of the admin site
$Username="*** Email address is removed for privacy ***"
$AdminPassword=Read-Host -Prompt "Password" -AsSecureString
$AdminUrl="yoursite-name"

Change-Feature -Username $Username -AdminPassword $AdminPassword -Url $AdminUrl
