andreas killer say:
Sub MergeWorkbooks()
  Dim WB As Workbook
  Dim WS As Worksheet
 
  With Application.FileDialog(msoFileDialogOpen)
    'Ask the user for the files to import
    .AllowMultiSelect = True
    .Title = "Select files to import"
    .ButtonName = "Import"
    .FilterIndex = 2 'Excel files by default
    If Not .Show Then Exit Sub
    'Screen off, runs faster
    Application.ScreenUpdating = False
    'For each file
    For i = 1 To .SelectedItems.Count
      'Open it readonly as is
      Set WB = Workbooks.Open(.SelectedItems(i), False, True)
      'For each worksheet
      For Each WS In WB.Worksheets
        'Copy the sheet into our file
        WS.Copy After:=ThisWorkbook.Sheets(ThisWorkbook.Sheets.Count)
        'Refer it
        Set WS = ThisWorkbook.Sheets(ThisWorkbook.Sheets.Count)
        'Remove the formulas
        WS.UsedRange = WS.UsedRange.Value
      Next
      'Close the file
      WB.Close False
    Next
    Application.ScreenUpdating = True
  End With
End Sub
