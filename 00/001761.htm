Try:

'=========>>
Option Explicit

'--------->>
Public Function AlphaOnly(sStr As String) 
    Dim oRegExp As Object

    Set oRegExp = CreateObject("VBScript.RegExp")

    With oRegExp
        .IgnoreCase = True
        .Global = True
        oRegExp.Pattern = "\d"

        AlphaOnly = oRegExp.Replace(sStr, vbNullString)
    End With
End Function
'<<=========

This function can be used as a UDF on a worksheet.  For example:

   =AlphaOnly(A1)

It may also be used in VBA, e.g:

'=========>>
Public Sub Tester()
    Dim sStr As String

    sStr = AlphaOnly("abc123defGH")
    MsgBox sStr
End Sub
