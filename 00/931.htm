[2]. Add a module named "basWebAccess" to the VBA project, and then copy the following code to the module:

Option Explicit

'******************************************
'Enumeration:Cutting types of string.
'******************************************
Public Enum StringCuttingTypes
    IncludingBegin = -1
    IncludingNone = 0
    IncludingEnd = 1
    IncludingAll = 2
End Enum

Public objXmlHttp           As New xmlHttp  'XMLHTTP Object.
Public strPostedResponse    As String       'Postback data from server.

'#################
'For URL-Encoding.
'#################
Public Function URLEncode(ByVal strToEncode As String, _
    Optional blnUsePlusRatherThanHexForSpace As Boolean = False) As String
   
    Dim strTemp As String
    Dim intCurrentChar As Integer
   
    intCurrentChar = 1
   
    Do Until intCurrentChar - 1 = Len(strToEncode)
        Select Case Asc(Mid(strToEncode, intCurrentChar, 1))
            Case 48 To 57, 65 To 90, 97 To 122
                strTemp = strTemp & Mid(strToEncode, intCurrentChar, 1)
            Case 32
                strTemp = IIf(blnUsePlusRatherThanHexForSpace, _
                              strTemp & "+", _
                              strTemp & "%" & Hex(32))
        Case Else
            strTemp = strTemp & "%" & _
                      Format(Hex(Asc(Mid(strToEncode, intCurrentChar, 1))), "00")
        End Select
        intCurrentChar = intCurrentChar + 1
    Loop
   
    URLEncode = strTemp
End Function

'#################
'For URL-Decoding.
'#################
Public Function URLDecode(strToDecode As String) As String
    Dim strTemp As String
    Dim intCurrentChar As Integer
   
    intCurrentChar = 1
   
    Do Until intCurrentChar - 1 = Len(strToDecode)
      Select Case Mid(strToDecode, intCurrentChar, 1)
        Case "+"
          strTemp = strTemp & " "
        Case "%"
          strTemp = strTemp & Chr(Val("&h" & _
             Mid(strToDecode, intCurrentChar + 1, 2)))
           intCurrentChar = intCurrentChar + 2
        Case Else
          strTemp = strTemp & Mid(strToDecode, intCurrentChar, 1)
      End Select
   
    intCurrentChar = intCurrentChar + 1
    Loop
   
    URLDecode = strTemp
End Function

'##############################################
'【Function Name】
' CutString.
'【Function Description】
' Cut string according to given conditions.
'【Parameter Meaning】
' [1].strOri    -->> The original string；
' [2].strBegin  -->> The beginning string;
' [3].strEnd    -->> The ending string；
' [3].sctOption -->> Cutting types of string.
'【Return Type】
' String    -->>    The cutted string.
'##############################################
Public Function CutString(ByVal strOri As String, _
    ByVal strBegin As String, _
    ByVal strEnd As String, _
    Optional ByVal sctOption As StringCuttingTypes = IncludingAll) As String
   
    Dim lBeginPos As Long, lEndPos As Long
   
    lBeginPos = InStr(strOri, strBegin)
    lEndPos = InStr(strOri, strEnd)
   
    If lBeginPos <> 0 And lEndPos <> 0 And lEndPos >= lBeginPos Then
        Select Case sctOption
            Case IncludingBegin
                CutString = Mid(strOri, lBeginPos, (lEndPos - lBeginPos))
            Case IncludingNone
                CutString = Mid(strOri, lBeginPos + 1, (lEndPos - lBeginPos) - 1)
            Case IncludingEnd
                CutString = Mid(strOri, lBeginPos + Len(strBegin), ((lEndPos + Len(strEnd)) - (lBeginPos + Len(strBegin))))
            Case IncludingAll
                CutString = Mid(strOri, lBeginPos, ((lEndPos + Len(strEnd)) - lBeginPos))
        End Select
    End If
End Function

'#############################################
'【Function Name】
' GetHtml.
'【Function Description】
' Get the source code for the specified Url.
'【Parameter Meaning】
' strUrl    -->>    The URL string；
'【Return Type】
' String    -->>    The source code string.
'#############################################
Public Function GetHtml(ByVal strUrl As String) As String
    GetHtml = ""
    Dim xmlHttp As Object
    Set xmlHttp = CreateObject("Microsoft.XMLHTTP")
    xmlHttp.Open "GET", strUrl, False
    xmlHttp.sEnd
    If xmlHttp.readyState = 4 Then GetHtml = StrConv(xmlHttp.responseBody, vbUnicode)
    Set xmlHttp = Nothing
End Function

'###################################################
'Get any section of the delimiter-separated string.
'###################################################
Public Function Get_nStr(ByVal wStr As String, _
    ByVal Mark As String, _
    ByVal n As Integer, _
    Optional ByVal Chk As Boolean = False, _
    Optional d As Integer) As String
   
    Dim wArr
   
    If (wStr = "") Or (Mark = "") Then Exit Function
    If Len(Mark) > Len(wStr) Then Exit Function
   
    wArr = Split(wStr, Mark)
    d = UBound(wArr) + 1
   
    If d = 1 Then
        d = 0
        Exit Function
    Else
        If n < 1 Then Exit Function
    End If
   
    If Chk = True Then n = d - n + 1
    Get_nStr = wArr(n - 1)
End Function

The ISODD/ISEVEN functions are part of the Analysis ToolPak add-in in excel versions 2003 and earlier. Since the ATP is a separate file that's why you get the error.

These functions were added to the Excel application in Excel 2007 so you should not get the error in that version (and later).

If you need something that is backward compatible (will work in any version of Excel) then you can use something like this.

For ISODD:

MOD(cell_ref,2)>0

For ISEVEN:

MOD(cell_ref,2)=0
