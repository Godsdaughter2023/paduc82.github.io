Public Function GetFinishDate(datFinish as Date) as String
' this function will accept the Finish1 field and

Select Case datFinish

GetFinishDate = "TBD"

GetFinishDate = "ASAP"
Case #10/1/2000#
GetFinishDate = "N/A"
Case Else
GetFinishDate = Month(datFinish) & "/" & Day(datFinish)
End Select
End Function
