hansv mvp say:

Without VBA:


Press Ctrl+H.

Click More >>.

Click 'Use wildcards'.


Enter (_)(*)(_) in the Find what box.

Enter \2 in the Replace box.

With the insertion point still in the Replace with box, click Format, then Font...

Tick the Subscript check box, then click OK.

Click Replace All.


Change the text in the Find what box to (^94)(*)(^94)

^94 is the code for ^ (^ has a special meaning so you can't use it directly)

Click in the Replace with box.

Click Format, then Font...

Tick the Superscript check box (you'll probably have to click it twice), then click OK.

Click Replace All.

If you have to have a macro after reading Hans' reply you can use this:

Sub Demo()
Dim oRng As Word.Range
Dim arrTerms() As String
Dim lngIndex As Long

  arrTerms = Split("(_)(*)(_)|(^94)(*)(^94)", "|")
  For lngIndex = 0 To UBound(arrTerms)
    Set oRng = ActiveDocument.Range
    With oRng.Find
      .ClearFormatting
      .Text = arrTerms(lngIndex)
      .Format = True
      With .Replacement
        .ClearFormatting
        If lngIndex = 0 Then
          .Font.Subscript = True
        Else
          .Font.Superscript = True
        End If
        .Text = "\2"
      End With
      .MatchWildcards = True
      .Execute Replace:=wdReplaceAll
    End With
  Next lngIndex
End Sub

    Macro to create list of files in a directory

    Sub GrabFolderFiles()

    '

    ' GrabFolderFiles Macro

    ' Macro created 22/12/98 by Peter Paul William Weiss

    ' Grab Folder Files: I sometimes like to have a print out of the contents

    ' of a folder/sub directory. This macro does the job. To run the macro, point

    ' Word to the folder in question, using File | Open. Then close the "Open

    ' Dialog" box. Run the macro, which will create a new page and provide a "list"

    ' of the files - name, date last modified and file size on a newly created

    ' page. The date is set up for the European system, you might have to modify

    ' for the US.

     

    ' declare our variables

    Dim cDir$, cFile$

    ' Our variables for the directory and files.

    Dim cExtension$

    ' The type of file we are looking for

     

    Application.Documents.Add

    ' create a new document

     

    cDir = Application.Options.DefaultFilePath(wdCurrentFolderPath) & "\"

    ' the directory we are looking grabbing our files

    ' from. Change this if you do not want the default

    ' Word document path

    cExtension = "*"

    ' the type of files we are looking for

    ' change to "*" if you want all files.

     

    cFile = Dir(cDir & "*." & cExtension, vbNormal)

    ' Retrieve the first file.

     

    Do While cFile <> ""

    ' Start the loop and stay

    ' there until there are no more files.

    ' exits immediately if no file found

     

    Selection.TypeText Text:=cFile & _

    vbTab & _

    Format(FileDateTime(cDir & cFile), "Long Date") & _

    vbTab & _

    Format(FileLen(cDir & cFile), "###,###,###,###b")

    'type the filename, a tab, File date, tab and file size

     

    Selection.TypeParagraph

    ' and hit the enter key

     

    cFile = Dir

    ' Get next file name.

     

    Loop

    ' until finished with all the files

     

    Selection.WholeStory

    ' select the entire text

     

    Selection.ConvertToTable _

    Separator:=wdSeparateByTabs, _

    NumColumns:=3, _

    Format:=wdTableFormatNone

     

    ' and convert it to a table. This is why we

    ' used a tab when we keyed in the details

     

    Selection.Cells.AutoFit

    ' resize the table to fit all cells

     

    Selection.ParagraphFormat.Alignment = wdAlignParagraphRight

    ' set the alignment to the right

     

    Selection.EndKey Unit:=wdStory

     

    Selection.TypeParagraph

    Selection.TypeParagraph

    Selection.TypeText Text:="These are the files in:-  "

    Selection.TypeText Text:=cDir

    Selection.TypeParagraph

     

    Selection.HomeKey Unit:=wdStory

     

    ' and go back to the beginning of the table

     

    End Sub
