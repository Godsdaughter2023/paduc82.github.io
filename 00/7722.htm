Sub DeleteRowsWithoutWord()
  Dim Col As Variant, LastRow As Long, LastCol As Long
  Const CharToLeave = "40"
  Const DataStartRow As Long = 3
  Const ColToSearch As String = "K"
  LastRow = Cells(Rows.Count, ColToSearch).End(xlUp).Row
  LastCol = Cells.Find(What:="*", SearchOrder:=xlByColumns, SearchDirection:=xlPrevious, LookIn:=xlFormulas).Column
  On Error Resume Next
  Application.ScreenUpdating = False
  Range(Cells(1, ColToSearch), Cells(LastRow, ColToSearch)).Copy Cells(1, LastCol + 1)
  With Range(Cells(1, LastCol + 1), Cells(LastRow, LastCol + 1))
    .Replace "*" & CharToLeave & "*", "", xlWhole, MatchCase:=False
    .SpecialCells(xlCellTypeConstants).EntireRow.Delete
  End With
  Application.ScreenUpdating = True
End Sub
