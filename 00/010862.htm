If the images are different sizes it wouldn't be simple to distribute them with vba. This might help a bit:

It is only a start you would need to work on it!

Sub toGrid_SortOf()
Dim oshp As Shape
Dim osld As Slide
Dim vert As Single
Dim hor As Single
Dim lrgW As Single
Set osld = ActiveWindow.Selection.SlideRange(1)
vert = 10
hor = 10
For Each oshp In osld.Shapes
If oshp.Type = msoPicture Then
If vert > ActivePresentation.PageSetup.SlideHeight + oshp.Height + 10 Then
hor = hor + lrgW + 10
lrgW = 0
vert = 10
End If
oshp.Left = hor
oshp.Top = vert
vert = vert + oshp.Height + 10
If oshp.Width > lrgW Then lrgW = oshp.Width + 10
End If
Next oshp
End Sub
