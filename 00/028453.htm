gord dibben say:

Try this routine.  Adjust ranges to suit.

Sub Add_Checkboxes()
   'user will add series of questions to A2:A21
   'create "Agree yes/no" checkboxes in B2:B21
   'assign true/false results to C2:C21
   'create formula in D2:D21 to return yes or no
   'hide column C
Dim myCBX As CheckBox
Dim myCell As Range
Dim RAN As Range

Set RAN = ActiveSheet.Range("B2:B21")
Application.ScreenUpdating = False
With ActiveSheet
      .CheckBoxes.Delete
    For Each myCell In RAN

    With myCell
        Set myCBX = .Parent.CheckBoxes.Add _
            (Top:=.Top, _
             Width:=12, _
             Left:=.Left + ((.Width - 12) / 2), _
             Height:=.Height)
             .Offset(0, 1).ColumnWidth = 0
             .RowHeight = 19
        With myCBX
            .LinkedCell = myCell.Offset(0, 1).Address(external:=True)
            .Caption = ""
            .Value = xlOff
        End With
    
    End With
    
    Next myCell
    
    With Range("D2")
      .Value = "=IF(C2=TRUE,""Yes"",""No"")"
      .AutoFill Destination:=Range("D2:D21"), Type:=xlFillDefault
    End With
    
    Range("A1").Value = "Question"
    Range("B1").Value = "Do you agree?"
    
End With
Application.ScreenUpdating = True
End Sub

Sub Add_Checkboxes_Across()
Dim myCBX As CheckBox
Dim myCell As Range
Dim RAN As Range

             .RowHeight = 24

             .Offset(-1, 0).HorizontalAlignment = xlCenter
             .Offset(1, 0).RowHeight = 0

        End With

    Next myCell
    
    With Range("I5")
