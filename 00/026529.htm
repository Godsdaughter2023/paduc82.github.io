The explicit rounding avoids anomalies of arithmetic with the binary representation of decimal fractions that often cause infinitesimal differences with the expected decimal result. For example, IF(10.10 - 10 = 0.10, TRUE) returns FALSE(!), whereas IF(ROUND(10.10 - 10, 2) = 0.10, TRUE) returns TRUE.

jllatham say:

By now you should have received the workbooks I sent with the Holiday indicating code in them.

Here is the code that does all the work.  Much, as you will note, has its origins at Chip Pearson's website.  My major contribution to this was to translate the algorithm for calculating Easter that is published by the U.S. Naval Observatory into VBA, and from there to calculate other related dates such as Ash Wednesday and Good Friday.

Note that when the ReturnType parameter of IsItAHoliday() is TRUE, then the date that the holiday is observed is returned, not necessarily the date of the holiday itself.  Example: if Christmas falls on a Saturday, then with ReturnType set to TRUE the returned date would be December 24 - the Friday before Christmas itself.

The complete code:

Option Explicit
'Rules for U.S. Holidays (from http://aa.usno.navy.mil/faq/docs/holidays.php
'Fixed by date  - if on Saturday, (may be) observed on preceding Friday
'               - if on Sunday, (may be) observed on following Monday
'New Year's Day:    January 1
'Independence Day:  July 4
'Veteran's Day:     November 11
'Christmas Day:     December 25
' The following holidays are observed on by date only:
'Martin Luther King's Birthday                     Third Monday in January
'Washington 's Birthday (President's Day)          Third Monday in February
'Memorial Day                                      Last Monday in May
'Labor Day                                         First Monday in September
'Columbus Day                                      Second Monday in October
'Thanksgiving                                      Fourth Thursday in November
'
'additionally, Easter Sunday, Ash Wednesday, Holy Thursday and Good Friday are included
'as detected dates.
'

Public Function IsItAHoliday(theDay As Integer, theMonth As Integer, theYear As Integer, Optional ReturnType As Boolean) As Variant
'Inputs:    theDay integer equating to a valid day of associated month
'           theMonth month value 1-12
'           theYear integer of a valid year (years before 1900 will not have Easter related dates calculated)
'           ReturnType:
'               missing or False = name of the holday returnd
'               True = return actual date
'Outputs:   valid date but not a holiday - empty string
'           invalid date combination = "Not a Date"
'           holiday/observance day - depends on ReturnType
'               ReturnType  = False = Name of the holiday (default)
'                           = True  = date the holiday is observed
'
'Coded by:  Jerry Latham
'           with the grateful assistance of routines
'           made available by Chip Pearson as provided from 'http://cpearson.com/excel/holidays.htm
'
    Dim testDate As Variant
    Dim checkDate As Date
   
    Application.Volatile
    IsItAHoliday = "" ' default response
    If IsMissing(ReturnType) Then
        ReturnType = False
    End If
    testDate = theMonth & "/" & theDay & "/" & theYear
    If Not IsDate(testDate) Then
        IsItAHoliday = "Not a Date"
        Exit Function
    End If
    testDate = DateSerial(theYear, theMonth, theDay)
'start by testing against fixed dates and return the date it is observed on
'special case (as in 2005) when Jan 1 falls on a Saturday, then it is observed
'on Dec 31 of the previous year.
    If testDate = FindObservanceDay(DateSerial(theYear, 1, 1)) Or _
        testDate = DateSerial(theYear, 12, 31) And Weekday(DateSerial(theYear, 12, 31)) = 6 Then
        'it is New Year's Day
        'find observed date and exit
        IsItAHoliday = "New Year's Observed"
        If ReturnType = True Then
            IsItAHoliday = testDate
        End If
        Exit Function
    End If
    If testDate = FindObservanceDay(DateSerial(theYear, 7, 4)) Then
        'it is U.S. Independence day
        'find observed date and exit
        IsItAHoliday = "Independence Day Observed"
        If ReturnType = True Then
            IsItAHoliday = testDate
        End If
        Exit Function
    End If
    If testDate = FindObservanceDay(DateSerial(theYear, 11, 11)) Then
        'it is U.S. Veteran's day
        'find observed date and exit
        IsItAHoliday = "Veteran's Day Observed"
        If ReturnType = True Then
            IsItAHoliday = testDate
        End If
        Exit Function
    End If
    If testDate = FindObservanceDay(DateSerial(theYear, 12, 25)) Then
        'it is Christmas Day
        'find observed date and exit
        IsItAHoliday = "Christmas Observed"
        If ReturnType = True Then
            IsItAHoliday = testDate
        End If
        Exit Function
    End If
'
'now into variable date holiday section
'while Easter is not a U.S. Federal Holiday, it
'is a significant Holiday in a large part of the world, find it also
'
'Martin Luther King Day (3rd Monday in January)
    If testDate = NDow(theYear, 1, 3, 2) Then
        'it is MLK Day
        IsItAHoliday = "Martin Luther King, Jr. Day"
        If ReturnType = True Then
            IsItAHoliday = testDate
        End If
        Exit Function
    End If

'Washington 's Birthday (President's Day)          Third Monday in February
    If testDate = NDow(theYear, 2, 3, 2) Then
        'it is President's Day
        IsItAHoliday = "President's Day"
        If ReturnType = True Then
            IsItAHoliday = testDate
        End If
        Exit Function
    End If

'Memorial Day                                      Last Monday in May
'have to find out how many Mondays in May (month)
'
    If testDate = NDow(theYear, 5, DOWsInMonth(theYear, 5, 2), 2) Then
        'it is Memorial Day
        IsItAHoliday = "Memorial Day"
        If ReturnType = True Then
            IsItAHoliday = testDate
        End If
        Exit Function
    End If

'Labor Day                                         First Monday in September
    If testDate = NDow(theYear, 9, 1, 2) Then
        'it is Labor Day
        IsItAHoliday = "Labor Day"
        If ReturnType = True Then
            IsItAHoliday = testDate
        End If
        Exit Function
    End If

'Columbus Day                                      Second Monday in October
    If testDate = NDow(theYear, 10, 2, 2) Then
        'it is Columbus Day
        IsItAHoliday = "Columbus Day"
        If ReturnType = True Then
            IsItAHoliday = testDate
        End If
        Exit Function
    End If

'Thanksgiving                                      Fourth Thursday in November
    If testDate = NDow(theYear, 11, 4, 5) Then
        'it is Thanksgiving Day
        IsItAHoliday = "Thanksgiving Day"
        If ReturnType = True Then
            IsItAHoliday = testDate
        End If
        Exit Function
    End If
   
'Easter and related days
    If theYear > 1899 Then
        'easter calculator does not work prior to 1900
        checkDate = CalculateEaster(theYear)
    Else
        Exit Function ' all done with whatever results we have
    End If
    'is it Ash Wednesday?
    If testDate = (checkDate - 46) Then
        'return Ash Wed. as a holiday
        IsItAHoliday = "Ash Wednesday"
        If ReturnType = True Then
            IsItAHoliday = testDate
        End If
        Exit Function
    End If
    If testDate = (checkDate - 2) Then
        'return Good Friday as holiday
        IsItAHoliday = "Good Friday"
        If ReturnType = True Then
            IsItAHoliday = testDate
        End If
        Exit Function
    End If
    If testDate = (checkDate - 3) Then
        'return Holy Thursday as holiday
        IsItAHoliday = "Holy Thursday"
        If ReturnType = True Then
            IsItAHoliday = testDate
        End If
        Exit Function
    End If
    If testDate = checkDate Then
        'this is Easter, doesn't move
        'Good Friday is 2 days earlier and
        'Lent starts 46 days earlier
        IsItAHoliday = "Easter Sunday"
        If ReturnType = True Then
            IsItAHoliday = testDate
        End If
        Exit Function
    End If
   
End Function

Public Function CalculateEaster(anyYear As Variant) As Variant
'Algorithm Source:
' http://aa.usno.navy.mil/faq/docs/easter.php
'requires integer math (use \ for division, not /)
'
'Valid for dates AFTER 1899
'
'Adapted for VBA use by
'Jerry Latham
'This routine may be reused so long as:
'   acknowledgment of the source is made in the code segment, and
'   no 'value added' charge is applied based on the inclusion of
'   this process.
'
    Dim c, d, N, k, I, j, l, M, Y As Integer
    Y = anyYear
    If Y < 1900 Then
        CalculateEaster = "Invalid prior to 1900"
        Exit Function
    End If
    c = Y \ 100
    N = Y - 19 * (Y \ 19)
    k = (c - 17) \ 25
    I = c - c \ 4 - (c - k) \ 3 + 19 * N + 15
    c = Y \ 100
    N = Y - 19 * (Y \ 19)
    k = (c - 17) \ 25
    I = c - c \ 4 - (c - k) \ 3 + 19 * N + 15
    I = I - 30 * (I \ 30)
    I = I - (I \ 28) * (1 - (I \ 28) * (29 \ (I + 1)) * ((21 - N) \ 11))
    j = Y + Y \ 4 + I + 2 - c + c \ 4
    j = j - 7 * (j \ 7)
    l = I - j
    M = 3 + (l + 40) \ 44
    d = l + 28 - 31 * (M \ 4)
    CalculateEaster = DateSerial(Y, M, d)
End Function

Public Function FindObservanceDay(anyHoliday As Variant) As Date
'if a date is a Saturday, return date of preceding Friday
'if a date is a Sunday, return date of following Monday
    FindObservanceDay = anyHoliday ' default
    If Weekday(anyHoliday, vbSunday) = vbSaturday Then
        FindObservanceDay = anyHoliday - 1 ' subtract 1 day
    ElseIf Weekday(anyHoliday, vbSunday) = vbSunday Then
        FindObservanceDay = anyHoliday + 1 ' move to Monday
    End If
End Function

Public Function ThanksgivingDate(Yr As Integer) As Date
'This function graciously made available by Chip Pearson, MSFT MVP Excel
'http://cpearson.com/excel/holidays.htm
'
   ThanksgivingDate = DateSerial(Yr, 11, 29 - Weekday(DateSerial(Yr, 11, 1), vbFriday))
End Function

Public Function NDow(Y As Integer, M As Integer, N As Integer, DOW As Integer) As Date
'This function graciously made available by Chip Pearson, MSFT MVP Excel
'http://cpearson.com/excel/holidays.htm
'
'this finds date of the Nth Day of Week in a month
'- where DOW = 1 for Sunday, 2 for Monday, etc.
    NDow = DateSerial(Y, M, (8 - Weekday(DateSerial(Y, M, 1), (DOW + 1) Mod 8)) + ((N - 1) * 7))
End Function

Public Function DOWsInMonth(Yr As Integer, M As Integer, DOW As Integer) As Integer
'This function graciously made available by Chip Pearson, MSFT MVP Excel
'http://cpearson.com/excel/holidays.htm
'
'this function finds the count of a particular day of the week in a given month/year
'use for finding things like Memorial Day (last Monday in May)
' - so we need to know # of Mondays in May to find Memorial Day date
'
    On Error GoTo EndFunction
   
    Dim I As Integer
    Dim Lim As Integer
    Lim = Day(DateSerial(Yr, M + 1, 0))
    DOWsInMonth = 0
    For I = 1 To Lim
        If Weekday(DateSerial(Yr, M, I)) = DOW Then
            DOWsInMonth = DOWsInMonth + 1
        End If
    Next I
    Exit Function
EndFunction:
    DOWsInMonth = 0
End Function
