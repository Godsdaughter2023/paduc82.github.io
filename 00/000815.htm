Option Explicit

Sub InsertAssembly()
  Dim S As Shape
  Dim Source As Range, Dest As Range

  If VarType(Application.Caller) <> vbString Then
    MsgBox "Please click the 'Insert Assembly Rows' button", vbInformation, "InsertAssembly"
    Exit Sub
  End If

  'Refer to the rows to copy
  Set Source = Rows("7:16")
 
  'Refer to the calling button
  Set S = ActiveSheet.Shapes(Application.Caller)
  'Get the cell behind
  Set Dest = S.TopLeftCell

  'Copy the source above the button
  Application.ScreenUpdating = False
  With Source
    Dest.Resize(.Rows.Count).EntireRow.Insert
    .Hidden = False
    .Copy
    Dest.Offset(-Source.Rows.Count, 1 - Dest.Column).Select
    ActiveSheet.Paste
    .Hidden = True
    Dest.Offset(-Source.Rows.Count).Select
  End With
  Application.CutCopyMode = False

  'Make unique "Bevel" names
  'Bug seen in XL2010 and XL2016: The copied shapes are not renamed
  RenameBevels S.Parent
End Sub

Private Sub RenameBevels(ByVal Ws As Worksheet)
  Dim S As Shape
  Dim i As Long, j As Long
  For j = 1 To 0 Step -1
    i = 0
    For Each S In Ws.Shapes
      If InStr(1, S.Name, "Bevel", vbTextCompare) = 1 Then
        i = i + 1
        S.Name = "Bevel " & i + j * 100
      End If
    Next
  Next
End Sub

Sub InsertLines()
  Dim S As Shape
  Dim Dest As Range
  Dim NumberOfRows As Long

  If VarType(Application.Caller) <> vbString Then
    MsgBox "Please click an 'Insert ???' button", vbInformation, "InsertLines"
    Exit Sub
  End If

  NumberOfRows = Application.InputBox("How many lines to insert?", "Insert Lines", Type:=1)
  If NumberOfRows <= 0 Then Exit Sub

  'Refer to the calling button
  Set S = ActiveSheet.Shapes(Application.Caller)
  'Get the cell behind
  Set Dest = S.TopLeftCell
  'Insert the lines
  Application.ScreenUpdating = False
  With Dest
    .Resize(NumberOfRows).EntireRow.Insert xlShiftDown, xlFormatFromLeftOrAbove
    .Offset(-NumberOfRows).Select
  End With
End Sub

Sometimes, add-ins can cause problems. To determine, if add-in is a problem start your application in safe mode and see if problem has gone away.

To start in Safe Mode -

Hold CTRL key and click on application icon and don't release the CTRL key till it asks you for Safe mode confirmation.

If yes, then start application normally and disable add-ins one by one and start application again every time you disable an add-in to determine the culprit add-in.
To disable add-in > File > Options > Add-ins > Down below you have Manage: Com Add-in : Press Go > Untick the Add-ins to disable and tick to enable them
