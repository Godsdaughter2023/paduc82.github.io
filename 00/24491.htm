jllatham say:

Here is macro code that will look at all open workbooks and set the PrintArea from cell A1 over and down to the last cell truly in use on the sheet for all sheets in each other open workbook.

To use the code:  Open up Excel and create a new workbook.  Press [Alt]+[F11] to enter the Visual Basic Editor (VBE).  Use Insert -> Module from the VBE menu and copy and paste the code below into the module.  Close the VBE by pressing [Alt]+[Q] (or using it's red-x).

Save the workbook as a macro enabled workbook, type .xlsm or .xlsb.

To use it: open that workbook, then open the other workbooks you want to reset the print areas in.

Go to the [View] tab and click the {Macros} dropdown and choose "View Macros" - find the SetPrintAreas macro in the list, highlight it and click the [Run] button.  Once the "Task Completed" message comes up and you [OK] it, then save all workbooks with changes.

The code:

Sub SetPrintAreas()
  Dim anyWB As Workbook
  Dim anyWS As Worksheet
  Dim lastCol As Long
  Dim lastRow As Long
  Dim paAddress As String
  For Each anyWB In Application.Workbooks
    'don't do this for the workbook this code is in
    If anyWB.Name <> ThisWorkbook.Name Then
      'this will go through every sheet in
      'the other workbook(s) and set up the
      'PrintArea to be A1:--  where -- is the
      'last cell actually used based on last
      'column and last row with data in them.
      For Each anyWS In anyWB.Worksheets
        'find last used row on the sheet
        On Error Resume Next
        lastRow = anyWS.Cells.Find(What:="*", _
         After:=anyWS.Cells(1), _
         Lookat:=xlPart, _
         LookIn:=xlFormulas, _
         SearchOrder:=xlByRows, _
         SearchDirection:=xlPrevious, _
         MatchCase:=False).Row
        On Error GoTo 0
        'find last used column on the row
        On Error Resume Next
        lastCol = anyWS.Cells.Find(What:="*", _
         After:=anyWS.Cells(1), _
         Lookat:=xlPart, _
         LookIn:=xlFormulas, _
         SearchOrder:=xlByColumns, _
         SearchDirection:=xlPrevious, _
         MatchCase:=False).Column
        On Error GoTo 0
        paAddress = "A1:" & Cells(lastRow, lastCol).Address
        anyWS.PageSetup.PrintArea = paAddress
      Next ' next worksheet in a workbook
    End If ' end test for this workbook
  Next ' next workbook currently open
  'release assigned resources back to the system
  Set anyWS = Nothing
  Set anyWB = Nothing
  MsgBox "Print Areas have been set.", vbOKOnly, "Task Completed"
End Sub

Sub SetPrintAreas()
  Const fileTypes = "*.xls*" ' all Excel file types
  Dim strStartFolder As String
  Dim fileToOpen As String
  Dim processedCount As Long
  Dim myWS As Worksheet
  Dim anyWB As Workbook
  Dim anyWS As Worksheet
  Dim lastRow As Long
  Dim lastCol As Long
 
  'Choose the root folder
  strStartFolder = BrowseFolderDialog("Select Root Folder")
  If strStartFolder = "False" Then
      Exit Sub ' user cancelled
  End If
  If Right(strStartFolder, 1) <> Application.PathSeparator Then
    strStartFolder = strStartFolder & Application.PathSeparator
  End If
  Set myWS = ActiveSheet
  With myWS
    .Cells.ClearContents
    .Range("A1") = "Files Processed:"
  End With
  'so we can recover if an error happens
  On Error GoTo errHandler
  'Increase the execution speed of the macro
  Application.ScreenUpdating = False
  Application.Calculation = xlCalculationManual
  'turn off interrupt processing to prevent any
  'auto macros from running in other workbooks
  Application.EnableEvents = False
  fileToOpen = Dir$(strStartFolder & fileTypes)
  Do While fileToOpen <> ""
    If fileToOpen <> ThisWorkbook.Name Then
      'record the name of the workbook we are
      'about to process
      Application.ScreenUpdating = True
      myWS.Range("A" & Rows.Count).End(xlUp).Offset(1, 0) = _
       fileToOpen
      Application.ScreenUpdating = False
      Application.DisplayAlerts = False
      Set anyWB = _
       Workbooks.Open(strStartFolder & fileToOpen, False, False)
      Application.DisplayAlerts = True
      'this will go through every sheet in
      'the other workbook(s) and set up the
      'PrintArea to be A1:--  where -- is the
      'last cell actually used based on last
      'column and last row with data in them.
      For Each anyWS In anyWB.Worksheets
        'find last used row on the sheet
        On Error Resume Next
        lastRow = anyWS.Cells.Find(What:="*", _
         After:=anyWS.Cells(1), _
         Lookat:=xlPart, _
         LookIn:=xlFormulas, _
         SearchOrder:=xlByRows, _
         SearchDirection:=xlPrevious, _
         MatchCase:=False).Row
        'find last used column on the row
        On Error Resume Next
        lastCol = anyWS.Cells.Find(What:="*", _
         After:=anyWS.Cells(1), _
         Lookat:=xlPart, _
         LookIn:=xlFormulas, _
         SearchOrder:=xlByColumns, _
         SearchDirection:=xlPrevious, _
         MatchCase:=False).Column
        On Error GoTo errHandler
        If lastRow > 0 And lastCol > 0 Then
          Application.DisplayAlerts = False
          'can get error if it is a chart sheet
          On Error Resume Next
          anyWS.PageSetup.PrintArea = _
           "A1:" & Cells(lastRow, lastCol).Address
          If Err <> 0 Then
            Err.Clear
          End If
          On Error GoTo errHandler
          Application.DisplayAlerts = True
        Else
          'probably empty sheet
          anyWS.PageSetup.PrintArea = ""
        End If
      Next ' next worksheet in a workbook
      processedCount = processedCount + 1
      'close and save the workbook
      Application.DisplayAlerts = False
      anyWB.Close True
      Application.DisplayAlerts = True
    End If
    fileToOpen = Dir$() ' get next Excel filename in folder
  Loop
errHandler:
  Application.ScreenUpdating = True
  Application.Calculation = xlAutomatic
  Application.EnableEvents = True
  If Err.Number <> 0 Then
    MsgBox Err.Description, vbExclamation, "Something has gone wrong :-("
  End If
  Set myWS = Nothing
  Set anyWB = Nothing
  Set anyWS = Nothing
  On Error GoTo 0
  MsgBox "Reset Print Area for " & processedCount & " files.", _
   vbOKOnly + vbInformation, "Task Completed"
End Sub

Function BrowseFolderDialog(Title As String, _
        Optional InitialFolder As String = vbNullString, _
        Optional InitialView As Office.MsoFileDialogView = _
            msoFileDialogViewList) As String
    Dim V As Variant
    Dim InitFolder As String
    With Application.FileDialog(msoFileDialogFolderPicker)
        .Title = Title
        .InitialView = InitialView
        If Len(InitialFolder) > 0 Then
            If Dir(InitialFolder, vbDirectory) <> vbNullString Then
                InitFolder = InitialFolder
                If Right(InitFolder, 1) <> "\" Then
                    InitFolder = InitFolder & "\"
                End If
                .InitialFileName = InitFolder
            End If
        End If
        .Show
        On Error Resume Next
        Err.Clear
        V = .SelectedItems(1)
        If Err.Number <> 0 Then
            V = vbNullString
        End If
    End With
    BrowseFolderDialog = CStr(V)
    On Error GoTo 0
End Function

You could try this:

Create a simple HTML file containing the embed code:

<html>
