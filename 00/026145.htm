gord dibben say:

To re-insert the blank rows after the sorting is complete run this macro,

Sub InsertBreak_At_Change()
Dim I As Long
    For I = Columns(1).Rows.Count To 1 Step -1
        If Selection(I).row = 1 Then Exit Sub
        Application.ScreenUpdating = False
        If Selection(I) <> Selection(I - 1) And Not IsEmpty _
                (Selection(I - 1)) Then
            With Selection(I)
                .entirerow.Insert
            End With
        End If
    Next
    Application.ScreenUpdating = True
End Sub
