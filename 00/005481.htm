rich007a say:

If it were me, and if I hadn't seen the link that Graham had posted, I'd do all this with VBA, either from within Word, or from within Excel.  For example, in Word you could use:

 

Sub UseExcelFromWord()
' This procedure requires a reference be set to
' Microsoft Excel 12.0 Object Library
' via VBE > Tools > References.
'
    Dim XL As Excel.Application
    Dim bWeStartedExcel As Boolean
    Dim xlWb As Excel.Workbook
    Dim xlWs As Excel.Worksheet
    Dim xlChtObj As Excel.ChartObject
    Dim xlCht As Excel.Chart
    Dim xlPnt As Excel.Point
    
     'Get Excel if it's running:
    On Error Resume Next
    Set XL = GetObject(, "Excel.Application")
    If Err <> 0 Then
        On Error GoTo ExitPoint
         'Excel wasn't running, start it from code:
        Set XL = CreateObject("Excel.Application")
        bWeStartedExcel = True
    End If
    On Error GoTo 0 'ExitPoint
   
     'Open Excel file from Template ( use XL instead of Application):
    Set xlWb = XL.Workbooks.Add()
    Set xlWs = xlWb.Sheets(1)
   
     'Do things with the Excel worksheet like this:
    xlWs.Range("A1").Value = "A Chart from Excel"
    xlWs.Range("A2").Value = 2
    xlWs.Range("A3").Value = 2
    xlWs.Range("A4").Value = 3

     'Add a chart:
    Set xlCht = xlWb.Charts.Add
    xlCht.ChartType = xlPie
    xlCht.SetSourceData Source:=xlWs.Range("A1:A4")
    Set xlPnt = xlCht.SeriesCollection(1).Points(1)
    xlPnt.Explosion = 20
   
    xlCht.Location Where:=xlLocationAsObject, Name:=xlWs.Name
    Set xlChtObj = xlWs.ChartObjects(xlWs.ChartObjects.Count)
    xlChtObj.Width = 200
    xlChtObj.Height = 300
   
     'Copy the chart in Excel
    xlChtObj.Copy
   
     'Paste chart into Word doc
    Selection.Paste
   
ExitPoint:
    On Error Resume Next
    If bWeStartedExcel Then XL.Quit
    Set xlWb = Nothing
    Set XL = Nothing
End Sub
