You could use code like this - if the date cells for the template are based on cell A1

Sub TestMacro()

    Dim i As Integer

    Dim sh As Worksheet

    

    For i = 1 To 12

        Worksheets("Template").Copy After:=Worksheets(Worksheets.Count)

        With Worksheets(Worksheets.Count)

            .Name = Format(DateSerial(Year(Date), i, 1), "MMMM")

            .Range("A1").Value = DateSerial(Year(Date), i, 1)

        End With

    Next i

    

End Sub
