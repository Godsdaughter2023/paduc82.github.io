A funny task for a Sunday morning. :-)

To make it usable for others, I use the current contents of the clipboard.
But to get a string from the clipboard, a reference to "Microsoft Forms Object Library" must be set for my code.
Set it manually or run Sub SetRef once.

I also added a couple of string routines to align the words left-padded or centered in the columns.

Andreas.

Option Explicit

Private Sub SetRef()
  On Error Resume Next
  'Microsoft Forms 2.0 Object Library
  Application.VBE.ActiveVBProject.References.AddFromGuid _
    "{0D452EE1-E08F-101A-852E-02608C4D0BB4}", 2, 0
End Sub

Private Function GetClip() As String
  Dim MyData As Object
  If MyData Is Nothing Then Set MyData = New DataObject
  MyData.GetFromClipboard
  On Error Resume Next
  GetClip = MyData.GetText
End Function

Sub Main()
  Dim S As String
  Dim Lines() As String, This() As String
  Dim All() As Variant
  Dim Lengths() As Long
  Dim i As Integer, j As Integer

  ' Day     Month     Year     Model
  ' 1         6         2013    ABC
  '  2           5         2012    XYZ
  '   3   15        2011      EFG

  'Get a string from the clipboard, assume it looks like above
  'Note:
  '  You can copy the lines above from the VBA editor and run this example
  '  Or copy some code lines and run this sub for a funny output
  S = GetClip
  'Get the lines
  Lines = Split(S, vbCrLf)

  'Remove superfluous blanks and empty lines
  j = -1
  For i = 0 To UBound(Lines)
    Lines(i) = TrimDouble(Trim$(Lines(i)))
    If Len(Lines(i)) > 0 Then
      j = j + 1
      If i > j Then Lines(j) = Lines(i)
    End If
  Next
  'Nothing but empty lines?
  If j < 0 Then
    MsgBox "Copy some text into the clipboard and try again"
    Exit Sub
  End If
  ReDim Preserve Lines(0 To j)
 
  'At this point the data in the array look like this:
  'Day Month Year Model
  '1 6 2013 ABC
  '2 5 2012 XYZ
  '3 15 2011 EFG

  'Now split each line into words, save each in All
  ReDim All(0 To j)
  j = 0
  For i = 0 To UBound(All)
    All(i) = Split(Lines(i))
    'Get the max. number of words in each line
    If UBound(All(i)) > j Then j = UBound(All(i))
  Next

  'Get the max. length in each column
  ReDim Lengths(0 To j)
  For i = 0 To UBound(All)
    For j = 0 To UBound(All(i))
      If Len(All(i)(j)) > Lengths(j) Then Lengths(j) = Len(All(i)(j))
    Next
  Next

  'Format each word
  For i = 0 To UBound(All)
    For j = 0 To UBound(All(i))
      'Right pad
      All(i)(j) = Pad(All(i)(j), Lengths(j))
      'Left pad
      'All(i)(j) = LeftPad(All(i)(j), Lengths(j))
      'Centered
      'All(i)(j) = Center(All(i)(j), Lengths(j))
    Next
    Lines(i) = Join(All(i), " ")
  Next

  'Get the string for the comment
  S = Join(Lines, vbCrLf)

  With ActiveCell
    'Delete an old comment if any
    If Not .Comment Is Nothing Then .Comment.Delete
    'Add the new one
    .AddComment S
    'We need a fixed-width font
    .Comment.Shape.TextFrame.Characters.Font.Name = "Courier"
    'Show all
    .Comment.Shape.TextFrame.AutoSize = True
  End With
End Sub

Function TrimDouble(ByVal S As String, _
    Optional ByVal Delim As String = " ", _
    Optional ByVal Compare As VbCompareMethod = vbTextCompare) As String
  'Liefert einen String in dem keine mehrfachen (Leer-)Zeichen mehr sind
  Dim DoubleDelim As String
  DoubleDelim = Delim & Delim
  Do While InStr(1, S, DoubleDelim, Compare) > 0
    S = Replace$(S, DoubleDelim, Delim, Compare:=Compare)
  Loop
  TrimDouble = S
End Function

Function strRepeat(ByVal Count As Long, ByVal S As String) As String
  'Gibt einen sich Count wiederholenden Text zurück.
  If Count < 1 Or Len(S) < 1 Then Exit Function
  If Len(S) > 1 Then
    'Platz reservieren:
    strRepeat = Space$(Len(S) * Count)
    'Text in erste Position kopieren:
    Mid$(strRepeat, 1) = S
    'Ggf. Rest durch Selbst-Kopie erledigen:
    If Count > 1 Then Mid$(strRepeat, Len(S) + 1) = strRepeat
  Else
    strRepeat = String$(Count, S)
  End If
End Function

Function CharStr(ByVal Character As String, ByVal Length As Long, _
    Optional ByVal Left2Right As Boolean = True) As String
  'Return a string with Character in specified Length
  Dim i As Long
  If Len(Character) > 1 Then
    If Left2Right Then
      CharStr = Mid$(strRepeat(Length \ Len(Character) + 1, Character), 1, Length)
    Else
      CharStr = Mid$(strRepeat(Length \ Len(Character) + 1, Character), _
        Len(Character) - Length Mod Len(Character) + 1, Length)
    End If
  Else
    CharStr = String$(Length, Character)
  End If
End Function

Function Pad(ByVal S As String, ByVal Length As Long, _
    Optional ByVal Character As String = " ") As String
  'Return a String right-padded to Length with Character
  Dim i As Long
  If Len(S) > Length Then
    Pad = S
  Else
    If Len(Character) > 1 Then
      Pad = S & CharStr(Character, Length - Len(S))
    Else
      Pad = S & String$(Length - Len(S), Character)
    End If
  End If
End Function

Function LeftPad(ByVal S As String, ByVal Length As Long, _
    Optional ByVal Character As String = " ") As String
  'Return a String left-padded to Length with Character
  Dim i As Long, Temp As String
  If Len(S) > Length Then
    LeftPad = S
  Else
    If Len(Character) > 1 Then
      LeftPad = CharStr(Character, Length - Len(S), False) & S
    Else
      LeftPad = String$(Length - Len(S), Character) & S
    End If
  End If
End Function

Function Center(ByVal S As String, ByVal Length As Long, _
    Optional ByVal Character As String = " ") As String
  'Return a String centered in a String of Character with specified Length
  Dim i As Long, j As Long, Temp As String
  If Len(S) > Length Then
    Center = S
  Else
    i = (Length - Len(S)) \ 2
    If Len(Character) > 1 Then
      Center = Pad("", Length - i - Len(S), Character) & S & Pad("", i, Character)
    Else
      Center = String$(Length - i - Len(S), Character) & S & String$(i, Character)
    End If
  End If
End Function
