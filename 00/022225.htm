john sr wilson say:

You need to be clearer what you are trying to achieve. Changing the footer on the Master may not change the footers on the actual slides. (Try it manually)

This might do what you need:

Sub footers()
   Dim osld As Slide
   Dim ocust As CustomLayout

'change this to the text needed  

Const strFooter1 As String = "abc"
   Const strFooter2 As String = "xyz"
   'change master
   With ActivePresentation.SlideMaster.HeadersFooters.Footer
      Select Case .Text
      Case strFooter2
         .Text = strFooter1
      Case strFooter1
         .Text = strFooter2
      End Select
   End With

   'change layouts
   For Each ocust In ActivePresentation.SlideMaster.CustomLayouts
      With ocust.HeadersFooters.Footer
         Select Case .Text
         Case strFooter2
            .Text = strFooter1
         Case strFooter1
            .Text = strFooter2
         End Select
      End With
   Next ocust

   'change slides
   For Each osld In ActivePresentation.Slides
      With osld.HeadersFooters.Footer
         .Visible = True
         Select Case .Text
         Case strFooter2
            .Text = strFooter1
         Case strFooter1
            .Text = strFooter2
         End Select
      End With
   Next osld
End Sub

Function SplitName(FullName)
    Dim arr
    Dim i As Long
    Dim n As Long
    Dim ret(1 To 2) As String
    FullName = Application.Trim(Replace(FullName, Chr(160), ""))
    If FullName <> "" Then
        arr = Split(FullName)
        n = UBound(arr)
        Select Case LCase(arr(0))
            Case "mohd", "mohd.", "mohammad", "mohammed"
                Select Case n
                    Case 0
                        ret(1) = arr(0)
                    Case 1
                        ret(1) = arr(0) & " " & arr(1)
                    Case Else
                        ret(1) = arr(0) & " " & arr(1)
                        For i = 2 To n
                            ret(2) = ret(2) & " " & arr(i)
                        Next i
                        ret(2) = Mid(ret(2), 2)
                End Select
            Case Else
                If Len(arr(0)) = 1 Then
                Select Case n
                    Case 0
                        ret(1) = arr(0)
                    Case 1
                        ret(1) = arr(0) & " " & arr(1)
                    Case Else
                        ret(1) = arr(0) & " " & arr(1)
                        For i = 2 To n
                            ret(2) = ret(2) & " " & arr(i)
                        Next i
                        ret(2) = Mid(ret(2), 2)
                End Select
                Else
                    ret(1) = arr(0)
                    If n > 0 Then
                        For i = 1 To n
