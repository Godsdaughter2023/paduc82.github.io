mike h.. say:

Hi,


Try this macro on a copy of your date. It assumes the sheet with data on is the active sheet and that the data has a row header.



Sub somesub()
'A, B, D, E, G
Dim x As Long, LastRow As Long
Dim y As Long
LastRow = Cells(Cells.Rows.Count, "A").End(xlUp).Row
For x = 2 To LastRow
    For y = LastRow To x + 1 Step -1
        If _
        UCase(Cells(y, 1)) = UCase(Cells(x, 1).Value) And _
        UCase(Cells(y, 2)) = UCase(Cells(x, 2).Value) And _
        UCase(Cells(y, 4)) = UCase(Cells(x, 4).Value) And _
        UCase(Cells(y, 5)) = UCase(Cells(x, 5).Value) And _
        UCase(Cells(y, 7)) = UCase(Cells(x, 7).Value) Then
     
        Cells(x, 3) = Cells(x, 3) + Cells(y, 3)
        Cells(x, 6) = Cells(x, 6) + Cells(y, 6)
        Cells(x, 8) = Cells(x, 8) + ", " + Cells(y, 8)
        Rows(y).EntireRow.Delete
        LastRow = LastRow - 1
        End If
   
    Next
Next
End Sub
