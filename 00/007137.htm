Public Sub SplitToFiles()

' MACRO SplitToFiles
' Last update: 2012-03-04
' Author: mtone
' Version 1.1
' Description:
' Loops through a specified column, and split each distinct values into a separate file by making a copy and deleting rows below and above
'
' Note: Values in the column should be unique or sorted.
'
' The following cells are ignored when delimiting sections:
' - blank cells, or containing spaces only
' - same value repeated
' - cells containing "total"
'
' Files are saved in a "Split" subfolder from the location of the source workbook, and named after the section name.

Dim osh As Worksheet ' Original sheet
Dim iRow As Long ' Cursors
Dim iCol As Long
Dim iFirstRow As Long ' Constant
Dim iTotalRows As Long ' Constant
Dim iStartRow As Long ' Section delimiters
Dim iStopRow As Long
Dim sSectionName As String ' Section name (and filename)
Dim rCell As Range ' current cell
Dim owb As Workbook ' Original workbook
Dim sFilePath As String ' Constant
Dim iCount As Integer ' # of documents created

iCol = Application.InputBox("Enter the column number used for splitting", "Select column", 1, , , , , 1)
iRow = Application.InputBox("Enter the starting row number (to skip header)", "Select row", 6, , , , , 1)
iFirstRow = iRow

Set osh = Application.ActiveSheet
Set owb = Application.ActiveWorkbook
iTotalRows = osh.UsedRange.Rows.Count
sFilePath = Application.ActiveWorkbook.Path

If Dir(sFilePath + "\AgRpt Split by Agencies_yymmdd", vbDirectory) = "" Then
    MkDir sFilePath + "\AgRpt Split by Agencies_yymmdd"
End If

'Turn Off Screen Updating  Events
Application.EnableEvents = False
Application.ScreenUpdating = False

Do
    ' Get cell at cursor
    Set rCell = osh.Cells(iRow, iCol)
    sCell = Replace(rCell.Text, " ", "")

    If sCell = "" Or (rCell.Text = sSectionName And iStartRow <> 0) Or InStr(1, rCell.Text, "total", vbTextCompare) <> 0 Then
        ' Skip condition met
    Else
        ' Found new section
        If iStartRow = 0 Then
            ' StartRow delimiter not set, meaning beginning a new section
            sSectionName = rCell.Text
            iStartRow = iRow
        Else
            ' StartRow delimiter set, meaning we reached the end of a section
            iStopRow = iRow - 1

            ' Pass variables to a separate sub to create and save the new worksheet
            CopySheet osh, iFirstRow, iStartRow, iStopRow, iTotalRows, sFilePath, sSectionName, owb.fileFormat
            iCount = iCount + 1

            ' Reset section delimiters
            iStartRow = 0
            iStopRow = 0

            ' Ready to continue loop
            iRow = iRow - 1
        End If

        CopySheet osh, iFirstRow, iStartRow, iStopRow, iTotalRows, sFilePath, sSectionName, owb.fileFormat
        iCount = iCount + 1

        Exit Do

Public Sub CopySheet(osh As Worksheet, iFirstRow As Long, iStartRow As Long, iStopRow As Long, iTotalRows As Long, sFilePath As String, sSectionName As String, fileFormat As XlFileFormat)
     Dim ash As Worksheet ' Copied sheet
     Dim awb As Workbook ' New workbook

     ' Copy book
     osh.Copy
     Set ash = Application.ActiveSheet

     ' Delete Rows after section
     If iTotalRows > iStopRow Then
         DeleteRows ash, iStopRow + 1, iTotalRows
     End If

     ' Delete Rows before section
     If iStartRow > iFirstRow Then
         DeleteRows ash, iFirstRow, iStartRow - 1
     End If

     ' Select left-topmost cell
     ash.Cells(1, 1).Select

     ' Clean up a few characters to prevent invalid filename
     sSectionName = Replace(sSectionName, "/", " ")
     sSectionName = Replace(sSectionName, "\", " ")
     sSectionName = Replace(sSectionName, ":", " ")
     sSectionName = Replace(sSectionName, "=", " ")
     sSectionName = Replace(sSectionName, "*", " ")
     sSectionName = Replace(sSectionName, ".", " ")
     sSectionName = Replace(sSectionName, "?", " ")

     ' Save in same format as original workbook
     ash.SaveAs sFilePath + "\AgRpt Split by Agencies_yymmdd\" + sSectionName, fileFormat

     ' Close
     Set awb = ash.Parent
     awb.Close SaveChanges:=False
End Sub

Option Explicit

Sub ExportDatabaseToSeparateFiles()
    'Export is based on the value in the desired column
    Dim rngC As Range
    Dim shtExtract As Worksheet
    Dim shtData As Worksheet
    Dim strName As String
    Dim rngKeyValues As Range
    Dim strKeyCol As String
    Dim iKField As Integer
    Dim strPath As String
    Dim strFName As String

    strPath = Application.ActiveWorkbook.Path & "\"
    strFName = "AgRpt Split by Agencies_"

    If Dir(strPath & strFName & Format(Date, "yymmdd"), vbDirectory) = "" Then
        MkDir strPath & strFName & Format(Date, "yymmdd")
    End If

    strPath = strPath & strFName & Format(Date, "yymmdd") & "\"

    Set shtData = Application.ActiveSheet

    'strKeyCol = InputBox("What column letter to use as key?") 'Use this if variable
    strKeyCol = "A"  'Change to column with key values

    Set rngKeyValues = Intersect(shtData.UsedRange, shtData.Range(strKeyCol & ":" & strKeyCol))

    Set rngKeyValues = rngKeyValues.SpecialCells(xlCellTypeConstants)    'Use for constants
    'Set rngKeyValues = rngKeyValues.SpecialCells(xlCellTypeFormulas) 'Use if formulas

    Set rngKeyValues = rngKeyValues.Offset(1, 0).Resize(rngKeyValues.Rows.Count - 1, 1)    'Assumed one row of header values
    iKField = rngKeyValues.Column - ActiveCell.CurrentRegion.Cells(1).Column + 1

    For Each rngC In rngKeyValues
        On Error GoTo NoSheet
        strName = Application.ActiveWorkbook.Worksheets(rngC.Value).Name
        GoTo SheetExists:
NoSheet:
        Set shtExtract = Application.ActiveWorkbook.Worksheets.Add(before:=shtData)
        shtExtract.Name = rngC.Value
        With rngC.CurrentRegion
            .AutoFilter Field:=iKField, Criteria1:=rngC.Value
            .SpecialCells(xlCellTypeVisible).Copy _
                    shtExtract.Range("A1")
            shtExtract.Cells.EntireColumn.AutoFit
            .AutoFilter
        End With
        Resume
SheetExists:
    Next rngC

    For Each shtExtract In Application.ActiveWorkbook.Worksheets
        If shtExtract.Name <> shtData.Name Then
            If Not IsError(Application.Match(shtExtract.Name, rngKeyValues, False)) Then
                shtExtract.Move
                Application.ActiveWorkbook.SaveAs strPath & Application.ActiveWorkbook.ActiveSheet.Name & Format(Date, "_yymmdd") & ".xlsx"
                Application.ActiveWorkbook.Close False
            End If
        End If
    Next shtExtract
End Sub

Function GFileName(strFilename As String) As String
    Dim RemChar As String
    Dim i As Integer
    
    RemChar = "#<$+%!`&*‘|{?“=}/:\@{}"
   
    GFileName = strFilename
    
    For i = 1 To Len(RemChar)
        GFileName = Replace(GFileName, Mid(RemChar, i, 1), "")
    Next i

    GFileName  = Trim(GFileName)
End Function

Option Explicit

Sub ExportDatabaseToSeparateFiles3()
    'Export is based on the value in the desired column
    Dim rngC As Range
    Dim shtExtract As Worksheet
    Dim shtData As Worksheet
    Dim strName As String
    Dim rngKeyValues As Range
    Dim strKeyCol As String
    Dim iKField As Integer
    Dim strPath As String
    Dim strFName As String
    Dim lngRow As Long

    strPath = Application.ActiveWorkbook.Path & "\"
    strFName = "AgRpt Split by Agencies_"

    If Dir(strPath & strFName & Format(Date, "yymmdd"), vbDirectory) = "" Then
        MkDir strPath & strFName & Format(Date, "yymmdd")
    End If

    strPath = strPath & strFName & Format(Date, "yymmdd") & "\"

    Set shtData = Application.ActiveSheet

    strKeyCol = InputBox("What column letter to use as key?")    'Use this if variable
    'strKeyCol = "A"  'Change to column with key values
    lngRow = CLng(InputBox("What row for the first key value?"))

    Set rngKeyValues = shtData.Range(shtData.Range(strKeyCol & lngRow), shtData.Cells(shtData.Rows.Count, strKeyCol).End(xlUp))
    
    iKField = rngKeyValues.Column - rngKeyValues.CurrentRegion.Cells(1).Column + 1

    For Each rngC In rngKeyValues
        If rngC.Value <> "Total" And Application.WorksheetFunction.CountIf(Range(rngKeyValues.Cells(1, 1), rngC), rngC) = 1 Then
            On Error GoTo NoSheet
            strName = Application.ActiveWorkbook.Worksheets(GFileName(rngC.Value)).Name
            GoTo SheetExists:
NoSheet:
            Set shtExtract = Application.ActiveWorkbook.Worksheets.Add(before:=shtData)
            shtExtract.Name = GFileName(rngC.Value)
            With rngKeyValues.Offset(-1).EntireRow
                .AutoFilter Field:=iKField, Criteria1:=rngC.Text
                .SpecialCells(xlCellTypeVisible).Copy _
                        shtExtract.Range("A1")
                shtExtract.Cells.EntireColumn.AutoFit
                .AutoFilter
            End With
            Resume
        End If
SheetExists:
    Next rngC

    For Each shtExtract In Application.ActiveWorkbook.Worksheets
        If shtExtract.Name = shtData.Name Then Exit Sub
        shtExtract.Move
        Application.ActiveWorkbook.SaveAs strPath & Application.ActiveWorkbook.ActiveSheet.Name & Format(Date, "_yymmdd") & ".xlsx"
        Application.ActiveWorkbook.Close False
    Next shtExtract
End Sub

Function GFileName(strFilename As String) As String
    Dim RemChar As String
    Dim i As Integer
    
    RemChar = "#<$+%!`&*‘|{?“=}/:\@{}"
   
    GFileName = strFilename
    
    For i = 1 To Len(RemChar)
        GFileName = Replace(GFileName, Mid(RemChar, i, 1), "")
    Next i

    GFileName = Trim(GFileName)
End Function
