We can add an indice to the file automatically if it exists, no problem. And I would do the whole process like this way. (Place the code in a new normal module)

Andreas.

 

EDIT: Bug in function NewFileName removed.


Option Explicit
Option Compare Text

#Const Develop = False

Sub Save_Emails_As_MSG_Files()
  Dim Folder As MAPIFolder
  Dim Item As Object
  Dim mAttachment As Attachment
  Dim Path As String, Subject As String, Value As String
  Dim Success As Boolean

  'From Microsoft Scripting Runtime
  Const TemporaryFolder = 2
  Dim fso As Object 'Scripting.FileSystemObject
  'From Microsoft Excel
  Dim xl As Object 'Excel.Application

  On Error GoTo ErrorHandler

#If Develop Then
  'Get the currently selected folder
  Set Folder = ActiveExplorer.CurrentFolder
#Else
  'choose folder called "4)  ADIs Completed"
  'NOTE:  You must create this folder within your Inbox
  Set Folder = Outlook.Session.GetDefaultFolder(olFolderInbox).Folders("4)  ADIs Completed")
#End If

  'Initialize our helper objects
  Set fso = CreateObject("Scripting.FileSystemObject")
  Set xl = CreateObject("Excel.Application")

  On Error GoTo 0

  'Visit all items
  For Each Item In Folder.Items
    'Is it a mailitem?
    If Item.Class = olMail Then
      'Visit all attachments
      For Each mAttachment In Item.Attachments
        Select Case mAttachment.Type
          Case olByValue
            'This is an attached file
            Select Case fso.GetExtensionName(mAttachment.FileName)
              Case "xls", "xlsx", "xlsm"
                'Save a copy of the Excel file into the temporary folder with a unique name
                Path = fso.BuildPath( _
                  fso.GetSpecialFolder(TemporaryFolder), _
                  fso.GetTempName & "." & fso.GetExtensionName(mAttachment.FileName))
                mAttachment.SaveAsFile Path
                'Try open it and get the value
                Success = GetValue(xl, Path, Value)
                'Delete the temporary file
                Kill Path
                'Got it?
                If Success Then
                  'Make a valid file name
                  Subject = Trim$(ValidFileName(Item.Subject))
                  Value = Trim$(ValidFileName(Value))
                  'Set a default if failed
                  If Subject = "" Then Subject = "(No subject)"
                  'Make an unique file name
                  Path = NewFileName("C:\COMPLETED\" & Subject & Value & ".msg")
                  'Save it
                  Item.SaveAs Path
                  Item.Move Outlook.Session.GetDefaultFolder(olFolderDeletedItems)
                  'Done
                  Exit For
                End If
            End Select
        End Select
      Next
    End If
  Next
  'Close Excel
  xl.Quit
  Exit Sub

ErrorHandler:
  MsgBox Err.Number & " - " & Err.Description
End Sub

Private Function GetValue(ByVal Excel As Object, ByVal Path As String, ByRef Value As String) As Boolean
  'Get the value from the workbook
  Dim xlBook As Object 'Excel.Workbook
  On Error GoTo ExitPoint
  'Open the workbook
  Set xlBook = Excel.Workbooks.Open(Path)
  'Get the value
  Value = CStr(xlBook.Sheets("Journal").Range("G3"))
  'Success
  GetValue = True
ExitPoint:
  If Not xlBook Is Nothing Then xlBook.Close False
End Function

Private Function ValidFileName(ByVal FName As String) As String
  'Return a filename without invalid chars
  Const InvalidChars = "\/:*?""<>|"
  Dim i As Integer, p As Long
  Dim Digit As String
  For i = 1 To Len(InvalidChars)
    Digit = Mid$(InvalidChars, i, 1)
    p = InStr(FName, Digit)
    Do While p > 0
      Mid$(FName, p, 1) = vbNullChar
      p = InStr(FName, Digit)
    Loop
  Next
  ValidFileName = Replace(FName, vbNullChar, "")
End Function

Private Function NewFileName(ByVal FullName As String) As String
  'Return a indicated filename that did not exists but begins with fullname
  '  i.E.: c:\temp\test.xla => c:\temp\test(1).xla
  Dim i As Long, LeftParen As Long, RightParen As Long, DotPos As Long
  Dim LeftPart As String, RightPart As String
  If Dir$(FullName) <> "" Then
    LeftParen = InStrRev(FullName, "(")
    RightParen = InStrRev(FullName, ")")
    On Error GoTo AddParen
    i = Mid$(FullName, LeftParen + 1, RightParen - LeftParen - 1)
    LeftPart = Left$(FullName, LeftParen)
    RightPart = Mid$(FullName, RightParen)
    GoTo DoIt
AddParen:
    DotPos = InStrRev(FullName, ".")
    If DotPos = 0 Then
      LeftPart = FullName & "("
      RightPart = ")"
    Else
      LeftPart = Left$(FullName, DotPos - 1) & "("
      RightPart = ")" & Mid$(FullName, DotPos)
    End If
DoIt:
    Do
      i = i + 1
      NewFileName = LeftPart & i & RightPart
    Loop Until Dir$(NewFileName) = ""

  Else

    NewFileName = FullName
  End If
End Function
