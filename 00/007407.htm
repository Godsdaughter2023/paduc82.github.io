hansv mvp say:

Here is such a macro. It will prompt you to select the document. The code uses late binding to automate Word, so you don't need to set a reference to the Word object library.

Sub CheckDoc()
    Dim strFile As String
    Dim blnStart As Boolean
    Dim objWord As Object
    Dim objDoc As Object
    Dim rngCell As Range
    strFile = Application.GetOpenFilename("Word documents (*.doc*),*.doc*")
    If strFile = "False" Then
        MsgBox "No document selected...", vbExclamation
        Exit Sub
    End If
    On Error Resume Next
    Set objWord = GetObject(Class:="Word.Application")
    If objWord Is Nothing Then
        Set objWord = CreateObject(Class:="Word.Application")
        If objWord Is Nothing Then
            MsgBox "Can't start Word...", vbExclamation
            Exit Sub
        End If
        blnStart = True
    End If
    On Error GoTo ErrHandler
    Set objDoc = objWord.Documents.Open(Filename:=strFile, AddToRecentFiles:=False)
    Application.ScreenUpdating = False
    For Each rngCell In Range(Range("A1"), Range("A1").End(xlDown))
        With objDoc.Content.Find
            .ClearFormatting
            .Text = rngCell.Value
            .MatchCase = False
            .MatchWholeWord = False
            .MatchWildcards = False
            .Format = False
            If .Execute Then
                rngCell.Offset(0, 1).Value = "Matched"
            Else
                rngCell.Offset(0, 1).Value = "Not found"
            End If
        End With
    Next rngCell
ExitHandler:
    On Error Resume Next
    Application.ScreenUpdating = True
    objDoc.Close SaveChanges:=False
    If blnStart Then
        objWord.Quit SaveChanges:=False
    End If
    Exit Sub
ErrHandler:
    MsgBox Err.Description, vbExclamation
    Resume ExitHandler
End Sub

amit tandon say:

Try below:

=IF(MID(A2,3,1)="a",REPLACE(A2,3,1,"z"),A2)

ONLY if the 3rd character is "a" will it be replaced by "z". This formula is case-insensitive so that means that even if the 3rd character is "A" (caps) it will be still replaced with "z".

To make the formula case-sensitive, use below:

=IF(EXACT(MID(A2,3,1),"a"),REPLACE(A2,3,1,"z"),A2)

This will replace only "a" and NOT "A".
