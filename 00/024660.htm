Sub Unique_Rands()
Dim FillRange As Range
Dim UpperNum As Long
Dim LowerNum As Long, Nums As Long
Set FillRange = Selection
LowerNum = Application.InputBox("Please enter Lower Number.", Title:="Randoms", Type:=1)
UpperNum = Application.InputBox("Please enter Upper Number.", Title:="Randoms", Type:=1)


If UpperNum > 0 And LowerNum < 0 Then
    Nums = UpperNum - LowerNum
Else
    Nums = UpperNum - LowerNum + 1
End If

If Nums < Selection.Cells.Count Or LowerNum = 0 Or UpperNum = 0 Then
    oops = MsgBox("Not enough numbers to fill range or zero number", vbCritical, "Randoms")
    Exit Sub
End If

For Each c In FillRange
    Do
        c.Value = WorksheetFunction.RandBetween(LowerNum, UpperNum)
    Loop Until WorksheetFunction.CountIf(FillRange, c.Value) < 2 And _
        c.Value <> 0
Next
End Sub
