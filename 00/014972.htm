per jessen say:

 

I have set up a (workbook and) sheet reference to the B3 statement, and removed the select statements. Note that I only open QuoteRecords once, to speed the code up a bit.

Remeber that events will stay disabled if your macro does not enable events, unlike .ScreenUpdating, which automatically is enabled when the macro end.

Sub ProcessFolder()
Dim strPath As String
Dim strFile As String
Dim wbk As Workbook
Dim dbWbk As Workbook

Application.ScreenUpdating = False
Application.DisplayAlerts = False

QuoteRecords = "D:\QUOTES DATABASE.XLS"
Set dbWbk = Workbooks.Open(Filename:=QuoteRecords)

strPath = "D:\Old Quotes\"
strFile = Dir(strPath & "*.xls*")
Do While strFile <> ""
    Set wbk = Workbooks.Open(strPath & strFile)
    'Workbooks.Open(strPath & strFile).Activate
   
    'If wbk has more than one worksheet, you should add a sheet reference
    'to the copy statements below
    Range("C18").Copy Destination:=Range("K1") 'Estimator
    Range("C19").Copy Destination:=Range("L1") 'Sales Person
    Range("D5").Copy Destination:=Range("M1") 'Original Estimate
    Range("C4").Copy Destination:=Range("N1") 'Quote Number
    Range("C8").Copy Destination:=Range("O1") 'Customer
    Range("C16").Copy Destination:=Range("P1") 'Description
    Range("K1:P1").Copy
   
    'Set selected cell to bottom of list
    '**** Change sheet name below to suit *****
    dbWbk.Worksheets("Desitnation Sheet").Range("B3").End(xlDown).Offset(1, 0).PasteSpecial Paste:=xlPasteValues
       
    wbk.Close savechanges:=False
    Name strPath & strFile As strPath & strFile & ".old"
    strFile = Dir
Loop
dbWbk.Close savechanges:=True
With Application
    .ScreenUpdating = True
    .EnableEvents = True
End With
End Sub
