Put this into a standard codemodule, and run ProtectSheet - change Sheet1 to the sheet name of interest - to allow the sheet to be protected but be changed through code:

Sub ProtectSheet()
    Sheets("Sheet1").Protect "Password", UserInterfaceOnly:=True
End Sub

'This is the code that locks the cell

Sub LockCell(strAdd As String)
     Sheets("Sheet1").Range(strAdd).Locked = True
End Sub

Then add this to the worksheet's codemodule:

Private Sub Worksheet_Change(ByVal Target As Range)
    Application.OnTime Now() + TimeValue("12:00:00"), "'LockCell """ & Target.Address & """'"
End Sub

sFile = Dir(SourcePath & "*.csv")
wb.Sheets.Add before:=wb.Sheets(1)
c = 1
Do While sFile <> ""
Set wb1 = Workbooks.Open(SourcePath & sFile, xlDelimited, local:=True)
L = wb1.Sheets(1).UsedRange.Rows.Count
wb1.Sheets(1).Range(Cells(N, sC), Cells(L, sC)).Copy wb.Sheets(1).Cells(1, c)
c = c + 1
wb1.Close False
sFile = Dir()
Loop
Application.DisplayAlerts = False
wb.Sheets(1).Copy
ActiveWorkbook.SaveAs Filename:=TargetPath, FileFormat:=xlCSV, local:=True
ActiveWorkbook.Close False
wb.Sheets(1).Delete
Application.DisplayAlerts = True
Application.ScreenUpdating = True

'open csv file via notepad /if you want ignore the below 2 lines

Dim npad As Variant
npad = Shell("C:\WINDOWS\notepad.exe " & TargetPath, 1)
End Sub
