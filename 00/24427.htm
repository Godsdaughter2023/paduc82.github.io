Option Explicit

Sub Test()
  Dim ff As Integer
  Dim FName As String, Contents As String
  Dim Data, Words, Temp, Item
  Dim C As New Collection
  Dim i As Long, j As Long, m As Long

  'Ask for the file
  With Application.FileDialog(msoFileDialogOpen)
    .Filters.Clear
    .Filters.Add "text files", "*.txt"
    If Not .Show Then Exit Sub
    FName = .SelectedItems(1)
  End With

  'Read in all data
  ff = FreeFile
  Open FName For Binary Access Read As #ff
  Contents = Space(LOF(ff))
  Get #ff, , Contents
  Close #ff

  'Split into lines
  Data = Split(Contents, vbCrLf)
  For i = 0 To UBound(Data)
    'Split each line into words
    Words = Split(Data(i), vbTab)
    ReDim Temp(0 To UBound(Words))
    'Parse the contents
    For j = 0 To UBound(Words)
      If IsDate(Words(j)) Then
        'Convert dates
        Temp(j) = CDate(Words(j))
      Else
        Item = TrimAllWhite(Words(j))
        If IsNumeric(Item) Then
          'Convert numbers
          Temp(j) = CDbl(Item)
        Else
          'This is a string
          Temp(j) = Words(j)
        End If
      End If
    Next
    'Remember max. column
    If m < UBound(Temp) Then m = UBound(Temp)
    'Store
    C.Add Temp
  Next
 
  'Compile all data into a 2D array
  ReDim Data(1 To C.Count, 1 To m + 1) As Variant
  i = 0
  For Each Item In C
    i = i + 1
    For j = 0 To UBound(Item)
      Data(i, j + 1) = Item(j)
    Next
  Next
  'Flush into the sheet
  Range("A1").Resize(UBound(Data), UBound(Data, 2)).Value = Data
End Sub

Private Function TrimAllWhite(ByVal s As String, Optional ByVal IncludeBlanks As Boolean) As String
  'Return a string with white space removed
  Dim i As Long, A As Long
  A = 31 + Abs(IncludeBlanks)
  For i = 1 To Len(s)
    Select Case AscW(Mid$(s, i, 1))
      Case Is <= A, 128 To 160
        Mid$(s, i, 1) = vbNullChar
    End Select
  Next
  TrimAllWhite = Replace$(s, vbNullChar, "")
End Function
