Dave

 

Sub AddDateToNames()
  Dim basicPath As String
  Dim startingFileName As String
  Dim newFileName As String
  Dim myFSO As Object
  Dim dateText As String
 
  On Error Resume Next
  With Application.FileDialog(msoFileDialogFolderPicker)
      .Show
      basicPath = .SelectedItems(1)
      If Err <> 0 Then
        'user cancelled
        Err.Clear
        On Error GoTo 0
        Exit Sub
      End If
  End With

  On Error GoTo 0
  basicPath = basicPath & Application.PathSeparator
  Set myFSO = CreateObject("Scripting.FileSystemObject")
  startingFileName = Dir$(basicPath)
  filecount = 0

  Do While startingFileName <> ""
    dateText = myFSO.getfile(startingFileName).datecreated - 9
    'filenames may not contain / or : so...
    'change / and/or : to underscore characters
    dateText = Format(dateText, "yyyy-mm-dd;@")
    checkextension = Mid(startingFileName, InStrRev(startingFileName, "."))
    checkfile = Replace(startingFileName, checkextension, "")
   
    If dateText <> Right(checkfile, 10) Then
   
    newFileName = Left(startingFileName, Len(startingFileName) - 4) & _
     " " & dateText & Right(startingFileName, 4)
     
    On Error Resume Next
   
    Name basicPath & startingFileName As basicPath & newFileName
    Worksheets("Sheet1").Range("A" & Rows.Count).End(xlUp).Offset(1, 0) = basicPath & startingFileName
    Worksheets("Sheet1").Range("B" & Rows.Count).End(xlUp).Offset(1, 0) = basicPath & newFileName


    filecount = filecount + 1
    Else: End If
    startingFileName = Dir$() ' get next Excel filename
  Loop

  Set myFSO = Nothing
  MsgBox "Total Files Processed: " & filecount, vbOKOnly, "Job Done"
End Sub

What you're after can't be done without macros, so you'll need to consider whether whatever app the documents are viewed in will support those. For your 'No' formfields, you could attach a macro such as the following to the 'on exit' property:

Sub Hilite()
Const Pwd As String = ""
With ActiveDocument
  If .ProtectionType <> wdNoProtection Then .Unprotect Pwd
  With Selection.FormFields(1)
    If .CheckBox.Value = True Then
      .Range.HighlightColorIndex = wdYellow
    Else
      .Range.HighlightColorIndex = wdNoHighlight
    End If
  End With
  .Protect Type:=wdAllowOnlyFormFields, NoReset:=True, Password:=Pwd
End With
End Sub
