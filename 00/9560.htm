For that we must compare the column headings with the files names, exclude files that are not listed,... many possibilities for errors. It is much easier if we store the filename as header.

Andreas.

Option Explicit

Sub Test()
  Dim FNames As New Collection, FName As Variant
  Dim WB As Workbook
  Dim Dest As Range

  'Install an error handler
  On Error GoTo ErrorHandler

  'Note: We have to search all files first!
  'Otherwise a save of a file causes an endless loop

  'Search for Excel files in our directory
  FName = Dir(ThisWorkbook.Path & "\*.xls")
  Do While FName <> ""
    'Skip our file
    If FName <> ThisWorkbook.Name Then
      'Store the name in our collection
      FNames.Add FName
    End If
    'Search the next file
    FName = Dir
  Loop
  If FNames.Count = 0 Then
    MsgBox "No files found"
    Exit Sub
  End If

  'Setup the first destination for the data
  Set Dest = Range("B1")
  'Screen off, runs faster
  Application.ScreenUpdating = False
 
  'Perform action on each file
  For Each FName In FNames
    'Open the file
    Set WB = Workbooks.Open(ThisWorkbook.Path & "\" & FName)
    'Access the sheet
    With WB.ActiveSheet
      'Convert column D and R to values
      .Columns("D").NumberFormat = "General"
      Intersect(.UsedRange, .Columns("D")).Value = _
        Intersect(.UsedRange, .Columns("D")).Value
      .Columns("R").NumberFormat = "General"
      Intersect(.UsedRange, .Columns("R")).Value = _
        Intersect(.UsedRange, .Columns("R")).Value
      'Change some text to values
      .Columns("R").Replace "A", 1, LookAt:=xlWhole
      .Columns("R").Replace "B", 0.5, LookAt:=xlWhole
      'Store the filename
      Dest = Left$(WB.Name, InStrRev(WB.Name, ".") - 1)
      'Calculate and store the results
      Dest.Offset(1) = WorksheetFunction.Sum(.Columns("D"))
      Dest.Offset(2) = WorksheetFunction.Sum(.Columns("R"))
      'Next column
      Set Dest = Dest.Offset(, 1)
    End With
    'Save and close the file
    WB.Close True
  Next
  Application.ScreenUpdating = True
  Exit Sub

ErrorHandler:
  Application.ScreenUpdating = True
  If Err.Source = "" Then Err.Source = Application.Name
  If MsgBox(Err.Source & " Error " & Err.Number & ": " & Err.Description & vbNewLine & _
      "Enter debug mode?", vbOKCancel + vbDefaultButton2) = vbOK Then
    Stop 'Press F8 twice
    Resume
  End If
End Sub

If Me.Dirty Then Me.Dirty = False
