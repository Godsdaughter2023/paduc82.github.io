Option Explicit

Sub Suppress_CopyPaste()

    Next oCtrl

    For Each oCtrl In Application.CommandBars.FindControls(ID:=19)

    Next oCtrl

    For Each oCtrl In Application.CommandBars.FindControls(ID:=19)
        oCtrl.Enabled = True
    Next oCtrl

Private Sub Worksheet_SelectionChange(ByVal Target As Range)
