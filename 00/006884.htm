hansv mvp say:

Try this version:

 

Function STDEVIF(SearchRange As Range, Criteria As Range, STDEVRange As Range)
    Dim Cell As Range
    Dim MyOffset As Long
    Dim MyString As String
    Dim MyArray() As Single
    Dim n As Long
    ReDim MyArray(0)
    n = -1
    If SearchRange.Row = STDEVRange.Row And _
            SearchRange.Rows.Count = STDEVRange.Rows.Count And _
            SearchRange.Columns.Count = 1 And _
            STDEVRange.Columns.Count = 1 Then
        MyOffset = STDEVRange.Column - SearchRange.Column
        For Each Cell In SearchRange
            If Criteria = Cell Then
                If Cell.Offset(0, MyOffset) = "ND" Then
                    n = n + 1
                    ReDim Preserve MyArray(UBound(MyArray) + 1)
                    MyArray(n) = 0
                ElseIf Cell.Offset(0, MyOffset) <> "" And IsNumeric(Cell.Offset(0, MyOffset)) Then
                    n = n + 1
                    ReDim Preserve MyArray(n)
                    MyArray(n) = Cell.Offset(0, MyOffset)
                End If
            End If
        Next Cell
        STDEVIF = Application.StDev(MyArray)
    End If
End Function
