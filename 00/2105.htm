Based on Hans' UDF:

Function StockReserve(Stock As Double, StockDate As Long, DemandDate As Range, Demand As Range)
   Dim i As Long, s As Double, r As Long
   
   For i = 1 To DemandDate.Count
      If StockDate <= DemandDate(i) Then
         s = s + Demand(i)
         r = r + 1
         If s > Stock Then
            StockReserve = r - 1
            Exit Function
         End If
      End If
   Next i
   StockReserve = r
End Function
