Option Explicit

Dim ZoomStates As New Collection
Dim ZoomState As Double, ZoomTarget As Range

Private Sub Workbook_Open()
  Workbook_SheetActivate ActiveSheet
End Sub

Private Sub Workbook_SheetActivate(ByVal Sh As Object)
  'Gets the current zoom state for this sheet
  Dim Item
  On Error GoTo AddState
  Item = ZoomStates(Sh.Name)
  ZoomState = Item(0)
  Set ZoomTarget = Item(1)
  Exit Sub
AddState:
  ZoomState = ActiveWindow.Zoom
  Set ZoomTarget = ActiveCell
  ZoomStates.Add Array(ZoomState, ZoomTarget), Sh.Name
End Sub

Private Sub Workbook_SheetChange(ByVal Sh As Object, ByVal Target As Range)
  'Zoom out if necessary
  If ActiveWindow.Zoom <> ZoomState Then
    ActiveWindow.Zoom = ZoomState
    CenterTarget
  End If
End Sub

Private Sub Workbook_SheetSelectionChange(ByVal Sh As Object, ByVal Target As Range)
  Dim Item
  On Error Resume Next
  If Target.Validation.Type <> xlValidateList Then
    'Zoom out if necessary
    If ActiveWindow.Zoom <> ZoomState Then
      ActiveWindow.Zoom = ZoomState
      CenterTarget
    End If
  Else

    Set ZoomTarget = Target
    'Update the target
    Item = ZoomStates(Sh.Name)
    Set Item(1) = Target
    ZoomStates(Sh.Name) = Item
    CenterTarget
  End If
End Sub

Private Sub CenterTarget()
  'Move the target into the users view
  Dim r As Long, c As Long
  With ActiveWindow
    '25% below the top to ensure that the dropdown is visible
    r = ZoomTarget.Row - .VisibleRange.Rows.Count * 0.25
    '40% from the left to ensure that the number of columns on the right is greater as on the left

    If r < 1 Then r = 1
    If c < 1 Then c = 1
    .ScrollRow = r
    .ScrollColumn = c
  End With
End Sub

ken sheridan say:

For most business purposes the Format property of the controls showing currency values in forms and reports would be set to something like this:

 

£#,##0.00;(£#,##0.00)[Red]

 

This will show the values with the GBP symbol, with positive amounts or zero in the ForeColor colour set for the control (usually black), and negative values in the conventional format of being enclosed within parentheses and in red.

 

You can also set the Format property of the relevant columns (fields) in the tables in the same way, though this is less important as the tables should never be exposed to the user in raw datasheet view.


Do not be tempted to use the generic format of 'Currency' (not to be confused with the Currency data type, which you should use).  If you do this and send your file to someone whose system is set to use a different currency, the values will show as that currency, not as GBP, so you might find that you are quoting something at 1,000 GBP and it will appear to someone in the Euro Zone as 1,000 EUR, which you would not want to sell the product at!
