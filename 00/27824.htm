Dim bolDisable As Boolean   'Must be at top of module before any subs.

Private Sub CheckBox1_Click()
    If bolDisable Then Exit Sub
    If CheckBox1 = True Then
        Call SetCheck(CheckBox1.Name, CheckBox1.GroupName)
    End If
End Sub

'The following sub is called from the individual CheckBox Click events
Sub SetCheck(strChkBox As String, strGrp As String)
    Dim ctrl As OLEObject
   
    bolDisable = True
   
    For Each ctrl In ActiveSheet.OLEObjects
        If ctrl.Object.GroupName = strGrp Then
            ctrl.Object = False
        End If
    Next ctrl
    ActiveSheet.OLEObjects(strChkBox).Object = True
   
    bolDisable = False
   
End Sub

You can e.g. combine the OLEobjects of each group in a collection and store that into a dictionary with the GroupName as key.

Andreas.

Private Sub Example_OLEGroupItems()
  Dim Container As Collection
  Dim OO As OLEObject

  Set Container = OLEGroupItems("MyOLEGroup")
  If Container Is Nothing Then Exit Sub
  For Each OO In Container
    Debug.Print OO.Caption
  Next
End Sub

Static Function OLEGroupItems(ByVal GroupName As String, _
    Optional ByVal Ws As Worksheet) As Collection
  'Returns a collection with all ActiveX controls matching GroupName, nothing if not found
  Dim Dict As Object 'Scripting.Dictionary
  Dim OO As OLEObject, Item As Variant, Key As String

  If Ws Is Nothing Then Set Ws = ActiveSheet
  If Ws Is Nothing Then Exit Function

  If Dict Is Nothing Then
    Set Dict = CreateObject("Scripting.Dictionary")
    Dict.CompareMode = vbTextCompare
AddControls:
    With Dict
      .Add Ws.Name, Ws.OLEObjects.Count
      For Each OO In Ws.OLEObjects
        For Each Item In Array("Forms.CheckBox.1", "Forms.OpionButton.1")
          If StrComp(Item, OO.progID, vbTextCompare) = 0 Then
            Key = Ws.Name & "!" & OO.Object.GroupName
            If Not Dict.Exists(Key) Then Dict.Add Key, New Collection
            Dict(Key).Add OO
            Exit For
          End If
        Next
      Next
    End With
  End If

  With Dict
    If Not .Exists(Ws.Name) Then GoTo AddControls
    Key = Ws.Name & "!" & GroupName
    If .Exists(Key) Then Set OLEGroupItems = .Item(Key)
  End With
End Function

I have actually come up with a simple viable alternative using Forms Control CheckBoxes. (Code below)

Create the Forms Control CheckBoxes.

Right click each checkbox in turn and assign the common macro name to the control.

Right click the first checkbox of a group.

Hold the Ctrl key and right click the remaining checkboxes in the required group so all checkboxes for the required group are selected.

Release the Ctrl key and Right click in one of the selected checkboxes.

Select Group -> Group and the Group is established.

The same code sub is run irrespective of which checkbox is clicked and if the checkbox is checked (xlOn) then all of the checkboxes in the group are turned off and then the clicked checkbox is turned back on. (Saves using an If statement to test for the checked box.)

The Forms controls have the advantage in that the names of the checkboxes and groups are irrelevant and does not need a separate event sub for each checkbox.

Sub ProcessChkBoxes()
    'Processes Forms Control CheckBoxes based on Groups
    Dim shpChk As Shape
    Dim shpGrp As Shape
    Dim shp As Shape
    Dim i As Long
    Set shpChk = ActiveSheet.Shapes(Application.Caller)
    If shpChk.ControlFormat.Value = xlOn Then   'xlOn is Checked. (Calling CheckBox)
        Set shpGrp = shpChk.ParentGroup
        For i = 1 To shpGrp.GroupItems.Count
            Set shp = shpGrp.GroupItems(i)
            shp.ControlFormat.Value = xlOff     'xlOff is Checked. turns off all in the Group
        Next i
        shpChk.ControlFormat.Value = xlOn       'This is calling CheckBox. Gets turned back on
    End If
End Sub

Similar method for ActiveX CheckBoxes

I tested a similar method using ActiveX controls and it also works but requires a separate click event for each checkbox.

Dim bolDisable As Boolean   'Must be at top of module before any subs.
Private Sub CheckBox1_Click()
    If bolDisable Then Exit Sub
    If CheckBox1 = True Then
        Call ProcessActiveXChecks(Me, Shapes("CheckBox1"))
    End If
End Sub

'Following code can be in the same worksheet module as the click events. If placing in a standard module then dimension as Public bolDisable in the standard module; not in worksheet module.

Sub ProcessActiveXChecks(ws As Worksheet, shpChk As Shape)
    'Process ActiveX CheckBoxes in Shapes Groups. Called from ActiveX Click Events
    bolDisable = True
    Dim shpGrp As Shape
    Dim oleObj As OLEObject
    Dim i As Long
   
    Set shpGrp = shpChk.ParentGroup
    For i = 1 To shpGrp.GroupItems.Count
        Set oleObj = ws.OLEObjects(shpGrp.GroupItems(i).Name)
        oleObj.Object.Value = False
    Next i
    Set oleObj = ws.OLEObjects(shpChk.Name)
    oleObj.Object.Value = True
    bolDisable = False
End Sub

The following code is a utility to create forms control checkboxes in groups. To test just copy my code Sub ProcessChkBoxes() from above plus the code below into a standard module.

Run the code below and when the Inputbox pops up, follow the instructions to select the ranges where the groups of checkboxes are required. You will see that the Input creates a Union of the separate ranges and then groups the checkboxes that are created in each area of the union.

Using the utility, additional groups of checkboxes can be added at any time and as I said previously, the names of the checkboxes and groups are irrelevant in the processing code to uncheck boxes in the group. The utility currently places the checkbox at the right of a cell but that could be easily modified.

Sub CreateCheckBoxes()
    'Use this code to create checkboxes with groups
    'At InputBox select first range and then hold Ctrl Key to select remaining ranges for the CheckBox groups.
    Dim strPrompt As String
    Dim ws As Worksheet
    Dim rngUnion As Range
    Dim rngArea As Range
    Dim cel As Range
    Dim arrChkBoxes()
    Dim i As Long
    'Dim j As Long  'Only required for re-naming controls.
    Dim shp As Object
    Dim shpGrp As Shape
    Dim dblLeft As Double
    Dim dblTop As Double
    Dim dblWidth As Double
    Dim dblHeight As Double
       
    strPrompt = "1. Select the first range for the first group of CheckBoxes." & vbCrLf _
                & "2. Hold the Ctrl Key and select the range for the second group." & vbCrLf _
                & "3. Repeat until all ranges are selected for the groups of CheckBoxes." & vbCrLf _
                & "4. When all required ranges are selected, click OK."
   
    On Error Resume Next    'Cannot assign the Cancel to an object variable
    Set rngUnion = Application.InputBox(Prompt:=strPrompt, Type:=8)
    On Error GoTo 0
   
    If rngUnion Is Nothing Then
        MsgBox "User cancelled. Processing terminated."
        Exit Sub
    End If
   
    Set ws = rngUnion.Parent
    Application.Goto ws.Cells(1, 1)    'Activate worksheet with the selected ranges
   
    For Each rngArea In rngUnion.Areas
        i = 1
        ReDim arrChkBoxes(1 To i)   'Reset to one element. (Can't redim preserve without at least one element)
        For Each cel In rngArea
            dblLeft = cel.Left + cel.Width - cel.Height  'All CheckBoxes positioned at right of cell
            dblWidth = cel.Height               'All CheckBoxes square (Height and Width same)
            dblHeight = cel.Height              'Set Height to height of row
            dblTop = cel.Top                    'Set Top to top of row
            Set shp = ActiveSheet.CheckBoxes.Add(dblLeft, dblTop, dblWidth, dblHeight)
            'j = j + 1
            'shp.Name = "Check Box  " & j       'Errors if name duplicated. Best to not rename
            ReDim Preserve arrChkBoxes(1 To i)
            arrChkBoxes(i) = shp.Name   'Array is used to create the Group
            i = i + 1
            shp.OnAction = "ProcessChkBoxes"    'Edit this line to reflect required sub to be called
            shp.Characters.Text = ""
        Next cel
        If UBound(arrChkBoxes) > 1 Then     'Must have more than one checkbox to group
            Set shpGrp = ActiveSheet.Shapes.Range(arrChkBoxes()).Group
        End If
    Next rngArea
  
End Sub

Sub SetCheck(strChkBox As String, strGrp As String)

  If Not Application.EnableEvents Then Exit Sub
  Application.EnableEvents = False
  For Each ctrl In OLEGroupItems(strGrp)

  Next
  Application.EnableEvents = True
End Sub
