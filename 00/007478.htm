jay freedman say:

There is no single command that corresponds to the "Accept This Change" entry on the Accept button's menu. However, it's possible to simulate it with a macro. Install this macro (see http://www.gmayor.com/installing_macro.htm if needed) and assign a keyboard shortcut to the macro:

Sub AcceptOnly()
    With Selection.Range.Revisions
        If .Count > 0 Then
            .Item(1).Range.Select
        End If
    End With
    Selection.Range.Revisions.AcceptAll
End Sub

If you also want a shortcut to reject the current revision and stay in place, the macro for that is similar:

Sub RejectOnly()
    With Selection.Range.Revisions
        If .Count > 0 Then
            .Item(1).Range.Select
        End If
    End With
    Selection.Range.Revisions.RejectAll
End Sub

shane devenshire say:

Hi,

Dim cell As Range
Dim cel As Range

Sub Main()
    Application.ScreenUpdating = False
    Sheets("Menu").Activate
    For Each cell In Range("B27:B51")
        If cell <> "" Then
            CabComponents
            Doors
            DrawerBoxes
            HardwareList
        End If
    Next cell
End Sub
Sub CabComponents()
    For Each cel In Sheets(cell.Value).Range("I19:I64")
        If cel > 0 Then
            Sheets(cell.Value).Range("B" & cel.Row & ":K" & cel.Row).Copy
            If Sheets("Cab Components").Range("B35") = "" Then
                Sheets("Cab Components").Range("B35").PasteSpecial xlValues
            Else
                Sheets("Cab Components").Range("B34").End(xlDown).Offset(1, 0).PasteSpecial xlValues
            End If
        End If
    Next cel
End Sub

Sub Doors()
    For Each cel In Sheets(cell.Value).Range("I68:I92")
        If cel > 0 Then
            Sheets(cell.Value).Range("B" & cel.Row & ":I" & cel.Row).Copy
            If Sheets("Doors").Range("B25") = "" Then
                Sheets("Doors").Range("B25").PasteSpecial xlValues
            Else
                Sheets("Doors").Range("B24").End(xlDown).Offset(1, 0).PasteSpecial xlValues
            End If
        End If
    Next cel
End Sub

Sub DrawerBoxes()
    For Each cel In Sheets(cell.Value).Range("I96:I101")
        If cel > 0 Then
            Sheets(cell.Value).Range("B" & cel.Row & ":I" & cel.Row).Copy
            If Sheets("Drawer Boxes").Range("B25") = "" Then
                Sheets("Drawer Boxes").Range("B25").PasteSpecial xlValues
            Else
                Sheets("Drawer Boxes").Range("B24").End(xlDown).Offset(1, 0).PasteSpecial xlValues
            End If
        End If
    Next cel
End Sub

Sub HardwareList()
    For Each cel In Sheets(cell.Value).Range("G120:G125")
        If cel > 0 Then
            Sheets(cell.Value).Range("B" & cel.Row & ":I" & cel.Row).Copy
            If Sheets("Hardware List").Range("DB18") = "" Then
                Sheets("Hardware List").Range("DB18").PasteSpecial xlValues
            Else
                Sheets("Hardware List").Range("DB17").End(xlDown).Offset(1, 0).PasteSpecial xlValues
            End If
        End If
    Next cel
End Sub

I could shorten it a bit but it might become more obtuse.  I'm not sure what you are using the Worksheet_Activate code in most of the sheets for, but it is not necessary for the above code and it slows down the execution.
