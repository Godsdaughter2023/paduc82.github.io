This code shows just the basics how to use a dictionary to count the words, build a collection with all item objects parallel to the dictionary, get the page numbers for each word and how to store the result in excel.

Andreas.

Sub Main()
  Dim D As Object
  Dim Item As Range
  Dim AllItems As New Collection
  Dim AllWords, AllCounts, AllPages
  Dim I As Long
  Dim xlApp As Excel.Application
  Dim xlWorkbook As Excel.Workbook
  Dim xlWorksheet As Excel.Worksheet
  '
  'Get or open excel
  On Error Resume Next
  Set xlApp = GetObject(, "Excel.Application")
  If xlApp Is Nothing Then
    Set xlApp = CreateObject("Excel.Application")
    If xlApp Is Nothing Then
      MsgBox "Excel is not accessable"
      Exit Sub
    End If
  End If
  'Add a new WorkBook
  Set xlWorkbook = xlApp.Workbooks.Add
  'Get current sheet
  Set xlWorksheet = xlWorkbook.ActiveSheet
  '
  'Create a dictionary
  Set D = CreateObject("Scripting.Dictionary")
  'Ignore the spelling of the words
  D.CompareMode = vbTextCompare
  'Count all words
  For Each Item In ActiveDocument.Words
    If Not D.Exists(Item.Text) Then
      'Add the text to our dictionary
      D.Add Item.Text, 1
      'Store the item in the collection too
      AllItems.Add Item
    Else
      'Count this word
      D.Item(Item.Text) = D.Item(Item.Text) + 1
    End If
  Next
  'Get all words in one array
  AllWords = D.Keys
  AllCounts = D.Items
  'Retrieve all page numbers using the collection
  ReDim AllPages(0 To UBound(AllWords))
  For Each Item In AllItems
    AllPages(I) = Item.Information(wdActiveEndPageNumber)
    I = I + 1
  Next
  '
  'Write the arrays to excel
  With xlWorksheet
    'Write header
    .Range("A1").Resize(1, 3) = Array("Words", "Count", "Page")
    'Write words
    .Range("A2").Resize(UBound(AllWords) + 1, 1) = _
      xlApp.WorksheetFunction.Transpose(AllWords)
    'Write count
    .Range("B2").Resize(UBound(AllCounts) + 1, 1) = _
      xlApp.WorksheetFunction.Transpose(AllCounts)
    'Write page numbers
    .Range("C2").Resize(UBound(AllPages) + 1, 1) = _
      xlApp.WorksheetFunction.Transpose(AllPages)
  End With
End Sub
