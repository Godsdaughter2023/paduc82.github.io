dirk goldgar say:

How about this approach?  Have the code export to .xls, then automate an instance of Excel to convert that to .xlsx format?

Here's a quick & dirty procedure that demonstrates this approach:

'------ start of code ------

Sub ExportReportToXLSX(reportName As String, outputFile As String)

    On Error GoTo Err_Handler

    ' Export a report to Excel, in XLSX format (not supported by DoCmd.OutputTo).
    '
    ' Arguments:
    '   reportName      - the name of the report to be exported
    '   outputFile      - the path & name of the output Excel workbook

    Const xlOpenXMLWorkbook = 51

    Dim objXL       As Object     ' Excel.Application
    Dim objWB       As Object     ' Excel.Workbook
    Dim strWorkXLS  As String
    Dim blnOpened   As Boolean
    
    If Not (CurrentProject.AllReports(reportName).IsLoaded) Then
        DoCmd.OpenReport reportName, acViewReport, WindowMode:=acHidden
        blnOpened = True
    End If

    strWorkXLS = CurrentProject.path & "\$TEMP$_" & reportName & ".xls"
    
    If fncFileExists(strWorkXLS) Then
        Kill strWorkXLS
    End If
    
    DoCmd.OutputTo acOutputReport, reportName, acFormatXLS, CurrentProject.path & "\$TEMP$_" & reportName & ".xls", False

    Set objXL = CreateObject("Excel.Application")
    Set objWB = objXL.Workbooks.Open(strWorkXLS)
    objWB.SaveAs outputFile, xlOpenXMLWorkbook
    
    Set objWB = Nothing
    
    objXL.Quit
    Set objXL = Nothing

    Kill strWorkXLS
    
    If blnOpened Then
        DoCmd.Close acReport, reportName, acSaveNo
    End If
    
Exit_Point:
    On Error Resume Next
    Set objWB = Nothing
    If Not objXL Is Nothing Then
        objXL.Quit
        Set objXL = Nothing
    End If
    Kill strWorkXLS
    If blnOpened Then
        DoCmd.Close acReport, reportName, acSaveNo
    End If
    Exit Sub
    
Err_Handler:
    MsgBox Err.Description, vbExclamation, "ERROR " & Err.Number
    Resume Exit_Point
    
End Sub
'------ end of code ------

The above function uses another function, fncFileExists, to determine whether or not the work file exists.  There are various implementations to this floating around, so feel free to pick or write your own.
