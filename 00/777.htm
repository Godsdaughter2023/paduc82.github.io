ossiemac say:
Even specifying DMY format in FieldInfo

When specifying the date format with TextToColumns you should specify the date format that is being used in the Source Text Data; not the date format that you want to end up with.

 

Therefore if the Source Text data is in MDY format then select the MDY format in the TextToColumns. The dates should then get translated to the regional date format of your computer. The reason for this is that if a date is in text format and the day is =< than 12, then Excel cannot determine if it is MDY or DMY so it is necessary to specify before the data is processed.

 

There are a couple of methods of handling the second part of your question where the date columns are not constant.

 

Your post indicates that you are using VBA so I am assuming that you have sufficient knowledge of VBA to handle the following answer but if not then please get back to me and I will provide some further explanation

 

What you could do is in TextToColumns options, specify TEXT format for ALL columns and then manually identify the columns to be converted. The reason for specifying Text for ALL columns is that if the columns are left as General then TextToColumns will import any MDY dates with a day <= 12 to DMY dates and they will not be the correct date.

 

Now with all of the data in Text format, copy the worksheet to a Macro Enabled workbook that contains the code below and you can use a DoubleClick event to manually identify each column in turn with the MDY dates and the code will convert one column at a time. The code requires that you double click the first cell in the column.

 

Private Sub Worksheet_BeforeDoubleClick(ByVal Target As Range, Cancel As Boolean)

    Cancel = True

    If Target.Row = 1 Then

        Target.EntireColumn.TextToColumns Destination:=Target, DataType:=xlFixedWidth, _

            FieldInfo:=Array(0, 3), TrailingMinusNumbers:=True

    End If

End Sub

 

If you want to fully automate and your code is in a different workbook and you are simply opening the txt file which becomes a txt workbook.

Import with TextToColumns and again set all columns to TEXT in the TextToFomat options. Then code like the following could identify the columns that appear to have MDY dates and re-apply TextToColumns to just those columns.

 

Following Code Modified since originally posted. Unnecessary parameters removed in case your txt file does not exactly match delimiters etc with the test one that I used. Have also added test for Numeric data as per later post.

Sub ConvertDateColumns()
    Dim wb As Workbook
    Dim ws As Worksheet
    Dim rng As Range
    Dim cel As Range
    Dim arrSplit
    Dim strYear As String
    Dim strMth As String
    Dim strDay As String
   
    Set wb = Workbooks("MyTextData.txt")    'File needs to be open
   
    Set ws = wb.Sheets("MyTextData")    'worksheet name same as txt file name
   
    With ws
        'Set Test range to row 2. Assumes column headers in row 1.
        Set rng = .Range(.Cells(2, 1), .Cells(2, .Columns.Count).End(xlToLeft))
    End With
   
    For Each cel In rng
        'If date delimiter in Text data is not "/" then edit next line to correct delimiter
        arrSplit = Split(cel.Value, "/")    'Creates zero based array of Mth, Day and Year
       
        If UBound(arrSplit) = 2 Then        'If not 3 elements then not a date
            strMth = arrSplit(0)            'First element of array
            strDay = arrSplit(1)            'Second element of array
            strYear = arrSplit(2)           'Third element of array
       
            If IsDate(strDay & "/" & strMth & "/" & strYear) Then
               
                cel.EntireColumn.TextToColumns Destination:=cel.Offset(-1, 0), _
                    FieldInfo:=Array(1, 3)
            End If
        Else
            If IsNumeric(cel.Value) Then
                cel.EntireColumn.TextToColumns Destination:=cel.Offset(-1, 0), _
                        FieldInfo:=Array(0, 1)
                       
                cel.EntireColumn.NumberFormat = "0"     'Edit format as required
            End If
        End If
    Next cel
   
    Set cel = Nothing
    Set rng = Nothing
    Set ws = Nothing
    Set wb = Nothing
End Sub

Private Function Min(Arg, ParamArray Args())
  'Liefert das kleinste Element
  Dim i As Long
  On Error Resume Next
  Min = Arg
  For i = LBound(Args) To UBound(Args)
    If Args(i) < Min Then Min = Args(i)
  Next
End Function

Function DateString2Date(ByVal DateTimeString As String, _
    Optional ByVal DateTimeMask As String = "DD.MM.YYYY") As Date
  Dim Delimiter As String, Temp As String, This As String
  Dim i As Long
  Dim DateArr, MaskArr
  Dim Day As Integer, Month As Integer, Year As Integer
  Dim Hour As Integer, Minute As Integer, Second As Integer
  'Unify the delimiters
  For i = 1 To Len(DateTimeMask)
    Temp = Mid$(DateTimeMask, i, 1)
    Select Case Temp
      Case "D", "T", "M", "Y", "J", "h", "m", "s"
      Case Else
        If Len(Delimiter) = 0 Then
          Delimiter = Temp
        ElseIf Temp <> Delimiter Then
          DateTimeMask = Replace$(DateTimeMask, Temp, Delimiter)
          DateTimeString = Replace$(DateTimeString, Temp, Delimiter)
        End If
    End Select
  Next
  If Len(Delimiter) = 0 Then
    'Add a delimiter between different formatters
    Delimiter = "-"
    For i = Min(Len(DateTimeString), Len(DateTimeMask)) To 2 Step -1
      Temp = Mid$(DateTimeMask, i - 1, 1)
      This = Mid$(DateTimeMask, i - 0, 1)
      Select Case Temp
        Case "D", "T", "M", "Y", "J", "h", "m", "s"
          If Temp <> This Then
            DateTimeMask = Left$(DateTimeMask, i - 1) & Delimiter & Mid$(DateTimeMask, i)
            DateTimeString = Left$(DateTimeString, i - 1) & Delimiter & Mid$(DateTimeString, i)
          End If
      End Select
    Next
  End If
  'Split the strings
  DateArr = Split(DateTimeString, Delimiter)
  MaskArr = Split(DateTimeMask, Delimiter)
  'Get the DMY parts
  For i = 0 To Min(UBound(DateArr), UBound(MaskArr))
    If IsNumeric(DateArr(i)) Then
      Select Case Left$(MaskArr(i), 1)
        Case "h":      Hour = DateArr(i)
        Case "m":      Minute = DateArr(i)
        Case "s":      Second = DateArr(i)
        Case "D", "T": Day = DateArr(i)
        Case "M":      Month = DateArr(i)
        Case "Y", "J": Year = DateArr(i)
      End Select
    End If
  Next
  'Calculate the date
  If (Year > 0) Or (Month > 0) Or (Day > 0) Then
    If Year = 0 Then Year = VBA.Year(Date)
    If Month = 0 Then Month = 1
    If Day = 0 Then Day = 1
    DateString2Date = DateSerial(Year, Month, Day) + TimeSerial(Hour, Minute, Second)
  Else
    DateString2Date = TimeSerial(Hour, Minute, Second)
  End If
End Function
