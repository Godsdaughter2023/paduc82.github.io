jllatham say:

What part of it doesn't seem to be working for you?

 

Let's try a rewrite in a very 'generic' form of the code, where you could have many names in both lists, not just 2 names.  I'll provide some Const entries that you can edit to make it work with any two sheets, and any columns that might be involved

 

Sub GetInvoiceTotals()
'change these Const values as required
'by changing these here, you can get the code to
'work with any set up involving 2 lists of names and
'a column of numbers to total and report on
  'name of sheet with multiple entries of mike, pat, etc.
  Const mainSheetName = "Sheet1"
  Const mainNamesColumn = "A" ' column names are in
  Const mainAmtsColumn = "B" ' column invoice amounts are in
  Const mainFirstEntryRow = 1 ' where first name is entered
  'name of the sheet where you want to put the totals

  'this could even be the same as mainSheetName but then

  ' totalsNameColumn would need to be changed
  Const totalsSheetName = "Sheet2"
  Const totalsNameColumn = "A" ' column with 1 entry of each name
  Const totalsAmtColumn = "C" ' where to put grand totals
  Const totalsFirstEntryRow = 1 ' where first name is entered
'end of user redefinable Const values

 

  'working variables
  Dim mainWS As Worksheet
  Dim mainNamesList As Range
  Dim anyMainNameCell As Range
 
  Dim totalsWS As Worksheet
  Dim totalsNamesList As Range
  Dim anyTotalsNameCell As Range
 
  Dim grandTotal As Variant ' could be integer, long, currency, etc.
  Dim anyLastRow As Long
 
  'set up to work with the source list
  Set mainWS = Worksheets(mainSheetName) ' references Sheet1
  'set reference to all entries based on names on the source sheet
  'should point to row 20 in your list
  anyLastRow = mainWS.Range(mainNamesColumn & Rows.Count).End(xlUp).Row
  'build a reference to cells A1:A20
  'this should be the same as if we did this (if anyLastRow = 20)
  'Set mainNamesList = mainWS.Range("A" & 1 & ":" & "A" & 20)
  Set mainNamesList = mainWS.Range(mainNamesColumn & mainFirstEntryRow & _
   ":" & mainNamesColumn & anyLastRow)
 
  'set up to work with the 'destination' worksheet
  'it could even be the same worksheet, but of course it would
  'involve different columns, and usually fewer rows
  Set totalsWS = Worksheets(totalsSheetName) ' reference to Sheet2
  'we do the same thing here to find out how long the list of names is
  anyLastRow = totalsWS.Range(totalsNameColumn & Rows.Count).End(xlUp).Row
  'build a reference to cells A1:A2 (only Mike & Pat in the list)
  'this should be the same as if we did this (if anyLastRow = 2)
  'Set totalsNamesList = totalsWS.Range("A" & 1 & ":" & "A" & 2)
  Set totalsNamesList = totalsWS.Range(totalsNameColumn & mainFirstEntryRow & _
   ":" & totalsNameColumn & anyLastRow)
  'now we are ready to actually do the work
  'we will go through each name in the short list on Sheet2
  'and compare those names up to each name on Sheet1 and
  'where they match, we will add up the invoice amounts
  For Each anyTotalsNameCell In totalsNamesList
    'reset grandTotal to zero for this person
    grandTotal = 0
    'now go through the long list of names on Sheet1
    For Each anyMainNameCell In mainNamesList
      'need to remove any leading/trailing spaces
      'from both names and convert to UPPERCASE to
      'compare in VBA
      If UCase(Trim(anyTotalsNameCell)) = UCase(Trim(anyMainNameCell)) Then
        'the names match, add in the invoice amount
        'we know we need to get the amount from the same row that we
        'matched a name to at this point, but we need to get that
        'from column B
        'we could have used an .Offset, which is faster, but this works also.
        grandTotal = grandTotal + mainWS.Range(mainAmtsColumn & anyMainNameCell.Row)
      End If
    Next ' end anyMainNameCell loop
    'report the results on the second sheet on the same row with
    'the name we are looking at on that sheet
    totalsWS.Range(totalsAmtColumn & anyTotalsNameCell.Row) = grandTotal
  Next ' end anyTotalsNameCell loop
  'the job is done, the next steps are not required
  'in theory - but I consider it a Best Practice to
  'ensure no "memory leaks" if the code is run many times
  Set totalsNamesList = Nothing
  Set totalsWS = Nothing
  Set mainNamesList = Nothing
  Set mainWS = Nothing
  'announce job completed
  MsgBox "Task Accomplished"
End Sub
