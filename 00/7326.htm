Private Sub Worksheet_Change(ByVal Target As Range)
    Dim OldValue
    If Target.Address <> "$B$1" Then Exit Sub
    Application.EnableEvents = False
    With Me
        If .Range("B1") > .Range("A1") Then

Private Sub Worksheet_Change(ByVal Target As Range)
  Dim A As Range, B As Range, C As Range, D As Range
  Dim AChange As Boolean, BChange As Boolean
  
  'Only if one value changes
  If Target.Count > 1 Then Exit Sub
  'Where is the change?
  AChange = Target.Column = 1
  BChange = Target.Column = 2
  
  'Only in column A and B
  If Not AChange And Not BChange Then Exit Sub
  
  'Refer to the cells in the current row
  Set A = Range("A" & Target.Row)
  Set B = Range("B" & Target.Row)
  Set C = Range("C" & Target.Row)
  Set D = Range("D" & Target.Row)
  
  'Events off, otherwise we call ourself
  Application.EnableEvents = False
  
  'Your rules:
  If A > B Then
    D = C
  End If
  If BChange Then
    C = 0
  End If
  
  'Events on
  Application.EnableEvents = True
End Sub

Sub CountCcolorR1(range_data As Variant, C_criteria As Variant, lngTotal As Long)
'Jim Cone - July 2015 - Portland, Oregon USA
'Counts the quantity of cells in the specified range that are the same color as the selected cell.
'Only valid on xl2010 or later excel versions.
'Standard and conditional formatted colors are counted.
'Merged cells are considered one cell.

 On Error GoTo Black
 Dim datax As Range
 Dim xcolor As Long
 Dim ycolor As Long
 Dim arrMerged() As Variant
 Dim N As Long
 Dim Ndx As Long
 Dim blnFlag As Boolean
 
 If TypeName(C_criteria) = "Range" And TypeName(range_data) = "Range" Then
   ReDim arrMerged(1 To range_data.Cells.Count, 1 To 1)
   Set arrMerged(1, 1) = range_data(range_data.Cells.Count).Offset(1, 1)
   
   xcolor = C_criteria.DisplayFormat.Interior.Color
   For Each datax In range_data.Cells
      ycolor = datax.DisplayFormat.Interior.Color
      If datax.MergeArea.Cells.Count > 1 Then
        For Ndx = 1 To range_data.Cells.Count
          If Not IsEmpty(arrMerged(Ndx, 1)) Then
            If Not Application.Intersect(datax, arrMerged(Ndx, 1)) Is Nothing Then
              blnFlag = True
              Exit For
            End If
          Else
            Exit For
          End If
        Next 'Ndx
        If Not blnFlag Then
          N = N + 1
          Set arrMerged(N, 1) = datax.MergeArea
          If ycolor = xcolor Then lngTotal = lngTotal + 1
        Else
          blnFlag = False
        End If
      Else
        If ycolor = xcolor Then lngTotal = lngTotal + 1
      End If
   Next 'datax
 End If
 Exit Sub
 
Black:
 MsgBox Err.Number & " " & Err.Description
End Sub
