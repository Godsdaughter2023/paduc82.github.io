Copy the code below into a regular module and use a formula like this:

Andreas.

Function Encode(ByVal What) As String
  Dim i As Long, j As Long
  Dim CodeA As String, CodeB As String
  CodeA = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ"
  CodeB = "ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789"
  For i = 1 To Len(What)
    j = InStr(CodeB, Mid$(What, i, 1))
    If j > 0 Then
      Mid$(What, i, 1) = Mid$(CodeA, j, 1)
    End If
  Next
  Encode = What
End Function
