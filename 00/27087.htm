andreas killer say:
You can't copy macros from one application to another one, because the other application did not know of the constants. Place the line
  Option Explicit
at the top of the module, compile your project and you'll see you miss a lot.

Never use SELECT, SELECTION in your macros, it's slow and error prone.

Don't use strings to create the address of cells, use the range object directly.

And what you try to do is IMHO really a bad idea. A much easier way is to create a template file in Excel, format the heading and the second row as the data should look like. After you copy the data into the sheet, copy the formats from row 2 down to the end of the data.

If a very special formatting is really needed, create a macro in the template and run that from Access, e.g.: Apxl.Run "MySpecialMacro"

That is very easy and reduces your macro to a very few lines.

Andreas.


Public Sub SendTQ2XLWbSheet(ByVal strTQName As String, ByVal strSheetName As String, _
    ByVal strFilePath As String, ByVal strTemplateFilePath As String)
  ' strTQName is the name of the table or query you want to send to Excel
  ' strSheetName is the name of the sheet you want to send it to
  ' strFilePath is the name and path of the file you want to send this data into.

  Dim rst As DAO.Recordset
  Dim Apxl As Object  'Excel.Application
  Dim xlWBk As Object 'Excel.Workbook
  Dim xlWSh As Object 'Excel.Worksheet
  Dim xlRng As Object 'Excel.Range
  Dim fld As DAO.field
  Dim FileExists As Boolean, WasOpen As Boolean
  Dim i As Long

  'Get or open Excel
  On Error Resume Next
  Set Apxl = GetObject(, "Excel.Application")
  WasOpen = True
  If Apxl Is Nothing Then
    Set Apxl = CreateObject("Excel.Application")
    WasOpen = False
  End If
  If Apxl Is Nothing Then Err.Raise 1000, "SendTQ2XLWbSheet", "Excel is not accessible"
  On Error GoTo ErrorHandler

  ' Check if Excel Workbook exists
  FileExists = WorkBookExists(strFilePath)
  If Not FileExists Then
    Set xlWBk = Apxl.Workbooks.Add(strTemplateFilePath)
  Else
    Set xlWBk = Apxl.Workbooks.Open(strFilePath)
  End If
  ' Check if Excel Worksheet name exists
  If Not SheetExists(strSheetName, xlWBk) Then
    Set xlWSh = xlWBk.Worksheets.Add
    xlWSh.Name = strSheetName
  Else
    Set xlWSh = xlWBk.Worksheets(strSheetName)
  End If

  xlWSh.Unprotect "Password"

  Set rst = CurrentDb.OpenRecordset(strTQName)
  'Get column headers
  With rst
    For i = 1 To .Fields.Count
      xlWSh.Cells(1, i) = .Fields(i - 1).Name
    Next i
  End With
  'Copy the data
  rst.MoveFirst
  xlWSh.Range("A2").CopyFromRecordset rst
  rst.Close
 
  'Copy the formats from row 2 down to the end of data
  Set xlRng = xlWSh.UsedRange
  Set xlRng = xlRng.Resize(xlRng.Rows.Count - 2).Offset(2)
  xlWSh.Rows(2).Copy
  xlRng.PasteSpecial -4122 'xlPasteFormats
  Apxl.CutCopyMode = False
 
  ' does the "autofit" for all columns
  xlWSh.Cells.EntireColumn.AutoFit

  xlWSh.Protect "Password"

  'Save the Workbook
  If FileExists Then
    xlWBk.Close SaveChanges:=True
  Else
    xlWBk.SaveAs strFilePath
  End If
  'Quit Excel
  If Not WasOpen Then Apxl.Quit

  Exit Sub
ErrorHandler:
  DoCmd.SetWarnings True
  If Err.Source = "" Then Err.Source = Application.Name
  Debug.Print "Source     : " & Err.Source
  Debug.Print "Error      : " & Err.Number
  Debug.Print "Description: " & Err.Description
  If MsgBox("Error " & Err.Number & ": " & vbNewLine & vbNewLine & _
      Err.Description & vbNewLine & vbNewLine & _
      "Enter debug mode?", vbOKCancel + vbDefaultButton2, Err.Source) = vbOK Then
    Stop 'Press F8 twice
    Resume
  End If
End Sub
