Hi Greg,

A short look into the Object Browser shows that AppendChildNode is a Sub:


Your code looks a little complicated, IMHO we can simplify it a bit.

Andreas.


Option Explicit

Const MyMainNode = "DocumentElement_Test"
Const MyNameSpace = "http://somenamespace/GKM"
Private m_oXMLPart As CustomXMLPart

Property Get AddressBook() As CustomXMLPart
  Dim MyXML As String
  'Already initialized?
  If m_oXMLPart Is Nothing Then
    On Error Resume Next
    'Get it from namespace
    Set m_oXMLPart = ThisDocument.CustomXMLParts.SelectByNamespace(MyNameSpace)
    'Success?
    If m_oXMLPart Is Nothing Then
      'No, add a new one
      MyXML = "<?xml version='1.0'?><@MN xmlns='@NS'></@MN>"
      MyXML = Replace(MyXML, "@MN", MyMainNode)
      MyXML = Replace(MyXML, "@NS", MyNameSpace)
      Set m_oXMLPart = ThisDocument.CustomXMLParts.Add(MyXML)
    End If
  End If
  Set AddressBook = m_oXMLPart
End Property

Sub AddChildNodesInDocumentElement()
  Dim oEntry As CustomXMLNode
  With AddressBook.DocumentElement
    'Add a new node
    .AppendChildNode "Entry", , msoCustomXMLNodeElement
    'Get it
    Set oEntry = .LastChild
    'Add 2 sub nodes
    With oEntry
      .AppendChildNode "FirstName", , msoCustomXMLNodeElement, "Greg"
      .AppendChildNode "LastName", , msoCustomXMLNodeElement, "Maxey"
    End With
  End With
End Sub
