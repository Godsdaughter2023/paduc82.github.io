bill manville say:

I got a bit lost in your description, but I think you are looking for code that will automatically truncate the content of a cell into which you enter data to as many whole words as will fit in no more than 35 characters.  This should do it.  In the module behind the worksheet:

Private Sub Worksheet_Change(ByVal Target As Excel.Range)
  Dim C As Range

  For Each C In Target.Cells
    If Len(C.Text)>35 Then
      C.Value = Left(C.Text, InStrRev(Left(C.Text,35)," ")-1)
    End If
  Next

End Sub

If you want to limit the action to, for example, column A of the worksheet then change this to

Private Sub Worksheet_Change(ByVal Target As Excel.Range)
  Dim C As Range
  If Intersect(Me.Range("A:A"), Target) Is Nothing Then Exit Sub
  For Each C In Intersect(Me.Range("A:A"), Target).Cells
    If Len(C.Text) > 35 Then
      C.Value = Left(C.Text, InStrRev(Left(C.Text, 35), " ") - 1)
    End If
  Next

End Sub
