doug robbins - mvp office apps & services (word) say:

You would need to select the list and run a macro containing the following code

 

' Macro created 12/02/2006 by Doug Robbins to apply/re-apply reverse sequential numbering

Dim i As Long, Numrange As Range

Set Numrange = Selection.Range

With Numrange

    For i = 1 To .Paragraphs.Count

        If IsNumeric(.Paragraphs(i).Range.Characters(1)) Then

            With .Paragraphs(i).Range

                While IsNumeric(.Characters(1))

                    .Characters(1).Delete

                Wend

                .Characters(1).Delete

            End With

        End If

        .Paragraphs(i).Range.InsertBefore .Paragraphs.Count - i + 1 & vbTab

    Next i

End With
