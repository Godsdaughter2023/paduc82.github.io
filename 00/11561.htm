bernie deitrick say:

Just use an IIF:

Sub test()
  Dim r As Long
    r = 2
    Do Until Range("C" & r).Value = ""
        Range("B" & r).Value = Range("C" & r).Text & " - " & _
            Range("K" & r).Text & " - " & Range("L" & r).Text & " - " & _
            Range("M" & r).Text & IIf(Range("N" & r).Text <> "", " - " _
            & Range("N" & r).Text, "") & " - " & Date
        r = r + 1
    Loop
End Sub
