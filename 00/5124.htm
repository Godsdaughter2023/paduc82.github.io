i want to lock all other column except today's date in first raw

can you please help me with the VBA code 

basic idea is to prevent entry in all other columns except today

[Moved from Office/Excel/Windows 10/Office 365 for business]

andreas killer say:
Copy the code below into the code module ThisWorkbook.
Customize the sheet name in Workbook_Open
Save the file as XLSM, close and reopen the file

Andreas. 

Option Explicit
Private Sub Workbook_Open()
  Dim Ws As Worksheet
  Dim Where As Range
  'Refer to the sheet, customize the sheet name:
  Set Ws = Sheets("Sheet1")
  
  'Find today in the first row
  Set Where = FindDatePrim(Ws.Range("1:1"), Date)
  'Found?
  If Where Is Nothing Then
    MsgBox "today not found"
    Exit Sub
  End If
  With Ws
    'Unprotect the sheet
    .Unprotect
    'Unlock all cells (that includes yesterday column)
    .Cells.Locked = False
    'Lock the todays column
    Where.EntireColumn.Locked = True
    'Protect the sheet
    .Protect
  End With
End Sub
Function FindDatePrim(ByVal Where As Range, ByVal What As Double) As Range
  'Search for a date in a single range, returns the cell if found or nothing
  Dim Data, i As Long, j As Long, Value As Double
  'Read in all values
  Data = Where.Value2
  'Compare each cell
  For j = 1 To UBound(Data, 2)
    For i = 1 To UBound(Data)
      If Data(i, j) = What Then
        'Return the cell if found
        Set FindDatePrim = Where.Cells(i, j)
        Exit Function
      End If
    Next
  Next
End Function
