Four possible solutions

The code for the VBA is

Private Sub Worksheet_Change(ByVal Target As Range)

If Intersect(Target, Range("F1:F100")) Is Nothing Or Target.Cells.Count > 1 Then Exit Sub
myValue = Target.Value
Set Lookup_Range = Range("J1:J8")
x = Application.WorksheetFunction.VLookup(myValue, Lookup_Range, 1, False)
If x <> myValue Then
  MsgBox "Incorrect, must be A, B, C, D, E, NS, Ung, or NR"
End If
End Sub
