Sub Random()()

    Dim StartState As Single

    Application.Calculation = xlCalculationManual
 

'Declare the variables
    Dim i As Double
    Dim a As Integer                                                
    Dim b As Integer                                                
    Dim Array(10000) As Integer                                 
    Dim Destination As Range                                        
           
    Set Destination = Range("A2")
    Set Destination = Destination.Resize(UBound(Array), 1)              
    
    a = 0

    For i = LBound(Array) To UBound(Array)

    Next i
