hansv mvp say:

The CONCATENATE function won't help. You could use code - try this as starting point; you can modify and expand it as needed:

Sub SetValue()
    Dim s1 As String
    Dim s2 As String
    s1 = Range("A1").Text
    s2 = Range("A2").Text
    Range("A3").Value = s1 & s2
    With Range("A3").Characters(1, Len(s1)).Font
        .Name = Range("A1").Font.Name
        .Color = Range("A1").Font.Color
        .Bold = Range("A1").Font.Bold
        .Italic = Range("A1").Font.Italic
    End With
    With Range("A3").Characters(Len(s1) + 1).Font
        .Name = Range("A2").Font.Name
        .Color = Range("A2").Font.Color
        .Bold = Range("A2").Font.Bold
        .Italic = Range("A2").Font.Italic
