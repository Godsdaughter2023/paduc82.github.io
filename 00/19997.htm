                            Dim p As Object, dScaleFactor As Double

                            .RowHeight = 50

                            Set p = ActiveSheet.Shapes.AddPicture(Pic_Name, msoFalse, msoCTrue, _

                                    .Left + 10, _

                                    .Top + 18, _

                                    60, _

                                    60)

                                

                            dScaleFactor = 0.4

                            p.LockAspectRatio = msoTrue

                            p.ScaleWidth dScaleFactor, msoTrue

                           

                            ' Office 2003 or earily has to do ScaleHeight() also in order to keep the aspect ratio

                            If Application.Version <= 11 Then

                                p.ScaleHeight dScaleFactor, msoTrue

                            End If

 

                            .RowHeight = p.Height + 25

                            .Select

                            Set p = Nothing

                        End With

                    Else

                        MissList = MissList & ArtCode & ".jpg" & Chr(13)

                    End If

                End If

            Next

        End If

        i = i + 1

    Wend

   

    If MissList <> "" Then

        MissList = "Following article .jpg missing :" & Chr(13) & MissList

        MsgBox MissList, vbOKOnly

    End If

   

    Exit Sub

Next_Article:

    Resume Next

End Sub

    Call AddImage
