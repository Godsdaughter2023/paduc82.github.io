How about:

 

Sub ald()
Dim s As String
Dim rE As Range, rDel As Range
s = "**ABC**"
Set rE = Intersect(Range("E:E"), ActiveSheet.UsedRange)
Set rDel = Nothing
For Each r In rE
    If r.Text = s Then
        If rDel Is Nothing Then
            Set rDel = r
        Else
            Set rDel = Union(rDel, r)
        End If
    End If
Next
rDel.EntireRow.Delete
End Sub
