Private Sub Workbook_SheetActivate(ByVal Sh As Object)
If Sh.Name = "Sheet1" Then
    Application.ExecuteExcel4Macro "SHOW.TOOLBAR(""Ribbon"",False)"
Else
    Application.ExecuteExcel4Macro "SHOW.TOOLBAR(""Ribbon"",TRUE)"
End If
End Sub

We can eliminate the If..Then structure from the above and make it a one-liner (just change the "Sheet1" inside the line of code to the actual worksheet's name)...

Private Sub Workbook_SheetActivate(ByVal Sh As Object)
  Application.ExecuteExcel4Macro "SHOW.TOOLBAR(""Ribbon""," & (Sh.Name <> "Sheet1") & ")"
End Sub

 

Private Sub Workbook_SheetActivate(ByVal Sh As Object)
Dim v As Variant
v = Split("Sheet1,Sheet2,", ",")
 If Not IsError(Application.Match(Sh.Name, v, 0)) Then
    Application.ExecuteExcel4Macro "SHOW.TOOLBAR(""Ribbon"",False)"
Else
    Application.ExecuteExcel4Macro "SHOW.TOOLBAR(""Ribbon"",TRUE)"
End If
End Sub

We can eliminate the If..Then structure from the above and also remove the call-out to the worksheet Match function as well (using native VB functions instead)...

Private Sub Workbook_SheetActivate(ByVal Sh As Object)
  '  Comma delimited list of sheet names... no spaces around the commas
  Const HideRibbonSheets As String = "Sheet1,Sheet2,Sheet4"
  Application.ExecuteExcel4Macro "SHOW.TOOLBAR(""Ribbon""," & (InStr(1, "," & _
            HideRibbonSheets & ",", "," & Sh.Name & ",", vbTextCompare) = 0) & ")"
End Sub
