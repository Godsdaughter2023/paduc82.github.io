Option Explicit

Private Const LOGPIXELSX = 88
Private Const LOGPIXELSY = 90
#If Win64 Then
Private Declare PtrSafe Function GetDC Lib "user32" (ByVal hWnd As LongPtr) As LongPtr
Private Declare PtrSafe Function ReleaseDC Lib "user32" (ByVal hWnd As LongPtr, ByVal hDC As LongPtr) As Long
Private Declare PtrSafe Function GetDeviceCaps Lib "gdi32" (ByVal hDC As LongPtr, ByVal nIndex As Long) As Long
#Else
Private Declare Function GetDC Lib "user32" (ByVal hWnd As Long) As Long
Private Declare Function ReleaseDC Lib "user32" (ByVal hWnd As Long, ByVal hDC As Long) As Long
Private Declare Function GetDeviceCaps Lib "gdi32" (ByVal hDC As Long, ByVal nIndex As Long) As Long
#End If

Private Type POINTAPI
  x As Long
  y As Long
End Type
#If Win64 Then
Private Declare PtrSafe Function GetCursorPos Lib "user32" (ByRef lpPoint As POINTAPI) As Long
Private Declare PtrSafe Function SetCursorPos Lib "user32" (ByVal x As Long, ByVal y As Long) As Long
#Else
Private Declare Function GetCursorPos Lib "user32" (ByRef lpPoint As POINTAPI) As Long
Private Declare Function SetCursorPos Lib "user32" (ByVal x As Long, ByVal y As Long) As Long
#End If

Sub Example_RangeToScreenPixels()
  'Move the mouse cursor to the top left edge of the current cell, which must be visible
  SetCursorPos RangeToScreenPixelsX(ActiveCell), RangeToScreenPixelsY(ActiveCell)
End Sub

Function RangeToScreenPixelsX(ByVal r As Range) As Long
  'Returns the horizontal position of a cell on screen in pixels
  'Excel must be the active window if this function is called, otherwise the position is wrong!
  #If Win64 Then
  Dim hDC As LongPtr
  #Else
  Dim hDC As Long
  #End If
  Static dpp As Double
  If dpp = 0 Then
    hDC = GetDC(0)
    dpp = GetDeviceCaps(hDC, LOGPIXELSX) / 72
    ReleaseDC 0, hDC
  End If
  With ActiveWindow
    RangeToScreenPixelsX = .PointsToScreenPixelsX(r.Left * dpp)
  End With
End Function

Function RangeToScreenPixelsY(ByVal r As Range) As Long
  'Returns the vertical position of a cell on screen in pixels
  'Excel must be the active window if this function is called, otherwise the position is wrong!
  #If Win64 Then
  Dim hDC As LongPtr
  #Else
  Dim hDC As Long
  #End If
  Static dpp As Double
  If dpp = 0 Then
    hDC = GetDC(0)
    dpp = GetDeviceCaps(hDC, LOGPIXELSY) / 72
    ReleaseDC 0, hDC
  End If
  With ActiveWindow
    RangeToScreenPixelsY = .PointsToScreenPixelsY(r.Top * dpp)
  End With
End Function

Sub Example_ObjectFromMouseCursor()
  Dim O As Object
  Set O = ObjectFromMouseCursor
  If O Is Nothing Then
    MsgBox "Outside of the sheet"
  Else
    If TypeOf O Is Range Then
      MsgBox O.Address(0, 0)
    Else
      MsgBox O.Name
    End If
  End If
End Sub

Function ObjectFromMouseCursor() As Object
  'Returns the object below the mouse cursor
  'Excel must be the active window if this function is called, otherwise the function fails!
  Dim p As POINTAPI
  GetCursorPos p
  With ActiveWindow
    Set ObjectFromMouseCursor = .RangeFromPoint(p.x, p.y)
  End With
End Function

rich michaels - mvp (greatcirclelearning.com) say:
If I understand correctly you need to setup a 2-page template.

To do this first open a blank new document and do a Save AS, giving your template a name and choose the "dotx" format. Then open the Header and from the Header and Footer tab on the Ribbon choose mark the different first page option.

Graphics that you want to appear on the first only, must be inserted and positioned as you want them to appear, into the first page header. This includes any graphic that would be appearing in the footer area of the page. Note: The terms used here can be confusing. So saying it another way... A graphic that is to appear at the very bottom of the page must be inserted into the "Header" layer of the document. 

Word documents have "layers" there is a Header layer, Footer Layer, and a Body Layer. However, all background graphics that you want appearing in static places on specific pages must be placed into the Header layer.

When you return to the Body layer, the background images will appear washed out. This is done intentionally so that the writer can see the text they are typing into the body layer of the document. When the document is printed all of the background images will appear in full vibrance.

Hope this helps

ronrosenfeld say:

I believe what you want is the FV function:

=FV(PerCent,Term,Rent*12)

Note that we multiply the Rent by 12 since it is a monthly rent.  You may have to adjust the sign to get a positive or negative result to your liking -- it is a convention that payouts are negative, and income is positive, in the Excel financial formulas.
