It’s not feasible to set IMAP Mail folders as default data file. norman david jones say:

Hi Song.USA

Yes, your code works. Please adapt the code to save the new files in the same folder as the Emaildata file.

Try:

   

'==========>>
Option Explicit

'---------->>
Public Sub Tester()
'\\ Note it is necessary to set a reference to
    '\\ the Microsoft Scripting Runtime Library
    '\\ Menu |Tools | References
    Dim WB As Workbook, newWB As Workbook
    Dim SH As Worksheet
    Dim srcRng As Range, destRng As Range
    Dim rCell As Range
    Dim LRow As Long, LCol As Long
    Dim arrIn As Variant, arrOut As Variant
    Dim arrItems As Variant, arrHeaders As Variant
    Dim i As Long, j As Long, k As Long
    Dim oDic As Scripting.Dictionary
    Dim sPath As String

    Dim CalcMode As Long
    Const iStep As Long = 499
    Const aStr As String = "Emaildata"

On Error GoTo errHandler              [EDIT] This line has been subsequently un-commented

                                                                 See the following post.
    Set oDic = New Scripting.Dictionary


    Set WB = ThisWorkbook
    Set SH = WB.Sheets("Sheet1")                                    
    sPath = WB.Path & Application.PathSeparator

    With SH
        LRow = LastRow(SH)
        LCol = LastCol(SH)
        Set srcRng = .Range("A2").Resize(LRow - 1, LCol)
    End With

    arrHeaders = srcRng.Rows(0).Value

    For i = 1 To LRow Step iStep
        arrIn = srcRng.Cells(i, 1).Resize(iStep, LCol).Value
        oDic.Add Key:=i, Item:=arrIn
        Erase arrIn
    Next i

    With Application
        CalcMode = .Calculation
        .Calculation = xlCalculationManual
        .ScreenUpdating = False
    End With

    j = oDic.Count
    arrItems = oDic.Items
    For k = 1 To j
        arrOut = arrItems(k - 1)
        Set newWB = Application.Workbooks.Add(1)
        Set destRng = newWB.Sheets(1).Range("A2").Resize(iStep, LCol)
        destRng.Rows(0).Value = arrHeaders
        destRng.Value = arrOut
        With newWB
            Application.DisplayAlerts = False
            .SaveAs Filename:=sPath & aStr & k, _
                    FileFormat:=51
            Application.DisplayAlerts = True

Function LastRow(SH As Worksheet, _
                 Optional Rng As Range)
    If Rng Is Nothing Then
        Set Rng = SH.Cells
    End If

    On Error Resume Next
    LastRow = Rng.Find(What:="*", _
                       After:=Rng.Cells(1), _
                       Lookat:=xlPart, _
                       LookIn:=xlFormulas, _
                       SearchOrder:=xlByRows, _
                       SearchDirection:=xlPrevious, _
                       MatchCase:=False).Row
    On Error GoTo 0
End Function

adam vero say:

Don't change AF19.

Go to Insert > Name > Define (in 2003 or older), or Formulas ribbon > Defined Names group > Define Name (2007+)

Put in a name such as SailRange1. Now, instead of pointing at a cell range, use the formula in here as the definition of the name (names can refer to constants, cells, or formulae). So in the "refers to" box, put =INDIRECT($AF$19)and click OK.

(tip:use F2 to make the box directly editable if you need to, otherwise if you use arrow keys you get a mess as it shoves cell refs in too)

Now use SailRange1 as the source for your validation.

By keeping the formula in the name it seems to avoid the compatibility check successfully.
