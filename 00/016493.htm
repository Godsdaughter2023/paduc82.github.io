bernie deitrick say: With the data sheet active:

Sub ExportDpartmentSheetsFromDatabase()
    'Based on the value in the "Department" column
    Dim myCell As Range
    Dim mySht As Worksheet
    Dim myName As String
    Dim myArea As Range

    Set myArea = Cells.Find("Department").CurrentRegion.Columns(1).Offset(1, 0).Cells

    Set myArea = myArea.Resize(myArea.Rows.Count - 1, 1)

    For Each myCell In myArea
        If myCell.Value <> "All" And myCell.Value <> "" Then
            On Error GoTo NoSheet
            myName = Worksheets(myCell.Value).Name
            GoTo SheetExists:
NoSheet:
            Set mySht = Worksheets.Add
            mySht.Name = myCell.Value
            With myCell.CurrentRegion
                .AutoFilter Field:=1, Criteria1:=myCell.Value, Operator:=xlOr, _
                            Criteria2:="=All"
                .SpecialCells(xlCellTypeVisible).Copy _
                        mySht.Range("A1")
                mySht.Cells.EntireColumn.AutoFit
                .AutoFilter
            End With
            Resume
        End If
SheetExists:
    Next myCell
End Sub

Sub ExportDpartmentSheetsFromDatabase()
    'Based on the value in the "Department" column
    Dim myCell As Range
    Dim mySht As Worksheet
    Dim myName As String
    Dim myArea As Range

    Set myArea = Cells.Find("Department").CurrentRegion.Columns(1).Offset(1, 0).Cells

    Set myArea = myArea.Resize(myArea.Rows.Count - 1, 1)

    For Each myCell In myArea
        If myCell.Value <> "All" And myCell.Value <> "" Then
            On Error GoTo NoSheet
            myName = Worksheets(myCell.Value).Name
            GoTo SheetExists:
NoSheet:
            Set mySht = Worksheets.Add
            mySht.Name = myCell.Value
            With myCell.CurrentRegion
                .AutoFilter Field:=1, Criteria1:=myCell.Value, Operator:=xlOr, _
                            Criteria2:="=All"
                .SpecialCells(xlCellTypeVisible).Copy _
                        mySht.Range("A1")
                mySht.Cells.EntireColumn.AutoFit
                .AutoFilter
            End With
            Resume
        End If
SheetExists:
    Next myCell

End Sub

If you add the attachment to an empty message and save it as a Template e.g. Attachment.oft then the following Outlook macro will create a new empty message with the attachment already attached.  Change the path where indicated to your local path.

Sub NewMailUsingTemplate()
Dim olApp As Object
Dim oMail As Object
Dim i As Long
Dim strMessagePath As String
Dim strTemplate As String
    On Error Resume Next
    strTemplate = "D:\Word 2010 Templates\Attachment.oft"
    On Error Resume Next
    Set olApp = GetObject(, "Outlook.Application")
    If Err <> 0 Then
        Set olApp = CreateObject("Outlook.Application")
    End If
    On Error GoTo 0
    Set oMail = olApp.CreateItemFromTemplate(strTemplate)
    oMail.Display
    Set oMail = Nothing
    Set olApp = Nothing
End Sub
