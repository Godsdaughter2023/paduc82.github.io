Sub TestMacro()
    Dim wb1 As Workbook
    Dim wb2 As Workbook
    Dim ws1 As Worksheet
    Dim ws2 As Worksheet

    Application.ScreenUpdating = False
    Set wb1 = Workbooks("Source.xls")
    Set wb2 = Workbooks("Target.xls")
    For Each ws1 In wb1.Worksheets
        On Error GoTo noSheet
        Set ws2 = wb2.Worksheets(ws1.Name)
        ws1.UsedRange.Copy ws2.Cells(Rows.Count, 1).End(xlUp)(2)

TryNext:
    Next ws1
    Application.ScreenUpdating = True
    Exit Sub
   
noSheet:
    Resume TryNext
End Sub
