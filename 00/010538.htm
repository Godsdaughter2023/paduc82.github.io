andreas killer say:
There is always a risk when we use GETOBJECT / CREATEOBJECT to get the embedded document, because it is possible that we access an other already running instance.

A better way is to use the OLEOBJECT to get the reference to the object.

Andreas.

Sub Test()
  Dim Oo As OLEObject
  Dim wDoc As Object 'Word.Document
 
  'Search for the embedded Word document
  For Each Oo In ActiveSheet.OLEObjects
    If InStr(1, Oo.progID, "Word.Document", vbTextCompare) > 0 Then
      'Open the embedded document
      Oo.Verb xlVerbPrimary
      'Get the document inside
      Set wDoc = Oo.Object
     
      'Copy the contents to cell A1
      wDoc.Content.Copy
      Range("A1").PasteSpecial xlPasteValues
     
      'Select any cell to close the document
      Range("A1").Select
      'Done
      Exit For
    End If
  Next
End Sub

graham mayor say:
Both approaches still present a problem if there is more than one Word document embedded. If that is the case, one approach would be to set the Title document properties of the embedded documents and then check for that title as you loop through the objects e.g. as follows. If you want to close the Word application after it is opened by the macro, you would also need to check whether the macro opened it.

Sub Test()
Dim Oo As OLEObject
Dim wDoc As Object        'Word.Document
Dim wdApp As Object       'Word application
Dim bCreated As Boolean

    On Error Resume Next
    Set wdApp = GetObject(, "Word.Application")
    If Err <> 0 Then
        Set wdApp = CreateObject("Word.Application")
        bCreated = True
    End If
    On Error GoTo 0

    ActiveSheet.OLEObjects(1).Verb Verb:=xlPrimary
    'Search for the embedded Word document
    For Each Oo In ActiveSheet.OLEObjects
        If InStr(1, Oo.progID, "Word.Document", vbTextCompare) > 0 Then
            'Open the embedded document
            Oo.Verb xlVerbPrimary
            'Get the document inside
            Set wDoc = Oo.Object
            'Check the title property to ensure it is the correct document
            If UCase(wDoc.BuiltinDocumentProperties("Title")) = "DOCUMENT TITLE" Then
                'Copy the contents to cell A1
                wDoc.Content.Copy
                Range("A1").PasteSpecial xlPasteValues
                wDoc.Close 0
                Range("A1").Select
                Exit For
                'Done
            End If
            wDoc.Close 0
            Range("A1").Select
        End If
    Next
    'Close Word if the macro opened it
    If bCreated Then wdApp.Quit
    Set wdApp = Nothing
    Set wDoc = Nothing
    Set Oo = Nothing
End Sub
