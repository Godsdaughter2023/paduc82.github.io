You mention delete cells in capital letters and delete highlight in red rows which is it?

Sub delete_Me()
Dim LastRow As Long, x As Long
LastRow = Cells(Cells.Rows.Count, "A").End(xlUp).Row
For x = LastRow To 1 Step -1

    End If
Next
End Sub
