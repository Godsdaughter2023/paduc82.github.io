Sub MakeSlipTimeSheets()
' for each sheet with a name like "Gate <n> Codes", create a new sheet "Gate <n> Slip Times"
' containing initially the same as Gate <n> Codes; then remove rows which do not contain
' Relay (1), Relay (2) or Relay (3) in column F
  Dim WS As Worksheet
  Dim lRow As Long
  For Each WS In ThisWorkbook.Worksheets
    If WS.Name Like "Gate * Codes" Then
      WS.Copy After:=WS
      ActiveSheet.Name = Replace(WS.Name, "Codes","Slip Times")
      lRow = Range(Rows.Count, "F").End(xlUp).Row  ' last used cell in F
      Do Until lRow = 0
        Select Case Cells(lRow, "F").Value
          Case "Relay (1)", "Relay (2)", "Relay (3)"
            ' keep
          Case Else
            ' lose

            Rows(lRow).Delete
        End Select
        lRow = lRow - 1
      Loop
    End If
  Next
End Sub
