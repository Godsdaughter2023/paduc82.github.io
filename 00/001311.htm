Sub ReverseIt()
  Dim wshSrc As Worksheet
  Dim wshTrg As Worksheet
  Dim r As Long
  Dim m As Long
  Dim t As Long
  Dim strConcat As String
  Set wshSrc = ActiveSheet
  Set wshTrg = Worksheets.Add
  wshTrg.Range("A1") = wshSrc.Range("A1")
  wshTrg.Range("B1") = wshSrc.Range("B1")
  wshTrg.Range("C1") = wshSrc.Range("C1")
  wshSrc.Range("A1").CurrentRegion.Sort Key1:=wshSrc.Range("A1"), _
    Key2:=wshSrc.Range("B1"), Header:=xlYes
  m = wshSrc.Range("A" & wshSrc.Rows.Count).End(xlUp).Row
  t = 1
  For r = 2 To m
    If wshSrc.Range("A" & r) <> wshSrc.Range("A" & (r - 1)) Then
      If t > 1 Then
        wshTrg.Range("C" & t) = strConcat
      End If
      t = t + 1
      wshTrg.Range("A" & t) = wshSrc.Range("A" & r)
      wshTrg.Range("B" & t) = wshSrc.Range("B" & r)
      strConcat = wshSrc.Range("C" & r)
    Else
      strConcat = strConcat & "/" & wshSrc.Range("C" & r)
    End If
  Next r
  wshTrg.Range("C" & t) = strConcat
End Sub
