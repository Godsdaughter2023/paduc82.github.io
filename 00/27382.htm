bernie deitrick say: Copy the code below, right-click the sheet tab, select "View Code" and paste the code into the window that appears.

Private Sub Worksheet_SelectionChange(ByVal Target As Range)
If Target.Column <> 2 Then Exit Sub
If Target.Cells.Count > 1 Then Exit Sub
If Target.Value = "" Then Exit Sub
Cells(Target.Row, "C").Value = Application.UserName
Cells(Target.Row, "D").Value = Environ$("computername")
Cells(Target.Row, "E").Value = Date
Cells(Target.Row, "F").Value = Time
Cells(Target.Row, "C").Resize(1, 4).Locked = True
End Sub

Hi,

you have two sheets

source sheet and target sheet (new sheet)

 

try this code...

 

 

Sub CopyPaste_Mirror()
'18 Nov 2013
Const rng As String = "A1:BD10"
Dim ws1 As Worksheet, ws2 As Worksheet
Set ws1 = Sheets("Sheet1")   '<<< source sheet name, change
Application.ScreenUpdating = False
Set ws2 = Sheets.Add '<<< target sht
ws1.Range(rng).Copy Range("A1")
For Each rr In ws1.Range(rng)   '<<< format target range
Range(rr.Address).RowHeight = rr.RowHeight
Range(rr.Address).ColumnWidth = rr.ColumnWidth
Next
Application.ScreenUpdating = True
End Sub
