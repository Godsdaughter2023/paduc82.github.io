Function zfraction2(ByVal pdou As Double, _
                    Optional multiplx _
                    As Variant) As String
    '*******************************************
    'Purpose:  Converts input into a fraction
    '          accurate to the nearest multiplx
    '
    'Coded by: raskew
    '
    'Inputs:   From debug window: ?
    '          (1) ? zfraction2(16.974,256)
    '          (2) ? zfraction2(16.974)   'defaults to 64ths
    '          (3) ? zfraction2(16)
    '
    'Output:   (1) 16-249/256
    '          (2) 16-31/32
    '          (3) 16
    '*******************************************

    Dim I               As Integer
    Dim j               As Integer
    Dim intHold         As Integer
    Dim Num             As Integer
    Dim Denom           As Integer
    Dim strHold         As String
    Dim IsWhole         As Boolean
    Dim IsZilch         As Boolean


    'Set default to 64 if multiplx is missing
    multiplx = IIf(IsMissing(multiplx) = True, 64, multiplx)

    'capture the integer portion of zdec
    intHold = Int(pdou)
    pdou = pdou - intHold

    'create a fraction, expressed in multiplxths (e.g. 64ths)
    'num = numerator
    Num = CInt(multiplx * pdou)

    'denom = denominator
    Denom = multiplx

    'reduce the fraction
    j = Num

    For I = j To 2 Step -1
        If Num Mod I = 0 And Denom Mod I = 0 Then
            Num = Num / I
            Denom = Denom / I
            j = Num
            Exit For
        End If
    Next I

    'express the results as a string
    'are the results 1/1?
    IsWhole = Num / Denom = 1

    'Is Num = 0 (will occur if only a whole number is input)
    IsZilch = Num = 0
    intHold = intHold + Abs(IsWhole)
    strHold = IIf(intHold > 0, LTrim(Str(intHold)) & IIf(Not IsWhole And Not IsZilch, Chr(45), ""), "")
    zfraction2 = strHold & IIf(Not IsWhole And Not IsZilch, LTrim(Str(Num)) & "/" & LTrim(Str(Denom)), "")
End Function

 

Function ConvertFraction(strGetNumber As String) As Double
    ' Will convert a fraction, such as 12 3/4 to
    ' it's decimal equivalent, 12.75
    Dim dblFraction     As Double
    Dim intPosition     As Integer
    Dim strTop          As String
    Dim strBottom       As String
    Dim dblWhole        As Double
    Dim strFraction     As String
    On Error GoTo Err_Convert
    intPosition = InStr(strGetNumber, "/")
    If intPosition = 0 Then
        ConvertFraction = strGetNumber  ' Not a whole number
        Exit Function
    End If

    intPosition = InStr(strGetNumber, " ")
    If intPosition > 0 Then
        dblWhole = Val(Left(strGetNumber, intPosition - 1))
    Else
        dblWhole = 0
    End If
    strFraction = Mid(strGetNumber, intPosition + 1)
    intPosition = InStr(strFraction, "/")
    strTop = Left(strFraction, intPosition - 1)
    strBottom = Mid(strFraction, intPosition + 1)
    dblFraction = Val(strTop) / Val(strBottom)
    ConvertFraction = dblWhole + dblFraction

Exit_Function:
    Exit Function

Err_Convert:
    MsgBox "Error #: " & Err.Number & "   " & Err.Description, _
    vbInformation
    Resume Exit_Function
End Function

Daniel's code will get you part of the way there, but note that it's very specific: it converts a text string containing a number and an (optional) fraction into a double float number. It does NOT account for the ' and " symbols. That is, it would convert the text string

 

2 1/2

 

to 2.5, but it will not be able to handle 12' 2-1/2".

 

Your problem is doubly complex since both ' and " are string delimiters in VBA! It's a bit tricky to parse them out of strings - a string literal such as

 

"12' 2-1/2""

 

won't work, because the " inches symbol will be misread as the closing quotemark.

 

You're going to need some nontrivial code to parse through the string, and it's going to need to account for ', ", - and / as meaningful parts of the string. It's certainly doable but more code than I want to tackle this late in the evening... I'll note this thread and try to come back to it tomorrow.

Public Function ImpToDec(Imperial As String, Optional ReturnUnits As String = "Inches")
    Dim RetMultiplier As Integer
   
    Select Case ReturnUnits
        Case "Feet"
            RetMultiplier = 12
        Case "Inches"
            RetMultiplier = 1
        Case Else
            RetMultiplier = 1
    End Select
   
    Dim Constructor As String
    Constructor = ""
   
    Constructor = IIf(InStr(1, Imperial, "'"), "(", "") & Replace(Imperial, "'", ")*12")
    Constructor = Replace(Constructor, """", "")
    Constructor = Replace(Constructor, " ", "+")
    Constructor = Replace(Constructor, "-", "+")
   
    ImpToDec = Evaluate(Constructor) / RetMultiplier
   
End Function
