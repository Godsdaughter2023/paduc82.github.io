jllatham say:

Try this for starters:

Create a new workbook.  Delete all sheets from it except Sheet1.  In cell A2 of Sheet1 enter the year for your calendar.  You can dress up this sheet as much as you want, and give it any name on its tab that you want, you just have to enter the year into cell A2 for the code below to work.

 

There are 2 pieces of code: one goes into the ThisWorkbook code module, the other goes into a regular code module.  Instructions for putting code into those locations can be found on this page:

Into ThisWorkbook:  http://www.contextures.com/xlvba01.html#Workbook

Into a Regular Code module: http://www.contextures.com/xlvba01.html#videoreg

 

If you want a workbook that already has the code in it then download this workbook (macros must be enabled for it to work)  If nothing else, it's a starting point.

WeeklyCalendarMaker.xlsm 78KB (contains macros)

http://www.mediafire.com/view/yexka4k5zppy8f6/WeeklyCalendarMaker.xlsm

 

Here is the code for the ThisWorkbook code module:


Private Sub Workbook_Open()
  Dim calWeekNumber As Integer
  Dim calYear As Integer
 
  On Error Resume Next
  If Not IsEmpty(Sheet1.Range("A2")) Then
    calYear = Sheet1.Range("A2")
  Else
    calYear = Year(Now())
  End If
  calWeekNumber = Application.WorksheetFunction.WeekNum(Now())
  ThisWorkbook.Worksheets(calYear & sheetNamePart & calWeekNumber).Activate
  If Err <> 0 Then
    Err.Clear
  End If
  On Error GoTo 0
End Sub

 

and here is the code for the regular code module:

 

Public Const sheetNamePart = " Week-"

Sub CreateWeeklyCalendars()
  Dim calYear As Integer
  Dim calDate As Date
  Dim calWeekNumber As Integer
  Dim dayLoop As Integer
  Dim anyWS As Worksheet
  'get rid of old calendar sheets if they exist
  If ThisWorkbook.Worksheets.Count > 1 Then
    Application.ScreenUpdating = False
    Application.DisplayAlerts = False
    For Each anyWS In ThisWorkbook.Worksheets
      If anyWS.Name <> Sheet1.Name Then
        anyWS.Delete
      End If
    Next
    Application.DisplayAlerts = False
  End If

  If Not IsEmpty(Sheet1.Range("A2")) Then
    calYear = Sheet1.Range("A2")
  Else
    calYear = Year(Now())
  End If
  ' =DATE(A2,1,-3)-WEEKDAY(DATE(A2,1,3))+1*7
  calDate = DateSerial(calYear, 1, -3) - Weekday(DateSerial(calYear, 1, 3)) + 1 * 7
  calWeekNumber = 0
  Do Until calDate > DateSerial(calYear + 1, 1, 1)
    'add a sheet for the week
    calWeekNumber = calWeekNumber + 1
    ThisWorkbook.Worksheets.Add after:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count)
    ActiveSheet.Name = calYear & sheetNamePart & calWeekNumber
    For dayLoop = 1 To 7
      Cells(1, dayLoop).HorizontalAlignment = xlCenter
      Select Case dayLoop
        Case Is = 1
          Cells(1, dayLoop) = "Sunday"
        Case Is = 2
          Cells(1, dayLoop) = "Monday"
        Case Is = 3
          Cells(1, dayLoop) = "Tuesday"
        Case Is = 4
          Cells(1, dayLoop) = "Wednesday"
        Case Is = 5
          Cells(1, dayLoop) = "Thursday"
        Case Is = 6
          Cells(1, dayLoop) = "Friday"
        Case Is = 7
          Cells(1, dayLoop) = "Saturday"
      End Select
      With Cells(2, dayLoop)
        .NumberFormat = "dd-mmm-yy"
        .Value = calDate
        .EntireColumn.AutoFit
        .HorizontalAlignment = xlCenter
      End With
      calDate = calDate + 1
    Next
  Loop
 
End Sub
