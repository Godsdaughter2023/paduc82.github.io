On Error Resume Next
MkDir saveFolder
On Error GoTo 0
     For Each objAtt In itm.attachments

      debug.print "Found: " & itm.filename 'so regardless, you'll have a list of the attachments

      If LCase(Right(objAtt.FileName, 4)) <> ".jpg" And LCase(Right(objAtt.FileName, 4)) <> ".png"

          debug.print "Confirmed as acceptable file type"

          objAtt.SaveAsFile saveFolder & dateFormat & "STT_NL" & LCase(Right(objAtt.FileName, 4))

          debug.print "Successful save"

          Set objAtt = Nothing
          End If
     Next
End Sub
