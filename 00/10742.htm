bob jones aka: cybertaz say:
The most significant difference is that Office 365 is not Excel/Word/PPT. It is a collection of 'cloud' computing services primarily for document access & sharing. It does provide web apps which are basically [quite] limited tools for working with Office documents as a convenience. However, my understanding is that although the web apps are available for Apple iOS5 (which is what is used by iPhone, iPod & iPad), Macs don't use iOS5. They use OS X as their operating system, so the web apps can't be used by a Mac.

IOW, you create the file on your Mac or iPhone, store it on the cloud, then you can use your iPhone to make certain types of changes to it while you're out of the office. When you get back you can open the file on your Mac & pick up where you left off using the full features of your Office 2011 software.

Bottom line - No, there is no "upgrade" because Office 2011 & Office 365 are 2 separate & distinct products which are not interchangeable. Additionally, Office 365 is a subscription service.

Would you perhaps have any sugestions how to make a list of all the subs and with where all the subs they are called on?

Option Explicit

Sub Test()
  Dim Wb As Workbook
  Dim vbComp As VBIDE.VBComponent
  Dim i As Long, j As Long
  Dim mName As String, pName As String, pKind As VBIDE.vbext_ProcKind, pLine As Long
 
  Dim pKindNames(0 To 3) As String
  pKindNames(vbext_pk_Proc) = "Proc"
  pKindNames(vbext_pk_Get) = "Property Get"
  pKindNames(vbext_pk_Let) = "Property Let"
  pKindNames(vbext_pk_Set) = "Property Set"
 
  Dim Dict As Object 'Scripting.Dictionary
  Dim Key As String
  Dim pItem, Items
  Dim pNames As New Collection
  Dim pData As New Collection
 
  Dim pBody As String
  Dim f As Long, t As Long
  Dim Data
 
  Set Wb = Workbooks("AK.xlam")
 
  'Get all proc names
  Set Dict = CreateObject("Scripting.Dictionary")
  For Each vbComp In Wb.VBProject.VBComponents
    With vbComp.CodeModule
      mName = vbComp.Name
      For i = .CountOfDeclarationLines + 1 To .CountOfLines
        pName = .ProcOfLine(i, pKind)
        pItem = Array(mName, pName, pKind)
        Key = Join(pItem, "|")
        pItem = Array(mName, pName, pKind, i)
        If Not Dict.Exists(Key) Then
          Dict.Add Key, pItem
          pNames.Add pName
        End If
      Next
    End With
  Next
 
  'Scan for proc calls
  Items = Dict.Items
  For i = 0 To UBound(Items)
    mName = Items(i)(0)
    pName = Items(i)(1)
    pKind = Items(i)(2)
    pLine = Items(i)(3)
    Set vbComp = Wb.VBProject.VBComponents(mName)
    With vbComp.CodeModule
      f = .ProcStartLine(pName, pKind)
      t = .ProcCountLines(pName, pKind)
      pBody = .Lines(f, t)
      pData.Add Array(mName, pName, pKindNames(pKind), pLine, .Lines(.ProcBodyLine(pName, pKind), 1))
      For Each pItem In pNames
        If InStr(pBody, pItem) > 0 Then
          If pName <> pItem Then
            pData.Add Array("->", pItem)
          End If
        End If
      Next
    End With
  Next
 
  'Output
  ReDim Data(1 To pData.Count, 1 To 5)
  i = 0
  For Each pItem In pData
    i = i + 1
    For j = 0 To UBound(pItem)
      Data(i, j + 1) = pItem(j)
    Next
  Next
  Sheets.Add
  Range("A1:D1") = Array("Module", "Proc", "Kind", "Line")
  Range("A2").Resize(UBound(Data), UBound(Data, 2)).Value = Data
End Sub
