$usermailbox = get-mailbox –recipienttypedetails usermailbox

Sub DivideUpandColor()

Dim DivideInfo As Range
Dim LastRow, Breakout, Loop1, StartRow, NextRow, Nameloop As Long
Dim MsgTxt1, TitleTxt1, DfltTxt1, UsrResponse1, Name As String

Dim i as Long, delt as Long, v as variant, v1 as variant, cnt as long
    
    MsgTxt1 = "How many people will this data by divided by?"
    TitleTxt1 = "Insert #"
    DfltTxt1 = "Enter info here"
   
UsrResponse1 = InputBox(MsgTxt1, TitleTxt1, DfltTxt1)
Cells.Interior.ColorIndex = xlNone
v1 = Array(6, 45, 8, 19, 22, 24, 35, 38, 39, 4)  ' colors for sections
LastRow = Range("E" & Rows.Count).End(xlUp).Row

Oddcheck = Int(LastRow / 2) * 2
If Oddcheck < LastRow Then LastRow = LastRow + 1

Set DivideInfo = Range("A2:AZ" & LastRow)

Breakout = (LastRow - 1) \ UsrResponse1
ReDim v(1 To UsrResponse1 + 1)
cnt = 1
remainder = (LastRow - 1) Mod UsrResponse1
Loop1 = 2
Do While Loop1 < LastRow - 3
   v(cnt) = Loop1
   Loop1 = Loop1 + Breakout
   If cnt <= remainder Then
     Loop1 = Loop1 + 1
   End If
   cnt = cnt + 1
Loop
delt = 1 - LBound(v1)
v(cnt) = LastRow + 1
For i = LBound(v) To UBound(v) - 1
  Range(Cells(v(i), "A"), Cells(v(i + 1) - 1, "A")).Resize(, 52).Interior.ColorIndex = v1(i - delt)
Next

End Sub

Additionally, you can set Hotmail to always show attachments from the specified email address, if you know the sender is a safe source.

These are the steps on how to unblock your email attachments:

 

1. Sign in to your Microsoft account.

2. Click on the gear icon on the upper right part of the screen and then More mail settings.

3. Under Preventing junk email, click on Filters and reporting.

4. Under Block content from unknown senders, click on the radio button beside Show attachments, pictures, and links for senders with good reputations option and then click on Save.
