steve rindsberg say:

This little macro will do it for you.  The link below explains how to use VBA code.

How do I use VBA code in PowerPoint
http://www.pptfaq.com/FAQ00033_How_do_I_use_VBA_code_in_PowerPoint.htm

 

Sub BazillionsOfBlockArrows()

    Dim oSh As Shape
    Dim x As Long
    Dim lStartingNumber As Long
   
    lStartingNumber = 1234   ' change this as needed
   
    For x = 0 To 19     ' This'll make 20
        ' add an arrow at 1" from the left, 1" from the top
        ' make it 2" wide, 1" hight
        ' change values as needed (they're in points, 72 points to the inch)
        Set oSh = ActiveWindow.Selection.SlideRange.Shapes.AddShape(msoShapeRightArrow, 72, 72, 144, 72)
        oSh.TextFrame.TextRange.Text = CStr(lStartingNumber + x)
        ' move each new arrow down by 1/4 inch
        oSh.Top = oSh.Top + (x * 18)
    Next
   
End Sub
