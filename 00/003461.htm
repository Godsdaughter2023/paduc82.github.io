Try:

Sub Demo()
Application.ScreenUpdating = False
Dim Tbl As Table
For Each Tbl In ActiveDocument.Tables
  With Tbl
    .PreferredWidthType = wdPreferredWidthPoints
    .PreferredWidth = CentimetersToPoints(15.494)

  End With
Next
Application.ScreenUpdating = True

Try below - cell A1 can have both date & time or only time:

=IF((MOD(A1,1)>=8/24)*(MOD(A1,1)<17/24),INFORMATION!C1,IF(OR(MOD(A1,1)>=17/24,MOD(A1,1)<1/24),INFORMATION!C2,INFORMATION!C3))
