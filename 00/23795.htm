Sub SortDates()

Const rng As String = "A1:G" '<< change

Const myCol As String = "A" '<< change

Dim r As Long, i As Integer

r = ActiveSheet.UsedRange.Rows.Count

Application.ScreenUpdating = False

For i = 1 To r

If Cells(i, myCol) = "" Then Cells(i, myCol) = 0

Next i

Range(rng & r).Sort Key1:=Range(myCol & "2"), Header:=xlYes

For Each rr In Range(myCol & "1:" & myCol & r)

If rr.Value = 0 Then rr.Value = ""

Next

Application.ScreenUpdating = True

End Sub
