Function ApplicationRun(ByVal Wb As Workbook, ByVal Ws As Variant, _
    ByVal MacroName As String, ParamArray Args()) As Variant
  'Runs a specific (private) macro
  Dim MacroCall As String
  'Full qualified syntax is "Workbook!Module.Macro" or "Workbook!Macro"
  If Wb Is Nothing Then Set Wb = ActiveWorkbook
  MacroCall = Wb.Name & "!"
  If IsObject(Ws) Then
    If Not Ws Is Nothing Then MacroCall = MacroCall & Ws.CodeName & "."
  Else
    MacroCall = MacroCall & Ws & "."
  End If
  MacroCall = MacroCall & MacroName
  Select Case UBound(Args)
    Case -1: ApplicationRun = Application.Run(MacroCall)
    Case 0: ApplicationRun = Application.Run(MacroCall, Args(0))
    Case 1: ApplicationRun = Application.Run(MacroCall, Args(0), Args(1))
    Case 2: ApplicationRun = Application.Run(MacroCall, Args(0), Args(1), Args(2))
    Case 3: ApplicationRun = Application.Run(MacroCall, Args(0), Args(1), Args(2), _
        Args(3))
    Case 4: ApplicationRun = Application.Run(MacroCall, Args(0), Args(1), Args(2), _
        Args(3), Args(4))
    Case 5: ApplicationRun = Application.Run(MacroCall, Args(0), Args(1), Args(2), _
        Args(3), Args(4), Args(5))
    Case 6: ApplicationRun = Application.Run(MacroCall, Args(0), Args(1), Args(2), _
        Args(3), Args(4), Args(5), Args(6))
    Case 7: ApplicationRun = Application.Run(MacroCall, Args(0), Args(1), Args(2), _
        Args(3), Args(4), Args(5), Args(6), Args(7))
    Case 8: ApplicationRun = Application.Run(MacroCall, Args(0), Args(1), Args(2), _
        Args(3), Args(4), Args(5), Args(6), Args(7), Args(8))
    Case 9: ApplicationRun = Application.Run(MacroCall, Args(0), Args(1), Args(2), _
        Args(3), Args(4), Args(5), Args(6), Args(7), Args(8), Args(9))
  End Select
End Function

john mcghie say:

iPhoto will re-size a batch of photos for you.  It has a rather strange user interface, but once you figure it out, it will do a whole batch for you.

Alternatively, the following macros will resize a pic for you: You add two toolbar buttons, one calls the InsertGraphic macro, the other calls the InsertLargeGraphic macro.  The two constants right up the top set the actual size the picture will end up at, in points. You can add a third button, for the SizePic macro: that will work to set the size of a picture that is already in your document.

Either macro calls the built-in Insert Picture dialog to prompt you for the picture file.

These macros will work with ANY raster graphic (.bmp, jpeg, gif, etc).

    

Public Const PicBorderColour As Long = 7420928
Public Const PicSize As Long = 252
Public Const LargePicSize As Long = 496

Dim myPicture As InlineShape

Sub InsertGraphic()
'    CheckGraphicStyle
    GraphicInsert
    SizePic myPicture, PicSize
    ProcessGraphic myPicture, PicSize
End Sub
Sub InsertLargeGraphic()
'    CheckGraphicStyle
    GraphicInsert
    SizePic myPicture, LargePicSize
    ProcessGraphic myPicture, LargePicSize
End Sub

Function GraphicInsert()
Dim sty As Style

For Each sty In ActiveDocument.Styles
    If sty.NameLocal = "Graphic" Then Exit For
Next ' sty
If sty Is Nothing Then
    Application.OrganizerCopy Source:=Application.NormalTemplate, _
            Destination:=ActiveDocument.FullName, _
            Name:="Graphic", _
            Object:=wdOrganizerObjectStyles
End If

    Selection.Paragraphs(1).Style = "Graphic"
    Selection.Collapse

    With Dialogs(wdDialogInsertPicture)
        If .Display = -1 Then
        Set myPicture = ActiveDocument.InlineShapes.AddPicture(.Name, False, True, Selection.Range)
        Else
            End
        End If
    End With
End Function

Sub SizePic(Optional myPicture As InlineShape, Optional mySize As Long)
'
' Size Picture Macro
' Macro written 14/09/2004 by John McGhie
'

If myPicture Is Nothing Then ' Called directly without passing a picture
    mySize = PicSize ' Load picture size
    With Selection
        If .Type = wdSelectionShape Then .ShapeRange.ConvertToInlineShape ' it's floating, make inline
        Set myPicture = .InlineShapes(1) ' Set mPicture
    End With
End If

With myPicture
    .LockAspectRatio = False
    ScaleFactor = (.Height / .Width)
    .Width = mySize
    .Height = .Width * ScaleFactor
    .LockAspectRatio = True
End With

End Sub

Function ProcessGraphic(myPicture As InlineShape, mySize As Long)
Dim ScaleFactor As Double
'Dim aBorder As Long

With myPicture
        With .Borders(wdBorderLeft)
            .LineStyle = wdLineStyleSingle
            .LineWidth = wdLineWidth100pt
            .Color = PicBorderColour
        End With
        With .Borders(wdBorderRight)
            .LineStyle = wdLineStyleSingle
            .LineWidth = wdLineWidth100pt
            .Color = PicBorderColour
        End With
        With .Borders(wdBorderTop)
            .LineStyle = wdLineStyleSingle
            .LineWidth = wdLineWidth100pt
            .Color = PicBorderColour
        End With
        With .Borders(wdBorderBottom)
            .LineStyle = wdLineStyleSingle
            .LineWidth = wdLineWidth100pt
            .Color = PicBorderColour
        End With
        .Borders.Shadow = False
End With

    With Selection
        .MoveRight Unit:=wdCharacter, Count:=2
        If .Style <> "Caption" Then
            .MoveLeft Unit:=wdCharacter, Count:=1
            .TypeParagraph
            NormalTemplate.AutoTextEntries("FigCap").Insert Where:=.Range, RichText:=True
            .Paragraphs.Style = ActiveDocument.Styles("Caption")
            .Collapse direction:=wdCollapseEnd
        End If
    End With
End Function
