rick rothstein, mvp - excel say:

This macro will remove only trailing spaces from the cells in all your columns leaving any multiple internal spacing that may exist as is...

Sub RemoveTrailingSpaces()
  Dim Col As Range, combo As String
  Application.ScreenUpdating = False
  For Each Col In ActiveSheet.UsedRange.Columns
    combo = Join(WorksheetFunction.Transpose(Col), Chr(1))
    Do While InStr(combo, " " & Chr(1))
      combo = Replace(combo, " " & Chr(1), Chr(1))
    Loop
    Col = WorksheetFunction.Transpose(Split(combo, Chr(1)))
  Next
  Application.ScreenUpdating = True
End Sub

Sub FromExcelToWord()

'24 Feb 2013
If MsgBox("select a range ?", vbOKCancel) = vbCancel Then Exit Sub
Dim wdApp As Object
Set wdApp = CreateObject("Word.Application")
Dim wdDoc
Set wdDoc = wdApp.Documents.Add
wdApp.Visible = True
Application.ScreenUpdating = False
Selection.Copy
wdDoc.Content.PasteSpecial Link:=False, DataType:=20, Placement:=wdInLine, _
DisplayAsIcon:=False
Application.CutCopyMode = False
Dim docName
docName = "myWord File.docx"   '<<< Word Document Name, change

Application.EnableEvents = False

Application.EnableEvents = True
MsgBox "Word Doc: " & vbCr & "SaveAs This Workbook Path " & vbCr & _
 "name = " & docName & " and Close.."

wdApp.Quit
End Sub
