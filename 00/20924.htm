fV 03.21
21-11-05
RM-127
(c) Nokia.

11) done :)

  getFileSize = -1
  If fso.FileExists(ServerName & strPath & strFilename) Then
    getFileSize = fso.GetFile(ServerName & strPath & strFilename).Size
  Else
    getFileSize = fso.GetFile(ServerName & "c$" & strPath & strFilename).Size
  End If
ExitPoint:
End Function

Private Function TrimSlashes(ByVal S As String) As String
  Do While Left(S, 1) = "\"
    S = Mid(S, 2)
  Loop
  Do While Right(S, 1) = "\"
    S = Left(S, Len(S) - 1)
  Loop
  TrimSlashes = S
End Function
