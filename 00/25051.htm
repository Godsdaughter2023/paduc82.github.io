bernie deitrick say:

This will delete all non-selected sheets if 2 or more sheets are selected  OR  if only one sheet is selected, it will delete all sheets (except Sheet1) whose name does not start with the values in B2 or B3: if B2 and B3 are blank, and only one sheet is selected, all sheets (except Sheet1) will be deleted. 

Sub Copy2()
    Dim FName As String
    Dim FPath As String
    Dim Sh As Worksheet
    Dim shs() As Boolean
    Dim i As Integer
    Dim str1 As String
    Dim str2 As String
    
    FPath = Sheets("Sheet1").Range("A1").Text
    FName = Sheets("Sheet1").Range("A2").Text

    Application.DisplayAlerts = False
    
    If ActiveWindow.SelectedSheets.Count > 1 Then
        ReDim shs(1 To Worksheets.Count)
        
        For Each Sh In ActiveWindow.SelectedSheets
            shs(Sh.Index) = True
        Next Sh
        
        For i = Worksheets.Count To 1 Step -1
            If Not shs(i) And Worksheets(i).Name <> "Sheet1" Then Worksheets(i).Delete
        Next i
    Else
        For Each Sh In Worksheets
            str1 = Worksheets("Sheet1").Range("B2").Value
            str2 = Worksheets("Sheet1").Range("B3").Value
            If str1 = "" Then str1 = "BBBBBBBB"
            If str2 = "" Then str2 = "CCCCCCCC"
            If Not (Sh.Name Like str1 & "*") And Not (Sh.Name Like str2 & "*") Then
                If Sh.Name <> "Sheet1" Then Sh.Delete
            End If
        Next Sh
    End If
    
    Application.DisplayAlerts = True

    ThisWorkbook.SaveAs Filename:=FPath & "\" & FName
      
End Sub

Usually, when you have difficulties with a tool provided by a third-party, the best place to get support is a forum/group that supports that tool. Usually, you'll find something like that on the web-site of the software manufacturer.
