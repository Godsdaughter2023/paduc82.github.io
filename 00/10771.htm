I fogot to put braces. Hence use below function 

Function ArrReturn(Rng As Range, CellInput) As String
    Dim Cell As Range
    Dim Length As Long
    Application.Volatile
    If CellInput = "" Then
        ArrReturn = "Input is Blank"
        Exit Function
    End If
    For Each Cell In Rng
        If Cell = CellInput Then
            ArrReturn = ArrReturn & """" & Cell.Offset(0, 1) & """" & ";"
        End If
    Next Cell
    If ArrReturn <> "" Then
        Length = Len(ArrReturn)
        ArrReturn = "{" & Mid(ArrReturn, 1, Length - 1) & "}"
    End If
End Function
