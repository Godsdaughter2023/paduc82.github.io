graham mayor say:
Maybe something like

Sub CountCategories()
Dim olItem As Outlook.MailItem
Dim iFolder As Folder
Dim iRed As Integer
Dim iBlue As Integer
Dim iNone As Integer

    iRed = 0
    iBlue = 0
    iNone = 0

    Set iFolder = Session.GetDefaultFolder(olFolderInbox)
    For Each olItem In iFolder.Items
        Select Case olItem.Categories
            Case Is = "Red Category": iRed = iRed + 1
            Case Is = "Blue Category": iBlue = iBlue + 1
            Case Else: iNone = iNone + 1
        End Select
    Next olItem
    MsgBox "Red: " & iRed & vbCr & _
    "Blue: " & iBlue & vbCr & _
    "Unflagged: " & iNone
Cleanup:
    Set olItem = Nothing
    Set iFolder = Nothing

End Sub
