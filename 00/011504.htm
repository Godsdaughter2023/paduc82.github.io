Function MyAverage(LookupValue As Variant, LookUpArray As Variant, NthValues As Integer)
   Dim intNameFound           As Integer
   Dim intNextName            As Integer
   Dim rngLookupArr           As Range
   Dim strSheet               As String
   Dim rngAvg                 As Range
   
   Set rngLookupArr = Intersect(LookUpArray, Sheets(LookUpArray.Parent.Name).UsedRange)
   
   intNameFound = WorksheetFunction.Match(LookupValue, rngLookupArr.Columns(1), 0)
   
   intNextName = LookUpArray.Columns(1).Find(What:="*", After:=LookUpArray.Cells(intNameFound, 1), LookAt:=xlPart, _
      SearchDirection:=xlByRows, MatchCase:=False).Row - 1
   
   If intNextName < intNameFound Then intNextName = rngLookupArr.Rows.Count
   
   With Sheets(LookUpArray.Parent.Name)
   Set rngAvg = Intersect(.Range(.Cells(intNameFound, LookUpArray.Columns(2).Column), .Cells(intNextName, LookUpArray.Columns(2).Column)), _
      .Range(.Cells(WorksheetFunction.Max(intNextName - NthValues + 1, intNameFound), LookUpArray.Columns(2).Column), .Cells(intNextName, LookUpArray.Columns(2).Column)))
   End With
   
   MyAverage = WorksheetFunction.Average(rngAvg)
   
End Function

Use this function in Sheet1 to get the average of the last 6 values or the last values if less than 6.

=MyAverage(C1,Sheet2!C:D,6)

One approach is with User Defined Types (Class module) and a VBA Macro.

To enter this Macro (Sub), <alt-F11> opens the Visual Basic Editor.
Ensure your project is highlighted in the Project Explorer window.

From the Top Menu, select Insert/Class and past the code below into the window that opens.

Then select the Class Module -- probably named Class1,  select <F4> and change the name property to

NameAvg

Then, from the top menu, select Insert/Module and
paste the code below into the window that opens.

===============================================

Private pName As String
Private pCount As Long
Private pData As Double

Public Property Get Name() As String
    Name = pName
End Property
Public Property Let Name(Value As String)
    pName = Value
End Property

Public Property Get Count() As Long
    Count = pCount
End Property
Public Property Let Count(Value As Long)
    pCount = Value
End Property

Public Property Get Data() As Double
    Data = pData
End Property
Public Property Let Data(Value As Double)
    pData = Value
End Property

===============================================

Change the Sheet names and ranges as appropriate.  I assumed Sheet1 and Sheet2; and that you had a labels row so we started with Row 2.

To use this Macro (Sub), <alt-F8> opens the macro dialog box. Select the macro by name, and <RUN>.

=====================================
Option Explicit
Sub AverageNames()
    Dim WS1 As Worksheet, WS2 As Worksheet
    Dim R1 As Range, R2 As Range
    Dim V1 As Variant, V2 As Variant
    Dim I As Long
    Dim cNameAvg As NameAvg
    Dim colNameAvg As Collection
    
Set WS1 = Worksheets("Sheet1")
Set WS2 = Worksheets("Sheet2")

With WS1
    Set R1 = .Range("C2", .Cells(.Rows.Count, "C").End(xlUp)).Resize(columnsize:=2)
End With

With WS2
    Set R2 = .Range("C2", .Cells(.Rows.Count, "D").End(xlUp))
End With

V1 = R1
V2 = R2

'Clean up the data
For I = 1 To UBound(V1)
    V1(I, 1) = Trim(Replace(V1(I, 1), Chr(160), ""))
Next I
For I = 1 To UBound(V2)
    V2(I, 1) = Trim(Replace(V2(I, 1), Chr(160), ""))
Next I
    
'Get all averages
Set colNameAvg = New Collection
For I = 1 To UBound(V2, 1)
    If Trim(V2(I, 1)) <> "" Then
        Set cNameAvg = New NameAvg
        With cNameAvg
            .Name = V2(I, 1)
            .Count = 1
            .Data = V2(I, 2)
            colNameAvg.Add cNameAvg, .Name
        End With
    Else
        With colNameAvg(cNameAvg.Name)
            .Count = .Count + 1
            .Data = .Data + V2(I, 2)
        End With
    End If
Next I

For I = 1 To UBound(V1)
    With colNameAvg(V1(I, 1))
        V1(I, 2) = .Data / .Count
    End With
Next I

R1 = V1
    
End Sub

john mcghie say:
Well, Suzanne is much more capable with philosophy than I, but I think I can solve your little problem :-)

In Word 2011, the "Insert>Symbol dialog is deprecated and just about to be removed.

The Advanced Symbol dialog is only half done, so far.

You will be able to add keystrokes for most macrons in Tools>Customise Keyboard...  Scroll the left column to see "Common Symbols" at the bottom.

Any that are not there, you can get with the Character Viewer, which is a Mac application.  Look it up in the Apple Help: it has to be enabled before you will see it.

To learn about the Character Viewer, see the Apple help topic:

Typing special characters and symbols


On the Mac, many of the macrons have keystrokes already assigned by the system, which will work anywhere.  You can reveal these with the Keyboard Viewer.  See the Apple Help topic:

Typing characters with accent marks


Of course, any characters you can't type by any of the above, you can create an AutoText for.  Use the Character Viewer to place the character in the document, store it as an AutoText, and assign a keystroke to the AutoText.


AutoText works the same as you're used to, and it's in the Word help if you get stuck.


Hope this helps
