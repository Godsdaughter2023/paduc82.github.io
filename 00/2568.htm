Even when you choose to delete your out-of-date security info, MS don't fully remove it for 30 days.

Sub final()
    Dim variable As String
    Dim wbk As Workbook
    Dim wsh As Worksheet
    Dim rng As Range
    Dim f As Boolean
    variable = Sheet1.Cells(8, 10)
    On Error Resume Next
    Set wbk = Workbooks("bhargav.xlsx")
    On Error GoTo 0
    If wbk Is Nothing Then
        Set wbk = Workbooks.Open(Filename:="C:\Users\BHARGAV\Desktop\bhargav.xlsx")
        f = True
    End If
    Set wsh = wbk.Worksheets(1)
    Set rng = wsh.Cells.Find(What:=variable, LookAt:=xlPart, MatchCase:=False)
    If Not rng Is Nothing Then
        With rng.Interior
            .Pattern = xlSolid
            .PatternColorIndex = xlAutomatic
            .Color = 65535
            .TintAndShade = 0
            .PatternTintAndShade = 0
        End With
        With rng.Font
            .Color = -16776961
            .TintAndShade = 0
        End With
    End If
    If f Then
        wbk.Close SaveChanges:=True
    Else
        wbk.Save
    End If
End Sub

Here is the code with lots of comments. Hopefully that will help you to understand it better.

Sub final()
    ' Declare the variables that we will use
    Dim variable As String
    Dim wbk As Workbook
    Dim wsh As Worksheet
    Dim rng As Range
    Dim f As Boolean
    ' The text to search for
    variable = Sheet1.Cells(8, 10)
    ' Try to refer to the workbook if it is open
    On Error Resume Next
    Set wbk = Workbooks("bhargav.xlsx")
    On Error GoTo 0
    ' If the workbook was not open, wbk will be Nothing...
    If wbk Is Nothing Then
        ' ... so open the workbook
        Set wbk = Workbooks.Open(Filename:="C:\Users\BHARGAV\Desktop\bhargav.xlsx")
        ' and set a Boolean variable to True to indicate that we opened the workbook
        f = True
    End If
    ' Refer to the first worksheet in the workbook
    Set wsh = wbk.Worksheets(1)
    ' Try to find the search text
    Set rng = wsh.Cells.Find(What:=variable, LookAt:=xlPart, MatchCase:=False)
    If Not rng Is Nothing Then
        ' If it is found, set the fill colour of the cell...
        With rng.Interior
            .Pattern = xlSolid
            .PatternColorIndex = xlAutomatic
            .Color = 65535
            .TintAndShade = 0
            .PatternTintAndShade = 0
        End With
        ' ... and the text colour
        With rng.Font
            .Color = -16776961
            .TintAndShade = 0
        End With
    End If
    ' Did we have to open the workbook?
    If f Then
        ' If so, save and close it
        wbk.Close SaveChanges:=True
    Else
        ' Otherwise, just save it
        wbk.Save
    End If
End Sub
