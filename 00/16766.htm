Function ConCatRange(CellBlock As Range) As String
'for non-contiguous cells  =ccr((a1:a10,c4,c6,e1:e5))

Dim cell As Range
Dim sbuf As String
    For Each cell In CellBlock
    On Error GoTo fred
    If Len(cell.text) > 0 Then sbuf = sbuf & cell.text & ", "
        Next
    ConCatRange = Left(sbuf, Len(sbuf) - 2)
fred:
End Function
