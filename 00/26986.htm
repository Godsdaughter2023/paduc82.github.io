per jessen say:

Then you can use this one:

 

Sub ChangeCells()
Dim Temp(1 To 2, 1 To 2) As Variant

If Selection.Cells.Count <> 2 Then Exit Sub

For Each cell In Selection
    c = c + 1
    Temp(c, 1) = cell.Address
    Temp(c, 2) = cell.Value
Next
Range(Temp(1, 1)) = Temp(2, 2)
Range(Temp(2, 1)) = Temp(1, 2)
End Sub
