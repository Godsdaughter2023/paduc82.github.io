Private Sub WdApp_MailMergeBeforeRecordMerge( _
 ByVal Doc As Document, Cancel As Boolean)
    Dim rngChartPie As Word.Range
    Dim rngChartBar As Word.Range

    If bFirstRecord Then
        iFileCodeStart = ThisDocument.MailMerge.DataSource.DataFields("code").Value
        bFirstRecord = False
    End If
    iFileCodeEnd = ThisDocument.MailMerge.DataSource.DataFields("code").Value

    'If something is wrong, don't continue
    'processing each record
    If CancelMerge = True Then
        Debug.Print "Cancelled. Record: " & CStr(recordIndex)
        Cancel = True
        appXl.Quit
        Set appXl = Nothing
        Exit Sub
    End If

    'The file containing the data for the merge
    'should only be opened once. Therefore,
    'we track when the merge has started
    If BeforeMergeExecuted = False Then
        BeforeMergeExecuted = True
        'MsgBox sMergeMessage, vbCritical + vbOKOnly
        Set DataXlPie = OpenChartDataFile(Doc.Path, PieChartDataXl)
        Set DataXlBar = OpenChartDataFile(Doc.Path, BarChartDataXl)
    End If
   
    If DataXlPie Is Nothing Or DataXlBar Is Nothing Then
        MsgBox "The data document could not be opened."
        CancelMerge = True
        Cancel = True
        Exit Sub
    End If
 
    
    If Doc.Bookmarks.Exists("PieChart") And Doc.Bookmarks.Exists("BarChart") Then
        
        Set rngChartPie = Doc.Bookmarks("PieChart").Range
        EditChart rngChartPie, DataXlPie


        Set rngChartBar = Doc.Bookmarks("BarChart").Range
        EditChart rngChartBar, DataXlBar

        DoEvents
        Cancel = False
    Else
        MsgBox "The bookmark " & "PieChart or BarChart" & " is missing."
        Cancel = True
        appXl.Quit
        Set appXl = Nothing
        Exit Sub
    End If
   
    'Make sure the changes to the chart
    'are carried over to the merge result
    rngChartPie.Fields.Update
    rngChartBar.Fields.Update
    Set rngChartPie = Nothing
    Set rngChartBar = Nothing
End Sub

bill manville say:

You can't run a macro in a workbook that is closed.

You can't run a macro that affects a workbook that is closed.

 

You can put code in Inventory.xlsm that opens stock.xlsx, updates it and closes it, e.g.

 

Sub Format()
    Workbooks.Open "S:/Data/stock.xlsx"
    Range("Table2[stockstatus]").Select
    Selection.NumberFormat = "General"
    Selection.TextToColumns Destination:=Range("Table2[stockstatus]"), DataType:=xlDelimited, _
        TextQualifier:=xlNone, ConsecutiveDelimiter:=False, Tab:=True, Semicolon _
        :=False, Comma:=False, Space:=False, Other:=False, FieldInfo:=Array(1, _
        1), TrailingMinusNumbers:=True
    ActiveWorkbook.Save
    Workbooks("stock.xlsx").Close
End Sub
