graham mayor say:
Outlook will normally reply using the address the message was sent to. New messages will use the default account. To reply using a particular account based on what folder the current message is filed in you will need a macro e.g. as follows
Change the names of the folders to match exactly what you have and the accounts you wish to associate with those folders (you can add more as required). Use the macro to reply to your messages

Sub ReplyUsingAccount()
Dim oAccount As Outlook.Account
Dim objItem As Outlook.MailItem
Dim oMail As Outlook.MailItem
Dim i As Long
Dim strAcc As String

Set objItem = ActiveExplorer.Selection.Item(1)
Select Case ActiveExplorer.CurrentFolder
    Case "Folder Name1": strAcc = "account1ATsomewhere.com"
    Case "Folder Name2": strAcc = "account2ATsomewhere.com"  
'etc
    Case Else: strAcc = "defaultaccATsomewhere.com"
End Select
For Each oAccount In Application.Session.Accounts
    If oAccount.DisplayName = strAcc Then
        Set oMail = objItem.Reply
        With oMail
            .SendUsingAccount = oAccount
            .Display
        End With
        Exit For
    End If
Next oAccount
Set oAccount = Nothing
Set objItem = Nothing
Set oMail = Nothing
End Sub
