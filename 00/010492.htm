That is because Excel only goes to 15 digits of accuracy. 

Use the NumberFormat to display in Scientific and set to more than 15 Decimal places and it will  convert the number so it only displays zeros after the 15 digits.

Sub WritetoWord()

'Step 1:  Declare your variables
Dim MyRange As Excel.Range
Dim wd As Object
Dim wdDoc As Object
Dim WdRange As Object
Dim bStarted As Boolean
Dim oVars As Object

    'Step 2:  Open the target Word document
    On Error Resume Next
    Set wd = GetObject(, "Word.Application")
    If Err Then
        Set wd = CreateObject("Word.Application")
        bStarted = True
    End If

    Set wdDoc = wd.Documents.Open _
                (ThisWorkbook.Path & "\" & "NW22 Plan Clean-up Ltr.rtf")
    wd.Visible = True

    'Step 3:  Write the data to a named docvariable
    Set oVars = wdDoc.variables
    On Error Resume Next
    oVars("varDate").Value = Format(Sheets("Sheet1").Range("L1"), "d MMMM yyyy")

    'Step 4:  Update the docvariable field in the target document

    '{DocVariable varDate }
    wdDoc.Range.Fields.Update
   
    'Step 5:  Close and save the document
    wdDoc.Close -1
    'Step 6:  Memory cleanup
    If bStarted Then wd.Quit
    Set wd = Nothing
    Set wdDoc = Nothing
    Set WdRange = Nothing
