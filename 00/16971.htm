Sub InsertRows()

Sub SplitIntoPages()

    Dim docSingle As Document
    Dim rngPage As Range
    Dim iCurrentPage As Integer
    Dim iPageCount As Integer
    Dim strNewFileName As String

    Application.ScreenUpdating = False

    Set docMultiple = ActiveDocument
   
     'instantiate the range object:
    Set rngPage = docMultiple.Range

     'get the document's page count:
    iPageCount = docMultiple.Content.ComputeStatistics(wdStatisticPages)
       
    Do Until iCurrentPage > iPageCount
        If iCurrentPage = iPageCount Then
            rngPage.End = ActiveDocument.Range.End 'last page (there won't be a next page)
        Else
             'Find the beginning of the next page:
             'Must use the Selection object (the Range.Goto method will not work on a page)
            Selection.GoTo wdGoToPage, wdGoToAbsolute, iCurrentPage + 1
           
             'Set end of range to the point between the pages:
            rngPage.End = Selection.Start
        End If
       
        rngPage.Copy                    'copy page to clipboard
        Set docSingle = Documents.Add   'create a new document
        docSingle.Range.Paste           'paste to new document
       
         'remove any manual page break to prevent a second blank:
        docSingle.Range.Find.Execute Findtext:="^m", ReplaceWith:=""

        strNewFileName = docSingle.Paragraphs(1).Range.Text
       
         'Remove multiple spaces:

        docSingle.Close                 'close the new document
        rngPage.Collapse wdCollapseEnd  'go to the next page

    Application.ScreenUpdating = True 'restore the screen updating
    
     'Destroy the objects:
    Set docMultiple = Nothing
    Set docSingle = Nothing
    Set rngPage = Nothing

oldDoc.Save

    .EndKey Unit:=wdStory
    Letters = .Information(wdActiveEndPageNumber)
    .HomeKey Unit:=wdStory

Counter = 1

While Counter < Letters + 1
    Application.ScreenUpdating = False
    ActiveDocument.Bookmarks("\page").Range.Cut
    Set newDoc = Documents.Add

        .Paste
        .EndKey Unit:=wdStory

    Set oRng = newDoc.Paragraphs(1).Range
    oRng.End = oRng.End - 1

                  FileFormat:=wdFormatDocument
    '+++++++++++++++++++++++
   
    newDoc.Close
    Counter = Counter + 1
    Application.ScreenUpdating = True
Wend
oldDoc.Close wdDoNotSaveChanges
Documents.Open sLoad

graham mayor say:
A couple of points:

1. You don't have to open the header view to access the table. You can set the range directly to the cell in the table.
2. It is not clear what the two replace functions are for as we don't know what's in the table cell, but they may be superfluous.
3. The macro assumes a single value in the table which will be used to name the document, in the currently active folder, given that you haven't specified a path.

Sub Rename2()
Dim oTable As Table
Dim rng As Range
Dim strNewFileName As String
    Set oTable = ActiveDocument.Sections(1).Headers(wdHeaderFooterPrimary).Range.Tables(1)
    Set rng = oTable.Cell(row:=1, Column:=2).Range
    rng.End = rng.End - 1
    strNewFileName = rng.Text & ".docx"
    'Remove multiple spaces:
    While InStr(strNewFileName, "  ") > 0
        strNewFileName = Replace(strNewFileName, "  ", " ")        'replace 2 spaces with 1
    Wend
    'Remove carriage return:
    strNewFileName = Replace(strNewFileName, Chr(13), "")
    ActiveDocument.SaveAs2 strNewFileName, FileFormat:=12
End Sub

4. If we are to assume that there are lots of different documents each with a similar table, then you need to loop through the documents in question e.g. as follows. In this case the documents are saved in a sub folder of the existing folder which if not present is created. The original documents are unaffected.

This should all work provided there are no illegal filename characters in the cell.

Sub BatchProcess()
Dim strFilename As String
Dim strPath As String
Dim strNewPath As String
Dim oDoc As Document
Dim fDialog As FileDialog
Dim oTable As Table
Dim oRng As Range

    Set fDialog = Application.FileDialog(msoFileDialogFolderPicker)
    With fDialog
        .Title = "Select folder and click OK"
        .AllowMultiSelect = False
        .InitialView = msoFileDialogViewList
        If .Show <> -1 Then
            MsgBox "Cancelled By User", , _
                   "List Folder Contents"
            Exit Sub
        End If
        strPath = fDialog.SelectedItems.Item(1) & "\"
        strNewPath = strPath & "Renamed\"
        If Not FolderExists(strNewPath) Then
            MkDir strNewPath
        End If
    End With
    strFilename = Dir$(strPath & "*.doc")
    While Len(strFilename) <> 0
        WordBasic.DisableAutoMacros 1
        Set oDoc = Documents.Open(strPath & strFilename)

        Set oTable = oDoc.Sections(1).Headers(wdHeaderFooterPrimary).Range.Tables(1)
        Set oRng = oTable.Cell(row:=1, Column:=2).Range
        oRng.End = oRng.End - 1
        strFilename = oRng.Text & ".docx"
        'Remove multiple spaces:
        While InStr(strFilename, "  ") > 0
            strFilename = Replace(strFilename, "  ", " ")        'replace 2 spaces with 1
        Wend
        'Remove carriage return:
        strFilename = Replace(strFilename, Chr(13), "")
        oDoc.SaveAs2 Filename:=strNewPath & strFilename, FileFormat:=12
        oDoc.Close SaveChanges:=wdDoNotSaveChanges
        strFilename = Dir$()
        WordBasic.DisableAutoMacros 0
    Wend
    Set oTable = Nothing
    Set oRng = Nothing
    Set oDoc = Nothing
    MsgBox "Process complete"
End Sub
Public Function FolderExists(ByVal PathName As String) As Boolean
   Dim nAttr As Long
   ' Grab the attributes, test valid values for folder bit.
   On Error GoTo NoFolder
   nAttr = GetAttr(PathName)
   If (nAttr And vbDirectory) = vbDirectory Then
      FolderExists = True
   End If
NoFolder:
End Function
