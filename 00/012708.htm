joel engineer say:

See code below.  To get the weighted number I just repeated the number the number of groups it appeared.  I then assigned a random number to each number in the list and sorted the list of number by the random number.  then I selected the number by the sorted list making sure I only picked each number once.

 

Sub WeightdRandomNumbes()

Const NumbersToSelect = 267

   'below is the input data
   'first number is the range of numbers and second number is the weight
   inputdata = Array(Array("1-108", 4), Array("217-324", 4), Array("433-540", 4), _
                   Array("649-756", 4), Array("865-972", 4), Array("325-432", 3), _
                   Array("109-216", 2), Array("541-648", 1), Array("757-864", 1))

   Range("A1") = "Input data"
   Range("B1") = "Random Number"
   Range("C1") = "Selected Numbers"
  
   RowCount = 2
   For i = LBound(inputdata, 1) To UBound(inputdata, 1)
       InputRange = Split(inputdata(i)(0), "-")
       LowerRange = InputRange(0)
       UpperRange = InputRange(1)
       Weight = inputdata(i)(1)
      
       For inputnumber = LowerRange To UpperRange
          'add data to worksheet
          For repeatcount = 1 To Weight
             Range("A" & RowCount) = inputnumber
             RowCount = RowCount + 1
          Next repeatcount
        Next inputnumber
   Next i
  
   'add add a random number to each input number
   LastRow = Range("A" & Rows.Count).End(xlUp).Row
   For RowCount = 2 To LastRow
      Range("B" & RowCount) = Rnd()
  
   Next RowCount
  
   'sort number by random value
   Range("A1:B" & LastRow).Sort _
      Header:=xlYes, _
      key1:=Range("B1"), _
      order1:=xlAscending
     
    RowCount = 2
    ResultsRowCount = 2
    NumbersTaken = 0
    'take numbers by randoom order checking to make sure it doesn't already exist
    Do
       Number = Range("A" & RowCount)
       If RowCount = 2 Then    ' first row don't check if number exists
          Range("C" & RowCount) = Number
       Else
          'test if number already choosen
          Set c = Columns("C").Find(what:=Number, LookIn:=xlValues, lookat:=xlWhole)
          If c Is Nothing Then
             Range("C" & ResultsRowCount) = Number
             ResultsRowCount = ResultsRowCount + 1
             NumbersTaken = NumbersTaken + 1
          End If
        End If
       RowCount = RowCount + 1
    Loop While (NumbersTaken <= NumbersToSelect) And (RowCount <= LastRow)
   

End Sub

You cannot delete rows or columns from a worksheet.

You can hide them only.

You may now decide how many rows you want visible and re-hide those you don't need to view.
