'===================================================================================
' The original Monte Carlo simulation came from work by Jack Dalgren, Copyright 2004
' This version includes new functionality which calculates the Criticality Index,
' Significance Index, Schedule Sensitivity Index, and Cruciality Index
' (as explained here http://www.pmknowledgecenter.com/node/131) while also adding
' enhanced formatting to the excel sheet output.
'===================================================================================
Option Explicit
'set some global variables
Dim iter As Integer, numTasks As Integer
Public exportedTasks As Tasks, monteCarloTasks As Tasks
Dim myCancel As Boolean

Sub setExportedTasks()
Dim exportcount As Integer
'filters to get collection of tasks for export
'with flag10 set to yes
'warn if export is more than 25 tasks
'Tasks are filtered for a positive value in flag10
FilterEdit Name:="_MCexportedTasks", TaskFilter:=True, Create:=True, OverwriteExisting:=True, FieldName:="Flag10", test:="equals", Value:="yes", ShowInMenu:=True, ShowSummaryTasks:=False
FilterApply Name:="_MCexportedTasks"
'then the tasks are selected
SelectAll
'if there are no tasks that meet the Criteria we display a warning and quit
If ActiveSelection = 0 Then
'this line is needed to tell the "monte" subprocedure to exit too.
myCancel = True
MsgBox "You have no tasks to export data for" & Chr(13) _
& "Please check the flag10 field to be sure that some tasks are marked YES"
Exit Sub
End If
'take the selection and turn it into a collection of tasks
Set exportedTasks = ActiveSelection.Tasks
exportcount = exportedTasks.Count

'we don't want to operate on more than 25
'and even that is a large number. 5 or so would make sense
If exportcount > 25 Then
'this asks if it is ok and if it is not OK then we exit as before
If MsgBox("You are exporting " & exportcount & " tasks" & Chr(13) & "Are you sure you want to continue?", vbOKCancel, "Large Export Warning") = vbCancel Then
myCancel = True
Exit Sub
End If
End If
End Sub
Sub setMonteCarloTasks()
Dim t As task
'filters to get collection of tasks for export
'with flag11 set to yes
For Each t In ActiveProject.Tasks

'this part does two things
If Not t Is Nothing Then
'first it stores the duration into duration1 so we can restore it
'back to what it was when we finish calculating
t.Duration1 = t.Duration
t.Flag11 = "No"
If Not t.Summary Then
'second it checks to see if optimistic and pessimistic durations
'are entered for tasks. If they are then flag11 is set to indicate
'that it is a task that should be calculated. If there is missing data or
'bad data we ignore the task
'Probably a good idea to have a routine that helps users enter
'and check these before starting
If (t.Duration >= t.Duration2) And (t.Duration <= t.Duration3) Then
t.Flag11 = "Yes"
End If
End If
End If
Next t
'now we filter for valid tasks
FilterEdit Name:="_MCTasks", TaskFilter:=True, Create:=True, OverwriteExisting:=True, FieldName:="Flag11", test:="equals", Value:="yes", ShowInMenu:=False, ShowSummaryTasks:=False
FilterApply Name:="_MCTasks"
SelectAll
'check to make sure there is at least one
If ActiveSelection = 0 Then
MsgBox "You have no valid tasks for the macro to work on"
myCancel = True
Exit Sub
End If
'and finally set them as the tasks we will recalculate
Set monteCarloTasks = ActiveSelection.Tasks
'Get the number of tasks to calculate for the sensitivity analysis
numTasks = monteCarloTasks.Count

End Sub

Function TriDist(ByVal prob As Single, ByVal opt As Single, ByVal expect As Single, ByVal pess As Single)
'this function returns a value from within the triangularity probablility
Dim x, d As Double
x = (expect - opt) / (pess - opt)


If prob <= x Then TriDist = opt + (prob * (pess - opt) * (expect - opt)) ^ 0.5
If prob > x Then TriDist = pess - ((1 - prob) * (pess - opt) * (pess - expect)) ^ 0.5

End Function

Sub calculateIndices(ByRef xlbook As Workbook, ByVal xlApp As Excel.Application)
    
Dim i As Integer
Dim dates As String
Dim xlRow As Excel.Range
Dim t As task
Dim ts As Integer

Dim xlSheet As New Excel.Worksheet

'Criticality Index, Sensitivity Index, Schedule Sensitivity Index, Cruciality Index (Spearman's Rank Correlation)
Dim CI() As Double, SI() As Double, SSI() As Single, CRIr() As Double, CRIt() As Double, CRIp() As Double
'AD= Activity Duration
Dim taskCounter As Integer, SL() As Long
'AvgAD = Average Activity Duration
Dim AD() As Double, AvgAD() As Double, stDevAD() As Double
'SD = Simulated Project Duration, AvgSD = Aerage Simulated Project Duration,
Dim SD() As Double, AvgSD As Double, stDevSD As Double, rankAD() As Single, rankSD() As Single

'Redefine sensitivity analysis variables
ReDim CI(numTasks), SI(numTasks), CRIr(numTasks), CRIt(numTasks), CRIp(numTasks)
ReDim AD(iter, numTasks), AvgAD(numTasks)
ReDim SD(iter), SSI(numTasks), stDevAD(numTasks), rankAD(iter, numTasks), rankSD(iter)
ReDim SL(iter, numTasks)

Set xlSheet = xlbook.Worksheets.Add
xlSheet.Name = "Simulations"
Set xlRow = xlSheet.Range("A1")

ts = 0
'write out the task names for exported tasks in the first row
For Each t In exportedTasks
    xlRow = t.Name
    Set xlRow = xlRow.Offset(0, 1)
    ts = ts + 1
Next t

'increment to take into account SD duration
xlRow = "Simulated Project Duration"
'Format Column
xlSheet.Columns(numTasks + 1).EntireColumn.AutoFit
Set xlRow = xlRow.Offset(0, 1)
ts = ts + 1

'iter is the number of iterations chosen by the user
'this loop sets the duration to a random but representative
'value within the triangular distribution, recalcs the project and then
'writes the results to excel one time for each iteration
For i = 1 To iter
    Set xlRow = xlRow.Offset(1, 0)
    Set xlRow = xlRow.Offset(0, -ts)
    
    'Here we call the TriDist function which returns a value we will use for
    'the duration. We do this for all valid tasks
    For Each t In monteCarloTasks
        If t.Duration2 > t.Duration3 Then
            MsgBox "Task " & t.ID & " has an optimistic duration (Duration2) that is greater than the pessimistic duration (Duration3). Please check."
            Exit Sub
        ElseIf t.Duration2 = 0 And t.Duration3 = 0 And t.Duration = 0 Then
                t.Duration = 0
        ElseIf t.Duration2 = t.Duration3 Then
                MsgBox "Task " & t.ID & " has a pessimistic duration equal to its optimistic duration. Please check."
        Else
                t.Duration = TriDist(Rnd(), t.Duration2, t.Duration, t.Duration3)
        End If
    Next t
    'once the new duration is set we recalculate the schedule
    Application.CalculateProject
    
    '==============================================================================================
    'Sensitivity Intermediate Calculations
    'project total simulated duration "SD"
    SD(i) = Round(ActiveProject.ProjectSummaryTask.Duration / (60 * 8), 0)
    'Partial calculation for average of all project durations
    AvgSD = AvgSD + SD(i)
    '==============================================================================================
    
    'Initialiting counter
    taskCounter = 1
    
    'Need to set here, rather than in above loop so that project can recalculate
    For Each t In exportedTasks

        xlRow = Round(t.Duration / (60 * 8), 0)
        Set xlRow = xlRow.Offset(0, 1)
        
        '==============================================================================================
        'Sensitivity Intermediate Calculations
        'A. Partial Calculation of Criticality Index; Add up whether a task is CRIttical or not over all iterations
        If t.Critical Then
            CI(taskCounter) = CI(taskCounter) + 1
        End If
        'store durations for calculations later
        AD(i, taskCounter) = Round(t.Duration / (60 * 8), 0)
        'Partial calculation for average of all each task durations
        AvgAD(taskCounter) = AvgAD(taskCounter) + AD(i, taskCounter)
        SL(i, taskCounter) = Round(t.FreeSlack / (60 * 8), 0)
        
        
        If i > 1 Then
            Dim temp As Integer
            'for temp = i to
            For temp = i - 1 To 1 Step -1
                If (AD(temp, taskCounter) - AD(i, taskCounter)) * (SD(temp) - SD(i)) > 0 Then
                    CRIt(taskCounter) = CRIt(taskCounter) + 1
                End If
            Next temp
        End If
        taskCounter = taskCounter + 1
    Next t
    xlRow = SD(i)
    Set xlRow = xlRow.Offset(0, 1)
    
Next i

'Once all simulations are compulte, average or "Expected Value" project duration of all the simulations of the schedule
AvgSD = Round(AvgSD / iter, 0)

'Average of each task
For taskCounter = 1 To numTasks
    AvgAD(taskCounter) = Round(AvgAD(taskCounter) / iter, 0)
    'Calculate the CRIt
    CRIt(taskCounter) = 4 * CRIt(taskCounter) / (iter ^ 2 - iter) - 1
Next taskCounter

'Set up calculations for the standard deviation of all the Simulated Durations
For i = 1 To iter
    'calculate variance for each task
    stDevSD = stDevSD + (SD(i) - AvgSD) ^ 2

    For taskCounter = 1 To numTasks
        stDevAD(taskCounter) = stDevAD(taskCounter) + (AD(i, taskCounter) - AvgAD(taskCounter)) ^ 2
        'Calculate intermediate value of Significance Index
         SI(taskCounter) = SI(taskCounter) + AD(i, taskCounter) * SD(i) / ((SL(i, taskCounter) + AD(i, taskCounter)) * AvgSD)
        
         CRIr(taskCounter) = CRIr(taskCounter) + (AD(i, taskCounter) - AvgAD(taskCounter)) * (SD(i) - AvgSD)
         If i = iter Then
             CRIr(taskCounter) = Abs(CRIr(taskCounter) / (stDevAD(taskCounter) * stDevSD) ^ 0.5)
        End If
    Next taskCounter
Next i
'Std Dev of the Project Simulation Durations
stDevSD = (stDevSD / (iter - 1)) ^ 0.5

'When we are done we reset the duration to what it was at the beginning
'this value has been store in the setMonteCarloTasks subprocedure
For Each t In monteCarloTasks
t.Duration = t.Duration1
Next t

'Output Sensitivity analysis variables
Set xlSheet = xlbook.Worksheets.Add
xlSheet.Name = "Sensitivity Analysis Data"

'Set xlRow = xlapp.ActiveCell
xlSheet.Cells(2, 1) = "Critical Index"
xlSheet.Cells(3, 1) = "Significance Index"
xlSheet.Cells(4, 1) = "Schedule Sensitivity Index"
xlSheet.Cells(5, 1) = "Cruciality Index Kendall's Tau Rank"
xlSheet.Cells(6, 1) = "Cruciality Index Person's Product-Moment"
xlSheet.Cells(7, 1) = "Cruciality Index Spearman's Rank Correlation"

Set xlRow = xlSheet.Cells(1, 2)
taskCounter = 1

For Each t In exportedTasks
    'Identify each task
    xlRow = t.Name

    'Record Critical Index Value for each task
    Set xlRow = xlRow.Offset(1, 0)
    CI(taskCounter) = CI(taskCounter) / iter
    xlRow = CI(taskCounter)
       
    'Record Significance Index Value for each task
    Set xlRow = xlRow.Offset(1, 0)
    SI(taskCounter) = SI(taskCounter) / iter
    xlRow = SI(taskCounter)
    
    'Record Schedule Sensitivity Index Value for each task
    Set xlRow = xlRow.Offset(1, 0)
    stDevAD(taskCounter) = (stDevAD(taskCounter) / (iter - 1)) ^ 0.5
    xlRow = stDevAD(taskCounter) * CI(taskCounter) / stDevSD
        
    'Record Cruciality Index for each task
    Set xlRow = xlRow.Offset(1, 0)
    xlRow = CRIt(taskCounter)
        
    'Record CRIr for each task
    Set xlRow = xlRow.Offset(1, 0)
    xlRow = CRIr(taskCounter)
        
    taskCounter = taskCounter + 1
    Set xlRow = xlRow.Offset(-5, 1)
Next t

    'format the sheet
    xlSheet.Columns("A:A").EntireColumn.AutoFit
    'format each indice to highlight the max values
    For i = 2 To 7
        Dim cell As Excel.Range
        Dim max As Double
        Set xlRow = xlSheet.Range( _
            xlSheet.Cells(i, 2), xlSheet.Cells(i, numTasks + 1))
        max = xlApp.WorksheetFunction.max(xlRow)
        For Each cell In xlRow
            If cell.Value = max Then
                cell.Interior.ColorIndex = 3
            End If
        Next cell
    Next i
    
    
    
End Sub
