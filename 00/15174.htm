  Set oCC = ActiveDocument.SelectContentControlsByTag("Meeting Day").Item(1)

      'We probably have a valid user selected date, but lets check to be sure.
      If Len(oRng) > Len(oRng.Words(1)) + 2 Then
        'Eliminate the day of the week from the range
        oRng.MoveStart wdCharacter, Len(oRng.Words(1)) + 2
        'What is left is probably a valid date, but check to be sure.
        If IsDate(oRng) Then
