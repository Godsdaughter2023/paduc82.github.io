doug robbins - mvp office apps & services (word) say:

Not by using the Page Layout tab>Page Background>Page Color facility.

However, if you have a Section Breaks for each page, you can insert a full page shape into the header of the first page, then unlink the headers in each Section from those in the previous section and fill each of the now separate shapes with a different color.  Only thing is, if your text flows from one page to another, the color will flow with it.

However, once again, you could overcome that issue by inserting text boxes on each page that are all linked and have their fill color set to none and then you text will flow from one textbox to the next without causing the color to run.

Sub Example_Variations()
  Dim c As Collection, Item, i As Integer

  Debug.Print
  Set c = Variations(Array("A", "B", "C", "D"), 2)
  i = 0
  For Each Item In c
    i = i + 1
    Debug.Print i; ": "; Join(Item)
  Next

Function Variations(Arr, ByVal Count As Integer) As Collection
  'Return all possible variations of count elements in Arr as arrays in a collection
  Dim Index() As Long, Pos() As Long
  Dim Result()
  Dim i As Long
  Dim j As Integer, k As Integer
  'Create a collection
  Set Variations = New Collection
  'Return empty collection if Count exceed the bound's of Arr
  If LBound(Arr) + Count - 1 > UBound(Arr) Or Count < 1 Then Exit Function
  'Setup space for index and result array
  ReDim Index(1 To Count) As Long
  ReDim Pos(1 To Count) As Long
  ReDim Result(1 To Count)
  'Setup index for first combination
  For i = 1 To Count
    Index(i) = LBound(Arr) + i - 1
  Next
  Do
    'Build variation
    For i = 1 To Count
      Result(i) = Arr(Index(i))
    Next
    'Save it
    Variations.Add Result
    'Get next index
    i = Count
    j = 0
    Do
      'Max. position for this index reached?
      If Index(i) = UBound(Arr) Then
Again:
        j = j + 1
        'Set index before in next loop
        i = i - 1
        If i < 1 Then Exit Function
      Else
        'Increment position
        Index(i) = Index(i) + 1
        'Verify this position is unused
        For k = 1 To i - 1
          'Position found?
          If Index(k) = Index(i) Then
            'Yes, try next
            Index(i) = Index(i) + 1
            'Restart the FOR loop
            k = 0
            'If we run out of bounds try next index
            If Index(i) > UBound(Arr) Then GoTo Again
          End If
        Next
        'Search slots for remaining indices
        For j = 1 To j
          i = i + 1
          Index(i) = LBound(Arr)
          For k = 1 To i - 1
            If Index(k) = Index(i) Then
              Index(i) = Index(i) + 1
              k = 0
              If Index(i) > UBound(Arr) Then GoTo Again
            End If
          Next
        Next
        Exit Do
      End If
    Loop
  Loop
End Function
