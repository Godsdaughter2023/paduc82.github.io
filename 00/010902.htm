jeeped say:
That gives a much better understanding of the process. In this case, a manual entry in column E can be used as the trigger to evaluate the value in column L and move the information into a new row if necessary. Note that the manual entry in column E is the trigger, not the change in column L.

It would be better to write the formula in column L when the value in column E is entered. For the purpose of this exercise, I will assume 8 hours in a workday and that the formula in column L is simply =MAX(E1-8,0).

Right-click the worksheet's name tab and when the VBE opens, paste the following into the pane titled something like Book1 - Sheet1 (Code), 

Private Sub Worksheet_Change(ByVal Target As Range)
    If Not Intersect(Target, Me.Columns("E").SpecialCells(xlConstants, xlNumbers)) _
      Is Nothing Then
        Application.EnableEvents = False
        Dim e As Range, dWrkHrs As Double
        dWrkHrs = 8
        For Each e In Intersect(Target, Me.Columns("E"))
            e.Offset(0, 7).FormulaR1C1 = "=IFERROR(MAX(RC[-7]-" & dWrkHrs & ", 0),"""")"
            If Application.Sum(e.Offset(0, 7).Value) > 0 Then
                Me.Cells(Rows.Count, "A").End(xlUp).Offset(1, 0) = e.Offset(0, -4).Value
                Me.Cells(Rows.Count, "A").End(xlUp).Offset(0, 2) = e.Offset(0, -2).Value
                Me.Cells(Rows.Count, "A").End(xlUp).Offset(0, 4) = e.Offset(0, 7).Value
                Me.Cells(Rows.Count, "A").End(xlUp).Offset(0, 11).FormulaR1C1 = _
                  "=IFERROR(MAX(RC[-7]-" & dWrkHrs & ", 0),"""")"
            End If
            Do While Me.Cells(Rows.Count, "L").End(xlUp).Value > 0
                Me.Cells(Rows.Count, "A").End(xlUp).Offset(1, 0) = _
                  Me.Cells(Rows.Count, "A").End(xlUp).Value
                Me.Cells(Rows.Count, "C").End(xlUp).Offset(1, 0) = _
                  Me.Cells(Rows.Count, "C").End(xlUp).Value
                Me.Cells(Rows.Count, "E").End(xlUp).Offset(1, 0) = _
                  Me.Cells(Rows.Count, "L").End(xlUp).Value
                Me.Cells(Rows.Count, "L").End(xlUp).Offset(1, 0).FormulaR1C1 = _
                  "=IFERROR(MAX(RC[-7]-" & dWrkHrs & ", 0),"""")"
            Loop
        Next e
    End If
FallThrough:
    Application.EnableEvents = True
End Sub

Note that the variable dWrkHrs is assigned a value of 8. Tap Alt+Q to return to your worksheet. When you type a value into column E that is great than the value assigned to dWrkHrs, a new row will be created and populated with the values from the entry row,

I'm not sure what you wanted to do with the original value from column. If you want it reset to a maximum of dWrkHrs, then post back and some modifications may be made.

ossiemac say:

Replace the previous code with the code below. Unfortunately it is not possible to use "DisplayFormat" in a UDF that is called from a worksheet so cannot get the color that is set by the Conditional formatting. "DisplayFormat" works in other VBA code and even in a UDF that is called by other VBA code but not when called from the worksheet.

Therefore I have used the values in the cells as set to apply the Conditional formatting colors.

Function CountColors(rngToCount As Range, rngColor As Range) As Long
    Dim rngCel As Range
    Dim i As Long
   
    Set rngCel = rngToCount.Cells(1, 1) 'First cell of range with colors to be counted
   
    Do
        'Test if cell is part of a Merged area AND contains Conditional Formatting
        If rngCel.MergeArea.Cells.Count > 1 And rngCel.MergeArea.FormatConditions.Count > 0 Then
           
            Select Case rngColor.Interior.Color 'The sample color ajdacent to function on worksheet
                Case 5296274    'Green
                    'Test the cell value against values used in Conditional formatting
                    If rngCel.Value >= 1 And rngCel.Value <= 8 Then
                        CountColors = CountColors + 1
                    End If
                   
                Case 65535  'Yellow
                    'Test the cell value against values used in Conditional formatting
                    If rngCel.Value >= 9 And rngCel.Value <= 18 Then
                        CountColors = CountColors + 1
                    End If
                   
                Case 255    'Red
                    'Test the cell value against values used in Conditional formatting
                    If rngCel.Value >= 19 And rngCel.Value <= 26 Then
                        CountColors = CountColors + 1
                    End If
            End Select
           
            With rngCel.MergeArea
                'Move to the next cell past the current merged area
                Set rngCel = .Cells(.Cells.Count).Offset(1, 0)
            End With
        Else
            'Else in case there are unmerged cells or no Format Conditions between the merged areas
            Set rngCel = rngCel.Offset(1, 0)
        End If
    Loop While Not Intersect(rngCel, rngToCount) Is Nothing
   
End Function
