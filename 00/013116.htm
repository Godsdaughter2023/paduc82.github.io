  Dim hFile As Integer
  Dim Contents As String
  Dim Lines, SplitLen, Result
  Dim i As Long, j As Long, p As Long
 
  'Define the lenghts
  SplitLen = Array(10, 4, 4, 8, 10, 10)

  'Prompt for the file
  With Application.FileDialog(msoFileDialogFilePicker)
    .InitialFileName = "C:\Users\JoeDoe\Desktop\Combined Issue Check Process\"
    .Filters.Add "Text files", "*.txt"
    If Not .Show Then Exit Sub
    FName = .SelectedItems(1)
  End With

  'Read the file
  hFile = FreeFile
  Open FName For Binary Access Read As #hFile
  Contents = Space(LOF(hFile))
  Get #hFile, , Contents
  Close #hFile
 
  'Split into lines
  Lines = Split(Contents, vbCrLf)
  ReDim Result(0 To UBound(Lines), 0 To UBound(SplitLen) + 1)
 
  'Split the lines by lengths
  For i = 0 To UBound(Lines)
    p = 1
    For j = 0 To UBound(SplitLen)
      Result(i, j) = Mid(Lines(i), p, SplitLen(j))
      p = p + SplitLen(j)
    Next
    'The remainder
    Result(i, j) = Mid(Lines(i), p)
  Next
 
  With Worksheets("All Data")
    'Clear previous results
    .UsedRange.ClearContents
    'Flush the data into the sheet
    .Range("A1").Resize(UBound(Result) + 1, UBound(Result, 2) + 1).Value = Result
  End With
End Sub
