Use the windows API SetForegroundWindow / GetForegroundWindow instead until the focus is back to Word.

Andreas.

Private Declare Sub Sleep Lib "kernel32" (ByVal dwMilliseconds As Long)
Private Declare Function GetForegroundWindowA Lib "user32" Alias "GetForegroundWindow" () As Long
Private Declare Function SetForegroundWindowA Lib "user32" Alias "SetForegroundWindow" ( _
    ByVal hWnd As Long) As Long

Function SetForegroundWindow(ByVal hWnd As Long) As Boolean
  Dim i As Long
  'If IsIconic(hWnd) Then ShowWindow hWnd, SW_RESTORE
  Do
    SetForegroundWindowA hWnd
    DoEvents
    If GetForegroundWindowA = hWnd Then
      SetForegroundWindow = True
      Exit Function
    End If
    Sleep 100
    i = i + 1
  Loop Until i = 10
End Function

Function GetForegroundWindow() As Long
  GetForegroundWindow = GetForegroundWindowA
End Function

'=========>>

'--------->>
Public Sub Tester()

Note that the source file should not be open when this macro is executed, otherwise a run-time permission error #70 will be encountered.
