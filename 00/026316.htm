hansv mvp say:

Quotes within quoted strings must be doubled, and formulas should begin with =.

 

    Application.EnableEvents = False
    Set FormatRange = Range("A4:V" & LastRow)
    With FormatRange
        .FormatConditions.Delete
        With .FormatConditions.Add(Type:=xlExpression, Formula1:="=AND($B4<>"""",$B5="""",$D5<>"""")")
            .Interior.ColorIndex = 40 '' to colour orange
            .StopIfTrue = False
        End With
        With .FormatConditions.Add(Type:=xlExpression, Formula1:="=AND($B4="""",$B5="""",$D5<>"""")")
            .Interior.ColorIndex = 37 '' to colour blue
            .StopIfTrue = False
        End With
        With .FormatConditions.Add(Type:=xlExpression, Formula1:="=$D4<>""""")
            .Borders.LineStyle = xlContinuous  ''to put border round cells
            .StopIfTrue = False
        End With
    End With
