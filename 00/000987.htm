tasos_k say:

Hi,

if i understand you want to split an xl file (only col A and B)

in separate csv files every 250 entries ?

 

if so,

try this code

(expected csv files named: 1.csv, 2.csv,....7.csv,...  in same folder

with existing xl file)

 

 

Sub SplitInSeparateCSVFiles()
'Oct 28 2013
Const N As Long = 250   ' << split every 250 entries
Const myDelim As String = ","     '<< select comma or semicolon
Dim WS As Worksheet
Set WS = Sheets("Sheet1") '<<  source sht name, change
Dim r As Long, x As Long, i As Long, j As Long, t As Long
r = WS.Range("A:B").Find(What:="*", SearchOrder:=xlByRows, SearchDirection:=xlPrevious).Row
Dim myPath As String
myPath = ThisWorkbook.Path & "\"
Dim v() As Variant
ReDim v(1 To 2)
t = 1
For i = 1 To r Step N
Open myPath & t & ".csv" For Output As #1
For x = i To N + i - 1
For j = 1 To 2
v(j) = WS.Cells(x, j)
Next
Print #1, Join(v, myDelim)
Next x
Close #1
t = t + 1
Next
End Sub
