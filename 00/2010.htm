phantom2200 say: This issue has nothing to do with Nokia.

Its the way USB is designed to work.

The main key point is to differentiate USB and Mini-USB.

The Nokia has a Mini USB connector while PCs have regular USB.

The regular USB port (as in the case of your flash drive) consists of 6 pins, 4 on the pins are always use for the transfer of data, and 2 pins of power. This is the process of how USB works. (same principle also applies to firewire)

However, with the case of Mini USB, due to the small size used, there is not enough room to fit 6 pins, so instead 4 pins are fitted, all being to transfer data, this meant the two power pins were neglected.

Therefore, a Mini USB device will never carry power, be it on a Phone, PDA, PC, PS3 or what ever device.

Its the limitation of Mini USB and nothing to do with Nokia. If the Nokia used regular USB, then yes, you can also charge the phone but that would mean more room required to incorporate the USB port on the phone due to its size compared with Mini USB.

Public Function SetFileDateTime(ByVal FileName As String, _
  ByVal TheDate As String) As Boolean
'************************************************
'PURPOSE:    Set File Date (and optionally time)
'            for a given file)
        
'PARAMETERS: TheDate -- Date to Set File's Modified Date/Time
'            FileName -- The File Name

'Returns:    True if successful, false otherwise
'************************************************
If Dir(FileName) = "" Then Exit Function
If Not IsDate(TheDate) Then Exit Function

Dim lFileHnd As Long
Dim lRet As Long

Dim typFileTime As FILETIME
Dim typLocalTime As FILETIME
Dim typSystemTime As SYSTEMTIME

With typSystemTime
    .wYear = Year(TheDate)
    .wMonth = Month(TheDate)
    .wDay = Day(TheDate)
    .wDayOfWeek = Weekday(TheDate) - 1
    .wHour = Hour(TheDate)
    .wMinute = Minute(TheDate)
    .wSecond = Second(TheDate)
End With

lRet = SystemTimeToFileTime(typSystemTime, typLocalTime)
lRet = LocalFileTimeToFileTime(typLocalTime, typFileTime)

lFileHnd = CreateFile(FileName, GENERIC_WRITE, _
    FILE_SHARE_READ Or FILE_SHARE_WRITE, ByVal 0&, _
    OPEN_EXISTING, 0, 0)
   
lRet = SetFileTime(lFileHnd, ByVal 0&, ByVal 0&, _
         typFileTime)

CloseHandle lFileHnd
SetFileDateTime = lRet > 0

End Function

Sub AddCoverPages()
' Minimise screen flickering
Application.ScreenUpdating = False

Dim strFolder As String, strFile As String, wdDoc As Document
Dim Rslt, strPDF As String, strDocNm As String, strDoc As String

If Rslt = "A" Then
  strPDF = "C:\Users\walter.k.adasczik\Documents\1 BLANK AREA 5 Access-Request-Form_May 2018.pdf"
ElseIf Rslt = "B" Then
  strPDF = "C:\Users\walter.k.adasczik\Documents\(SCR) Temporary Redlake Camera.pdf"
Else
  Exit Sub
End If

If strFolder = "" Then Exit Sub
If FSO Is Nothing Then Set FSO = CreateObject("Scripting.FileSystemObject")
strDocNm = ActiveDocument.FullName
strFile = Dir(strFolder & "\*.doc", vbNormal)
Do While strFile <> ""
  strDoc = strFolder & "\" & strFile
  If strDoc <> strDocNm Then
    ' Store the file's current Date/Time stamp.
    Set oItem = FSO.GetFile(strDoc)
    StrDtTm = oItem.DateLastModified
    Set wdDoc = Documents.Open(FileName:=strDoc, AddToRecentFiles:=False, ReadOnly:=False, Visible:=False)
    With wdDoc
      .Range.InsertBefore vbCr & Chr(12)
      .InlineShapes.AddOLEObject ClassType:="AcroExch.Document.DC", FileName:=strPDF, _
        LinkToFile:=False, DisplayAsIcon:=False, Range:=.Range.Characters.First
      .Close SaveChanges:=True
    End With
    DoEvents
    ' Reset the file's Date/Time stamp.
    Set oItem = FSO.GetFile(strDoc)
    If oItem.DateLastModified <> StrDtTm Then oItem.DateLastModified = StrDtTm
    Set oItem = Nothing
  End If
  strFile = Dir()
Loop
' Restore screen updating
Application.ScreenUpdating = True
End Sub

strDocNm = ActiveDocument.FullName
strFile = Dir(strFolder & "\*.doc", vbNormal)
Do While strFile <> ""
  strDoc = strFolder & "\" & strFile
  If strDoc <> strDocNm Then
    ' Store the file's current Date/Time stamp.
    Set oItem = FSO.GetFile(strDoc)
    StrDtTm = oItem.DateLastModified
    Set wdDoc = Documents.Open(FileName:=strDoc, AddToRecentFiles:=False, ReadOnly:=False, Visible:=False)
    With wdDoc
      .Range.InsertBefore vbCr & Chr(12)
      '.InlineShapes.AddOLEObject ClassType:="AcroExch.Document.DC", FileName:=strPDF, _
        LinkToFile:=False, DisplayAsIcon:=False, Range:=.Range.Characters.First
      .Close SaveChanges:=True
    End With
    DoEvents
    ' Reset the file's Date/Time stamp.
    SetFileDateTime strDoc, StrDtTm
  End If
  strFile = Dir()
Loop
Set oItem = Nothing: Set wdDoc = Nothing
' Restore screen updating
Application.ScreenUpdating = True
End Sub
