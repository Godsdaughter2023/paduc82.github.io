Sub apply_changes()

    'ERROR HANDLING
    On Error GoTo ErrorHandling

    'SET PROPERTIES
    Application.ScreenUpdating = False
    Application.Calculation = False

    'IF opening (successful)

        'RESET SHEETS TO ALLOW FOR FOLLOWING STEPS
        Dim c As Range, TheRow As Long, TheColumn As Long
        Dim ws As Worksheet
        Set ws = Sheets("FREIGHT OVERVIEW")
        Set c = ws.Cells.Find("Voyage Number", LookIn:=xlValues, LookAt:=xlWhole)
        TheRow = c.Row
        TheColumn = c.Column
        If TheRow > 1 Then
            ws.Range("A1").Resize(TheRow - 1).EntireRow.Delete
        End If
        If TheColumn > 1 Then
            ws.Range("A1").Resize(, TheColumn - 1).EntireColumn.Delete
        End If
        Worksheets("FREIGHT OVERVIEW").Pictures.Delete

        Dim Mc As Range, MTheRow As Long, MTheColumn As Long
        Dim Mws As Worksheet
        Set Mws = Sheets("MANIFEST")
        Set Mc = Mws.Cells.Find("Voyage", LookIn:=xlValues, LookAt:=xlWhole)
        MTheRow = Mc.Row
        MTheColumn = Mc.Column
        If MTheRow > 1 Then
            Mws.Range("A1").Resize(MTheRow - 1).EntireRow.Delete
        End If
        If MTheColumn > 1 Then
            Mws.Range("A1").Resize(, MTheColumn - 1).EntireColumn.Delete
        End If
        Worksheets("MANIFEST").Pictures.Delete

        'STEP 1 - Insert columns, populate and hide unrequired
        Sheets("FREIGHT OVERVIEW").Range("B1").EntireColumn.Insert
        Sheets("FREIGHT OVERVIEW").Range("I1").EntireColumn.Insert
        Sheets("FREIGHT OVERVIEW").Range("T1").EntireColumn.Insert
        Sheets("FREIGHT OVERVIEW").Range("U1").EntireColumn.Insert

        Dim FOLastRow As Long
        FOLastRow = Sheets("FREIGHT OVERVIEW").Range("A" & Rows.Count).End(xlUp).Row
        Sheets("FREIGHT OVERVIEW").Range("B2").Formula = "=LEFT(A2,LEN(A2)-4)"
        Sheets("FREIGHT OVERVIEW").Range("B2").AutoFill Destination:=Sheets("FREIGHT OVERVIEW").Range("B2:B" & FOLastRow)
        Application.Calculation = xlCalculationAutomatic
            Sheets("FREIGHT OVERVIEW").Range("I2").Formula = "=RIGHT(H2,LEN(H2)-5)"
            Sheets("FREIGHT OVERVIEW").Range("I2").AutoFill Destination:=Sheets("FREIGHT OVERVIEW").Range("I2:I" & FOLastRow)
            Sheets("FREIGHT OVERVIEW").Range("T2").Formula = "=IF(W2=""C"",S2,"""")"
            Sheets("FREIGHT OVERVIEW").Range("T2").AutoFill Destination:=Sheets("FREIGHT OVERVIEW").Range("T2:T" & FOLastRow)
            Sheets("FREIGHT OVERVIEW").Range("U2").Formula = "=IF(W2=""P"",S2,"""")"
            Sheets("FREIGHT OVERVIEW").Range("U2").AutoFill Destination:=Sheets("FREIGHT OVERVIEW").Range("U2:U" & FOLastRow)
        Application.Calculation = False

        Worksheets("FREIGHT OVERVIEW").Columns("D:G").Hidden = True
        Worksheets("FREIGHT OVERVIEW").Columns("J").Hidden = True
        Worksheets("FREIGHT OVERVIEW").Columns("L").Hidden = True
        Worksheets("FREIGHT OVERVIEW").Columns("O:R").Hidden = True
        Worksheets("FREIGHT OVERVIEW").Columns("V").Hidden = True

        'STEP 2 - Sort FREIGHT OVERVIEW by column I (A-Z)
        Dim strDataRange As Range
        Dim keyRange As Range
        Set strDataRange = Worksheets("FREIGHT OVERVIEW").Range("A1:XFD495")
        Set keyRange = Worksheets("FREIGHT OVERVIEW").Range("I1")
        strDataRange.Sort key1:=keyRange, Header:=xlYes

        'STEP 3 - Merge cells that have the same ARMACUP BL number and freight code
        Dim FOlngRow As Long

        With Sheets("FREIGHT OVERVIEW")
            Dim columnToMatchABL As Integer: columnToMatchABL = 9
            Dim columnToMatchFC As Integer: columnToMatchFC = 14
            Dim columnToSumFO As Integer: columnToSumFO = 19

            FOlngRow = .Cells(65536, columnToMatchABL).End(xlUp).Row
            .Cells(columnToMatchABL).CurrentRegion.Sort key1:=.Cells(columnToMatchABL), Header:=xlYes

            Do
                If .Cells(FOlngRow, columnToMatchABL) = .Cells(FOlngRow - 1, columnToMatchABL) Then
                    If .Cells(FOlngRow, columnToMatchFC) = .Cells(FOlngRow - 1, columnToMatchFC) Then
                        .Cells(FOlngRow - 1, columnToSumFO) = .Cells(FOlngRow - 1, columnToSumFO) + .Cells(FOlngRow, columnToSumFO)
                        .Rows(FOlngRow).Delete
                    End If
                End If

                FOlngRow = FOlngRow - 1
            Loop Until FOlngRow = 1
        End With

        'STEP 4 - Filter results

        Worksheets("FREIGHT OVERVIEW").Range("N1:N495").AutoFilter _
            field:=1, _
            Criteria1:="SRC", _
            VisibleDropDown:=False

        'STEP 5 - Copy column C (Vessel name) to column A on VESSEL DATA
        Sheets("FREIGHT OVERVIEW").Range("C3:C495").Copy
        Sheets("VESSEL DATA").Range("A5").PasteSpecial xlPasteValues
        Sheets("FREIGHT OVERVIEW").Range("C3").Copy
        Sheets("CREATE VESSEL").Range("U10").PasteSpecial xlPasteValues

        'STEP 6 - Copy column B (Voyage number) to column B on VESSEL DATA.
        Sheets("FREIGHT OVERVIEW").Range("B3:B495").Copy
        Sheets("VESSEL DATA").Range("B5").PasteSpecial xlPasteValues
        Sheets("FREIGHT OVERVIEW").Range("B3").Copy
        Sheets("CREATE VESSEL").Range("U11").PasteSpecial xlPasteValues

        'STEP 7 - Copy column I (ARMACUP BL number) to column F on VESSEL DATA
        Sheets("FREIGHT OVERVIEW").Range("I3:I495").Copy
        Sheets("VESSEL DATA").Range("F5").PasteSpecial xlPasteValues

        'STEP 8 - Copy column K (JP ref) to column H on VESSEL DATA
        Sheets("FREIGHT OVERVIEW").Range("K3:K495").Copy
        Sheets("VESSEL DATA").Range("H5").PasteSpecial xlPasteValues

        'STEP 9 - Copy column M (Consignee) to column K on VESSEL DATA
        Application.Calculation = xlCalculationAutomatic
            Sheets("FREIGHT OVERVIEW").Range("M3:M495").Copy
            Sheets("VESSEL DATA").Range("K5").PasteSpecial xlPasteValues
        Application.Calculation = False

        'STEP 10 - Copy column S (BL Actual Charges) to column AJ on VESSEL DATA
        Sheets("FREIGHT OVERVIEW").Range("S3:S495").Copy
        Sheets("VESSEL DATA").Range("AJ5").PasteSpecial xlPasteValues

        'STEP 11 - Remove current filter and then filter column N by ORA
        Sheets("FREIGHT OVERVIEW").Cells.AutoFilter

        Worksheets("FREIGHT OVERVIEW").Range("N1:N495").AutoFilter _
            field:=1, _
            Criteria1:="ORA", _
            VisibleDropDown:=False

        'STEP 12 - Check column W (BL Terms) to see if P or C, then copy column T and U (BL Actual Charges) into respective columns of VESSEL DATA sheet
        Application.Calculation = xlCalculationAutomatic
            Sheets("FREIGHT OVERVIEW").Range("T2:U495").Copy
            Sheets("VESSEL DATA").Range("AH5").PasteSpecial xlPasteValues
        Application.Calculation = False

        'MOVE TO MANIFEST
        'STEP 1 - Insert columns, populate and hide unrequired
        Sheets("MANIFEST").Range("C1").EntireColumn.Insert

        Dim LastRow As Long
        LastRow = Sheets("MANIFEST").Range("B" & Rows.Count).End(xlUp).Row
        Application.Calculation = xlCalculationAutomatic
            Sheets("MANIFEST").Range("C2").Formula = "=RIGHT(B2,LEN(B2)-5)"
            Sheets("MANIFEST").Range("C2").AutoFill Destination:=Sheets("MANIFEST").Range("C2:C" & LastRow)
        Application.Calculation = False

        Worksheets("MANIFEST").Columns("A").Hidden = True
        Worksheets("MANIFEST").Columns("D:AI").Hidden = True
        Worksheets("MANIFEST").Columns("AL:AU").Hidden = True
        Worksheets("MANIFEST").Columns("AW:AZ").Hidden = True

        'STEP 2 - Merge cells with same BL number
        Dim MlngRow As Long

        With Sheets("MANIFEST")
            Dim columnToMatch As Integer: columnToMatch = 3
            Dim columnToConcatenate As Integer: columnToConcatenate = 37
            Dim columnToSumM As Integer: columnToSumM = 36

            MlngRow = .Cells(65536, columnToMatch).End(xlUp).Row
            .Cells(columnToMatch).CurrentRegion.Sort key1:=.Cells(columnToMatch), Header:=xlYes

            Do
                If .Cells(MlngRow, columnToMatch) = .Cells(MlngRow - 1, columnToMatch) Then
                    .Cells(MlngRow - 1, columnToConcatenate) = .Cells(MlngRow - 1, columnToConcatenate) & " / " & .Cells(MlngRow, columnToConcatenate)
                    .Cells(MlngRow - 1, columnToSumM) = .Cells(MlngRow - 1, columnToSumM) + .Cells(MlngRow, columnToSumM)
                    .Rows(MlngRow).Delete
                End If

                MlngRow = MlngRow - 1
            Loop Until MlngRow = 1
        End With

        'STEP 3 - Sort by column C and copy both column AJ (Quantity) and column AK (Description) to I5 of VESSEL DATA
        Dim MstrDataRange As Range
        Dim MkeyRange As Range
        Set MstrDataRange = Worksheets("MANIFEST").Range("A1:S495")
        Set MkeyRange = Worksheets("MANIFEST").Range("C1")
        MstrDataRange.Sort key1:=MkeyRange, Header:=xlYes

        Sheets("MANIFEST").Range("AJ2:AK495").Copy
        Sheets("VESSEL DATA").Range("I5").PasteSpecial xlPasteValues
        Sheets("MANIFEST").Range("AV2:AV495").Copy
        Sheets("VESSEL DATA").Range("D5").PasteSpecial xlPasteValues

        'MOVE TO VESSEL DATA
        'STEP 1 - Organise contents by Port, SOU first
        Dim VstrDataRange As Range
        Dim VkeyRange As Range
        Set VstrDataRange = Worksheets("VESSEL DATA").Range("A5:BC500")
        Set VkeyRange = Worksheets("VESSEL DATA").Range("E5")
        VstrDataRange.Sort key1:=VkeyRange, Header:=xlNo, Order1:=xlDescending

        'STEP 2 - Delete rows between A5 and A201 if blank
        Application.EnableEvents = False
            Dim DeleteRng As Range
            Dim DeleteI As Long
            Set DeleteRng = Worksheets("VESSEL DATA").Range("A5:A500")
            With DeleteRng
                ' Loop through all cells of the range
                ' Loop backwards, hence the "Step -1"
                For DeleteI = .Rows.Count To 1 Step -1
                    If .Item(DeleteI) = "" Then
                        ' Since cell is empty, delete the whole row
                        .Item(DeleteI).EntireRow.Delete
                    End If
                Next DeleteI
            End With
        Application.EnableEvents = True

        'STEP 3 - Apply filters
        Sheets("VESSEL DATA").Range("A4:BC4").AutoFilter

        'SET PROPERTIES
        Application.ScreenUpdating = True
        Application.Calculation = xlCalculationAutomatic

        'Message of succession
        MsgBox "Applying changes has been successful. Please continue to stage 3."

    'Else

        'SET PROPERTIES
        Application.ScreenUpdating = True
        Application.Calculation = xlCalculationAutomatic

        'Message of failure

    'END IF

    Exit Sub

ErrorHandling:
    MsgBox GetErrorMsg(Err.Number)

    Application.ScreenUpdating = True
    Application.Calculation = xlCalculationAutomatic

End Sub
Function GetErrorMsg(ErrorNo As Long) As String
    Select Case ErrorNo
        Case 9
            GetErrorMsg = "Oops! An unavoidable error has occured (" & str(Err.Number) & " )." & vbCrLf & vbCrLf & _
                          "Most common fault:" & vbCrLf & _
                          "No spreadsheets have been imported during stage 1." & vbCrLf & vbCrLf & _
                          "Please ensure that you've imported the spreadsheets as required."
        Case 91
            GetErrorMsg = "Oops! An unavoidable error has occured (" & str(Err.Number) & " )." & vbCrLf & vbCrLf & _
                          "Most common fault:" & vbCrLf & _
                          "No voyage number has been found on the freight overview spreadsheet." & vbCrLf & vbCrLf & _
                          "Resolution:" & vbCrLf & _
                          "This error is explained in further detail in the ARMACUP manual in the Error Handling section. Please follow the steps to resolve the issue." & vbCrLf & vbCrLf & _
                          "Please close the generator ensuring you DO NOT SAVE and try again."
        Case Else
            GetErrorMsg = "Oops! An unavoidable error has occured (" & str(Err.Number) & " ). Please manually set up your vessel."
    End Select
End Function
