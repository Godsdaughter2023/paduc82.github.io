Public Sub AddControl(C As Control, ByVal ControlPostion As frmControlType)
  'Übernimmt die Positions-/Größenkontrolle des übergebenen Steuerelements
  If (ControlPostion And frmControlPosToRight) = frmControlPosToRight Then
    'Man kann nicht die rechte Position und die Breite gleichzeitig ändern!
    ControlPostion = ControlPostion And Not frmControlWidth
  End If
  If (ControlPostion And frmControlPosToButtom) = frmControlPosToButtom Then
    'Man kann nicht die untere Position und die Höhe gleichzeitig ändern!
    ControlPostion = ControlPostion And Not frmControlHeight
  End If
  objControls.Add C, ControlPostion
End Sub

Public Sub AdjustControls()
  'Kontrolliert die Steuerelemente bei einer Größenänderung
  Dim i As Integer, ControlTypes(), Controls()
  If IsIconic Then
    'Excel in den Vordergrund bringen, sonst wechselt Windows zum
    'nächsten Task wenn die Userform modal / gebunden ist!
    SetForegroundWindow hwndForm
    'SetForegroundWindow FindWindow("xlmain", vbNullString)
    Exit Sub
  End If
#If AddResizeControl Then
  If Not objResizer Is Nothing Then
    If IsZoomed Then
      objResizer.MousePointer = fmMousePointerDefault
    Else
      objResizer.MousePointer = fmMousePointerSizeNWSE
    End If
  End If
#End If
  With objForm
    'Die min. Größe der Form auf die Anfangsgröße beschränken!
    If .Width < InitWidth Then .Width = InitWidth
    If .Height < InitHeight Then .Height = InitHeight

    'Die übergebenen Steuerelemente kontrollieren
    ControlTypes = objControls.Items
    Controls = objControls.Keys
    For i = 0 To objControls.Count - 1
      If (ControlTypes(i) And frmControlPosToRight) = frmControlPosToRight Then
        Controls(i).Left = Controls(i).Left + .Width - LastWidth
      End If
      If (ControlTypes(i) And frmControlPosToButtom) = frmControlPosToButtom Then
        Controls(i).Top = Controls(i).Top + .Height - LastHeight
      End If
      If (ControlTypes(i) And frmControlWidth) = frmControlWidth Then
        Controls(i).Width = Controls(i).Width + .Width - LastWidth
      End If
      If (ControlTypes(i) And frmControlHeight) = frmControlHeight Then
        Controls(i).Height = Controls(i).Height + .Height - LastHeight
      End If
    Next
    LastWidth = .Width
    LastHeight = .Height
  End With
End Sub

japanzaman say:

To do what you want will require a slightly different approach than what John gave you, or at least it seems that way to me. This macro will do what you are asking, but it assumes you ONLY want the last period in every paragraph deleted. It will not delete every period in the paragraph. You could easily adjust this to work for all punctuation, or any characters you want to get rid of. I should also mention that if there is a period in the paragraph, it will be deleted regardless of where it happens to be in the paragraph as I have no way of knowing how long these paragraphs are going to be. This means it could end up deleting a period you want to keep if it happens to be the last in the paragraph. 

 

Sub deleteperiods()

Dim sld As Slide, shp As Shape, periodgone As Boolean

 

For Each sld In ActivePresentation.Slides

   For Each shp In sld.Shapes

      If shp.HasTextFrame = msoTrue Then

         With shp.TextFrame.TextRange

            For I = 1 To .Paragraphs.Count

               periodgone = False

               If .Paragraphs(I).ParagraphFormat.Bullet = msoTrue Then

                  For k = .Paragraphs(I).Characters.Count To 1 Step -1

                     If Not .Paragraphs(I).Characters(k).Text = Chr(32) And periodgone = True Then 'we are done with this paragraph

                        Exit For

                     End If

                     If .Paragraphs(I).Characters(k).Text = Chr(32) Then

                        .Paragraphs(I).Characters(k).Delete

                     End If

                     If .Paragraphs(I).Characters(k).Text = "." Then

                        .Paragraphs(I).Characters(k).Delete

                        periodgone = True

                     End If

                  Next

               End If

            Next

         End With

      End If

   Next

Next

 

End Sub

hansv mvp say:

You can create a procedure (sub) that is called by all three:

 

Private Sub ChkClothing_Click()
    Call HandleClick
End Sub

Private Sub ChkThis_Click()
    Call HandleClick
End Sub

Private Sub ChkThat_Click()
    Call HandleClick
End Sub

Private Sub HandleClick()
    If ChkClothing Or ChkThis Or ChkThat Then
        txtItem.Visible = True
        Item.Visible = True
        txtItem.SetFocus
    Else
        txtItem.Text = ""
        txtItem.Visible = False
        Item.Visible = False
    End If
End Sub

amit tandon say:

Refer below image:

"Open" between 2 dates - 2 options:

Formula in cell F2 - refers to start / end dates in cells E2 & E3:

=COUNTIFS($B$2:$B$100,"Open",$A$2:$A$100,">="&E2,$A$2:$A$100,"<="&E3)

Formula in cell F3 - start / end dates hardcoded within the formula:

=COUNTIFS($B$2:$B$100,"Open",$A$2:$A$100,">="&DATE(2016,7,15),$A$2:$A$100,"<="&DATE(2016,11,1))

Formula in cell F5 - "Open" in a specific month & year, as per date in E5:

=COUNTIFS($B$2:$B$100,"Open",$A$2:$A$100,">="&DATE(YEAR($E5),MONTH($E5),1),$A$2:$A$100,"<="&EOMONTH($E5,0))

Formula in cell F7 - "Open" in a specific month (irrespective of the year), as per date in E7:

=SUMPRODUCT(($B$2:$B$100="Open")*(MONTH($A$2:$A$100)=MONTH($E7)))
