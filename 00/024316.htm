alessandro iavarone say: Hi,
I found a solution for my issue:

First, I enabled "Developer" tab, from Options. Then, pressing ALT-F11, I opened VBA editor. Here you can write this code:



Private WithEvents mySentMails As Outlook.Items

Private Sub Application_Startup()
    Dim myNameSpace As Outlook.NameSpace
    Dim myMAPIFolder As Outlook.Folder

    Set myNameSpace = Application.GetNamespace("MAPI")
    Set myMAPIFolder = myNameSpace.Folders.Item("EXCHANGE_FOLDER").Folders("Sent Item")
    Set mySentMails = myMAPIFolder.Items
End Sub

Private Sub mySentMails_ItemAdd(ByVal Item As Object)
    Dim myNameSpace As Outlook.NameSpace
    Dim myPSTFolder As Outlook.Folder

    Set myNameSpace = Outlook.GetNamespace("MAPI")
    Set myPSTFolder = myNameSpace.GetDefaultFolder(olFolderSentMail)

    If TypeOf Item Is Outlook.MailItem Then
        Item.Move myPSTFolder
    End If

    Set myNameSpace = Nothing
    Set myPSTFolder = Nothing
End Sub


In the "OnStartup" event, you define the objects that you will use during your session. Change "EXCHANGE_FOLDER" with the name of your Exchange account (tipically Last Name First Name).

Jokes aside, why do you use FormulaR1C1 when you don't need it?

When you want to create formulas with a macro it is much easier if you write the formula manually into the sheet first (to be sure it works) and then use the same formula in VBA!

Sub CalcAwardAmts()

    Dim Rng As Range
    'Insert a column
    Columns("E").Insert
    'Get the used cells in column A
    Set Rng = Range("A1", Range("A" & Rows.Count).End(xlUp))
    'Map to column E
    Set Rng = Rng.Offset(, 4)
    'Write the formula into this cells
    Rng.Formula = "=VLOOKUP(A1,Sheet2!$E$1:$F$10,1,FALSE)"
End Sub

When you want to modify a formula in VBA, don't chop up the formula and try to combine them with many """&""&"

A much easier way it to use REPLACE and the Address property of the Range object.

Note: The #Const line of the following code must be at the top of the module!

#Const Version = 2

Sub Insert_Right_Demo()
    Dim Rng As Range
    Dim Yr As Range
    Dim Formula As String
    
    'Insert a column
    Columns("E").Insert
    'Get the used cells in column A
    Set Rng = Range("A1", Range("A" & Rows.Count).End(xlUp))
    'Map to column E
    Set Rng = Rng.Offset(, 4)
    
    'Setup the basic formula
    Formula = "=2013-YEAR(F1)"
    
    'Where is the year?
    Set Yr = Worksheets("Sheet2").Range("D1")
    
#If Version = 1 Then
    'Version A (exchange the value)
    Formula = Replace(Formula, "2013", Yr.Value)
#Else
    'Version B (use the cell reference, absolute style)
    Formula = Replace(Formula, "2013", Yr.Address(External:=True))
#End If
    
    Rng.Formula = Formula
End Sub

If you want the path in the footer then insert a Filename field in the footer with a \p switch.
{ FILENAME \p } - use CTRL+F9 to set the bracket pair.

or

Use  a macro to insert the filename and path at the cursor http://www.gmayor.com/installing_macro.htm   

Sub InsertFileNameAndPath()
With ActiveDocument
    If Len(.Path) = 0 Then .Save
    Selection.TypeText .FullName
End With
End Sub


or

put the filename and path at the end of the document with a macro

Sub InsertFileNameAtBookmark()
Dim oBM As Bookmark
Dim oRng As Range
Dim bFound As Boolean
    With ActiveDocument
        If Len(.Path) = 0 Then .Save
        For Each oBM In ActiveDocument.Bookmarks
            If oBM.name = "bmPath" Then
                Set oRng = oBM.Range
                bFound = True
                Exit For
            End If
        Next oBM
        If Not bFound Then
            ActiveDocument.Range.InsertParagraphAfter
            Set oRng = ActiveDocument.Paragraphs.Last.Range
            oRng.Collapse wdCollapseEnd
            oRng.Paragraphs.Alignment = _
            wdAlignParagraphRight
        End If
        oRng.Text = ActiveDocument.FullName
        oRng.Font.name = "Arial"
        oRng.Font.Size = 8
        ActiveDocument.Bookmarks.Add "bmPath", oRng
    End With
End Sub
