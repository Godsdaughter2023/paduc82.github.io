Try this:

Sub MoveData()
    Dim wsOp As Worksheet, wsA As Worksheet
    Dim rng As Range, rngHdr As Range
    Dim lnLastRow As Long
   
    Set wsOp = Worksheets("Operations")
    Set wsA = Worksheets("Country A")
   
     'Derivative Class column “D8” = Security Type on Standard Dashboard column “I2”
    Set rngHdr = wsOp.Range("8:8").Find( _
        What:="Security Type", _
        After:=wsOp.Cells(8, wsOp.Columns.Count), _
        LookIn:=xlValues, LookAt:=xlWhole, _
        SearchOrder:=xlByRows, SearchDirection:=xlNext, _
        MatchCase:=False, SearchFormat:=False)
       
    If Not rngHdr Is Nothing Then
        lnLastRow = rngHdr.End(xlDown).Row
        Set rng = wsOp.Range(rngHdr, wsOp.Cells(lnLastRow, rngHdr.Column))
        'Set rng = wsOp.Range(rngHdr.Offset(1, 0), wsOp.Cells(lnLastRow, rngHdr.Column)) 'if you only want to copy the data
        rng.Copy
        wsA.Cells(wsA.Rows.Count, "I").End(xlUp).Offset(1, 0).PasteSpecial Paste:=xlPasteValues
        Set rngHdr = Nothing
    Else
        MsgBox "Couldn't find 'Security Type' in row 8 of sheet '" & wsOp.Name & "'"
    End If
   
     'Derivative Ticker column “E8” = Security Alias on Standard Dashboard column “F2”
    Set rngHdr = wsOp.Range("8:8").Find( _
        What:="Security Alias", _
        After:=wsOp.Cells(8, wsOp.Columns.Count), _
        LookIn:=xlValues, LookAt:=xlWhole, _
        SearchOrder:=xlByRows, SearchDirection:=xlNext, _
        MatchCase:=False, SearchFormat:=False)
       
    If Not rngHdr Is Nothing Then
        lnLastRow = rngHdr.End(xlDown).Row
        Set rng = wsOp.Range(rngHdr, wsOp.Cells(lnLastRow, rngHdr.Column))
        'Set rng = wsOp.Range(rngHdr.Offset(1, 0), wsOp.Cells(lnLastRow, rngHdr.Column)) 'if you only want to copy the data
        rng.Copy
        wsA.Cells(wsA.Rows.Count, "F").End(xlUp).Offset(1, 0).PasteSpecial Paste:=xlPasteValues
        Set rngHdr = Nothing
    Else
        MsgBox "Couldn't find 'Security Alias' in row 8 of sheet '" & wsOp.Name & "'"
    End If
   
     'Fund column “F8”= Portfolio Group on Standard Dashboard column “A2”
    Set rngHdr = wsOp.Range("8:8").Find( _
        What:="Portfolio Group", _
        After:=wsOp.Cells(8, wsOp.Columns.Count), _
        LookIn:=xlValues, LookAt:=xlWhole, _
        SearchOrder:=xlByRows, SearchDirection:=xlNext, _
        MatchCase:=False, SearchFormat:=False)
       
    If Not rngHdr Is Nothing Then
        lnLastRow = rngHdr.End(xlDown).Row
        Set rng = wsOp.Range(rngHdr, wsOp.Cells(lnLastRow, rngHdr.Column))
        'Set rng = wsOp.Range(rngHdr.Offset(1, 0), wsOp.Cells(lnLastRow, rngHdr.Column)) 'if you only want to copy the data

        wsA.Cells(wsA.Rows.Count, "A").End(xlUp).Offset(1, 0).PasteSpecial Paste:=xlPasteValues
        Set rngHdr = Nothing
    Else
        MsgBox "Couldn't find 'Portfolio Group' in row 8 of sheet '" & wsOp.Name & "'"
    End If
   
    Application.CutCopyMode = False
End Sub


All of which can be "simplified" like this:

Sub CallMoveSomeData()
    Call MoveSomeData(strSrcSht:="Operations", strHdrTxt:="Security Type", strDestCol:="I", strDestSht:="Country A")
    Call MoveSomeData(strSrcSht:="Operations", strHdrTxt:="Security Alias", strDestCol:="F", strDestSht:="Country A")
    Call MoveSomeData(strSrcSht:="Operations", strHdrTxt:="Portfolio Group", strDestCol:="A", strDestSht:="Country A")
End Sub

Sub MoveSomeData(strSrcSht As String, strHdrTxt As String, strDestCol As String, strDestSht As String)
    Dim wsSrc As Worksheet, wsDest As Worksheet
    Dim rng As Range, rngHdr As Range
    Dim lnLastRow As Long
   
    On Error Resume Next
    Set wsSrc = Worksheets(strSrcSht)
    If wsSrc Is Nothing Then MsgBox "Sheet '" & strSrcSht & "' could not be found in the active workbook!": Exit Sub
    Set wsDest = Worksheets(strDestSht)
    If wsDest Is Nothing Then MsgBox "Sheet '" & strDestSht & "' could not be found in the active workbook!": Exit Sub
    On Error GoTo 0
   
     'Find a cell in row 8 in the source worksheet containing strHdrTxt:
    Set rngHdr = wsSrc.Range("8:8").Find( _
        What:=strHdrTxt, _
        After:=wsSrc.Cells(8, wsSrc.Columns.Count), _
        LookIn:=xlValues, LookAt:=xlWhole, _
        SearchOrder:=xlByRows, SearchDirection:=xlNext, _
        MatchCase:=False, SearchFormat:=False)
       
    If Not rngHdr Is Nothing Then
        lnLastRow = rngHdr.End(xlDown).Row
        'lnLastRow = wsSrc.Cells(wsSrc.Rows.Count, rngHdr.Column).End(xlUp).Row 'alternative if there could be gaps
        Set rng = wsSrc.Range(rngHdr, wsSrc.Cells(lnLastRow, rngHdr.Column))
        'Set rng = wsSrc.Range(rngHdr.Offset(1, 0), wsSrc.Cells(lnLastRow, rngHdr.Column)) 'if you only want to copy the data
        rng.Copy
        wsDest.Cells(wsDest.Rows.Count, strDestCol).End(xlUp).Offset(1, 0).PasteSpecial Paste:=xlPasteValues
        Application.CutCopyMode = False
    Else
        MsgBox "Couldn't find '" & strHdrTxt & "' in row 8 of sheet '" & wsSrc.Name & "'"
    End If
End Sub

There's no particular reason that Word 2010 can't use macros from Word 2013, but there are several possible pitfalls.

In Word 2013, did you save the document as a Macro-Enabled (*.docm) file? If not, then the macros would not be saved in the file -- but you would have seen a warning message about that.

On the computer with Word 2010, make sure the document is stored in a folder that's marked as a Trusted Location (or store it in any folder and then make that folder trusted). To do that, go to File > Options > Trust Center > Trust Center Settings > Trusted Locations.

There are some email servers that will prevent mailing a macro-containing document, but they usually just remove the entire attachment file -- removing just the macros while keeping the rest of the document is far outside what email programs should attempt to do. At any rate, sharing the drive would have gotten around that.

As a last resort (because it's more work than should be necessary), you can export the macros' code to a *.bas file in Word 2013 and then import that file into a new document in Word 2010. In the macro editor, right-click the module containing the macros, choose Export, and give a name and location for the file. Copy the file (which is plain text) to the other computer and reverse the process, choosing Import in the right-click menu.

There are other things you can try without the registry edit.
Empty your temporary cache, in Internet Explorer, Tools, Internet Options,
Delete, you need to delete the temporary files.
 
Open the Organizer from Start, Microsoft Office Tools, Microsoft Clip Organizer.

If your Organizer becomes corrupt you will have numerous error messages. The
best solution here is to create a new catalog.
Find the file Mstore10.mgc, rename it to Mstore10.old. It is in a folder similar
to this:
C:\Users\<<user>>\AppData\Roaming\Microsoft\Clip Organizer
If you are using Windows XP:
C:\Documents and Settings\<<user>>\Application Data\Microsoft\Clip Organizer
 
The Organizer will create a new catalog file the next time it is opened.
