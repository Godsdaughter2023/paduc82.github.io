john - project say:
Nathalie,
Okay, first set up your file by entering your "projects" using manual mode. That will allow you to set both the start date and finish date without constraints. Once you have you base file set, run the following macro. It will set up the subtasks under each of your project tasks. Note: the code assumes a standard 8 hour work day and 20 day month.

Sub MakeSubTsk()
Dim t As Task
Dim NumSum As Integer, NumSub As Integer
Dim i As Integer, j As Integer
Set area = ActiveProject.Tasks
NumSum = area.Count
j = 1
'first, set object as the set of existing "projects"
For Each t In area
    t.Manual = False
    'find out how integer number of month "fit" into the
    '   the existing project time span
    NumSub = Int(t.Duration / 9600)
    'second, add tasks under each existing project
    For i = NumSub To 1 Step -1
        t.OutlineChildren.Add Name:="subtask " & i, before:=t.ID + 1
    Next i
    j = j + 1
    If j = NumSum + 1 Then Exit For
Next t
'third, convert added tasks into subtasks, set their duration
'   to one month and link each subtask
For Each t In ActiveProject.Tasks
    If t.Flag1 = False Then
        t.Duration = 9600
        t.OutlineLevel = 2
        If ActiveProject.Tasks(t.ID - 1).Flag1 = False Then
            t.Predecessors = t.ID - 1
        End If
    End If
Next t
    
End Sub

stephenbb say: Figured it out:



Dim d As Variant, e As Variant, f, g As Integer

d = Array("INTERNET-LOGINS.xlsm", "Book3.xlsm")
e = Array((1), (2), (3))

For f = LBound(d) To UBound(d)
Application.Workbooks(d(f)).Activate

For g = LBound(e) To UBound(e) 'just first three sheets
If ActiveWorkbook.Sheets.Count < 3 Then End 'if few than 3 sheets then end
Sheets(e(g)).Select
Cells(1, 1).Value = "hello"
Next

Next
