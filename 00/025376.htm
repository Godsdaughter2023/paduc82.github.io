Here's how you'd go about it:

Sub Test()
    Dim oSh As Shape
    Dim oSl As Slide

    ' Look at each shape on each slide in the active pres
    For Each oSl In ActivePresentation.Slides
        For Each oSh In oSl.Shapes
            ' Is the shape a control?
            If oSh.Type = msoOLEControlObject Then
                ' Is it a checkbox?
                If InStr(UCase(oSh.OLEFormat.ProgID), "FORMS.CHECKBOX") > 0 Then
                    ' BINGO, found what we're after, so ...

                    ' to toggle the value
                    oSh.OLEFormat.Object.Value = Not (oSh.OLEFormat.Object.Value)
                    ' or simply assign it the value you want:
                    ' oSh.OLEFormat.Object.Value = True
                    ' oSh.OLEFormat.Object.Value = False

                End If
            End If
        Next    ' Shape
    Next    ' Slide
End Sub

(watch out for line breaks)
