Hi,

 

Here in the UK we don't use this but I assume where you are the keypad on the phone is the same i.e. like below so an A,B or C becomes 2 etc. Here's a user defined function that will do it. ALT+F11 to open VB editor, right click 'ThisWorkbook' and inser modyle and paste the code below in on the right. Close VB editor.

 

Back on the worksheet call with

 

=AlphaNum(A1)

 

Where a1 is a cell with this Alpha number in

 

Function AlphaNum(rng As String) As String
Dim x As Long
For x = 1 To Len(rng)
If UCase(Mid(rng, x, 1)) Like "[A-C]" Then
   rng = Replace(rng, Mid(rng, x, 1), 2)
ElseIf UCase(Mid(rng, x, 1)) Like "[D-F]" Then
   rng = Replace(rng, Mid(rng, x, 1), 3)
ElseIf UCase(Mid(rng, x, 1)) Like "[G-I]" Then
   rng = Replace(rng, Mid(rng, x, 1), 4)
ElseIf UCase(Mid(rng, x, 1)) Like "[J-L]" Then
   rng = Replace(rng, Mid(rng, x, 1), 5)
ElseIf UCase(Mid(rng, x, 1)) Like "[M-O]" Then
   rng = Replace(rng, Mid(rng, x, 1), 6)
ElseIf UCase(Mid(rng, x, 1)) Like "[P-S]" Then
   rng = Replace(rng, Mid(rng, x, 1), 7)
ElseIf UCase(Mid(rng, x, 1)) Like "[T-V]" Then
   rng = Replace(rng, Mid(rng, x, 1), 8)
ElseIf UCase(Mid(rng, x, 1)) Like "[W-Z]" Then
   rng = Replace(rng, Mid(rng, x, 1), 9)
End If
Next
AlphaNum = rng
End Function

Mike has beaten me too it, but here is my version of a UDF:

Function AlphaPhone(str As String)
    Dim i As Long
    For i = 1 To Len(str)
        Select Case UCase(Mid(str, i, 1))
            Case "A" To "C": Mid(str, i, 1) = 2
            Case "D" To "F": Mid(str, i, 1) = 3
            Case "G" To "I": Mid(str, i, 1) = 4
            Case "J" To "L": Mid(str, i, 1) = 5
            Case "M" To "O": Mid(str, i, 1) = 6
            Case "P" To "S": Mid(str, i, 1) = 7
            Case "T" To "V": Mid(str, i, 1) = 8
            Case "W" To "Z": Mid(str, i, 1) = 9
        End Select
    Next i
    AlphaPhone = str
End Function
