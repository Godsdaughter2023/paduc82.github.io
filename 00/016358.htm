sfc /scannow
Dism /Online /cleanup-image /restorehealth

4) To reset corrupted update database Copy following script to notepad and save it WUreset.bat, open CMD as administrator and run WUreset.bat

@ECHO OFF

echo.
PAUSE
echo.
net stop wuauserv
net stop CryptSvc
net stop BITS
net stop msiserver
ren %windir%system32\catroot2 catroot2.old
ren %windir%Software\Distribution sold.old
net start wuauserv
net start cryptSvc
net start bits
net start msiserver
echo.
echo Task completed successfully.
echo.

How do I convert a number into its word format in MS WORD - Microsoft Community

fazi67 say:

How do I convert a number into its word format in MS WORD

I have the below sentence where i have to type in the number format and i would like to have the word format next to it.

Today I transferred AED 1,200,345.78 , i.e One Million Two Hundred Thousand Three Hundred Forty Five Dirhams and Seventy Eight Fils

 

When I type in the amount of 1,200,345.78, I would like to have its word representation automatically next to it....as seen in the underlined section above.

graham mayor say:
The compiler errors suggest that you have not copied correctly. There are four functions and a sub with some text between them. Obviously you don't want the text included and if you have added a macro name as shown on the web page, you would get

Sub NumbertoWord()
End Sub

in the module. You need to replace this with the code listing.

The listing assumed that the macro and the functions would be in separate modules, but they can be in the same one with the following changes. The following is ALL that should be in the module.

Option Explicit
Sub AddNumberAsText()
Dim oRng As Range
Dim strNum As String
Dim strNumText As String
Const strList As String = "0123456789,."
    Set oRng = Selection.Range
    oRng.End = oRng.Start
    oRng.MoveStartWhile Cset:=strList, Count:=wdBackward
    oRng.MoveEndWhile Cset:=strList
    strNum = Replace(oRng.Text, Chr(44), "")
    strNumText = Replace(SpellNumber(strNum), "  ", " ")
    oRng.InsertBefore "AED "
    oRng.Collapse wdCollapseEnd
    oRng.Text = ", i.e. " & strNumText
End Sub

'****************
' Main Function *
'****************
Function SpellNumber(ByVal MyNumber)
Dim Dirhams, Fils, Temp
Dim DecimalPlace, Count

    ReDim Place(9) As String
    Place(2) = " Thousand "
    Place(3) = " Million "
    Place(4) = " Billion "
    Place(5) = " Trillion "

    ' String representation of amount
    MyNumber = Trim(str(MyNumber))

    ' Position of decimal place 0 if none
    DecimalPlace = InStr(MyNumber, ".")
    'Convert Fils and set MyNumber to Dirham amount
    If DecimalPlace > 0 Then
        Fils = GetTens(Left(Mid(MyNumber, DecimalPlace + 1) & "00", 2))
        MyNumber = Trim(Left(MyNumber, DecimalPlace - 1))
    End If

    Count = 1
    Do While MyNumber <> ""
        Temp = GetHundreds(Right(MyNumber, 3))
        If Temp <> "" Then Dirhams = Temp & Place(Count) & Dirhams
        If Len(MyNumber) > 3 Then
            MyNumber = Left(MyNumber, Len(MyNumber) - 3)
        Else
            MyNumber = ""
        End If
        Count = Count + 1
    Loop

    Select Case Dirhams
        Case ""
            Dirhams = "No Dirhams"
        Case "One"
            Dirhams = "One Dirham"
        Case Else
            Dirhams = Dirhams & " Dirhams"
    End Select

    Select Case Fils
        Case ""
            Fils = " and No Fils"
        Case "One"
            Fils = " and One Fil"
        Case Else
            Fils = " and " & Fils & " Fils"
    End Select

    SpellNumber = Dirhams & Fils
End Function
 
'*******************************************
' Converts a number from 100-999 into text *
'*******************************************
Function GetHundreds(ByVal MyNumber)
    Dim Result As String
 
    If Val(MyNumber) = 0 Then Exit Function
    MyNumber = Right("000" & MyNumber, 3)
 
    'Convert the hundreds place
    If Mid(MyNumber, 1, 1) <> "0" Then
        Result = GetDigit(Mid(MyNumber, 1, 1)) & " Hundred "
    End If
 
    'Convert the tens and ones place
    If Mid(MyNumber, 2, 1) <> "0" Then
        Result = Result & GetTens(Mid(MyNumber, 2))
    Else
        Result = Result & GetDigit(Mid(MyNumber, 3))
    End If
 
    GetHundreds = Result
End Function
 
'*********************************************
' Converts a number from 10 to 99 into text. *
'*********************************************
Function GetTens(TensText)
    Dim Result As String
 
    Result = ""           'null out the temporary function value
    If Val(Left(TensText, 1)) = 1 Then   ' If value between 10-19
        Select Case Val(TensText)
            Case 10: Result = "Ten"
            Case 11: Result = "Eleven"
            Case 12: Result = "Twelve"
            Case 13: Result = "Thirteen"
            Case 14: Result = "Fourteen"
            Case 15: Result = "Fifteen"
            Case 16: Result = "Sixteen"
            Case 17: Result = "Seventeen"
            Case 18: Result = "Eighteen"
            Case 19: Result = "Nineteen"
            Case Else
        End Select
      Else                                 ' If value between 20-99
        Select Case Val(Left(TensText, 1))
            Case 2: Result = "Twenty "
            Case 3: Result = "Thirty "
            Case 4: Result = "Forty "
            Case 5: Result = "Fifty "
            Case 6: Result = "Sixty "
            Case 7: Result = "Seventy "
            Case 8: Result = "Eighty "
            Case 9: Result = "Ninety "
            Case Else
        End Select
         Result = Result & GetDigit _
            (Right(TensText, 1))  'Retrieve ones place
      End If
      GetTens = Result
   End Function
 
'*******************************************
' Converts a number from 1 to 9 into text. *
'*******************************************
Function GetDigit(Digit)
    Select Case Val(Digit)
        Case 1: GetDigit = "One"
        Case 2: GetDigit = "Two"
        Case 3: GetDigit = "Three"
        Case 4: GetDigit = "Four"
        Case 5: GetDigit = "Five"
        Case 6: GetDigit = "Six"
        Case 7: GetDigit = "Seven"
        Case 8: GetDigit = "Eight"
        Case 9: GetDigit = "Nine"
        Case Else: GetDigit = ""
    End Select
End Function
