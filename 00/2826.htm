You either need to do this slide by slide or use a little vba code

Sub zap_Ani()
Dim osld As Slide
Dim a As Integer
For Each osld In ActivePresentation.Slides
For a = osld.TimeLine.MainSequence.Count To 1 Step -1

Next a
Next osld
End Sub
