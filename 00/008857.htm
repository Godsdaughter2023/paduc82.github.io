Sub Print_Noncontiguous()
    Dim Destrange As Range
    Dim Smallrng As Range
    Dim Newsh As Worksheet
    Dim Ash As Worksheet
    Dim lr As Long

    Application.ScreenUpdating = False

    Set Ash = ActiveSheet
    Set Newsh = Worksheets.Add
    Ash.Select

    lr = 1

     For Each Smallrng In Ash.Range("A1:J161 , A204: J245 , A386: J428 , A505: J1130").Areas

           'Add more Ranges as required

        Smallrng.Copy
        Set Destrange = Newsh.Cells(lr, 1)

        Destrange.PasteSpecial xlPasteFormats
        lr = lr + Smallrng.Rows.Count
    Next Smallrng

    Newsh.Columns.AutoFit
    Newsh.PrintPreview   'Change PrintPreview to PrintOut when ready

    Application.DisplayAlerts = False
    Newsh.Delete    'Optional
    Application.DisplayAlerts = True

    Application.ScreenUpdating = True

End Sub
