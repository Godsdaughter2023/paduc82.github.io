Sub Export_MultiCriteria()
'July 08, 2014
'multi criteria
'export specific columns
'expected results in existing sheet
Const N As Long = 2   '<< headers in row 2, change
Const myCol As String = "D"   '<< criteria in column D, change

Const shN As String = "OS tracker"   '<< target sheet name, change
Dim v As Variant, vC As Variant, vCrit As Variant
v = Array("Sheet1", "Sheet2", "Sheet3", "Sheet4", "Sheet5")  '<< sheet names, change

vC = Array("A", "C", "D", "K")   '<< export specific columns (20), change/add new columns

vCrit = Array("y", "Yes", "abc")   '<< criteria [y - Yes - abc], change

Dim r As Long, L As Long, x As Long, x1 As Long, x2 As Long, t As Long
Application.ScreenUpdating = False
L = Sheets(shN).Cells(Rows.Count, "A").End(xlUp).Row
If L < N + 1 Then
'do nothing
Else
Sheets(shN).Range("A" & N + 1).Resize(L, UBound(vC) + 1).ClearContents
End If
t = N + 1
For x = 0 To UBound(v)
For x2 = 0 To UBound(vCrit)
If Not IsError(Application.Match(vCrit(x2), Sheets(v(x)).Range(myCol & ":" & myCol), 0)) Then
With Sheets(v(x))
.AutoFilterMode = False
r = .Cells(Rows.Count, myCol).End(xlUp).Row
.Range(myCol & N & ":" & myCol & r).AutoFilter field:=1, Criteria1:=vCrit, Operator:=xlFilterValues
For x1 = 0 To UBound(vC)
.Range(.Cells(N + 1, vC(x1)), .Cells(r, vC(x1))).SpecialCells(xlCellTypeVisible).Copy
Sheets(shN).Cells(t, x1 + 1).PasteSpecial xlPasteValues
Application.CutCopyMode = False
Next
.AutoFilterMode = False
End With
t = Sheets(shN).Cells(Rows.Count, "A").End(xlUp).Row + 1
End If
GoTo nnext
Next
nnext:
Next
Application.ScreenUpdating = True
End Sub

pamcaswell say: This method works when figures are inline with text.

First revise the Caption paragraph style to look like you want it to and apply it to your caption and additional information.  With the cursor in the cursor in the Caption style paragraph, create a new style for the additional information called, say, Legend (a la Words into Type, 3rd ed.).  It will be based on the Caption style. 

Put a return (Â¶) between the caption and legend, *and apply the Legend style to the additional info*. With the cursor in the caption paragraph, press Ctrl+Alt+Enter  to apply the style separator to the paragraph mark between the two paragraphs. The style separator is special hidden formatting that Word applies to the next paragraph mark after the cursor position. The two parts of your figure title will end and start on the same line.

With the cursor at the beginning of the first caption in your document, insert a caption label & number.  Add the punctuation and spacing you want, and apply italics as direct formatting to the label and number only. Select the label, number, punctuation, and space before the caption text and go to  Insert tab > Quick Parts > AutoText. Click
When you generate the  table of figures, you will get the italic caption label and caption text only.


My experience with W2007 is that the style separator sometimes spaces the lines oddly when used in a text box or table. (The last time, it was only two or three out of about 20 captions.)
