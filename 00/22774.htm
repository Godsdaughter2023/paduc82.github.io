hansv mvp say:

Here is a more flexible version. You can change the colors in the Colors array, of course.

Sub HighlightWords()
    Dim Word As Range
    Dim WordCollection(2) As String
    Dim Colors(2) As WdColorIndex
    Dim CurrentColor As WdColorIndex
    Dim i As Long
    'Define list.
    'If you add or delete, change value above in Dim statement.
    WordCollection(0) = "down by"
    WordCollection(1) = "challenging"
    WordCollection(2) = "difficult"
    ' Colors
    Colors(0) = wdRed
    Colors(1) = wdGreen
    Colors(2) = wdBlue
    ' Keep current highlight color
    CurrentColor = Options.DefaultHighlightColorIndex
    ' Speed up execution by hiding what we're doing
    Application.ScreenUpdating = False
    With ActiveDocument.Content.Find
        'Clear existing formatting and settings in Find feature.
        .ClearFormatting
        .Replacement.ClearFormatting
        .Replacement.Text = ""
        .Forward = True
        .Wrap = wdFindContinue
        .Format = True
        .MatchCase = False
        .MatchWholeWord = True
        .MatchWildcards = False
        .MatchSoundsLike = False
        .MatchAllWordForms = False
        'Set highlight to replace setting.
        .Replacement.Highlight = True
        'Cycle through document and find words in collection.
        For i = 0 To 2
            'Set highlight color.
            Options.DefaultHighlightColorIndex = Colors(i)
            ' Replace all
            .Execute FindText:=WordCollection(i), Replace:=wdReplaceAll
        Next i
    End With
    ' Show the result
    Application.ScreenUpdating = True
    ' Restore highlight color
    Options.DefaultHighlightColorIndex = CurrentColor
End Sub
