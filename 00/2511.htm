Set rst = CurrentDb.OpenRecordset("t_ExcelExport_auto")

    With CurrentDb

           Set rst = .OpenRecordset("t_ExcelExport_auto")

           With rst

                    .MoveNext

                 .Close

            End With  ' rst

Public Sub UnprotectMe(Optional sht As Worksheet)
If IsMissing(sht) Then
    ActiveSheet.Unprotect Password:=pwd
Else
    sht.Unprotect Password:=pwd
End If
End Sub

'''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''

Public Sub ProtectMe(Optional sht As Worksheet)
If IsMissing(sht) Then
    ActiveSheet.Protect Password:=pwd, DrawingObjects:=True, Contents:=True, Scenarios:=True _
        , AllowFormattingColumns:=True, AllowSorting:=True, AllowFiltering:=True _
        , AllowUsingPivotTables:=True
       
ElseIf sht Is Nothing Then
    ActiveSheet.Protect Password:=pwd, DrawingObjects:=True, Contents:=True, Scenarios:=True _
        , AllowFormattingColumns:=True, AllowSorting:=True, AllowFiltering:=True _
        , AllowUsingPivotTables:=True
Else
    sht.Protect Password:=pwd, DrawingObjects:=True, Contents:=True, Scenarios:=True _
        , AllowFormattingColumns:=True, AllowSorting:=True, AllowFiltering:=True _
        , AllowUsingPivotTables:=True
End If
End Sub
