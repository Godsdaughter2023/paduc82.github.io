Change the sub to a function, move FName and the Range into the header as argument and 99% of the work is done.

Option Explicit

Sub SaveAsCSV()
  Dim MyPath As String
  Dim MyFileName As String
  'The path and file names:
  MyPath = "C:\test\"
  MyFileName = "File 01" & Format(Date, "ddmmyy")
  'Makes sure the path name ends with "\":
  If Not Right(MyPath, 1) = "\" Then MyPath = MyPath & "\"
  'Makes sure the filename ends with ".csv"
  If Not Right(MyFileName, 4) = ".csv" Then MyFileName = MyFileName & ".csv"

  If Not SaveAsCSVPrim(MyPath & MyFileName, Range("a1").CurrentRegion) Then
    MsgBox "Failed"
  Else
    MsgBox "Success"
  End If
End Sub

Private Function SaveAsCSVPrim(ByVal FName As String, ByVal Where As Range) As Boolean
  Dim fso As Object 'Scripting.FileSystemObject
  Dim ts As Object 'Scripting.TextStream
  Dim Data, Temp
  Dim i As Long, j As Long

  On Error GoTo ExitPoint

  'Get the data
  Data = Where.Value
  If Not IsArray(Data) Then
    ReDim Data(1 To 1, 1 To 1)
    Data(1, 1) = Where.Value
  End If

  'Create the file
  Set fso = CreateObject("Scripting.FileSystemObject")
  Set ts = fso.CreateTextFile(FName)
  ReDim Temp(1 To UBound(Data))
  For i = 1 To UBound(Data)
    Temp(i) = Join(ArrRow(Data, i), ",")
  Next
  ts.Write Join(Temp, vbCrLf)
  ts.Close
ExitPoint:
  SaveAsCSVPrim = True
End Function

Private Property Get ArrRow(Arr, ByVal Nth As Long) As Variant
  'Return the Nth row of a 2 dimensional array as one dimensional array
  Dim Result(), i As Long
  On Error GoTo ExitPoint
  ReDim Result(LBound(Arr, 2) To UBound(Arr, 2))
  For i = LBound(Arr, 2) To UBound(Arr, 2)
    Result(i) = Arr(Nth, i)
  Next
ExitPoint:
  ArrRow = Result
End Property
