andreas killer say:

The code above has a glitch, Tristate must be the 4th argument, not the 3rd.

I would like to show a comprehensible example, therefor we need at first some Unicode chars.

- Make a new file

- Open this website and copy an Arabic headline:

  http://www.bbc.co.uk/arabic/

- Paste the headline into cell A1 (the cell shows the UniCode chars as in the website)

- Open the VBA editor

- Add a regular module

- Copy the code below my name and paste it into that module

- Customize the path\filename in both macros to your needs

- Execute Example_WriteUnicodeTextfile to create the text file

- You can open that text file with Notepad to be sure it contains the Arabic text

- Clear cell A1

- Run Example_ReadUnicodeTextfile to read the contents back

BTW, you can create the text file with Notepad too, but you have to change the encoding from ANSI to Unicode in the SaveAs dialog.

Andreas.

Option Explicit

'Imported from Scripting.Tristate:
Enum Tristate
  TristateFalse = 0 'Open file as ASCII
  TristateTrue = -1 'Open file as Unicode
  TristateUseDefault = -2 'Open file as specified in your OS
End Enum

Sub Example_WriteUnicodeTextfile()
  Dim fso As Object 'Scripting.FileSystemObject
  Dim ts As Object 'Scripting.TextStream
  Dim Contents As String
 
  'Get the contents from a cell
  Contents = Range("A1")
 
  'Create the object
  Set fso = CreateObject("Scripting.FileSystemObject")
  'Create the file in Unicode mode
  Set ts = fso.CreateTextFile("c:\test.txt", Unicode:=True)
  'Write the contents
  ts.Write Contents
  'Close it
  ts.Close
End Sub

Sub Example_ReadUnicodeTextfile()
  Dim fso As Object 'Scripting.FileSystemObject
  Dim ts As Object 'Scripting.TextStream
  Dim Contents As String
 
  'Create the object
  Set fso = CreateObject("Scripting.FileSystemObject")
  'Open the file in Unicode mode
  Set ts = fso.OpenTextFile("c:\test.txt", Format:=TristateTrue)
  'Get the contents
  Contents = ts.ReadAll
  'Close it
  ts.Close
 
  'Write the contents into a cell
  Range("A1") = Contents
End Sub

                        net start msiserver
