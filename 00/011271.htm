<Quote Calendar control (mscal.ocx)

The Microsoft Calendar control (mscal.ocx) is not available in Access 2010. An alternative is to use the date picker control in Access 2010. You can use this formulas to call the code below from the sheet:
H47:  =NoOfTubes(B9:AF45)
S48:  =NoOfTubes(B9:AF45,0)
S49:  =NoOfTubes(B9:AF45,255)

You can use this formula to get the correct number of tubes in each row:
AG9:AG45:  {=TubesAtRow(B9:AF45)}

Note:
  If you change the fill color of a circle you must press CTRL-ALT-F9 to update the formulas!

Copy the code below into a regular module.

I recommend to call the macro AdjustCircles to adjust the cells and shapes, because some of the circles are not in place and looks like ovals and have no fill.

Andreas.


Option Explicit

Function NoOfTubes(ByVal Where As Range, Optional ByVal RGB) As Long
  Dim Sh As Shape
  'Search for ovals
  For Each Sh In Where.Parent.Shapes
    If Sh.Type = msoAutoShape Then
      If Sh.AutoShapeType = msoShapeOval Then
        'Count if the left upper edge of the shape intersect with Where
        If Not Intersect(Where, Sh.TopLeftCell) Is Nothing Then
          'Special color?
          If IsMissing(RGB) Then
            'Count all
            NoOfTubes = NoOfTubes + 1
          Else
            'Count this color only
            If Sh.Fill.ForeColor.RGB = RGB Then
              NoOfTubes = NoOfTubes + 1
            End If
          End If
        End If
      End If
    End If
  Next
End Function

Function TubesAtRow(ByVal Where As Range) As Variant
  Dim Sh As Shape
  Dim Result()
  Dim i As Long
  ReDim Result(1 To Where.Rows.Count, 1 To 1)
  'Search for ovals
  For Each Sh In Where.Parent.Shapes
    If Sh.Type = msoAutoShape Then
      If Sh.AutoShapeType = msoShapeOval Then
        'Count if the left upper edge of the shape intersect with Where
        If Not Intersect(Where, Sh.TopLeftCell) Is Nothing Then
          i = Sh.TopLeftCell.Row - Where.Row + 1
          Result(i, 1) = Result(i, 1) + 1
        End If
      End If
    End If
  Next
  TubesAtRow = Result
End Function

Sub AdjustCircles()
  Dim Sh As Shape
  Dim Where As Range
  Set Where = Range("B9:AF45")
  'Make quadratic cells
  Where.RowHeight = Where(1, 1).RowHeight
  Where.ColumnWidth = Where(1, 1).RowHeight * 1.25 * Where(1, 1).ColumnWidth / Where(1, 1).Width
  'Search for ovals
  For Each Sh In Where.Parent.Shapes
    If Sh.Type = msoAutoShape Then
      If Sh.AutoShapeType = msoShapeOval Then
        If Not Intersect(Where, Sh.TopLeftCell) Is Nothing Then
          Sh.Width = 12
          Sh.Height = 12
          If Sh.TopLeftCell.Column <> Sh.BottomRightCell.Column Then
            'Move to edge
            Sh.Left = Sh.TopLeftCell.Left + Sh.TopLeftCell.Width - Sh.Width / 2
          Else
            'Move to middle
            Sh.Left = Sh.TopLeftCell.Left + Sh.TopLeftCell.Width / 2 - Sh.Width / 2
          End If
          Sh.Top = Sh.TopLeftCell.Top + Sh.TopLeftCell.Height - Sh.Height / 2
          Select Case Sh.Fill.ForeColor.RGB
            Case vbBlack, vbRed
            Case Else
              'Change to white color
              Sh.Fill.ForeColor.RGB = vbWhite
              Sh.Fill.Visible = msoTrue
          End Select
          'Dont resize
          Sh.Placement = xlMove
        End If
      End If
    End If
  Next
End Sub
