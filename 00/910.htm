dana.d say:

Here's a simplified version of a program:

Sub MainProg()
    Dim v1, v2, M
    v1 = [C3:C5]
    v2 = [D3:D4]
    M = Outer(v1, v2)
    [E3].Resize(UBound(M), 2) = M
End Sub

Function Outer(x, y)
'// A simplified Outer Function
    Dim M, j, k, p
    ReDim M(1 To UBound(x) * UBound(y), 1 To 2)
    For j = 1 To UBound(x)
        For k = 1 To UBound(y)
            p = p + 1
            M(p, 1) = x(j, 1)
            M(p, 2) = y(k, 1)
        Next k
    Next j
    Outer = M
End Function
