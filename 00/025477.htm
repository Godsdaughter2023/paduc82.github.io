Sub Test()
  MoveResizeButton ActiveSheet.Shapes("Button 1"), _
    Range("A3"), Range("B10")
End Sub

Sub MoveResizeButton(ByVal Button As Shape, _
    ByVal TopLeftCell As Range, ByVal BottomRightCell As Range)
  Const BorderSize = 3
  'Maybe the button does not exists?
  On Error GoTo ExitPoint
  With Button
    .Top = TopLeftCell.Top + BorderSize
    .Left = TopLeftCell.Left + BorderSize
    .Width = BottomRightCell.Left - TopLeftCell.Left + _
      BottomRightCell.Width - 2 * BorderSize
    .Height = BottomRightCell.Top - TopLeftCell.Top + _
      BottomRightCell.Height - 2 * BorderSize
  End With
ExitPoint:
End Sub
