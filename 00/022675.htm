macropod say:

Depending on how much of the description spills over to the second line, you could use Word's 'Fit text' function to compress the description to fit on just one line, thereby allowing you to consistently format for a single-lined description. For example:

Sub MailMergeToDoc()
Application.ScreenUpdating = False
Dim Tbl As Table, Cll As Cell, sCllWdth As Single, sParWdth As Single
With ActiveDocument
  With .Tables(1)
    For Each Cll In .Range.Cells
      Cll.WordWrap = False
    Next
    With .Cell(1, 1)
      sCllWdth = .Width
      With .Range.Paragraphs(1)
        sCllWdth = sCllWdth - .LeftIndent - .RightIndent
      End With
    End With
  End With
  .MailMerge.Execute
End With
With ActiveDocument
  For Each Tbl In .Tables
    For Each Cll In Tbl.Range.Cells
      If Len(Cll.Range) > 2 Then
        With Cll.Range.Paragraphs(1).Range
          If .Characters.Last.Previous.Information(wdVerticalPositionRelativeToPage) <> _
            .Characters.First.Information(wdVerticalPositionRelativeToPage) Then
            .FitTextWidth = sCllWdth
          End If
        End With
      End If
    Next
  Next
End With
Application.ScreenUpdating = True
End Sub

Alternatively, if the line following the description can be deleted when the description spans two lines, you could use:

Sub MailMergeToDoc()
Application.ScreenUpdating = False
Dim Tbl As Table, Cll As Cell, Par As Paragraph
With ActiveDocument.MailMerge.Execute
With ActiveDocument
  For Each Tbl In .Tables
    For Each Cll In Tbl.Range.Cells
      If Len(Cll.Range) > 2 Then
        With Cll.Range.Paragraphs(1).Range
          If .Characters.Last.Previous.Information(wdVerticalPositionRelativeToPage) <> _
            .Characters.First.Information(wdVerticalPositionRelativeToPage) Then
            .Characters.Last.Delete
          End If
        End With
      End If
    Next
  Next
End With
Application.ScreenUpdating = True
End Sub

Note: Being named MailMergeToDoc, either macro (of which you doucment can only contain one) will intercept Word's Finish & Merge>Edit Individual Documents command, making the process entirely transparent.


For PC macro installation & usage instructions, see: http://www.gmayor.com/installing_macro.htm
