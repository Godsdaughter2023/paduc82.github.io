rich007a say:

Try this:

    Dim cht As Chart
    Dim ser As Series
   
    Set cht = ActiveChart
    Set ser = cht.SeriesCollection(1)
    ser.Format.Line.Visible = msoFalse
    ser.Format.Line.Visible = msoTrue
    ser.Format.Line.ForeColor.RGB = RGB(255, 50, 0)
