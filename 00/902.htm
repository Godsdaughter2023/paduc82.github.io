Sub AutoSum()
Dim cel1 As String, cel2 As String
cel1 = ActiveCell.Offset(-1, 0).End(xlUp).Address
cel2 = ActiveCell.Offset(-1).Address
ActiveCell.Value = "=sum(" & (cel1) & ":" & (cel2) & ")"
End Sub
