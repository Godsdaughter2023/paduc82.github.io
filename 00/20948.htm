tom ogilvy say:

Here is a revision that truncates the name to 30 characters:  (tested and worked for me)

[ of course the sheet name must be unique in the first 30 characters ]

Sub NameSheets()
Dim sh As Worksheet, sh1 As Worksheet, rw As Long, sName As String
On Error Resume Next
  Set sh1 = Worksheets("Summary")
On Error GoTo 0
  If sh1 Is Nothing Then
      Worksheets.Add Before:=Worksheets(1)
      Set sh1 = ActiveSheet
      sh1.Name = "Summary"
  Else
      sh1.Cells.Clear
  End If
rw = 2
For Each sh In Worksheets
 If sh.Name <> sh1.Name Then

  sName = sh.Range("A2").Value

  If Len(sName) > 30 Then sName = Left(sName, 30)

  sh.Name = sName
  If InStr(1, sName, " ", vbTextCompare) = 0 Then
      sh1.Cells(rw, 1).Hyperlinks.Add Anchor:=sh1.Cells(rw, 1), Address:="", SubAddress:= _
        sName & "!A2", TextToDisplay:=sh.Name
  Else
      sh1.Cells(rw, 1).Hyperlinks.Add Anchor:=sh1.Cells(rw, 1), Address:="", SubAddress:= _
       "'" & sName & "'!A2", TextToDisplay:=sh.Name
  End If
  sh1.Cells(rw, 2).Value = sh.Range("A5").Value
  rw = rw + 1
 End If
Next
End Sub
