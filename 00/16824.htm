Ignore my first post I missed the bit about the first 2 and last worksheets. Try this instead. It will toggle the protection on all sheets except the first 2 and the last one.

Sub sonic2()
Dim sh As Worksheet
For Each sh In ThisWorkbook.Worksheets
If sh.Index > 2 And sh.Index < Sheets.Count Then
If sh.ProtectContents = True Then
    sh.Unprotect "password"
Else
    sh.Protect "password"
End If
End If
Next sh
End Sub
