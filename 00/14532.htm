Yes, you can use the getVisible callback to set the visibility of a tab based on the current user's username:
ribbonX:
#tab
 id="tabMyTab"
 label="Tab for me only"
 getVisible="mod.getVis"#
 ...
#/tab#
(again, replace # with < or >, and replace mod with your module's name)

VBA:
'Callback for tabMyTab getVisible
Sub getVis(control As IRibbonControl, ByRef returnedVal)
    Dim UserName As String
    UserName = Environ("USERNAME")
    If UserName = "MyName" Then
        returnedVal = "true"
    Else
        returnedVal = "false"
    End If
End Sub
