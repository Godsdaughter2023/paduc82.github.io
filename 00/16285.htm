If I understand you right your 'Master File' has some named ranges, now you want to copy some data into that ranges and resize that ranges to fit the data?

That's easy, make a new file, select some cells and name them "Test". Then execute this macro and check the named range afterwards.

Sub Test()
  Dim R As Range
 
  'Get the range from the name
  Set R = Names("Test").RefersToRange
 
  'Increase it by 5 rows
  Set R = R.Resize(R.Rows.Count + 5, R.Columns.Count)
 
  'Change the formula in the named range to fit the new range
  Names("Test").RefersTo = "=" & R.Address(External:=True)
End Sub

Sub Test2()
  Dim Source As Name, Dest As Name
  Dim Data
 
  'Visit each named range
  For Each Dest In ThisWorkbook.Names
    'Get the same name in the opened workbook
    Set Source = GetName(Dest.Name, ActiveWorkbook)
    'Found?
    If Not Source Is Nothing Then
      'Read the values into an array
      Data = Source.RefersToRange.Value
      'Adjust the destination size
      With Dest.RefersToRange.Resize(UBound(Data), UBound(Data, 2))
        'Copy the values
        .Value = Data
        'Adjust the named range
        Dest.RefersTo = "=" & .Address(External:=True)
      End With
    End If
  Next
End Sub

Sub AddtoBoard()
  Dim vFile As Variant
  Dim wbCopyTo As Workbook
  Dim wsCopyTo As Worksheet
  Dim wbCopyFrom As Workbook
  Dim wsCopyFrom As Worksheet
  Dim Source As Range
  Dim Dest As Range
 
  Set wbCopyTo = ActiveWorkbook
  Set wsCopyTo = ActiveSheet
 
  'Find the bottom right cell of the used range
  Set Dest = RangeLastCell(wsCopyTo.Cells)
  'Go to the next row, first column
  Set Dest = Dest.Offset(1, 1 - Dest.Column)

  '-------------------------------------------------------------
  'Open file with data to be copied
  vFile = Application.GetOpenFilename("Excel Files (*.xl*)," & _
    "*.xl*", 1, "Select Excel File", "Open", False)

  'If Cancel then Exit
  If TypeName(vFile) = "Boolean" Then
    Exit Sub
  Else
    Set wbCopyFrom = Workbooks.Open(vFile)
    Set wsCopyFrom = wbCopyFrom.Worksheets(1)
  End If

  '--------------------------------------------------------------
  'Copy Range
  Set Source = wsCopyFrom.Range("A18:I42")
  Source.Copy
  Dest.PasteSpecial Paste:=xlPasteValues
  'Remove empty cells if any
  Set Dest = SpecialCells(Dest.Resize(Source.Rows.Count, Source.Columns.Count), _
    xlCellTypeBlanks)
  If Not Dest Is Nothing Then Dest.Delete shift:=xlShiftUp
  'Close file that was opened
  wbCopyFrom.Close SaveChanges:=False
End Sub

Private Function SpecialCells(ByVal R As Range, ByVal Typ As XlCellType, _
    Optional ByVal Value As XlSpecialCellsValue = &H17) As Range
  'Avoid the SpecialCells-BUG to return all cells from the current region
  On Error Resume Next
  Select Case Typ
    Case xlCellTypeConstants, xlCellTypeFormulas
      Set SpecialCells = Intersect(R, R.SpecialCells(Typ, Value))
    Case Else
      Set SpecialCells = Intersect(R, R.SpecialCells(Typ))
  End Select
End Function

Private Function RangeLastCell(Optional Bereich As Range) As Range
  'Liefert die letzte verwendete Zelle in Bereich
  Dim R As Range, C As Range
  If Bereich Is Nothing Then Set Bereich = ActiveSheet.Cells
  Set R = Bereich.Cells(Bereich.Rows.Count, Bereich.Columns.Count)
  If IsEmpty(R) And Not R.Address = Bereich.Cells(1, 1).Address Then
    Set C = Bereich.Find("*", After:=R, SearchOrder:=xlByColumns, _
      SearchDirection:=xlPrevious)
    If C Is Nothing Then
      Set RangeLastCell = Bereich(1, 1)
    Else
      Set R = Bereich.Find("*", After:=R, SearchOrder:=xlByRows, _
        SearchDirection:=xlPrevious)
      Set RangeLastCell = Bereich.Cells(R.Row - Bereich.Row + 1, C.Column - Bereich _
        .Column + 1)
    End If
  Else
    Set RangeLastCell = R
  End If
End Function
