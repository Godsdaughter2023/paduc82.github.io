Hi,

try this function. You can call it also in a query.

Option Compare Database
Option Explicit

Function FindAcronyms(yourWord As String)
Dim I As Integer
Dim ctr As Integer

FindAcronyms = Null
For I = 1 To Len(yourWord)
    If Asc(Mid(yourWord, I, 1)) <= 90 And _
       Asc(Mid(yourWord, I, 1)) > = 65 Then
       If ctr > 0 Then
          FindAcronyms = FindAcronyms & Mid(yourWord, I - 1, 1)
       End If
       ctr = ctr + 1
    Else
       If ctr > 1 Then
          FindAcronyms = FindAcronyms & Mid(yourWord, I - 1, 1) & ", "
       End If
       ctr = 0
    End If
Next

If ctr > 1 Then
   FindAcronyms = FindAcronyms & Mid(yourWord, I - 1, 1)
End If

If Right(FindAcronyms, 2) = ", " Then
   FindAcronyms = Left(FindAcronyms, Len(FindAcronyms) - 2)
End If

End Function

Timestamp Corrected TS Formula
09:15 09:08 =A2-(0.000694444444444444*7)
13:20 13:13 =A3-(0.000694444444444444*7)

0.000694444444444444 = 1 minute 

or

Timestamp Corrected TS Formula
09:15 09:08 =A2-TIME(0,7,0)
13:20 13:13 =A3-TIME(0,7,0)
