Another way is to create the shapes one by one.

Andreas.

Sub Test()
  Dim S As Shape
  Dim O As Oval
  Set S = AddShapeOval(Range("P3"), 5)
  S.OLEFormat.Object.Characters.Text = "12"
End Sub

Function AddShapeOval(ByVal C As Range, ByVal Number As Integer, _
    Optional ByVal RowsInC As Integer = 3, Optional ByVal ColumnsInC As Integer = 3) As Shape
  Dim a As Integer, b As Integer
 
  On Error Resume Next
  'Try access an existing shape
  Set AddShapeOval = ActiveSheet.Shapes(C.Address(0, 0) & "-" & Number)
  If AddShapeOval Is Nothing Then
    'Create one if failed
    Set AddShapeOval = ActiveSheet.Shapes.AddShape(msoShapeOval, 0, 0, _
      C.Width / RowsInC, C.Height / ColumnsInC)
  End If
  'Calculate the position
  a = (Number - 1) Mod ColumnsInC
  b = (Number - 1) \ RowsInC
  'Move and rename it
  AddShapeOval.Left = C.Left + C.Width / 3 * a
  AddShapeOval.Top = C.Top + C.Height / 3 * b
  AddShapeOval.Name = C.Address(0, 0) & "-" & Number
End Function
