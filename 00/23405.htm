       If iMin < 15 Then

          Else
             If iMin < 30 Then

             Else

               Else
                  If iMin < 45 Then

                        oMin = 0
                     End If
                  End If
               End If
            End If
         End If
      End If
    End If
    RoundTime = TimeSerial(iHour, oMin, 0)
