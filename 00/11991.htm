tasos_k say: Hi,
i assume that:
1) your workbook has 12 sheets, at least (months: Jan,Feb, Mar, Apr,....Dec)
2) in each month sheet:
a) in row1are  headers
b) Total sum in row 33
now,
export  data from month sheet (row 33),
in to a new sheet named "ALL Months"
if so, try this code...

1) press ALT+F11 to open VB editor.

2) Insert > module and paste the code below on the right

(from Sub ExportMonths()..........End Sub)


Sub ExportMonths()

Dim newWS As Worksheet

Dim v As Variant

v = Array("Jan", "Feb", "Mar", "Apr", "May", "Jun", _

"Jul", "Aug", "Sep", "Oct", "Nov", "Dec")

Application.ScreenUpdating = False

For i = Sheets.Count To 1 Step -1

Application.DisplayAlerts = False

If Sheets(i).Name = "ALL Months" Then Sheets(i).Delete

Application.DisplayAlerts = True

Next i

Set newWS = Sheets.Add(before:=Sheets(1))

newWS.Name = "ALL Months"

newWS.[A1] = "months"

newWS.[B1] = "common carrier"

newWS.[C1] = "FEDX/UPS"

newWS.[D1] = "other"

newWS.[E1] = "Total"

For i = 0 To UBound(v)

newWS.Cells(i + 2, 1) = Sheets(v(i)).Name

Sheets(v(i)).Range("B33:E33").Copy

newWS.Cells(i + 2, 2).PasteSpecial xlValues

Next i

Application.CutCopyMode = False

newWS.[A14] = "Total"

For i = 2 To 5

newWS.Cells(14, i).FormulaR1C1 = "=SUM(R[-12]C:R[-1]C)"

Next i

newWS.Range("A1:E1,A14").Font.Bold = True

newWS.Columns("A:E").AutoFit

Application.ScreenUpdating = True

End Sub


3) Close VB editor.

....................................

5) To run the macro, press ALT+F8, 

select <ExportMonths> from the list and click the run button.
