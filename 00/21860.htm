What Word displays in a citation depends on both the referencing format (e.g. APA, Chicago) and the number of titles in your bibliography by the author concerned. Word typically only displays the title in a citation if there is more than one title by the same author in your bibliography, regardless of how many time a particular title is referenced in citations. The fact you have citations in endnotes suggest you're not using the APA format - Chicago, perhaps. Do be aware that Microsoft's implementation of the various referencing formats is poor - especially for Chicago, where what's on offer isn't even close to conforming to either of Chicago's two referencing options.

  1. Download this file
    https://dl.dropboxusercontent.com/u/35239054/FileSearch.cls
  2. Make a new Excel file
  3. Import the downloaded file into the VBA editor
  4. Paste the code below into a regular module
  5. Execute sub Main
  6. Save the file as "CombinedFile.xlsm"

Option Explicit

Sub Main()
  Dim FS As New FileSearch
  Dim File
  Dim Wb As Workbook
  Dim ThisSheet As Worksheet

  With FS
    'Start the search here
    .LookIn = "C:\"
    'For all Excel files
    .FileName = "*.xls*"
    'Everywhere
    .SearchSubFolders = True
    'Show a message
    Application.StatusBar = "Searching for files..."
    DoEvents
    'Do it
    .Execute

    'For each foudn file
    For Each File In .FoundFiles
      'Show a message
      Application.StatusBar = "Opening " & File
      DoEvents
      'Open it
      Set Wb = Workbooks.Open(File, False, True)
      'If the sheet is found
      If SheetExists("PropertyData") Then
        'Add a new here in our file
        Set ThisSheet = ThisWorkbook.Worksheets.Add
        'Write the name in A1
        ThisSheet.Range("A1") = File
        'Copy the data below
        Wb.Sheets("PropertyData").UsedRange.Copy ThisSheet.Range("A2")
      End If
      'Close the file
      Wb.Close False
    Next
  End With
End Sub

Private Function SheetExists(ByVal SheetNameOrIndex As Variant, _
    Optional ByVal Wb As Workbook = Nothing) As Boolean
  'True if sheet SheetNameOrIndex exists
  On Error Resume Next
  If Wb Is Nothing Then Set Wb = ActiveWorkbook
  SheetExists = Not Wb.Sheets(SheetNameOrIndex) Is Nothing
End Function
