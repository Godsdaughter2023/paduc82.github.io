Hi,

try this code,

data in active sheet

[update: Sep 23, 2017]

Sub Convert Data()
'Sep 14, 2017
Const sC As String = "E" ' << results in col E and to the right
Dim ws As Worksheet
Set ws = ActiveSheet
Dim r As Range
Dim L As Long, t As Long
Dim v As Variant
v = Array("PART", "DATE", "START", "STOP", "TIME", "MINS")
L = ws.Cells(Rows.Count, 1).End(xlUp).Row
ws.Cells(1, sC).Resize(L, UBound(v) + 1).ClearContents
ws.Cells(1, sC).Resize(1, UBound(v) + 1) = v
t = 2
For Each r In ws.Range("A:A").SpecialCells(xlCellTypeConstants)
If r.Value = v(0) Then
ws.Cells(t, sC).Value = r.Offset(, 1).Value 'part
'
ws.Cells(t, sC).Offset(, 1).Value = r.Offset(1, 1).Value 'date
ws.Cells(t, sC).Offset(, 1).NumberFormat = "dd/mm/yyyy"
'
ws.Cells(t, sC).Offset(, 2).Value = r.Offset(1, 2).Value 'start
ws.Cells(t, sC).Offset(, 2).NumberFormat = "hh:mm:ss"
'
ws.Cells(t, sC).Offset(, 3).Value = r.Offset(2, 2).Value 'stop
ws.Cells(t, sC).Offset(, 3).NumberFormat = "hh:mm:ss" '<<<< here
'
ws.Cells(t, sC).Offset(, 4).Value = ws.Cells(t, sC).Offset(, 3).Value - ws.Cells(t, sC).Offset(, 2).Value 'time
ws.Cells(t, sC).Offset(, 4).NumberFormat = "hh:mm:ss"
'
ws.Cells(t, sC).Offset(, 5).FormulaR1C1 = "=HOUR(RC[-1])*60+MINUTE(RC[-1])+SECOND(RC[-1])/60" 'mins
ws.Cells(t, sC).Offset(, 5).NumberFormat = "00.00"
t = t + 1
End If
Next
End Sub
