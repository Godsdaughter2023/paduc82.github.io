Sub Excel2Email()
    Dim OutApp As Object
    Dim OutMail As Object
    Dim strbody As String
        Set OutApp = CreateObject("Outlook.Application")
        Set OutMail = OutApp.CreateItem(0)

        strbody = "Hello,<br><br>" & _
                  "This is Google home page  :<br>" & _
                  "Click on this link to open Google : " & _
                  "<A HREF=""https://www.google.com/" & _
                  """>Click Here</A>" & _
                  "<br><br>Thanks," & _
                  "<br><br>My Name</font>"

        On Error Resume Next
        With OutMail
            .To = ""
            .CC = ""
            .BCC = ""
            .Subject = "My Subject"
            .HTMLBody = strbody
            '.Body = strbody
            .Display   'or use .Send
        End With
        On Error GoTo 0

        Set OutMail = Nothing
        Set OutApp = Nothing

End Sub
