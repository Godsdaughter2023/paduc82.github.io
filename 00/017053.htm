suzanne s. barnhill say:

You can hyperlink back to the source if there is only one. In order to do that, you would have to bookmark the original location (the hyperlink) in order to hyperlink to that bookmark.

It is often the case, however, that there are several references (hyperlinks) to the same photo, so it wouldn't be practical to link back to the original place. If any editing has been done in the document before following the hyperlink, you can use the keyboard shortcut Shift+F5 to return to the previous position. But in any case, you can return using the WebGoBack command (Alt+Left Arrow).

So your only way out is to write it by your own, this one is a little bit quick and dirty, but I think a good start.

Add a class module, name it "CalendarEvent" and put this code in there:

--- schnipp ---
Public WithEvents DayBox As MSForms.Label
Public Parent As Calendar

Private Sub DayBox_Click()
  If Not Parent.LastDayBox Is Nothing Then
    'Restore backcolor
    Parent.LastDayBox.DayBox.BackColor = RGB(255, 255, 255)
  End If
  'This day becomes blue
  DayBox.BackColor = RGB(0, 0, 255)
  'Remember ourself
  Set Parent.LastDayBox = Me
End Sub

Private Sub DayBox_DblClick(ByVal Cancel As MSForms.ReturnBoolean)
  'Raise the event
  Parent.ReturnThisDate
  'Close the userform
  Unload Parent
End Sub
--- schnapp ---

Add a userform, name it "Calendar" and put this code in there:

--- schnipp ---
Option Explicit

Dim CP As New ControlPositioner
Dim CECol As New Collection
Public LastDayBox As CalendarEvent
Event ReturnDate(ByVal SelectedDate As Date)

Sub ReturnThisDate()
  RaiseEvent ReturnDate(LastDayBox.DayBox.Tag)
End Sub

Private Sub UserForm_Initialize()
  Dim L As MSForms.Label
  Dim I As Integer, J As Integer
  Dim ToDay As Date, ThisDay As Date
  Dim CE As CalendarEvent
   'Setup a date for the month that should displayed
  ToDay = Now
  'Determine the first monday before this month
  ThisDay = DateSerial(Year(ToDay), Month(ToDay), 1)
  If Weekday(ThisDay) <> vbMonday Then
    ThisDay = ThisDay - Weekday(ThisDay) + vbMonday
  End If
  'Startposition of our controls
  CP.Left = 5
  CP.Top = 5
  'The first row is a titel of the week days
  For I = 1 To 7
    Set L = Me.Controls.Add("Forms.Label.1")
    L.Caption = WeekdayName(I, True)
    L.Width = 20
    CP.Add L, I, 1, True, True
  Next
  'The next rows are the days
  For J = 1 To 5
    For I = 1 To 7
      Set L = Me.Controls.Add("Forms.Label.1")
      L.Caption = Format(ThisDay, "DD")
      L.Width = 20
      L.BackColor = IIf(Month(ThisDay) = Month(ToDay), RGB(255, 255, 255), RGB(128, 128, 128))
      L.Tag = ThisDay
      CP.Add L, I, J + 1, True, True
       Set CE = New CalendarEvent
      Set CE.DayBox = L
      Set CE.Parent = Me
      CECol.Add CE
       ThisDay = ThisDay + 1
    Next
  Next
  CP.MoveControls 3, True
End Sub
--- schnapp ---

'A class module to position controls in a rectangled area.

Option Explicit

Private Controls As New Collection
Private FirstColumn As Integer
Private FirstRow As Integer
Private LastColumn As Integer
Private LastRow As Integer

'Position of Area, set before call MoveControls!
Public Left As Single
Public Top As Single

'Size of Area, ask after call MoveControls!
Public Width As Single
Public Height As Single

Sub Add(C As Control, _
    ByVal Column As Integer, ByVal Row As Integer, _
    Optional ByVal AutoSizeWidth As Boolean = False, _
    Optional ByVal AutoSizeHeight As Boolean = False)
  Dim CP(0 To 2) As Variant
  Dim SaveHeight As Single, SaveWidth As Single
   Set CP(0) = C
  CP(1) = Column
  CP(2) = Row
   If Controls.Count = 0 Then
    FirstColumn = Column
    FirstRow = Row
    LastColumn = Column
    LastRow = Row
  Else
    FirstColumn = WorksheetFunction.Min(Column, FirstColumn)
    FirstRow = WorksheetFunction.Min(Row, FirstRow)
    LastColumn = WorksheetFunction.Max(Column, LastColumn)
    LastRow = WorksheetFunction.Max(Row, LastRow)
  End If
   If AutoSizeWidth Or AutoSizeHeight Then
    SaveHeight = C.Height
    SaveWidth = C.Width
     If AutoSizeWidth Then C.Width = 999
    If AutoSizeHeight Then C.Height = 999
    C.AutoSize = True
    C.AutoSize = False
    If Not AutoSizeWidth Then C.Width = SaveWidth
    If Not AutoSizeHeight Then C.Height = SaveHeight
  End If
   Controls.Add CP
End Sub

Sub MoveControls( _
    Optional Space As Single = 3, _
    Optional CenterHorizontal As Boolean = False, _
    Optional CenterVertikal As Boolean = False)
  Dim CP '(0 To 2) as Variant
  Dim C As Control, Column As Integer, Row As Integer
  Dim I As Integer, J As Integer
  Dim Pos As Single
  Dim Widths() As Single
  Dim Heights() As Single
   If Controls.Count <= 1 Then Exit Sub
  ReDim Heights(FirstRow To LastRow) As Single
  ReDim Widths(FirstColumn To LastColumn) As Single
   For I = 1 To Controls.Count
    CP = Controls(I)
    Set C = CP(0)
    Column = CP(1)
    Row = CP(2)
    Widths(Column) = WorksheetFunction.Max(Widths(Column), C.Width)
    Heights(Row) = WorksheetFunction.Max(Heights(Row), C.Height)
  Next
   Height = Space
  For I = FirstRow To LastRow
    Height = Height + Heights(I) + Space
  Next
  Width = Space
  For I = FirstColumn To LastColumn
    Width = Width + Widths(I) + Space
  Next
   For I = 1 To Controls.Count
    CP = Controls(I)
    Set C = CP(0)
    Column = CP(1)
    Row = CP(2)
     With C
      Pos = Top + Space
      For J = FirstRow To Row - 1
        Pos = Pos + Heights(J) + Space
      Next
      .Top = Pos
      If CenterVertikal Then .Top = .Top + Heights(J) / 2 - .Height / 2
       Pos = Left + Space
      For J = FirstColumn To Column - 1
        Pos = Pos + Widths(J) + Space
      Next
      .Left = Pos
      If CenterHorizontal Then .Left = .Left + Widths(J) / 2 - .Width / 2
    End With
  Next
End Sub

Option Explicit

Dim CP As New ControlPositioner
Dim CECol As New Collection
Public LastDayBox As CalendarEvent
Event ReturnDate(ByVal SelectedDate As Date)

Sub ReturnThisDate()
  RaiseEvent ReturnDate(LastDayBox.DayBox.Tag)
End Sub

Private Sub UserForm_Initialize()
  Dim L As MSForms.Label
  Dim I As Integer, J As Integer
  Dim ToDay As Date, ThisDay As Date
  Dim CE As CalendarEvent
   'Setup a date for the month that should displayed
  ToDay = Now
  'Determine the first monday before this month
  ThisDay = DateSerial(Year(ToDay), Month(ToDay), 1)
  If Weekday(ThisDay) <> vbMonday Then
    ThisDay = ThisDay - Weekday(ThisDay) + vbMonday
  End If
  'Startposition of our controls
  CP.Left = 5
  CP.Top = 5
  'The first row is a titel of the week days
  For I = 1 To 7
    Set L = Me.Controls.Add("Forms.Label.1")
    L.Caption = WeekdayName(I, True)
    L.Width = 20
    CP.Add L, I, 1, True, True
  Next
  'The next rows are the days
  For J = 1 To 5
    For I = 1 To 7
      Set L = Me.Controls.Add("Forms.Label.1")
      L.Caption = Format(ThisDay, "DD")
      L.Width = 20
      L.BackColor = IIf(Month(ThisDay) = Month(ToDay), RGB(255, 255, 255), RGB(128, 128, 128))
      L.Tag = ThisDay
      CP.Add L, I, J + 1, True, True
       Set CE = New CalendarEvent
      Set CE.DayBox = L
      Set CE.Parent = Me
      CECol.Add CE
       ThisDay = ThisDay + 1
    Next
  Next
  CP.MoveControls 3, True
End Sub
