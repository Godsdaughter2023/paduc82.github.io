bernie deitrick say:

I would use a macro to create the CSV - this example assumes the dates are in A and start in A2, and the created csv will be saved in the same folder as the file with the macro: I have never seen a time format with the numbers 2 and 4 after the hour - remove them from the Format command if they really aren't supposed to be there....

Sub ExportToCSV()
    Dim fName As String
    Dim WholeLine As String
    Dim FNum As Integer
    Dim RowNdx As Long
    Dim ColNdx As Long
    Dim StartRow As Long
    Dim EndRow As Long
    Dim StartCol As Long
    Dim EndCol As Long
    
    fName = "Export.csv"
    
    Application.ScreenUpdating = False
    On Error GoTo EndMacro:
    FNum = FreeFile
    
    StartRow = 1
    StartCol = 1
    EndRow = Cells(Rows.Count, "A").End(xlUp).Row
    EndCol = Cells(1, Columns.Count).End(xlToLeft).Column
    
    Open fName For Output Access Write As #FNum
    
    For RowNdx = StartRow To EndRow
        WholeLine = Format(Cells(RowNdx, 1).Value, "yyyy-mm-dd hh24:mm:ss")
        For ColNdx = StartCol + 1 To EndCol
             WholeLine = WholeLine & "," & Cells(RowNdx, ColNdx).Text
        Next ColNdx
        Print #FNum, WholeLine
    Next RowNdx
    
EndMacro:
    On Error GoTo 0
    Application.ScreenUpdating = True
    Close #FNum
    
End Sub

Do the Same on the Time Tab for your Required Time format

Then open Excel and export a sheet as csv and check the resulting date format
