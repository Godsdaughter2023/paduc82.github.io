Sub ImportContactsFromOutlook(strTableName As String)
'from
'https://support.microsoft.com/en-us/help/290792/how-to-programmatically-export-outlook-items-to-access
   ' This code is based in Microsoft Access.

   ' Set up DAO objects
   Dim rst As DAO.Recordset
   Set rst = CurrentDb.OpenRecordset(strTableName)

   ' Set up Outlook objects.
   Dim ol As New outlook.Application
   Dim olns As outlook.NameSpace
   Dim cf As outlook.MAPIFolder
   Dim c As outlook.ContactItem
   Dim objItems As outlook.Items
   Dim Prop As outlook.UserProperty
    Dim blIsEX As Boolean
    Dim blEmail1IsState As Boolean
    Dim blEmail2IsState As Boolean
    Dim blEmail3IsState As Boolean
    Dim blOneEmailExists As Boolean
    Dim blFirstLastNotNull As Boolean
    Dim blNoneAreState As Boolean
   
'added below as original code didn't have it
    Dim iNumContacts As Integer
    Dim i As Integer
   
   Set olns = ol.GetNamespace("MAPI")
   Set cf = olns.GetDefaultFolder(olFolderContacts)
   Set objItems = cf.Items
   iNumContacts = objItems.Count
   If iNumContacts <> 0 Then
      For i = 1 To iNumContacts

         'pseudo
         'not EX
         'one email has address (at least one has an email)
         'none are state
         'last name not null
         'first name not null

         If TypeName(objItems(i)) = "ContactItem" Then
            Set c = objItems(i)
                    
                blIsEX = objItems(i).Email1AddressType = "EX"
                blEmail1IsState = Right(c.Email1Address, Len(c.Email1Address) - InStr(1, c.Email1Address, "@")) = "michigan.gov"
                blEmail2IsState = Right(c.Email2Address, Len(c.Email2Address) - InStr(1, c.Email2Address, "@")) = "michigan.gov"
                blEmail3IsState = Right(c.Email3Address, Len(c.Email3Address) - InStr(1, c.Email3Address, "@")) = "michigan.gov"
                blNoneAreState = (Not blEmail1IsState) And (Not blEmail2IsState) And (Not blEmail3IsState)
                blOneEmailExists = ((Not c.Email1Address = "") Or (Not c.Email2Address = "") Or (Not c.Email3Address = ""))
                blFirstLastNotNull = ((Not c.FirstName = "") And (Not c.LastName = ""))
                    
                If (Not blIsEX) And blOneEmailExists And blNoneAreState And blFirstLastNotNull Then
                    rst.AddNew
                    rst![First Name] = c.FirstName
                    rst![Last Name] = c.LastName
                    If Not c.BusinessTelephoneNumber = "" Then rst![Business Phone] = c.BusinessTelephoneNumber
                    If Not c.MobileTelephoneNumber = "" Then rst![Mobile Phone] = c.MobileTelephoneNumber
                    If (Not c.Email1Address = "") Then ' if the email address is not an Exchange address. Assumes that we obtain emails of internal users elsewhere
                    rst![E-mail Address] = c.Email1Address
                    End If
                    If (Not c.Email2Address = "") Then ' if the email address is not an Exchange address. Assumes that we obtain emails of internal users elsewhere
                    rst![E-mail Address2] = c.Email2Address
                    End If
                    If (Not c.Email3Address = "") Then ' if the email address is not an Exchange address. Assumes that we obtain emails of internal users elsewhere
                    rst![E-mail Address3] = c.Email3Address
                    End If
                    If Not c.BusinessAddressStreet = "" Then rst!Address = c.BusinessAddress
                    If Not c.BusinessAddressCity = "" Then rst!City = c.BusinessAddressCity
                    If Not c.BusinessAddressState = "" Then rst![State/Province] = c.BusinessAddressState
                    If Not c.BusinessAddressPostalCode = "" Then rst![ZIP/Postal Code] = c.BusinessAddressPostalCode
                    If Not c.CompanyName = "" Then rst![Company] = c.CompanyName
                    If Not c.JobTitle = "" Then rst![Job Title] = c.JobTitle
                    rst.Update
                End If
         End If
      Next i
      rst.Close
      MsgBox "Finished."
   Else
      MsgBox "No contacts to export."
   End If

End Sub

WEEKNUM function

I have a large data where transaction dates need to be converted to the week number of year they pertain to. The condition now is

  1. Week starts on Sunday,hence ends on Saturday 
  2. Financial year starts on 1st October, ends on 30th September

Keeping the above in mind, the formula should return the value. Can it be worked around weeknum formula or something else ?

ronrosenfeld say: One problem is to determine how to define Week 1.   One standard is that all weeks are seven days long, and Week 1 of the year is the first week that contains a Thursday (that standard ISO8601, also has the week starting on a Monday).  So sometimes the last few days of the previous year are in the subsequent year; other times the first few days after Oct 1 are in the previous year.

But other conventions can be used.  Including short weeks at the start/end of the year.

So how do you define Week 1?

himpj say:
One problem is to determine how to define Week 1.   One standard is that all weeks are seven days long, and Week 1 of the year is the first week that contains a Thursday (that standard ISO8601, also has the week starting on a Monday).  So sometimes the last few days of the previous year are in the subsequent year; other times the first few days after Oct 1 are in the previous year.

But other conventions can be used.  Including short weeks at the start/end of the year.

So how do you define Week 1?

That precisely is the constraint. Now lets take example of "01st October 2014" 

This date falls on Wednesday, which according to our organisation is the first day of financial year. Now even though technically , working week starts on a Sunday, which 5th October 2014 would be first Sunday of the financial year; we will need to ignore that in our case. 

So to set the rule, here it goes

  1. Week # 1 starts on 1st October, regardless of whichever day it falls on
  2. count 7 days from first day of first week. All such days will be indicated, falling within week # 1.
  3. Start week # 2
  4. Repeat steps as point # 2 above to calculate second week. All such days will be indicated, falling within week # 2.
  5. so on
  6. Last week of the year concludes on 30th September 2015
