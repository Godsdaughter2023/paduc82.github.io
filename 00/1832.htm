It is easy enough to forward a message using a macro to add-in the text you require. The following macro will add the indicated text to the start of the forwarded message with your default signature for the account in question. You can format the range however you wish . Select a message to forward and run the macro.

Sub ForwardWithText()
Dim olItem As Outlook.MailItem
Dim olOutMail As Outlook.MailItem
Dim olInsp As Outlook.Inspector
Dim wdDoc As Object
Dim oRng As Object
Dim sName() As String
    If Application.ActiveExplorer.Selection.Count = 0 Then
        MsgBox "No message selected!", vbCritical, "Error"
        Exit Sub
    End If
    Set olItem = Application.ActiveExplorer.Selection.item(1)
    Set olOutMail = olItem.Forward
    With olOutMail
        .BodyFormat = olFormatHTML
        Set olInsp = .GetInspector
        Set wdDoc = olInsp.WordEditor
        Set oRng = wdDoc.Range(0, 0)
        With oRng
            .Text = "This is the text of your accompanying message"
            'format the range
            .Font.Name = "Calibri"
            .Font.Size = 14
            .Font.Italic = True
            .Font.Color = RGB(255, 0, 0) 'red
        End With
        .Display
    End With
lbl_Exit:
    Exit Sub
End Sub
