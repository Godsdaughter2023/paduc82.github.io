graham mayor say:
If you add the space as Doug suggests you will include words that are not part of the string. I can't think of a way of doing this with the replace function; however it is not that far adrift from your requirement in your other thread and thus can be done with a macro.

The macro below should work with both straight quotes and smart quotes. What it won't do is find a single quote at the start of a paragraph, so if you have any text strings that begin a paragraph you will have to run a separate search for those.

Sub SmallCaps()
Dim oRng As Range
Dim oStory As Range
Dim strAsk As String
    For Each oStory In ActiveDocument.StoryRanges
        Set oRng = oStory
        With oRng.Find
            Do While .Execute(FindText:=" '")
                oRng.MoveEndUntil Chr(39) & Chr(146)
                oRng.Start = oRng.Start + 2
                If Len(oRng.Text) > 1 Then
                    oRng.Select
                    strAsk = MsgBox("Small Caps?", vbYesNoCancel)
                    If strAsk = vbCancel Then GoTo Cleanup
                    If strAsk = vbYes Then
                        oRng.Start = oRng.Start + 1
                        oRng.Case = wdLowerCase
                        oRng.Font.SmallCaps = True
                    End If
                End If
                oRng.Collapse wdCollapseEnd
            Loop
        End With
    Next oStory
    ActiveWindow.View.Type = wdPrintView
    ActiveWindow.ActivePane.View.SeekView = wdSeekMainDocument
    ActiveWindow.View.Type = wdPrintView
Cleanup:
    Set oRng = Nothing
    Set oStory = Nothing
End Sub

Sub test()
    Dim rngC As Range
    Dim wsS As Worksheet
    Dim wsT As Worksheet
    Dim i As Integer
    Dim lngR As Long

    Set wsT = Worksheets("Output")
