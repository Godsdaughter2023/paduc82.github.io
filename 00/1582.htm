    Hi,

    for/next 'method', is rather slow if you have many rows

    so,

    try and this

    make a copy before....

    Assuming that in row 1 are headers

    Sub Delete_Rows()
    'Mar. 17, 2018
    On Error GoTo errh
    Const sC As String = "F"
    Dim ws As Worksheet
    Set ws = ActiveSheet 'data in active sheet
    Dim r As Long
    ws.AutoFilterMode = False
    r = ws.Cells(Rows.Count, sC).End(xlUp).Row
    With ws.Cells(1, sC).Resize(r, 2)
    .AutoFilter Field:=1, Criteria1:="=0"
    .AutoFilter Field:=2, Criteria1:="=0"
    End With
    ws.Cells(2, sC).Resize(r - 1).SpecialCells(xlCellTypeVisible).EntireRow.Delete
    ws.AutoFilterMode = False: Exit Sub
    errh:
    MsgBox "nothing found"
    ws.AutoFilterMode = False
    End Sub

    hansv mvp say:

    Try this version. Change the constant strDoc to the path and filename of the Word document that you want to open.

     

    Sub CopyRangeToWord()
        Const strRangeToCopy As String = "A1:J80"
        ' Modify path and filename as needed
        Const strDoc = "C:\Documents\MyDocument.docx"
        Dim appWord As Object
        Dim doc As Object
        Dim rng As Object
        Range(strRangeToCopy).Copy
        On Error Resume Next
        Set appWord = GetObject(, "Word.Application")
        On Error GoTo 0
        If appWord Is Nothing Then
            Set appWord = CreateObject("Word.Application")
        End If
        Set doc = appWord.Documents.Open(strDoc)
        With doc.Tables(1)
            Set rng = .Range
            .Delete
        End With
        rng.Paste
        doc.Save
        appWord.Visible = True
        Application.CutCopyMode = False
    End Sub

     

    If you'd rather close the document instead of making it visible:

     

    Sub CopyRangeToWord()
        Const strRangeToCopy As String = "A1:J80"
        ' Modify path and filename as needed
        Const strDoc = "C:\Documents\MyDocument.docx"
        Dim appWord As Object
        Dim doc As Object
        Dim rng As Object
        Dim fStarted As Boolean
        Range(strRangeToCopy).Copy
        On Error Resume Next
        Set appWord = GetObject(, "Word.Application")
        On Error GoTo 0
        If appWord Is Nothing Then
            Set appWord = CreateObject("Word.Application")
            fStarted = True
        End If
        Set doc = appWord.Documents.Open(strDoc)
        With doc.Tables(1)
            Set rng = .Range
            .Delete
        End With
        rng.Paste
        doc.Close SaveChanges:=True
        If fStarted Then
            appWord.Quit
        End If
        Application.CutCopyMode = False
    End Sub

    add months to a date - Microsoft Community

    bfosterneedshelp say:

    add months to a date

    I have a start date for a report and I want project when monthly interim reports are due based on the begin date.  How do I add months to a date in a cell?
    ronrosenfeld say:
    On Sat, 29 Jan 2011 20:31:57 +0000, bfosterneedshelp wrote:
     
    >I have a start date for a report and I want project when monthly interim reports are due based on the begin date.  How do I add months to a date in a cell?
     
    Use one of these two formulas, with begin date in A1:
     
    =EDATE(A1,2)
     
    =MIN(DATE(YEAR(A1),MONTH(A1)+{3,2},DAY(A1)*{0,1}))
     
    The second will work in any version of Excel.
     
    The first requires either Excel 2007 or later; or installation of the Analysis ToolPak (for earlier versions).
