andreas killer say:
Well, I would never do an analysis in this kind, IMHO it is unnecessary to create formulas first, then evaluate them using VBA, and then deleting them again. Why not make the same analysis entirely in VBA? It's faster and easier to understand at a later time.

Anyway, take a look at the macro below, is it not easier to understand what it does?

Andreas.

Sub Subregion_AK()
  Dim R As Range
  Dim Data, ID, Value
  Dim i As Long

  'Screen off
  Application.ScreenUpdating = False
 
  With Worksheets("Subregion")
    'Find the last used cell in column D
    Set R = .Range("D" & Rows.Count).End(xlUp)

    'Calculation off
    Application.Calculation = xlCalculationManual
   
    'Create the basic formulas
    .Range("A7").Formula = "=C7"
    .Range("A8").Formula = "=IF(C8="""",A7,C8)"
    .Range("B7").Formula = _
      "=IF(INDEX(E7:R7,MATCH(""?*"",INDEX(""""&E7:R7,0),0))=R7,0,A7)"

    'Fill them down
    .Range("A8", .Range("A" & R.Row)).FillDown
    .Range("B7", .Range("B" & R.Row)).FillDown
   
    'Let the formulas calculate
    Application.Calculation = xlCalculationAutomatic

    'column A contains all the ID from pivot through excel formula
    'suppose if column B contains column A id then next

    'Read in all data
    Data = .Range("A7", .Range("B" & R.Row))
    'Visit each row
    For i = 1 To UBound(Data)
      'ID changed or last row?
      If (Data(i, 1) <> ID) Or (i = UBound(Data)) Then
        'Note: ID is empty at the first row
        If Not IsEmpty(ID) Then
          'Show non-zero pivot items only
          .PivotTables("PivotTable239").PivotFields("orderid") _
            .PivotItems(ID).Visible = Value <> 0
        End If
        'New ID, set the first value
        ID = Data(i, 1)
        Value = Data(i, 2)
      Else
        'Store non-zero values
        If Data(i, 2) <> 0 Then Value = Data(i, 2)
      End If
    Next

    'Remove the formulas
    .Columns("A:B").ClearContents
  End With
  Application.ScreenUpdating = True
End Sub
