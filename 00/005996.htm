Sub SaveCSV()
   Dim r    As Variant
   Dim file As Object

   'please change file path
   Set file = CreateObject("Scripting.FileSystemObject").CreateTextFile("E:\test\test.csv", True)

   With ActiveSheet
      .UsedRange
      For Each r In .Range(.[A1], .Cells.SpecialCells(xlCellTypeLastCell)).Rows
         If r.Columns.Count = 1 Then r = Array(r.Value)
         file.WriteLine Join(WorksheetFunction.Transpose(WorksheetFunction.Transpose(r)), ",")
      Next
   End With
End Sub
