    Set oHDoc = IE.Document

    Dim oHEle As HTMLUListElement
    Dim Table As Object 'HTMLElementCollection
    Dim TableCell As HTMLTableCell

    Dim i As Long, j As Long
    Do While oHEle Is Nothing
      Set oHEle = oHDoc.getElementById("valuation_history_table")
      DoEvents
    Loop
    Set Table = oHEle.getElementsByClassName("row_data")
    For Each TableCell In Table
      If TableCell.cellIndex > i Then
        i = TableCell.cellIndex
      Else
        Exit For
      End If
    Next

    Dim Data
    ReDim Data(1 To Table.Length / i, 1 To i)
    j = 1
    For Each TableCell In Table
      Data(j, TableCell.cellIndex) = Val(TableCell.outerText)
      If TableCell.cellIndex = i Then j = j + 1
    Next
    Range("A1").Resize(UBound(Data), UBound(Data, 2)).Value = Data
