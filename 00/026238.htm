tasos_k say:

Hi,

try and this vba macro,

data in activesheet, in column A

results in columns B and C

(make a copy before you run this code)

Sub SplitData_TwoColumns()
'Apr. 22, 2016

On Error Resume Next

Dim x As Long, N As Long, vN As Long
v = Array("011", "0124", "022", "07272") '<< change/add as needed
With Columns("B:C")
.ClearContents
.NumberFormat = "@"
End With
For Each r In Range("A:A").SpecialCells(xlCellTypeConstants)

N = Len(r.Value)

For x = 0 To UBound(v)
vN = Len(v(x))
If Left(r.Value, vN) = v(x) Then
r.Offset(, 1).Value = v(x)
r.Offset(, 2).Value = Right(r.Value, N - vN)
GoTo nnext
End If
Next
nnext:
Next

On Error GoTo 0

End Sub
