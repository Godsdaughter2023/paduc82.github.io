andreas killer say:
AFAIK there is no way to accomplish this directly from VBA.

Unfortunately the register tabs inside the "Navigation" commandbar are accessible objects (the commandbar itself is a child window inside the application), so we need a lot of API code.


Sub Main()
  Dim hWnd As Long, uWnd As Long
  Dim iaNavigation As IAccessible, iaRegister As IAccessible
 
  'Show the navigation commandbar
  ThisDocument.ActiveWindow.DocumentMap = True
 
  'Find the window of our document
  hWnd = FindWindowLike("OpusApp", ThisDocument.Name & "*")
  'Find the navigation window inside
  'OpusApp\MsoCommandBarDock "MsoDockLeft"\MsoCommandBar "Navigation"\ _
    MsoWorkPane "Navigation"\NUIPane\NetUIHWND
  uWnd = FindChildWindow(hWnd, "NetUIHWND")
 
  'Get the "File" register (this contains the 3 tabs)
  Set iaNavigation = FindAccessibleObject(uWnd, 3)
  'Get the second tab
  Set iaRegister = FindAccessibleObject(iaNavigation, 1)
  'The default action of this object is "Toggle", execute it
  iaRegister.accDoDefaultAction ByVal 0&
End Sub

Function IsChildVisible(ByVal hWnd As Long) As Boolean
  Dim R As RECT, T As RECT
  Dim tWnd As Long
  GetWindowRect hWnd, R
  tWnd = WindowFromPoint((R.Right + R.Left) / 2, (R.Top + R.Bottom) / 2)
  GetWindowRect tWnd, T
  IsChildVisible = tWnd = hWnd
  If Not IsChildVisible Then
    tWnd = GetWindow(tWnd, GW_CHILD)
    Do
      IsChildVisible = tWnd = hWnd
      tWnd = GetWindow(tWnd, GW_HWNDNEXT)
    Loop Until (tWnd = 0) Or IsChildVisible
  End If
End Function

Private Function EnumFindWindowLikeProc(ByVal hWnd As Long, ByVal lParam As Long) As Long
  Dim HasTitle As Boolean, HasClass As Boolean, HasID As Boolean, HasContents As Boolean
  Dim Contents As String, i As Long
  If Not IsMissing(wCtlID) Then
    i = GetDlgCtrlID(hWnd)
    HasID = i = wCtlID
  Else
    HasID = True
  End If
  If Not IsMissing(wClass) Then
    Contents = Space$(256)
    i = GetClassNameA(hWnd, Contents, Len(Contents) - 1)
    HasClass = Left$(Contents, i) Like wClass
  Else
    HasClass = True
  End If
  If Not IsMissing(wTitle) Then
    Contents = Space$(GetWindowTextLengthA(hWnd) + 1)
    i = GetWindowTextA(hWnd, Contents, Len(Contents))
    HasTitle = Left$(Contents, i) Like wTitle
  Else
    HasTitle = True
  End If
  If Not IsMissing(wContents) Then
    i = SendMessage(hWnd, WM_GETTEXTLENGTH, 0, 0)
    If i > 0 Then
      Contents = Space$(i + 1)
      i = SendMessage(hWnd, WM_GETTEXT, i + 1, ByVal Contents)
      HasContents = Trim$(Left$(Contents, i)) Like wContents
    End If
  Else
    HasContents = True
  End If
  If HasTitle And HasClass And HasID And HasContents Then
    wResult.Add hWnd
    EnumFindWindowLikeProc = lParam
  Else
    EnumFindWindowLikeProc = 1
  End If
End Function

Function FindWindowLike( _
    Optional ByVal ClassName, Optional ByVal Title, Optional ByVal ID, Optional ByVal Contents, _
    Optional ByVal FindAll As Boolean = False) As Variant
  'Sucht ein/alle Fenster LIKE ClassName und/oder LIKE Title
  wCtlID = ID
  wClass = ClassName
  wTitle = Title
  wContents = Contents
  Set wResult = New Collection
  If FindAll Then
    EnumWindows AddressOf EnumFindWindowLikeProc, 1
    Set FindWindowLike = wResult
  Else
    EnumWindows AddressOf EnumFindWindowLikeProc, 0
    If wResult.Count > 0 Then FindWindowLike = wResult(1)
  End If
End Function

Function FindThreadWindow(ByVal hWnd As Long, _
    Optional ByVal ClassName As Variant, Optional ByVal Title As Variant, Optional ByVal ID, Optional ByVal Contents, _
    Optional ByVal FindAll As Boolean = False) As Variant
  'Sucht ein/alle "Nicht Kind"-Fenster LIKE ClassName und/oder LIKE Title die zu diesem Fenster gehören
  Dim ThreadID As Long, ProcessID As Long
  wCtlID = ID
  wClass = ClassName
  wTitle = Title
  wContents = Contents
  Set wResult = New Collection
  ThreadID = GetWindowThreadProcessId(hWnd, ProcessID)

  Else
    EnumThreadWindows ThreadID, AddressOf EnumFindWindowLikeProc, 0
    If wResult.Count > 0 Then FindThreadWindow = wResult(1)
  End If
End Function

Function FindDirectChildWindow(ByVal hWnd As Long, _
    Optional ByVal ClassName As Variant, Optional ByVal Title As Variant, Optional ByVal ID, Optional ByVal Contents, _
    Optional ByVal FindAll As Boolean = False) As Variant
  'Sucht ein/alle "Kind"-Fenster dieser Ebene LIKE ClassName und/oder LIKE Title die zu diesem Fenster gehören
  Dim hWndChild As Long
  wCtlID = ID
  wClass = ClassName
  wTitle = Title
  wContents = Contents
  Set wResult = New Collection

  hWndChild = GetWindow(hWnd, GW_CHILD)
  Do While hWndChild <> 0
    If EnumFindWindowLikeProc(hWndChild, 0) = 0 Then
      If Not FindAll Then
        FindDirectChildWindow = hWndChild
        Exit Function
      End If
    End If
    hWndChild = GetWindow(hWndChild, GW_HWNDNEXT)
  Loop
  If FindAll Then Set FindDirectChildWindow = wResult
End Function

Function FindEditWindow(ByVal hWnd As Long, ByVal hChild As Long, _
    Optional ByVal Direction As MsoBarPosition = msoBarRight, _
    Optional ByVal Space As Long = 5, Optional ByVal Tol As Long = 5, _
    Optional ByVal Visible As Boolean = False, _
    Optional ByVal FindAll As Boolean = False) As Variant
  Dim Item
  Dim CR As RECT, R As RECT, T As RECT
  Dim X As Long, Y As Long

  GetWindowRect hChild, CR
  If Tol <= 0 Then Tol = 1
  If Space < 0 Then Space = 0
  With CR
    Select Case Direction
      Case msoBarRight
        Y = (.Top + .Bottom) / 2
        X = .Right + Space
      Case msoBarLeft
        Y = (.Top + .Bottom) / 2
        X = .Left - Space
      Case msoBarBottom
        X = (.Left + .Right) / 2
        Y = .Bottom + Space
      Case msoBarTop
        X = (.Left + .Right) / 2
        Y = .Top - Space
      Case Else
        X = (.Left + .Right) / 2
        Y = (.Top + .Bottom) / 2
    End Select
    .Left = X - Tol
    .Right = X + Tol
    .Top = Y - Tol
    .Bottom = Y + Tol
  End With

  If FindAll Then Set FindEditWindow = New Collection
  For Each Item In FindChildWindow(hWnd, "*", FindAll:=True)
    GetWindowRect Item, R
    If IntersectRect(T, CR, R) Then
      If FindAll Then
        If Not Visible Or (Visible And IsChildVisible(Item)) Then FindEditWindow.Add Item
      Else
        If Not Visible Or (Visible And IsChildVisible(Item)) Then
          FindEditWindow = Item
          Exit Function
        End If
      End If
    End If
  Next
End Function

Function FindAccessibleObject(ByVal hWndOrAccessible As Variant, _
    Optional ByVal Index As Long = -1) As Variant
  Dim i As Long, Count As Long, lRet As Long
  Dim GUID As UUID
  Dim iAcc As IAccessible, iChild As Object
  Dim Av()
  Dim FindAll As Boolean

  If Index < 0 Then
    FindAll = True
    Set FindAccessibleObject = New Collection
  Else
    FindAll = False
    Set FindAccessibleObject = Nothing
  End If

  On Error GoTo ExitPoint
  If TypeOf hWndOrAccessible Is IAccessible Then
    Set iAcc = hWndOrAccessible
  Else
    With GUID
      .Data1 = &H618736E0
      .Data2 = &H3C3D
      .Data3 = &H11CF
      .Data4(0) = &H81
      .Data4(1) = &HC
      .Data4(2) = &H0
      .Data4(3) = &HAA
      .Data4(4) = &H0
      .Data4(5) = &H38
      .Data4(6) = &H9B
      .Data4(7) = &H71
    End With
    lRet = AccessibleObjectFromWindow(ByVal hWndOrAccessible, OBJID_CLIENT, GUID, iAcc)
  End If

  Count = iAcc.accChildCount
  If Count = 0 Then Exit Function
  If Index >= Count Then Exit Function
  If Index >= 0 Then Count = 1 Else Index = 0
  ReDim Av(0 To Count - 1)
  lRet = AccessibleChildren(iAcc, Index, Count, Av(0), Count)
  If Count > UBound(Av) + 1 Then Count = UBound(Av) + 1

  For i = 0 To Count - 1
    If TypeOf Av(i) Is Object  Then
      Set iChild = Av(i)
    Else
      Set iChild = iAcc.accChild(Av(i))
    End If
    If FindAll Then
      FindAccessibleObject.Add iChild
    Else
      Set FindAccessibleObject = iChild
    End If
  Next
ExitPoint:
End Function

tom van stiphout say:

I would definitely reboot the machine. All kinds of fleeting problems are fixed that way.

Then I would run Microsoft Updates to ensure I have all latest patches and service packs.

Then I would decompile the front-end database (search the web for "Access decompile"), as well as compact both the FE and the BE.

LDB files can be deleted at will (if they are not in use) but that doesn't help in itself. Access creates and overwrites these files as-needed.

 

gina whipp say:

I'm hoping the *entire* database isn't on the Server.  I am hoping you have split it and only the Backend is on the Server and the Front end is on your Local.  That said...

1. Close the Database

2. Bring up Task Manager

3. Kill the MSACCESS.EXE file you see in the Process Tab

If you can't do the above and you have to power off the PC, then yes, you delete the Lock file.  And, because this will keep happening unless you split the database.  Split it OR move a copy to your Local work on it and then copy back to the Server.  For more information on splitting databases...
