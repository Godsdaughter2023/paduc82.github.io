Rather than trying to loop through each of the folders in 2 loops, I grab all the CSV filenames and put them into one array, then do the same for all .XLS* files - then loop through those.  DIR$() makes it a bit difficult to do the loop within a loop (although I think it could actually be done). 

Anyhow, I hope this is a little help to you.  If I can get my home network straightened out this weekend I'll check back (new modem killed everything - no internet, no email for the moment).

Sub CopyFromCSVIntoXLS()
'assumptions
' #1 - as stated, the CSV and XLS files are in two different folders
' #2 - The contents of each CSV file's sheet is to be copied into
' each of the XLS workbooks.
' #3 - you have the path to each of the folders setup before we
'      actually get into the code.  I'm going to just show
'      variables CSV_FolderPath and XLS_FolderPath as being
'      used here, they'd need to have full path, including the
'      final \ character as part of them.  This code will
'      provide that.
'
  Dim csvBook As Workbook
  Dim csvSheet As Worksheet
  Dim xlsBook As Workbook
  Dim xlsSheet As Worksheet
 
  Dim csvFileName As String
  Dim xlsFileName As String
  'we will put all the filenames of each type
  'into 2 arrays - easier to work with this way
  Dim csvFiles() As String
  Dim xlsFiles() As String
  'these are to loop through the 2 arrays
  Dim csvLoop As Long
  Dim xlsLoop As Long
 
  Dim CSV_FolderPath As String
  Dim XLS_FolderPath As String
  'nice routine to select folders can be found at, this shows
  'how to use it here
  CSV_FolderPath = BrowseFolder("Select CSV Folder")
  If CSV_FolderPath = vbNullString Then
    Exit Sub ' user cancelled!
  End If
  If Right(CSV_FolderPath, 1) <> Application.PathSeparator Then
    CSV_FolderPath = CSV_FolderPath & Application.PathSeparator
  End If
  XLS_FolderPath = BrowseFolder("Select XLS Folder")
  If XLS_FolderPath = vbNullString Then
    Exit Sub ' user cancelled!
  End If
  If Right(XLS_FolderPath, 1) <> Application.PathSeparator Then
    XLS_FolderPath = XLS_FolderPath & Application.PathSeparator
  End If
  'initialize the two dynamic arrays
  ReDim csvFiles(1 To 1)
  ReDim xlsFiles(1 To 1)
  'initialize both csvFilename
  'to get the sequence started
  csvFileName = Dir$(CSV_FolderPath & "*.csv")
  Do While csvFileName <> ""
    csvFiles(UBound(csvFiles)) = CSV_FolderPath & csvFileName
    'make room for another
    ReDim Preserve csvFiles(1 To UBound(csvFiles) + 1)
    'get next filename from the system
    csvFileName = Dir$()
  Loop
  'csvFiles() will have an empty element at the top
  'of it, get rid of it
  If UBound(csvFiles) > 1 Then
    ReDim Preserve csvFiles(1 To UBound(csvFiles) - 1)
  Else
    'no csv files were found in the folder!
    MsgBox "No CSV files in folder " & CSV_FolderPath, _
     vbOKOnly vbExclamation, "Quitting"
    Exit Sub
  End If
  'do same for XLS files
  xlsFileName = Dir$(XLS_FolderPath & "*.xls*") ' gets .xls, .xlsm, .xlsx, etc.
  Do While xlsFileName <> ""
    xlsFiles(UBound(xlsFiles)) = XLS_FolderPath & xlsFileName
    'make room for another
    ReDim Preserve xlsFiles(1 To UBound(xlsFiles) + 1)
    'get next filename from the system
    xlsFileName = Dir$()
  Loop
  'csvFiles() will have an empty element at the top
  'of it, get rid of it
  If UBound(xlsFiles) > 1 Then
    ReDim Preserve xlsFiles(1 To UBound(xlsFiles) - 1)
  Else
    'no csv files were found in the folder!
    MsgBox "No Excel files in folder " & XLS_FolderPath, _
     vbOKOnly vbExclamation, "Quitting"
    Exit Sub
  End If
'now the real work begins
'this will make it all take place faster
  Application.ScreenUpdating = False
  For csvLoop = LBound(csvFiles) To UBound(csvFiles)
    'open the csv file for use
    Set csvBook = Workbooks.Open(csvFiles(csvLoop))
    Set csvSheet = csvBook.Worksheets(1)
    'now copy what is needed into the Excel workbooks
    'one at a time
    For xlsLoop = LBound(xlsFiles) To UBound(xlsFiles)
      Set xlsBook = Workbooks.Open(xlsFiles(xlsLoop))
      'set the sheet you want to copy into
      Set xlsSheet = xlsBook.Worksheets("SheetName") ' change as needed
      '*******************************************************
      'now in this section, do your copying from the CSV file
      'by referencing csvSheet
      'into the Excel workbook by referencing xlsSheet
      '*******************************************************
      'when all work with this Excel workbook is done, close
      'it and save changes
      xlsBook.Close True
    Next ' end of xlsLoop
    'at this point we now close the CSV file, but do not save
    'any changes that may have taken place (shouldn't be any)
    csvBook.Close False
  Next ' end of csvLoop
  'final housekeeping cleanup
  Set csvSheet = Nothing
  Set csvBook = Nothing
  Set xlsSheet = Nothing
  Set xlsBook = Nothing
  'announce job done
  MsgBox "Copy actions have been completed.", _
   vbOKOnly vbInformation, "Task Finished"
End Sub

'this is the browse for a folder using a dialog from
'http://www.cpearson.com/excel/browsefolder.aspx
'near the bottom of the page
Function BrowseFolder(Title As String, _
        Optional InitialFolder As String = vbNullString, _
        Optional InitialView As Office.MsoFileDialogView = _
            msoFileDialogViewList) As String
    Dim V As Variant
    Dim InitFolder As String
    With Application.FileDialog(msoFileDialogFolderPicker)
        .Title = Title
        .InitialView = InitialView
        If Len(InitialFolder) > 0 Then
            If Dir(InitialFolder, vbDirectory) <> vbNullString Then
                InitFolder = InitialFolder
                If Right(InitFolder, 1) <> "\" Then
                    InitFolder = InitFolder & "\"
                End If
                .InitialFileName = InitFolder
            End If
        End If
        .Show
        On Error Resume Next
        Err.Clear
        V = .SelectedItems(1)
        If Err.Number <> 0 Then
            V = vbNullString
        End If
    End With
    BrowseFolder = CStr(V)
End Function

Adding domains from another Office365 tenant is not supported.

    .Title = "Select the folder that contains the files that you want to import."
    If .Show = -1 Then
        strFolder = .SelectedItems(1) & "\"
    Else
        MsgBox "You did not select a folder."
        Exit Sub
    End If
End With
strFile = Dir$(strFolder & "*.xlsx")
counter = 0
While strFile <> ""
    Set Source = Workbooks.Open(strFolder & strFile)
    With Target.Sheets(1).Range("A1")
        i = .CurrentRegion.Rows.Count
        For m = 1 To Source.Sheets(1).Range("A1").CurrentRegion.Rows.Count - 1
            For n = 0 To Source.Sheets(1).Range("A1").CurrentRegion.Columns.Count - 1
                .Offset(i, n) = Source.Sheets(1).Range("A1").Offset(m, n)
            Next n
            i = i + 1
        Next m
    End With
    counter = counter + 1
    Source.Close xlDoNotSaveChanges
    strFile = Dir$()
Wend
MsgBox ("Finished and processed " & counter & " files") 'This displayes a message box inc. the value of Counter

End Sub
