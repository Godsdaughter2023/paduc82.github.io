doug robbins - mvp office apps & services (word) say:

Using the following code, modified as necessary to add any additional street address types, should insert a carriage return to separate the street address from the City

Dim arrFind As Variant
Dim i As Long
arrFind = Split("Ave|St|Street|Rd|Road|Rte|Route|Hwy|Highway|Circ|Circle|Circuit|Prom|Promenade|Blv|Boulevard|Trail", "|")
For i = LBound(arrFind) To UBound(arrFind)
    Selection.HomeKey wdStory
    Selection.Find.ClearFormatting
    With Selection.Find
        Do While .Execute(FindText:=arrFind(i), Forward:=True, _
            MatchWildcards:=False, Wrap:=wdFindStope) = True
            With Selection
                .Range.InsertAfter vbCr
                .Collapse wdCollapseEnd
            End With
        Loop
    End With
Next i

norman david jones say:

Hi Mill95,

Assuming that the data in column A is of the type 01.01.2013.xls, without using a second column, in order to convert the data to dates and sort it, try something like:

'=========>>

Option Explicit


'--------->>
Public Sub Change2Date()
    Dim SH As Worksheet
    Dim ResultRange As Range, rCell As Range
    Dim LRow As Long

    Set SH = ActiveSheet

    With SH
        LRow = .Cells(.Rows.Count, "A").End(xlUp).Row
        Set ResultRange = .Range("A2:A" & LRow)
    End With

    For Each rCell In ResultRange.Cells
        With rCell
            .NumberFormat = "general"
            .Value = Replace(.Value, ".xls", vbNullString)
            .Value = CDate(Replace(.Value, ".", "/"))
        End With
    Next rCell
    ResultRange.Sort Key1:=Range("A2"), _
                     Order1:=xlAscending, _
                     Header:=xlNo, _
                     OrderCustom:=1, _
                     MatchCase:=False, _
                     Orientation:=xlTopToBottom

End Sub
'<<=========

             SearchOrder:=xlByRows, _
             MatchCase:=False, _
             SearchFormat:=False, _
             ReplaceFormat:=False
        
         ' Assume MDY format
         .Offset(0, 1).TextToColumns _
             Destination:=.Offset(0, 1), _
             DataType:=xlDelimited, _
             TextQualifier:=xlDoubleQuote, _
             ConsecutiveDelimiter:=False, _
             Tab:=False, _
             Semicolon:=False, _
             Comma:=False, _
             Space:=False, _
             Other:=False, _
             FieldInfo:=Array(1, xlMDYFormat), _
             TrailingMinusNumbers:=True
     End With
        
 End Sub
