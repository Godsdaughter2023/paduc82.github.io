jay freedman say:

A script to do this is included in Andreas Killer's post at https://answers.microsoft.com/en-us/msoffice/forum/msoffice_excel-msoffice_custom/how-to-insert-a-hyperlink-in-excel-to-a-word/0a6d4725-4af1-408c-b9e2-f1c1cad6b6c7. As Andreas explains, you need to make a copy of the VBScript file for each template you want to "hyperlink", and name the copy the same as the template except for the .vbs extension.

If you don't want to do this -- for instance, if you have many templates to handle, and you already have hyperlinks to them in your spreadsheet, then you can use this all-VBA macro to create a document from the address in the hyperlink in the selected cell (it assumes there's only one hyperlink per cell).

Note: In order to use this macro as written, you must click Tools > References in the macro editor menu and check the box for the Microsoft Word Object Model.

Sub NewFromHyperlinkToTemplate()

    Dim strTemplatePath As String

    Dim strTemplateExtension As String

    Dim oWord As Word.Application

    Dim WordWasNotRunning As Boolean

    If Selection.Hyperlinks.Count = 1 Then

        strTemplatePath = Selection.Hyperlinks(1).Address

        strTemplateExtension = _

            Right(strTemplatePath, _

            Len(strTemplatePath) - InStrRev(strTemplatePath, "."))

        Select Case strTemplateExtension

            Case "dot", "dotx", "dotm"

                'Get existing instance of Word if it's open; otherwise create a new one

                On Error Resume Next

                Set oWord = GetObject(, "Word.Application")

                If Err Then

                    Set oWord = New Word.Application

                    WordWasNotRunning = True

                End If

                On Error GoTo Err_Handler

                oWord.Visible = True

                oWord.Activate

                oWord.Documents.Add Template:=strTemplatePath

                'Make sure you release object references.

                Set oWord = Nothing

                'quit

                Exit Sub

Err_Handler:

                    MsgBox "Word caused a problem. " & Err.Description, vbCritical, "Error: " _

                            & Err.Number

                    If WordWasNotRunning Then

                        oWord.Quit

                    End If

            Case "xlt", "xltx", "xltm"

                Workbooks.Add Template:=strTemplatePath

            Case Else

                ' do nothing

        End Select

    End If

End Sub


You can assign a Quick Access Toolbar button to run the macro.
