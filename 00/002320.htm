Function GetFileName( _
    Optional ByVal FileFilter As String = "Alle Dateien", _
    Optional ByVal FileMask As String = "*.*", _
    Optional ByVal FilterIndex As Integer = 1, _
    Optional ByVal InitialFileName As String = "", _
    Optional ByVal InitialPath As String = "", _
    Optional ByVal Title As String = "", _
    Optional ByVal MultiSelect As Boolean = False, _
    Optional ByVal flags As OFN_FLAGS = 0, _
    Optional ByVal FileFilterDelimiter As String = "|", _
    Optional ByVal OpenOrSave As Boolean = True, _
    Optional ByVal DialogView As OFN_DialogView = 0) As Variant
  Dim FF() As String, FM() As String
  Dim i As Long
  Dim OFName As OPENFILENAME
  Dim fso As Object
  Dim Result As Boolean

  Set fso = CreateObject("Scripting.FileSystemObject")

  OFName.lStructSize = Len(OFName) 'Set Size of structure
  OFName.hwndOwner = GetActiveWindow 'Set the parent window
  OFName.hInstance = GetCurrentThreadId 'Set the application's instance

  'Parse the filter
  FF = Split(FileFilter, FileFilterDelimiter)
  FM = Split(FileMask, FileFilterDelimiter)
  If UBound(FF) > UBound(FM) Then
    i = UBound(FM) + 1
    ReDim Preserve FM(0 To UBound(FF)) As String
    For i = i To UBound(FM)
      FM(i) = "*.*"
    Next
  ElseIf UBound(FM) > UBound(FF) Then
    i = UBound(FF) + 1
    ReDim Preserve FF(0 To UBound(FM)) As String
    For i = i To UBound(FF)
      FF(i) = "Dateien"
    Next
  End If

  'Build the filter
  For i = 0 To UBound(FF)
    OFName.lpstrFilter = OFName.lpstrFilter & FF(i) & " (" & FM(i) & ")" & vbNullChar & _
      FM(i) & vbNullChar
  Next
  OFName.lpstrFilter = OFName.lpstrFilter & vbNullChar

  'Select a filter
  If FilterIndex < 1 Then FilterIndex = 1 Else If FilterIndex > UBound(FF) + 1 Then _
    FilterIndex = UBound(FF) + 1
  OFName.nFilterIndex = FilterIndex

  'set the maximum length of a returned file
  OFName.nMaxFile = 4096
  'Setup default filename
  OFName.lpstrFile = fso.GetFileName(InitialFileName)
  'Setup default extension
  OFName.lpstrDefExt = fso.GetExtensionName(InitialFileName) & vbNullChar
  'create a buffer for the file
  OFName.lpstrFile = OFName.lpstrFile & String$(OFName.nMaxFile - Len(OFName.lpstrFile), _
    vbNullChar)

  'Set the maximum length of a returned file title
  OFName.nMaxFileTitle = MAX_PATH
  'Create a buffer for the file title
  OFName.lpstrFileTitle = String$(OFName.nMaxFileTitle, vbNullChar)
  'Set the initial directory
  OFName.lpstrInitialDir = InitialPath & vbNullChar
  'Set the title
  OFName.lpstrTitle = Title

  'Set flags
  OFName.flags = flags
  If MultiSelect Then
    OFName.flags = OFName.flags Or OFN_ALLOWMULTISELECT
  Else
    OFName.flags = OFName.flags And Not OFN_ALLOWMULTISELECT
  End If

  Select Case DialogView
    Case LargeIcon, List, Report, SmallIcon, Thumbnails
      OFN_View = DialogView
      OFName.flags = OFName.flags Or OFN_ENABLEHOOK Or OFN_EXPLORER
      OFName.lpfnHook = GetAddressOf(AddressOf OFNHookProc)
  End Select

  'Show the dialog
  If OpenOrSave Then
    Result = GetOpenFileNameA(OFName)
  Else
    Result = GetSaveFileNameA(OFName)
  End If
  If Result Then
    i = lstrlenA(OFName.lpstrFile)
    If MultiSelect Then
      If OFName.nFileExtension > 0 Then
        GetFileName = Array(Mid$(OFName.lpstrFile, 1, i))
      Else
        InitialPath = Mid$(OFName.lpstrFile, 1, i)
        OFName.lpstrFile = Mid$(OFName.lpstrFile, i + 2)
        i = InStr(OFName.lpstrFile, vbNullChar & vbNullChar)
        If i = 0 Then
          GetFileName = Array(InitialPath)
        Else
          OFName.lpstrFile = Mid$(OFName.lpstrFile, 1, i - 1)
          FF = Split(OFName.lpstrFile, vbNullChar)
          For i = 0 To UBound(FF)
            FF(i) = fso.BuildPath(InitialPath, FF(i))
          Next
          GetFileName = FF
        End If
      End If
    Else
      GetFileName = Mid$(OFName.lpstrFile, 1, lstrlenA(OFName.lpstrFile))
    End If
  Else
    If MultiSelect Then
      GetFileName = Array()
    Else
      GetFileName = False
    End If
  End If
End Function

Private Function OFNHookProc(ByVal hDialog As Long, ByVal Message As Long, _
    ByVal wParam As Long, ByRef lParam As NMHDR) As Long
  Dim hWndLVParent As Long, hWndListView As Long
  Static blnInitDone As Boolean
  Static blnFirstView As Boolean
  If Message = WM_NOTIFY Then
    Select Case lParam.code
      Case CDN_INITDONE
        'Dialog wurde initialisiert
        blnInitDone = True
        blnFirstView = False
      Case CDN_FOLDERCHANGE
        ' Es wurde in einen Ordner gewechselt.
        If blnInitDone Or OFN_View = Report Then
          GoSub SwitchView
          blnInitDone = False
        End If
      Case CDN_SELCHANGE
        'Der Anwender hat seine Auswahl geändert.
        If Not blnFirstView Then
          'Ansicht umschalten (Geht sonst nicht unter Win2K!?)
          GoSub SwitchView
          blnFirstView = True
        End If
      Case CDN_SHAREVIOLATION
        'Eine Zugriffsverletzung ist aufgetreten.
      Case CDN_HELP
        'Der Anwender hat den Hilfe-Button betätigt.
      Case CDN_FILEOK
        'Der Anwender hat den OK-Button betätigt.
      Case CDN_TYPECHANGE
        'Die Dateityp-Vorauswahl wurde geändert.
    End Select
  End If
  OFNHookProc = Message
  Exit Function

SwitchView:
  hWndLVParent = FindWindowEx(GetParent(hDialog), 0, "ShellDll_DefView", vbNullString)
  If hWndLVParent <> 0 Then SendMessage hWndLVParent, WM_COMMAND, ByVal OFN_View, ByVal 0&
  If OFN_View = Report Then
    hWndListView = FindWindowEx(hWndLVParent, 0, "SysListView32", vbNullString)
    If hWndListView <> 0 Then ColumnAutoSize hWndListView
  End If
  Return
End Function

Private Function ColumnAutoSize(ByVal hwnd As Long, _
    Optional ByVal Column As Long = -1, Optional ByVal Width As Long = 0) As Variant
  'Spaltenbreite optimal einstellen
  Const LVM_FIRST = &H1000
  Const LVM_GETITEMCOUNT = (LVM_FIRST + 4)
  Const LVM_GETCOLUMNWIDTH = (LVM_FIRST + 29)
  Const LVM_SETCOLUMNWIDTH = (LVM_FIRST + 30)
  Const LVM_GETHEADER = (LVM_FIRST + 31)
  Const LVSCW_AUTOSIZE = -1
  Const LVSCW_AUTOSIZE_USEHEADER = -2

  Const HDM_FIRST As Long = &H1200
  Const HDM_GETITEMCOUNT As Long = (HDM_FIRST + 0)

  Dim flags As Long, hWndHeader As Long
  Dim Count As Long
  Static Result

  hWndHeader = SendMessage(hwnd, LVM_GETHEADER, 0, 0)
  Count = SendMessage(hWndHeader, HDM_GETITEMCOUNT, 0&, 0&)
  Select Case Column
    Case Is > 0
      SendMessage hwnd, LVM_SETCOLUMNWIDTH, ByVal Column - 1, ByVal Width
      Result = Column <= Count
    Case 0
      ReDim Result(0 To Count - 1) As Long
      For Column = 0 To Count - 1
        Result(Column) = SendMessage(hwnd, LVM_GETCOLUMNWIDTH, ByVal Column, 0)
      Next
    Case Else
      If IsEmpty(Result) Then
        ReDim Result(0 To Count - 1) As Long
      Else
        ReDim Preserve Result(0 To Count - 1) As Long
      End If
      LockWindowUpdate hwnd
      For flags = LVSCW_AUTOSIZE_USEHEADER To LVSCW_AUTOSIZE
        For Column = 0 To Count - 1
          SendMessage hwnd, LVM_SETCOLUMNWIDTH, ByVal Column, ByVal flags
          Width = SendMessage(hwnd, LVM_GETCOLUMNWIDTH, ByVal Column, 0)
          If Width > Result(Column) Then Result(Column) = Width
        Next
      Next
      For Column = 0 To Count - 1
        Width = Result(Column)
        SendMessage hwnd, LVM_SETCOLUMNWIDTH, ByVal Column, ByVal Width
      Next
      LockWindowUpdate False
  End Select
  ColumnAutoSize = Result
End Function

The solution to silence Cortana forever is found in the Windows Volume Mixer.

1. Open Volume Mixer
(Right-click or press and hold the Volume icon on the lower right corner, called the notification area in the taskbar, and click Open Volume Mixer)

2. Side scroll PAST ALL the different vertical sliders till you find the ONE slider named "Cortana." My Cortana slider was found at the far right end, but this may differ for you.

3. Slide the Cortana slider all the way down.

Cortana will be MUTED FOREVER -- BUT YOU WILL STILL HAVE ALL YOUR OTHER SOUNDS WORKING AT NORMAL VOLUME as they are set in your mixer. Volume Mixer is very handy for customizing your sound experience in Windows. I use it a lot for many apps like Skype, music, and even browsers sometimes. Volume Mixer allows you individually adjust the volume of each app making sound. The main system volume (called "Speakers" in the Volume Mixer) adjusts all app volumes simultaneously while maintaining the proportionate volume of each app as they are set in the mixer. So the mixer and the main volume work together.

So setting Cortana to silent in Volume Mixer means you can still use her/it, but it/she is always silent.

Alternatively, you have the option of adjusting her volume to very quiet but still audible. It's all doable in the Mixer.
