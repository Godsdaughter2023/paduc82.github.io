Function Grouper(varRange, y)
    Grouper = ""
    First = True
    For Each ce In varRange
      If Not First And ce = y Then
        Grouper = Grouper & ", " & ce.Offset(0, -1).Value

dave peterson say:

I put some test data in A1:a2 of Sheet1.

I put a table in A1:B10 of Sheet2.  The code will match up the values in A1 (and
A2) of Sheet1 to the values in B1:B10 of sheet2.  If there's a match, then it
retrieves the values and formatting from the column to the left (A1:A10 of
sheet2).

And then it plops this value in the cell to the right of the original cell
(B1:B2 of sheet2).

Option Explicit
Sub DoTheGrouping()
    Dim myCell As Range
    Dim myRng As Range
    Dim myRngToInspect As Range

    Set myRng = Worksheets("Sheet1").Range("a1:A2")
    Set myRngToInspect = Worksheets("Sheet2").Range("b1:b10")

    For Each myCell In myRng.Cells
        Call GrouperValFormat(CurCell:=myCell, LookAtRng:=myRngToInspect)
    Next myCell

End Sub
Sub GrouperValFormat(CurCell As Range, LookAtRng As Range)
    Dim myCell As Range
    Dim myStr As String
    Dim mySeparator As String
    Dim TotChars As Long
    Dim cCtr As Long

    mySeparator = ", "
    myStr = ""
    For Each myCell In LookAtRng.Cells
        If myCell.Value = CurCell.Value Then
            myStr = myStr & mySeparator & myCell.Offset(0, -1).Value
        End If
    Next myCell

    If myStr = "" Then
        'do nothing
    Else
        'chop the separator at the beginning
        myStr = Mid(myStr, Len(mySeparator) + 1)
    End If

    With CurCell.Offset(0, 1)
        .Value = myStr

        If myStr = "" Then
            'do nothing
        Else
            TotChars = 0
            For Each myCell In LookAtRng.Cells
                If myCell.Value = CurCell.Value Then
                    For cCtr = 1 To Len(myCell.Offset(0, -1).Value)
                        With .Characters(Start:=TotChars + cCtr, Length:=1).Font
                            .ColorIndex = myCell.Offset(0, -1).Characters _
                                           (Start:=cCtr, Length:=1) _
                                                .Font.ColorIndex
                            .Bold = myCell.Offset(0, -1).Characters _
                                           (Start:=cCtr, Length:=1).Font.Bold
                        End With
                    Next cCtr
                    TotChars = TotChars + Len(myCell.Offset(0, -1).Value) _
                                  + Len(mySeparator)
                End If
            Next myCell
        End If
    End With

End Sub
