Sub MkDir_SplitFiles()
'Apr. 01, 2016
Const Nmax As Long = 800 '<<number of files, in each folder
Dim N As Long, t As Long, r As Long
Dim sPath
sPath = "c:\folderA\folderB\MAIN FOLDER\"  '<< path, main folder
Dim obj As Object, mainFd As Object, Fd As Object, fl As Object
Dim targetFd
targetFd = "c:\Users\Tasos\Desktop\" & Format(Now, "yyyymmdd_hhmmss") & "\" '<< change path
MkDir targetFd
t = 1
Dim newFd
newFd = targetFd & "\folder" & t & "\"
MkDir newFd

Sheets.Add
r = 1
Cells(r, 1) = "folder" & t
Set obj = CreateObject("Scripting.FileSystemObject")
Set mainFd = obj.getfolder(sPath)
N = 0
For Each Fd In mainFd.subfolders
For Each fl In obj.getfolder(Fd).Files
N = N + 1
r = r + 1
Cells(r, 1) = fl.Name
obj.CopyFile fl.Path, newFd
If N = Nmax Then
t = t + 1
newFd = targetFd & "\folder" & t & "\"
MkDir newFd
N = 0
r = r + 2
Cells(r, 1) = "folder" & t
End If
Next
Next
End Sub

These are the XML tags for the "normal" paragraph:

<w:p w:rsidR="003B482E" w:rsidRDefault="0080199D"><w:

and these are the tags for the indented paragraph

<w:p w:rsidR="003B482E" w:rsidRDefault="0080199D" w:rsidP="0080199D"><w:pPr><w:

If the superfluous tags are removed, the indentation is removed.
