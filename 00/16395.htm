john sr wilson say:

This should only affect selected slides (CTRL Click to select what you need)

Sub try_this()
    Dim oplc As Shape
    Dim osld As Slide
    Dim L As Long
    Dim oshp As Shape
    Dim otr As TextRange
    For Each osld In ActiveWindow.Selection.SlideRange
    For Each oplc In osld.Shapes
            If oplc.Type = msoPlaceholder Then
                If oplc.PlaceholderFormat.Type = ppPlaceholderObject Then
                 Exit For
                End If
            End If
        Next oplc
        For L = osld.Shapes.Count To 1 Step -1
            If Not osld.Shapes(L).Type = msoPlaceholder Then
                If osld.Shapes(L).HasTextFrame Then
                    If osld.Shapes(L).TextFrame.HasText Then
                        oplc.TextFrame.TextRange.InsertAfter osld.Shapes(L).TextFrame.TextRange.Text
                    End If
                End If
                osld.Shapes(L).Delete
            End If
        Next L
    Next osld
End Sub
