     .InitialFileName = DestPath
     If Not .Show Then Exit Sub
     DestPath = .SelectedItems(1)
     If Right(DestPath, 1) <> "\" Then DestPath = DestPath & "\"
  End With
 
  'Ask for the cells with filenames / filemasks
  On Error Resume Next
  Set Where = Application.InputBox("Select cells with filemasks", Type:=8)
  If Where Is Nothing Then Exit Sub
  Set Where = Intersect(Where, Where.Parent.UsedRange)
  If Where Is Nothing Then Exit Sub
 
  'Build the mask array
  FileMasks = Array()
  For Each R In Where
    If Not IsEmpty(R) Then
      ReDim Preserve FileMasks(0 To UBound(FileMasks) + 1)
      FileMasks(UBound(FileMasks)) = R.Value
    End If
  Next
 
  With FS
    'Perform the file search
    .LookIn = SourcePath
    .FileName = FileMasks
    .SearchSubFolders = True
    If .Execute = 0 Then
      MsgBox "No files found"
      Exit Sub
    End If
    
    'Copy the files
    For Each ThisFile In .FoundFiles
      FName = Mid(ThisFile, InStrRev(ThisFile, "\") + 1)
      If Dir(DestPath & FName) <> "" Then
        Select Case MsgBox(DestPath & FName & vbCrLf & "exits. Overwrite?", vbYesNoCancel)
          Case vbNo
            GoTo SkipFile
          Case vbCancel
            Exit Sub
        End Select
      End If
      VBA.FileCopy ThisFile, DestPath & FName
SkipFile:
    Next
  End With
End Sub
