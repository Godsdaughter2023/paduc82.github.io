Option Explicit

Private Sub CommandButton1_Click()
    If Me.TextBox1.Text = "" Then
        MsgBox "Enter a filename, without extension"

        GoTo lbl_Exit
    End If

        MsgBox "Select the document type"
        Me.ComboBox1.SetFocus
        GoTo lbl_Exit
    End If
    Me.Hide
    Me.Tag = 1
lbl_Exit:
    Exit Sub
End Sub

Me.Hide

End Sub

Private Sub UserForm_Initialize()
    With Me
        Me.Caption = "Save Documents"
        Me.CommandButton1.Caption = "Continue"
        Me.CommandButton2.Caption = "Cancel"

            .AddItem

            .List(.ListCount - 1, 2) = "C:\Desktop\test5\"

You can add as many document types as you wish as long as you include the three associated folders as shown. You can repeat folders, but you must not omit any or it will crash.

The main macro is modified to use this userform as follows:

Sub Macro1()
Dim strpath1 As String
Dim strpath2 As String
Dim strpath3 As String
Dim strName As String
Dim oFrm As New UserForm1

    With oFrm
        .Show
        If .Tag = 0 Then GoTo lbl_Exit
        strName = .TextBox1.Text
        strpath1 = .ComboBox1.Column(1)
        strpath2 = .ComboBox1.Column(2)
        strpath3 = .ComboBox1.Column(3)
    End With
    Unload oFrm

    'create the folders if not present
    CreateFolders strpath1
    CreateFolders strpath2
    CreateFolders strpath3
    'Save to strpath2
    ActiveDocument.SaveAs Filename:=strpath2 & strName & ".docx", _
                          FileFormat:=wdFormatXMLDocument, AddToRecentFiles:=False
    'Saveto strpath1
    ActiveDocument.SaveAs Filename:=strpath1 & strName & ".docx", _
                          FileFormat:=wdFormatXMLDocument, AddToRecentFiles:=False
    'Save as PDF in strpath3
    ActiveDocument.ExportAsFixedFormat _
            OutputFileName:=strpath3 & strName & ".pdf", _
            ExportFormat:=wdExportFormatPDF, _
            OpenAfterExport:=False, _
            OptimizeFor:=wdExportOptimizeForPrint, _
            Range:=wdExportAllDocument, _
            From:=1, To:=1, _
            Item:=wdExportDocumentContent, _
            IncludeDocProps:=True, _
            KeepIRM:=True, _
            CreateBookmarks:=wdExportCreateNoBookmarks, _
            DocStructureTags:=True, _
            BitmapMissingFonts:=True, _
            UseISO19005_1:=False
    'Print the document
    Application.PrintOut Filename:="", Range:=wdPrintAllDocument, Item:= _
                         wdPrintDocumentWithMarkup, Copies:=1, Pages:="", PageType:= _
                         wdPrintAllPages, Collate:=True, Background:=True, PrintToFile:=False, _
                         PrintZoomColumn:=0, PrintZoomRow:=0, PrintZoomPaperWidth:=0, _
                         PrintZoomPaperHeight:=0
lbl_Exit:
    Exit Sub
End Sub

daniel pineault say:

I'd use a function such as:

Function GetXLSWrkBkCount() As Long

'Return the count of the number of currently open Excel workbooks

'Uses Late Binding, so no reference libraries required

    Dim oExcel          As Object
 
    On Error Resume Next
    Set oExcel = GetObject(, "Excel.Application")    'Bind to existing instance of Excel

    If Err.Number <> 0 Then    'Could not get instance of Excel, so create a new one
        Err.Clear
        GoTo Error_Handler_Exit
    End If
    On Error GoTo Error_Handler
   
    GetXLSWrkBkCount = oExcel.workbooks.Count
 
Error_Handler_Exit:
    On Error Resume Next
    Set oExcel = Nothing
    Exit Function
 
Error_Handler:
    MsgBox "The following error has occured" & vbCrLf & vbCrLf & _
           "Error Number: " & Err.Number & vbCrLf & _
           "Error Source: Export2XLS" & vbCrLf & _
           "Error Description: " & Err.Description _
           , vbOKOnly + vbCritical, "An Error has Occured!"
    Resume Error_Handler_Exit
End Function

and then call it

Public Sub test()

    Debug.Print "there are :"; GetXLSWrkBkCount; " open workbooks."

End Sub
