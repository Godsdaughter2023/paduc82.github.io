Assuming you have named the slide correctly

Sub moveTo()
Dim osld As Slide
Set osld = ActivePresentation.Slides("PrePracticeSlide")
SlideShowWindows(1).View.GotoSlide (osld.SlideIndex)

You also do not have to name the slide you can use it's unique ID

Select the slide and type ?ActiveWindow.Selection.SlideRange(1).SlideId into the Immediate window (ENTER)

Sub moveTo_byID()
Dim osld As Slide

This works for me:

Dim numPages As Integer

   ActiveDocument.PageSetup.DifferentFirstPageHeaderFooter = True
   With ActiveDocument.Sections(1).Footers(wdHeaderFooterFirstPage)
    .Range.Paragraphs.Alignment = wdAlignParagraphRight
    .Range.Bookmarks.add name:="Bar", _
  Range:=ActiveDocument.Sections(1).Footers(wdHeaderFooterFirstPage).Range
   End With
 
  ActiveDocument.Bookmarks("Bar").Select

   .InsertFile ("C:\Logo.doc")
   .Font.Size = 8

   .TypeText Text:=Chr(13) & Chr(13) & "Special text" & Chr(13)
  End With
 
  ActiveDocument.Sections(1).Footers(1) _
    .Range.Bookmarks.add name:="Bar2", _
    Range:=ActiveDocument.Sections(1).Footers(1).Range
 

 

    .Paragraphs.Alignment = wdAlignParagraphRight
   .InsertFile ("C:\Logo.doc")
   .Font.name = "Arial"
   .Font.Size = 8
    .TypeText Text:=Chr(13) & Chr(13) & "" & Chr(13)
End With
 
  ActiveDocument.Sections.Last.Footers.Item(1).PageNumbers.add 1, FirstPage:=True
  numPages = ActiveDocument.Range.Information(wdActiveEndPageNumber)
 

 ActiveDocument.Sections(1).Footers(wdHeaderFooterFirstPage).PageNumbers.ShowFirstPageNumber = False
 End If
 
  ActiveDocument.ActiveWindow.View.Type = wdNormalView
 
  With ActiveDocument.ActiveWindow.View
    .Type = wdPrintView
    .SeekView = wdSeekMainDocument
  End With

dirk goldgar say:

Given that table structure and the RowGenerator table I described above, something along the lines of the following query should give what you need:

----- start of SQL -----

SELECT FormRows.*, TblInvB.LineNo, TblInvB.ItemNo, TblInvB.ItemDescription, TblInvB.UnitPrice, TblInvB.Qty, TblInvB.ExtPrice
FROM
(
    SELECT TblInvA.*, RowGenerator.RowNum
    FROM TblInvA, RowGenerator
) AS FormRows
LEFT JOIN TblInvB

ON (FormRows.FormNo = TblInvB.FormNo) AND (FormRows.RowNum = TblInvB.LineNo)
ORDER BY FormRows.FormNo, FormRows.RowNum;
----- end of SQL -----

That is based on the understanding that the field "LineNo" in TblInvB has consecutive ascending values from 1 to however many detail lines there are foer each FormNo, but not more than 10.  I believe that is what you said.
