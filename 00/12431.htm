Option Explicit

Function WHours(rngV As Range) As Double

    Dim c As Range

    Dim v As Variant

    Dim dteS As Date

    Dim dteF As Date

    For Each c In rngV

        If HasNumbers(c.Value) Then

            v = Split(Replace(Replace(Replace(LCase(c.Value), " ", "-"), "p", ":00 PM"), "a", ":00 AM"), "-")

            dteF = TimeValue(v(UBound(v)))

            dteS = TimeValue(v(UBound(v) - 1))

            WHours = WHours + (IIf(dteF < dteS, 1, 0) + dteF - dteS) * 24

        End If

    Next c

End Function

Function HasNumbers(strV As String) As Boolean

    Dim i As Integer

    HasNumbers = False

    For i = 1 To Len(strV)

        If IsNumeric(Mid(strV, i, 1)) Then

            HasNumbers = True

            Exit Function

        End If

    Next i
