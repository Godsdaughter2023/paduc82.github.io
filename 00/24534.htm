Sub Highlight_Rows_Sheet2()
Dim ws1 As Worksheet, ws2 As Worksheet
Set ws1 = Sheets("Sheet1") ' << sheet1 name
Set ws2 = Sheets("Sheet2") ' << sheet2 name
Dim r1 As Range, r2 As Range
Dim nColor As Long ' each color has a unique number
Dim c2 As Long
c2 = ws2.Range("A1").CurrentRegion.Columns.Count
For Each r1 In ws1.Range("A:A").SpecialCells(xlCellTypeConstants)
If r1.Interior.ColorIndex <> xlNone Then
nColor = r1.Interior.Color
If WorksheetFunction.CountIf(ws2.Range("A:A"), r1) > 0 Then
Set r2 = ws2.Range("A:A").Find(What:=r1, SearchOrder:=xlByRows, SearchDirection:=xlPrevious)
ws2.Cells(r2.Row, 1).Resize(, c2).Interior.Color = nColor
End If
End If
Next
End Sub
