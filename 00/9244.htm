Try this,

Sub mcrAddPurch2Mat()
    For Each c In Sheets("PurchStock").Range("A1:A55")
        If Len(c) Then
            iMatch = Application.Match(c.Value, Sheets("MatStock").Range("A1:A55"), False)
            If IsError(iMatch) Then
                c.Interior.ColorIndex = 3 'mark not found as red
            Else
                Sheets("MatStock").Cells(iMatch, 2) = c.Offset(, 1).Value + _
                  Sheets("MatStock").Cells(iMatch, 2).Value 
            End If
        End If
    Next c
End Sub

You will note that I added a red highlight to any cell in PurchStock that is not found in MatStock.
