{penlight} say:

I realized the issue is that VBA does not allow discontiguous selections. This is why you can't copy all blue text. 

The closest workaround I found is to copy all blue text into a new document to separate it out. 

Sub separatebluetext()
Application.ScreenUpdating = False
Dim StrTxt As String, Doc As Document
With ActiveDocument.Range
  With .find
    .ClearFormatting
    .Replacement.ClearFormatting
    .Text = ""
    .Replacement.Text = ""
    .Font.color = RGB(0, 0, 255)
    .Highlight = True
    .Format = True
    .Forward = True
    .Wrap = wdFindStop
    .Execute
  End With
  Do While .find.Found
    StrTxt = StrTxt & .Text
    If Right(.Text, 1) <> Space(1) Then StrTxt = StrTxt & Space(1)
    .Collapse wdCollapseEnd
    .find.Execute
  Loop
End With
Set Doc = Documents.Add
Doc.Range.Text = StrTxt
Application.ScreenUpdating = True
End Sub

a. user say:

You'll need to qualify your references for Selection.  For example, you'll need to replace...

With Selection

with

With .ActiveWindow.Selection

And, you'll need to replace...

Selection.Paste

with

.ActiveWindow.Selection.Paste

Also, if you only want to copy Columns A through C from your workbook, replace...

Range("A1", Range("C1").End(xlDown).End(xlToRight)).Copy

with

Range("A1", Range("C1").End(xlDown)).Copy

Maybe something like this...

Sub CreateWordDoc()

    Dim wdApp As Word.Application
    Dim Savename As String
    
    Set wdApp = New Word.Application
   
    Range("A1", Range("C1").End(xlDown)).Copy
    
    With wdApp
       .Visible = True
       .Activate
       .Documents.Add
        With .ActiveWindow.Selection
              .TypeText Text:="My Report"
              .ParagraphFormat.Alignment = wdAlignParagraphCenter
              .BoldRun
              .Font.Size = 16
              .TypeText "My Report"
              .BoldRun
              .TypeParagraph
              .Font.Size = 11
              .ParagraphFormat.Alignment = wdAlignParagraphLeft
              .TypeParagraph
              .Paste
        End With
        Savename = Environ("UserProfile") & "\My Documents\DFAC Weekly Task Report" & Format(Now, "yyyy-mm-dd") & ".doc"
        .ActiveDocument.SaveAs Savename
        .ActiveDocument.Close
        .Quit
    End With

End Sub
