gord dibben say:

Easy with a User Function installed in a General Module in your workbook.

Enter date in Sheet1 A1

Select Sheet2 then Shift + Click on last sheet.

In Sheet2 A1 enter   =PrevSheet(A1) + 1

Format in your favorite date format then ungroup the sheets.

Function PrevSheet(Rg As Range)
'accounts for more than one workbook open
'Bob Phillips  October 4, 2009
Dim N As Variant
    With Application.Caller.Parent
        N = .Index
        Do
            If N = 1 Then
                PrevSheet = CVErr(xlErrRef)
                Exit Do
            ElseIf TypeName(.Parent.Sheets(N - 1)) <> "Chart" And _
                .Parent.Sheets(N - 1).Visible = xlSheetVisible Then
                PrevSheet = .Parent.Sheets(N - 1).Range(Rg.Address).Value
                Exit Do
            End If
            N = N - 1
        Loop
    End With
End Function
