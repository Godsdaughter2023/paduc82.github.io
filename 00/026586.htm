This is actually not the relevant setting, as it applies only to line spacing within a paragraph. The correct option is "Suppress space before after a hard page or column break," which must not be enabled. Unfortunately, it is enabled by default in Word 2013, and there is no way to disable it except to set the Compatibility to Word 2010, as follows:

Press Alt+F11 to open the macro editor, and press Ctrl+G to open the Immediate window at the bottom of the editor. In that window, paste this line and then press Enter:

 

ActiveDocument.SetCompatibilityMode wdWord2010

 

Close the macro editor. Then open the Options dialog, go to the Advanced options, and scroll to the bottom. Change the dropdown "Lay out this document as if created in" to "Microsoft Word 2010". At that point, the list of 2010 layout options will appear below the dropdown.

Sub WordToExcel()
' Set reference to Microsoft Word 12.0 Object Library in the VBE
' via Tools > References...
'
    Dim oWd As Word.Application
    Dim wdDoc As Word.Document
    Dim wdRng As Word.Range
    Dim i As Long, j As Long
    Dim bWdStartedByUs As Boolean
   
    On Error Resume Next
    Set oWd = GetObject(, "Word.Application")
    On Error GoTo ErrorHandler
    If oWd Is Nothing Then
        Set oWd = CreateObject("Word.Application")
        bWdStartedByUs = True
    End If
   
    Set wdDoc = oWd.ActiveDocument
'    Set oDoc = oWd.Documents.Open("C:\Work\Files\Macros\wordtoexcel\Test1.docx")
   
    For i = 1 To wdDoc.Paragraphs.Count
        If wdDoc.Paragraphs(i).Style = "Heading 1" Then
            If InStr(wdDoc.Paragraphs(i), "Procedure") > 0 Then
                If wdDoc.Paragraphs(i + 1).Style <> "Heading 1" Then
                    Set wdRng = wdDoc.Paragraphs(i + 1).Range
                    For j = i + 2 To wdDoc.Paragraphs.Count
                        If wdDoc.Paragraphs(j).Style = "Heading 1" Then
                            i = j - 1
                            Exit For
                        Else
                            wdRng.SetRange Start:=wdRng.Start, End:=wdDoc.Paragraphs(j).Range.End
                        End If
                    Next j
                Else
                    MsgBox "This Procedure section is empty!"
                End If
            End If
        End If
        If Not wdRng Is Nothing Then
            wdRng.Copy
           
             'IN EXCEL:
            Range("A" & Range("A" & ActiveSheet.Rows.Count).End(xlUp).Row + 2) _
                .PasteSpecial xlPasteValues
           
            Set wdRng = Nothing
        End If
    Next i

ErrorHandler:
    Word.Application.ScreenUpdating = True
    Excel.Application.ScreenUpdating = True
   
'    If bWdStartedByUs Then oWd.Quit
    Set wdDoc = Nothing
    Set oWd = Nothing
End Sub
