ossiemac say:

Either of the following 2 examples. Note that it is better to use either Ucase or Lcase then Proper when comparing strings for worksheet names because you do not know how the user will enter the name and irrespective of the case, 2 worksheets cannot have the same name.

 

Sub AddWorksheet_1()
    Dim Title As String
    Dim Param As String
    Dim Message As String
    Dim defaultRef As String
    Dim Sht As Worksheet
    Dim WorksheetExists As Boolean
   
    Title = "Ask for the parameter to analyse"
    Message = "What parameter are you analysing?"
    defaultRef = "Asat"
   
    Param = InputBox(Message, Title, defaultRef)
    
    For Each Sht In ThisWorkbook.Worksheets
        'Following line ignores case in comparison
        If UCase(Sht.Name) = UCase(Param) Then
            WorksheetExists = True
            Exit For
        End If
    Next Sht
   
    If WorksheetExists = False Then
        Worksheets.Add(After:=Worksheets(Worksheets.Count)).Name = Param
    Else
        MsgBox "Worksheet " & Param & " Exists"
    End If


End Sub

 


Sub AddWorksheet2()

    Dim Title As String
    Dim Param As String
    Dim Message As String
    Dim defaultRef As String
    Dim Sht As Worksheet
   
    Title = "Ask for the parameter to analyse"
    Message = "What parameter are you analysing?"
    defaultRef = "Asat"
   
    Param = InputBox(Message, Title, defaultRef)
   
    On Error Resume Next
    Set Sht = Worksheets(Param)
   
    If Err.Number > 0 Then      'Worksheet does not exist
        On Error GoTo 0         'Resume error trapping ASAP
        Worksheets.Add(After:=Worksheets(Worksheets.Count)).Name = Param
    Else
        On Error GoTo 0         'Resume error trapping ASAP
        MsgBox "Worksheet " & Param & " Exists"
    End If   

End Sub
