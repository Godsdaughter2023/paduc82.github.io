$lcid = "1033" #(3081 for en-au)

$tenantadminsiteurl = "https://xxxxx-admin.sharepoint.com/"

$tenantadminusername = "*** Email address is removed for privacy ***"

$tenantadminpassword = "xxxxxx"

$securepassword = convertto-securestring $tenantadminpassword -asplaintext -force

# path may have "15" instead of "16"

add-type -path "c:\program files\common files\microsoft shared\web server extensions\16\isapi\microsoft.sharepoint.client.dll"

add-type -path "c:\program files\common files\microsoft shared\web server extensions\16\isapi\microsoft.sharepoint.client.runtime.dll"

add-type -path "c:\program files\common files\microsoft shared\web server extensions\16\isapi\microsoft.sharepoint.client.taxonomy.dll"

$clientcontext = new-object microsoft.sharepoint.client.clientcontext($tenantadminsiteurl)

$credentials = new-object microsoft.sharepoint.client.sharepointonlinecredentials($tenantadminusername, $securepassword)

$clientcontext.credentials = $credentials

$taxonomysession = [microsoft.sharepoint.client.taxonomy.taxonomysession]::gettaxonomysession($clientcontext)

$taxonomysession.updatecache()

$clientcontext.load($taxonomysession)

$clientcontext.executequery()

$termstore = $taxonomysession.getdefaultkeywordstermstore()

$clientcontext.load($termstore)

$clientcontext.executequery()

$group = $termstore.creategroup("search dictionaries", [guid]"d87b6a37-c801-4a36-9046-6296d4779c87")

$clientcontext.load($group)

$clientcontext.executequery()

$termset1 = $group.createtermset("company inclusions", [guid]"c866ca65-f095-4a16-9249-028d500f7703", $lcid)

$clientcontext.load($termset1)

$clientcontext.executequery()

$termset1.setcustomproperty("iscustomdictionary", "true")

$termset1.owner = [string]::empty

$clientcontext.executequery()

$termset2 = $group.createtermset("company exclusions", [guid]"d41a92ec-33dc-4720-bd03-3a00287653e4", $lcid)

$clientcontext.load($termset2)

$clientcontext.executequery()

$termset2.setcustomproperty("iscustomdictionary", "true")

$termset2.owner = [string]::empty

$clientcontext.executequery()

$termset3 = $group.createtermset("query spelling inclusions", [guid]"22035077-2e30-450f-99ac-889a3bec24d8", $lcid)

$clientcontext.load($termset3)

Option 2

 

1. Show the values are % of column first

2. Drag row labels to column labels instead.  % of column will now become % of subtotal
