mike h.. say:

Hi,

 

This works on column A and puts the text in Col B. Right click the sheet tab, view code and paste the code below in. Close VB editoe and enter the values in Col A. Change the text you want to the correct text

 

Private Sub Worksheet_Change(ByVal Target As Range)
Dim Word As String, fcolor As Variant
If Target.Cells.Count > 1 Or IsEmpty(Target) Then Exit Sub
If Not Intersect(Target, Range("A:A")) Is Nothing Then
Application.EnableEvents = False
Select Case UCase(Target.Value)
    Case Is = "N"
        fcolor = vbRed
        Word = "This"
    Case Is = "D"
        fcolor = vbBlue
        Word = "That"
    Case Is = "R"
        fcolor = vbGreen
        Word = "Other"
End Select
Target.Font.Color = fcolor
Target.Offset(, 1) = Word
Application.EnableEvents = True
End If
End Sub
