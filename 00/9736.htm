hansv mvp say:

You have to use the newer Create method instead of Add to create the pivotcache. Create lets you specify the version of the pivotcache:

 

Sub MakeTable()
    Dim ws As Worksheet
    Dim PTCache As PivotCache
    Dim pt As PivotTable
    Dim PivotDataRange As Range

    Set ws = Worksheets.Add
    Set PivotDataRange = Sheets("CleanData").Range("DataRange")
    Set PTCache = ThisWorkbook.PivotCaches.Create _
        (SourceType:=xlDatabase, _
        SourceData:=PivotDataRange, _
        Version:=xlPivotTableVersion14)
    Set pt = PTCache.CreatePivotTable _
        (TableDestination:=ws.Cells(10, 2), _
        TableName:="allData2", _
        DefaultVersion:=xlPivotTableVersion14)
End Sub
