Public Sub SetDefaultsTheme(frm As Form)
'Purpose  : Set values for form & default controls.
'           Form must be open in Design View.
'DateTime : 5/21/2004 10:31
'Author   : Bill Mosca
 
    On Error Resume Next
 
    With frm
        .Section(acDetail).BackColor = eTmpltColor.FORM_BACK
        .Section(acHeader).BackColor = eTmpltColor.FORM_HEADER
        .Section(acFooter).BackColor = eTmpltColor.FORM_BACK
        .ScrollBars = 0     '0=neither;1=Horizontal;2=Vertical;3=Both
        .RecordSelectors = False
        .NavigationButtons = False
        .DividingLines = False
        .AutoResize = True
        .AutoCenter = True
        .AllowDesignChanges = False
 
    End With
 
    With frm.DefaultControl(acTextBox)
        .ForeColor = eTmpltColor.DETAIL_FORE_COLOR
        .FontName = pstrFont1
        .Height = 0.1979 * 1440
        .fontsize = 10
    End With
 
    With frm.DefaultControl(acLabel)
        .ForeColor = eTmpltColor.DETAIL_FORE_COLOR
        .FontName = pstrFont1
        .Height = 0.1979 * 1440
        .fontsize = 10
        .BackStyle = 1  'Normal 0=Transparent
        .BackColor = eTmpltColor.DETAIL_LBL_BACK
        .BorderStyle = 1    'Solid
        .BorderColor = eTmpltColor.DETAIL_BORDER
        .BorderWidth = 0
        .TextAlign = 2  '0=General;1=Left;2=Center;3=Right
 
    End With
 
    With frm.DefaultControl(acComboBox)
        .ForeColor = eTmpltColor.THEME_COLOR
        .FontName = pstrFont1
        .Height = 0.1979 * 1440
        .fontsize = 10
    End With
 
    With frm.DefaultControl(acListBox)
        .ForeColor = eTmpltColor.THEME_COLOR
        .FontName = pstrFont1
        .fontsize = 10
    End With
 
    With frm.DefaultControl(acOptionGroup)
        .BorderStyle = acEffectEtched   'Etched overrides all other border settings
        .BorderColor = 0
        .BorderWidth = 0
    End With
 
    With frm.DefaultControl(acCheckBox)
        .BorderStyle = acEffectSunken
        .BorderColor = 0
        .BorderWidth = 0
    End With
 
    DoCmd.Save acForm, frm.name
 
End Sub

  3. For both types of data source, the field names must contain only alphanumeric characters (No @,#,$,%,&,(,), etc) and the field names must not start with a numeric character (0-9). The number of characters in the field names, including spaces, must not be more than 40.

  4. For a data source in the form of an Excel worksheet, the data must start in the second row of the worksheet and there should be no empty rows within the range of data that is to be processed.

  5. If there are multiple addressees in either the To or CC fields of the data source, the email addresses need to be separated by a semi-colon.

NOTE: The MergeTools applications cannot handle “Compound” MergeFields such as the «AddressBlock» or «GreetingLine».  Instead of using those fields, you will need to insert the individual merge fields.
