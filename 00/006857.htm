You want something simple, so with the code below you can specify the exact columns or

The code looks up, starting in the last cell in each column, and stops at the first cell with data.

The code then uses the largest number, from all the columns searched, and adds 1 to it.

'---
Function WhereAmI() As Long
 Dim Rw As Long
 Dim BlankRow As Long
 Dim N As Long
 Dim rng As Range
 
 Set rng = ActiveSheet.Range(Columns(1), Columns(3)) ' <<<specify columns
'or
'Set rng = ActiveSheet.UsedRange ' alternative
 
 For N = rng.Column To rng.Columns.Count
'or
'For N = rng.Columns(1).Column To rng.Columns(rng.Columns.Count).Column ' alternative
    Rw = Cells(Rows.Count, N).End(xlUp).Row
    If Rw > BlankRow Then BlankRow = Rw
 Next
 WhereAmI = BlankRow + 1
End Function

'******CODE STARTS HERE - ok to copy this line also.
  'be sure to include $ symbols befor column and row identifiers.
  ' cell where you enter customer's name to get info about
  Const CustomerNameCell = "$H$2"
  ' H3, H4 and H5 (next 3 rows below the above cell)
  ' also used regardless of how you define it.
  '
Private Sub Worksheet_Change(ByVal Target As Range)
  'assumes this code is part of the main data sheet's code
  'and that there is a sheet named
  'WorkingDataSheet in the workbook
  ' If WorkingDataSheet does not exist, it will be created.
  '
  'some constants you can change to match setup
  Const srcNameCol = "A" ' column customer names are in
  Const srcDateCol = "B" ' column visit dates are in
  Const srcTimeCol = "D" ' column that time of visits are in
  Const firstDataRow = 2 ' row first customer entry is on
  'end of user redefinable Const values.
  '*************************************
 
  'these constants for working data sheet, leave alone
  Const wdNameCol = "A"
  Const wdDateCol = "B"
  Const wdTimeCol = "C"
  Const wdVisitCol = "D" ' where we will put time between same day visits
  Const wdUniqueCol = "E" ' where we combine name and date to create easy matchup list
  Dim srcWS As Worksheet
  Dim wdWS As Worksheet
  Dim lastRow As Long
  Dim tableRange As Range
  Dim anyTableCell As Range
  Dim sKey1 As Range
  Dim sKey2 As Range
  Dim sKey3 As Range
  Dim calcTimeFormula As String
  Dim makeUniqueFormula As String
  Dim multiVisitFormula1 As String
  Dim multiVisitFormula2 As String
  ' to track different days customer visited
  Dim visitDates() As Date
  ' visit loop counter to work through visitDates() as needed
  Dim VLC As Long
  ' name from H2 on source sheet
  Dim cName As String
  ' offset from customer name column to date of visit cell
  Dim offset2VDate As Integer
 
  'only runs rest of code if the one cell identified
  'at the address as CustomerNameCell changes
  If Target.Address <> CustomerNameCell Then
    Exit Sub
  End If
  'test if any work even needs to be done
  'must be at least 1 name in srcNameCol
  lastRow = ActiveSheet.Range(srcNameCol & Rows.Count).End(xlUp).Row
  If lastRow < firstDataRow Then
    Exit Sub ' nothing to work with
  End If
  'create formulas we will need later in R1C1 format
  calcTimeFormula = _
   "=IF(RC[-3]=R[1]C[-3],IF(RC[-2]=R[1]C[-2],(R[1]C[-1]-RC[-1])*1440," _
    & Chr$(34) & Chr$(34) & ")," _
    & Chr$(34) & Chr$(34) & ")"

  makeUniqueFormula = _
   "=RC[-4] & " & Chr$(34) & " : " & Chr$(34) & " & RC[-3]"
 
  multiVisitFormula1 = "=COUNTIFS(A:A,Sheet1!$H$2,D:D," _
   & Chr$(34) & ">" & Chr$(34) & " & 0,D:D," _
   & Chr$(34) & "<=" & Chr$(34) & " & 20)"

  multiVisitFormula2 = "=COUNTIFS(A:A,Sheet1!$H$2,D:D," _
   & Chr$(34) & ">" & Chr$(34) & " & 20)"
 
  Set srcWS = ActiveSheet
  cName = UCase(Trim(srcWS.Range(CustomerNameCell))) ' from H2
 
  ' improve performance, prevent screen flicker
  Application.ScreenUpdating = False
  On Error Resume Next
  Set wdWS = ThisWorkbook.Worksheets("WorkingDataSheet")
  If Err <> 0 Then
    ThisWorkbook.Worksheets.Add _
     after:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count)
    ActiveSheet.Name = "WorkingDataSheet"
    Set wdWS = ActiveSheet ' the WorkingDataSheet
    srcWS.Activate ' back to the source data sheet
    Err.Clear
  End If
  'hide the WorkingDataSheet from view
  wdWS.Visible = xlSheetHidden
  On Error GoTo 0 ' clear error trapping

  wdWS.Cells.Clear ' remove any previous results
  Set tableRange = srcWS.Range(srcNameCol & "1:" & srcNameCol & lastRow)
  tableRange.Copy
  wdWS.Range(wdNameCol & 1).PasteSpecial xlPasteValues
  Set tableRange = srcWS.Range(srcDateCol & "1:" & srcDateCol & lastRow)
  tableRange.Copy
  wdWS.Range(wdDateCol & 1).PasteSpecial xlPasteValues
  Set tableRange = srcWS.Range(srcTimeCol & "1:" & srcTimeCol & lastRow)
  tableRange.Copy
  wdWS.Range(wdTimeCol & 1).PasteSpecial xlPasteValues
  Application.CutCopyMode = False
'sort the data we have on the analysis sheet by NAME - DATE - Time of Day
'at this point all work, other than displaying the results is done
'on/to the wdWS worksheet, WorkingDataSheet
  lastRow = wdWS.Range(wdNameCol & Rows.Count).End(xlUp).Row
  Set tableRange = wdWS.Range(wdNameCol & "1:" & wdTimeCol & lastRow)
  Set sKey1 = wdWS.Range(wdNameCol & firstDataRow)
  Set sKey2 = wdWS.Range(wdDateCol & firstDataRow)
  Set sKey3 = wdWS.Range(wdTimeCol & firstDataRow)
  'this method of sorting will work with Excel 2003, 2007 and 2010
  tableRange.Sort Key1:=sKey1, Order1:=xlAscending, Key2:=sKey2, _
      Order2:=xlAscending, Key3:=sKey3, Order3:=xlAscending, Header:= _
      xlGuess, OrderCustom:=1, MatchCase:=False, Orientation:=xlTopToBottom, _
      DataOption1:=xlSortNormal, DataOption2:=xlSortNormal, DataOption3:= _
      xlSortNormal
  wdWS.Range(wdVisitCol & 1) = "TBV-Minutes"
  Set tableRange = wdWS.Range(wdVisitCol & firstDataRow & ":" _
   & wdVisitCol & lastRow)
  tableRange.FormulaR1C1 = calcTimeFormula
 
  wdWS.Range(wdUniqueCol & 1) = "Unique ID"
  Set tableRange = wdWS.Range(wdUniqueCol & firstDataRow & ":" _
   & wdUniqueCol & lastRow)
  tableRange.FormulaR1C1 = makeUniqueFormula
  ReDim visitDates(1 To 1)
  Set tableRange = wdWS.Range(wdNameCol & firstDataRow & ":" _
   & wdNameCol & lastRow)
  offset2VDate = Range(wdDateCol & 1).Column - Range(wdNameCol & 1).Column
  For Each anyTableCell In tableRange
    If UCase(Trim(anyTableCell)) = cName Then
      For VLC = LBound(visitDates) To UBound(visitDates)
        If visitDates(VLC) = anyTableCell.Offset(0, offset2VDate) Then
          Exit For ' already in list
        End If
      Next
      If VLC > UBound(visitDates) Then
        'new date, add to list
        visitDates(UBound(visitDates)) = anyTableCell.Offset(0, offset2VDate)
        ReDim Preserve visitDates(1 To UBound(visitDates) + 1)
      End If
    End If
  Next
  'visitDates() will have empty element - get rid of it
  If UBound(visitDates) > 1 Then
    ReDim Preserve visitDates(1 To UBound(visitDates) - 1)
  End If
  If UBound(visitDates) = 1 Then
    If visitDates(UBound(visitDates)) <> 0 Then
      srcWS.Range(CustomerNameCell).Offset(1, 0) = _
       "Visited on just " & UBound(visitDates) & " day."
    Else
      srcWS.Range(CustomerNameCell).Offset(1, 0) = _
       "No Visits Found."
    End If
  Else
    srcWS.Range(CustomerNameCell).Offset(1, 0) = _
     "Visited on " & UBound(visitDates) & " separate days."
  End If
  wdWS.Range("F2").Formula = multiVisitFormula1
  srcWS.Range(CustomerNameCell).Offset(2, 0) = _
   "Multiple visits on same day less than 20 minutes apart: " & wdWS.Range("F2")
  wdWS.Range("F3").Formula = multiVisitFormula2
  srcWS.Range(CustomerNameCell).Offset(3, 0) = _
   "Multiple visits on same day at least 20 minutes apart: " & wdWS.Range("F3")
'
  Set tableRange = Nothing
  Set srcWS = Nothing
  Set wdWS = Nothing
End Sub
