Sub ExtractData()

  Dim ws1, ws2 As Worksheet

  Dim r, x, Lrow1, Lrow2 As Long

  Set ws1 = Sheets("Sheet1")

  Set ws2 = Sheets("Sheet2")

  Lrow1 = ws1.Cells(Rows.Count, "A").End(xlUp).Row

  ws1.Range("A1").EntireRow.Insert

  ws1.Range("a1") = "---"

  ws1.Range("A1:A" & Lrow1 + 1).AdvancedFilter Action:=xlFilterInPlace, Unique:=True

  ws1.Range("A2:A" & Lrow1).Copy

  ws2.Range("A1").PasteSpecial xlValues

ws1.Range("A1:A" & Lrow1 + 1).AdvancedFilter Action:=xlFilterInPlace, Unique:=False

ws1.Range("A1").EntireRow.Delete

  Lrow2 = ws2.Cells(Rows.Count, 1).End(xlUp).Row

  For r = 1 To Lrow1

x = Application.Match(ws1.Cells(r, 1), ws2.Range("A1:A" & Lrow2), 0)

ws2.Cells(x, ws2.Cells(x, Columns.Count).End(xlToLeft).Column).Offset(, 1) = ws1.Cells(r, 2)

  Next r

End Sub

To run the macro, press ALT+8,
