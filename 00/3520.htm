You could use a macro similar to the following:

' Macro created 26/10/01 by Doug Robbins to update links in a document
'
Dim alink As Field, linktype As Range, linkfile As Range
Dim linklocation As Range, i As Integer, j As Integer, linkcode As Range
Dim Message, Title, Default, Newfile
Dim counter As Integer

 

counter = 0
For Each alink In ActiveDocument.Fields
    If alink.Type = wdFieldLink Then 

        Set linkcode = alink.Code
        i = InStr(linkcode, Chr(34))
        Set linktype = alink.Code
        linktype.End = linktype.Start + i
        j = InStr(Mid(linkcode, i + 1), Chr(34))
        Set linklocation = alink.Code
        linklocation.Start = linklocation.Start + i + j - 1
        If counter = 0 Then
            Set linkfile = alink.Code
            linkfile.End = linkfile.Start + i + j - 1
            linkfile.Start = linkfile.Start + i
            Message = "Enter the modified path and filename following this Format " & linkfile
            Title = "Update Link"
            Default = linkfile
            Newfile = InputBox(Message, Title, Default)
        End If
        linkcode.Text = linktype & Newfile & linklocation
        counter = counter + 1
    End If
Next alink

Assuming it's only the name that changes (i.e. not the path as well), try:

Sub UpdateExcelLinks()
Application.ScreenUpdating = False
Dim i As Long, j As Long, StrNm As String

Sub UpdateExcelLinks()
Application.ScreenUpdating = False
Dim Rng As Range, Fld As Field, Shp As Shape, iShp As InlineShape, i As Long, j As Long, StrNm As String
StrNm = Trim(InputBox("What is the new Excel filename & extension?", "Filename Input"))
If StrNm = "" Or InStr(StrNm, ".xls") = 0 Then Exit Sub
With ActiveDocument
  For Each Rng In .StoryRanges
    With Rng
      For i = .Fields.Count To 1 Step -1
        With .Fields(i)
          If .Type = wdFieldLink Then
            j = j + 1
            With .LinkFormat
              .SourceFullName = .SourcePath & "\" & StrNm
            End With
          End If
        End With
      Next
      For i = .ShapeRange.Count To 1 Step -1
        With .ShapeRange(i)
          If Not .LinkFormat Is Nothing Then
            With .LinkFormat
              If .SourceFullName <> .SourcePath & "\" & StrNm Then
                j = j + 1
                .SourceFullName = .SourcePath & "\" & StrNm
              End If
            End With
          End If
        End With
      Next
      For i = .InlineShapes.Count To 1 Step -1
        With .InlineShapes(i)
          If Not .LinkFormat Is Nothing Then
            With .LinkFormat
              If .SourceFullName <> .SourcePath & "\" & StrNm Then
                j = j + 1
                .SourceFullName = .SourcePath & "\" & StrNm
              End If
            End With
          End If
        End With
      Next
    End With
  Next
End With

pennyrt say:

In word 2010, i'm using line numbering however the line spacing changes between line numbers when using paragraphs. how do if fix this?

I set up a word document for line numbering to start new on each page.  however, when beginning a paragraph by tabbing, the line spacing changes.  The document is set for double spacing but clearly it gets bigger when using paragraphs.  Please help if you can.
doug robbins - mvp office apps & services (word) say:
I set up a word document for line numbering to start new on each page.  however, when beginning a paragraph by tabbing, the line spacing changes.  The document is set for double spacing but clearly it gets bigger when using paragraphs.  Please help if you can.


I am not sure what you mean by starting a paragraph by tabbing.  A new paragraph is started by using the Enter key.

If the paragraphs are formatted as having space before or after them, that space is respected by the line numbering so that the numbering lines up with the actual text in the document.

There are two options to achieve uniform spacing of the line numbering, both of which may be unsatisfactory:

  1. Do not have space between the paragraphs by formatting the paragraphs so that they do not have space before or after them and press the Enter key only once to create a new paragraph
  2. Format the paragraphs as above, but press the Enter key twice to start a paragraph so there will be a space (a one line empty paragraph) between the text in the actual paragraphs.  The problem with this approach is that a line number will be assigned to the empty paragraph.
