Hi,

 

Change the column you want to fill to the correct one

 

Sub FillBlanks()
Dim Area As Range, LastRow As Long
Dim MyCol As String
MyCol = "A"
  On Error Resume Next
  LastRow = Cells.Find(What:="*", SearchOrder:=xlRows, _
            SearchDirection:=xlPrevious, LookIn:=xlFormulas).Row
  For Each Area In Columns(MyCol).Resize(LastRow). _
                   SpecialCells(xlCellTypeBlanks).Areas
    Area.Value = Area(1).Offset(-1).Value
  Next
End Sub

rick rothstein, mvp - excel say:

Give this macro a try..

Sub FillBlanks()
  Dim LastRow As Long
  Const StartRow As Long = 6
  LastRow = Cells.Find(What:="*", SearchOrder:=xlRows, _
            SearchDirection:=xlPrevious, LookIn:=xlValues).Row
  With Cells(StartRow, "A").Resize(LastRow - StartRow + 1)
    .SpecialCells(xlBlanks).FormulaR1C1 = "=R[-1]C"
    .Value = .Value
  End With
End Sub

vijay a. verma say:

1. ALT+F11

2. Locate your workbook name and right click > insert module

3. Copy and paste below code

Function GetHyperlink(Rng As Range) As String

    If Rng(1).Hyperlinks.Count Then
        GetHyperlink = Rng.Hyperlinks(1).Address
        If Len(Rng.Hyperlinks(1).SubAddress) > 0 Then
            GetHyperlink = GetHyperlink & "#" & Rng.Hyperlinks(1).SubAddress
        End If
    Else
        GetHyperlink = ""
    End If
        
End Function

Function GetHlink(Hlink As Range) As String

    If Hlink(1).Hyperlinks.Count Then
        GetHlink = Hlink.Hyperlinks(1).Address
    Else
        GetHlink = ""
    End If

End Function

4. In your cell put following formula for A1 (replace A1 with any other reference) and drag down the formula

=GetHyperlink(A1)

= GetHlink(A1)
