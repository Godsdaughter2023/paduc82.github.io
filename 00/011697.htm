Private Function MonthMatrix(Year, Month, FirstDayOfWeek) As Variant
    Dim MM(1 To 7, 1 To 6)
    Dim FirstDay As Long 'the weekday
    Dim LastDay As Long  'last date
    Dim I As Long, J As Long, K As Long
   
FirstDay = Weekday(DateSerial(Year, Month, 1), FirstDayOfWeek)
LastDay = Day(DateSerial(Year, Month + 1, 0))
   
K = 0
For J = 1 To UBound(MM, 2)
    For I = 1 To UBound(MM, 1)
        If K = 0 Then I = FirstDay
        K = K + 1
        MM(I, J) = K
        If K > LastDay - 1 Then Exit For
    Next I
    If K > LastDay - 1 Then Exit For
Next J

For J = 1 To UBound(MM, 2)
    For I = 1 To UBound(MM, 1)
        If IsEmpty(MM(I, J)) Then MM(I, J) = ""
    Next I
Next J
MonthMatrix = MM
End Function

Private Function MonthStartDay(Year) As Variant
    Dim I As Long, J As Long
    Dim WDStart As Long
    Dim MSD(1 To 12) As Long
    Dim Flag As Boolean
    Dim MSday() As Long
    ReDim MSday(0)

Flag = True
For WDStart = 1 To 7
    For I = 1 To 12
        MSD(I) = Weekday(DateSerial(Year, I, 1), WDStart)
    Next I
   
    For J = 1 To 12
        Select Case J
            Case Is = 1, 2, 4, 5, 7, 8, 10, 11
                If (MSD(J) - 1 + Day(DateSerial(Year, J + 1, 0))) > 35 Then Flag = False
        End Select
    Next J
    If Flag = True Then
        If MSday(0) = 0 Then
            MSday(0) = WDStart
        Else
            ReDim Preserve MSday(0 To UBound(MSday) + 1)
            MSday(UBound(MSday)) = WDStart
        End If
    End If
    Flag = True
Next WDStart
MonthStartDay = MSday
End Function
