rich michaels - mvp (greatcirclelearning.com) say:

This should run a little better for you.

Sub AlignToCentre()
'
' AlignToCentre
    Dim shp As Word.Shape
    If Selection.InlineShapes.Count = 1 Then
        Set shp = Selection.InlineShapes(1).ConvertToShape
        shp.WrapFormat.Type = wdWrapSquare
        shp.Select
        With Selection.ShapeRange
            .RelativeHorizontalPosition = wdRelativeHorizontalPositionPage
            .Left = wdShapeCenter
            .RelativeVerticalPosition = wdRelativeVerticalPositionPage
            .Top = InchesToPoints(1)
        End With
    ElseIf Selection.ShapeRange.Count = 1 Then
        With Selection.ShapeRange
            .RelativeHorizontalPosition = wdRelativeHorizontalPositionPage
            .Left = wdShapeCenter
            .RelativeVerticalPosition = wdRelativeVerticalPositionPage
            .Top = InchesToPoints(1)
        End With
    End If
End Sub

The alignment settings seem only to work with the ShapeRange method of VBA because they need the anchored paragraph included as part of the range.
