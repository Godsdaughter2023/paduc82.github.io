Try this:

 

Dim RngToDelete as Range 
Range("Table1").Select
    ActiveSheet.ListObjects("Table1").Range.AutoFilter Field:=1, Criteria1:="5"

    Set RngToDelete = Selection.SpecialCells(xlCellTypeVisible)

   Selection.AutoFilter
   RngToDelete.EntireRow.Delete
    ActiveSheet.ListObjects("Table1").Range.AutoFilter Field:=1

    Range("Table1[[#Headers],[A]]").Select
