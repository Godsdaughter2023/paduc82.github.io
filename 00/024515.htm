The following macro converts date strings and values in the selected range between mm/dd/yyyy and dd/mm/yyyy. The output is in the form of date values that you can format as dates via Format|Cells|Number|Date.


Sub ConvertDateFormat()
Dim DtRange As Range, oCell As Range, oTxt As String
If Selection.Cells.Count = 1 Then
  Set DtRange = ActiveCell
Else
  Set DtRange = Selection
End If
With Application
  On Error Resume Next ' In case there are no xlConstants or convertible dates in DtRange
  For Each oCell In DtRange.SpecialCells(xlConstants)
    oTxt = oCell.Text
    If UBound(Split(oTxt, "/")) = 2 Then _
      oCell.Value = CDate(Split(oTxt, "/")(1) & "/" & Split(oTxt, "/")(0) & "/" & Split(oTxt, "/")(2))
  Next oCell
End With
End Sub


Note that no conversion is performed if the date is embedded in a formula or the output would be an invalid date.

Private Sub Example_DateString2Date()

Private Function Min(Arg, ParamArray Args())
  Dim i As Long
  On Error Resume Next
  Min = Arg
  For i = LBound(Args) To UBound(Args)
    If Args(i) < Min Then Min = Args(i)
  Next
End Function

Function DateString2Date(ByVal DateTimeString As String, _
    Optional ByVal DateTimeMask As String = "DD.MM.YYYY hh:mm:ss AM/PM", _
    Optional Identifier) As Date
  'Converts a date string into a date
  'Note:
  '  The identifiers in DateTimeMask can be in english/VBA or your local language
  '  DateTimeMask should have at least one separator
  '  Its not necessary that DateTimeMask must have the same format used in _
    DateTimeString, only the order of the identifiers must fit.
  '  Date identifiers must be uppercase, Time identifiers must be lowercase
  'Identifier must be an array with the identifiers in this order:
  '  Year, Month, Day, Hour, Minute, Second
  Dim DoubleDelim As String, Delimiter As String, Temp As String, This As String
  Dim i As Long, j As Long
  Dim DateArr, MaskArr
  Dim Day As Integer, Month As Integer, Year As Integer
  Dim AMPM As Integer, Hour As Integer, Minute As Integer, Second As Integer
  Dim IDs As String

  If Not IsArray(Identifier) Then
    'Load the identifiers in your local language
    ReDim Identifier(0 To 5) As String
    For i = 0 To 5
      'Note: xlYearCode = 19 up to xlSecondCode = 24
      Identifier(i) = Application.International(i + xlYearCode)
      'In some languages the IDs do not match with our conditions
      If i < 3 Then
        'Uppercase date identifiers
        Identifier(i) = UCase(Identifier(i))
      Else
        'Lowercase time identifiers
        Identifier(i) = LCase(Identifier(i))
      End If
    Next
  End If

  If UBound(Identifier) - LBound(Identifier) + 1 <> 6 Then
    Err.Raise 13, "DateString2Date", "Invalid array: Identifier"
  End If

  'Build a string of all identifiers for faster processing
  IDs = "DMYhmns"
  For i = LBound(Identifier) To UBound(Identifier)
    IDs = IDs & Identifier(i)
  Next

  'Did we have a time modifier?
  i = InStrRev(DateTimeMask, "AM/PM", -1, vbTextCompare)
  If i > 0 Then
    'Remove it in both strings
    DateTimeMask = Left$(DateTimeMask, i - 1) & Mid$(DateTimeMask, i + 5)
    i = InStrRev(DateTimeString, "AM", -1, vbTextCompare)
    j = InStrRev(DateTimeString, "PM", -1, vbTextCompare)
    If j > i Then
      AMPM = 12
      i = j
    End If
    If i > 0 Then
      DateTimeString = Left$(DateTimeString, i - 1) & Mid$(DateTimeString, i + 2)
    End If
  End If
  DateTimeString = Trim$(DateTimeString)
  DateTimeMask = Trim$(DateTimeMask)

  'Unify the delimiters
  For i = 1 To Len(DateTimeMask)
    Temp = Mid$(DateTimeMask, i, 1)
    j = InStr(IDs, Temp)
    If j = 0 Then
      If Len(Delimiter) = 0 Then
        Delimiter = Temp
      ElseIf Temp <> Delimiter Then
        DateTimeMask = Replace$(DateTimeMask, Temp, Delimiter)
        DateTimeString = Replace$(DateTimeString, Temp, Delimiter)
      End If
    End If
  Next

  If Len(Delimiter) = 0 Then
    'Add a delimiter between different identifiers
    Delimiter = "-"
    For i = Min(Len(DateTimeString), Len(DateTimeMask)) To 2 Step -1
      Temp = Mid$(DateTimeMask, i - 1, 1)
      This = Mid$(DateTimeMask, i - 0, 1)
      j = InStr(IDs, Temp)
      If j > 0 Then
        If Temp <> This Then
          DateTimeMask = Left$(DateTimeMask, i - 1) & Delimiter & Mid$(DateTimeMask, i)
          DateTimeString = Left$(DateTimeString, i - 1) & Delimiter & Mid$( _
            DateTimeString, i)
        End If
      End If
    Next
  End If

  'Fill all blanks with delimiters
  DateTimeString = Replace$(DateTimeString, " ", Delimiter)
  'Remove double delimiters
  DoubleDelim = Delimiter & Delimiter
  Do While InStr(1, DateTimeMask, DoubleDelim) > 0
    DateTimeMask = Replace$(DateTimeMask, DoubleDelim, Delimiter)
  Loop
  Do While InStr(1, DateTimeString, DoubleDelim) > 0
    DateTimeString = Replace$(DateTimeString, DoubleDelim, Delimiter)
  Loop

  'Split the strings
  DateArr = Split(DateTimeString, Delimiter)
  MaskArr = Split(DateTimeMask, Delimiter)
  'Get the parts
  For i = 0 To Min(UBound(DateArr), UBound(MaskArr))
    If IsNumeric(DateArr(i)) Then
      Select Case Left$(MaskArr(i), 1)
        Case "Y", Identifier(LBound(Identifier)):          Year = DateArr(i)
        Case "M", Identifier(LBound(Identifier) + 1):      Month = DateArr(i)
        Case "D", Identifier(LBound(Identifier) + 2):      Day = DateArr(i)
        Case "h", Identifier(LBound(Identifier) + 3):      Hour = DateArr(i) + AMPM
        Case "m", "n", Identifier(LBound(Identifier) + 4): Minute = DateArr(i)
        Case "s", Identifier(LBound(Identifier) + 5):      Second = DateArr(i)
      End Select
    End If
  Next

  'Calculate the date
  If (Year > 0) Or (Month > 0) Or (Day > 0) Then
    If Year = 0 Then Year = VBA.Year(Date)
    If Month = 0 Then Month = 1
    If Day = 0 Then Day = 1
    DateString2Date = DateSerial(Year, Month, Day) + TimeSerial(Hour, Minute, Second)
  Else
    DateString2Date = TimeSerial(Hour, Minute, Second)
  End If
End Function
