**** Code Start

 

  Dim rsAccess As ADODB.Recordset
  Dim cnnAccess As ADODB.Connection
  Dim strDB As String
  Dim strDbSQL As String
 
On Error Resume Next
    Application.DisplayAlerts = False
    Sheets("Lbr_Dstrb_Data").Delete
    Application.DisplayAlerts = True
On Error GoTo Err_Hndlr
    Sheets.Add 'Before:=Sheets("SQL")
    ActiveSheet.Name = "Lbr_Dstrb_Data"

  Sheets("Lbr_Dstrb_Data").Activate
 
  Range("A5").Select
  ActiveCell.Value = "Emp_ID"
  ActiveCell.Offset(0, 1).Value = "Emp_Name"
  ActiveCell.Offset(0, 2).Value = "Dept_ID"
  ActiveCell.Offset(0, 3).Value = "Proj"
  ActiveCell.Offset(0, 4).Value = "Act_ID"
  ActiveCell.Offset(0, 5).Value = "Total"
  ActiveCell.Offset(0, 6).Value = "Pay_Date_7"
  ActiveCell.Offset(0, 7).Value = "Alloc_Pct"
 
  Set cnnAccess = New ADODB.Connection
  Set rsAccess = New ADODB.Recordset
  
  strDB = Range("LD_Path").Value & Range("LD_File").Value
 
  cnnAccess.ConnectionString = _
     "Provider=Microsoft.ACE.OLEDB.12.0;" & _
     "Data Source=" & strDB & ";"

 

  cnnAccess.Open

 

strDbSQL = _
  "SELECT tbl_TS_FTEs.Emp_ID, tbl_Labor_Dist.Emp_Name, tbl_Labor_Dist.Dept_ID, tbl_Labor_Dist.Proj, tbl_Labor_Dist.Act_ID, Sum(tbl_Labor_Dist.Amount) AS Total_Amt, Left([Pay_Date],7) AS Expr1 " & _
  "FROM tbl_TS_FTEs LEFT JOIN tbl_Labor_Dist ON tbl_TS_FTEs.Emp_ID = tbl_Labor_Dist.Emp_ID " & _
  "GROUP BY tbl_TS_FTEs.Emp_ID, tbl_Labor_Dist.Emp_Name, tbl_Labor_Dist.Dept_ID, tbl_Labor_Dist.Proj, tbl_Labor_Dist.Act_ID, Left([Pay_Date],7) " & _
  "HAVING (((Left([Pay_Date],7))='" & Range("CYYYCM").Value & "'));"


  rsAccess.Open strDbSQL, cnnAccess, adOpenKeyset, adLockOptimistic, adCmdText
  Range("A6").Select
 
  ActiveCell.CopyFromRecordset rsAccess
 
  rsAccess.Close
  Set rsAccess = Nothing
  Set cnnAccess = Nothing

 

  Exit Sub

Err_Hndlr:
  MsgBox "Err #: " & Err.Number & "  Description: " & Err.Description
   Exit Sub
  Resume Next

**** Code End

With VBA, resp. Application.VBE.VBProjects? No.

The only alternate way is to extract the vbaproject.bin from the Excel file, extract that file and read the module files directly.
