You can use GetSaveAsFileName to get VBA to display a Save As dialog and get the file name from that. For example,

 


Dim FName As Variant
Dim Filter As String

Filter = "Excel Files (*.xlsm),*.xlsm"
FName = Application.GetSaveAsFilename(FileFilter:=Filter)
If FName = False Then
    ' user cancelled. get out
    Exit Sub
Else
    Application.DisplayAlerts = False
    ActiveWorkbook.SaveAs FName
    Application.DisplayAlerts = True
End If
