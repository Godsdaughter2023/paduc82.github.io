Function GetVolumeControl(ByVal hmixer As Long, _
                        ByVal componentType As Long, _
                        ByVal ctrlType As Long, _
                        ByRef mxc As MIXERCONTROL) As Boolean

' This function attempts to obtain a mixer control.
' Returns True if successful.
   Dim mxlc As MIXERLINECONTROLS
   Dim mxl As MIXERLINE
   Dim hmem As Long
   Dim rc As Long

   mxl.cbStruct = Len(mxl)
   mxl.dwComponentType = componentType

   ' Obtain a line corresponding to the component type
   rc = mixerGetLineInfo(hmixer, _
                         mxl, _
                         MIXER_GETLINEINFOF_COMPONENTTYPE)

   If (MMSYSERR_NOERROR = rc) Then
       mxlc.cbStruct = Len(mxlc)
       mxlc.dwLineID = mxl.dwLineID
       mxlc.dwControl = ctrlType
       mxlc.cControls = 1
       mxlc.cbmxctrl = Len(mxc)

       ' Allocate a buffer for the control
       hmem = GlobalAlloc(&H40, Len(mxc))
       mxlc.pamxctrl = GlobalLock(hmem)
       mxc.cbStruct = Len(mxc)

       ' Get the control
       rc = mixerGetLineControls(hmixer, _
                                 mxlc, _
                                 MIXER_GETLINECONTROLSF_ONEBYTYPE)

       If (MMSYSERR_NOERROR = rc) Then
           GetVolumeControl = True

           ' Copy the control into the destination structure
           CopyStructFromPtr mxc, mxlc.pamxctrl, Len(mxc)
       Else
           GetVolumeControl = False
       End If
       GlobalFree (hmem)
       Exit Function
   End If

   GetVolumeControl = False
End Function

 

 

Function SetVolumeControl(ByVal hmixer As Long, _
                        mxc As MIXERCONTROL, _
                        ByVal volume As Long) As Boolean
' This function sets the value for a volume control.
' Returns True if successful

   Dim mxcd As MIXERCONTROLDETAILS
   Dim vol As MIXERCONTROLDETAILS_UNSIGNED
   Dim hmem As Long
   Dim rc As Long

   mxcd.item = 0
   mxcd.dwControlID = mxc.dwControlID
   mxcd.cbStruct = Len(mxcd)
   mxcd.cbDetails = Len(vol)

   ' Allocate a buffer for the control value buffer
   hmem = GlobalAlloc(&H40, Len(vol))
   mxcd.paDetails = GlobalLock(hmem)
   mxcd.cChannels = 1
   vol.dwValue = volume

   ' Copy the data into the control value buffer
   CopyPtrFromStruct mxcd.paDetails, vol, Len(vol)

   ' Set the control value
   rc = mixerSetControlDetails(hmixer, _
                              mxcd, _
                              MIXER_SETCONTROLDETAILSF_VALUE)

   GlobalFree (hmem)
   If (MMSYSERR_NOERROR = rc) Then
       SetVolumeControl = True
   Else
       SetVolumeControl = False
   End If
End Function

Example workbook

From the information you provided I have assumed that you have data on more than one worksheet and the overdue tasks from each worksheet are to be on the Overview sheet.

Code is inserted below for information. To install the code in your production workbook.

Sub Macro1()
    Dim ws As Worksheet
    Dim rngDestin As Range
    Dim strDate As String
    Dim rngList As Range
    Dim rngVisible As Range
    Dim lstObj As ListObject
    Dim lngLastRow As Long
    Dim lngLastCol As Long
   
    strDate = "<" & Format(Date, "dd-mmm-yy")
   
    'Clear existing data form the Overview sheet. (Leave column headers in place)
    With Worksheets("Overview")
        On Error Resume Next    'In case no table in worksheet "Overview"
        Set rngList = .ListObjects(1).Range
        .ListObjects(1).Unlist  'Remove table
        On Error GoTo 0
       
        With .Cells     'Clear table formatting
            .Interior.ColorIndex = xlColorIndexNone
            .Font.ColorIndex = xlColorIndexAutomatic
            .Borders.LineStyle = xlLineStyleNone
        End With
       
        'Next line clear all but the column headers
        .Range(.Cells(2, "A"), .Cells.SpecialCells(xlCellTypeLastCell)).Clear
    End With
   
    For Each ws In ThisWorkbook.Worksheets  'Loop through all worksheets
        If ws.Name <> "Overview" Then    'Skip the "Overview" Sheet
            With ws
                If .ListObjects(1).AutoFilter.FilterMode Then
                    .ListObjects(1).AutoFilter.ShowAllData
                    '.ShowAllData    'Remove any existing filters
                End If
               
                .ListObjects(1).Range.AutoFilter Field:=2, Criteria1:=strDate
               
                Set rngVisible = Nothing        'Initialize to nothing
               
                With .ListObjects(1).Range
                'Next line. If > one row in column 1 is visible then more than column headers visible
                    If .Columns(1).SpecialCells(xlCellTypeVisible).Count > 1 Then
                        Set rngVisible = .Offset(1, 0) _
                                            .Resize(.Rows.Count - 1, .Columns.Count) _
                                            .SpecialCells(xlCellTypeVisible)
                    End If
                End With
               
                With Worksheets("Overview")
                    'Set destination to next blank row
                    Set rngDestin = .Cells(.Rows.Count, "A").End(xlUp).Offset(1, 0)
                End With
               
                If Not rngVisible Is Nothing Then
                    'Copy visible data to Overview sheet
                    rngVisible.Copy Destination:=rngDestin
                End If
                If .ListObjects(1).AutoFilter.FilterMode Then
                    .ListObjects(1).AutoFilter.ShowAllData    'Remove filters
                End If
            End With
        End If
    Next ws
   
    With Worksheets("Overview")
        lngLastRow = LastRowOrCol(True, .Cells)
        lngLastCol = LastRowOrCol(False, .Cells)
       
        Set lstObj = .ListObjects.Add(xlSrcRange, .Range(.Cells(1, 1), _
                    .Cells(lngLastRow, lngLastCol)), , xlYes)
                   
        lstObj.Name = "Overview"    'Name the table.
       
        .Columns.AutoFit
       
        'Conditional formatting not required on Overview (all missed)
        .Cells.FormatConditions.Delete
       
    End With
   
    Application.Goto Worksheets("Overview").Range("A1")
   
End Sub


Function LastRowOrCol(bolRowOrCol As Boolean, Optional rng As Range) As Long
    'If optional Rng must include the worksheet reference if not the active sheet
    'Finds the last used row or column in a worksheet
    'First parameter is True for Last Row or False for last Column
    'Third parameter is optional
        'Must be specified if not ActiveSheet
   
    Dim lngRowCol As Long
    Dim rngToFind As Range
   
    If rng Is Nothing Then
        Set rng = ActiveSheet.Cells
    End If
   
    If bolRowOrCol Then
        lngRowCol = xlByRows
    Else
        lngRowCol = xlByColumns
    End If
   
    With rng
        Set rngToFind = rng.Find(What:="*", _
                LookIn:=xlFormulas, _
                LookAt:=xlPart, _
                SearchOrder:=lngRowCol, _
                SearchDirection:=xlPrevious, _
                MatchCase:=False)
    End With
   
    If Not rngToFind Is Nothing Then
        If bolRowOrCol Then
            LastRowOrCol = rngToFind.Row
        Else
            LastRowOrCol = rngToFind.Column
        End If
    End If
   
End Function

Journey of upgrading Windows 8.1 Pro to Windows 10 TP - Microsoft Community

tarund. say:

Journey of upgrading Windows 8.1 Pro to Windows 10 TP

Hello everyone,

I never do in-place upgrades of any Windows version. I always do a fresh install. However, I felt fairly confident about upgrading my perfectly working setup of Windows 8.1 on my Lenovo Yoga 2 Pro. I've the shared experience below and will answer any questions/concerns you may have.

Installation

1) Downloaded ISO from Insider website

2) Created USB drive using Rufus (http://rufus.akeo.ie/)

3) Started setup from USB drive

4) "Ready to install" screen showed to confirm

'Windows Technical Preview'

'Keep Windows settings, personal files and apps'

5) I clicked the 'Install' button and started praying around 1:27 PM EST

6) At 1:54 PM EST, it rebooted twice (very quickly) and the "Setting up 10%" screen (this is the initial Windows setup screen we’re all familiar with) showed up.

7) At 1:58 PM EST, it completed setup and saw the ‘Evaluation copy. Build 9841’ in the bottom right of the screen.

8) At 1:59 PM EST, ‘Getting devices ready **%" message

9) At 2:00 PM EST, ‘Getting ready’ message

10) At 2:01 PM EST, ‘Applying PC settings 1%" message

11) At 2:09 PM EST, reboot and ‘Updating system 100%’ message and another reboot

12) ‘Setting up a few more things 1%’ message, finished at 2:11 PM EST with ‘Getting ready’ message

13) ‘Get online’ screen popped up

14) Set up my settings rather than using Express settings

15) Signed into my Microsoft account

16) Approved the sign-in request on my Android phone using Microsoft account app

17) At 2:14 PM EST, saw messages:

‘Hi’

‘We’re setting up things for you’

‘You can get new apps from the Store’ (below, ‘Installing your apps’)

18) At 2:15 PM EST, I was on the Desktop with the words ‘Windows Technical Preview’ and ‘Evaluation copy. Build 9841’

19) At 2:18 PM EST, Kaspersky prompts to reinstall itself due to a ‘Windows 8.1’ update (clearly, Kaspersky isn’t updated to handle Windows 10) yet it crashed the system and rebooted. Not a big deal and was expecting this to happen.

Applications

I was able to launch all other applications without hiccups. Launching Word 2013 did launch the configuration setup but that only took 20 secs to finish. Also, I had to reinstall the Synaptics driver for the touchpad – not a big deal since it’s available on Lenovo’s website (latest version from Synaptics website doesn’t install because of signature verification failure).

Graphics

Graphics performance is slightly sluggish but that is most likely due to the Intel Graphics drivers not being fully optimized for this version. I may try a beta version of the Intel drivers if available. Didn’t have to install any other drivers.

IE

IE 11.0.8 is installed. A bit odd considering I’m running 11.0.12 on my Windows 7 machine. Maybe this was done for stability and compatibility reasons?

Windows Explorer

Windows Explorer ‘Navigation pane’ icons have been updated. This is clearly a work in progress as these get updated across the OS. Can’t wait to see how they approach this but definitely excited to see an icon refresh.

Continuum

Continuum seems promising but definitely needs polish. A user should be able to drag applications from Desktop to Desktop rather clicking on the app, drop-down menu, and chose 'Move to Desktop *'.

WiFi

WiFi does't seem to connect until I login. Not sure if that's a new security feature or a bug. Will follow up with Microsoft about it.

Windows Updates

There were couple of updates to install:

Windows Defender update

KB3001512 to address 3 issues (http://support.microsoft.com/kb/3001512)

Issuer Distribution Point (IDP) sets are removed from Windows Tech Preview or Windows Server Tech Preview. This change prevents Microsoft Surface devices and Nokia tablets from obtaining firmware or driver updates.

Adobe Flash update does not contain premium video playback.

Some problems in the Compatibility View list for Internet Explorer 11.

Also, under Update and Recovery - there is a 'Preview builds' update check. Users of technical preview should be able to upgrade within the OS, very cool. I hope that works well and doesn't brick devices in the future.

My impression

Upgrading from Windows 8.1 to Windows 10 in 50 minutes is pretty solid. This includes keeping and being able to launch almost all applications, keeping all settings and data (of course, this all depends on how much data you may have). From my perspective, for a technical preview, this is a fairly stable build for as long as I’ve used it. I’ll put it through the paces tonight at home and see how it holds up for the next few days. If it’s solid enough, I may end up upgrading my desktop at home as well – but that’s too scary of a thought for now. I’ll come back to that later.

I will try to add to this as much as possible, including providing feedback to Microsoft through the built-in feedback tool.
