This should do it as expected.

Function TextFunction(InputFormula As Variant)
   Dim strFormula       As String
   Dim r                As Range
   Dim Result           As String
   Dim intPos           As Integer
   Dim strAddress       As String
   Dim x                As Range
   Dim n                As Integer
   Dim var              As Variant
  
   If TypeName(Application.Caller) = "Range" Then
    
      Set r = Range(Application.Caller.Address)
    
      strFormula = r.Formula
      strFormula = Mid(strFormula, InStr(1, strFormula, "(") + 1)
      strFormula = Left(strFormula, Len(strFormula) - 1)
      Set r = Range(strFormula)
      strFormula = r.Formula & "|"
     
      intPos = 1
    
      Do While intPos <= Len(strFormula)
         Select Case Mid(strFormula, intPos, 1)
            Case "+", "-", "*", "/", "^", "(", ")", ",", "|"
               If strAddress = "" Then
                  Result = Result & Mid(strFormula, intPos, 1)
               ElseIf IsNumeric(strAddress) Then
                  Result = Result & strAddress & Mid(strFormula, intPos, 1)
                  strAddress = ""
               Else
                  On Error GoTo Err_Handler
                  Result = Result & Range(strAddress).Text & Mid(strFormula, intPos, 1)
                  strAddress = ""
               End If
            Case "="
            Case Else
               strAddress = strAddress & Mid(strFormula, intPos, 1)
         End Select
NextChar:
         intPos = intPos + 1
      Loop
   End If
 
  
   Result = Replace(Result, "ROUND(", "") & ","
   var = Split(Result, ",")
  
   TextFunction = Replace(var(0), "|", "") & "="
  
   Exit Function
Err_Handler:
 
   Result = Mid(Result, 2) & strAddress & Mid(strFormula, intPos, 1)
   strAddress = ""
   GoTo NextChar
End Function
