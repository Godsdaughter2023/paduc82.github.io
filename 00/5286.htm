Public Sub MacroName()
 
    Dim ws1 As Worksheet
    Dim pt1 As PivotTable
 
    For Each ws1 In Workbooks("Source Workbook.xlsm").Worksheets
 
        'I assume that all worksheets with a pivot table have blank C1, because if it were a data sheet, C1 should be a column name
 
        If ws1.Range("C1").Value = "" Then
 
            'Copy the Pivot worksheet as a new worksheet
             ws1.Copy ThisWorkbook.Sheets(Sheets.Count)
            
             'Get the pivot table name in the worksheet
             For Each pt1 In ActiveSheet.PivotTables
                ptName = pt1.Name
             Next pt1
            
 
            'Remove the Department column from pivot table
             ActiveSheet.PivotTables(ptName).PivotFields("Department").Orientation = xlHidden

            'Paste the new worksheet as value
             Cells.Select
             Selection.Copy
             Selection.PasteSpecial Paste:=xlPasteValues, Operation:=xlNone, SkipBlanks:=False, Transpose:=False

            file_name = ActiveSheet.Range("B1").Value

            With ActiveSheet
             .Name = file_name
             .Columns("C:C").Delete
             .Rows("1:2").Delete
             .Move
             End With

            'Save the new workbook, define your path
             ActiveWorkbook.SaveAs Filename:="C:\Users\wymanwmwong\Desktop\" & file_name & ".xlsx" _
                     , FileFormat:=xlOpenXMLWorkbook, CreateBackup:=False

            ActiveWorkbook.Close False
 
        End If
 
    Next ws1
 
End Sub

rick rothstein, mvp - excel say:

Here is a modification to your modification of Bob's original code that should stop that problem...

Sub Setup_Plan_Records()
    On Error Resume Next
    Range("AD3:AM300").SpecialCells(xlCellTypeBlanks).Delete xlToLeft
    On Error GoTo 0
    N = 1
    For I = 3 To Range("A10000").End(xlUp).Row
        For J = 2 To Cells(I, 256).End(xlToLeft).Column
            Sheets("Import Plans").Cells(N, 1).Value = Cells(I, 1).Value
            Sheets("Import Plans").Cells(N, 2).Value = Cells(I, J).Value
            N = N + 1
        Next
    Next
End Sub
