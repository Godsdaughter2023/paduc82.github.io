If you set the Styles Pane Options to show "All styles" in alphabetical order, you should be able to find List Number, List Number 2, List Number 3, etc. There are five levels built in, but of course you could add others by defining new styles as needed.

andreas killer say:

Sub Main()
  Const Nth = 11
  Dim Keep As Range, This As Range, Del As Range

  'Find the last cell in row 1
  Set Keep = Range("A1").Offset(, Columns.Count - 1).End(xlToLeft)
  'We want to keep the nth column from the right
  If Keep.Column - Nth < 1 Then
    MsgBox "Not possible"
    Exit Sub
  End If
  'Refer to the column to keep
  Set Keep = Keep.Offset(, -Nth)
  'Clear all numbers in this column if any
  Set This = SpecialCells(Keep, xlCellTypeConstants, xlNumbers)
  If Not This Is Nothing Then
    This.ClearContents
  End If
  'Refer to the right cells
  Set Del = Keep.Offset(, 1).Resize(, Nth)
  'Are there cells on the left?
  If Keep.Column > 1 Then
    'Add the left cells
    Set Del = Union(Range("A1", Keep.Offset(, -1)), Del)
  End If
  'Delete the entire columns
  Del.EntireColumn.Delete
End Sub

Private Function SpecialCells(ByVal R As Range, ByVal Typ As XlCellType, _
    Optional ByVal Value As XlSpecialCellsValue = &H17) As Range
  'Avoid the SpecialCells-BUG to return all cells from the current region
  On Error Resume Next
  Select Case Typ
    Case xlCellTypeConstants, xlCellTypeFormulas
      Set SpecialCells = Intersect(R, R.SpecialCells(Typ, Value))
    Case xlCellTypeConstants Or xlCellTypeFormulas
      'Special feature: Return all used cells
      Set SpecialCells = Intersect(R, R.SpecialCells(xlCellTypeConstants, Value))
      If SpecialCells Is Nothing Then
        Set SpecialCells = Intersect(R, R.SpecialCells(xlCellTypeFormulas, Value))
      Else
        Set SpecialCells = Union(SpecialCells, Intersect(R, R.SpecialCells( _
          xlCellTypeFormulas, Value)))
      End If
    Case Else
      Set SpecialCells = Intersect(R, R.SpecialCells(Typ))
  End Select
End Function
