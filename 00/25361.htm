Sub AutoF_MultipleCriteria_Array()
'Aug 31, 2016
Dim ws As Worksheet
Set ws = ActiveSheet '<< data in active sht
Dim v As Variant
Dim t As Long, r As Long, x As Long
Dim rng As Range
t = ws.Cells(Rows.Count, "A").End(xlUp).Row ' << number of criteria in col A
