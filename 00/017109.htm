graham mayor say:
The code to open a word document based on a template from another application is fairly straightforward e.g.

Sub OpenWord()
Dim wdApp As Object
Dim wdDoc As Object
Dim strMyTemplate As String
    strMyTemplate = "D:\Path\Letter1.dotm" 'The template in question

    On Error Resume Next
    Set wdApp = GetObject(, "Word.Application")
    If Err Then
        Set wdApp = CreateObject("Word.Application")
    End If
    Set wdDoc = wdApp.Documents.Add(strMyTemplate)

    wdApp.Visible = True
    wdApp.Activate
    Set wdDoc = Nothing
    Set wdApp = Nothing
End Sub
