hansv mvp say:

You can't do this with a single rule, you have to add a rule for each value.

 

Sub conditionalFormatingColours()
    Dim arrColours
    Dim i As Long
    arrColours = Array(192, 255, 49407, 65535, 5296274, 5287936, 15773696, 12611584, 6299648, 10498160)
    For i = 0 To 9
        With Sheet5.Range("AB2:AP198").FormatConditions.Add _
                (Type:=xlCellValue, Operator:=xlEqual, Formula1:="=" & (i + 1))
            .SetFirstPriority
            With .Interior
                .PatternColorIndex = xlAutomatic
                .Color = arrColours(i)
               .TintAndShade = 0
            End With
            .StopIfTrue = False
        End With
    Next i
End Sub

Sub From_XML_To_XL()
On Error GoTo errh
Dim myWB As Workbook, WB As Workbook
Set myWB = ThisWorkbook
Dim myPath
myPath = "c:\folder1\folder2\xml folder\" '<<< change path

Dim myFile
myFile = Dir(myPath & "*.xml")

Dim t As Long
t = 1

Application.ScreenUpdating = False
Do While myFile <> ""
Set WB = Workbooks.OpenXML(Filename:=myPath & myFile)
WB.Sheets(1).UsedRange.Copy myWB.Sheets(1).Cells(t, "A")
WB.Close False
t = myWB.Sheets(1).UsedRange.Rows.Count + 2

myFile = Dir()
Loop
Application.ScreenUpdating = True

myWB.Save
Exit Sub

End Sub

Sub From_XML_To_XL_02()
On Error GoTo errh
Dim myWB As Workbook, WB As Workbook
Set myWB = ThisWorkbook
Dim myPath
myPath = "c:\folder1\folder2\xml folder\" '<<< change path
Dim myFile

L = myWB.Sheets(1).Cells(Rows.Count, 1).End(xlUp).Row
ReDim v(t To L)

Next
myWB.Sheets(1).Cells(t, 1) = Join(v, ",")
t = t + 1
myWB.Sheets(1).Cells(t, 1).Resize(L).Clear
myFile = Dir()
Loop

Application.ScreenUpdating = True
myWB.Save
Exit Sub
