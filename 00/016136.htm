Product code validation

sandro peruz say:

Ciao Scott,

I would be inclined to proceed in this way.

I would keep the producCode field as text.

A field in your customer table which shows the length of the product code for that specific customer, productLength is its name.

in the table, recordsource of the form used to populate customer and product code, an id_customer  field like field name of the previous table as combo box, for which its sql extracts id_customer, customerName and productLength.

Then in your form, some vba code to check what user enters in the productCode control :

 

Get a look at this little demo : http://1drv.ms/1CPLJfNvalidation.accdb

 

Option Compare Database
Option Explicit
Dim bool As Boolean

Private Sub product_BeforeUpdate(Cancel As Integer)

    Cancel = True
    Me.product.Undo
    MsgBox "product code length must be: " & Me.id_customer.Column(2) & " Digits.", vbCritical
End If
 
End Sub

    Me.product.Undo
    bool = Not bool
End If

Private Sub product_KeyDown(KeyCode As Integer, Shift As Integer)

    MsgBox "Just digits are acceptable", vbInformation
    bool = Not bool
End If  
   
End Sub

Private Function chkDigit(keyPressedCode As Integer) As Boolean
  
   Select Case keyPressedCode
            Case 48 To 57, 96 To 105
                 chkDigit = Not chkDigit
            Case vbKeyDelete, vbKeyReturn, vbKeyBack, vbKeyNumlock
                 chkDigit = Not chkDigit
   End Select

I am afraid you are missing the point. You cannot use mail merge in the way you are attempting. Word mail merge is a simple one to one merge. ie. one record (worksheet row) per 'document'.

You cannot nest Next or NextIF fields so you cannot make the table 'grow' in the way you imagine by conditionally adding next record fields..

Here is a complete step by step guide to getting Windows 10 Technical Preview to work on your Mac via Bootcamp. (As a dual boot)

  • A USB drive, 1GB or more

Let’s do it!

  1. Boot your mac into OSX
  2. Download the Windows Technical Preview ISO (64-bit ONLY)
  3. Open disk utility
    1. Drag the ISO file that you downloaded into the sidebar
    2. Insert a CD (this can also be done with a USB, though it’s much harder)
    3. Select the ISO file in the sidebar and click ‘Burn’ at the top
      • Mount the disc when done
      • Verify the disc
    4. Wait for it to finish (should take somewhere around 5-30 minutes)

      • Choose both options (download support software, and install windows 7 or later version)

      • Click ‘Install Now’

      • Select the volume (on disk 0) called BOOTCAMP. Click format, and press ok.
      • Press ‘Next’ at the bottom right
      • Wait around 30 minutes for everything to install properly.

Sub Sort_Copy()
    Dim fltRng As Range
    Dim CriRng As Range
        
    Set fltRng = Range("A1:I13")
    Set CriRng = Range("K1:S2")
    
    Range("K4").CurrentRegion.Clear
    
    With fltRng
        .AdvancedFilter xlFilterCopy, CriRng, Range("K4"), False
    End With
    Mail_Selection_Range_Outlook_Body
End Sub

Sub Mail_Selection_Range_Outlook_Body()
'For Tips see: http://www.rondebruin.nl/win/winmail/Outlook/tips.htm
'Don't forget to copy the function RangetoHTML in the module.
'Working in Excel 2000-2013
    Dim rng As Range
    Dim OutApp As Object
    Dim OutMail As Object

    Set rng = Nothing
    On Error Resume Next
    'Only the visible cells in the selection
    Set rng = Range("K4").CurrentRegion.SpecialCells(xlCellTypeVisible)
    'You can also use a fixed range if you want
    'Set rng = Sheets("YourSheet").Range("D4:D12").SpecialCells(xlCellTypeVisible)
    On Error GoTo 0

    If rng Is Nothing Then
        MsgBox "The selection is not a range or the sheet is protected" & _
               vbNewLine & "please correct and try again.", vbOKOnly
        Exit Sub
    End If

    With Application
        .EnableEvents = False
        .ScreenUpdating = False
    End With

    Set OutApp = CreateObject("Outlook.Application")
    Set OutMail = OutApp.CreateItem(0)

    On Error Resume Next
    With OutMail
        .To = "" 'please enter your required mail id
        .CC = ""
        .BCC = ""
        .Subject = "This is the Subject line"
        .HTMLBody = RangetoHTML(rng)
        .Display   'or use .Display
    End With
    On Error GoTo 0

    With Application
        .EnableEvents = True
        .ScreenUpdating = True
    End With

    Set OutMail = Nothing
    Set OutApp = Nothing
End Sub


Function RangetoHTML(rng As Range)
' Changed by Ron de Bruin 28-Oct-2006
' Working in Office 2000-2013
    Dim fso As Object
    Dim ts As Object
    Dim TempFile As String
    Dim TempWB As Workbook

    TempFile = Environ$("temp") & "\" & Format(Now, "dd-mm-yy h-mm-ss") & ".htm"

    'Copy the range and create a new workbook to past the data in
    rng.Copy
    Set TempWB = Workbooks.Add(1)
    With TempWB.Sheets(1)
        .Cells(1).PasteSpecial Paste:=8
        .Cells(1).PasteSpecial xlPasteValues, , False, False
        .Cells(1).PasteSpecial xlPasteFormats, , False, False
        .Cells(1).Select
        Application.CutCopyMode = False
        On Error Resume Next
        .DrawingObjects.Visible = True
        .DrawingObjects.Delete
        On Error GoTo 0
    End With

    'Publish the sheet to a htm file
    With TempWB.PublishObjects.Add( _
         SourceType:=xlSourceRange, _
         Filename:=TempFile, _
         Sheet:=TempWB.Sheets(1).Name, _
         Source:=TempWB.Sheets(1).UsedRange.Address, _
         HtmlType:=xlHtmlStatic)
        .Publish (True)
    End With

    'Read all data from the htm file into RangetoHTML
    Set fso = CreateObject("Scripting.FileSystemObject")
    Set ts = fso.GetFile(TempFile).OpenAsTextStream(1, -2)
    RangetoHTML = ts.readall
    ts.Close
    RangetoHTML = Replace(RangetoHTML, "align=center x:publishsource=", _
                          "align=left x:publishsource=")

    'Close TempWB
    TempWB.Close savechanges:=False

    'Delete the htm file we used in this function
    Kill TempFile

    Set ts = Nothing
    Set fso = Nothing
    Set TempWB = Nothing
End Function
