chemcal say:

Hi Rahul,

In my long years’ experience, the best way is to adjust the sizes of all the objects/charts as gathered in one Excel worksheet in advance, and then transfer them into a Word document one by one in the proper positions. The size adjustments can be realized by means of the macro:

 

Sub ChartSizeEn()

'Sets the size of the selected object/objects in mm.

'The objects can be either of Chart or Picture type only.

'Selected can be either the object whose sizes should be

'- first copied or

'- changed.

'Sequential inputs:

'- height:

'   - number -> new height size,

'   - "m" -> saving of the selected object height for changing in the next round,

'   - no input -> the selected object height should be maintained.

'- width:

'   - number -> new width size,

'   - "s" -> setting the width to form a square,

'   - "a" -> setting the width to keep the aspect ratio,

'   - no input -> the selected object width should be maintained,

'For size change, a whole series of charts can be also selected

 

  Dim ErrMsg As String, MB As Long, ChH As Double, ChW As Double, NewSizes As Boolean, _

    S As Variant, Sp As Variant, NewChartHeight As String, NewChartWidth As String, _

    ObjDim As Variant, I As Long

  Static ChartHeight As Double, ChartWidth As Double, NChart As Long

  Const PointSize As Single = 2.831 

'coefficient can be adjusted by measuring the Word actual size (i.e. in the print)

  Const Title As String = "Object size"

NewMeasures:

  On Error Resume Next

  NChart = 1

  NChart = Selection.Count

  ErrMsg = "no object"

  On Error GoTo ErrExit

  Set S = Selection

  If NChart > 1 Then

    Set Sp = S(1)

  Else

    Set Sp = S

  End If

  If Sp.Parent.Type = -4167 Then Set ObjDim = Sp Else Set ObjDim = Sp.Parent.Parent

  Select Case Sp.Parent.Type

  Case -4169, -4167, -4111, 4, 3

  Case Else

    ErrMsg = "chart/picture": GoTo ErrExit

  End Select

ActualSize:

  If ChartHeight = 0 Or NewSizes Then ChartHeight = ObjDim.Height / PointSize

  If ChartWidth = 0 Or NewSizes Then ChartWidth = ObjDim.Width / PointSize

  If NewSizes Then Exit Sub

  NewChartHeight = InputBox("H e i g h t /mm" & vbCrLf & _

    "Cancel:  keep actual height" & vbCrLf & _

    "'m':     save actual sizes", Title, Format(ChartHeight, "0"))

  If LCase(NewChartHeight) = "m" Then

    MB = MsgBox("H e i g h t : " & Format(ObjDim.Height / PointSize, "0") & vbCrLf & _

      "W i d t h    : " & Format(ObjDim.Width / PointSize, "0"), vbOKOnly, Title)

    NewSizes = True: GoTo NewMeasures

  End If

  If NewChartHeight <> "" Then ChartHeight = CDbl(NewChartHeight)

  ErrMsg = "enough large size"

  If ChartHeight <= 10 Then GoTo ErrExit

  NewChartWidth = InputBox("W  I d t h  /mm" & vbCrLf & _

    "Cancel:  keep actual width" & vbCrLf & _

    "'a':     keep actual aspect" & vbCrLf & _

    "'s':     square object pattern", Title, Format(ChartWidth, "0"))

  Select Case LCase(NewChartWidth)

  Case "": Exit Sub

  Case "s": ChartWidth = ChartHeight

  Case "a": ChartWidth = ChartHeight * (Sp.Width / Sp.Height)

  Case Else: ChartWidth = CDbl(NewChartWidth)

  End Select

  If ChartWidth <= 10 Then GoTo ErrExit

  ErrMsg = "a proper object"

  ChH = ChartHeight * PointSize

  ChW = ChartWidth * PointSize

  On Error GoTo ErrExit

  Application.ScreenUpdating = False

  If NChart = 1 Then

    ObjDim.Height = ChH

    ObjDim.Width = ChW

    Sp.Select

  Else

    For I = 1 To NChart

      S(I).Height = ChH

      S(I).Width = ChW

    Next I

    S(NChart).Select

  End If

  Application.ScreenUpdating = True

  On Error GoTo 0

  Exit Sub

ErrExit:

  MB = MsgBox("There was not selected " & ErrMsg, vbCritical, Title)

  On Error GoTo 0

End Sub

Any amendments improving this operation are welcome.

mccake say:

Resolved, by chance really I guess, but I can only explain it the following way.

 

  The error message seemed to be intermittent, in that deleting a few rows did 'initially' clear the error for the user that had the pivot table refresh problem.
