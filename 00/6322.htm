Illustrates various means of restring results by entering/selecting values in a dialogue form, including two methods involving the use of a multi-select list box.  If the method which filters a report by means of the WhereCondition argument were to be extended to use two list boxes, adding one for cities in addition to that for employees in the demo, the code would be as follows;

    Dim varItem As Variant
    Dim strEmployeeIDList As String
    Dim strCityList As String
    Dim strCriteria As String
    Dim ctrl As Control
    
    Set ctrl = Me.lstEmployees
    
    If ctrl.ItemsSelected.Count > 0 Then
        For Each varItem In ctrl.ItemsSelected
            strEmployeeIDList = strEmployeeIDList & "," & ctrl.ItemData(varItem)
        Next varItem
        
        ' remove leading comma
        strEmployeeIDList = Mid(strEmployeeIDList, 2)
        
        strCriteria = "EmployeeID In(" & strEmployeeIDList & ")"
        
    End If

    Set ctrl = Me.lstCities
    
    If ctrl.ItemsSelected.Count > 0 Then
        If Len(strCriteria) > 0 Then
              strCriteria = strCriteria & " And "
        End if

        For Each varItem In ctrl.ItemsSelected
            strCityList = strCityList & ",""" & ctrl.ItemData(varItem) & """"
        Next varItem
        
        ' remove leading comma
        strCityList = Mid(strCityList, 2)
        
        strCriteria = strCriteria & "City In(" & strCityList & ")"
        
    End If

    If Len(strCriteria) > 0 Then
        DoCmd.OpenReport "rptEmployees", _
            view:=acViewPreview, _
            WhereCondition:=strCriteria
    Else
        MsgBox "No criteria selected", vbInformation, "Warning"
    End if
