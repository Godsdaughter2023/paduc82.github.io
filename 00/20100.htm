I think Chip's code will work with just a minor modification - but I haven't tested it:

 

Sub AAA()
    Dim R As Range
    Dim DataRange As Range
    Dim Total As Double
    Dim TheYear As String
    Dim TheMonth As String
    Dim TheFacility As String
   
    ' get your values from the linked cells.
    TheYear = "2011"
    TheMonth = "may"
    TheFacility = "123"
   
    Set DataRange = Worksheets("Sheet2").Range("A1:D10") '<< CHANGE
    For Each R In DataRange.Columns(1).Cells
        If R(1, 1).Value = TheYear Or LCase(Trim(TheYear)) = "all" Then
            If R(1, 2).Value = TheMonth Or LCase(Trim(TheMonth)) = "all" Then
                If R(1, 3).Value = TheFacility Or LCase(Trim(TheFacility)) = "all" Then
                    Total = Total + R(1, 4).Value
                End If
            End If
        End If
    Next R
    MsgBox Total
End Sub
