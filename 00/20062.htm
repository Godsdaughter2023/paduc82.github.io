sloophijinx say:

Regretably both these suggestions have been tried and they return your own username or login.

The following code in a class module always returns the current user's login name and is better than the above applicaion.username or Environ("username").  What I want is to find the same thing for another workbook already in use (preferably withut first having to open it).

 

'Written by Bruce Copland
'Class module to access system components named cSysyInfo
'This version is usable on W7 or XP both 32 or 64 bit systems

Option Explicit

'Declaration required for User Network Login Name
#If VBA7 Then
    Private Declare PtrSafe Function GetUserName _
        Lib "advapi32.dll" _
        Alias "GetUserNameA" _
        (ByVal lpbuffer As String, _
        ByRef nsize As LongPtr) As LongPtr
#Else
    Private Declare Function GetUserName _
        Lib "advapi32.dll" _
        Alias "GetUserNameA" _
        (ByVal lpbuffer As String, _
        ByRef nsize As Long) As Long
#End If

Public Property Get stUserLoginName() As String
'Returns users network login name
#If VBA7 Then
    Dim lAPIResult As LongPtr
    Dim lBufflen As LongPtr
#Else
    Dim lAPIResult As Long
    Dim lBufflen As Long
#End If
    Dim stBuff As String * 255
   
    lBufflen = 255
    lAPIResult = GetUserName(stBuff, lBufflen)
    stUserLoginName = Left(stBuff, InStr(stBuff, vbNullChar) - 1)
End Property

In standard module:

 

Function FunReadUserLogin(Optional bInterActive As Boolean = False) As String
    'Requires Class module cSysInfo
    On Error GoTo ERRORHANDLER
    Dim stProcedureName As String
    stProcedureName = "FunReadUserLogin"

    '-----------------------------------------------------------------------------------
    Dim oSysInfo As New cSysInfo
    FunReadUserLogin = oSysInfo.stUserLoginName
    If bInterActive Then
        MsgBox Prompt:= _
               "User's Login Name = " & oSysInfo.stUserLoginName, _
               Buttons:=vbInformation, _
               Title:="System Information"
    End If
    Set oSysInfo = Nothing
    '================================================

    Exit Function
ERRORHANDLER:
    ErrorMessage stProcedureName
End Function
