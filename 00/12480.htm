bernie deitrick say: Do you want the macro to write values to a cell? or a function that returns a value? or do you just want to write a formula in a cell using VBA?  Lots of options...for example, to put the values into column C based on columns A and B, for each row:

Sub TestMacro()
    Dim rngA As Range
    Dim rngB As Range
    Dim rngC As Range

    For Each rngA In Range("A2", Cells(Rows.Count, 1).End(xlUp))
        Set rngB = rngA.Offset(0, 1)
        Set rngC = rngA.Offset(0, 2)
        If rngA.Value <> "" And rngB.Value <> "" Then
            If rngB.Value <= rngA.Value + 14 Then
                rngC.Value = rngA.Value + 14
            Else
                rngC.Value = "LATE"
            End If
        End If
    Next rngA
End Sub
