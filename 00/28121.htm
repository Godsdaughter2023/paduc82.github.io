    In my spreadsheet there are two tabs - "ORDER" is the order form where customer selects specification they want to buy, and "data sheet" - from where drop-down options are taken (Validation list - 494 options, A2:A495 on the data sheet).

     

    I use Excel 2010. I inserted ComboBox on my "ORDERS" spreadsheet. In Developer I clicked Design Mode. In properties of the ComboBox I entered name TempCombo, set font to Arial 14, ListRows - 15, MatchEntry - 1-fmMatchEntryComplete.

    LinkedCell and ListFillRange are blank.

     

    Then I closed Design Mode.

    Right clicked on "ORDERS" tab, clicked on View Code and inserted this code (got this from another forum):

     

    Option Explicit

    ' Developed by Contextures Inc.

    ' www.contextures.com

    Private Sub TempCombo_KeyDown(ByVal _

            KeyCode As MSForms.ReturnInteger, _

            ByVal Shift As Integer)

        'Hide combo box and move to next cell on Enter and Tab

        Select Case KeyCode

            Case 9

                ActiveCell.Offset(0, 1).Activate

            Case 13

                ActiveCell.Offset(1, 0).Activate

            Case Else

                'do nothing

        End Select

     

    End Sub

     

    Private Sub Worksheet_BeforeDoubleClick(ByVal Target As Range, Cancel As Boolean)

    Dim str As String

    Dim cboTemp As OLEObject

    Dim ws As Worksheet

    Set ws = ActiveSheet

    On Error GoTo errHandler

     

    If Target.Validation.Type = 3 Then

    Cancel = True

    End If

     

    Set cboTemp = ws.OLEObjects("TempCombo")

      On Error Resume Next

      With cboTemp

        .ListFillRange = ""

        .LinkedCell = ""

        .Visible = False

      End With

    On Error GoTo errHandler

      If Target.Validation.Type = 3 Then

        Application.EnableEvents = False

        str = Target.Validation.Formula1

        str = Right(str, Len(str) - 1)

        With cboTemp

          .Visible = True

          .Left = Target.Left

          .Top = Target.Top

          .Width = Target.Width + 15

          .Height = Target.Height + 5

          .ListFillRange = ws.Range(str).Address

          .LinkedCell = Target.Address

        End With

        cboTemp.Activate

      End If

     

    errHandler:

      Application.EnableEvents = True

      Exit Sub

     

    End Sub

     

    Private Sub Worksheet_SelectionChange(ByVal Target As Range)

    There were 2 problems:
    - the list feeding the combobox is not on the same sheet as the combobox, and the way ListFillRange was set didn't work in that case
    - my additional code omitted .Object which gave errors


    These were compounded by the fact that if an error occurred it would not be reported.

    So, the revised code for the DoubleClick procedure is


    Private Sub Worksheet_BeforeDoubleClick(ByVal Target As Range, Cancel As Boolean)
    Dim str As String
    Dim cboTemp As OLEObject
    Dim ws As Worksheet
    Set ws = ActiveSheet
    On Error GoTo errHandler

    If Target.Validation.Type = 3 Then
    Cancel = True
    End If

    Set cboTemp = ws.OLEObjects("TempCombo")
      On Error Resume Next
      With cboTemp
        .ListFillRange = ""
        .LinkedCell = ""
        .Visible = False
      End With
    On Error GoTo errHandler
      If Target.Validation.Type = 3 Then
        Application.EnableEvents = False
        str = Target.Validation.Formula1
        str = Right(str, Len(str) - 1)
        With cboTemp
          .Visible = True
          .Left = Target.Left
          .Top = Target.Top
          .Width = Target.Width + 15
          .Height = Target.Height + 5
          .ListFillRange = str
          .LinkedCell = Target.Address
          .Object.Font.Name = "Arial"
          .Object.Font.Size = 12
        End With
        cboTemp.Activate
      End If
     
    TidyUp:
      Application.EnableEvents = True
      Exit Sub
    errHandler:
      MsgBox "Error " & Err & ": " & Error & vbNewLine & "in Worksheet_BeforeDoubleClick"
      Resume TidyUp

    End Sub
