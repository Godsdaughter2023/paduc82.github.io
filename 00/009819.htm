vijay a. verma say:
You can not control the name through Powerpoint. You will have 2 choices -

1. After export, rename in the folder either manually or through a bulk file rename utility. Jason has given a link.

2. Or use a macro to export. Below is a macro which I have written specifically for this purpose . To run this Macro -
ALT+F11 in your Powerpoint
Right click on your file name in left side pane
Insert Module
Copy and paste the code below
ALT+F8 to open Macro window and Run it. 

Sub PptSaveAsPng()
    Dim FolderName, Arr
    Dim DirName As String, PptName As String, FileName As String
    Dim Sld As Slide
    Dim i As Long
    FileName = "MathSlide"
    With Application.FileDialog(msoFileDialogFolderPicker)
        .Title = "Select a Folder to Store Exported Slides"
        .Show
        If .SelectedItems.Count = 0 Then
            MsgBox "Cancelled"
            Exit Sub
            Else
            FolderName = .SelectedItems(1)
        End If
    End With
    Arr = Split(ActivePresentation.Name, ".")
    PptName = Arr(LBound(Arr))
    On Error Resume Next
    DirName = FolderName & "\" & PptName
    MkDir DirName
    If Err.Number <> 0 Then
        Kill DirName & "\*.*"
    End If
    On Error GoTo 0
    For Each Sld In ActivePresentation.slideS
        i = i + 1
        Sld.Export DirName & "\" & FileName & i & ".png", "png"
    Next Sld
End Sub
