    Dim db                    As DAO.Database
    Dim rs                    As DAO.Recordset
    Dim stText                As String
    Dim sTOEmailList          As String
    Dim strOracleLot          As String

    Set db = CurrentDb()

                              "FROM tbl_EmailAddresses;", dbOpenSnapshot)
