gary's student say:

Hwre is a simple example.  We find the cells and draw an arrow:

 

 

Sub PointTheWay()
Dim rFrom As Range, rTo As Range, r As Range, Arrow As Shape
For Each r In ActiveSheet.UsedRange
    If r.Value = 1 Then Set rFrom = r
    If r.Value = 2 Then Set rTo = r
Next
Set Arrow = ActiveSheet.Shapes.AddLine( _
    beginx:=rFrom.Left + rFrom.Width / 2, _
    beginy:=rFrom.Top + rFrom.Height / 2, _
    endx:=rTo.Left + rTo.Width / 2, _
    endy:=rTo.Top + rTo.Height / 2)
   
With Arrow.Line
    .EndArrowheadStyle = msoArrowheadTriangle
    .EndArrowheadLength = msoArrowheadLengthMedium
    .EndArrowheadWidth = msoArrowheadWidthMedium
End With
End Sub

Sub Test()
  CreateArrow Range("A3"), Range("F5")
End Sub

Sub CreateArrow(FromCell As Range, ToCell As Range, _
  Optional WS As Worksheet = Nothing)
  Dim Arrow As Shape
  If FromCell Is Nothing Or ToCell Is Nothing Then Exit Sub
  If FromCell.Address = ToCell.Address Then Exit Sub
  If WS Is Nothing Then Set WS = ActiveSheet
  Set Arrow = WS.Shapes.AddLine( _
    FromCell.Left + FromCell.Width / 2, _
    FromCell.Top + FromCell.Height / 2, _
    ToCell.Left + ToCell.Width / 2, _
    ToCell.Top + ToCell.Height / 2)
  With Arrow.Line
    .EndArrowheadStyle = msoArrowheadTriangle
  End With
End Sub
