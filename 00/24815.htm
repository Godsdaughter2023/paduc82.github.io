hansv mvp say:

You could override the Save As command, but keep in mind that it's dangerous: if the current document contains macros and if you save it as .docx, you'll lose the macros.

This macro should be stored in a standard module in your default template Normal.dotm:

Public Sub FileSaveAs()
  With Dialogs(wdDialogFileSaveAs)
    .Format = wdFormatXMLDocument
    .Show
  End With
End Sub

graham mayor say:

Unlike Doug, I live in the Eastern Med, so had a bit longer to play. :) Based on your example, the following should do the job. It's a bit Heath Robinson, but it should be OK. Display the worksheet and run the macro, which will create a new sheet 'Label Merge Data'

Option Explicit

Sub FixData()
Dim xlSheet As Worksheet
Dim xlNewSheet As Worksheet
Dim i As Long, j As Long, k As Long
    Set xlSheet = ActiveSheet
    Set xlNewSheet = Sheets.Add
    xlNewSheet.Name = "Label Merge Data"
    xlSheet.UsedRange.UnMerge
    xlNewSheet.Cells(1, 1) = xlSheet.Cells(1, 1)
    xlNewSheet.Cells(1, 2) = xlSheet.Cells(1, 2)
    xlNewSheet.Cells(1, 3) = xlSheet.Cells(1, 3)
    xlNewSheet.Cells(1, 4) = xlSheet.Cells(1, 4)
    xlNewSheet.Cells(1, 5) = xlSheet.Cells(1, 5)

    j = 2
    For i = 2 To xlSheet.UsedRange.Rows.Count
        xlNewSheet.Cells(j, 1) = xlSheet.Cells(i, 1)
        xlNewSheet.Cells(j, 2) = xlSheet.Cells(i, 2)
        xlNewSheet.Cells(j, 3) = xlSheet.Cells(i, 3)
        xlNewSheet.Cells(j, 4) = xlSheet.Cells(i, 4)
        xlNewSheet.Cells(j, 5) = xlSheet.Cells(i, 5)
        For k = 6 To 10
            If Not xlSheet.Cells(i, k) = "" Then
                j = j + 1
                xlNewSheet.Cells(j, 1) = xlSheet.Cells(i, 1)
                xlNewSheet.Cells(j, 2) = xlSheet.Cells(i, 2)
                xlNewSheet.Cells(j, 3) = xlSheet.Cells(i, 3)
                xlNewSheet.Cells(j, 4) = xlSheet.Cells(i, 4)
                xlNewSheet.Cells(j, 5) = xlSheet.Cells(i, k)
            End If
        Next k
        j = j + 1
    Next i
    For i = xlNewSheet.UsedRange.Rows.Count To 1 Step -1
        If xlNewSheet.Cells(i, 5) = "" Then
            xlNewSheet.Rows(i).Delete
        End If
    Next i
End Sub
