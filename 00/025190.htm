lars-Åke aspelin say:

Dear experts,

Is it possible to plot data in wagon wheel style using excel 2007 or 2010. If yes, please suggest the procedure.

Actually i am interested in plotting 20 shop status on one parameter using wagon wheel.  Depending upon the numbers of shops the area of circle will be divided in same nos. of pie  and the radial level ( radius) from the centre of each shop will represent its  present status.

 

I agrees that it can be done very easily using the bar charts but the requirement is very specific.

Assuming that your shop status values are positive and located in cells B1:B20 and the shop names are in cells A1:A20,
try this macro to produce a doughnut chart with one "wheel segment" per shop and with the segment "height" increasing by the the corresponding shop status from 0 giving a vanishing segment upto a maximum value that will give a segment reaching from the wheel axis all the way out to the wheel periphery. An extra helper sheet will be used to hold the data needed for the doughnut chart.
The color of each segment will be the same as the background color of the corresponding status cell in column B unless that cell has a white background in which case the segment will be "autocolored".

Sub waggon_wheel_chart()
  Dim wsdata As Worksheet
  Dim shopnames, shopstatus, wheelsegments As Range
  Dim steps As Integer
  Dim ch As Chart
  Dim se As Series
 
  Set shopnames = Worksheets("Sheet1").Range("A1:A20")
  Set shopstatus = shopnames.Offset(, 1)
  steps = 20  ' more steps give a more accurate chart
  maxstatus = WorksheetFunction.Max(shopstatus)  ' or a fixed value
 
  Set wsdata = Worksheets.Add
  shopnames.Copy Destination:=wsdata.Range("A1")
  Set wheelsegments = wsdata.Range("B1").Resize(shopstatus.Rows.Count)
  For Each c In wheelsegments: c.Value = 1: Next c
 
  Set ch = ActiveWorkbook.Charts.Add
  ch.ChartType = xlDoughnut
  ch.DoughnutGroups(1).DoughnutHoleSize = 10  ' 10 seems to be the minimum
  ch.SetSourceData Source:=shopnames.Resize(, 2)
  ch.HasLegend = True
 
  For i = 1 To steps + 1
    Set se = ch.SeriesCollection.Add(Source:=wheelsegments, RowCol:=xlColumns, SeriesLabels:=False)
    For j = 1 To shopstatus.Rows.Count
      If shopstatus(j).Interior.Color <> RGB(255, 255, 255) Then
        se.Points(j).Interior.Color = shopstatus(j).Interior.Color 'skip this if you want autocolors
      End If
      If shopstatus(j) < maxstatus / steps * (i - 0.5) Then
        se.Points(j).Interior.Color = RGB(255, 255, 255)
        If shopstatus(j) >= maxstatus / steps * (i - 1.5) Then
           se.Points(j).HasDataLabel = True
           se.Points(j).DataLabel.Text = shopstatus(j)
        End If
      End If
    Next j
    If i = steps + 1 Then
      se.Interior.Color = RGB(160, 160, 160)
    End If
  Next i
  ActiveChart.SeriesCollection(1).Delete
End Sub


Hope this helps / Lars-Åke

    ' Creates a PivotTable report from the table on Sheet1
    ' by using the PivotTableWizard method with the PivotFields
    ' method to specify the fields in the PivotTable.
    Dim objTable As PivotTable, objField As PivotField
   
    ' Select the sheet and first cell of the table that contains the data.
    ActiveWorkbook.Sheets("RawData").Select
    Range("A1").Select
   
    ' Create the PivotTable object based on the Employee data on Sheet1.
    Set objTable = Sheet1.PivotTableWizard
   
    ' Specify filter fields.
    Set objField = objTable.PivotFields("Request_Type")
    objField.Orientation = xlPageField
    Set objField = objTable.PivotFields("Completion_Code")
    objField.Orientation = xlPageField
    Set objField = objTable.PivotFields("Category")
    objField.Orientation = xlPageField
    Set objField = objTable.PivotFields("Team_Assigned")
    objField.Orientation = xlPageField
   
    ' Specify column fields.
    Set objField = objTable.PivotFields("Date_Submitted")
    objField.Orientation = xlRowField
   
    ' Filter Fields
    ActiveSheet.PivotTables("PivotTable1").PivotFields("Request_Type"). _
        ClearAllFilters
    ActiveSheet.PivotTables("PivotTable1").PivotFields("Request_Type").CurrentPage _
        = "Ticket"
    ActiveSheet.PivotTables("PivotTable1").PivotFields("Completion_Code"). _
        CurrentPage = "(All)"
    With ActiveSheet.PivotTables("PivotTable1").PivotFields("Completion_Code")
        .PivotItems("Duplicate").Visible = False
        .PivotItems("Withdrawn").Visible = False
    End With
    ActiveSheet.PivotTables("PivotTable1").PivotFields("Completion_Code"). _
        EnableMultiplePageItems = True
    ActiveSheet.PivotTables("PivotTable1").PivotFields("Category").ClearAllFilters
    ActiveSheet.PivotTables("PivotTable1").PivotFields("Category").CurrentPage = _
        "Benefits"
   
    ' Specify a data field with its summary
    ' function and format.
    Set objField = objTable.PivotFields("Ticket_Number")
    objField.Orientation = xlDataField
    objField.Function = xlCount
    objField.NumberFormat = "Number"
   
    ' Rename sheet
      Sheets("Sheet2").Select
    Sheets("Sheet2").Move After:=Sheets(2)
    Sheets("Sheet2").Select
    Sheets("Sheet2").Name = "Benefits"
    Range("D34").Select
 End Sub
