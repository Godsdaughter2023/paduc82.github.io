rich007a say:

This example might help:

 

Sub CallMyAutoFilter()
     ' Call the MyAutoFilter procedure, filter the 2nd column in the
     ' Autofilter range (col E) with the value in A1:
    MyAutoFilter rngCriteria:=Worksheets("Sheet1").Range("A1"), lnField:=2
   
    Stop 'The above will be undone by the following....
   
     ' Call the MyAutoFilter procedure, filter the 1st column in the
     ' Autofilter range with the value in A2:
    MyAutoFilter rngCriteria:=Worksheets("Sheet1").Range("A2"), lnField:=1
End Sub


Sub MyAutoFilter(rngCriteria As Range, Optional lnField As Long = 1)
' Note that Field is The integer offset of the field on which you want to base the filter
' (from the left of the list; the leftmost field is field one).  In this example, field=1
' refers to column D, since rngToFilter is D5:F5.
'
    Dim ws As Worksheet
    Dim rngToFilter As Range
   
    Set ws = Worksheets("Sheet1")
    Set rngToFilter = ws.Range("D5:F5") 'Range of headings for data to be filtered
   
    If Not ws.AutoFilterMode Then rngToFilter.AutoFilter
   
    On Error Resume Next
    ws.ShowAllData
    On Error GoTo 0
   
    rngToFilter.AutoFilter Field:=lnField, Criteria1:=rngCriteria(1, 1).Value
End Sub
