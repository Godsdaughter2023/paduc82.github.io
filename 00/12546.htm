jeeped say:
The runtime error 13 is a type mismatch which leads me to believe that you are attempting to run this on text values or no value at all (blank cell(s)). as that will produce the error when processing a mathematical operation like subtraction.

While it would be best if you fixed the data so that text and/or blank values were not processed, the following will use the SUM() worksheet function for error control by converting text and blank cells to 0 (zero).

Public Function udf_Min_Abs_Diff(rMinuends As Range, rSubtrahend As Range) As Double
    Dim dRtrn As Double, dMin As Double, r As Range, app As Application
    Set app = Application
    dMin = 1E+99
    For Each r In rMinuends
        If Abs(app.Sum(r.Value) - app.Sum(rSubtrahend)) < dMin Then
            dRtrn = app.Sum(r.Value)
            dMin = Abs(app.Sum(r.Value) - app.Sum(rSubtrahend))
        End If
    Next r
    Set app = Nothing
    udf_Min_Abs_Diff = dRtrn
End Function
