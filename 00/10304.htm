norman david jones say:

Hi Bruce,

 To Hide the relevant rows on demand, try something like:

  • Alt-F11 to open the VBA editor
  • Alt-IM to insert a new code module
  • In the new module, paste the following code
'=========>>
Option Explicit

'--------->>
Public Sub HideRows()
    Dim WB As Workbook
    Dim SH As Worksheet
    Dim Rng As Range
    Dim rCell As Range
    Dim hideRng As Range
    Dim iLastRow As Long
    Dim CalcMode As Long

    Set WB = ThisWorkbook
    Set SH = WB.Sheets("Sheet1")           '<<==== Change to suit

    With SH
        iLastRow = LastRow(SH, .Columns("A:C"))
        Set Rng = .Range("A2:A" & iLastRow)
    End With

   On Error GoTo XIT
    With Application
        CalcMode = .Calculation
        .Calculation = xlCalculationManual
        .ScreenUpdating = False
    End With

    For Each rCell In Rng.Columns(1).Cells
        With rCell
            If IsEmpty(.Value) _
               And IsEmpty(.Offset(0, 1).Value) _
               And Not IsEmpty(.Offset(0, 2).Value) Then
                If hideRng Is Nothing Then
                    Set hideRng = rCell
                Else
                    Set hideRng = Union(rCell, hideRng)
                End If
            End If
        End With
    Next rCell

    If Not hideRng Is Nothing Then
    Application.EnableEvents = False
            hideRng.EntireRow.Hidden = True
    Else
        'nothing found, do nothing
    End If
XIT:
    With Application
        .Calculation = CalcMode
        .ScreenUpdating = True
        .EnableEvents = True
    End With
End Sub

'--------->>
Public Function LastRow(SH As Worksheet, _
                        Optional Rng As Range)
    If Rng Is Nothing Then
        Set Rng = SH.Cells
    End If

    On Error Resume Next
    LastRow = Rng.Find(What:="*", _
                       after:=Rng.Cells(1), _
                       Lookat:=xlPart, _
                       LookIn:=xlFormulas, _
                       SearchOrder:=xlByRows, _
                       SearchDirection:=xlPrevious, _
                       MatchCase:=False).Row
    On Error GoTo 0
End Function
'<<=========
  • Alt-Q to close the VBA editor and return to Excel
  • Alt-F8 to open the macro window
  • Select HideRows | Run

If you wish to hide/unhide rows dynamically, as cells are populated or cleared, having run the above macro, try as follows:

  • Right-click the sheet tab
  • Select the View Code option from the resultant context menu
  • Paste the following event code:
'=========>>
Option Explicit

'--------->>
Private Sub Worksheet_Change(ByVal Target As Range)
    Dim Rng As Range, rRow As Range

    Set Rng = Intersect(Target, Me.Range("A:C"))

    If Not Rng Is Nothing Then
        For Each rRow In Rng.Rows
            With rRow.EntireRow
                .Hidden = IsEmpty(.Cells(1).Value) _
                          And IsEmpty(.Cells(2).Value) _
                          And Not IsEmpty(.Cells(3).Value)
            End With
        Next rRow
    End If
End Sub
'<<=========
  • Alt-Q to close the VBA editor and return to Excel.

talking moose say:
Outlook has no mechanism to perform blanket purges of messages from just anywhere. However, you can designate folders to purge mail older than a certain number of days using a schedule.

Go to Outlook menu --> Preferences... --> Schedules and click the + (plus) button to create a new schedule. Set the new schedule to run whenever you prefer such as startup, quit or on a timed schedule. Set the action to delete mail from your selected folder that is more than X number of days old.

Hope this helps!
