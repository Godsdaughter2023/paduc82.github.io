macropod say:

This kind of thing is usually only required after a server change, in which case you have more than just Word links to external files to worry about. For example, when templates other than Word’s Normal template are used to create a document, the template’s path & name are stored with the document. If that path is a network path, a change to the server name will break the link. The result can be significant delays in opening the documents on the new server. See: http://support.microsoft.com/?kbid=830561. The same effect occurs when the file is opened on a computer attached to a different network.

The following macro can be used to update links & template paths. If a new template path can’t be found, to point it to Word’s Normal template. Code is included to restore the original date/time stamps of the updated files.

Simply replace however much of the old & new template/link paths differ in the variables StrOldServer, and StrNewServer. You can, of course, comment-out/delete the code relating to template path updates if that's not a concern.

Option Explicit
Dim FSO As Object 'a FileSystemObject
Dim oFolder As Object 'the folder object
Dim oSubFolder As Object 'the subfolders collection
Dim oFiles As Object 'the files object
Dim StrOldServer As String, StrNewServer As String 'the StrOldServer and StrNewServer paths.
Dim i As Long, j As Long
'
Sub Main()
' Minimise screen flickering
Application.ScreenUpdating = False
Dim StrFolder As String
' Browse for the starting folder
StrFolder = GetTopFolder
If StrFolder = "" Then Exit Sub
' Insert however much of the left end of the server
' paths differ as the StrOldServer and StrNewServer variables.
StrOldServer = "\\TSB\VOL1": StrNewServer = "\\TSLSERVER\Files"
i = 0: j = 0
' Search the top-level folder
Call GetFolder(StrFolder & "\")
' Search the subfolders for more files
Call SearchSubFolders(StrFolder)
' Return control of status bar to Word
Application.StatusBar = ""
' Restore screen updating
Application.ScreenUpdating = True
MsgBox i & " of " & j & " files updated.", vbOKOnly
End Sub
'
Function GetTopFolder() As String
GetTopFolder = ""
Set oFolder = CreateObject("Shell.Application").BrowseForFolder(0, "Choose a folder", 0)
If (Not oFolder Is Nothing) Then GetTopFolder = oFolder.Items.Item.Path
Set oFolder = Nothing
End Function
'
Sub SearchSubFolders(strStartPath As String)
If FSO Is Nothing Then Set FSO = CreateObject("Scripting.FileSystemObject")
Set oFolder = FSO.GetFolder(strStartPath)
Set oSubFolder = oFolder.subfolders
For Each oFolder In oSubFolder
  Set oFiles = oFolder.Files
  ' Search the current folder
  Call GetFolder(oFolder.Path & "\")
  ' Call ourself to see if there are subfolders below
  Call SearchSubFolders(oFolder.Path)
Next
Set FSO = Nothing
End Sub
'
Sub GetFolder(StrFolder As String)
Dim strFile As String
strFile = Dir(StrFolder & "*.doc")
' Process the files in the folder
While strFile <> ""
  ' Update the status bar is just to let us know where we are
  Application.StatusBar = StrFolder & strFile
  Call UpdateRefPaths(StrFolder & strFile)
  strFile = Dir()
Wend
End Sub
'
Sub UpdateRefPaths(strDoc As String)
' This sub updates the template paths for files after a server change.
Dim strPath As String, oItem As Object, StrDtTm As String
Dim Rng As Range, Fld As Field, Shp As Shape, iShp As InlineShape
Dim StrOldPath As String, StrNewPath As String
' Store the file's current Date/Time stamp.
If FSO Is Nothing Then Set FSO = CreateObject("Scripting.FileSystemObject")
Set oItem = FSO.GetFile(strDoc)
StrDtTm = oItem.DateLastModified
' Open the document
Documents.Open strDoc, AddToRecentFiles:=False, ReadOnly:=False, Format:=wdOpenFormatAuto
With ActiveDocument
  If .ProtectionType = wdNoProtection Then
    ' Update the template path
    strPath = Dialogs(wdDialogToolsTemplates).Template
    If LCase(Left(strPath, Len(StrOldServer))) = LCase(StrOldServer) Then
      ' Update the file counter for changed files
      i = i + 1
      ' Get the new template path & name
      strPath = StrNewServer & Mid(strPath, Len(StrOldServer) + 1)
      ' Check whether the template exists
      If Dir(strPath) <> "" Then
        ' If found, update the path
        .AttachedTemplate = StrNewServer & Mid(strPath, Len(StrOldServer) + 1)
      Else
        ' If not found, reset the template to 'Normal'
        .AttachedTemplate = ""
        ' Output an error report in the document from which the macro is run.
        ThisDocument.Range.InsertAfter vbCr & "Template: " & strPath & " not found for " & strDoc
      End If
    End If
    ' Update link paths for Word's INCLUDEPICTURE, INCLUDETEXT, LINK, RD and HYPERLINK fields.
    ' Only HYPERLINK fields pointing to documents on the old server are updated.
    ' Fields in all story ranges in the document are processed.
    For Each Rng In .StoryRanges
      ' Go through the shapes in the story range.
      For Each Shp In Rng.ShapeRange
        With Shp
          ' Skip over shapes that don't have links to external files.
          If Not .LinkFormat Is Nothing Then
            With .LinkFormat
              ' Replace the link to the external file if it differs.
              StrOldPath = Left(.SourceFullName, InStrRev(.SourceFullName, "\"))
              StrNewPath = Replace(StrOldPath, StrOldServer, StrNewServer)
              If StrOldPath <> StrNewPath Then
                .SourceFullName = Replace(.SourceFullName, StrOldPath, StrNewPath)
              End If
            End With
          End If
        End With
      Next Shp
      ' Go through the inlineshapes in the story range.
      For Each iShp In Rng.InlineShapes
        With iShp
          ' Skip over inlineshapes that don't have links to external files.
          If Not .LinkFormat Is Nothing Then
            With .LinkFormat
              ' Replace the link to the external file if it differs.
              StrOldPath = Left(.SourceFullName, InStrRev(.SourceFullName, "\"))
              StrNewPath = Replace(StrOldPath, StrOldServer, StrNewServer)
              If StrOldPath <> StrNewPath Then
                .SourceFullName = Replace(.SourceFullName, StrOldPath, StrNewPath)
              End If
            End With
          End If
        End With
      Next iShp
      ' Go through the fields in the story range.
      For Each Fld In Rng.Fields
        With Fld
          ' Skip over fields that don't have links to external files.
          If Not .LinkFormat Is Nothing Then
            With .LinkFormat
              ' Replace the link to the external file if it differs.
              StrOldPath = Left(.SourceFullName, InStrRev(.SourceFullName, "\"))
              StrNewPath = Replace(StrOldPath, StrOldServer, StrNewServer)
              If StrOldPath <> StrNewPath Then
                .SourceFullName = Replace(.SourceFullName, StrOldPath, StrNewPath)
              End If
            End With
          End If
        End With
      Next Fld
    Next Rng
  Else
    ' Output a 'protected' file report in the document from which the macro is run.
    ThisDocument.Range.InsertAfter vbCr & strDoc & " protected. Not updated."
  End If
  .Close SaveChanges:=True
End With
' Update the main file counter
j = j + 1
' Let Word do its housekeeping
DoEvents
' Reset the file's Date/Time stamp.
Set oItem = FSO.GetFile(strDoc)
If oItem.DateLastModified <> StrDtTm Then oItem.DateLastModified = StrDtTm
Set oItem = Nothing
End Sub

Assuming you have a proper table with one row of headers (column names) with no empty rows or columns (empty individual cells are OK), and no merged cells there are at least two ways to accomplish your task.
