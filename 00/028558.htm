Function TrimWhite(ByVal S As String) As String
  'Return a string with white space removed
  Dim i As Long
  For i = 1 To Len(S)
    If Asc(Mid$(S, i, 1)) < 32 Then Mid$(S, i, 1) = vbNullChar
  Next
  TrimWhite = Replace$(S, vbNullChar, "")
End Function
