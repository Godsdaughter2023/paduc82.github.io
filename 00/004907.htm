Hello ANNIECHAN1.

As far as I know this is a software bug that occurs in all versions of Outlook
Try these options:

Ensure that you are using drag and drop to copy/move items of the same type.
The drag & drop feature works only when you copy/move items between the folders of the same type. Therefore, you cannot copy e.g. calendar items to a folder with emails.
Press the ESC key repeatedly.
Make sure that the Outlook window is active, and press the ESC key several times. After that, you should be able to use the drag & drop feature again.
Use the Folder List view.
Try to change the view to Folders and try dragging and dropping your items again. We have discovered that Outlook behaves differently depending on the view type.
Reset Outlook folders.
It might happen that certain Outlook folders are corrupted (for example, they do not have a valid type identifier), which causes the problem with drag & drop. You can use a command-line switch that restores missing folders at the default delivery location and recreates their properties:

Close your Outlook.
Launch the Run command (e.g. by pressing the Windows logo key + R).
Type the following command: outlook.exe /resetfolders, and click OK to execute it. This should repair the default Outlook folders and restore the drag & drop capability.

Dim vShp() As Variant

Private Sub Align_Shapes(sLayout As String)


    Set ppApp = Application
    Set ppPres = ppApp.ActivePresentation
    iCur = ActiveWindow.View.Slide.SlideNumber
    Set ppSld = ppPres.Slides(iCur)

    Select Case sLayout

        Case "FourSquare"
            iMax = 4
            i = 0
            For Each ppShp In ActiveWindow.Selection.SlideRange.Shapes.Range
                If IsTitle(ppShp) = False Then i = i + 1
            Next ppShp
            If i < iMax Or i > iMax Then
                MsgBox ("You have selected " & i & " shapes on the current slide." & Chr(10) & _
                        "You must have " & iMax & " shapes selected to align 'FourSquare'.")
                Exit Sub
            Else
                'Correct number of shapes
                ReDim vShp(4)
                'Build array of selected shapes
                i = 0
                For Each ppShp In ActiveWindow.Selection.SlideRange.Shapes.Range
                    If IsTitle(ppShp) = False Then vShp(i) = ppShp.Name
                    i = i + 1
                Next ppShp
                'Get height of Title
                For Each ppShp In ppSld.Shapes
                    If IsTitle(ppShp) Then iTitleHeight = ppShp.Height
                Next ppShp
                'Make shapes same size
                If optSameWidth.Value = True Or optSameBoth.Value = True Then
                    ppSld.Shapes(vShp(1)).Width = ppSld.Shapes(vShp(0)).Width
                    ppSld.Shapes(vShp(2)).Width = ppSld.Shapes(vShp(0)).Width
                    ppSld.Shapes(vShp(3)).Width = ppSld.Shapes(vShp(0)).Width
                End If
                If optSameHeight.Value = True Or optSameBoth.Value = True Then
                    ppSld.Shapes(vShp(1)).Height = ppSld.Shapes(vShp(0)).Height
                    ppSld.Shapes(vShp(2)).Height = ppSld.Shapes(vShp(0)).Height
                    ppSld.Shapes(vShp(3)).Height = ppSld.Shapes(vShp(0)).Height
                End If
                'PowerPoint Slides are 720 width and 540 height
                'Top Left
                ppSld.Shapes(vShp(0)).Top = iTitleHeight + (((270 - iTitleHeight) - ppSld.Shapes(vShp(0)).Height) / 2)
                ppSld.Shapes(vShp(0)).Left = (360 - ppSld.Shapes(vShp(0)).Width) / 2
                'Bottom Left
                ppSld.Shapes(vShp(1)).Top = iTitleHeight + (270 - iTitleHeight) + (((270 - iTitleHeight) - ppSld.Shapes(vShp(0)).Height) / 2)
                ppSld.Shapes(vShp(1)).Left = (360 - ppSld.Shapes(vShp(0)).Width) / 2
                'Top Right
                ppSld.Shapes(vShp(2)).Top = iTitleHeight + (((270 - iTitleHeight) - ppSld.Shapes(vShp(0)).Height) / 2)
                ppSld.Shapes(vShp(2)).Left = 360 + ((360 - ppSld.Shapes(vShp(0)).Width) / 2)
                'Bottom Right
                ppSld.Shapes(vShp(3)).Top = iTitleHeight + (270 - iTitleHeight) + (((270 - iTitleHeight) - ppSld.Shapes(vShp(0)).Height) / 2)
                ppSld.Shapes(vShp(3)).Left = 360 + ((360 - ppSld.Shapes(vShp(0)).Width) / 2)
            End If
        Case "Diamond"
            'Do nothing yet
        Case "Triangle"
            'Do nothing yet
        Case "Circle"
            'Do nothing yet
    End Select

End Sub

 

Function IsTitle(oSh As Shape) As Boolean

    IsTitle = False
    If oSh.Type = msoPlaceholder Then
        If oSh.PlaceholderFormat.Type = ppPlaceholderCenterTitle Or oSh.PlaceholderFormat.Type = ppPlaceholderTitle Then IsTitle = True
    End If

End Function
