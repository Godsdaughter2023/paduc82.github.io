Get-AppxPackage -allusers Microsoft.SkypeApp | Remove-AppxPackage

get-appxpackage -AllUsers Microsoft.SkypeApp | foreach { Add-AppxPackage -DisableDevelopmentMode -Register "$($_.InstallLocation)\AppXManifest.xml" }
