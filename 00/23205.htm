Sub BreakOnSection()
    ' Select a folder

      .Title = "Select the folder into which the documents will be saved."
      If .Show = -1 Then

    For i = 1 To ((ActiveDocument.Sections.Count) - 1)

        ActiveDocument.SaveAs

Sub Merge_To_Individual_Files()
'Merges one record at a time to the folder containing the mailmerge main document.
' Sourced from: http://windowssecrets.com/forums/showthread.php/163017-Word-Mailmerge-Tips-amp-Tricks
Application.ScreenUpdating = False
Dim StrFolder As String, StrName As String, MainDoc As Document, i As Long, j As Long
Set MainDoc = ActiveDocument
With MainDoc
  StrFolder = .Path & Application.PathSeparator
  For i = 1 To .MailMerge.DataSource.RecordCount
    With .MailMerge
      .Destination = wdSendToNewDocument
      .SuppressBlankLines = True
      With .DataSource
        .FirstRecord = i
        .LastRecord = i
        .ActiveRecord = i
        If Trim(.DataFields("Accident_Ref_No")) = "" Then Exit For
        StrName = .DataFields("Accident_Ref_No")
      End With
      .Execute Pause:=False
    End With
    For j = 1 To 255
      Select Case j
        Case 1 To 31, 33, 34, 37, 42, 44, 46, 47, 58 - 63, 91 - 93, 96, 124, 147, 148
        StrName = Replace(StrName, Chr(j), "")
      End Select
    Next
    StrName = Trim(StrName)
    With ActiveDocument
      .SaveAs filename:=StrPath & StrName & ".docx", FileFormat:=wdFormatXMLDocument, AddToRecentFiles:=False
      ' and/or:
      .SaveAs filename:=StrPath & StrName & ".pdf", FileFormat:=wdFormatPDF, AddToRecentFiles:=False
      .Close SaveChanges:=False
    End With
  Next i
End With
Application.ScreenUpdating = True
End Sub

linley meslier say:

Use a registry editor to change the values back to Calendar and Contacts.

 

Go to HKCU\Software\Microsoft\Today\Keys\112\OPen and change the value for Calendar to "\Windows\Calendar.exe" (including quotes)

Repeat the same steps for Contacts HKCU\Software\Microsoft\Today\Keys\113 and type ":MSContacts"

 

You do this at your own risk.
