Function GetNthWord(From As String, Delim As String, n As Long) As String
    Dim lngStart As Long
    Dim lngEnd As Long
    Dim lngLen As Long
    Dim i As Long
    Dim lngCount As Long
    Dim blnInQuote As Boolean
    lngCount = 1
    lngStart = 1
    lngLen = Len(From)
    For i = 1 To lngLen
        Select Case Mid(From, i, 1)
            Case Delim
                If blnInQuote Then
                    ' ignore
                ElseIf lngCount = n Then
                    GetNthWord = Mid(From, lngStart, i - lngStart)
                    Exit Function
                Else
                    lngCount = lngCount + 1
                    lngStart = i + 1
                End If
            Case Chr(34) ' double quote
                If blnInQuote Then
                    If lngCount = n Then
                        GetNthWord = Mid(From, lngStart, i - lngStart)
                        Exit Function
                    Else
                        blnInQuote = False
                    End If
                Else
                    blnInQuote = True
                    lngStart = i + 1
                End If
        End Select
    Next i
    If lngCount = n Then
        GetNthWord = Mid(From, lngStart)
    End If
End Function

Function GetNthWord(strFrom As String, _
    strDelim As String, n As Long) As String
   
    Dim arrSplit As Variant, arrWords As Variant
    Dim i As Long, j As Long
   
    On Error GoTo ErrHandler
   
    arrSplit = Split(strFrom, strDelim)

    ReDim arrWords(0)
    j = UBound(arrWords)
   
    For i = LBound(arrSplit) To UBound(arrSplit)
        ReDim Preserve arrWords(j)
        If Left(arrSplit(i), 1) = """" Then 'found first word in spaced word:
            If Right(arrSplit(i), 1) = """" Then  'spaced word only contains one word!:
                arrWords(j) = Mid(arrSplit(i), 2, Len(arrSplit(i)) - 2)
            Else
                arrWords(j) = Mid(arrSplit(i), 2)   'remove the speachmark
                If i < UBound(arrSplit) - LBound(arrSplit) Then
                    Do
                        i = i + 1
                        If Right(arrSplit(i), 1) = """" Then 'found last word in spaced word:
                            arrWords(j) = arrWords(j) & " " & Left(arrSplit(i), Len(arrSplit(i)) - 1)
                        Else
                            arrWords(j) = arrWords(j) & " " & arrSplit(i)
                        End If
                    Loop While Right(arrSplit(i), 1) <> """" And i < UBound(arrSplit) - LBound(arrSplit)
                End If
            End If
        Else
            arrWords(j) = arrSplit(i)
        End If
        j = j + 1
    Next i

     'Output:
    If n > UBound(arrWords) - LBound(arrWords) + 1 Then
        GetNthWord = ""
    Else
        GetNthWord = arrWords(n - 1)
    End If
   
    On Error GoTo 0
Exit Function

ErrHandler:
    GetNthWord = ""
End Function

chip pearson say:

Try the following function:

Function NthWord(InputString As String, _
        SplitChar As String, Nth As Long) As String
    Dim T As String
    Dim SS() As String
    Dim N  As Long
    Dim InQuote As Boolean
    Dim C As String
   
    For N = 1 To Len(InputString)
        C = Mid(InputString, N, 1)
        If StrComp(C, Chr(34), vbTextCompare) = 0 Then
            InQuote = Not InQuote
        End If
        If InQuote And StrComp(C, SplitChar, vbTextCompare) = 0 Then
            T = T & Chr(0)
        Else
            T = T & C
        End If
    Next N
    SS = Split(T, SplitChar)
    NthWord = Replace(Replace(SS(Nth - 1), Chr(0), SplitChar), _
        Chr(34), vbNullString)
End Function


InputString is the string to split into words, and Nth is the 1-based word you want to return for InString. You can jazz it up with some error handling (e.g., if Nth < 0) but the core logic works fine as long as the inputs are valid. You cannot use Chr(0) as the SplitChar. Any other character is valid.

 

You can call this function with code like the following:

 

Dim Nth As Long
Dim S As String
Dim SplitChar As String
Dim Word As String
S = "one ""two two"" three ""four four"""
SplitChar = Chr(32)
Word = NthWord(S, SplitChar, 3)
Debug.Print Word

These are not Styles, but Header Building Blocks.

Insert > Header

Whisp and Semaphore both have elements of what you are discussing. [Edit: These are in Word 2016, not in Word 2010.]

If you go into Insert > Header > Edit Header, you can certainly create your own.

To save one you would use Alt+F3 and give your Header a name. Many people use an _underscore at the beginning of the name so that it will show up first in the list given in the Gallery. If you mean to distribute this to others, consider saving it in a separate template that can be distributed to be used as a Global Template placed in the Word Startup Folder.

Where can Building Blocks be stored?

Author and Title are Document Property Content Controls. There is no built-in document property for subtitle. You can certainly use one of the others for this purpose.  Repeating Data Using Document Property Content Controls and Other Mapped Content Controls

Date is likely a field. Using Date Fields in Microsoft Word
