    *************************************************************************************************

    Private Sub Worksheet_SelectionChange(ByVal Target As Range)


    Dim MyRng As Range

    Set MyRng = Range("A:A")  ''' Sets the Target Range

    On Error GoTo Stp     ''' Case rectangle linked to target row does not exist

    If Not Application.Intersect(Target, MyRng) Is Nothing Then

            TRow = Target.Row  '' Target row = Rectangle Index number

            ''Fill previous shapes with white color and fill Rectangle Index number =Target row with green color

            ActiveSheet.Shapes.SelectAll

            With Selection.ShapeRange.Fill

            .ForeColor.RGB = vbWhite

            Target.Activate

             ActiveSheet.Shapes("Rectangle " & TRow).Fill.ForeColor.RGB = vbGreen

            End With

    End If

    Stp:

    End Sub
