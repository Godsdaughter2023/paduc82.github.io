andreas killer say:

However, if anyone has a better approach of identifying the coordinates of the shape to return it to it's original position or another method, that would be very helpful.

Sure. ;-)

When you want to restore the position, you have the save the values of the Top, Left and Rotation properties before you touch the shape, then move it anywhere you like and finally restore the saved values.

Make a new file, add a regular module, copy the code below into the module, close the VBA editor, create a rounded rectangle anywhere on the visible screen, press Alt-F8 and run the code.

You can interrupt the code with CTRL-BREAK as noted in the status bar.

After the first run create an oval near the 1st shape and run the code again. Enjoy.

I'm on vacation next week, so an answer take a while if you have furthermore questions.

Andreas.

Option Explicit

#If Win64 Then
Private Declare PtrSafe Sub Sleep Lib "kernel32" (ByVal dwMilliseconds As Long)
#Else
Private Declare Sub Sleep Lib "kernel32" (ByVal dwMilliseconds As Long)
#End If

Sub Main()
  Dim C As New Collection
  Dim S As Shape
  Dim Item
  Dim NewTop As Double, NewLeft As Double, NewRot As Double
  Dim hDir As Integer, vDir As Integer
 
  'Be sure no shape is selected (makes the moving smoother)
  Range("A1").Select
 
  'Collect all rectangles, ovals, etc.
  ReDim Item(0 To 3)
  For Each S In ActiveSheet.Shapes
    If S.Type = msoAutoShape Then
      'Store the shape itself
      Set Item(0) = S
      'Save the properties that we modify
      Item(1) = S.Top
      Item(2) = S.Left
      Item(3) = S.Rotation
      'Store the array into the collection
      C.Add Item
    End If
  Next
 
  'Allow the user to interrupt
  On Error GoTo ExitPoint
  Application.EnableCancelKey = xlErrorHandler
  Application.StatusBar = "Press CTRL-Break to stop moving"
 
  'Setup the primary direction
  hDir = 1
  vDir = 1
 
  'Start an endless loop
  Do
    For Each Item In C
      'Get the shape back
      Set S = Item(0)
      'Move it
      With S
        .Top = .Top + 10 * vDir
        .Left = .Left + 10 * hDir
        .Rotation = .Rotation + 10 * vDir * hDir
      End With
      'Don't go bejond the visible screen
      With ActiveWindow.VisibleRange
        If S.Top + S.Height > .Height Then vDir = -1
        If S.Top < 10 Then vDir = 1
        If S.Left + S.Width > .Width Then hDir = -1
        If S.Left < 10 Then hDir = 1
      End With
      'Let Excel update the screen
      DoEvents
      'Wait a moment
      Sleep 20 / C.Count
    Next
  Loop
 
ExitPoint:
  'Restore the initial position
  For Each Item In C
    Set S = Item(0)
    With S
      .Top = Item(1)
      .Left = Item(2)
      .Rotation = Item(3)
    End With
  Next
  Application.StatusBar = False
End Sub
