     AND PR2.Sample_Event = PR1.Sample_Event
     GROUP BY Park_Unit,Macroplot_ID,Sample_Event) > 1
ORDER BY Park_Unit,Macroplot_ID,Sample_Event;

or this:

SELECT *
FROM PlotReports As PR1
WHERE Protocols="Trees - Individuals (metric)"
AND EXISTS
    (SELECT Park_Unit, Macroplot_ID,Sample_Event
     FROM PlotReports As PR2
     WHERE PR2.Park_Unit = PR1.Park_Unit
     AND PR2.Macroplot_ID = PR1.Macroplot_ID
     AND PR2.Sample_Event = PR1.Sample_Event
     GROUP BY Park_Unit, Macroplot_ID,Sample_Event
     HAVING COUNT(*) >1)
ORDER BY Park_Unit,Macroplot_ID,Sample_Event;

or this:

SELECT *
FROM PlotReports As PR1
WHERE Protocols="Trees - Individuals (metric)"
AND EXISTS
    (SELECT *
     FROM PlotReports As PR2
     WHERE PR2.Park_Unit = PR1.Park_Unit
     AND PR2.Macroplot_ID = PR1.Macroplot_ID
     AND PR2.Sample_Event = PR1.Sample_Event
     AND PR2.MyID <> PR1.MyID)
ORDER BY Park_Unit,Macroplot_ID,Sample_Event;

In the last one 'MyID' is the table's primary key column, e.g. an autonumber.  The first and last of the above alternatives will probably be the most efficient.
