Hint in Textbox Ms Access : Code - Microsoft Community

techittk say:

Hint in Textbox Ms Access : Code

This code test just only one textbox (Update)

1. Set default value to UserID (Design View: Format)

2. Set FontItalic to Yes (Design View: Format)

Private Sub txtid_Change()
    If Me.txtid.Text = "UserID" Then
        Me.txtid.Text = "UserID"
        Me.txtid.FontItalic = True
    ElseIf Me.txtid.Text = "" Then
        Me.txtid.Text = "UserID"
        Me.txtid.FontItalic = True
    End If
End Sub

Private Sub txtid_KeyPress(KeyAscii As Integer)
    If Me.txtid.Text = "UserID" Then
        Me.txtid = ""
        Me.txtid.FontItalic = False
    End If
    Select Case KeyAscii
    Case 65 To 90, 97 To 122, 48 To 57
    Case vbKeyBack
    Case Else
        KeyAscii = 0
        Exit Sub
    End Select
End Sub

Private Sub txtid_LostFocus()
    If Me.txtid.Text = "UserID" Then
        Me.txtid.Text = "UserID"
        Me.txtid.FontItalic = True
    ElseIf Me.txtid.Text = "" Then
        Me.txtid.Text = "UserID"
        Me.txtid.FontItalic = True
    End If
End Sub
