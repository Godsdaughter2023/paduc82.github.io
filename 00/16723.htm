Sub SendEmail()

    Dim ws As Worksheet
    Dim strFileToAttach As String
    Dim objOutlook As Object
    Dim objMailItem As Object
    Dim objNameSpace As Object
    Dim strFrom As String
    Dim arrEmpId()
    Dim arrEmpIdUnique()
    Dim rngToSend As Range
    Dim rngCel As Range
    Dim i As Long
    Dim j As Long
    Dim varTemp
    Dim rngTo As Range
    Dim rngCC As Range
    Dim strBody As String
    Dim rng As Range
    Dim dteReport As Date
    Dim lngCounter As Long
    Dim lngNumbToSend As Long
    Dim lngColNumb As Long
   
    '******************************************************************************
    'Edit code between asterisk lines to suit your project
    Set ws = Sheets("Sheet1")   'Edit "Sheet1" to the worksheet with the data
    lngNumbToSend = 10           'Edit to number of emails to send in a batch.
    lngColNumb = ws.Columns("J").Column   'Edit "J" to column for marking Report Date
   
    'Edit between the double quotes in following line with
    'the path and file name for attachment.
    strFileToAttach = "C:\Users\User\Documents\Word\My Test Doc.docx"
    '******************************************************************************
   
    dteReport = Date
    Do
        If Format(dteReport, "dddd") <> "Friday" Then
            dteReport = dteReport - 1
        Else
            Exit Do
        End If
    Loop
  
    With ws
        .AutoFilterMode = False
        .Range("A1").AutoFilter
        'Excludes the alredy sent lines
        With .AutoFilter.Range
            'Set filter to exclude emails already sent for report date
            .AutoFilter Field:=lngColNumb, Criteria1:="<>" _
                            & Format(dteReport, "dddd dd mmmm yyyy")
           
            'Assign visible cells of column A to a range variable (Excludes already sent)
            Set rngToSend = .Resize(.Rows.Count, 1) _
                            .SpecialCells(xlCellTypeVisible)
           
            'If only 1 cell then only column headers visible so finished
            If rngToSend.Cells.Count = 1 Then
                MsgBox "All emails have been sent for " _
                        & Format(dteReport, "dddd dd mmmm yyyy")
                ws.AutoFilterMode = False
                Exit Sub
            End If
           
            Set rngToSend = .Offset(1, 0) _
                            .Resize(.Rows.Count - 1, 1) _
                            .SpecialCells(xlCellTypeVisible)
        End With
        .AutoFilterMode = False
       
        'Copy the values in rngToSend to an array
        j = 0
        For Each rngCel In rngToSend
            j = j + 1
            ReDim Preserve arrEmpId(1 To j)
            arrEmpId(j) = rngCel.Value
        Next rngCel
    End With
   
    'Sort the array
    For i = LBound(arrEmpId) To UBound(arrEmpId) - 1
        For j = LBound(arrEmpId) To UBound(arrEmpId) - 1
            If arrEmpId(j) > arrEmpId(j + 1) Then
                varTemp = arrEmpId(j)
                arrEmpId(j) = arrEmpId(j + 1)
                arrEmpId(j + 1) = varTemp
            End If
        Next j
    Next i
   
    'Create array of unique values (up to 2nd last element)
    j = 0
    For i = LBound(arrEmpId) To UBound(arrEmpId) - 1
        If arrEmpId(i) <> arrEmpId(i + 1) Then
            j = j + 1
            ReDim Preserve arrEmpIdUnique(1 To j)
            arrEmpIdUnique(j) = arrEmpId(i)
        End If
    Next i
    'Add last element to Unique array
    j = j + 1
    ReDim Preserve arrEmpIdUnique(1 To j)
    arrEmpIdUnique(j) = arrEmpId(i)
   
    'In following line "<br />" are line feeds
    strBody = "Hi," & "<br />  <br />" _
                    & "Please find the below details for the week ending " _
                    & Format(dteReport, "dddd dd mmmm yyyy") & "<br />  <br />" _
                    & "Note: Attached is the instruction to create the time report." _
                    & "<br />  <br />"
   
    ws.Range("A1").AutoFilter
   
    On Error Resume Next
    Set objOutlook = GetObject(, "Outlook.Application")
    On Error GoTo 0
   
    If objOutlook Is Nothing Then
        Set objOutlook = CreateObject("Outlook.Application")
    End If
   
    Set objNameSpace = objOutlook.GetNamespace("MAPI")
   
    lngCounter = 0
    For i = LBound(arrEmpIdUnique) To UBound(arrEmpIdUnique)
        With ws.AutoFilter.Range
            .AutoFilter Field:=1, Criteria1:=arrEmpIdUnique(i)
            .AutoFilter Field:=lngColNumb, Criteria1:="<>" _
                            & Format(dteReport, "dddd dd mmmm yyyy")
           
            Set rng = .Resize(.Rows.Count, 1).SpecialCells(xlCellTypeVisible)
           
            If rng.Cells.Count = 1 Then GoTo lblNextLoop    'No data. Only column headers visible
           
            Set rng = .Resize(.Rows.Count, 7).SpecialCells(xlCellTypeVisible)
           
            Set rngTo = .Columns("H") _
                        .Offset(1, 0) _
                        .SpecialCells(xlCellTypeVisible)
                       
            Set rngCC = .Columns("I") _
                        .Offset(1, 0) _
                        .SpecialCells(xlCellTypeVisible)
                       
            .Columns("J").Offset(1, 0).Resize(.Rows.Count - 1, 1) _
                    .SpecialCells(xlCellTypeVisible) = Format(dteReport, "dddd dd mmmm yyyy")
        End With
       
        Set objMailItem = objOutlook.CreateItem(0)
       
        With objMailItem
            .To = rngTo.Cells(1, 1).Value
            .CC = rngCC.Cells(1, 1).Value
            .BCC = ""
            .Subject = "Data for Week ending " & Format(dteReport, "dddd dd mmmm yyyy")
            .HTMLBody = strBody & RangetoHTML(rng)
            .Attachments.Add (strFileToAttach)
            .Display
        End With
        DoEvents
       
        lngCounter = lngCounter + 1
        If lngCounter >= lngNumbToSend Then Exit For
       
lblNextLoop:

    Next i
           
    ws.ShowAllData
    ws.Columns(lngColNumb).AutoFit
    Set rngCel = Nothing
    Set rngToSend = Nothing
    Set rng = Nothing
    Set rngTo = Nothing
    Set rngCC = Nothing
    Set objMailItem = Nothing
    Set objNameSpace = Nothing
    Set objOutlook = Nothing

End Sub


Function RangetoHTML(rng As Range)
' From Ron de Bruin 28-Oct-2006
' Working in Office 2000-2010
' Reference: http://www.rondebruin.nl/mail/folder3/mail4.htm

    Dim fso As Object
    Dim ts As Object
    Dim TempFile As String
    Dim TempWB As Workbook
 
    TempFile = Environ$("temp") & "/" & Format(Now, "dd-mm-yy h-mm-ss") & ".htm"
 
    'Copy the range and create a new workbook to past the data in
    rng.Copy
    Set TempWB = Workbooks.Add(1)
    With TempWB.Sheets(1)
        .Cells(1).PasteSpecial Paste:=8
        .Cells(1).PasteSpecial xlPasteValues, , False, False
        .Cells(1).PasteSpecial xlPasteFormats, , False, False
        .Cells(1).Select
        .UsedRange.Columns.AutoFit
        .UsedRange.HorizontalAlignment = xlCenter
        Application.CutCopyMode = False
        On Error Resume Next
        .DrawingObjects.Visible = True
        .DrawingObjects.Delete
        On Error GoTo 0
    End With
 
    'Publish the sheet to a htm file
    With TempWB.PublishObjects.Add( _
         SourceType:=xlSourceRange, _
         Filename:=TempFile, _
         Sheet:=TempWB.Sheets(1).Name, _
         Source:=TempWB.Sheets(1).UsedRange.Address, _
         HtmlType:=xlHtmlStatic)
        .Publish (True)
    End With
 
    'Read all data from the htm file into RangetoHTML
    Set fso = CreateObject("Scripting.FileSystemObject")
    Set ts = fso.GetFile(TempFile).OpenAsTextStream(1, -2)
    RangetoHTML = ts.ReadAll
    ts.Close
    RangetoHTML = Replace(RangetoHTML, "align=center x:publishsource=", _
                          "align=left x:publishsource=")
 
    'Close TempWB
    TempWB.Close savechanges:=False
 
    'Delete the htm file we used in this function
    Kill TempFile
 
    Set ts = Nothing
    Set fso = Nothing
    Set TempWB = Nothing
End Function

A deep question about one of the fundamental features of Word.

All MS Word documents are "associated" with a template. Templates contain Word "Styles", some (not all!) Word configuration settings, shortcut key assignments, some generic text aka "Building Blocks", some macros and other miscellaneous "stuff".

When you create a simple new document it copies the information (specifically styles) from Normal.dotm and is still attached to Normal.DOTM.

Or when you create a new file you can select another "template" to work from. These templates are stored in various locations. They can be created by MS, or by you or your company.

"Global" templates are saved in a couple of specific locations defined in Word Options.  They give all documents access to shared information like macros.  Global templates can be used to deliver "Addons", tools that add user defined new features to Word.
