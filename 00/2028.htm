You have 5 dates that have and the sequence cannot be determined from their names.  I used Course_End_Date to calculate number of days since course attendance.


    EmpCourseStat --

SELECT DCRR20C_Consent.Learner_Employee_ID, DCRR20C_Consent.Course_Completion_Status, DCRR20C_Consent.Course_Plan_Date, DCRR20C_Consent.Course_Start_Date_Time, DCRR20C_Consent.Course_End_Date_Time, Max(DCRR20C_Consent.Course_End_Date_Time) AS MaxOfCourse_End_Date_Time, DCRR20C_Consent.Course_ID, IIf([Pass_Fail]="Pass",1,0) AS Pass, IIf([Pass_Fail]="Fail",1,0) AS Fail

FROM Courses INNER JOIN DCRR20C_Consent ON Courses.LAC = DCRR20C_Consent.Course_ID

GROUP BY DCRR20C_Consent.Learner_Employee_ID, DCRR20C_Consent.Course_Completion_Status, DCRR20C_Consent.Course_Plan_Date, DCRR20C_Consent.Course_Start_Date_Time, DCRR20C_Consent.Course_End_Date_Time, DCRR20C_Consent.Course_ID, IIf([Pass_Fail]="Pass",1,0), IIf([Pass_Fail]="Fail",1,0);



TRANSFORM Sum(EmpCourseStat.Pass) AS SumOfPass

SELECT EmpCourseStat.Learner_Employee_ID, EmpCourseStat.Course_ID, IIf([Pass]=0 And [Fail]=0,DateDiff("d",CVDate([MaxOfCourse_End_Date_Time]),Date()),"") AS Days_Since_Completion, Sum(EmpCourseStat.Pass) AS [Total Of Pass], Sum(EmpCourseStat.Fail) AS [Total Of Fail]

FROM EmpCourseStat

GROUP BY EmpCourseStat.Learner_Employee_ID, EmpCourseStat.Course_ID, IIf([Pass]=0 And [Fail]=0,DateDiff("d",CVDate([MaxOfCourse_End_Date_Time]),Date()),"")

PIVOT EmpCourseStat.Course_Plan_Date;
