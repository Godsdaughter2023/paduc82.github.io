Hi,

1) data on sheet named 'data' (change name)

2) in row 1 are headers

3) in column M: ignore zeros and empty cells

4) in column A: group data by names

(new sheet name: mm-dd-yyyy )

if so, try this...

{new Edit...]

in a regular module paste in the following:

Option Explicit

Const shName As String = "data"   '<< sheet name
Const CritC = "M"   '<< column M, ignore zeros and empty cells
Const SortC = "A"   '<<  names in column A

Const FormatD As String = "mm-dd-yyyy" '<< format date

Dim ws As Worksheet, sh As Worksheet
Dim r As Long, c As Long, i As Long

Sub ExportData_Group()

'step 1 and 2

'Sep 07, 2014
Set ws = Sheets(shName)
r = ws.Cells(Rows.Count, "A").End(xlUp).Row
c = ws.Cells(1, Columns.Count).End(xlToLeft).Column
Application.ScreenUpdating = False
Application.DisplayAlerts = False
For Each sh In Worksheets
If sh.Name = Format(Date, FormatD) Then sh.Delete
Next
Application.DisplayAlerts = True
Sheets.Add
ActiveSheet.Name = Format(Date, FormatD)

ws.Range("A1").Resize(r, c).Copy
Range("A1").PasteSpecial xlPasteValues
Range("A1").PasteSpecial xlPasteFormats

Application.CutCopyMode = False
With ActiveSheet.Sort
'.SortFields.Clear
.SortFields.Add Key:= Range(SortC & "2:" & SortC & c), SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
.SetRange Range(Cells(2, "A"), Cells(r, c))
.Header = xlNo
.MatchCase = False
.Orientation = xlTopToBottom
.SortMethod = xlPinYin
.Apply
End With
Range(CritC & ":" & CritC).Replace What:="0", Replacement:="", LookAt:=xlWhole
Range(CritC & ":" & CritC).SpecialCells(xlCellTypeBlanks).EntireRow.Delete
r = Cells(Rows.Count, "A").End(xlUp).Row
For i = r To 2 Step -1
If i - 1 = 1 Then
'nothing
Else
If Cells(i, SortC) <> Cells(i - 1, SortC) Then Rows(i).Resize(2).Insert
End If
Next
ActiveSheet.UsedRange.EntireColumn.AutoFit

Application.ScreenUpdating = True
End Sub

Sub HighlightRows()

'step 3

'Sep 07, 2014

Set ws = Sheets(shName)
r = ws.Cells(Rows.Count, "A").End(xlUp).Row
c = ws.Cells(1, Columns.Count).End(xlToLeft).Column
For i = 2 To r
ws.Range(ws.Cells(i, "A"), ws.Cells(i, c)).Interior.ColorIndex = xlNone
If ws.Cells(i, CritC).Value = "" Or ws.Cells(i, CritC).Value = 0 Then
'nothing
Else
ws.Range(ws.Cells(i, "A"), ws.Cells(i, c)).Interior.Color = RGB(255, 255, 0) 'yellow
End If
Next
End Sub

Sub Clear_InteriorColors()

'Sep 07, 2014

r = ws.Cells(Rows.Count, "A").End(xlUp).Row
c = ws.Cells(1, Columns.Count).End(xlToLeft).Column
ws.Range(ws.Cells(2, "A"), ws.Cells(r, c)).Interior.ColorIndex = xlNone
End Sub
