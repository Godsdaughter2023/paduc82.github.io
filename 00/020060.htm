tasos_k say: Hi,
try this code..
(before you run the code, make a copy)

Sub FromTxtToExcel()
Const myPath As String = "C:\Files Notepad\" 'change path name
Dim v As Variant
Dim r As Long
r = 2 'row 2
ActiveSheet.Cells.ClearContents
Open myPath & "Text File.txt" For Input As #1'change file name
Do While Not EOF(1)
Input #1, Line$
Rows(r).Columns(1) = Trim(Line$)
v = Split(Cells(r, "A"), " ")
For x = 0 To UBound(v)
Cells(r, 1 + x) = v(x)
Next x
r = r + 1
Loop
Close #1
End Sub
