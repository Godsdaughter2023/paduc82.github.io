' Arg 1 = Folder name in your inbox

Sub SaveEmailAttachmentsToFolder(OutlookFolderInInbox As String, _
                                 ExtString As String, DestFolder As String)
    Dim ns As NameSpace
    Dim reportFolder As MAPIFolder
    Dim inboxFolder As MAPIFolder
    Dim Item As Object
    Dim Atmt As Attachment
    Dim FileName As String
    Dim MyDocPath As String
    Dim I As Integer
    Dim wsh As Object
    Dim fs As Object
    Dim rootDir As MAPIFolder
    Dim Msg As Outlook.MailItem
   
    On Error GoTo ThisMacro_err

 

    Set ns = GetNamespace("MAPI")
    Set rootDir = ns.Folders.Item("Mailbox - My User")
    Set reportFolder = rootDir.Folders("Reports")
    Set finalFolder = rootDir.Folders("Lot Status")
    Set Msg = Application.CreateItem(olMailItem)
 
   
    I = 0
   ' Check subfolder for messages and exit of none found
    If reportFolder.Items.Count = 0 Then
        MsgBox "There are no messages in this folder : " & OutlookFolderInInbox, _
               vbInformation, "Nothing Found"
        Set reportFolder = Nothing
        Set ns = Nothing
        Exit Sub
    End If

   'Create DestFolder if DestFolder = ""
    If DestFolder = "" Then
        Set wsh = CreateObject("WScript.Shell")
        Set fs = CreateObject("Scripting.FileSystemObject")
        MyDocPath = wsh.SpecialFolders.Item("mydocuments")
        DestFolder = MyDocPath & "\"
        If Not fs.FolderExists(DestFolder) Then
            fs.CreateFolder DestFolder
        End If
    End If

    If Right(DestFolder, 1) <> "\" Then
        DestFolder = DestFolder & "\"
    End If

    ' Check each message for attachments and extensions
    For Each Item In reportFolder.Items
        For Each Atmt In Item.Attachments
            If LCase(Right(Atmt.FileName, Len(ExtString))) = LCase(ExtString) Then
                FileName = DestFolder & "LotStockStatus.xls"
                Atmt.SaveAsFile FileName
                I = I + 1
            End If
        Next Atmt
    Next Item
   
    ' Call this subroutine to edit the worksheet
    Delete_Rows

    ' Show this message when Finished
    If I > 0 Then
        MsgBox "You can find the files here : " _
             & DestFolder, vbInformation, "Finished!"
    Else
        MsgBox "No attached files in your mail.", vbInformation, "Finished!"
    End If
   
    ' Mark email as Read and move email back to Inbox
    Msg.UnRead = False
    Msg.Move finalFolder
      

    ' Clear memory
ThisMacro_exit:
    Set reportFolder = Nothing
    Set finalFolder = Nothing
    Set ns = Nothing
    Set fs = Nothing
    Set wsh = Nothing
    Exit Sub

    ' Error information
ThisMacro_err:
    MsgBox "An unexpected error has occurred." _
         & vbCrLf & "Please note and report the following information." _
         & vbCrLf & "Macro Name: SaveEmailAttachmentsToFolder" _
         & vbCrLf & "Error Number: " & Err.Number _
         & vbCrLf & "Error Description: " & Err.Description _
         , vbCritical, "Error!"
    Resume ThisMacro_exit

End Sub

 

Sub Delete_Rows()
' This macro deletes the first 5 rows of the LotStockStatus.xls spreadsheet.

 Dim XLApp As Object            ' Excel Application
 Dim XLwk As Excel.Workbook     ' Excel workbook
' DimXLwkdel as XLwk.add()       ' Excel delete function
 Dim numRow As Integer
 numRow = 1
 
 

 Set XLApp = New Excel.Application
 Set XLwk = XLApp.Workbooks.Open("C:\Users\my.user\My Documents\ExcelReports\LotStockStatus.xls")

 
 For numRow = 1 To 7
     XLApp.Rows(numRow).Delete
 Next numRow
   
 XLwk.Close SaveChanges:=True
  
 XLApp.Quit
 
 Set XLwk = Nothing
 Set XLApp = Nothing
 
End Sub

new-addresslist -name "test-resource" -recipientfilter {(((recipientdisplaytype -eq 'conferenceroommailbox') -or (recipientdisplaytype -eq 'syncedconferenceroommailbox')) -and (customattribute3 -eq "nullresource"))} -displayname "test room list"

new-addresslist -name "test" -recipientfilter {(recipienttype -eq 'usermailbox') -and (customattribute2 -eq "test")} –displayname “test address list”

new-globaladdresslist -name "gal-test " -recipientfilter {(customattribute2 -eq "test")}
6.create offline address book
new-offlineaddressbook -name "oab-test " -addresslists "gal_test"

new-addressbookpolicy -name "abp-test" -addresslists "test" -offlineaddressbook "\oab-test " -globaladdresslist "\gal-test" -roomlist "\ test-resource "

get-mailbox -resultsize unlimited | where {$_.customattribute2 -eq "test"} | set-mailbox -addressbookpolicy "abp-test"
9.hide test from address lists
get-mailbox -resultsize unlimited | where {$_.customattribute2 -eq "test"} | set-mailbox -hiddenfromaddresslistsenabled $true

john - project say:
Gridguy,
You should install the SP-3 update for Project 2007. You can get it at the URL I gave in my original response.

If by, "... get the data I need...", is the number of people per day, week, month, etc. then yes, a master file with a resource pool will give you the ability to view all resources across all projects. Note, the resource pool need not be separate from the master.

By "clicking on the resource column" in the Gantt Chart view I assume you mean the Resource Names field (column). There is no "resource" column in the Gantt Chart view unless you customized a field  or renamed the Resource Names field. Sorry if that sounds a little picky but using standard terminology makes helping you a whole lot easier.

If you entered all your resources, either group (generic) or individual in the Resource Sheet, then when you click on the Resource Names field in the Gantt Chart view, you will see it turn into a selection box such that you can resources for any given task. However, selecting resources this way only allows you to select a single resource. You can also type resources directly into the Resource Names field (e.g. plumbers[200%), meaning you have assigned 2 generic plumbers to that task.

However, a much better way to assign resources to your tasks is to open the Assign Resource window by clicking on the icon with the two little heads. The Assign Resources window will show all available resources from the Resource Sheet and the window will stay visible as you select tasks on the Gantt Chart. Click on a task and then select one or more resources from the Assign Resource window.
