bernie deitrick say:

You could use the selection change event - this will work as long as you are not depending on other hidden rows.

    1) Copy this code.
    2) Right-Click the sheet tab of interest.
    3) Select "View Code"
    4) Paste the code into the window that appears.
    5) Save the file as a macro-enabled .xlsm file.
    6) Make changes to the referenced ranges as needed - this will freze 1, 11, 21, 31 in succession, and finally have 41 be visible just below the other frozen rows - I hope you see the pattern in the code - just make sure that the higher rows are checked first....

Private Sub Worksheet_SelectionChange(ByVal Target As Range)
    Application.ScreenUpdating = False
    Application.EnableEvents = False
    
    ActiveWindow.FreezePanes = False
    Cells.EntireRow.Hidden = False
    
    If Target.Row > 40 Then
        Range("2:10,12:20,22:30,32:40").EntireRow.Hidden = True
        Range("A41").Select
        ActiveWindow.FreezePanes = True
        Target.Select
        GoTo FinishUp
    End If
    
    If Target.Row > 30 Then
        Range("2:10,12:20,22:30").EntireRow.Hidden = True
        Range("A31").Select
        ActiveWindow.FreezePanes = True
        Target.Select
        GoTo FinishUp
    End If

    If Target.Row > 20 Then
        Range("2:10,12:20").EntireRow.Hidden = True
        Range("A21").Select
        ActiveWindow.FreezePanes = True
        Target.Select
        GoTo FinishUp
    End If

    If Target.Row > 10 Then
        Range("2:10").EntireRow.Hidden = True
        Range("A11").Select
        ActiveWindow.FreezePanes = True
        Target.Select
        GoTo FinishUp
    End If
    
    Range("A2").Select
    ActiveWindow.FreezePanes = True
    Target.Select

FinishUp:
    Application.ScreenUpdating = True
    Application.EnableEvents = True

End Sub
