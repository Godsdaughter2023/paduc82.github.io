' Set up the non-reversed assignments in Normal.dotm
Sub resetArrowKeyAssignmentsinNormal()
CustomizationContext = Application.NormalTemplate

Application.KeyBindings.Add wdKeyCategoryCommand, "LineDown", 40

KeyBindings.Add KeyCode:=BuildKeyCode(40, wdKeyShift), KeyCategory:=wdKeyCategoryCommand, Command:="LineDownExtend"
KeyBindings.Add KeyCode:=BuildKeyCode(37, wdKeyShift), KeyCategory:=wdKeyCategoryCommand, Command:="CharLeftExtend"
KeyBindings.Add KeyCode:=BuildKeyCode(39, wdKeyShift), KeyCategory:=wdKeyCategoryCommand, Command:="CharRightExtend"
