andreas killer say:
You need a helper column for that, I assume that column N is empty. If not, increase the column used in the WITH instruction.

Andreas.


Option Explicit

Sub Test()
  Dim R As Range, Item, Icons()

  Icons = Array( _
    ActiveWorkbook.IconSets(xl3TrafficLights1).Item(1), _
    ActiveWorkbook.IconSets(xl3TrafficLights2).Item(3), _
    ActiveWorkbook.IconSets(xl3TrafficLights2).Item(1))

  'Screen off, runs faster
  Application.ScreenUpdating = False
 
  With Range("A1:N16")
    'Reset the autofilter
    .Parent.AutoFilterMode = False
    .AutoFilter
    'Filter each item
    For Each Item In Icons
      .AutoFilter Field:=11, Criteria1:=Item, Operator:=xlFilterIcon
      'Get the visible rows
      Set R = SpecialCells(.Parent.AutoFilter.Range, xlCellTypeVisible)
      'Write an "x" into the last column
      Intersect(.Columns(.Columns.Count), R.EntireRow) = "x"
    Next
    'Remove the filter settings
    .Parent.ShowAllData
    'Filter the "x"
    .AutoFilter Field:=.Columns.Count, Criteria1:="x"
    'clear the helper column
    .Columns(.Columns.Count).ClearContents
  End With
 
  Application.ScreenUpdating = True
End Sub

Private Function SpecialCells(ByVal R As Range, ByVal Typ As XlCellType, _
    Optional ByVal Value As XlSpecialCellsValue = &H17) As Range
  'Avoid the SpecialCells-BUG to return all cells from the current region
  On Error Resume Next
  Select Case Typ
    Case xlCellTypeConstants, xlCellTypeFormulas
      Set SpecialCells = Intersect(R, R.SpecialCells(Typ, Value))
    Case Else
      Set SpecialCells = Intersect(R, R.SpecialCells(Typ))
  End Select
End Function
