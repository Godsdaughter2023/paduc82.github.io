hansv mvp say:

If a user doesn't have Microsoft Project installed, the code won't work regardless of whether you use early binding or late binding.

Late binding can help if the developer and user have different versions of Project (for example if the developer has Project 2013 and the user has Project 2010), but not if the user doesn't have Project at all.

hansv mvp say:

You could use this (it does use late binding to avoid error messages):

Option Explicit
 
Public prjApp As Object
Public prjProject As Object
 
Sub GetDataFromMSProject()
    Application.DisplayAlerts = False
    Application.ScreenUpdating = False
    '
    On Error Resume Next
    Set prjApp = CreateObject("MSPRoject.Application")
    If prjApp Is Nothing Then
        MsgBox "Microsoft is not available on this computer!", vbExclamation
        GoTo ExitHandler
    End If
    On Error GoTo ErrHandler
    prjApp.Visible = True
    Set prjProject = prjApp.ActiveProject
    '
    ' Copy the tasks from the Project file
    '
    prjApp.ViewApply Name:="Resource &Usage"
    prjApp.SelectResourceField Row:=0, Column:="Text3"
    prjApp.SelectResourceColumn Column:="Name"
    prjApp.OutlineShowSubTasks
    prjApp.OutlineShowSubTasks
    prjApp.Sort Key1:="Text3"
    prjApp.SelectSheet
    prjApp.EditCopy
    '
    ' Paste the task data into Excel (your code here)
    Sheets("Everyone").Select
    Rows("3:10000").Delete
    Range("A3").Select
    ActiveSheet.Paste
    'Rows("3:3").Select
    'Selection.Delete Shift:=xlUp
    '
    ' Copy the hours from Project
    '
    Set prjProject = prjApp.ActiveProject
    prjApp.PaneNext
    prjApp.SelectTimescaleRange Row:=1, StartTime:=Now(), Width:=3, Height:=1047618
    prjApp.EditCopy
    '
    ' Paste the hours data into Excel (your code here)
    Range("I3").Select
    ActiveSheet.PasteSpecial Format:="Unicode Text", Link:=False, _
        DisplayAsIcon:=False, NoHTMLFormatting:=True
    '
    ' Close out the MS Project file and release the memory
    '
    'Set prjProject = Nothing
    'prjApp.Quit SaveChanges:=False
    'Set prjApp = Nothing
    'Columns("A:A").Select
    'Selection.Replace What:="Text1:", Replacement:="Project Name:", LookAt:= _
    '    xlPart, SearchOrder:=xlByRows, MatchCase:=False, SearchFormat:=False, _
    '    ReplaceFormat:=False
    '
    'Call Module1.CopyFormatting
    'Call Module1.SplitResourcesIntoTabs
    'Call Module7.ws
ExitHandler:
    Application.DisplayAlerts = True
    Application.ScreenUpdating = True
    Exit Sub
ErrHandler:
    MsgBox Err.Description, vbExclamation
End Sub

After replacing the original code with this version, select Tools > References... and clear the check box for Microsoft Project 12.0 Object Library. Then switch to Excel and save the workbook.
