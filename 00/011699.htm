willy denoyette say:

....because the default font is "MS Sans Serif"....

The default Windows font is not "MS Sans", it's Segoe UI. The default for Explorer for instance is Segoe UI 12pt.

Your program(s) tell you what the default Form fonts is.  The reason for this is that you use Windows .Forms during your application development (f.i VS) for which the default form font is set to "MS Sans", but you can set any font you like in your application.

Sub FileDialogMethod_01()
'Mar 19, 2014
Dim v As Variant
Dim wb As Workbook
Dim newPath As String
With Application.FileDialog(msoFileDialogFolderPicker)
.Title = "STEP1: select one Folder"
.AllowMultiSelect = False
If .Show = True Then
newPath = .SelectedItems(1) & "\"
Else
MsgBox "Cancel ALL"
Exit Sub
End If
End With
Dim wbName As String
With Application.FileDialog(msoFileDialogFilePicker)
.Title = "STEP2: press ctrl and select the files"
.Filters.Add "Excel Files", "*.xls*"
.AllowMultiSelect = True
If .Show = True Then
Application.ScreenUpdating = False
For Each v In .SelectedItems
Set wb = Workbooks.Open(v, False, False)
wbName = Replace(wb.Name, Left(wb.Name, 8), "")
Application.DisplayAlerts = False
wb.SaveAs Filename:=wb.Path & "\" & wbName
wb.Sheets("summary sheet").Activate   '<<< check sheet name
wbName = Replace(wbName, ".xlsm", ".pdf")
ActiveSheet.ExportAsFixedFormat Type:=xlTypePDF, _
Filename:=newPath & wbName, OpenAfterPublish:=False
Application.DisplayAlerts = True
wb.Close False
wbName = ""
Next
Application.ScreenUpdating = True
MsgBox "done"
Else
MsgBox "cancel, try again"
End If
End With
End Sub
