3. Hold CTRL key and click on Excel icon

Sub zz_AddToReg()
    Dim rngSelected As Range
    Dim rngSource As Range
    Dim rngDestin As Range
    Dim lngLastRow As Long
   
    'Test if selection is column B and only one cell selected
    If Selection.Column = 2 And Selection.Cells.Count = 1 Then
        Set rngSelected = Selection
        Set rngDestin = Cells(Rows.Count, "I").End(xlUp).Offset(1, 0)   'Next blank cell in column "I"
       
        Set rngSource = Range(rngSelected, rngSelected.Offset(0, 4))  'From Selected cell to 4 additional cells to right
       
        With rngSource
            'Resize destination cell to a range to match same number of cells as Source range
            'and set destination values equal to Source values (Equivalent to Copy -> PasteSpecial -> Values)
            rngDestin.Resize(.Rows.Count, .Columns.Count).Value = rngSource.Value
        End With
       
        lngLastRow = Cells(Rows.Count, "I").End(xlUp).Row
        Range(Cells(9, "I"), Cells(lngLastRow, "M")) _
            .RemoveDuplicates Columns:=Array(1, 2, 3, 4, 5), Header:=xlNo
           
        If lngLastRow <> Cells(Rows.Count, "I").End(xlUp).Row Then
            MsgBox "Duplicate."
        End If
    End If
 End Sub
