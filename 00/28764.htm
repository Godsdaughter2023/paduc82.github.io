Sub Protectpassword()
Dim mypath As String
Dim rname As String
Dim s As String
Dim rname1 As String
Dim x As Workbook
Dim r As String
Dim fd As String
Dim fd1 As String
Dim fd2 As String
Dim mypath1 As String
Dim T As String

Call createFolder(ThisWorkbook.path)
Set x = ThisWorkbook
s = Range("C4").Value
mypath = ThisWorkbook.path & "\" & s & "\"
mypath1 = ThisWorkbook.path & "\" & "Protected files" & "\"
rname = Dir(mypath & "*.xlsx")
    
    If rname = "" Then
        rname1 = Dir(mypath & "*.xls")
            
            
            Do While rname1 <> ""
            Application.DisplayAlerts = False
            Application.StatusBar = "Now working on " & rname1
            Workbooks.Open Filename:=mypath & rname1
            fd = Len(rname1) - 4
            fd1 = Left(rname1, fd)
            T = Format(Now(), "MMM-YYYY")
            fd2 = mypath1 & fd1 & "_" & T & ".xlsx"
            Workbooks(rname1).SaveAs Filename:=fd2, Password:="Raghu12", WriteRespassword:="Raghu12", ReadOnlyRecommended:=False, CreateBackup:=False
            ActiveWorkbook.Save
            ActiveWorkbook.Close

            
            rname1 = Dir()
            Loop

        
    Else
            Do While rname <> ""
            Application.DisplayAlerts = False
            Application.StatusBar = "Now working on " & rname
            Workbooks.Open Filename:=mypath & rname
            fd = Len(rname) - 5
            fd1 = Left(rname, fd)
            T = Format(Now(), "MMM-YYYY")
            fd2 = mypath1 & fd1 & "_" & T & ".xlsx"
            Workbooks(rname).SaveAs Filename:=fd2, Password:="Raghu12", WriteRespassword:="Raghu12", ReadOnlyRecommended:=False, CreateBackup:=False
            ActiveWorkbook.Save
            ActiveWorkbook.Close
            
            rname = Dir()
            Loop
    End If
  
MsgBox ("Files has been protected with password")
End Sub
