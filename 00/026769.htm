bernie deitrick, excel mvp 2000-2010 say:

Try this one:

Sub MakeTableFromBlocks2()

    Dim myR As Range
    Dim myA As Range
    Dim i As Integer
    Dim j As Integer
    Dim strTemp As String

    Set myR = Columns("A:A").SpecialCells(xlCellTypeConstants, 23)
    For Each myA In myR.Areas
        strTemp = ""
        Set myA = myA.CurrentRegion
        For i = 1 To myA.Rows.Count
            For j = 1 To myA.Columns.Count
                strTemp = strTemp & " " & myA.Cells(i, j).Value
            Next j
        Next i
        myA.Cells.Clear
        myA.Cells(1, 1).Value = Application.Trim(strTemp)
    Next myA

    Columns("A:A").SpecialCells(xlCellTypeBlanks).EntireRow.Delete
End Sub
