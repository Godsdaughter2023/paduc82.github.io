gary's student say: Select the cells you need to convert and run this small macro:

Sub TimeConverter()
Dim d As Date, dd As Double
Dim r As Range
For Each r In Selection
    d = r.Value
    dd = CDbl(d) * 24
    r.Clear
    r.Value = dd
Next
End Sub
