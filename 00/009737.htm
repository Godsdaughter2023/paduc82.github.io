shyam pillai say:

Hi,

Try this:

 

Sub Test()
Dim oCL As CustomLayout

'Pass the design and the name of the custom layout to get the reference
Set oCL = GetCustomLayout(ActivePresentation.Designs(1), "Title Slide")
If Not oCL Is Nothing Then
    ' Apply the layout to multiple slides
    With ActivePresentation.Slides
        .Range(Array(2, 3, 4)).CustomLayout = oCL
    End With
Else
    Debug.Print "Did not find specified custom layout in the design."
End If
End Sub

Function GetCustomLayout(Design As Design, LayoutName As String) As CustomLayout
Dim i As Integer

For i = 1 To Design.SlideMaster.CustomLayouts.Count
    With Design.SlideMaster.CustomLayouts
        If .Item(i).Name = LayoutName Then
            Set GetCustomLayout = .Item(i)
        End If
    End With
Next

End Function
