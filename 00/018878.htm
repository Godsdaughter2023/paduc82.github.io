jay freedman say:

There isn't any direct route to the Alt Text entry in the user interface, and the venerable (but sometimes unreliable) SendKeys function doesn't work in the Format XXX panes because the accelerators are messed up.

The macro gets a bit complicated because of the various kinds of objects that you want to handle from the same QAT button or shortcut. Try this and let me know whether it works for you.

Sub AddAltText()
    Dim shp As Shape
    Dim ilshp As InlineShape
    Dim tbl As Table
    Dim strAltTxt As String

    strAltTxt = InputBox( _

        Prompt:="Enter alt text for the selected object", _
        Title:="AltText", Default:=" ")
    If StrPtr(strAltTxt) = 0 Then Exit Sub   ' canceled

    Select Case Selection.Type
        Case wdSelectionShape
            Set shp = Selection.ShapeRange(1)
            shp.AlternativeText = strAltTxt
            shp.Title = "Alt Text"
        Case wdSelectionInlineShape
            ' When a Shape is set to In Line text wrapping,
            ' it's still a member of the ShapeRange and not the
            ' InlineShapes collection.
            On Error Resume Next
            Set ilshp = Selection.InlineShapes(1)
            If Err.Number = 0 Then
                ilshp.AlternativeText = strAltTxt
                ilshp.Title = "Alt Text"
            Else
                Err.Clear
                Set shp = Selection.ShapeRange(1)
                shp.AlternativeText = strAltTxt
                shp.Title = "Alt Text"
            End If
        Case Else
            If Selection.Information(wdWithInTable) Then
                ' If the Selection is in a nested table,
                ' find the outermost table and set the
                ' alt text there.
                ' If you want the alt text to attach to
                ' the nested table instead, delete the
                ' statements with the comment "<=".
                Dim oRg As Range                                 ' <=
                Dim nNestLev As Integer                          ' <=
                Set tbl = Selection.Tables(1)
                nNestLev = tbl.NestingLevel                      ' <=
                Set oRg = tbl.Range.Cells(1).Range.Characters(1) ' <=
                While nNestLev > 1                               ' <=
                    oRg.MoveStart wdCharacter, -1                ' <=
                    nNestLev = oRg.Tables(1).NestingLevel        ' <=
                Wend                                             ' <=
                Set tbl = oRg.Tables(1)                          ' <=
                tbl.Descr = strAltTxt
                tbl.Title = "Alt Text"
            ElseIf Selection.Type = wdSelectionIP Or _

                    Selection.Type = wdSelectionNormal Then
                MsgBox "Select a picture, shape, text box, or table and try again." _

                    & vbCr & "For a shape, click on the edge instead of the middle."
            End If
    End Select
End Sub

If you want to be able to set both the Title and the Alt Text to user-entered text, you'll need either two InputBox statements or a userform to gather the two strings.

Sub SplitEachWordInACell()

'Dim WB As Workbook
Dim ws As Worksheet
Dim wdApp As Object, wdDoc As Object
Dim myPath As String, s As String
Dim p As Long, r As Long, t As Long

Dim v As Variant
Set ws = ThisWorkbook.ActiveSheet

Application.ScreenUpdating = False
r = ws.Cells(Rows.Count, "A").End(xlUp).Row
If r < 3 Then r = 3
ws.Range("A3:A" & r).ClearContents

myPath = "C:\Folder1\Folder2\test01.doc" '<<< change path and doc. name as needed
Set wdApp = CreateObject("Word.Application")

With wdDoc
For p = 1 To .paragraphs.Count

ws.Cells(t, "A") = s
With ws.Cells(t, "A")
.Replace What:=Chr(9), Replacement:="@#$", LookAt:=xlPart
.Replace What:=Chr(13), Replacement:="@#$", LookAt:=xlPart
.Replace What:=Chr(32), Replacement:="@#$", LookAt:=xlPart

v = Split(ws.Cells(t, "A"), "@#$")

Next

Next p
.Close
