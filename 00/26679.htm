Cable TV ISPs commonly use the 10.x.x.x private network addressing scheme for routers and other devices kept behind their firewalls or other network edge devices.  Though these addresses shouldn't be able to leave the local network and so be seen by the remote servers or other authentication devices, in some cases this information can "leak" and cause such issues.

If the addresses are such private network numbers, then I wouldn't be worried at all.

If they're something else that appears to provide an actual Internet assigned IP address, then I'd first try to determine where in the world this number is supposedly assigned, since that might provide clues as to whether my own ISP controls these as well.  Or in the case they don't, the owner of that IP address range might provide a clue as to why this address is being seen.

Then you do not have sheets that contain only shapes and you never use more than 256 columns. The AddIn has BUGs.

Add the code below to a regular module in your personal workbook.

EDIT: The code could work better if a sheet has a password set. Sorry for any inconvenience.

Andreas.


Option Explicit

Sub ResetLastCell_ActiveWorkbook()
  Dim WS As Worksheet
  For Each WS In ActiveWorkbook.Worksheets
    If ResetLastCell(WS) = vbAbort Then Exit For
  Next
End Sub

Sub ResetLastCell_ActiveSheet()
  Dim Before As Range, After As Range
  Select Case ResetLastCell(ActiveSheet, Before, After)
    Case vbYes
      MsgBox "Last cell is now " & After.Address(0, 0) & _
        " (was " & Before.Address(0, 0) & ")", , "ResetLastCell"
    Case vbNo
      MsgBox "Last cell is unchanged " & Before.Address(0, 0), , _
        "ResetLastCell"
  End Select
End Sub

Function ResetLastCell(ByVal ThisSheet As Worksheet, _
    Optional ByRef LastCellBefore As Range, _
    Optional ByRef LastCellAfter As Range) As VbMsgBoxResult
  'Deletes unused formatting and resets the last cell.
  'Returns values as follows:
  '  vbYes = last cell is changed
  '  vbNo  = last cell is unchanged
  '  If the sheet is protected:
  '  vbAbort = user want to abort the whole process
  '  vbIgnore = user want to skip this sheet
  Dim R As Range, C As Range
  Dim WS As Object 'Worksheet
  Dim Sh As Shape
  Dim LastRow As Long, LastCol As Long
  Dim ThisRow As Long, ThisCol As Long
  Dim i As Long
  Dim PS(0 To 2) As Boolean
  Dim Pwd As Variant

  ResetLastCell = vbNo
  Set WS = ThisSheet
  If WS Is Nothing Then Exit Function
  On Error Resume Next
  'Get the protection state
  PS(0) = WS.ProtectDrawingObjects
  PS(1) = WS.ProtectContents
  PS(2) = WS.ProtectScenarios
  'Try to unprotect with empty password
  WS.Unprotect ""
  If Err Then
AskPwd:
    Err.Clear
    Pwd = Application.InputBox("Sheet '" & WS.Name & "' is protected." & _
      vbNewLine & "Password:", "ResetLastCell")
    If VarType(Pwd) = vbBoolean Then
      ResetLastCell = vbAbort
      Exit Function
    End If
    WS.Unprotect Pwd
    If Err Then
      ResetLastCell = MsgBox("Wrong Password.", vbAbortRetryIgnore, _
        "ResetLastCell")
      If ResetLastCell = vbRetry Then
        GoTo AskPwd
      Else
        Exit Function
      End If
    End If
  End If
  'Try to reset the last cell
  WS.UsedRange
  'Get the last cell
  Set LastCellBefore = WS.UsedRange.SpecialCells(xlCellTypeLastCell)
  ThisRow = LastCellBefore.Row
  ThisCol = LastCellBefore.Column
  'Find the really last used cell
  Set C = Union(WS.UsedRange.SpecialCells(xlCellTypeConstants), _
    WS.UsedRange.SpecialCells(xlCellTypeFormulas))
  If C Is Nothing Then Set C = WS.UsedRange.SpecialCells(xlCellTypeConstants)
  If C Is Nothing Then Set C = WS.UsedRange.SpecialCells(xlCellTypeFormulas)
  If Not C Is Nothing Then
    For Each R In C.Areas
      i = R.Row + R.Rows.Count - 1
      If i > LastRow Then LastRow = i
      i = R.Column + R.Columns.Count - 1
      If i > LastCol Then LastCol = i
    Next
  End If
  'Get the last cell behind the shapes
  For Each Sh In WS.Shapes
    If Sh.BottomRightCell.Row > LastRow Then
      LastRow = Sh.BottomRightCell.Row
    End If
    If Sh.BottomRightCell.Column > LastCol Then
      LastCol = Sh.BottomRightCell.Column
    End If
  Next
  If LastRow = 0 And ThisRow = 1 And ThisCol = 1 Then
    'Sheet is empty
    Set LastCellAfter = LastCellBefore
    'Test if A1 has a formatting
    If LastCellBefore.RowHeight <> WS.StandardHeight Then GoTo DeleteIt
    If LastCellBefore.ColumnWidth <> WS.StandardWidth Then GoTo DeleteIt
    If LastCellBefore.Font.ColorIndex <> xlColorIndexAutomatic Then GoTo DeleteIt
    If LastCellBefore.Interior.ColorIndex <> xlColorIndexNone Then GoTo DeleteIt
    For i = 1 To 12
      If LastCellBefore.Borders(i).ColorIndex <> xlColorIndexNone Then GoTo DeleteIt
    Next
  Else
DeleteIt:
    'Delete the overhang if necessary
    If ThisRow > LastRow Then
      Set C = WS.Rows(LastRow + 1).Resize(ThisRow - LastRow)
      Application.EnableEvents = False
      C.Delete
      Application.EnableEvents = True
    End If
    If ThisCol > LastCol Then
      Set C = WS.Columns(LastCol + 1).Resize(, ThisCol - LastCol)
      Application.EnableEvents = False
      C.Delete
      Application.EnableEvents = True
    End If
    'Reset the last cell
    WS.UsedRange
    Set LastCellAfter = WS.Cells.SpecialCells(xlCellTypeLastCell)
    Set LastCellBefore = WS.Cells(ThisRow, ThisCol)
    If (ThisRow > LastRow) Or (ThisCol > LastCol) Or _
        (LastCellAfter.Row < ThisRow) Or (LastCellAfter.Column < ThisCol) Then
      ResetLastCell = vbYes
    Else
      ResetLastCell = vbNo
    End If
  End If
  WS.Protect Pwd, PS(0), PS(1), PS(2)
  Err.Clear
End Function
