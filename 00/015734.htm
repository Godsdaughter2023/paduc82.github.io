Here is a custom VBA function (with a helper function) that you may use:

 

 

Function FindString(TextString As String, FindIn As Range) As String
    Dim rngCell As Range
    Dim strAddress As String
    Dim lngOffset As Long
    With FindIn
        Set rngCell = .Find(What:="Mutual*", LookAt:=xlWhole)
        If Not rngCell Is Nothing Then
            lngOffset = CheckString(TextString, rngCell)
            If lngOffset > 0 Then
                FindString = rngCell.Offset(lngOffset).Value
                Exit Function
            End If
            strAddress = rngCell.Address
            Do
                Set rngCell = .FindNext(After:=rngCell)
                If rngCell Is Nothing Then Exit Do
                If rngCell.Address = strAddress Then Exit Do
                lngOffset = CheckString(TextString, rngCell)
                If lngOffset > 0 Then
                    FindString = rngCell.Offset(lngOffset).Value
                    Exit Do
                End If
            Loop
        End If
    End With
End Function
 

Function CheckString(TextString As String, Cell As Range) As Long
    Dim strFound As String
    Dim i As Long
    strFound = Cell.Value
    For i = 1 To 5
        If strFound = TextString Then
            CheckString = i
            Exit For
        End If
        strFound = Application.Trim(strFound & " " & Cell.Offset(i).Value)
    Next i
End Function

mike middleton say:

You can get the "There is a problem ..." error message when the system List Separator doesn't match what you're using in the formula.

If you're running Windows 10, choose Start > Settings > Time & Language > Region & Language > Additional date, time, & regional settings > Change date, time, or number formats > Formats > Additional settings... > Numbers > Reset (to restore system default settings) > Yes to confirm Remove all customizations.

And, there are shorter paths to get to the Additional settings... in Windows 10, but the long path shown above works for me.
