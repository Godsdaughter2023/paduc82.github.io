Sub find_activeX()

Dim objX As Shape
For Each sht In Sheets
    With sht
        On Error Resume Next
        For Each objX In .Shapes

           Debug.Print sht.Name & ": " & objX.Name & " | " & objX.Left & " | " & objX.Type _
           & " | " & IIf(objX.Type = 8, "Normal Form Control", IIf(objX.Type = 12, "ActiveX Control", "Other"))

        Next
    End With
Next

End Sub
