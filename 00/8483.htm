Re-working slightly Bernard's code:

Sub makedate()
  Dim myCell, myYear, myMonth, myDay, j, LastRow
  Worksheets("Sheet1").Activate
  LastRow = Cells(Rows.Count, "A").End(xlUp).Row
  For j = 1 To LastRow
    Set myCell = Cells(j, "A")
    If myCell.NumberFormat = "General" And Len(myCell.Value) = 8 And _
     IsNumeric(myCell.Value) Then
      myYear = Left(myCell, 4)
      myMonth = Mid(myCell, 5, 2)
      myDay = Right(myCell, 2)
      mydate = DateSerial(myYear, myMonth, myDay)
      Cells(j, "A") = mydate
    End If
  Next j
End Sub

Re-working a little more my original code:

Sub FormatDates()
  Dim myRange As Range, c As Range
  Set myRange = Workbooks("YourWorkbookName").Sheets("YourSheetName") _
   .Range("A:A").End(xlUp)
  For Each c In myRange
    If IsDate(c.Value) Then
      c.NumberFormat = "dd/mm/yyyy"
    ElseIf IsNumeric(c.Value) Then
      If Int(Log(c.Value) / Log(10)) = 7 Then
        c.Value = DateSerial(Left(c.Value, 4), Mid(c.Value, 5, 2), Right(int(c.Value), 2))
        c.NumberFormat = "dd/mm/yyyy"
      End If
    End If
  Next c
End Sub
