Option Explicit

'Scheduling Priorities:
'https://msdn.microsoft.com/en-us/library/windows/desktop/ms685100%28v=vs.85%29.aspx
'Process Security and Access Rights
'https://msdn.microsoft.com/en-us/library/windows/desktop/ms684880%28v=vs.85%29.aspx

#If Win64 Then
Private Declare PtrSafe Function GetCurrentProcess Lib "kernel32" () As LongPtr
Private Declare PtrSafe Function GetCurrentThread Lib "kernel32" () As LongPtr
Private Declare PtrSafe Function SetPriorityClass Lib "kernel32" ( _
    ByVal hProcess As LongPtr, ByVal dwPriorityClass As Long) As Long
Private Declare PtrSafe Function GetPriorityClass Lib "kernel32" ( _
    ByVal hProcess As LongPtr) As Long
Private Declare PtrSafe Function SetThreadPriority Lib "kernel32" ( _
    ByVal hThread As LongPtr, ByVal nPriority As Long) As Long
Private Declare PtrSafe Function GetThreadPriority Lib "kernel32" ( _
    ByVal hThread As LongPtr) As Long
#Else
Private Declare Function GetCurrentProcess Lib "kernel32" () As Long
Private Declare Function GetCurrentThread Lib "kernel32" () As Long
Private Declare Function SetPriorityClass Lib "kernel32" ( _
    ByVal hProcess As Long, ByVal dwPriorityClass As Long) As Long
Private Declare Function GetPriorityClass Lib "kernel32" (ByVal hProcess As Long) As Long
Private Declare Function SetThreadPriority Lib "kernel32" ( _
    ByVal hThread As Long, ByVal nPriority As Long) As Long
Private Declare Function GetThreadPriority Lib "kernel32" (ByVal hThread As Long) As Long
#End If

Public Enum PRIORITY_CLASS
  REALTIME_PRIORITY_CLASS = &H100&
  HIGH_PRIORITY_CLASS = &H80&
  ABOVE_NORMAL_PRIORITY_CLASS = &H8000&
  NORMAL_PRIORITY_CLASS = &H20&
  BELOW_NORMAL_PRIORITY_CLASS = &H4000&
  IDLE_PRIORITY_CLASS = &H40&
End Enum

Private Const THREAD_BASE_PRIORITY_LOWRT As Long = 15&
Private Const THREAD_BASE_PRIORITY_MAX As Long = 2&
Private Const THREAD_BASE_PRIORITY_MIN As Long = -2&
Private Const THREAD_BASE_PRIORITY_IDLE As Long = -15&
Private Const THREAD_PRIORITY_ERROR_RETURN = &H7FFFFFFF

Public Enum THREAD_PRIORITY
  THREAD_PRIORITY_TIME_CRITICAL = THREAD_BASE_PRIORITY_LOWRT
  THREAD_PRIORITY_HIGHEST = THREAD_BASE_PRIORITY_MAX
  THREAD_PRIORITY_NORMAL = 0&
  THREAD_PRIORITY_LOWEST = THREAD_BASE_PRIORITY_MIN
  THREAD_PRIORITY_IDLE = THREAD_BASE_PRIORITY_IDLE
  THREAD_PRIORITY_ABOVE_NORMAL = (THREAD_PRIORITY_HIGHEST - 1&)
  THREAD_PRIORITY_BELOW_NORMAL = (THREAD_PRIORITY_LOWEST + 1&)
End Enum

Private Sub Example_Priority()
  Dim Class As Long, Thread As Long
  Dim sClass As String, sThread As String

  'Show current settings
  GoSub GetIt

  'Set a higher priority
  If Not SetPriority(ABOVE_NORMAL_PRIORITY_CLASS, THREAD_PRIORITY_HIGHEST) Then
    MsgBox "You don't have access rights for this"
    Exit Sub
  End If
  GoSub GetIt

  'Back to normal
  SetPriority NORMAL_PRIORITY_CLASS

  Exit Sub
GetIt:
  If Not GetPriority(Class, Thread) Then
    MsgBox "You don't have access rights for this"
    Exit Sub
  End If
  Select Case Class
    Case REALTIME_PRIORITY_CLASS:     sClass = "Realtime"
    Case HIGH_PRIORITY_CLASS:         sClass = "High"
    Case ABOVE_NORMAL_PRIORITY_CLASS: sClass = "Above Normal"
    Case NORMAL_PRIORITY_CLASS:       sClass = "Normal"
    Case BELOW_NORMAL_PRIORITY_CLASS: sClass = "Below Normal"
    Case IDLE_PRIORITY_CLASS:         sClass = "Idle"
  End Select
  Select Case Thread
    Case THREAD_PRIORITY_TIME_CRITICAL: sThread = "Time Critical"
    Case THREAD_PRIORITY_HIGHEST:       sThread = "Highest"
    Case THREAD_PRIORITY_ABOVE_NORMAL:  sThread = "Above Normal"
    Case THREAD_PRIORITY_NORMAL:        sThread = "Normal"
    Case THREAD_PRIORITY_BELOW_NORMAL:  sThread = "Below Normal"
    Case THREAD_PRIORITY_LOWEST:        sThread = "Lowest"
    Case THREAD_PRIORITY_IDLE:          sThread = "Idle"
  End Select
  Debug.Print "Class", sClass, "Thread", sThread
  Return
End Sub

Function GetPriority(ByRef Class As Long, Optional ByRef Thread As Long, _
    Optional ByVal hThread, Optional ByVal hProcess) As Boolean
  If IsMissing(hThread) Then hThread = GetCurrentThread
  If IsMissing(hProcess) Then hProcess = GetCurrentProcess
  Class = GetPriorityClass(hProcess)
  Thread = GetThreadPriority(hThread)
  GetPriority = Thread <> THREAD_PRIORITY_ERROR_RETURN And Class <> 0
End Function

Function SetPriority(ByVal Class As PRIORITY_CLASS, _
    Optional ByVal Thread As THREAD_PRIORITY, Optional ByVal hThread, _
    Optional ByVal hProcess) As Boolean
  'Note: The value in the Windows Task Manager is from the process
  Dim t As Long, c As Long
  If IsMissing(hThread) Then hThread = GetCurrentThread
  If IsMissing(hProcess) Then hProcess = GetCurrentProcess
  c = SetPriorityClass(hProcess, Class)
  t = SetThreadPriority(hThread, Thread)
  SetPriority = t <> 0 And c <> 0
End Function

mel77084 say:

When the Windows Update scans for updates and finds one to download and install, it shows a description of it while it's downloading and installing. That's where you would find what it's for. In the case of failure to install, the error message gives a code that can possibly give a clue as to why it failed, but will not tell what the update was intended for.

It's like if you want to catch the bus to go downtown to do something such as make a clothing purchase; however, when it comes, the driver tells you that the bus is already loaded to capacity, so you can't get on, and need to wait for the next one. In that instance, the error code was that you couldn't board because the bus was full; but that says nothing about what your

Often, the failure is because the auto-update is already in the process of installing an update, so there's nothing to be concerned about.

Sometimes, it's a driver for audio or video that fails, but if both audio and video seem to be working well, don't get in a hurry to worry about it. You likely didn't need it, and the error was that it being offered.

Just go with the flow, and don't be too surprised or put off by things that don't seem quite right.
