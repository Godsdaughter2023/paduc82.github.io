If you are going to edit the message text, you need a different approach to creating the message, using the Outlook e-mail editor e.g. as follows. I have put the areas to be replaced by your variables in bold text.

Dim OutApp As Object
Dim OutMail As Object
Dim olInsp As Object
Dim wdDoc As Object
Dim oRng As Object
Dim strBody As String
Dim strLink As String
Dim strLinkText As String

    strLink = "file://" & ActiveWorkbook.FullName
    strLinkText = ">Link to the file"
    strBody = "Hi " & "This is the text box reference" & vbCr & vbCr _
    & "I have input changes in Super Team League Updates." & vbCr

    On Error Resume Next
    'Get Outlook if it's running
    Set OutApp = GetObject(, "Outlook.Application")

    'Outlook wasn't running, start it from code
    If Err <> 0 Then
        Set OutApp = CreateObject("Outlook.Application")
    End If
    On Error Goto 0
    'Create a new mailitem
    Set OutMail = OutApp.CreateItem(0)
    With OutMail
        .To = "*** Email address is removed for privacy ***"
        .Subject = "This is the subject"
        .ReadReceiptRequested = True
        .BodyFormat = 2
        Set olInsp = .GetInspector
        Set wdDoc = olInsp.WordEditor
        Set oRng = wdDoc.Range
        oRng.collapse 1
        oRng.Text = strBody
        oRng.collapse 0
        wdDoc.Hyperlinks.Add Anchor:=oRng, _
                                     Address:=strLink, _
                                     SubAddress:="", _
                                     ScreenTip:="", _
                                     TextToDisplay:=strLinkText
        oRng.End = oRng.Paragraphs(1).Range.End
        oRng.collapse 0
        oRng.Text = vbCr
        oRng.collapse 0
        oRng.Text = "Thanks" & vbCr
        .Display 'This line must be retained
        '.Send 'This line is optional (after the process has been validated)
    End With

    'Clean up
    Set OutMail = Nothing
    Set OutApp = Nothing
    Set olInsp = Nothing
    Set wdDoc = Nothing
    Set oRng = Nothing
