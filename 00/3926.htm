andreas killer say:
There is no need for any calculation, get the position, size and rotation from the existing picture.

The edge point of a shape is always the same, regardless which rotation it has.
If you want to rotate a shape around a corner, you must calculate the difference to the real corner point before and after the rotation and move it over this distance.

Andreas.


Option Explicit

Public Enum shCornerPoint
  shTopLeft
  shTopRight
  shBottomLeft
  shBottomRight
End Enum

Sub InsertPictureOverExistingShape()
  Dim OldShape As Shape, NewShape As Shape
 
  'On this slide
  With ActivePresentation.Slides(1)
    'Refer to the existing picture
    Set OldShape = .Shapes(1)
    'Insert the new one with the same size and position
    Set NewShape = .Shapes.AddPicture("C:\test.JPG", False, True, _
      OldShape.Left, OldShape.Top, OldShape.Width, OldShape.Height)
    'Same orientation
    NewShape.Rotation = OldShape.Rotation
    'Rotate it
    RotateShape NewShape, -90, shTopLeft
  End With
End Sub

Sub RotateShape(ByVal S As Shape, ByVal Degree As Double, _
    Optional ByVal Corner As shCornerPoint = shTopLeft)
  'Rotate a shape around a corner point
  Dim R As Double, D As Double, Rot As Double
  Dim oX As Double, oY As Double
  Dim nX As Double, nY As Double
  With S
    'Distance corner to center point
    R = Sqr((.Width / 2) ^ 2 + (.Height / 2) ^ 2)
    'Angle to the corner point in radians
    Select Case Corner
      Case shTopLeft, shBottomRight
        D = Atn((.Height / 2) / (.Width / 2))
      Case shBottomLeft, shTopRight
        D = Atn((.Width / 2) / (.Height / 2))
    End Select
    'Get the shape rotation in radians
    Rot = .Rotation * Atn(1) * 4 / 180
    'Calculate the distance to the real corner point
    oX = Cos(Rot + D) * R
    oY = Sin(Rot + D) * R
    'Rotate the shape
    .Rotation = .Rotation + Degree
    'Calculate the distance again
    Rot = .Rotation * Atn(1) * 4 / 180
    nX = Cos(Rot + D) * R
    nY = Sin(Rot + D) * R
    'Move it to the new location
    Select Case Corner
      Case shBottomLeft
        .Left = .Left - oY + nY
        .Top = .Top - nX + oX
      Case shBottomRight
        .Left = .Left + oX - nX
        .Top = .Top + oY - nY
      Case shTopLeft
        .Left = .Left - oX + nX
        .Top = .Top - oY + nY
      Case shTopRight
        .Left = .Left + oY - nY
        .Top = .Top + nX - oX
    End Select
  End With
End Sub
