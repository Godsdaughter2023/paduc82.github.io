Sub Export()
'Set reference to Excel!
    Dim xlsApp As Excel.Application
    Dim xlsWb  As Workbook
    Dim xlsWs  As Worksheet
    Dim L As Long
    ' Dim Chart Data
    Dim PPTChart As PowerPoint.Chart
    Dim PPTChartSheet As Worksheet
    Set xlsApp = CreateObject("Excel.Application")
   
    'Change the path!
    Set xlsWb = xlsApp.Workbooks.Open("C:\Users\John\Desktop\data.xlsx")

    'Point at your chart
    Set PPTChart = ActivePresentation.Slides(1).Shapes("Chart 3").Chart
    'include this to make sure data is open
    PPTChart.ChartData.Activate

    Set PPTChartSheet = PPTChart.ChartData.Workbook.Sheets(1)

    With PPTChartSheet
