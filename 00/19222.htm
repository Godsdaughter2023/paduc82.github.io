amit tandon say:

Refer below image:

Enter below formula as an array formula (CTRL+SHIFT+ENTER) in cell G2 & copy down:

=IFERROR(INDEX($B$2:$B$100, SMALL(IF(ISNUMBER(SEARCH($E$1,$B$2:$B$100)),ROW($B$2:$B$100)-ROW($B$2)+1), ROWS($B$2:$B2))),"")

Enter below formula in cell F2 & copy to H2 & down the column:

=IFERROR(INDEX(A$2:A$100,MATCH($G2,$B$2:$B100,0)),"")

Column G returns all Project Names from column B which contain the text in cell E1 ("cat skirts")

Option Explicit

Dim WithEvents FS As FileSearch
Dim Y As Long

Private Sub FS_FoundFile(ByVal FullName As String, ByVal Attributes As Long, _
    ByVal DateCreated As Date, ByVal DateLastAccessed As Date, _
    ByVal DateLastModified As Date, ByVal Size As Variant, Cancel As Boolean)
'  Dim Result
'  Result = GetTrustees(FullName)
'  Y = Y + 1
'  Range("A" & Y).Resize(, UBound(Result) + 1) = Result

  Y = Y + 1
  Range("A" & Y) = FullName
End Sub

Private Sub FS_FoundFolder(ByVal FullName As String, ByVal Attributes As Long, _
    ByVal DateCreated As Date, ByVal DateLastAccessed As Date, _
    ByVal DateLastModified As Date, Cancel As Boolean)
  Dim Result
  Result = GetTrustees(FullName)
  Y = Y + 1
  Range("A" & Y).Resize(, UBound(Result) + 1) = Result
End Sub

Private Function GetTrustees(FullName As String) As Variant
  Dim wmiFileSecSetting As Object
  Dim wmiSecurityDescriptor As Object
  Dim RetVal As Long
  Dim wmiACL As Variant
  Dim wmiACE As Variant
  Dim TRUSTEE As Object
  Dim i As Long
  Dim Result() As Variant
 
  'https://gallery.technet.microsoft.com/scriptcenter/a7f4694d-c71b-4b47-a781-4c4f832e87d7#content
 
  On Error GoTo ErrorHandler
  Set wmiFileSecSetting = GetObject("winmgmts:Win32_LogicalFileSecuritySetting.path='" & FullName & "'")
  RetVal = wmiFileSecSetting.GetSecurityDescriptor(wmiSecurityDescriptor)
  wmiACL = wmiSecurityDescriptor.DACL
  ReDim Result(0 To UBound(wmiACL) + 1)
  Result(0) = FullName
  i = 1
  For Each wmiACE In wmiACL
    Set TRUSTEE = wmiACE.TRUSTEE
    Result(i) = TRUSTEE.Domain & "\" & TRUSTEE.Name
    i = i + 1
  Next
  GetTrustees = Result
  Exit Function
ErrorHandler:
  ReDim Result(0 To 1)
  Result(0) = FullName
  Result(1) = "Access denied"
  GetTrustees = Result
End Function

Sub Main()
  Cells.ClearContents
  Y = 0
  Set FS = New FileSearch
  With FS
    .LookIn = "H:\"
    .FileName = "*.*"
    .SearchSubFolders = False
    .Execute
  End With
End Sub

scottgem say:

It's not a case of improperly designed tables (yet!). These are data points to be collected for a survey/data analysis application,

Actually, based on this description it IS an improper design. A survey database requires a very specific structure. I am suspecting that you have a field for each data point to be collected. And that IS improper design. 

The proper design for such a database would look like this:

tblQuestion
QuestionID (PK Autonumber)
Question

tblRespondent
RespondentID (PK Autonumber)
FirstName
Lastname
Other info about respondent

tblResponse
ResponseID (PK Autonumber)
RespondentID (FK)
ResponseDate

tblAnswer
AnswerID (PK Autonumber)
ResponseID (FK)
QuestionID (FK)
Answer

Bottom line is you should should identify each data point as a RECORD in a table, by identifying the question being answered and the survey the answer is from. What it sounds like you have is a record for each survey and a field for each answer.
