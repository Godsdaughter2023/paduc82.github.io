jeovany cv say:

Hi Katie

Here the macro

Note:

Create a new sheet in the workbook and name it "Report"

Then use this macro

*****************************************************************************************


Sub SummaryReport()


Dim Wb As Workbook

Dim Ws As Worksheet

Dim SumWs As Worksheet

Dim Criteria As String

Dim Sumvalue As Long


Set Wb = ThisWorkbook

Set SumWs = Wb.Sheets("Report")

WONsum = 0

LOSTsum = 0

SUBMITTEDsum = 0

UNUSEDsum = 0


 Application.ScreenUpdating = False

 WsCount = Wb.Worksheets.Count

    For i = 1 To WsCount

    If Sheets(i).Name <> "Report" Then

            Criteria = UCase(Trim(Sheets(i).Range("H6").Value))

            Sumvalue = Sheets(i).Range("G41").Value

                 If Criteria = "WON" Then

                         WONsum = WONsum + Sumvalue

                 End If

                If Criteria = "LOST" Then

                         LOSTsum = LOSTsum + Sumvalue

                 End If

                If Criteria = "SUBMITTED" Then

                         SUBMITTEDsum = SUBMITTEDsum + Sumvalue

                 End If

                If Criteria = "UNUSED" Then

                         UNUSEDsum = UNUSEDsum + Sumvalue

                 End If

        End If

    Next i



SumWs.Activate

With ActiveSheet

   .Range("A1") = "Summary Report"

   .Range("A2") = "WON $"

   .Range("A3") = "LOST $"

   .Range("A4") = "SUBMITTED $"

   .Range("A5") = "UNUSED $"

   .Range("B1") = "TOTALS"

   .Range("B2") = WONsum

   .Range("B3") = LOSTsum

   .Range("B4") = SUBMITTEDsum

   .Range("B5") = UNUSEDsum

   .Range("A:B").EntireColumn.AutoFit

   .Range("A1:B1").Font.Bold = True

   .Range("A1:B1").HorizontalAlignment = xlCenter

End With


 Application.ScreenUpdating = True

MsgBox "Done"

End Sub

*************************************************************************************
