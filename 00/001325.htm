Sub Delete_Old_Names()
'Feb 01, 2017
Dim nItem As Name
Dim N As Integer, t As Integer
Dim x As Variant
N = ActiveWorkbook.Names.Count
If N = 0 Then MsgBox "no names": Exit Sub
t = 0
For Each nItem In ActiveWorkbook.Names
If InStr(1, CStr(nItem.RefersTo), "#REF!") > 0 Then
t = t + 1
x = MsgBox("Item " & t & vbCr & nItem.Name & ", " & nItem.RefersTo, vbYesNo, "Yes = delete [" & nItem.Name & "] , No = find next item")
If x = vbYes Then nItem.Delete: GoTo nnext
End If
nnext:
Next nItem
If t = 0 Then MsgBox "not found old names"
End Sub
