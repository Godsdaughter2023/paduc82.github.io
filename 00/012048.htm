rik_uk say:

This is what I use to reposition and autosize comments in a large spreadsheet I have. If you don't want to move your comments just delete or comment out the lines in bold.

Sub ResetComments()
Dim cmt As Comment
Dim ws As Worksheet
Dim acol As Integer
Dim tHeight As Single
Dim desiredWidth As Integer, desiredHeight As Integer
Dim longestLine As Integer, x As Integer, cpl As Integer, lineCount As Integer
Dim BoxText As String, subString As String
Dim ArrText
    Set ws = ActiveSheet
    Application.DisplayCommentIndicator = xlCommentAndIndicator
    For Each cmt In ActiveSheet.Comments
        cmt.Shape.Top = cmt.Parent.Top - 5
        cmt.Shape.Left = cmt.Parent.Offset(0, 1).Left + 15
    Next
    Application.DisplayCommentIndicator = xlCommentIndicatorOnly
    For Each cmt In ws.Comments
        desiredWidth = 250
        BoxText = cmt.Parent.Comment.Text
        ArrText = Split(BoxText, Chr(10))
        longestLine = 0
        For x = 0 To UBound(ArrText)
            If Len(ArrText(x)) > longestLine Then longestLine = Len(ArrText(x))
        Next
        cmt.Shape.TextFrame.AutoSize = True
        cpl = Fix(desiredWidth / cmt.Shape.Width * longestLine) - 2
        lineCount = 0
        For x = 0 To UBound(ArrText)
            lineCount = lineCount + 1
            Do
                subString = Left(ArrText(x), cpl)
                If Len(subString) = cpl Then subString = Left(subString, InStrRev(subString, " "))
                ArrText(x) = Replace(ArrText(x), subString, "")
                If Len(ArrText(x)) > 0 Then lineCount = lineCount + 1
            Loop Until Len(ArrText(x)) = 0
        Next
        tHeight = cmt.Shape.Height / (UBound(ArrText) + 1)
        desiredHeight = tHeight * lineCount
        If longestLine * 5 < desiredWidth Then desiredWidth = longestLine * 5
        cmt.Shape.Width = desiredWidth
        cmt.Shape.Height = desiredHeight
    Next
End Sub
