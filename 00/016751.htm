Private Sub cmdStudentEmail_Click()
    DoCmd.OpenForm "MassEmailCertify", acNormal, , , , acDialog
    Dim db          As DAO.Database
    Dim MailList    As DAO.Recordset
    Dim MyOutlook   As Outlook.Application
    Dim MyMail      As Outlook.MailItem
    Dim Subjectline As String
    Dim strRecip    As String
    Dim lngCount    As Long
    Dim file        As Object

    Const MAX_RECIPIENTS_PER_MAIL = 5000

    DoCmd.SetWarnings False

    On Error Resume Next
    Set MyOutlook = GetObject(, "Outlook.Application")
    If MyOutlook Is Nothing Then Set MyOutlook = New Outlook.Application
    lblStatus.Caption = ""

    ' Set up the database and query connections
    Set db = CurrentDb()

    If Not IsNull(Me.cboSubject) Then
        DoCmd.OpenQuery "qryEmailStudSubj"
    ElseIf Not IsNull(Me.cboDept) Then
        DoCmd.OpenQuery "qryEmailStudDept"
    End If

    Set MailList = db.OpenRecordset("tblEmailSend")
    MailList.MoveFirst
    Subjectline = MailList("year") & " " & MailList("Semester") & " "

    ' This creates the e-mail
    Set MyMail = MyOutlook.CreateItem(olMailItem)

    Do Until MailList.EOF
        lblStatus.Caption = "Email: " & CStr(lngCount)
        Me.Form.Repaint

        ' This addresses it
        With MyMail.Recipients.Add(MailList("EmailAddress"))
            If .Resolve Then
                .Type = olBCC
                lngCount = lngCount + 1
            Else
                .Delete
            End If
        End With

        'And on to the next one...
        MailList.MoveNext

        If MyMail.Recipients.Count = MAX_RECIPIENTS_PER_MAIL Or (MailList.EOF And MyMail.Recipients.Count > 0) Then
            MyMail.Body = "Dear " & Subjectline & "student:" & vbNewLine & vbNewLine
            MyMail.Display

            'create next e-mail
            If Not MailList.EOF Then Set MyMail = MyOutlook.CreateItem(olMailItem)
        End If
    Loop


    'Cleanup after ourselves
    Set MyMail = Nothing
    Set MyOutlook = Nothing
    MailList.Close
    Set MailList = Nothing
    db.Close
    Set db = Nothing
    Me.lblStatus.Caption = ""

End Sub
