dhookom say: I'm not sure how you want to use the data but you need to first set a reference to the "Microsoft ActiveX Data Objects 6.1 Library" or other most recent. Then you can write code like the following which will open the Employees table and loop through the records. You will need to substitute database and other information.

Public Sub ShowData()
Dim Conn1 As New ADODB.Connection
Dim Cmd1 As New ADODB.Command
Dim Errs1 As Errors
Dim Rs1 As New ADODB.Recordset

Dim i As Integer
Dim AccessConnect As String

' Error Handling Variables
Dim errLoop As Error
Dim strTmp As String

AccessConnect = "Driver={Microsoft Access Driver (*.mdb)};" & _
"Dbq=Northwind.mdb;" & _
"DefaultDir=C:\Data\Access\;" & _
"Uid=Admin;Pwd=;"
Conn1.Open AccessConnect
Set Rs1 = Conn1.Execute("SELECT * FROM Employees")
With Rs1
Do Until .EOF

.MoveNext
Loop
End With
Rs1.Close
Conn1.Close
End Sub
