Hi,


This will find the last row and column no matter which row column that's in and then the address.



Sub somesub()
Dim LastCol As Long, LastRow As Long
Dim LastColLetter As String, LastRowLetter As String


LastCol = Cells.Find(What:="*", After:=[A1], _
SearchOrder:=xlByColumns, _
SearchDirection:=xlPrevious).Column


LastRow = Cells.Find(What:="*", After:=[A1], _
SearchOrder:=xlByRows, _
SearchDirection:=xlPrevious).Row


LastColLetter = Cells.Find(What:="*", After:=[A1], _
SearchOrder:=xlByColumns, _
SearchDirection:=xlPrevious).Address(1, 0)


LastRowLetter = Cells.Find(What:="*", After:=[A1], _
SearchOrder:=xlByRows, _
SearchDirection:=xlPrevious).Address(1, 0)



End Sub
