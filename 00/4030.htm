Ok, but it got interesting - determining what color is selected in the dropdown is mostly impossible.
But, pick a cell to copy a color from and you are off and running.
'---

Sub HighlightWorkSheet_R2()
'Uses ConvertColorNumberToRGB function

 Dim objAny As Object
 Dim lngColor As Long

'Uses cell A1 to determine color to use - change to suit.  <<<<
'If cell has no color then Yellow is used.
 If Range("A1").Interior.ColorIndex <> xlColorIndexNone Then
   lngColor = Range("A1").Interior.Color
 Else
  'vbBlack vbRed vbYellow vbGreen vbCyan vbBlue vbWhite vbMagenta  <<<< more choices
   lngColor = vbYellow
 End If

 Set objAny = Selection
 If TypeName(objAny) = "Range" Then
   With objAny.Interior
    .Pattern = xlSolid
    .Color = lngColor
   End With
 ElseIf TypeName(objAny) = "TextBox" Then
  'call function

   lngColor = ConvertColorNumberToRGB(lngColor)
   With objAny.ShapeRange.Fill
     .Solid
     .Transparency = 0
     .ForeColor.RGB = lngColor
   End With

 Else

   Msgbox "Get a job.   " 'always good advice
 End If
End Sub
'---
 Function ConvertColorNumberToRGB(ByRef ColorNumber As Long) As Variant
'By Leo Heuser - RIP
  Dim Red As Long
  Dim Green As Long
  Dim Blue As Long

  Red = ColorNumber And &HFF
  Green = (ColorNumber And CLng(&HFF) * &H100) / &H100
  Blue = (ColorNumber And CLng(&HFF) * &H10000) / &H10000
  ConvertColorNumberToRGB = RGB(Red, Green, Blue)
 End Function
