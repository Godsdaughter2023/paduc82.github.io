        Set VTDoc = Documents.Open(p & VTFile)

andreas killer say:

Copy the code below into a regular module and use this formula:

C2:  =ExtractingNumbers(B2,"O/S")

Drag the formula down to the end of the data.

Andreas.

Option Explicit

Function ExtractingNumbers(ByVal S As String, Optional ByVal After) As String
  Dim i As Long, f As Long
  If Not IsMissing(After) Then
    'Only the part after
    i = InStr(1, S, After, vbTextCompare)
    If i > 0 Then S = Mid(S, i + Len(After))
  End If
  'Remove blanks
  S = Trim(S)
  'Scan for numbers and valid delimiters
  For i = 1 To Len(S)
    Select Case Mid(S, i, 1)
      Case "-", "0" To "9"
        'These signs are okay
      Case Else
        'Done
        Exit For
    End Select
  Next
  'Return the result
  ExtractingNumbers = Left(S, i - 1)
End Function
