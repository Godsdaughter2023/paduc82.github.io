Word really doesn't handle dates before 1/1/1901, let alone the Julian calendar, so the field code to do it properly is quite complex:

{QUOTE
{ASK StartDate "What is the date, in d-m-y format, please? Show dates B.C. as -ve and subtract 1 from the year number (eg: input 1 January 101 BC as -1-1-100)."}
{SET d{=-{=-{StartDate}-StartDate}/2}}
{SET m{=-({StartDate}*(-1)-{StartDate})/2}}
{SET y{=-({d}+{m}-{StartDate})}}{SET y{=y*IF(OR(d<0,y<1),-1,1)}}
{SET StartM"{=m}/0"}
{SET a{=INT((14-m)/12)}}
{SET b{=y+4800-a}}
{SET c{=m+12*a-3}}
{SET d{=abs(d)}}
{SET jd{=IF(y*10000+m*100+d<15821005,{=d+INT((153*c+2)/5)+365*b+INT(b/4)-32083},{=d+INT((153*c+2)/5)+365*b+INT(b/4)-INT(b/100)+INT(b/400)-32045})}}
{IF{=(d+m=0)+(d>4)*(d<15)*(m=10)*(y=1582)}= 1 "Date Input error. Please try again." "{{SET e{=INT((4*(jd+32044)+3)/146097)*(jd>2299160)}}
{SET f{=jd+32082-(38+INT(146097*e/4))*(jd>2299160)}}
{SET g{=INT((4*f+3)/1461)}}
{SET h{=f-INT(1461*g/4)}}
{SET i{=INT((5*h+2)/153)}}
{SET day{=h-INT((153*i+2)/5)+1}}
{SET mnth"{=i+3-12*INT(i/10)}/0"}
{SET yr{=100*e+g-4800+INT(i/10)}}
{SET yn{=ABS({yr})+({yr}<1) \# 0}}}{IF{=MOD(jd,7)}= 0 Mon {IF{=MOD(jd,7)}= 1 Tues {IF{=MOD(jd,7)}= 2 Wednes {IF{=MOD(jd,7)}= 3 Thurs {IF{=MOD(jd,7)}= 4 Fri {IF{=MOD(jd,7)}= 5 Satur Sun}}}}}}day, the {day \* Ordtext} day of {mnth \@ MMMM} {IF{yr}> 0 "in the year of our Lord, "}{=INT({yn}/100)*100 \*CardText}{IF{=MOD({yn},100)}<> 0 " and {=MOD({yn},100) \* CardText}}{yr \# "; B.C.; B.C."}"}}

 

Note: The field brace pairs (ie '{ }') for the above example are created in the body of the document via Ctrl-F9 (Cmd-F9 on a Mac) - you can't simply type them or copy & paste them from this message.

 

You can download a macro to urn the above into a working field from: http://www.gmayor.com/export_field.htm#TextToField

 

PS: The field code even gives you the day of the week! If you don't want that, delete:

{IF{=MOD(jd,7)}= 0 Mon {IF{=MOD(jd,7)}= 1 Tues {IF{=MOD(jd,7)}= 2 Wednes {IF{=MOD(jd,7)}= 3 Thurs {IF{=MOD(jd,7)}= 4 Fri {IF{=MOD(jd,7)}= 5 Satur Sun}}}}}}day,

 

Of course, not even the Julian Calenday existed before 45BC, so expressing dates before then in that form is somewhat anachronistic and even leap years were initially calculated incorrectly - see: http://en.wikipedia.org/wiki/Julian_calendar#Leap_year_error

Dim KeepRefreshing As Boolean
Sub StopRefresh()
KeepRefreshing = False
End Sub
Sub AutomaticRefresh()
If KeepRefreshing Then
Application.OnTime Now + TimeValue(ThisWorkbook.Worksheets("Now").Range("H19").Text), "RefreshThisSheet"
'Application.Caption = Sheets("Lab").Range("A50").Value
End If
End Sub
Sub StartAutoRefresh()
        ' This function sets it all going, by setting the flag and
        ' making the initial call to RefreshThisSheet
        KeepRefreshing = True
        RefreshThisSheet
End Sub
Sub RefreshThisSheet()
         ' This function refreshes the sheet, and then calls the AutomaticRefresh
         ' function, which will call it back in a couple of seconds
         Application.ScreenUpdating = True
         Application.CalculateFull
         AutomaticRefresh
End Sub
Sub RestartRefresh()
KeepRefreshing = False
StartAutoRefresh
End Sub
