  Dim Item, Ctrl As Control
  Dim Y As Long, Count As Integer
   On Error Resume Next
  For Each Item In Split(ControlTypes, ";")
    Set Ctrl = Me.Controls.Add("Forms." & Item & ".1")
    Count = Count + 1
    Ctrl.Caption = Count
    Ctrl.Value = Count
    Ctrl.Top = Y
    Ctrl.AddItem Count

  Next

  Me.Height = 2 * Y - Me.InsideHeight
End Sub
