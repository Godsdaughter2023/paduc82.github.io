hansv mvp say:

Looping might be slow. Here is a macro that does it in two steps. Don't run it twice on the same file!

Sub ReplaceFirstSpace()
    Selection.HomeKey Unit:=wdStory
    With Selection.Find
        .ClearFormatting
        .Replacement.ClearFormatting
        ' First line
        .MatchWildcards = False
        .Text = "^w"
        .Replacement.Text = "^t"
        .Execute Replace:=wdReplaceOne
        Selection.Collapse Direction:=wdCollapseEnd
        ' Other lines
        .MatchWildcards = True
        .Text = "([^13^l][!^s^32]{1,})[^32^s]{1,}"
        .Replacement.Text = "\1^t"
        .Wrap = wdFindStop
        .Execute Replace:=wdReplaceAll
    End With
End Sub
