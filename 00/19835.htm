bernie deitrick say:

For the different headers:

If "City" "centre" "center" and "town" are all allowed variants - use this version. But, first insert 4 rows above your headers on Summary, and enter the main headings into row 5. Then enter allowed variants (up to 4) as a list above row 5, filling up in order, in row 4, 3, 2, then 1.  

This version will look for the main heading "City" (A5) and if does not find it, will look for the heading from A4, then A3, etc. (unless those cells are blank).  It will only find one variant for each column.  If you need more than 5 variants, change the 5 in this line to higher number

iPossibles = 5 

and make sure to insert a suitable number or rows above row 5 so that you main list of headers is in the correct row.

Sub SummarizeSheets2()
    Dim xlCalc As XlCalculation
    Dim shtS As Worksheet
    Dim shtT As Worksheet
    Dim rngC As Range
    Dim rngH As Range
    Dim strN As String
    Dim strF As String
    Dim rngF As Range
    Dim lrow As Long
    Dim iOff As Integer
    Dim boolRecordSheet As Boolean
    Dim iPossibles As Integer

    strN = "Summary"

    With Application
        .ScreenUpdating = False
        xlCalc = .Calculation
        .Calculation = xlCalculationManual
        .EnableEvents = False
        .DisplayAlerts = False
    End With

    Set shtS = Worksheets(strN)

    boolRecordSheet = MsgBox("Record sheet source?", vbYesNo) = vbYes

    'Use row 5 for headers values, rows 4, 3, 2, and 1 for possible variants.
    'Increase the 5 for more allowed variations to search for

    iPossibles = 5

    With shtS
        Set rngH = .Range(.Range("A" & iPossibles), .Cells(iPossibles, .Columns.Count).End(xlToLeft))
    End With

    shtS.UsedRange.Offset(iPossibles).EntireRow.Delete

    For Each shtT In Worksheets
        If shtT.Name <> strN Then
            lrow = shtS.UsedRange.Rows.Count + 1
            iOff = 0
            For Each rngC In rngH
TryNext:
                strF = rngC.Offset(iOff, 0).Value
                If strF <> "" Then
                    Set rngF = shtT.Range("1:1").Find(strF, LookAt:=xlWhole, MatchCase:=False)
                    If Not rngF Is Nothing Then
                        shtT.Range(rngF.Offset(1), shtT.Cells(shtT.Rows.Count, rngF.Column).End(xlUp)).Copy _
                        shtS.Cells(lrow, rngC.Column)
                    Else
                        iOff = iOff - 1
                        If iOff > -iPossibles Then GoTo TryNext
                    End If
                End If
                iOff = 0
            Next rngC
            If boolRecordSheet Then
                shtS.Cells(lrow, rngH.Columns.Count + 1).Resize(shtS.UsedRange.Rows.Count - lrow + iPossibles).Value = shtT.Name
            End If
        End If
    Next shtT

    If boolRecordSheet Then
        shtS.Cells(iPossibles, rngH.Columns.Count + 1).Value = "Sheet Source"
    End If

    With Application
        .ScreenUpdating = True
        .EnableEvents = True
        .DisplayAlerts = True
        .Calculation = xlCalc
    End With
End Sub
