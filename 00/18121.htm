Sub IncreaseDate()
    Dim rngHour As Range, rngMinute As Range
    Dim rngMonth As Range, rngDate As Range
    Dim rngFindValue As Range
    Dim sht As Worksheet
    Dim varMonths As Variant, i As Integer
    
    Set sht = ThisWorkbook.Worksheets("summary")
    
    With sht
        Set rngHour = .Range("A2")
        Set rngMinute = .Range("B2")
        Set rngMonth = .Range("C2")
        Set rngDate = .Range("D2")
    End With
    
    varMonths = Split(rngMonth.Validation.Formula1, ",")

    Set rngFindValue = Range("Date_" & rngMonth.Value).Find(rngDate.Value, , , XlLookAt.xlWhole)
    If Not rngFindValue Is Nothing Then
        Set rngFindValue = rngFindValue.Offset(1, 0)

        'Not a new month
        If Trim(CStr(rngFindValue.Value)) <> "" Then
            rngDate.Value = rngFindValue.Value
            
        'New month
        Else
            If rngMonth.Value <> varMonths(UBound(varMonths)) Then 'Not dec
                For i = LBound(varMonths) To UBound(varMonths)
                    If rngMonth.Value = varMonths(i) Then Exit For
                Next i
                rngMonth.Value = varMonths(i + 1)
            End If
            rngDate.Value = Range("Date_" & rngMonth).Range("A1")
        End If
    End If
    
End Sub
