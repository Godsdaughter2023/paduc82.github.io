'
' PasteSpecialCtrlK Macro
'
' Keyboard Shortcut: Ctrl+k
'
Sub PasteSpecialCtrlK()
    Selection.PasteSpecial Paste:=xlPasteValues, Operation:=xlNone, SkipBlanks _
        :=False, Transpose:=False
    Application.CutCopyMode = False
    Selection.NumberFormat = "0.00"
End Sub

 

'

' Workbook_Open Macro (PERSONAL.XLSB)

'

Private Sub Workbook_Open()
    Application.OnKey "^k", "PERSONAL.XLSB!PasteSpecialCtrlK" '<<< Ctrl+k will run PasteSpecialCtrlK from the PERSONAL.XLSB file
End Sub

SELECT [Part Number], "A" AS TrialRun
