macropod say:

I need a solution that will make my pasted "test" word to appear automatically in the table of content

That doesn't even happen with the standard Table of Contents, which has to be refreshed before the changes appear. You need to accept the fact that what you're asking for is something Word is not designed to do. Accordingly, if something comparable to what you want is required, one has to devise a workaround and accept the limitations that come with such an approach. Given that stipulation, the following macro that might do the job. It assumes you have already created the 4-column table that is to be used for the custom 'Table of Contents' and that that table is the first in the document.

Sub Demo()
Application.ScreenUpdating = False
Dim h As Long, i As Long, j As Long, TblTOC As Table, TblTmp As Table, RngTmp As Range, strHead As String, Headings
Headings = ActiveDocument.GetCrossReferenceItems(wdRefTypeHeading)
With ActiveDocument.Range
  With .Tables(1)
    .Range.Fields.Update
    If .Rows.Count > 1 Then
      .Split 2
      Set TblTmp = ActiveDocument.Tables(2)
    End If
  End With
  Set TblTOC = .Tables(1)
  With .Find
    .ClearFormatting
    .Replacement.ClearFormatting
    .Text = ""
    .Style = "Heading 1"
    .Replacement.Text = ""
    .Forward = True
    .Wrap = wdFindStop
    .Format = True
    .MatchCase = False
    .MatchWholeWord = False
    .MatchWildcards = False
    .MatchSoundsLike = False
    .MatchAllWordForms = False
    .Execute
  End With
  Do While .Find.Found
    strHead = Split(.Text, vbCr)(0)
    While Asc(Left(strHead, 1)) < 33
      strHead = Right(strHead, Len(strHead) - 1)
    Wend
    With TblTOC
      .Rows.Add
      With .Cell(.Rows.Count, 1).Range
        .End = .End - 1
        .Fields.Add Range:=.Duplicate, Type:=wdFieldEmpty, Text:="SEQ Toc"
        .Style = "TOC 1"
      End With
      For h = 1 To UBound(Headings)
        'MsgBox Chr(34) & Trim(Headings(h)) & Chr(34) & vbCr & Chr(34) & strHead & Chr(34)
        If Trim(Headings(h)) = strHead Then
          With .Cell(.Rows.Count, 2).Range
            .End = .End - 1
            .InsertCrossReference _
              ReferenceType:=wdRefTypeHeading, _
                ReferenceKind:=wdContentText, _
                ReferenceItem:=h
            .Style = "TOC 1"
          End With
          With .Cell(.Rows.Count, 3).Range
            .End = .End - 1
            .InsertCrossReference _
              ReferenceType:=wdRefTypeHeading, _
              ReferenceKind:=wdPageNumber, _
              ReferenceItem:=h, _
              InsertAsHyperlink:=True
            .Style = "TOC 1"
          End With
          Exit For
        End If
      Next
    End With
    'The next If ... End If block is needed in case the heading is in a table
    If .Information(wdWithInTable) = True Then
      If .End = .Cells(1).Range.End - 1 Then
        .End = .Cells(1).Range.End
        .Collapse wdCollapseEnd
        If .Information(wdAtEndOfRowMarker) = True Then
          .End = .End + 1
        End If
      End If
    End If
    'The next line is  needed in case a heading includes the final paragraph break
    If .End = ActiveDocument.Range.End Then Exit Do
    .Collapse wdCollapseEnd
    .Find.Execute
  Loop
  TblTOC.Range.Fields.Update
  If Not TblTmp Is Nothing Then
    For i = 2 To TblTOC.Rows.Count
      For j = 1 To TblTmp.Rows.Count
        If Split(TblTOC.Cell(i, 2).Range.Text, vbCr)(0) = Split(TblTmp.Cell(j, 2).Range.Text, vbCr)(0) Then
          Set RngTmp = TblTmp.Cell(j, 4).Range
          RngTmp.End = RngTmp.End - 1
          TblTOC.Cell(i, 4).Range.FormattedText = RngTmp.FormattedText
          Exit For
        End If
      Next
    Next
    Set RngTmp = TblTmp.Range
    RngTmp.MoveStart wdParagraph, -1
    RngTmp.Delete
  End If
End With
Set TblTOC = Nothing: Set TblTmp = Nothing: Set RngTmp = Nothing
Application.ScreenUpdating = True
End Sub
