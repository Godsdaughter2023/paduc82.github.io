  Next

graham mayor say:
Without attempting to decipher your macro, it appears that there is insufficient space to accommodate the extra character when the number has two digits rather than one. You need to modify the Figure Caption style to increase the tab spacing to allow the extra character.

You would do better to use ranges to format the text e.g.

Sub InsertFigure(control As IRibbonControl)
Dim oRng As Range
    Set oRng = Selection.Range
    With oRng
        .Text = "<Insert your figure here>" & vbCr
        .Style = "Figure Space,fs"
        .Collapse wdCollapseEnd
        .Text = vbTab & "<Type your Figure Caption here>"
        .Collapse wdCollapseStart
        .InsertCaption Label:="Figure", TitleAutoText:="InsertCaption1", _
                       Title:="", Position:=wdCaptionPositionBelow, ExcludeLabel:=0
        With CaptionLabels("Figure")
            .NumberStyle = wdCaptionNumberStyleArabic
            .IncludeChapterNumber = False
            .ChapterStyleLevel = 1
            .Separator = wdSeparatorHyphen
        End With
        .Style = "Figure Caption"
    End With
End Sub
