Put the AutoOpen in the template project or but the Document_Open version in the "ThisDocument" module of the document.  Both are revised to but the number "in" vice at the bookmark.

Sub AutoOpen()
 Dim strSeqNumber As String
 Dim oRng As Word.Range
   'Get stored registry value
   strSeqNumber = GetSetting("Sequenced Documents", "Stored Values", "SeqNumber")
   If strSeqNumber = "" Then
     'Create initial value. (HKEY_Current_User/Software/VB and VBA program settings
     SaveSetting "Sequenced Documents", "Stored Values", "SeqNumber", "1"
     strSeqNumber = GetSetting("Sequenced Documents", "Stored Values", "SeqNumber")
   End If
   strSeqNumber = strSeqNumber + 1
   SaveSetting "Sequenced Documents", "Stored Values", "SeqNumber", strSeqNumber
   Set oRng = ActiveDocument.Bookmarks("docNumber").Range
   oRng.Text = Format(strSeqNumber, "00#")
   ActiveDocument.Bookmarks.Add "docNumber", oRng
   'ActiveDocument.SaveAs FileName:="path" & Format(strSeqNumber, "00#")
   Exit Sub
 End Sub
