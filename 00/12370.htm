STEPS TO SUBMIT AN OUTDATED CACHE REMOVAL REQUEST

You can submit an outdated cache removal request for a page that is live on the web (HTTP status code 200) and for which we still have old and outdated content in our cache by doing the following:

Instead of doing a pivot table in code, I just coded what you wanted.  Minimal formatting was added, but in the code you can see where to alter that if you wish.  All I did was bold and center the header row; and autofit.

One way to learn about how to do that would be to record a macro while you are formatting, and observe what comes out. 

Again, this replaces the Regular Module.  I re-used the Class Object when collecting items by Item, and not by item/date pair.

Option Explicit
 Sub PairCount() 'R is the table
     'Can have a range with any number of columns
   Dim wsSrc As Worksheet, wsRes As Worksheet, rRes As Range
   Dim vSrc As Variant, V() As Variant, vRes() As Variant
   Dim cP As cPairs, colP As Collection
   Dim I As Long, J As Long
   Dim sKey As String
  
'Set source and result worksheets and ranges
'Read source data into variant array for speed
Set wsSrc = Worksheets("sheet1")
Set wsRes = Worksheets("sheet2")
    Set rRes = wsRes.Range("A1")

    vSrc = .Range(.Cells(1, 1), .Cells(.Rows.Count, "A").End(xlUp)).Resize(columnsize:=2)
End With

 On Error Resume Next
 For I = 2 To UBound(V, 1) 'start at row 2 to exclude headers
     If vSrc(I, 1) <> "" Then 'exclude blanks
     For J = 1 To UBound(vSrc, 2)
         V(J) = vSrc(I, J)
     Next J
         Set cP = New cPairs
         With cP
             .RW = V
             .RwCNT = 1
             sKey = Join(V, "|")
             colP.Add cP, sKey
            
             Select Case Err.Number
                 Case 457 'it is a duplicate
                     colP(sKey).RwCNT = colP(sKey).RwCNT + 1
                     Err.Clear
                 Case Is <> 0 'some other error
                     Debug.Print "Error: " & Err.Number, Err.Description
                     Stop
             End Select
         End With
     End If
 Next I
 On Error GoTo 0
 
'Dimension results array
ReDim vRes(1 To colP.Count, 1 To 4)

For I = 1 To colP.Count
With colP(I)
    V = .RW
    vRes(I, 1) = V(1)
    vRes(I, 2) = V(2)
    vRes(I, 3) = .RwCNT
    vRes(I, 4) = Int(.RwCNT / 2)
End With
Next I

Set colP = New Collection
On Error Resume Next
For I = 2 To UBound(vRes)
    Set cP = New cPairs
    With cP
        .RW = vRes(I, 1)
        .RwCNT = vRes(I, 4)
        sKey = CStr(.RW)
        colP.Add cP, sKey
       
        Select Case Err.Number
            Case 457
                colP(sKey).RwCNT = colP(sKey).RwCNT + .RwCNT
                Err.Clear
            Case Is <> 0
                Debug.Print "Error: " & Err.Number, Err.Description
        End Select
    End With
Next I
On Error GoTo 0

ReDim vRes(0 To colP.Count, 1 To 2)
    vRes(0, 1) = "Item"
    vRes(0, 2) = "Duplicate Count"

    With colP(I)
        vRes(I, 1) = .RW
        vRes(I, 2) = .RwCNT
    End With
Next I

'Write to worksheet
Set rRes = rRes.Resize(rowsize:=UBound(vRes, 1) + 1, columnsize:=UBound(vRes, 2))
With rRes
    .EntireColumn.Clear
    .Value = vRes
   
    'Could format in various ways
    With .Rows(1)
        .Font.Bold = True
        .HorizontalAlignment = xlCenter
    End With
    .EntireColumn.AutoFit
   
End With
