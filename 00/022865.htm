Sub FormatFixer()
    'Written By: Richard V.Michaels, MVP
    'http:\\www.greatcirclelearning.com
    
    FixStyle "32", "Heading 1"
    FixStyle "30", "Heading 2"
    FixStyle "28", "Heading 3"
    FixStyle "26", "Heading 4"
    FixStyle "24", "Heading 5"
    FixStyle "22", "Heading 6"
    FixStyle "20", "Heading 7"
    
    Dim sec As Word.Section
    Dim hf As Word.HeaderFooter
    Dim fNote As Word.Footnote
    Dim rng As Word.Range
    
    
    For Each sec In ActiveDocument.Sections
        For Each hf In sec.Headers
            'three potential headers per section have to be scanned
            If hf.LinkToPrevious = True Then Exit For
            Set rng = hf.Range
            FixOtherLayers rng
        Next
        For Each hf In sec.Footers
            'three potential footers per section have to be scanned
            If hf.LinkToPrevious = True Then Exit For
            Set rng = hf.Range
            FixOtherLayers rng
      Next
    Next
    
    For Each fNote In ActiveDocument.Footnotes
        'footnotes are their own special collection that must be scanned
        Set rng = fNote.Range
        FixOtherLayers rng
        rng.Style = "Footnote Text "
    Next
    
     Set rng = ActiveDocument.Range
     FixOtherLayers rng
    
End Sub

Private Function FixStyle(ByRef fSize As String, ByRef sName As String)
    Selection.HomeKey Word.WdUnits.wdStory
    With Selection.Find
        .ClearFormatting
        .Font.Size = fSize
        .Format = True
        .Forward = True
        .Text = ""
        .Wrap = wdFindStop
        .Execute
        Do While .Found
            .Parent.Expand Word.WdUnits.wdParagraph
            Selection.ClearFormatting
            Selection.Style = sName
            Selection.Collapse Word.WdCollapseDirection.wdCollapseEnd
            .Execute
        Loop
    End With
End Function

Private Function FixOtherLayers(ByRef rng As Word.Range)
    With rng.Find
        .ClearFormatting
        .Replacement.ClearFormatting
        .Forward = True
        .Text = "(+)(*)(+)"
        'I didn't know the special character you used that surrounded the string
        'so for testing I just used a plus sign. The find is setup for 3 continguous groups
        'Group 1 is the special character, group 2 is the variable text, and group 3 is the ending
        'special character
        .MatchWildcards = True
        .Format = True
        .Replacement.Font.SmallCaps = True
        .Replacement.Text = "\2"
        'this wildcard replacement is for the variable text between the special characters
        'only the variable text is replaced and the special characters are dropped
        .Execute Replace:=Word.WdReplace.wdReplaceAll
    End With
End Function

    End With
End Function

I think the problem you are having with the special characters is because you assume Footnotes are part of the Footer layer of a Word document, which they are not. They are their own separate collection that must be scanned separately. I did not realize that these special character strings existed potential in footnotes as well as headers and footers. If they also potentially exist in the body of the document you will have to add code for that as well.

Here is revised code that incorporates a scan for special character strings in the footnotes. Since adding another collection scan made the code a bit messy, I tidied that up as well.

Good Luck
