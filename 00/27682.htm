For the whole workbook:

Function CountCcolor(range_data As Range, criteria As Range) As Long
    Dim datax As Range
    Dim xcolor As Long
    Dim sh As Worksheet
    
    xcolor = criteria.Interior.ColorIndex
    
    For Each sh In Worksheets
        For Each datax In sh.Range(range_data.Address)
        If datax.Interior.ColorIndex = xcolor Then
            CountCcolor = CountCcolor + 1
        End If
        Next datax
    Next sh
End Function

=CountCcolor(A1:A7,B1,{"Sheet1","Sheet2"})

Function CountCcolor(range_data As Range, criteria As Range, shN As Variant) As Long
