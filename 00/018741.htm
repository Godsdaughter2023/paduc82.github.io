    hansv mvp say:

    Try this version:

    Sub RemoveLastPeriod()
      Dim tbl As Table
      Dim oCell As Cell
      Dim par As Paragraph
      Application.ScreenUpdating = False
      For Each tbl In ActiveDocument.Tables
        For Each oCell In tbl.Range.Cells
          Set par = oCell.Range.Paragraphs(oCell.Range.Paragraphs.Count)
          If par.Range.ListFormat.ListType = wdListBullet And Asc(Right(par.Range.Text, 3)) = 46 Then
            oCell.Range.Characters(Len(oCell.Range.Text) - 2).Delete
          End If
        Next oCell
      Next tbl
      Application.ScreenUpdating = True
    End Sub
