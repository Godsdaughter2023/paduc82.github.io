Sub FillLists()
Dim lngIndex As Long
Dim oTbl As Table
Dim oCC As ContentControl
  Set oTbl = ActiveDocument.Tables(1)
  For Each oCC In ActiveDocument.ContentControls
    If oCC.Title = "Document List" Then
      For lngIndex = oCC.DropdownListEntries.Count To 2 Step -1
        oCC.DropdownListEntries(lngIndex).Delete
      Next lngIndex
      For lngIndex = 1 To oTbl.Rows.Count
        oCC.DropdownListEntries.Add fcnGetCellText(oTbl.Cell(lngIndex, 1))
      Next
    End If
  Next
lbl_Exit:
  Exit Sub
End Sub

Function fcnGetCellText(ByRef oCell As Word.Cell) As String
  'Replace the end of cell marker with a null string.

lbl_Exit:
  Exit Function
End Function
