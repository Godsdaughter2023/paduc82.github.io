Why are you using the Address bar as a search bar?   Use the Search bar, e.g. press Ctrl-e and then type your query there.

The symptoms that you are complaining about suggest that you have disabled search from the Address bar (e.g. in Options, Advanced tab, Search from the Address bar section) and that your ISP's DNS has a feature which does a search for you when you enter partial or invalid domain names and that that search just happens to be BING.   None of that would happen if you used the Address bar only with complete valid URLs and did any searches you need to do only in the Address bar.

 

Alternatively, if you must use the Address bar to search from, you should use the question mark space prefix (? ) as a signal that you do not want a DNS lookup to be done on what you are entering there.   That feature has been there since at least IE4 but I think that getting into the habit of pressing Ctrl-e instead would be more useful to you.


HTH

And then update the worksheet. All of this without the user having to manually access the worksheet themselves. Is this possable?

andreas killer say:
Yes, but we must set the RowSource to an empty string, otherwise the items in the ListBox are protected.

For the example below, write some values in the cells A1:A6, create a Userform with a Textbox and a Listbox, set the RowSource property of the Listbox to A1:A6.

If you click the ListBox, the item is loaded into the Textbox. If you modify the value in the Textbox, the item in the Listbox and in the worksheet is modified.

Andreas.

Option Explicit

Dim RowSourceRange As Range

Private Sub ListBox1_Click()
  TextBox1.Value = ListBox1.Value
End Sub

Private Sub TextBox1_Change()
  With ListBox1
    If .ListIndex < 0 Then Exit Sub
    .List(.ListIndex) = TextBox1.Value
    RowSourceRange(.ListIndex + 1) = TextBox1.Value
  End With
End Sub

Private Sub UserForm_Initialize()
  Dim R As Range
  With ListBox1
    Set RowSourceRange = Range(.RowSource)
    .RowSource = ""
    For Each R In RowSourceRange
      .AddItem R
    Next
  End With
End Sub

Option Explicit

Dim RowSourceRange As Range
Dim TextBox() As MSForms.TextBox

Private Sub ListBox1_Click()
  Dim ThisRow() As String, i As Integer
  With ListBox1
    'Should we work?
    If Not Application.EnableEvents Then Exit Sub
    'Prevent that this code runs the other event
    Application.EnableEvents = False
    'Copy them from the listbox into the textboxes
    For i = 0 To .ColumnCount - 1
      TextBox(i + 1) = .List(.ListIndex, i)
    Next
    Application.EnableEvents = True
  End With

  If Not Application.EnableEvents Then Exit Sub

    RowSourceRange(.ListIndex + 1, Number) = TextBox(Number).Value

Sub Download_and_Import()
Dim StartDate As Date
Dim EndDate As Date
Dim Current_Date As Date

StartDate = DateSerial(2013, 12, 6)
EndDate = DateSerial(Year(Now() - 1), Month(Now() - 1), Day(Now() - 1))

Current_Date = StartDate

Do

Downloads Current_Date
Current_Date = Current_Date + 1
Loop Until Current_Date > EndDate

Mass_Import_Optidrive StartDate, EndDate

End Sub



Function Mass_Import_Optidrive(Start_Date As Date, End_Date As Date, Optional Historical As Boolean)

Dim File_Date As Date

File_Date = Start_Date

Do


Import_OptiDrive_File Year(File_Date) & Right("0" & Month(File_Date), 2) & Right("0" & Day(File_Date), 2) & ".txt", "c:\"

File_Date = File_Date + 1

Loop Until File_Date > End_Date
End Function



Function Import_OptiDrive_File(strFileName As String, strPath As String)

Write_Schema strPath, strFileName, ";"

    Dim Data_Date As String
    Dim SQL_part1 As String
    Dim SQL_part2 As String
    Dim db As Database, rs As Recordset
    Set db = OpenDatabase(strPath, False, _
    False, "TEXT;Database=" & strPath & ";table=" & strFileName)
    Set rs = db.OpenRecordset(strFileName)

    Data_Date = Mid(strFileName, 5, 2) & "/" & Mid(strFileName, 7, 2) & "/" & Left(strFileName, 4)

rs.MoveFirst
While Not rs.EOF

'Had to split SQL as was over 255 char

SQL_part1 = "INSERT INTO `Data - OptiDrive RAW` ( `Date`, `Driver`, `Driver_No`, `Driving events`, `Average severity`, `Driving events indicator`, `Speeding indicator`, " & _
"`Speeding events`, `Speeding time`, `OptiDrive Indicator`, `Trend`)"
                                
                    SQL_part2 = " VALUES (#" & Data_Date & "#, '" & _
                                Replace(rs.Fields("drivername"), "'", "''") & "', '" & _
                                rs.Fields("driverno") & "', '" & _
                                rs.Fields("drivingevents_count") & "', '" & _
                                rs.Fields("drivingevents_avgseverity") & "', '" & _
                                rs.Fields("drivingevents_indicator") * 10 & "', '" & _
                                rs.Fields("speeding_indicator") * 10 & "', '" & _
                                rs.Fields("speeding_avgexcess") & "', '" & _
                                rs.Fields("speeding_time") & "', '" & _
                                rs.Fields("optidrive_indicator") * 10 & "', '" & _
                                rs.Fields("trend") & "');"

DoCmd.SetWarnings False
DoCmd.RunSQL SQL_part1 & SQL_part2
DoCmd.SetWarnings True

rs.MoveNext
Wend

End Function
