hansv mvp say:

Try this:

Sub AddCF()
    Dim mI As Long, mO As Long, mR As Long
    mI = Range("I" & Rows.Count).End(xlUp).Row
    mO = Range("O" & Rows.Count).End(xlUp).Row
    mR = Range("R" & Rows.Count).End(xlUp).Row
    With Range("I2:I" & mI & ",O2:O" & mO & ",R2:R" & mR).FormatConditions
        .Delete
        .Add(Type:=xlBlanksCondition).Interior.ColorIndex = xlColorIndexNone
        .Add(Type:=xlCellValue, Operator:=xlGreaterEqual, Formula1:="=5").Interior.Color = vbGreen
        .Add(Type:=xlCellValue, Operator:=xlGreater, Formula1:="=3").Interior.Color = RGB(255, 192, 0)
        .Add(Type:=xlCellValue, Operator:=xlLessEqual, Formula1:="=3").Interior.Color = vbRed
    End With
End Sub

Sub LoopThroughDirectory()

x = 1
