Function countformats(MyRange As Range, ChkRange As Range) As Long
Dim c As Range
For Each c In MyRange
    If c.Interior.Color = ChkRange.Interior.Color And _
    c.Interior.pattern = ChkRange.Interior.pattern And _
    c.Interior.PatternColor = ChkRange.Interior.PatternColor And _
    c.Borders(xlEdgeLeft).LineStyle = ChkRange.Borders(xlEdgeLeft).LineStyle And _
    c.Borders(xlEdgeLeft).Color = ChkRange.Borders(xlEdgeLeft).Color And _
