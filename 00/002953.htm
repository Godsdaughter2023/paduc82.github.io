Or you can resave the file as in .pptm format, then add a macro:

  Dim oShape As Shape, oSlide As Slide, x As Integer
  x = 1
  For Each oSlide In ActivePresentation.Slides
    If oSlide.SlideShowTransition.Hidden = msoFalse Then
      With oSlide.Shapes
        Set oshp = .AddTextbox(msoTextOrientationHorizontal, 620, 495, 100, 50)
        With oshp.TextFrame.TextRange
          With .Font
            .Name = "Arial"
            .Size = 12
          End With
          .Text = x
        End With
      End With
      x = x + 1
    End If
  Next oSlide
  Set oshp = Nothing
  Set oSlide = Nothing
End Sub
