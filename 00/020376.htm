Sub Copy_comments()
'
' Copy comments one cell to the right for 19 cells (to the left of the selected cell) and erase comment from current cell
' Still needs to copy last comment to the left back one cell - since that comment is usually correctly copied
'
    Dim coluna As Integer
    ActiveCell.Offset(rowOffset:=0, columnOffset:=-18).Activate
    For coluna = 1 To 19
        Selection.Copy
        ActiveCell.Offset(rowOffset:=0, columnOffset:=-1).Activate
        Selection.PasteSpecial Paste:=xlPasteComments, Operation:=xlNone, _
            SkipBlanks:=False, Transpose:=False
        ActiveCell.Offset(rowOffset:=0, columnOffset:=2).Activate
    Next coluna
    ActiveCell.Offset(rowOffset:=0, columnOffset:=-1).Activate
    Selection.ClearComments
End Sub
