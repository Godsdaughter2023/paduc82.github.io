Set up the tenant in Microsoft, do all the dns record changes but don't change the mx record until all accounts are set up in the tenant. In Outlook for android, I set the *** Email address is removed for privacy *** account as the default account. BUT any emails sent under these circumstances DO NOT appear in my outlook.com Sent Items folder. So I have no record of the sent email.

So, I also BCC *** Email address is removed for privacy ***, and this sends me the email too, and it appears in my outlook.com inbox. I added an inbox rule in outlook.com which takes email sent from *** Email address is removed for privacy ***, moves it to the Sent Items folder and marks it as read.

One other point ... on android, if I reply to an email the default reply address is again, the GUID address. The default account is apparently ignored in this case.

I have a bit more flexibility in DTO because of the availability of VBA. I have written code that automates what I have to do manually in Android. Yes, I have to use Outlook Redemption to achieve this because the SendUsingAccount property is ReadOnly. 

I also sink the Application.ItemSend event, in which I add *** Email address is removed for privacy *** as a BCC recipient. And again, the outlook.com inbox rule moves this to the outlook.com Sent Items folder. 

The only minor downside to this approach is that the emails moved to the sent folder do not show the BCC address. A BCCd email copied to the Sent folder ordinarily does show the BCC addresses. I can live with that.

bernie deitrick, excel mvp 2000-2010 say:

I noticed that, too, after I posted.  Send me an email 

bdeitrick at alum dot mit dot edu

 and I will sent you the file with other event code macros and some buttons for gameplay.

Otherwise, I will post the main code here. below.

 

Bernie

 

Sub MakeMove()
    Dim BlankCell As Range
    Dim DiagCell As Range
    Dim Board As Range
    Dim BestBlock As Range
    Dim BestWin As Range
    Dim Diagonal1 As Range
    Dim Diagonal2 As Range
    Dim CountX As Integer
    Dim CountO As Integer
    Dim PossWin As Integer
    Dim MaxWin As Integer
    Dim CountB As Integer
    Const WinningPhrase As String = "I WIN, SUCKER!!!!!"

    'Set up the playing field
    Set Board = Range("C6:E8")
    Set Diagonal1 = Range("C6,D7,E8")
    Set Diagonal2 = Range("E6,D7,C8")
    If Application.CountA(Board) = 9 Then
        Board.ClearContents
        Exit Sub
    End If

    'Look for winners for O
    For Each BlankCell In Board.SpecialCells(xlCellTypeBlanks)
        'Check blank cell for winning row or column
        If Application.CountIf(Intersect(Board, BlankCell.EntireRow), "O") = 2 Or _
           Application.CountIf(Intersect(Board, BlankCell.EntireColumn), "O") = 2 Then
            BlankCell.Value = "O"
            MsgBox WinningPhrase
            Exit Sub
        End If

        'Check blank cell's Diagonals - if on the diagonal
        If Not Intersect(Diagonal1, BlankCell) Is Nothing Then
            CountO = 0
            For Each DiagCell In Diagonal1
                If DiagCell.Value = "O" Then CountO = CountO + 1
            Next DiagCell

            If CountO = 2 Then
                BlankCell.Value = "O"
                MsgBox WinningPhrase
                Exit Sub
            End If
        End If

        If Not Intersect(Diagonal2, BlankCell) Is Nothing Then
            CountO = 0
            For Each DiagCell In Diagonal2
                If DiagCell.Value = "O" Then CountO = CountO + 1
            Next DiagCell
            If CountO = 2 Then
                BlankCell.Value = "O"
                MsgBox WinningPhrase
                Exit Sub
            End If
        End If

    Next BlankCell

    'If we've made it here, we haven't won
    'Look for imminent wins by X, and fill in the first blank that blocks a win
    For Each BlankCell In Board.SpecialCells(xlCellTypeBlanks)
        'Check blank cell's row
        If Application.CountIf(Intersect(Board, BlankCell.EntireRow), "X") = 2 Then
            BlankCell.Value = "O"
            Exit Sub
        End If
        'Check blank cell's column
        If Application.CountIf(Intersect(Board, BlankCell.EntireColumn), "X") = 2 Then
            BlankCell.Value = "O"
            Exit Sub
        End If
        'Check blank cell's diagonals - if on the diagonal
        'Check Diagonal 1
        If Not Intersect(Diagonal1, BlankCell) Is Nothing Then
            CountX = 0
            For Each DiagCell In Diagonal1
                If DiagCell.Value = "X" Then CountX = CountX + 1
            Next DiagCell
            If CountX = 2 Then
                BlankCell.Value = "O"
                Exit Sub
            End If
        End If
        'Check Diagonal 2
        If Not Intersect(Diagonal2, BlankCell) Is Nothing Then
            CountX = 0
            For Each DiagCell In Diagonal2
                If DiagCell.Value = "X" Then CountX = CountX + 1
            Next DiagCell
            If CountX = 2 Then
                BlankCell.Value = "O"
                Exit Sub
            End If
        End If
    Next BlankCell

    'If we make it to here, we haven't won, and there are no imminent wins for X, so
    'assign scores to blanks to find best move - making up the best lines that could
    'be winners for O after the next move
    MaxWin = 0
    For Each BlankCell In Board.SpecialCells(xlCellTypeBlanks)
        'Reset counters
        CountO = 0
        PossWin = 0

        If BestWin Is Nothing Then Set BestWin = BlankCell

        If Application.CountIf(Intersect(Board, BlankCell.EntireRow), "O") + _
           Application.CountIf(Intersect(Board, BlankCell.EntireRow), "") = 3 Then PossWin = PossWin + 1

        'Check blank cell's column
        If Application.CountIf(Intersect(Board, BlankCell.EntireColumn), "O") + _
           Application.CountIf(Intersect(Board, BlankCell.EntireColumn), "") = 3 Then PossWin = PossWin + 1

        'Check blank cell's Diagonals - if on the diagonal
        If Not Intersect(Diagonal1, BlankCell) Is Nothing Then
            CountO = 0
            CountB = 0
            For Each DiagCell In Diagonal1
                If DiagCell.Value = "O" Then CountO = CountO + 1
                If DiagCell.Value = "" Then CountB = CountB + 1
            Next DiagCell
            If CountO + CountB = 3 Then PossWin = PossWin + 1
        End If

        If Not Intersect(Diagonal2, BlankCell) Is Nothing Then
            CountO = 0
            CountB = 0
            For Each DiagCell In Diagonal2
                If DiagCell.Value = "O" Then CountO = CountO + 1
                If DiagCell.Value = "" Then CountB = CountB + 1
            Next DiagCell
            If CountO + CountB = 3 Then PossWin = PossWin + 1
        End If

        If PossWin > MaxWin Then
            Set BestWin = BlankCell
            MaxWin = PossWin
        End If
    Next BlankCell

    'Fill in the best move
    On Error GoTo SingleBlank
    BestWin.Value = "O"

    Exit Sub
SingleBlank:
    BlankCell.Value = "O"
End Sub
