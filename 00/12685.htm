myall_blues say: The problem with the approach you are trying is the col_index_num parameter (the third parameter) in the VLOOKUP formula. Simply making it an array formula doesn't increment that value. One solution is to use the COLUMN() formula to get the column number.
So for example this formula in B1 to get the value from the 2nd column of the array
=VLOOKUP($A1,$A$3:$E$7,COLUMN(),FALSE)
and then copy it to the right (C,D,E) for values from the 3rd, 4th and 5th columns of the array respectively.
If the formula were in column J (column 10) bu the array was still in $A$3:$E$7 then it needs to become =VLOOKUP($A1,$A$3:$E$7,COLUMN()-8,FALSE)

greg maxey say:

No.

However, you can create a custom group containing a dynamic menu using the CustomUI editor.  Then you can use the dynamic menu "GetContent" attribute to build the necessary ribbon controls dynamically.  For example the following code can be used to build a dynamic menu based on the content of a table in an Word document:

Option Explicit
Private rxIRibbonUI As IRibbonUI
Private Const NameSpace = "<menu xmlns=""http://schemas.microsoft.com/office/2006/01/customui"">" & vbNewLine

Sub rxGKMIRibbonUI_onLoad(ribbon As IRibbonUI)
Set rxIRibbonUI = ribbon
End Sub

Public Sub rxGKMGetContent(control As IRibbonControl, ByRef content)
Dim pMenuXML As String
pMenuXML = NameSpace
Select Case control.Id
  Case "rxdmnuGKMDemoDynamicMenu"
    pMenuXML = pMenuXML & BuildXMLFromTableItems(ActiveDocument.Tables(1))
  Case Else
    'For future use
End Select
pMenuXML = pMenuXML & vbNewLine & "</menu>"
pMenuXML = Replace(pMenuXML, "&", "&amp;&amp;") 'Added to prevent error entries with "&"
content = pMenuXML
End Sub
Private Sub rxGKMOnAction(control As IRibbonControl)
Select Case True
  Case IsNumeric(control.Tag)
    ActiveDocument.Tables(1).Rows(control.Tag).Range.Select
  Case Else
    rxIRibbonUI.Invalidate
End Select
End Sub
Function BuildXMLFromTableItems(ByRef oTbl As Word.Table) As String
Dim pXML As String
Dim i As Long
Dim lngCount As Long
lngCount = 0
If oTbl.Rows.Count > 999 Then
  MsgBox "Too many items to list", vbOKOnly + vbInformation, "Alert"
  BuildXMLFromTableItems = XMLBuild(ControlType:="button", _
                 Id:="rxGKMItems" & i, _
                 ImageMso:="MacroSecurity", _
                 Label:=Left(oTbl.Cell(i, 1).Range.Text, Len(oTbl.Cell(i, 1).Range.Text) - 2), _
                 Action:="rxGKMOnAction", _
                 Screentip:="Too many items", _
                 Supertip:="Too many items")
  Exit Function
End If
'Add Refresh button
pXML = pXML & XMLBuild(ControlType:="button", _
              Id:="rxGKMRefresh", _
              ImageMso:="RecurrenceEdit", _
              Label:="Refresh", _
              Action:="rxGKMOnAction", _
              Screentip:="Refresh menu", _
              Supertip:="Click this control to refresh the menu listing")
For i = 1 To oTbl.Rows.Count
  If Len(oTbl.Cell(i, 1).Range.Text) > 2 Then
    pXML = pXML & "<menuSeparator id=""rxGKMSep" & i & """/> " & vbNewLine & _
           XMLBuild(ControlType:="button", _
           Id:="rxGKMItem" & i, _
           Label:=Left(oTbl.Cell(i, 1).Range.Text, Len(oTbl.Cell(i, 1).Range.Text) - 2), _
           ImageMso:="FileNew", _
           Action:="rxGKMOnAction", _
           Tag:=i, _
           Screentip:="Go to item", _
           Supertip:="Click to select item in table")
    lngCount = lngCount + 1
  End If
Next i
'Check if any file button or dynamic menu controls were created. 'Create a "No Files" button if not
If lngCount = 0 Then
  pXML = pXML & XMLBuild(ControlType:="button", _
          Id:="rxGKMItems", _
          Label:="No items to list", _
          Screentip:="No items", _
          Supertip:="There are no items in this table to list")
End If
BuildXMLFromTableItems = pXML
End Function

Public Function XMLBuild(ByVal ControlType As String, ByVal Id As String, _
                         Optional ByVal Label As String, Optional ByVal ImageMso As String = "", _
                         Optional ByVal Image As String, Optional ByVal Action As String = "", _
                         Optional ByVal content As String = "", _
                         Optional ByVal Tag As String = "", _
                         Optional ByVal Screentip As String = "", _
                         Optional ByVal Supertip As String = "") As String
Dim pControlXML As String
pControlXML = "<" & ControlType & " " & vbNewLine & _
              "    id=""" & Id & """ " & vbNewLine
If Label <> "" Then pControlXML = pControlXML & "    label=""" & Label & """ " & vbNewLine
If ImageMso <> "" Then pControlXML = pControlXML & "    imageMso=""" & ImageMso & """ " & vbNewLine
If Image <> "" Then pControlXML = pControlXML & "    image=""" & Image & """ " & vbNewLine
If Action <> "" Then pControlXML = pControlXML & "    onAction=""" & Action & """ " & vbNewLine
If content <> "" Then pControlXML = pControlXML & "    getContent=""" & content & """ " & vbNewLine
If Tag <> "" Then pControlXML = pControlXML & "    tag=""" & Tag & """"
If Screentip <> "" Then pControlXML = pControlXML & "    screentip=""" & Screentip & """"
If Supertip <> "" Then pControlXML = pControlXML & "    supertip=""" & Supertip & """"
XMLBuild = pControlXML & "/>" & vbNewLine
End Function

macropod say:

Strictly speaking, since the Explanation columns seem to be related to the Strengths rather than to the individuals, you shouldn't need anything more than:

Strength A B C D E
Tom 21 52 32 2 5
Jane 2 60 80 10

1

This can be achieved by inserting the following formula onto a new sheet:

=OFFSET(Sheet1!$A$1,COLUMN()-1,ROW()-1)

where 'Sheet1' is the sheet name. and copying across 5 columns, then down as far as needed. The second row will be filled with the Explanation data:

Strength A B C D E
Explanation Blah blah blah Blah blah blah Blah blah blah Blah blah blah Blah blah blah
Tom 21 52 32 2 5
Jane 2 60 80 10 1

The merge could be done using the new worksheet with the setup as is, starting from the 2nd record, though I'd be inclined to replace 'Strength' with 'Name'. Indeed, one could even start from row 1 and use the first record to populate some bookmarks with the Explanation data, to make them available to all subsequent records.

Field coding in the mailmerge main document can be used to prioritise the strengths and output the corresponding Explanation. Such a field, for starting with the first record, might be coded as:

{IF{MERGESEQ}= 1 "{SET ExA {MERGEFIELD A}}{SET ExB {MERGEFIELD B}}{SET ExC {MERGEFIELD C}}{SET ExD {MERGEFIELD D}}{SET ExE {MERGEFIELD E}}" "{QUOTE{SET MaxVal {=MAX({MERGEFIELD A},{MERGEFIELD B},{MERGEFIELD C},{MERGEFIELD D},{MERGEFIELD E}) \# 0}} {IF {MERGEFIELD A}= MaxVal "A ({MERGEFIELD A}) {REF ExA}"}{IF {MERGEFIELD B}= MaxVal "B ({MERGEFIELD B}) {REF ExB}"}{IF {MERGEFIELD C}= MaxVal "C ({MERGEFIELD C}) {REF ExC}"}{IF {MERGEFIELD D}= MaxVal "D ({MERGEFIELD D}) {REF ExD}"}{IF {MERGEFIELD E}= MaxVal "E ({MERGEFIELD E}) {REF ExE}"}}"}

or:

{IF{MERGESEQ}= 1 "{SET ExA («A»}{SET ExB «B»}{SET ExC «C»}{SET ExD «D»}{SET ExE «E»}" "{QUOTE{SET MaxVal {=MAX(«A»,«B»,«C»,«D»,«E») \# 0}} {IF «A»= MaxVal "A («A») Explanation for A"}{IF «B»= MaxVal "B («B») Explanation for B"}{IF «C»= MaxVal "C («C») Explanation for C"}{IF «D»= MaxVal "D («D») Explanation for D"}{IF «E»= MaxVal "E («E») Explanation for E"}}"}

Note: The field brace pairs (i.e. '{ }') for the above example are all created in the document itself, via Ctrl-F9 (Cmd-F9 on a Mac); you can't simply type them or copy & paste them from this message. Nor is it practicable to add them via any of the standard Word dialogues. Likewise, you can't type or copy & paste the chevrons (i.e. '« »') - they're part of the actual mergefields, which you can insert from the mailmerge toolbar. The spaces represented in the field construction are all required.

No VBA required, which will be beneficial next time the updated data require mailmerging. That said, for a macro to convert the above text representations of the fields to working field codes, see the Convert Text Representations of Fields to Working Fields macro in the Mailmerge Tips and Tricks thread at:
http://www.msofficeforums.com/mail-merge/21803-mailmerge-tips-tricks.html
or:
http://windowssecrets.com/forums/showthread.php/163017-Word-Mailmerge-Tips-amp-Tricks
