Here you go:

Sub RemoveNames()
    Const c1 = "Print_Area"
    Const c2 = "Print_Titles"
    Dim i As Long
    Dim nm As Name
    For i = ActiveWorkbook.Names.Count To 1 Step -1
        Set nm = ActiveWorkbook.Names(i)
        If Right(nm.Name, Len(c1)) <> c1 And Right(nm.Name, Len(c2)) <> c2 And _
                Not TypeOf nm.Parent Is Workbook Then
            ActiveWorkbook.Names(i).Delete
        End If
    Next i
End Sub
