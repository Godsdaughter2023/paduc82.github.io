Private Sub SyncChk_Visibility_Master()

    Dim str_TotalRecords As Long, str_FilteredRecords As Long

    ' Count the records with matching continent from the table
    str_TotalRecords = DCount("Country", "tbl_Country", "Continent_ID= " & Nz(cbo_Continent))

    If str_TotalRecords > 0 Then
        ' Found records.  Count the records that also have Visible flag set to -1

        Select Case str_FilteredRecords
                Case 0 ' No records so set Master to unchecked

            Case 1 To (str_TotalRecords - 1) ' Some records so set Master to null
                Me!subfrm_Edit_Country!chk_Visibility_Master.TripleState = yes
                Me!subfrm_Edit_Country!chk_Visibility_Master.Value = Null
                Me!subfrm_Edit_Country!chk_Visibility_Master.TripleState = no
