andreas killer say:
If you want to run a macro afterwards, what should the macro do?

To find the last 6 ones from the bottom with a macro is really simple, see code below.

Andreas. 

Sub Test()
  Dim All As Range, Where As Range, This As Range, What As Range
  Dim FirstAddress As String
  
  'Refer to the cell that contains the name
  Set What = Range("C1")
  'Refer to the used cells in column C (from C11 down to the end)
  Set Where = Range("C11", Range("C" & Rows.Count).End(xlUp))
  'Find the first from the bottom
  Set This = Where.Find(What, LookIn:=xlValues, LookAt:=xlWhole, SearchDirection:=xlPrevious)
  'Found any?
  If This Is Nothing Then
    MsgBox What & " not found"
    Exit Sub
  End If
  'Remember this one
  Set All = This
  FirstAddress = This.Address
  Do
    'Find the next one upwards
    Set This = Where.FindPrevious(This)
    'Already found?
    If FirstAddress = This.Address Then Exit Do
    'Add it
    Set All = Union(All, This)
  Loop Until All.Count = 6
  'Here are the cells your are looking for:
  All.Select
End Sub
