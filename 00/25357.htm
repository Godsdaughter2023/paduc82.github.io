You need VB code for that. Right click the sheet tab, view code and paste the code below in on the right, Close VB editor and start entering numbers in a1 and b1 will keep the running sum

 

 

Private Sub Worksheet_Change(ByVal Target As Range)
If Target.Cells.Count > 1 Or IsEmpty(Target) Then Exit Sub
If Target.Address = "$A$1" Then
    If IsNumeric(Target) Then
        Application.EnableEvents = False
        Target.Offset(, 1) = Target.Offset(, 1) + Target.Value
        Application.EnableEvents = True
    End If
End If
End Sub
