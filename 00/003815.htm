gord dibben say:

You can use this macro.

Public Sub StripAll_But_NumText()
Dim rConsts As Range
Dim rCell As Range
Dim i As Long
Dim sChar As String
Dim sTemp As String
    On Error Resume Next
    Set rConsts = Selection.SpecialCells(xlCellTypeConstants)
    On Error GoTo 0
    If Not rConsts Is Nothing Then
        For Each rCell In rConsts
            With rCell
                For i = 1 To Len(.text)
                    sChar = Mid(.text, i, 1)
                    If sChar Like "[0-9a-zA-Z]" Then _
                            sTemp = sTemp & sChar
                Next i
                .Value = sTemp
            End With
            sTemp = ""
        Next rCell
    End If
End Sub

Sub Hide_Closed_Rows()
'using set column
Dim RngCol As Range
Dim i As Range
    Set RngCol = Range("H3", Range("H" & Rows.Count). _
            End(xlUp).Address)
    For Each i In RngCol
        If i.Value = "Closed" Then _
                i.EntireRow.Hidden = True
    Next i
End Sub

Hi Allan, first, thank you for answering.

As for network connectivity, yes, from all testing, the affected machines have full network connectivity; they can reach all non-restricted sites, our corporate tools can connect to corporate databases (Citrix, etc.), I can remote into the affected machines, I can access their C: drives from my machine.
