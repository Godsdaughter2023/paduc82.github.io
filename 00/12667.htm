mndef say:

How do I assign a macro to a control button in Word?

jay freedman say:
No, it isn't quite so simple in Word. Notice that the "legacy form field" tools on the Developer tab in Word don't include a button, whereas the "form controls" tools in Excel do include a button. Word simply doesn't have a control that behaves like the one in Excel.

In Word, you would need to use an ActiveX button or a MacroButton field (more on this below). An ActiveX button always activates one and only one macro, whose name is the button control's name with "_Click" tacked on the end. You can open that macro in the VBA editor by right-clicking the button and choosing View Code. You can either put your code directly inside this Click macro, or write a Call statement there to invoke another macro that you've already written.

As an alternative, you can insert a MacroButton field in a Word document. Click Insert > Quick Parts > Field to open the Field dialog. Enter the text you want to display in the document. In the list of field names, click MacroButton. In the list of macro names, click the one you want to run. Click OK. If you want to make it look like a button, you can select the field and apply a border and shading, or other formatting.

By default, a MacroButton field needs a double-click to activate it. To make it activate on a single click, add this macro to the form:

Sub AutoOpen()
    Options.ButtonFieldClicks = 1
End Sub

Sub Swap_Columns()

andreas killer say:
Just to make sure you understood what I was doing,
Your macro is much trickier than it seems at first look. Your questions are eligible and lead to the conclusion that it is much better as my solution.

I've tried around a bit and found that the constants of the clipboard format of Excel on different PCs is not the same. Therefore, it seems to be the best solution is to identify the format using its name. The macro below should work for the most situations.

During my experiments I've found that the file name of the source file can be found in the clipboard data of the HTML format:
  SourceURL:file:///Z:\File%20A.xls
But unfortunately if the file is not saved, the HMTL code did not contain the SourceURL line.

Andreas.

Option Explicit

#Const ParseFileName = False

Private Declare Function OpenClipboard Lib "user32.dll" (ByVal hwnd As Long) As Long
Private Declare Function CloseClipboard Lib "user32.dll" () As Long
Private Declare Function EnumClipboardFormats Lib "user32" (ByVal wFormat As Long) As Long
Private Declare Function GetClipboardFormatName Lib "user32" Alias "GetClipboardFormatNameA" ( _
    ByVal wFormat As Long, ByVal lpString As String, ByVal nMaxCount As Long) As Long

#If ParseFileName Then
Private Declare Function GetClipboardData Lib "user32.dll" (ByVal wFormat As Long) As Long
Private Declare Function GlobalLock Lib "kernel32.dll" (ByVal hMem As Long) As Long
Private Declare Function GlobalUnlock Lib "kernel32.dll" (ByVal hMem As Long) As Long
Private Declare Function lstrlen Lib "kernel32.dll" Alias "lstrlenA" ( _
    ByVal lpString As Long) As Long
Private Declare Function lstrcpy Lib "kernel32.dll" ( _
    ByVal lpStr1 As Any, ByVal lpStr2 As Any) As Long
#End If

Sub PasteAsLocalFormula()
  'If the clipbaord contains an Excel range, any formula is pasted unchanged, moving sheet and _
    cell references to the destination workbook.
  Dim S As String
  Dim i As Long, CF_Format As Long
  Dim SaveDisplayAlerts As Boolean, SaveScreenUpdating As Boolean
  Dim RangeInClipBoard As Boolean
#If ParseFileName Then
  Dim Handle As Long, Ptr As Long, FileName As String
#End If

  'Enumerate the clipboard formats
  If OpenClipboard(0) Then
    CF_Format = EnumClipboardFormats(0&)
    Do While CF_Format <> 0
      S = String(255, vbNullChar)
      i = GetClipboardFormatName(CF_Format, S, 255)
      S = Left(S, i)
      'If we find the format "Biff8", a range is in the clipboard
      'Note: Can be found in XL2000 up to XL2010
      RangeInClipBoard = InStr(1, S, "Biff8", vbTextCompare) > 0
     
#If ParseFileName Then
      If InStr(1, S, "HTML Format", vbTextCompare) > 0 Then
        Handle = GetClipboardData(CF_Format)
        Ptr = GlobalLock(Handle)
        S = Space$(lstrlen(ByVal Ptr))
        lstrcpy S, ByVal Ptr
        GlobalUnlock Ptr
        i = InStr(1, S, "SourceURL:file:///", vbTextCompare)
        If i > 0 Then
          i = i + 21
          FileName = Mid$(S, i, InStr(i, S, vbCrLf) - i)
          FileName = "[" & Replace(FileName, "%20", " ") & "]"
        End If
      End If
#Else
      If RangeInClipBoard Then Exit Do
#End If

      CF_Format = EnumClipboardFormats(CF_Format)
    Loop
    CloseClipboard
  End If

  If RangeInClipBoard And TypeOf Selection Is Range Then
    SaveDisplayAlerts = Application.DisplayAlerts
    SaveScreenUpdating = Application.ScreenUpdating
    'Supress errors, a referenced sheet may not exists in the destination file
    Application.DisplayAlerts = False
    Application.ScreenUpdating = False
    'Paste contents of cb as link
    ActiveSheet.Paste Link:=True
    'Name of source file with [] from first cell
    With Selection.Cells(1)
      i = InStr(1, .Formula, "[")
      S = Mid$(.Formula, i, InStr(1, .Formula, "]") - i + 1)
    End With
    'Repaste as Formulas
    Selection.PasteSpecial

    'Remove "[sourcefilename]" from formulas
    Selection.Replace What:=S, Replacement:="", LookAt:=xlPart
    Application.DisplayAlerts = SaveDisplayAlerts
    Application.ScreenUpdating = SaveScreenUpdating
  Else
    'Ignore errors, this could cause problems
    On Error Resume Next
    ActiveSheet.Paste
  End If
End Sub
