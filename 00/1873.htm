This should be close...

 

Sub Test()
'Copy information from job workbook to summary workbook
    Dim Key As String
    Dim target As Range
    Dim wbkSummary As Workbook
    Dim shtSummary As Worksheet
    Dim rngToCopy As Range
   
    Set wbkSummary = Windows("TimeReportingSummary.xlsx")
    Set shtSummary = wbkSummary.Sheets("Summary")
   
    With ThisWorkbook.Sheets("JobReport")
    Key = .Cells(2, 3).Value 'Project name
    Set rngToCopy = .Range("I8:X8") 'Cost, Hour, Role Data
    End With

   
   
    Set target = shtSummary.Columns(2).Find(What:=Key, _
                                            LookIn:=xlValues, _
                                            LookAt:=xlWhole, _
                                            MatchCase:=False) 'Find project name
    If target Is Nothing Then
        'Key not Found
        shtSummary.Cells(Rows.Count, 2).End(xlUp).Offset(1, 0).Value = Key
    Else
        target.Offset(0, 3).Resize(1, rngToCopy.Cells.Count).Value = rngToCopy.Value
    End If
End Sub
