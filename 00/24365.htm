suzanne s. barnhill say:

It's not as obvious as in previous versions, where you could change the printer and then the Cancel button in the Print dialog changed to Close, allowing you to change settings and close it without printing, but you can still do it:

  1. Press Ctrl+P or use File | Print to open the Print dialog (for want of a better word).
  2. Select a different printer and click Printer Properties to select any desired printer settings (such as duplexing).
  3. Reselect the Home tab to exit the "backstage" view.
  4. If you have the QuickPrint icon on your QAT, hover over it and you'll see that it reflects the new printer selection.

Note,  however, that Quick Print does always use the default Print settings, printing a single copy of the entire document, one page per sheet, etc. You will still need to open the Print tab if you want to print multiple copies, specific pages, multiple pages per sheet, or any of the other specializations in that dialog.

confusedatwork 2 say:

Updating Excel only certain rows from table list

I have a summary list of all transactions in a worksheet.  There is lets say 10 different customers, all have unique customer numbers that I can filter by.  I need to create ten different excel sheets and save them in separate folders, but the other nine cannot see the rest of the transactions.  Then they view the excel sheet through a FTP Site and just pull up that file corresponding to their business.

What is the easiest way to pull this data into the 10 separate spreadsheets automatically from the main list?

bernie deitrick say:

You can use a macro - this assumes that the key values are in column B, and that you have headers in the first row. The macro places dated files into a subfolder of the folder containing the data file.

Sub ExportDatabaseToSeparateFiles()
    Dim rngC As Range
    Dim i As Integer
    Dim shtExtract As Worksheet
    Dim shtData As Worksheet
    Dim strName As String
    Dim rngKeyValues As Range
    Dim strKeyCol As String
    Dim iKField As Integer
    Dim strPath As String
    Dim strFName As String
    
    Application.ScreenUpdating = False

    ActiveSheet.Copy
    ActiveSheet.Columns.Hidden = False
    
    ActiveSheet.Cells.RemoveSubtotal

    With ActiveSheet.Sort
        .SortFields.Clear
        .SortFields.Add Key:= _
        Range("B2"), SortOn:=xlSortOnValues, Order:=xlAscending, DataOption _
        :=xlSortNormal
   
        .SetRange ActiveSheet.UsedRange
        .Header = xlYes
        .MatchCase = False
        .Orientation = xlTopToBottom
        .Apply
    End With

    'Export is based on the value in the desired column

    strPath = Application.ThisWorkbook.Path & "\"
    strFName = "Export Folder "

    If Dir(strPath & strFName & Format(Date, "yyyy-mm-dd"), vbDirectory) = "" Then
        MkDir strPath & strFName & Format(Date, "yyyy-mm-dd")
    End If

    strPath = strPath & strFName & Format(Date, "yyyy-mm-dd") & "\"

    Set shtData = Application.ActiveSheet

    'strKeyCol = InputBox("What column letter to use as key?") 'Use this if the column is variable
    strKeyCol = "B"  'Change to column with key values

    Set rngKeyValues = Intersect(shtData.UsedRange, shtData.Range(strKeyCol & ":" & strKeyCol))

    Set rngKeyValues = rngKeyValues.SpecialCells(xlCellTypeConstants)    'Use for constants
    'Set rngKeyValues = rngKeyValues.SpecialCells(xlCellTypeFormulas) 'Use if formulas

    Set rngKeyValues = rngKeyValues.Offset(1, 0).Resize(rngKeyValues.Rows.Count - 1, 1)    'Assumed one row of header values
    iKField = rngKeyValues.Column

    For Each rngC In rngKeyValues
    If rngC.Value <> "" Then
        On Error GoTo NoSheet
        strName = Application.ActiveWorkbook.Worksheets(Left(rngC.Value, 31)).Name
        GoTo SheetExists:
NoSheet:
        Set shtExtract = Application.ActiveWorkbook.Worksheets.Add(before:=shtData)
        shtExtract.Name = Left(rngC.Value, 31)
        With rngC.Parent.UsedRange
            .AutoFilter Field:=iKField, Criteria1:=rngC.Value
            .SpecialCells(xlCellTypeVisible).Copy _
                    shtExtract.Range("A1")
            For i = 1 To .Columns.Count
                shtExtract.Columns(i).ColumnWidth = .Columns(i).ColumnWidth
            Next i
            shtExtract.Buttons.Delete
            .AutoFilter
        End With
        Resume
SheetExists:
    End If
    Next rngC

    On Error Resume Next
    For Each shtExtract In Application.ActiveWorkbook.Worksheets
        If shtExtract.Name <> shtData.Name Then
            shtExtract.Move
            Application.ActiveWorkbook.SaveAs strPath & Application.ActiveWorkbook.ActiveSheet.Name & Format(Date, "_yymmdd") & ".xlsx"
            Application.ActiveWorkbook.Close False
        End If
    Next shtExtract
    ActiveWorkbook.Close False
    
    Application.ScreenUpdating = True
End Sub
