I have sent you a Macro Enabled copy of the Workbook that contains a module with the following code in it:

Sub Transpose()
Dim i As Long, j As Long, k As Long, l As Long
Dim arrData As Variant
Dim lngID As Long
arrData = Sheets("Intv").Range("A1").CurrentRegion.Value
For i = 2 To UBound(arrData, 1)
    lngID = arrData(i, 1)
    With Sheets("All Schls").Range("A1")
        For j = 1 To .CurrentRegion.Rows.Count - 1
            If .Offset(j, 0) = lngID Then
                l = 0
Repeat:
                For k = 0 To 3
                    .Offset(j, 22 + k + l) = arrData(i, 6 + k)
                Next k
                If i < UBound(arrData, 1) Then
                    If arrData(i + 1, 1) = lngID Then
                        i = i + 1
                        l = l + 4
                        GoTo Repeat
                    End If
                End If
                Exit For
            End If
        Next j
    End With
Next i
End Sub
