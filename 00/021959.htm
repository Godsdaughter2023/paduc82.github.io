nspiv914 say: Yay!!!! Well I found a solution to this online....


Use this macro ....


Sub deletesound()

    Dim oSl As Slide
    Dim oSh As Shape
    Dim x As Long

    For Each oSl In ActivePresentation.Slides
        For x = oSl.Shapes.Count To 1 Step -1
            Set oSh = oSl.Shapes(x)
            If oSh.Type = msoMedia Then
                If oSh.MediaType = ppMediaTypeSound Then
                    oSh.Delete
                End If
            End If
        Next
    Next

End Sub
