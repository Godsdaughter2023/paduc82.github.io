Sub Reordena(ByRef divergentes() As String)

    Dim i As Integer
    Dim j As Integer
    Dim identificador_i As String
    Dim identificador_j As String
    Dim mudou As Boolean
    Dim parse As Integer
   
    ' Sai se não há nada para ser feito (quando está vazio ou quando só há um elemento)
    If LBound(divergentes) >= UBound(divergentes) - 1 Then Exit Sub
   
    For parse = 1 To 20 ' Permite até 21 reordenamentos, quando necessário pelo grau de dependências
        mudou = False
        ' Desconsidera o primeiro elemento da lista pois ele nunca é preenchido
        ' pela função ComparaValores
        For i = LBound(divergentes) + 1 To UBound(divergentes) - 1
            identificador_i = divergentes(i)
            For j = i + 1 To UBound(divergentes)
                identificador_j = divergentes(j)
                If ComparaDivergencias(identificador_i, identificador_j) = 1 Then
                    Swap divergentes, i, j
                    identificador_i = identificador_j
                    mudou = True
                End If
            Next
        Next
        If Not mudou Then Exit For
    Next
   
End Sub

Function ComparaDivergencias(ByVal div_id1 As String, ByVal div_id2 As String) As Integer

    ' Compara uma divergência com outra, passando os identificadores de cada uma delas
    ' Caso div_id1 seja dependente de div_id2, retorna +1 (div_id2 deve vir antes de div_id1)
    ' Caso div_id2 seja dependente de div_id1, retorna -1 (div_id1 deve vir antes de div_id2)
    ' Caso contrário, retorna 0 (é indiferente)
   
    Dim div1_depende_de As String
    Dim div2_depende_de As String
    Dim dependencias() As String
    Dim i As Integer
    Dim resultado As Integer
    resultado = 0
   
    div1_depende_de = PegaDependentes(div_id1)
    div2_depende_de = PegaDependentes(div_id2)
   
    If div1_depende_de <> "" And div2_depende_de <> "" Then
        resultado = 0
        ' Verifica se '1' depende de '2'
        dependencias = Split(div1_depende_de, ",")
        For i = LBound(dependencias) To UBound(dependencias)
            If div_id2 = Trim(dependencias(i)) Then
                resultado = 1
                Exit For
            End If
        Next
        If resultado = 0 Then
        ' Verifica se '2' depende de '1'
            dependencias = Split(div2_depende_de, ",")
            For i = LBound(dependencias) To UBound(dependencias)
                If div_id1 = Trim(dependencias(i)) Then
                    resultado = -1
                    Exit For
                End If
            Next
        End If
    End If
   
    ComparaDivergencias = resultado
   
End Function

Function ProximaLinha(ByVal planilha As String, ByVal linha As Integer)

End Function

Function PegaMensagem(ByVal identificador As String) As String
    Dim row As Integer
    Dim msg As String
    Dim texto As String
    row = 1

Function FormataMensagem(ByVal mensagem As String) As String

End Function

You can't sync POP3 pst files to outlook.com  - you need to move the calendar and contacts into the outlook.com pst then they will sync.  See http://www.slipstick.com/outlook/olc-hotmail/sync-calendar-contacts-using-outlook-com/ for the steps.

 

Many people think that Outlook will magically sync directly with outlook on their other computers, either alone or once they add an Outlook.com account. That isn't how it works - the outlook.com syncs to outlook but folders within outlook don't sync to outlook.com. 

 

Not supported: outlook pst <==> outlook.com <==> pst on another computer

Supported: outlook.com account in outlook <==> outlook.com <==> outlook.com account on another computer
