rich007a say:

I think you need just one line of code to do this.  The following:

    Sheets(Array("Sheet1", "Sheet2")).PrintOut From:=2, To:=2, Copies:=1

will print page 2 of Sheet1 and Sheet2.

If you want to you can use variables for the various parameters:

Sub PrintPagesOnSheets(strShtA As String, Optional strShtB As String = "", _
    Optional strShtC As String = "", Optional strShtD As String = "", _
    Optional lnFrom As Long = 1, Optional lnTo As Long = 0, Optional lnCopies As Long = 1)
   
     ' Always specifiy sheets from shtA in order without gaps, i.e.:
     ' sthrShtA:="Sheet1", strShtB:="Sheet3"    is ok
     ' sthrShtA:="Sheet1", strShtC:="Sheet3"    is NOT ok
   
    Dim vShtArr As Variant
   
     'Note default for lnTo is set to equal lnFrom if it is not specified.
    If lnTo = 0 Then lnTo = lnFrom
   
    If strShtB = "" Then
        vShtArr = strShtA
    ElseIf strShtC = "" Then
        vShtArr = Array(strShtA, strShtB)
    ElseIf strShtD = "" Then
        vShtArr = Array(strShtA, strShtB, strShtC)
    Else
        vShtArr = Array(strShtA, strShtB, strShtC, strShtD)
    End If
   
    Sheets(vShtArr).PrintOut From:=lnFrom, To:=lnTo, Copies:=lnCopies
End Sub

Sub ThisPrintsPage2OnSheets1And2()
    Call PrintPagesOnSheets(strShtA:="Sheet1", strShtB:="Sheet2", lnFrom:=2, lnCopies:=1)
End Sub

Hope that helps.

gleam2 say: Here is another solution which asks the user which page he or she wants to print and does not require the programmer or user to enter the sheet names.

Option Explicit
Sub PrintPage()
' ask user which page to print and print this page for all selected sheets
Dim PageNo As Integer, N1 As Integer
Dim Arr() As String
' ask user which page to print
    On Error Resume Next
    Application.DisplayAlerts = False
    PageNo = InputBox("Please enter page Number to print for all selected sheets")
    On Error GoTo 0
    Application.DisplayAlerts = True
    If PageNo < 1 Then Exit Sub ' Cancel hit or negative number entered
' Build up array of selected sheets and print required page on each sheet
    With ActiveWindow.SelectedSheets
        ReDim Arr(1 To .Count)
        For N1 = 1 To .Count
            Arr(N1) = .Item(N1).Name
' expression.PrintOut(From Page , To Page, Copies, Preview, ActivePrinter,
 '  PrintToFile, Collate, PrToFileName, IgnorePrintAreas)
            Sheets(Arr(N1)).PrintOut PageNo, PageNo
        Next N1
    End With
' Now reselect sheets originally selected
    ActiveWorkbook.Sheets(Arr).Select
End Sub

I think, between Gleam2 and myself, we established that that code will not work, because despite:

    Sheets(vShtArr).PrintOut From:=lnFrom, To:=lnTo, Copies:=lnCopies

being the same syntax to what is recorded by the macro recorder when you print a selection of sheets, it only prints out the first sheet in the group of sheets.

So instead you should use the code below, where we create a Sheets collection (selSheets) and loop through each Worksheet in that collection and print the desired page:

 

Sub PrintPage()
' Ask user which page to print and print this page for all selected sheets
'
    Dim PageNo As Variant, N1 As Integer
    Dim Arr() As String
    Dim ws As Worksheet
    Dim vActiveSht As Variant 'ActiveSheet could be Worksheet or Chart(sheet)
    Dim selShts As Sheets
   
    Application.ScreenUpdating = False
   
GetPageNo:
    PageNo = InputBox("Please enter page Number to print for all selected sheets" _
        , Default:=1)
    If Not IsNumeric(PageNo) Then   'Ensure user entered a number
        If MsgBox("Please enter a number or press Cancel to Exit.", _
            vbOKCancel + vbExclamation) = vbCancel Then Exit Sub
        GoTo GetPageNo
    End If
    PageNo = Int(PageNo)            'Make the numeric value entered an integer
    If PageNo < 1 Then              'Zero or negative number entered
        If MsgBox("Please enter a whole number greater than zero or press Cancel to Exit.", _
            vbOKCancel + vbExclamation) = vbCancel Then Exit Sub
        GoTo GetPageNo
    End If
   
    Set selShts = ActiveWorkbook.Sheets(Array("Civics", "Reading", "Writing", "Sample")) 'Worksheets to be printed
    Set vActiveSht = ActiveSheet    'Store the active sheet
   
    For Each ws In selShts          'Loop through all sheets to be printed
        ws.PrintOut PageNo, PageNo  'Print required page of sheet
    Next ws
   
    vActiveSht.Activate             'Re-activate the previously active sheet
    Application.ScreenUpdating = True
End Sub

steve rindsberg say:

Lucina or Lucida? 

If the computer they're viewing the presentation on doesn't have the needed font installed, Windows will substitute a different font. 

Lucida Calligraphy is edit-only embeddable, meaning that if you choose to embed it when you save your presentation, it'll be available at the other end while the presentation is open; that should prevent any font substitution problems.

In the Save As dialog box, click Tools, choose Save Options.

Put a check next to Embed fonts in the file.

If you simply want them to be able to view the file, choose "Embed only the characters used ..."

If you want them to be able to edit the text in the file, choose "Embed all characters ...".  The file will larger as a result.
