This worked for me in Outlook2007:

Sub test()
    Dim OL As Object
    Dim EmailItem As Object

    Set OL = CreateObject("Outlook.Application")
    Set EmailItem = OL.CreateItem(0)

    EmailItem.HTMLBody = "<html><p>This is my message body.</p>" & _
        "<p>This is my sig, and this is a picture in my signature:</p>" & _
        "<img src=C:\sigpic.png height=20 width=50>" & _
        "<p>This is the rest of my sig</p>"
   
    EmailItem.Display
End Sub
