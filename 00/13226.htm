Dim rngIndex As Range
Dim indexentry As Range

Dim arrnumbers As Variant
Dim strEntry As String
Set rngIndex = Selection.Range
For i = 1 To rngIndex.Paragraphs.Count
    Set indexentry = rngIndex.Paragraphs(i).Range
    indexentry.End = indexentry.End - 1
    indexentry.Text = Replace(indexentry.Text, ", ", "|")
    arrentry = Split(indexentry.Text, "|")
    strEntry = ""
    For j = LBound(arrentry) + 1 To UBound(arrentry)
        strEntry = strEntry & "|" & arrentry(j)
    Next j
    strEntry = Mid(strEntry, 2)
    arrnumbers = Split(strEntry, "|")
    For m = LBound(arrnumbers) To UBound(arrnumbers) - 1
        For n = m + 1 To UBound(arrnumbers)
            If arrnumbers(m) > arrnumbers(n) Then
                temp = arrnumbers(n)
                arrnumbers(n) = arrnumbers(m)
                arrnumbers(m) = temp
            End If
        Next n
    Next m
    strEntry = arrentry(LBound(arrentry))
    For j = LBound(arrnumbers) To UBound(arrnumbers)
        strEntry = strEntry & ", " & arrnumbers(j)
    Next j
    rngIndex.Paragraphs(i).Range.Text = strEntry & vbCr
Next i
