a. user say:

Well, I was nearly through the code I was experimenting with when I had to break off, but you can try the following in Word 2007 if you like. It makes a number of assumptions which may not be true (e.g. you may have protected your forms with a password, but this assumes there is no password) but it is obviously up to you to decide whether it is actually better for you to redevelop from scratch or attempt a "quick fix" which might not turn out to be quick.

Sub replaceContentControls()
' A text we'll use for the top value
Const strTopText As String = "[Choose a value]"
Dim cc As Word.ContentControl
Dim fld As Word.FormField
Dim i As Integer
Dim intControlCount As Integer
Dim intEntryCount As Integer
Dim rng As Word.Range
Dim strEntryTexts() As String

With ActiveDocument
  On Error Resume Next
  .Unprotect
  Err.Clear
  On Error GoTo 0
  For intControlCount = .ContentControls.Count To 1 Step -1
    ' copy each cc
    Set cc = .ContentControls(intControlCount)
    If cc.Type = wdContentControlDropdownList Then
      Set rng = cc.Range
      intEntryCount = cc.DropdownListEntries.Count
      ReDim strEntryTexts(intEntryCount)
      For i = 1 To intEntryCount
        strEntryTexts(i) = cc.DropdownListEntries(i).Value
        If strEntryTexts(i) = "" Then strEntryTexts(i) = strTopText
      Next
      cc.Delete
      Set cc = Nothing
      Set fld = rng.FormFields.Add(rng, wdFieldFormDropDown)
      For i = 1 To intEntryCount
        fld.DropDown.ListEntries.Add strEntryTexts(i), i
      Next
      Set fld = Nothing
      Set rng = Nothing
    End If
  Next
  .Protect wdAllowOnlyFormFields
End With
End Sub

graham mayor say:

As you have gathered protected forms and free editing are uneasy bedfellows. Where free editing is required, protected forms are not usually the best solution, and userforms would make better sense as a means of collecting and placing data - For the basics, see Word MVP FAQ - Userforms
http://word.mvps.org/FAQs/Userforms.htm

for a more in depth explanation, see
http://gregmaxey.mvps.org/Create_and_employ_a_UserForm.htm

However you could use macros run from toolbar buttons to insert and remove comments in an unprotected section, which would unlock and re-lock the form - something like

Sub Add_A_Comment()
Dim bProtected As Boolean
Dim strComment As String
'Unprotect the file
If ActiveDocument.ProtectionType <> wdNoProtection Then
    bProtected = True
    ActiveDocument.Unprotect Password:=""
End If
strComment = InputBox("Enter comment text", "Comment")
    Selection.Comments.Add Range:=Selection.Range
    Selection.TypeText Text:=strComment
    ActiveWindow.ActivePane.Close
If bProtected = True Then
    ActiveDocument.Protect _
    Type:=wdAllowOnlyFormFields, _
    NoReset:=True, _
    Password:=""
End If
End Sub

Sub Remove_A_Comment()
Dim bProtected As Boolean
Dim oRng As Range
'Unprotect the file
On Error GoTo Protect
If ActiveDocument.ProtectionType <> wdNoProtection Then
    bProtected = True
    ActiveDocument.Unprotect Password:=""
End If
Set oRng = Selection.Range
oRng.Start = oRng.Words(1).Start
oRng.End = oRng.Words(1).End
oRng.Comments(1).Delete
Protect:
If bProtected = True Then
    ActiveDocument.Protect _
    Type:=wdAllowOnlyFormFields, _
    NoReset:=True, _
    Password:=""
End If
End Sub
