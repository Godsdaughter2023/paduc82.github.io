Sub update_PortExcel_ExistingPO()
Dim pelRow As Long, pplRow As Long
Dim pelCol As Long, pplCol As Long
Dim PEws As Worksheet, PPws As Worksheet
Dim projCol As String, exstPOCol As String, noradCol As String
Dim fillRange As String, tblRange As String, rngProj As String, rnglookup As String
Dim c As Range

Set tblRange = .Range(rnglookup)

amit tandon say:

Refer below image:

Enter below formula as an array formula (CTRL+SHIFT+ENTER) in cell Q2 and copy down:

=IF($P2="","",SUM(IF(FREQUENCY(IF($L$2:$L$100=$P2,IF($B$2:$B$100<>"",MATCH($B$2:$B$100,$B$2:$B$100&"",0))),ROW($B$2:$B$100)-ROW($B$2)+1),1)))

In case you wish to get unique page ids in column P, & also sorted in ascending order, enter below formula as an array formula (CTRL+SHIFT+ENTER) in cell P2 (P1 should be blank) & copy down:

=IFERROR(INDEX($L$2:$L$100, MATCH(MIN(IF(COUNTIF(P$1:P1, $L$2:$L$100&"")=0, 1, MAX((COUNTIF($L$2:$L$100, "<"&$L$2:$L$100)+1)*2))*(COUNTIF($L$2:$L$100, "<"&$L$2:$L$100)+1)), COUNTIF($L$2:$L$100, "<"&$L$2:$L$100)+1, 0)), "")
