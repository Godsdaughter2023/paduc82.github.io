john mcghie say:

Hi Mandy:

I think the news is all bad...

If you never "named" the document, nothing was ever saved to disk.  The document existed only in the computer memory.

When you restarted Word, it cleaned the memory out: the document was gone forever at that point.

Autorecover depends on being able to read the saved copy to work: in this case, there wasn't one.  Similarly, "Always create backup" works only if there have been at least TWO saves.

And TimeMachine saves only what is written to disk.

I suggest you get into the habit (it needs to become a habit...) of assigning a file name and saving a document BEFORE you type the first character.  If you do, then the various backup mechanisms can swing into action.

Until you save, there is nothing to be found.

Sorry

Sub MoveDashboardSheets()
    Dim strPath As String
    Dim strFile As String
    Dim strName As String
    Dim lngPos As Long
    Dim wbkS As Workbook
    Dim wbkT As Workbook
    Dim wshS As Worksheet
    Dim wshT As Worksheet
    Dim colFiles As New Collection
    Dim varFile As Variant
    ' Prompt user for folder path
    With Application.FileDialog(4) ' msoFileDialogFolderPicker
        If .Show Then
            strPath = .SelectedItems(1)
        Else
            MsgBox "No folder selected!", vbExclamation
            Exit Sub
        End If
    End With
    ' Make sure path ends in \
    If Right(strPath, 1) <> "\" Then
        strPath = strPath & "\"
    End If
    ' Reduce screen flicker
    Application.ScreenUpdating = False
    Application.DisplayAlerts = False
    ' Create target workbook with one sheet
    Set wbkT = Workbooks.Add(Template:=xlWBATWorksheet)
    Set wshT = wbkT.Worksheets(1)
    ' Turn on error handling
    On Error GoTo ErrHandler
    ' Loop through the files
    strFile = Dir(strPath & "*.xls*")
    Do While strFile <> ""
        ' Add filename to collection
        colFiles.Add strFile
        ' Get the next filename
        strFile = Dir
    Loop
    For Each varFile In colFiles
        ' Open workbook
        Set wbkS = Workbooks.Open(Filename:=strPath & varFile)
        ' Reference to Dashboard sheet
        Set wshS = wbkS.Worksheets("Dashboard")
        ' Copy the sheet to the new workbook
        wshS.Copy After:=wshT
        ' Reference to new sheet
        Set wshT = wshT.Next
        ' Get position of "." in filename
        lngPos = InStrRev(varFile, ".")
        ' Get part before the "."
        strName = Left(varFile, lngPos - 1)
        strName = Replace(strName, "[", "")
        strName = Replace(strName, "]", "")
        strName = Left(strName, 19)
        strName = strName & " - Dashboard"
        ' Name the new sheet
        wshT.Name = strName
        ' Also copy the sheet to a separate workbook
        wshS.Copy
        ' Save the separate workbook
        ActiveWorkbook.SaveAs Filename:=strPath & strName & ".xlsx", _
            FileFormat:=xlOpenXMLWorkbook
        ' And close it
        ActiveWorkbook.Close
        ' Close the workbook
        wbkS.Close
    Next varFile
    ' Delete the first (empty) sheet of the new workbook
    wbkT.Worksheets(1).Delete
ExitHandler:
    Application.DisplayAlerts = True
    Application.ScreenUpdating = True
    Exit Sub
ErrHandler:
    MsgBox Err.Description, vbExclamation
    Resume ExitHandler
End Sub
