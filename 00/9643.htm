Sub macro_01()

Const cFirst As String = "X" '<<< data, first column
Const cLast As String = "AF" '<<< data, last column
Const N As Long = 60 '<< page break after 60 rows, change
Const rFirst As Long = 32 '<< data headers in row 32, change
Dim L As Long, r As Long, i As Long, x As Long
x = Cells(1, cLast).Column - Cells(1, cFirst).Column + 1
Dim ws As Worksheet
Set ws = Sheets("Sheet1") '<<< data on sheet1, change
L = ws.Cells(Rows.Count, cFirst).End(xlUp).Row
Application.ScreenUpdating = False
Sheets.Add
With ActiveSheet.PageSetup
.PrintArea = Cells(rFirst, cFirst).Resize(N + 2, x).Address
.LeftMargin = Application.InchesToPoints(0.5)
.RightMargin = Application.InchesToPoints(0.5)
.TopMargin = Application.InchesToPoints(0.5)
.BottomMargin = Application.InchesToPoints(0.5)
.HeaderMargin = Application.InchesToPoints(0.2)
.FooterMargin = Application.InchesToPoints(0.2)
.CenterHorizontally = True
.CenterVertically = False
.PaperSize = xlPaperA4
.Orientation = xlPortrait 'xlLandscape
.Zoom = False
.FitToPagesWide = 1
.FitToPagesTall = 1
End With

For i = rFirst + 1 To L Step N
ws.Cells(rFirst, cFirst).Resize(, x).Copy Cells(rFirst, cFirst)
ws.Cells(rFirst, cFirst).Resize(, x).Copy
Cells(rFirst, cFirst).PasteSpecial xlPasteColumnWidths
Rows(rFirst).HorizontalAlignment = xlLeft
Application.CutCopyMode = False
ws.Cells(i, cFirst).Resize(N, x).Copy Cells(rFirst + 1, cFirst)
r = Cells(Rows.Count, cFirst).End(xlUp).Row + 1
ws.Cells(rFirst, cFirst).Resize(, x).Copy Cells(r, cFirst)
Rows(r).HorizontalAlignment = xlLeft
ActiveSheet.PrintPreview ' PrintOut
ActiveSheet.Cells.Clear
Next
Application.DisplayAlerts = False
ActiveSheet.Delete
Application.DisplayAlerts = True
Application.ScreenUpdating = True
End Sub
