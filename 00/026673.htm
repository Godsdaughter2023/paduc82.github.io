try this vba macro,

Sub FolderPicker_MergeCSVFiles_01()
'Feb 13, 2015
Dim mywb As Workbook
Set mywb = ThisWorkbook
With Application.FileDialog(msoFileDialogFolderPicker)
If .Show = True Then
Sheets.Add
Dim shName As String
shName = mywb.ActiveSheet.Name
Dim sPath As String, sFile As String
sPath = .SelectedItems(1) & "\"
sFile = Dir(sPath & "*.csv")
Dim t As Long, r As Long, c As Long
Dim s As String
t = 1
Application.ScreenUpdating = False
Do Until sFile = ""
Set wb = Workbooks.Open(sPath & sFile, xlDelimited, local:=True)
r = Sheets(1).Cells.Find(what:="*", SearchOrder:=xlByRows, SearchDirection:=xlPrevious).Row
c = Sheets(1).Cells.Find(what:="*", SearchOrder:=xlByColumns, SearchDirection:=xlPrevious).Column
Sheets(1).Range("A1").Resize(r, c).Copy
With mywb.Sheets(shName)
.Cells(t, 3).PasteSpecial xlPasteValues
.Cells(t, 3).PasteSpecial xlPasteFormats
Application.CutCopyMode = False
s = Replace(wb.Name, ".CSV", "")
.Cells(t, 1).Resize(r).Value = s
.Cells(t, 2).Resize(r).Value = Sheets(1).Name
t = .Cells.Find(what:="*", SearchOrder:=xlByRows, SearchDirection:=xlPrevious).Row + 1
End With
wb.Close False
sFile = Dir()
Loop
With ActiveSheet.UsedRange
.Font.Size = 11
.RowHeight = 15
.VerticalAlignment = xlCenter
.HorizontalAlignment = xlLeft
.EntireColumn.AutoFit
End With
Application.ScreenUpdating = True
End If
End With
End Sub
