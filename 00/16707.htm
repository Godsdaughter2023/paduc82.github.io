Private strFileName As String
Private strFilePath As String
Private collFileName As New Collection
Private strSearchSubFolder As VbTriState

'This is your main sub
Public Sub fnFileDeatils()
Dim i

Private Function fnFindFile(ByVal sFol As String, sFile As String, Optional nDirs As Long) As Currency
Dim tFld As Object, FileName As String
Dim fld, fso As Object
Dim file As Object

   On Error GoTo Catch
   
   Set fso = CreateObject("Scripting.FileSystemObject")
   Set fld = fso.GetFolder(sFol)
   FileName = Dir(fso.BuildPath(fld.Path, sFile), vbNormal Or vbHidden Or vbSystem Or vbReadOnly)
    
    On Error GoTo Catch1
    For Each file In fld.Files
        fnFindFile = fnFindFile + FileLen(fso.BuildPath(fld.Path, FileName))
        If Len(FileName) <> 0 And file = fld.Path & "\" & FileName Then
            collFileName.Add Item:=file
            FileName = Dir()  ' Get next file
            DoEvents
        End If
    Next file

    If strSearchSubFolder = vbTrue Then
         nDirs = nDirs + 1
         If fld.SubFolders.Count > 0 Then
            For Each tFld In fld.SubFolders
               DoEvents
               fnFindFile = fnFindFile + fnFindFile(tFld.Path, sFile, nDirs)
            Next
         End If
    End If
    
    Set fso = Nothing
    Exit Function
    
Catch:
        If Err.Number = 76 Then
            MsgBox Err.Description
            GoTo Catch1
        End If
        FileName = ""
        Resume Next
Catch1:
    FileName = ""

The 32MB limit pertains to text only, exclusive of images or other objects. The likelihood of a Word document file getting to be anywhere near that size is unrealistically remote -- the document would be in excess of 4,000 pages [US Letter, standard margins & fonts] or roughly 3,585,280 words. That suggests 1 or both of the following:

  • The documents are seriously corrupt
  • The documents have been infected with a virus which has inflated their size

Without any details about the source or nature of the documents or any other insights on how they've been handled it's difficult to offer specific suggestions. If you don't already have one, though, I'd recommend that you get a good antivirus app, such as Sophos Anti-Virus
