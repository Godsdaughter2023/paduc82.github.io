You should be able to do this when starting a new presentation or possibly when editing existing ones:

In the PowerPoint options dialog box, click Language on the left

OK, I think I see what's happening.  I was checking this in PPT 2010, you're using PPT 2013 and they have changed this aspect of the program.  There's an option in Advanced called "Automatically switch keyboard to match language of surrounding text".  While I don't think this will solve the problem, it's worth investigating whether changing it will help.

Function ExtractData(ByVal text As String, ByVal pattern As String, ByVal no As Integer) As String

    Dim result As String

    Dim allMatches As Object

    Dim re As Object

    Set re = CreateObject("vbscript.regexp")

    

    re.pattern = pattern

    re.Global = True

    re.IgnoreCase = True

    Set allMatches = re.Execute(text)

    

    If allMatches.Count <> 0 Then

        result = allMatches.item(0).SubMatches.item(no)

    End If

    

    ExtractData = result

End Function

Function strip(ByVal str As String) As String

    Dim re As Object, result As String

    Set re = CreateObject("vbscript.regexp")

    re.MultiLine = True

    re.Global = True

    re.pattern = "^[\s\xA0]+|[\s\xA0]+$"

    strip = re.Replace(str, "")

End Function
