donleverton say:

Hi Terry.

I do something *somewhat* similar ... in that I display a subform with a set of records that meet certain criteria. The criteria sort of flags the "suspected" records for me, and the "Selected" (Yes/No) field allows me further control over which records I want to actually delete.

When the form opens, code runs that automatically resets all records to "No"

Here is the code for this form:

======================

 

Option Compare Database

Private Sub Form_Open(Cancel As Integer)
'Make sure that the selections are completely clear before we begin.
CurrentDb.Execute "UPDATE tblPartsInventory SET tblPartsInventory.Selected = False;", dbFailOnError

End Sub
Private Sub cmdSelectAll_Click()

Dim MySQL As String

MySQL = ""
MySQL = MySQL & "UPDATE DISTINCTROW (tblPartsInventory "
MySQL = MySQL & "LEFT JOIN tblPartsSold ON tblPartsInventory.PartID = tblPartsSold.PartID) "
MySQL = MySQL & "LEFT JOIN tblPartsRecd ON tblPartsInventory.PartID = tblPartsRecd.PartID "
MySQL = MySQL & "SET tblPartsInventory.Selected = True "
MySQL = MySQL & "WHERE (((tblPartsInventory.QtyOnHand)<1) "
MySQL = MySQL & "AND ((tblPartsSold.PartID) Is Null) "
MySQL = MySQL & "AND ((tblPartsRecd.PartID) Is Null));"

'Debug.Print MySQL
CurrentDb.Execute MySQL, dbFailOnError
Me!sbfNoPartsSales.Requery
End Sub

Private Sub cmdClearSelection_Click()
Dim MySQL As String

MySQL = ""
MySQL = MySQL & "UPDATE DISTINCTROW (tblPartsInventory "
MySQL = MySQL & "LEFT JOIN tblPartsSold ON tblPartsInventory.PartID = tblPartsSold.PartID) "
MySQL = MySQL & "LEFT JOIN tblPartsRecd ON tblPartsInventory.PartID = tblPartsRecd.PartID "
MySQL = MySQL & "SET tblPartsInventory.Selected = False "
MySQL = MySQL & "WHERE (((tblPartsInventory.QtyOnHand)<1) "
MySQL = MySQL & "AND ((tblPartsSold.PartID) Is Null) "
MySQL = MySQL & "AND ((tblPartsRecd.PartID) Is Null));"

CurrentDb.Execute MySQL, dbFailOnError
Me!sbfNoPartsSales.Requery

End Sub

Private Sub cmdDeleteSelected_Click()

Dim lngSelected As Variant
lngSelected = DCount("Selected", "tblPartsInventory", "Selected = True")

Dim Msg As String
Dim CR As String
CR = vbCrLf

If lngSelected > 0 Then

    Msg = ""
    Msg = Msg & "You currently have "
    Msg = Msg & lngSelected
    Msg = Msg & " records selected for deletion ..." & CR & CR
    Msg = Msg & "Press 'OK' to proceed with the deletion, " & CR
    Msg = Msg & "or press 'Cancel' to re-consider."

        If MsgBox(Msg, vbOKCancel, "Please CONFIRM!") = vbOK Then

            Dim MySQL As String

            MySQL = ""
            MySQL = MySQL & "DELETE tblPartsInventory.*, tblPartsInventory.Selected "
            MySQL = MySQL & "FROM tblPartsInventory "
            MySQL = MySQL & "WHERE (((tblPartsInventory.Selected)=True));"
Debug.Print MySQL

            CurrentDb.Execute MySQL, dbFailOnError
            Me!sbfNoPartsSales.Requery

            Msg = ""
            Msg = Msg & lngSelected
            Msg = Msg & " selected records have been deleted."
            MsgBox (Msg)

        End If

Else

        Msg = ""
        Msg = Msg & "You don't have any "
        Msg = Msg & "records selected ..." & CR & CR
        Msg = Msg & "Please make your selection and try again."
        MsgBox (Msg)

End If

End Sub
