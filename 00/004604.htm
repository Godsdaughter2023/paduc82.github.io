According to this support article, when you save a Microsoft Excel workbook file on a network drive and you do not have both Modify and Delete rights for the folder where you are saving the file, you may experience the following symptoms:

Your changes could not be saved to 'BOOK1.XLS’ but were saved to a temporary document named '4CCA4100'. Close the existing document, then open the temporary document and save it under a new name.

Where4CCA4100 is a random eight-character alphanumeric string, and BOOK1.XLS is the name of the file you are saving.

Once you have appropriate rights for the folder in network drive, the “Sharing Violation” error should also go away.

Here's one possible solution (you'll have to excuse the formatting this site screws up code copy/paste)

Private Sub btn_UpdateDepartDates_Click()
    'Updates Depart dates when used with Update Arrive Dates
    Dim rs                    As DAO.Recordset
    Dim varBookMark           As Variant

    On Error GoTo Error_Handler

    Set rs = Me.Recordset
    varBookMark = Me.Bookmark
    With rs
        .MoveFirst
        Do While Not .EOF
            If IsNull(!DepartDate) = False Then
                .Edit
                !ArriveDate = !DepartDate
                .Update
            End If
            .MoveNext
        Loop
    End With
    Me.Bookmark = varBookMark

Error_Handler_Exit:
    On Error Resume Next
    If Not rs Is Nothing Then Set rs = Nothing
    Exit Sub

Error_Handler:
    MsgBox "The following error has occured" & vbCrLf & vbCrLf & _
           "Error Number: " & Err.Number & vbCrLf & _
           "Error Source: btn_UpdateDepartDates_Click" & vbCrLf & _
           "Error Description: " & Err.Description & _
           Switch(Erl = 0, "", Erl <> 0, vbCrLf & "Line No: " & Erl) _
           , vbOKOnly + vbCritical, "An Error has Occured!"
    Resume Error_Handler_Exit
End Sub

Here's another approach

Private Sub btn_UpdateDepartDates_Click()
    'Updates Depart dates when used with Update Arrive Dates
    Dim rsFiltered            As DAO.Recordset
    Dim varBookMark           As Variant
    Dim bBookmarkMe           As Boolean

    On Error GoTo Error_Handler

    If Me.NewRecord = False Then
        bBookmarkMe = True
        varBookMark = Me.Bookmark
    End If
    With Me.Recordset
        .Filter = "[DepartDate] Is Not Null"
        Set rsFiltered = .OpenRecordset
    End With

    With rsFiltered
        .MoveFirst
        Do While Not .EOF
            .Edit
            !ArriveDate = !DepartDate
            .Update
            .MoveNext
        Loop
    End With
    Me.Requery
    If bBookmarkMe = True Then
        Me.Bookmark = varBookMark
    Else
        DoCmd.GoToRecord , , acNewRec
    End If

Error_Handler_Exit:
    On Error Resume Next
    If Not rsFiltered Is Nothing Then Set rsFiltered = Nothing
    Exit Sub

Error_Handler:
    MsgBox "The following error has occured" & vbCrLf & vbCrLf & _
           "Error Number: " & Err.Number & vbCrLf & _
           "Error Source: btn_UpdateDepartDates_Click" & vbCrLf & _
           "Error Description: " & Err.Description & _
           Switch(Erl = 0, "", Erl <> 0, vbCrLf & "Line No: " & Erl) _
           , vbOKOnly + vbCritical, "An Error has Occured!"
    Resume Error_Handler_Exit
End Sub
