What I would like to know is if their is a way for Word to see what type of Picture is in the Clipboard so that I can use the following to Place the Cover within my Word Document?

Here is my Macro which holds both ways of placing the Cover, and I have placed what I would like to do, if it is possible in [BOLD] and the Place commands are in [BOLDITALIC]:

Private Sub cmdYes_Click()
    Application.ScreenUpdating = False
    With Selection
        if "Image in Clipboard is of JPEG format" then
            .PasteAndFormat (wdPasteDefault) 'Place JPEG
        else
            .PasteSpecial _
                Link:=False, _
                DataType:=wdPasteDeviceIndependentBitmap, _
                Placement:=wdInLine, _
                DisplayAsIcon:=False 'Place PNG
        end if
        .MoveRight _
            Unit:=wdCharacter, _
            Count:=1
        .MoveLeft _
            Unit:=wdCharacter, _
            Count:=1, _
            Extend:=wdExtend
        With .InlineShapes(1)
            .LockAspectRatio = msoTrue
            .Height = CentimetersToPoints(2)

            'Remove Hyperlink from Picture
            If .Range.Hyperlinks.Count = 1 Then .Range.Hyperlinks(1).Delete
        End With
    End With
    Application.ScreenUpdating = True
    Unload Me
End Sub

graham mayor say:

I have tested the following with several images from the site quoted and the following seems to work

    Application.ScreenUpdating = False
    With Selection
        On Error Resume Next
        .PasteSpecial _
                Link:=False, _
                DataType:=wdPasteDeviceIndependentBitmap, _
                Placement:=wdInLine, _
                DisplayAsIcon:=False
        If Err.Number = 5342 Then
            .PasteSpecial _
                    Link:=False, _
                    DataType:=wdPasteHTML, _
                    Placement:=wdInLine, _
                    DisplayAsIcon:=False
        End If
        On Error GoTo 0
        .MoveRight _
                Unit:=wdCharacter, _
                Count:=1
        .MoveLeft _
                Unit:=wdCharacter, _
                Count:=1, _
                Extend:=wdExtend
        With .InlineShapes(1)
            .LockAspectRatio = msoTrue
            .Height = CentimetersToPoints(2)

            'Remove Hyperlink from Picture
            If .Range.Hyperlinks.Count = 1 Then
                .Range.Hyperlinks(1).Delete
            End If
        End With
    End With
    Application.ScreenUpdating = True
    Unload Me

You can also use the Application.FileDialog.

 

Public Function OpenFileDialog(Optional sFilter As String = "All", _
                                Optional sInitialDirectory As String = vbNullString) As String
                               
    Dim oFileDialog As Object               'Late binding
    Dim varFile As Variant
   
    'Constances for type of File Dialog.
    Const msoFileDialogOpen = 1
   
    Set oFileDialog = Application.FileDialog(msoFileDialogOpen)
   
    With oFileDialog

        ' Allow user to make multiple selections in dialog box '
        .AllowMultiSelect = False
               
        ' Set initial directory
        'If not set uses users last directory
        '.InitialFileName = Application.CurrentProject.Path & "\" 'Current directory
        If sInitialDirectory <> vbNullString Then .InitialFileName = sInitialDirectory
       
        ' Clear out the current filters, and add our own.'
        .Filters.Clear
        If sFilter = "All" Then .Filters.Add "All Files", "*.*"
        If sFilter = "Access" Then
            .Filters.Add "All Access Databases", "*.accdb;*.accde;*.mdb;*.mde"
            .Filters.Add "Access 2007 Databases", "*.accdb;*.accde"
            .Filters.Add "Access 97-2003 Databases", "*.mdb;*.mde"
        End If
        If sFilter = "Excel" Then
            .Filters.Add "Excel 2007 Workbook", "*.xlsx"
            .Filters.Add "Excel 97-2003 Workbook", "*.xls"
        End If
        If sFilter = "Word" Then
            .Filters.Add "All Word Document", "*.docx;*.docm;*.doc;*.rtf"
            .Filters.Add "Word 2007 Document", "*.docx"
            .Filters.Add "Word Macro-Enabled Document", "*.docm"
            .Filters.Add "Word 97-2003 Document", "*.doc"
            .Filters.Add "Rich Text Format", "*.rtf"
        End If
        If sFilter = "Word Template" Then
            .Filters.Add "All Word Templates", "*.dotx;*.dotm;*.dot"
            .Filters.Add "Word 2007 Template", "*.dotx;*.dotm"
            .Filters.Add "Word 97-2003 Template", "*.dot"
        End If
        If sFilter = "Graphic" Then
            .Filters.Add "Graphic Files", "*.jpg;*.jpe;*.jpeg;*.gif;*.bmp;*.png;*.ico;*.wmf;*.emf"
            .Filters.Add "Portable Network Graphics", "*.png"
            .Filters.Add "JPEG File Interchange Format", "*.jpg;*.jpe;*.jpeg;*.jfif"
            .Filters.Add "Bitmaps", "*.bmp"
            .Filters.Add "Device Independent Bitmaps", "*.dib"
            .Filters.Add "Graphice Interchange Format", "*.gif"
            .Filters.Add "Icons", "*.ico"
            .Filters.Add "Windows Metafiles", "*.wmf"
            .Filters.Add "Enhanced Metafiles", "*.emf"
            .Filters.Add "All Files", "*.*"
        End If
       
        ' Set default value in list
        .FilterIndex = 1 'Defaults to the first filter
       
        ' Set the title of the dialog box. '
        .Title = "Please select a file"
        .ButtonName = "Select"
       
       
        ' Show the dialog box. If the .Show method returns True, the '
        ' user picked at least one file. If the .Show method returns '
        ' False, the user clicked Cancel. '
        If .Show = True Then

            Dim strFilename As String
            Dim strFolderName As String
            Dim iPtr As Integer
            'Loop through each file selected and add it to our list box. '
            For Each varFile In .SelectedItems
                iPtr = InStrRev(varFile, "\")
                strFilename = Mid(varFile, iPtr + 1)
                strFolderName = Mid(varFile, 1, iPtr)
 '               oFileDialog.Execute
                OpenFileDialog = varFile
            Next
        Else
            'MsgBox "You clicked Cancel."
        End If
    End With
End Function

ken sheridan say:

Another thing you might want to consider is whether it would be better to use a Word template file, .dotx?  You can then fill and save as many instances of the template as you wish.  Here's some code I wrote many years ago to do this, but which still works in Access 2007 at least.  The path to the template file, which can be obtained by means of the common dialogue as described by Doug and Jack, is passed into the procedure.  In earlier versions of Access the Word window came to the top, but not so in 2007, and I've never been able to crack that particular nut:

 

Sub FillForm(strTemplate As String)

 

    ' Opens a document in Word and inserts values from

    ' current record in form fields.

    ' Accepts: path to Word template file - String

   

    On Error GoTo Err_Handler

   

    Dim objWord As Object

    Dim objDoc As Object

    Dim frm As Form

    Dim strAddress As String

   

    ' return reference to form

    Set frm = Forms!frmAddresses

   

    ' if Word open return reference to it

    ' else establish reference to it

    On Error Resume Next

    Set objWord = GetObject(, "Word.Application")

    If Err.Number = 429 Then

        Set objWord = CreateObject("Word.Application")

    End If

   

    AppActivate "Microsoft Word"

    On Error GoTo Err_Handler

   

    ' open Word document in maximised window

    objWord.Visible = True

    Set objDoc = objWord.Documents.Add(strTemplate)

    objWord.WindowState = wdWindowStateMaximize

   

    ' insert first and lastname in form fields

    objDoc.FormFields("Text1").Result = frm!FirstName

    objDoc.FormFields("Text2").Result = frm!LastName

   

    Set objDoc = Nothing

    Set objWord = Nothing

   

Exit_here:

    On Error GoTo 0

    Exit Sub

   

Err_Handler:

    MsgBox Err.Description & " (" & Err.Number & ")"

    Resume Exit_here

 

End Sub

    kosh vorlon say:

    Hi Afronubian,

     

    IT IS A SCAM!!!

     

    This is a fairly common SCAM to try to get you to respond with personal information so they can hack into your account, hijack your account, confirm your account is valid. I've seen this same message before from others. Pay it no mind. Do nothing it says. Click on no links in it, download no attachments, do not reply to it, add the sender to your blocked senders list, and delete it and any more like it you get.

     

    See: http://www.microsoft.com/security/online-privacy/msname.aspx.

     

    Do not worry.  Your account is perfectly fine. This was certainly not from Microsoft and they are not closing your account. The SCAM is just to scare you into responding before you think about what you are doing.

     

    You were wise to post here for confirmation. You've saved yourself a lot of headaches down the road. You have good instincts and practice safe online habits. Keep it up.
