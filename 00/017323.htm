Here's my very crude script. Now the contacts' website URLs are available in all three fields, making them available everywhere again: in Outlook, in the People app, on my phone.

Sub CopyURLFields()
Dim objApp As Application
Dim objNS As NameSpace
Dim objFolder As MAPIFolder
Dim objItems As Items
Dim objItem As Object
Set objApp = CreateObject("Outlook.Application")
Set objNS = objApp.GetNamespace("MAPI")
Set objFolder = objNS.PickFolder
If Not objFolder Is Nothing Then
    Set objItems = objFolder.Items
    For Each objItem In objItems
        ' make sure you have a Contact item
        If objItem.Class = olContact Then
            sWebPage = objItem.WebPage
            sPersonalHomePage = objItem.PersonalHomePage
            sBusinessHomePage = objItem.BusinessHomePage
            bDirty = False
            If (sWebPage = "") Then
                If (sPersonalHomePage <> "") Then
                    objItem.WebPage = sPersonalHomePage
                    bDirty = True
                ElseIf (sBusinessHomePage <> "") Then
                    objItem.WebPage = sBusinessHomePage
                    bDirty = True
                End If
            End If
            If (sPersonalHomePage = "") Then
                If (sWebPage <> "") Then
                    objItem.PersonalHomePage = sWebPage
                    bDirty = True
                ElseIf (sBusinessHomePage <> "") Then
                    objItem.PersonalHomePage = sBusinessHomePage
                    bDirty = True
                End If
            End If
            If (sBusinessHomePage = "") Then
                If (sWebPage <> "") Then
                    objItem.BusinessHomePage = sWebPage
                    bDirty = True
                ElseIf (sBusinessHomePage <> "") Then
                    objItem.BusinessHomePage = sPersonalHomePage
                    bDirty = True
                End If
            End If
            If (bDirty) Then
                objItem.Save
            End If
        End If
    Next
End If

Set objItem = Nothing
Set objFolder = Nothing
Set objNS = Nothing
Set objApp = Nothing

    4. You will see a checkbox that says : "I sign in frequently on this device. Don't ask me for a code". Tick the checkbox, and enter the code.

    Your computer/device is now Trusted. Whenever you use your account on that specific computer/device, you will no longer asked for a code.
