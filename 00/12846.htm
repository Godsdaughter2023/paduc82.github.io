graham mayor say:
OK, if the message is formatted exactly as you suggested, and without knowing what ABCDEF actually represents, the following should work:

Sub MessageToExcel()
Dim xlApp As Object
Dim xlWB As Object
Dim xlSheet As Object
Dim olItem As MailItem
Dim olOutMail As MailItem
Dim vText As Variant
Dim sText As String
Dim vItem As Variant
Dim i As Long
Dim rCount As Long
Dim strFile As String
Dim strNum As String
Dim bXstarted As Boolean
Const strPath As String = "D:\Path\WorkbookName.xlsx"        'the path of the workbook
Const strAddr As String = "someoneATsomewhere.com"         'the manager's e-mail address

    If Application.ActiveExplorer.Selection.Count = 0 Then
        MsgBox "No Items selected!", vbCritical, "Error"
        Exit Sub
    End If
    On Error Resume Next
    Set xlApp = GetObject(, "Excel.Application")
    If Err <> 0 Then
        Application.StatusBar = "Please wait while Excel source is opened ... "
        Set xlApp = CreateObject("Excel.Application")
        bXstarted = True
    End If
    On Error GoTo 0
    'Open the workbook to input the data
    Set xlWB = xlApp.Workbooks.Open(strPath)
    Set xlSheet = xlWB.Sheets("Sheet1")

    'Process each selected record
    Set olItem = ActiveExplorer.Selection.Item(1)
    sText = olItem.Body
    vText = Split(sText, Chr(13))
    'Find the next empty line of the worksheet
    rCount = xlSheet.Range("A" & xlSheet.Rows.Count).End(-4162).Row
    rCount = rCount + 1

    'Check each line of text in the message body
    For i = UBound(vText) To 0 Step -1
        If InStr(1, LCase(vText(i)), "loaded") > 0 Then
            vItem = Split(vText(i), Chr(58))
            strNum = Trim(vItem(2))
            vItem = Split(Trim(vItem(1)), Chr(32))
            strFile = Trim(vItem(0))
            xlSheet.Range("A" & rCount) = strFile
            xlSheet.Range("B" & rCount) = strNum
            Exit For
        End If
    Next i
    xlWB.Close 1
    Set olOutMail = olItem.Forward
    With olOutMail
        .To = strAddr
        .HTMLBody = "<p><font face=""Arial"" size=""4"" color=""#FF0000"">Workbook attached" _
                    & "</font></p>" & olItem.HTMLBody
        .Attachments.Add strPath
        .Display
    End With
    If bXstarted Then
        xlApp.Quit
    End If
    Set xlApp = Nothing
    Set xlWB = Nothing
    Set xlSheet = Nothing
    Set olItem = Nothing
End Sub
