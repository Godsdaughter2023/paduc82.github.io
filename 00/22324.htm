AFAIK this is not possible. The only thing I know is the net file [ID [/close]] command, but I never used it.

http://technet.microsoft.com/en-us/library/bb490702.aspx

But you said you develop a report, that means to me that only you create\modify this file and anybody else just view this file. If so, we can use a feature:

- Create the report local on your PC.
- Add this code to the code module "ThisWorkBook":

Option Explicit

Private Sub Workbook_BeforeClose(Cancel As Boolean)
  With ThisWorkbook
    If .ReadOnly Then .Saved = True
  End With
End Sub

Private Sub Workbook_SheetChange(ByVal Sh As Object, ByVal Target As Range)
  Static MsgHasShown As Boolean
  If Not MsgHasShown And ThisWorkbook.ReadOnly Then
    MsgBox "This file is write protected and you cannot saved any changes!"
    MsgHasShown = True
  End If
End Sub

- Save and close the file
- Open the windows explorer, right the file, select properties and write protect the file.
- Copy the file to the network

Now it doesn't matter if somebody opens the file, create the next report same way and overwrite the file on the network.

And if you want you can add a feature to the file: Read the file date in Workbook_Open and Workbook_BeforeClose (maybe Selection_Change or inside a periodic OnTime sub) and compare it. And if they are different, show a message to the user that the file has changed.

hansv mvp say:

Right-click the sheet tab.

Select 'View Code' from the context menu.

Copy the following code into the worksheet menu:

Private Sub Worksheet_Change(ByVal Target As Range)
    If Not Intersect(Range("B1:D21"), Target) Is Nothing Then
        MsgBox "B22: " & Range("B22").Text & vbCrLf & _
            "C22: " & Range("C22").Text & vbCrLf & _
            "D22: " & Range("D22").Text, vbInformation
    End If
End Sub

Switch back to Excel.

Make sure that you save the workbook in a format that supports macros (.xlsm, .xlsb, or .xls, but not .xlsx).

rich michaels - mvp (greatcirclelearning.com) say:

Hi John Christopher,

You can add this VBA code to your Normal template and it will adjust the Zoom automatically. I use it on my systems. I've also included below a screen clip of where you should add the code... it goes into the ThisDocument class of the Normal template. To open the VBE (Visual Basic Editor) press Alt+F11. Depending on your keyboard you might have to add the Fn (Function) key... Alt+Fn+F11.

Hope this helps (the code begins with "Option Explicit")

Option Explicit
'Created by Richard V. Michaels, Microsoft Word MVP

'These are VBA subroutines for you to add to the "ThisDocument" Class of Microsoft Word's
'Normal.dotm template. They automatically establish the Word preference settings for a
'professional writing environment. They function with Word 2011 and 2016 on the Mac
'and Word 2007, 2010, 2013, and 2016 on the PC.

'On a PC they work 100% of the time and on a Mac about 99% of the time. The one situation
'on a Mac where they do not work is if Word is open but no documents are open. In this
'scenario if you click on the Word icon in your dock, a new blank document is created by
'Word but the AutoNew, AutoOpen, and AutoExec routines do not execute. I don't know why
'and assume it is a shortcoming in the Word 2011 and 2016 Mac versions. If you use Word's
'menu bar instead and choose File > New, the AutoNew routine does execute and your
'special writing environment settings will be properly established.

'To obtain other products I have available for professional level writers and editors
'visit my publisher's website at: http://www.greatcirclelearning.com
Dim doc As Word.Document

Sub AutoExec()
    'what is contained in an AutoExec subroutine only executes from a globally loaded template
    'this means the template is either Word's Normal.dotm template or a custom template
    'of your own that you have placed in Word's Startup folder
    Application.OnTime Now + TimeValue("00:00:02"), "SetWritingEnvironment", False
    'the above command will cause the SetWritingEnvironment subroutine to execute 2 seconds
    'after Word opens. Adjust the time if you find that your system is slower and more time is needed
End Sub

Sub AutoNew()
    'this only executes if you select from Word, "File > New" and start a new blank document
    'it does not execute if you start a new document from a template that also contains an
    'AutoNew routine. The other template's AutoNew routine executes instead
    Application.OnTime Now + TimeValue("00:00:02"), "SetWritingEnvironment", False
End Sub

Sub AutoOpen()
    'this only executes if you open a new document either from Word > File > Open or
    'from File Finder on a Mac or File Explorer on a PC. There is no restriction if the other
    'document has an attached template with and AutoOpen routine... this one will work also.
    Application.OnTime Now + TimeValue("00:00:02"), "SetWritingEnvironment", False
End Sub

Sub SetWritingEnvironment()
    On Error GoTo errHandler
    Set doc = ActiveDocument
    
    'add or edit any of the following settings to make the writing environment settings
    'satisfactory for your needs.
    
    If doc.TrackRevisions = True Then doc.TrackRevisions = False
    If doc.ShowRevisions = False Then doc.ShowRevisions = True
    
    If AutoCorrect.ReplaceText = False Then AutoCorrect.ReplaceText = True
    If Not Application.Browser.Target = wdBrowsePage Then Application.Browser.Target = wdBrowsePage
    If doc.ActiveWindow.DisplayRulers = False Then doc.ActiveWindow.DisplayRulers = True
    If doc.ActiveWindow.DisplayVerticalRuler = False Then doc.ActiveWindow.DisplayVerticalRuler = True
    With doc.ActiveWindow.View
        If .ShowBookmarks = False Then .ShowBookmarks = True
        If .ShowFieldCodes = True Then .ShowFieldCodes = False
        If .ShowAll = False Then .ShowAll = True
        If Not .FieldShading = wdFieldShadingAlways Then .FieldShading = wdFieldShadingAlways
        If .TableGridlines = False Then .TableGridlines = True
        If Not .Type = wdPageView Then .Type = wdPageView
        If Not .Zoom.PageFit = wdPageFitBestFit Then .Zoom.PageFit = wdPageFitBestFit
    End With
    'the reason for putting a check for opposite on all of the above settings is so the code will not
    'update preference files and the normal template unnecessarily. However, that is still not
    'enough for the new blank temporary document that Word first opens and so I've included the following:
    doc.Saved = True
    'it stops the save message prompt if you close the blank document and quit Word
' end default settings

errHandler:
    Select Case Err.Number
        Case 0
            'Once you know this routine is running (exactly when you want it to on your machine)
            'remove the following Message Box command
            MsgBox "Writing Environment Now Established" & vbCr & _
                "http://www.greatcirclelearning.com", vbInformation, "Message from Rich Michaels, Word MVP"
            DoEvents
        Case Else
            MsgBox "Error Code: " & Err.Number & vbCr & _
                Err.Description, vbExclamation, "Set Writing Environment Failed"
            Err.Clear
    End Select
End Sub
