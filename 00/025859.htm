Try this:

Sub xy()
  x = ActiveSheet.ChartObjects(1).Chart.SeriesCollection(1).XValues
  y = ActiveSheet.ChartObjects(1).Chart.SeriesCollection(1).Values
  For i = LBound(x) To UBound(x)

  Next i
End Sub
