Can I extract footnotes/endnotes in a document but still retain formatting, including the numbering?

There's a macro in Windows that does this, but it doesn't work on the Mac. Is there a note-stripping macro for the Mac? Here's a link to the Windows discussion that has the macro:

http://answers.microsoft.com/en-us/office/forum/office_2010-word/can-i-extract-footnotesendnotes-a-document-but/620eb2e2-e489-4d13-a1e5-1b44290ce906?page=2

And here's the macro for Windows: 

Sub UnLinkNotes()
Application.ScreenUpdating = False
Dim Fld As Field, nRng As Range, eNote As Endnote, fNote As Footnote, nRef As String
With ActiveDocument
  'Convert any endnote/footnote cross-reference fields to text
  For Each Fld In .Fields
    If Fld.Type = wdFieldNoteRef Then Fld.Unlink
  Next
  'Delete the next two lines if you don't want to differentiate footnote and endnote references by colour
  .Styles("Footnote Reference").Font.Color = wdColorGreen
  .Styles("Endnote Reference").Font.Color = wdColorRed
  For Each eNote In .Endnotes
    With eNote
      With .Reference.Characters.First
        'The next two lines insert a brace after the endnote. Two lines
        'further down insert a brace before the endnote.
        'This is useful if you want to be able to re-create the endnotes later on.
        .InsertAfter "}"
        .Characters.Last.Font.Superscript = True
        .Collapse wdCollapseStart
        .InsertCrossReference wdRefTypeEndnote, wdEndnoteNumberFormatted, eNote.Index
        nRef = .Characters.First.Fields(1).Result
        .Characters.First.Fields(1).Unlink
        'The next two lines insert a brace before the footnote. Two lines
        'further up insert a brace after the footnote.
        'This is useful if you want to be able to re-create the footnotes later on.
        .InsertBefore "}"
        .Characters.First.Font.Superscript = True
      End With
      .Range.Cut
    End With
    'To position converted Endnotes at the end of the document, use:
    Set nRng = .Range
    'To position converted Endnotes at the end of the Section, use:
    'Set nRng = eNote.Range.Sections.First.Range
    'To let the code decide, based on whatever's in the document, use:
    'If .Range.EndnoteOptions.Location = wdEndOfSection Then
    '  Set nRng = eNote.Range.Sections.First.Range
    'Else
    '  Set nRng = .Range
    'End If
    With nRng
      .Collapse wdCollapseEnd
      .End = .End - 1
      If .Characters.Last <> Chr(12) Then .InsertAfter vbCr
      .InsertAfter nRef & " "
      With .Paragraphs.Last.Range
        .Style = "Endnote Text"
        .Words.First.Style = "Endnote Reference"
      End With
      .Collapse wdCollapseEnd
      .Paste
      If .Characters.Last.Next = Chr(12) Then .InsertAfter vbCr
    End With
  Next
  For Each eNote In .Endnotes
    eNote.Delete
  Next
  For Each fNote In .Footnotes
    With fNote
      With .Reference.Characters.First
        'The next two lines insert a square bracket after the footnote. Two lines
        'further down insert a square bracket before the footnote.
        'This is useful if you want to be able to re-create the footnotes later on.
        .InsertAfter "]"
        .Characters.Last.Font.Superscript = True
        .Collapse wdCollapseStart
        .InsertCrossReference wdRefTypeFootnote, wdFootnoteNumberFormatted, fNote.Index
        nRef = .Characters.First.Fields(1).Result
        .Characters.First.Fields(1).Unlink
        'The next two lines insert a square bracket before the footnote. Two lines
        'further up insert a square bracket after the footnote.
        'This is useful if you want to be able to re-create the footnotes later on.
        .InsertBefore "["
        .Characters.First.Font.Superscript = True
      End With
      .Range.Cut
    End With
    'The nearest a converted footnote can be positioned to its reference without
    'considerable extra work that can involve splitting paragraphs in the body of the
    'document and splitting footnote text across pages is the end of the same Section.
    'this can be done via:
    'Set nRng = fNote.Range.Sections.First.Range
    'The alternative is to position them at the end of the document, via:
    Set nRng = .Range
    With nRng
      .Collapse wdCollapseEnd
      .End = .End - 1
      If .Characters.Last <> Chr(12) Then .InsertAfter vbCr
      .InsertAfter nRef & " "
      With .Paragraphs.Last.Range
        .Style = "Footnote Text"
        .Words.First.Style = "Footnote Reference"
      End With
      .Collapse wdCollapseEnd
      .Paste
      If .Characters.Last = Chr(12) Then .InsertAfter vbCr
    End With
  Next
  For Each fNote In .Footnotes
    fNote.Delete
  Next
End With
Set nRng = Nothing
Application.ScreenUpdating = True
End Sub

Sub ReformatFootnotes()
