ken sheridan say: You need to open the form filtered on whatever is the primary key of its recordset, e.g. ProductID, where the value is the value of the ProductID column in the currently selected row of the subform's recordset.

You can see how this works if you make a small amendment to one of the form's in my FindRecord demo which you'll find in my public databases folder at:

https://onedrive.live.com/?cid=44CC60D7FEA42912&id=44CC60D7FEA42912!169

Note that if you are using an earlier version of Access you might find that the colour of some form objects such as buttons shows incorrectly and you will need to  amend the form design accordingly.  

If you have difficulty opening the link copy its text (NB, not the link location) and paste it into your browser's address bar.

This little demo file includes an option to Open simulated 'split form'.  Forget about the simulation of a split form, however, and assume that the parent form is, like yours, unbound.  If you add a button to the parent form and put the following in its Click event procedure:

    Dim strCriteria As String
    
    strCriteria = "AddressID = " & Nz(Me.sfcAddresses.Form.AddressID, 0)
    
    DoCmd.OpenForm "frmAddresses", _
        WhereCondition:=strCriteria, _
        WindowMode:=acDialog
        
    If Me.NewRecord Then Me.sfcAddresses.Requery

When you click the button it will open the frmAddresses form (normally used elsewhere in the demo) in dialogue mode at whatever address is selected in the subform.  The address can then be edited in frmAddresses.  

If the button is clicked while the subform is at an empty new record, frmAddresses will open at an empty new record, and a new address can be entered.  When code execution in the above procedure resumes after frmAddresses has been closed the subform will be requeried to include the new record.

suzanne s. barnhill say:

In a TOC, Word includes the page number represented by the PAGE field, which is the number that appears on the page (or would appear if printed). In order to add a part or chapter number to a page number, you must use the "Include chapter number" feature in the Page Number Format dialog, and the "chapter number" must be the auto number of a paragraph in one of Word's built-in heading styles. There is no exception to this. You can manually add chapter numbering in a variety of ways, but none of them will result in this number being included in the TOC.

For more, see http://word.mvps.org/FAQs/Numbering/ChapterNumber.htm.

Function FixBuffer(inputStr As String) As String

Dim outString As String
Dim thisBlock As String
Dim jspBlock As String

Dim jspStartTag As Long
Dim jspEndTag As Long
Dim tagLen As Long

On Error GoTo fixBuffer_ERR

tagLen = 5
outString = ""
jspStartTag = 0
jspEndTag = 1 - tagLen

If InStr(1, inputStr, "&lt;%") = 0 Then
    FixBuffer = inputStr
    Exit Function
End If

'Before processing JSP blocks, replace encoded JSP start and end tags
'within hyperlinks which are encoded differently
inputStr = Replace(inputStr, "..\%3C%25", "&lt;%")
inputStr = Replace(inputStr, "%25%3E", "%&gt;")

Do
    '<% is encoded when the htm file is saved
    jspStartTag = InStr(jspEndTag + tagLen, inputStr, "&lt;%")
    If jspStartTag = 0 Then
        thisBlock = Mid(inputStr, jspEndTag + tagLen)
        outString = outString + thisBlock
        Exit Do
    Else
        '%> is encoded when the htm file is saved
        beforeblock = SubStr(inputStr, jspEndTag + tagLen, jspStartTag - 1)
        jspEndTag = InStr(jspStartTag, inputStr, "%&gt;")
        
        ' FFO-9004: Added check to see if opening <% is closed
        If jspEndTag = 0 Then
            MsgBox ("An Error occurred: FixBuffer, An open jsp scriptlet tag (<%) exists and is not closed. A closing scriptlet (%>) is required before this document can be saved as a JSP")
            Exit Function
        Else
            jspBlock = SubStr(inputStr, jspStartTag, jspEndTag + tagLen - 1)
            
            jspBlock = FixJspBlock(jspBlock)
            If Left(jspBlock, 13) = "<%=startList(" Then
                jspBlock = jspBlock & "<%for (int i = 0; i < getCurrentNodeList().getLength(); i++) {Node node = getCurrentNodeList().item(i);if (node.getNodeType() == Node.ELEMENT_NODE) {setCurrentElement((Element) node);%>"
            ElseIf Left(jspBlock, 11) = "<%=endList(" Then
                jspBlock = "<%}}%>" & jspBlock
            ElseIf Left(jspBlock, 12) = "<%=elseList(" Then
                jspBlock = "<%}} if (getCurrentNodeList().getLength()==0) {{%>" & jspBlock
            End If
            thisBlock = beforeblock + jspBlock
            outString = outString + thisBlock
        End If
    End If
        
Loop While True

FixBuffer = outString

Exit Function

fixBuffer_ERR:
    MsgBox ("An Error occurred: FixBuffer, " + Err.Description)
    
End Function

Function RemoveTag(buf As String, TagToGo As String) As String
'return the supplied string with the tag removed, possible values for tag are
' <SPAN
'</SPAN
'' the trailing > will be found and the tag removed
Dim startTagPos As Long
Dim endPos As Long

On Error GoTo RemoveTag_ERR

    startTagPos = InStr(buf, TagToGo)
    If startTagPos > 0 Then
        ' find closing >
        endPos = InStr(startTagPos, buf, ">")
        If (endPos > 0) Then
            ' remove that much from the buffer
            ans = Left(buf, startTagPos - 1)
            ans = ans + Right(buf, Len(buf) - endPos)
        Else
            ' couldn't find the trailing > thing is *ucked
            ans = buf
           
        End If
    Else
        ans = buf
    End If
    RemoveTag = ans
    
    Exit Function
        
RemoveTag_ERR:
    MsgBox ("An Error has occurred, RemoveTag : " + Err.Description)

End Function
Function FixJspBlock(inputString As String) As String

Dim outString As String
Dim toProcess As String

outString = inputString

'Decode encoded characters


outString = Replace(outString, "Ã­", "'")
outString = Replace(outString, "&lt;", "<")
outString = Replace(outString, "&gt;", ">")
outString = Replace(outString, "Ã¬", """")
outString = Replace(outString, "Ã®", """")
outString = Replace(outString, "&quot;", """")
outString = Replace(outString, Chr$(160), "")
outString = Replace(outString, "&amp;&amp;", "&&")

'remove new lines within jsp tags (cannot split single statements over multiple lines)
outString = Replace(outString, Chr(10), " ")

outString = RemoveAllTag(outString, "<span")
outString = RemoveAllTag(outString, "</span")
outString = RemoveAllTag(outString, "<o:p")
outString = RemoveAllTag(outString, "</o:p")
outString = RemoveAllTag(outString, "<p")
outString = RemoveAllTag(outString, "</p")
outString = RemoveAllTag(outString, "<a")
outString = RemoveAllTag(outString, "</a")
outString = RemoveAllTag(outString, "<b")
outString = RemoveAllTag(outString, "</b")


Do While InStr(1, outString, "  ")
        outString = Replace(outString, "  ", " ")
Loop

outString = Trim(outString)


FixJspBlock = outString
    
End Function

Public Function setCustomProperty(name As String, newValue As String)

On Error GoTo setCustomProperty_ERR

    Dim Index As Long
    Dim pos As Long
    Dim blockSize As Integer
    Dim tmpName As String
    Dim tmpValue As String
    
    blockSize = 200
    pos = 1
    Index = 1
    
    While (pos <= Len(newValue))
        tmpValue = SubStr(newValue, pos, Min(pos + blockSize - 1, Len(newValue)))
        tmpName = name + str(Index)
        setSimpleProperty tmpName, tmpValue
        pos = pos + blockSize
        Index = Index + 1
    Wend

    'Blank out higher index value properties
    For Each prop In ActiveDocument.CustomDocumentProperties
        tmpName = name + str(Index)
        If prop.name = tmpName Then
            prop.Value = ""
            Index = Index + 1
        End If
    Next

Exit Function
setCustomProperty_ERR:
    MsgBox ("Error setting Custom Property, " + Err.Description)
    
End Function

Private Sub setSimpleProperty(name As String, newValue As String)

    For Each prop In ActiveDocument.CustomDocumentProperties
        If prop.name = name Then
            prop.Value = newValue
            Exit Sub
        End If
    Next
    
    'If we can't find it, add it
    With ActiveDocument.CustomDocumentProperties
        .Add name:=name, _
            LinkToContent:=False, _
            Type:=msoPropertyTypeString, _
            Value:=newValue
    End With
    
End Sub

    Dim simpleValue As String

    simpleValue = ""

        End If
        If prop.name = tmpName Then

        End If
    Next
    
    If multiValue = "" Then
        getCustomProperty = simpleValue

        getCustomProperty = multiValue

            'blank out previous simple value
            setSimpleProperty name, "No longer used"

    End If
    
    Exit Function
