Sub insrt_below()
Dim c As Range
For Each c In Range("A1:A8")
    If InStr(1, c, "*", vbTextCompare) > 0 Then
        Rows(c.Offset(1, 0).Row & ":" & c.Offset(1, 0).Row).Insert shift:=xlDown
    End If
Next c
End Sub

We apologize for the inconvenience this issue has caused you as well as for the delayed response.
