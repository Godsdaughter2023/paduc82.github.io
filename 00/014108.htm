Hello Mercina,

I would use a VBA code like this:
Sub sort_with_adjacent_rows()
Const sWorksheet = "Sheet1" 'Amend if necessary
Const sHelperColumn1 = "AA" 'Amend if necessary
Const sHelperColumn2 = "AB" 'Amend if necessary

Dim ws As Worksheet
Dim s As String
Dim i As Long, j As Long

Set ws = Sheets(sWorksheet)
j = ws.UsedRange.Rows.Count

'Fill helper columns
For i = 2 To j
    If ws.Cells(i, 3) <> "" Then s = ws.Cells(i, 3)
    ws.Range(sHelperColumn1 & i) = s
    ws.Range(sHelperColumn2 & i) = i
Next i

'Sort
With ws.Sort
    .SortFields.Clear
    .SortFields.Add Key:=Range(sHelperColumn1 & "2:" & sHelperColumn1 & j), _
        SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
    .SortFields.Add Key:=Range(sHelperColumn2 & "2:" & sHelperColumn2 & j), _
        SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
    .SetRange Range("A2:" & sHelperColumn2 & j)
    .Header = xlNo
    .MatchCase = False
    .Orientation = xlTopToBottom
    .SortMethod = xlPinYin
    .Apply
End With

'Empty helper columns
ws.Range(sHelperColumn1 & "2:" & sHelperColumn2 & j).ClearContents
End Sub
