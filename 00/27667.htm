Okay, no problem. Make a new file, copy the code below into a regular module and execute sub Main.

Please be patient, needs a few seconds to run (depending on your PC).

Option Explicit

Sub Main()
  Dim Bolt, Hole, Item, Result
  Dim B As Collection, H As Collection
  Dim i As Long, j As Long

  'All possible bolts
  ReDim Bolt(1 To 12)
  For i = 1 To UBound(Bolt)
    Bolt(i) = Chr(i + 64)
  Next
  'All possible holes
  ReDim Hole(1 To 12)
  For i = 1 To UBound(Hole)
    Hole(i) = i
  Next

  'Build the combinations
  Set B = VariationsR(Bolt, 3)
  Set H = Combinations(Hole, 3)
  'Create the result
  ReDim Result(1 To B.Count * H.Count, 1 To 6)
  i = 0
  For Each Hole In H
    For Each Bolt In B
      i = i + 1
      For j = 1 To 3
        Result(i, j) = Bolt(j)
        Result(i, j + 3) = Hole(j)
      Next
    Next
  Next
  'Output
  Range("A1").Resize(UBound(Result), UBound(Result, 2)).Value = Result
End Sub

Function Combinations(Arr, ByVal Count As Integer) As Collection
  'Return all possible combinations of count elements in Arr as arrays in a collection
  Dim Index() As Long
  Dim Result()
  Dim i As Long, j As Integer
  'Create a collection
  Set Combinations = New Collection
  'Return empty collection if Count exceed the bound's of Arr
  If LBound(Arr) + Count - 1 > UBound(Arr) Or Count < 1 Then Exit Function
  'Setup space for index and result array
  ReDim Index(1 To Count) As Long
  ReDim Result(1 To Count)
  'Setup index for first combination
  For i = 1 To Count
    Index(i) = LBound(Arr) + i - 1
  Next
  Do
    'Build combination
    For i = 1 To Count
      Result(i) = Arr(Index(i))
    Next
    'Save it
    Combinations.Add Result
    'Get next index
    i = Count
    j = 0
    Do
      'Max. position for this index reached?
      If Index(i) = UBound(Arr) - Count + i Then
        j = j + 1
        'Move index before up one step in next loop
        i = i - 1
        If i < 1 Then Exit Function
      Else
        'Increment position
        Index(i) = Index(i) + 1
        'Setup next indices
        For j = 1 To j
          i = i + 1
          Index(i) = Index(i - 1) + 1
        Next
        Exit Do
      End If
    Loop
  Loop
End Function

Function VariationsR(Arr, ByVal Count As Integer) As Collection
  'Return all possible variations with repetition of count elements in Arr as arrays in a collection
  Dim Index() As Long
  Dim Result()
  Dim i As Long
  'Create a collection
  Set VariationsR = New Collection
  'Return empty collection if Count exceed the bound's of Arr
  If LBound(Arr) + Count - 1 > UBound(Arr) Or Count < 1 Then Exit Function
  'Setup space for index and result array
  ReDim Index(1 To Count) As Long
  ReDim Result(1 To Count)
  'Setup index for first combination
  For i = 1 To Count
    Index(i) = LBound(Arr) '+ I - 1
  Next
  Do
    'Build variation
    For i = 1 To Count
      Result(i) = Arr(Index(i))
    Next
    'Save it
    VariationsR.Add Result
    'Get next index
    i = Count
    Do
      'Max. position for this index reached?
      If Index(i) = UBound(Arr) Then
        Index(i) = LBound(Arr)
        'Move index before up one step in next loop
        i = i - 1
        If i < 1 Then Exit Function
      Else
        'Increment position
        Index(i) = Index(i) + 1
        Exit Do
      End If
    Loop
  Loop
End Function
