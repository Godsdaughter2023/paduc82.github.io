tasos_k say:

Hi,

sample

activesheet: data/numbers in a1:a30, results (10 unique numbers) in column C

pic1

or

pic2 (select from random numbers, in coluumn A)

try this vba macro

[edit..]

Sub Rand_Unique_Values()

'## 08 Dec. 2018 oAnast##

Const sRng As String = "A1:A30" '<< ## data range / change as needed ##

Const N As Long = 10 ' '<< ## search for 10 unique values / change as needed##

Const sC As String = "C" '## results in column C ##

Dim rng As Range

Set rng = Range(sRng)

Dim N1 As Long, N2 As Long, N As Long

N1 = 1

N2 = rng.Rows.Count '## number of rows##

Range(sC & ":" & sC).ClearContents

If N > N2 Then Exit Sub '## always N=10 less or equal to N2=30 ##

Dim v As Variant

ReDim v(N1 To N2)

Dim i As Long, x As Long, y As Long

For i = N1 To N2

v(i) = Cells(i, "A")

Next i

For i = N2 To N1 Step -1

x = WorksheetFunction.RandBetween(N1, N2)

y = v(x)

v(x) = v(i)

v(i) = y

Next i

ReDim Preserve v(N1 To N2)

Cells(1, sC).Resize(N) = WorksheetFunction.Transpose(v)

Cells(1, sC).Resize(N).Sort Key1:=Cells(1, sC), Order1:=xlAscending, Header:=xlNo 'sort data

End Sub
