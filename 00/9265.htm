jim thomlinson say:

There are only a few possibilities here. There is no setting that would force the cursor down and to the left. Here are a few things that could generate that behaviour:

  • A macro
  • Using the tab key to move to the right. When you then hit enter it will move you down and to the left under where you started to tab
  • UnLocked cells with the option to select locked cells deactiveated. If the next unlocked cell is down and to the left then that is where the cursor will go.

    Sub FillArray(ByRef Arr, ByVal LowerBound As Long, ParamArray Args())
      'Füllt ein Array ab LowerBound mit Werten, initialisiert es falls nötig, führt ein/zwei dimensionale Array's zusammen
      Dim I As Long, J As Long, K As Long, Item As Long
      For I = 0 To UBound(Args)
        If IsArray(Args(I)) Then
          Select Case Dimension(Args(I))
            Case 1
              Item = Item + UBound(Args(I)) - LBound(Args(I)) + 1
            Case 2
              Item = Item + (UBound(Args(I)) - LBound(Args(I)) + 1) * (UBound(Args(I), 2) - LBound(Args(I), 2) + 1)
          End Select
        Else
          Item = Item + 1
        End If
      Next
      If Item = 0 Then Exit Sub
      On Error GoTo ExitPoint
      If Dimension(Arr) <= 0 Then ReDim Arr(LowerBound To LowerBound + Item - 1)
      Item = LowerBound
      For I = 0 To UBound(Args)
        If IsArray(Args(I)) Then
          Select Case Dimension(Args(I))
            Case 1
              For J = LBound(Args(I)) To UBound(Args(I))
                Arr(Item) = Args(I)(J)
                Item = Item + 1
              Next
            Case 2
              For K = LBound(Args(I), 2) To UBound(Args(I), 2)
                For J = LBound(Args(I)) To UBound(Args(I))
                  Arr(Item) = Args(I)(J, K)
                  Item = Item + 1
                Next
              Next
          End Select
        Else
          Arr(Item) = Args(I)
          Item = Item + 1
        End If
      Next
    ExitPoint:
    End Sub

     

    Function Dimension(Arr As Variant) As Long
      'Returns number of dimensions of an array or 0 for an undimensioned array or -1 if no array at all.
      If IsArray(Arr) Then
        On Error GoTo Done
        Do
          Dimension = Dimension + 1
        Loop While IsNumeric(UBound(Arr, Dimension))
      End If
    Done:
      Dimension = Dimension - 1
    End Function
