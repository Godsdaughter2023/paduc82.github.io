hansv mvp say:

Try this:

Sub ImportFromLastYear()
  Dim strFileName As String
  Dim wbkCur As Workbook
  Dim wbkOld As Workbook
  Dim wshCur As Worksheet
  Dim wshOld As Worksheet

  On Error GoTo ErrHandler

  ' Prompt user to locate last year's workbook
  strFileName = Application.GetOpenFilename(Title:="Please select last year's workbook")
  If strFileName = "False" Then
    MsgBox "No workbook selected!", vbExclamation
    Exit Sub
  End If
  Application.ScreenUpdating = False
  Set wbkCur = ActiveWorkbook
  ' Open last year's workbook
  Set wbkOld = Workbooks.Open(strFileName)
  ' Loop through sheets of this year's workbook
  For Each wshCur In wbkCur.Worksheets
    Set wshOld = Nothing
    On Error Resume Next
    ' Try to match with sheet in last year's workbook
    Set wshOld = wbkOld.Worksheets(wshCur.Name)
    On Error GoTo ErrHandler
    If Not wshOld Is Nothing Then
      wshOld.Range("M7:M15").Copy
      wshCur.Range("N7:N15").PasteSpecial Paste:=xlPasteValuesAndNumberFormats
    End If
  Next wshCur

ExitHandler:
  On Error Resume Next
  wbkOld.Close SaveChanges:=False
  Application.ScreenUpdating = True
  Exit Sub

ErrHandler:
  MsgBox Err.Description, vbExclamation
  Resume ExitHandler
End Sub

swischner say:

Yes. Of course. reinstalling OneNote will completely change the data model of OneNote objects.....

Ink "strokes" in OneNote (as in some other note taking programs) are handled as / converted to vector graphic objects and not stored pixel by pixel. This for example gives you the option to move a stroke which is overlapping another one without damaging the underlying object or leave a gap as it is simply "redrawn" at the new position.

It's also a prerequisite for handwriting recognition as the program needs to "know" that you have drawn a circle like shape to identify it as "o" or "0". It could not read an "o" if there is only a bunch of unrelated pixels which (for human eyes) happen to form a circle.

So if you draw a line of a certain length it is stored as a line with exactly that length. You can not shorten it by removing some pixels with an eraser. That is only possible in pixel based paint programs. 

But maybe all of these facts will totally change when you reinstall OneNote. Good luck :)

Public Sub XLS_SecureWrkBk(ByVal sFilePath As String, ByVal sFileName As String, ByVal sFileExt As String)
    Dim oExcel                As Object
    Dim oExcelWrkBk           As Object
    Const xlExcel8 = 56 'Excel 97-2003 Workbook (*.xls)

    On Error GoTo Error_Handler
   
    FileCopy sFilePath & sFileName & "." & sFileExt, sFilePath & "MyTmp" & "." & sFileExt
    Do While Len(Dir(sFilePath & "MyTmp" & "." & sFileExt)) = 0
        Sleep (200)
    Loop
    Kill sFilePath & sFileName & "." & sFileExt

    Set oExcel = CreateObject("Excel.Application")     'Start Excel
    oExcel.Visible = False   'Keep Excel hidden
    oExcel.ScreenUpdating = False
    oExcel.DisplayAlerts = False
    Set oExcelWrkBk = oExcel.WorkBooks.Open(sFilePath & "MyTmp" & "." & sFileExt)    'Start a new workbook
    oExcelWrkBk.SaveAs sFilePath & sFileName & "." & sFileExt, xlExcel8, "YourPassword"
    oExcelWrkBk.Close False
    Kill sFilePath & "MyTmp" & "." & sFileExt
    oExcel.Quit

Error_Handler_Exit:
    On Error Resume Next
    Set oExcelWrkBk = Nothing
    Set oExcel = Nothing
    Exit Sub

Error_Handler:
    MsgBox "The following error has occured" & vbCrLf & vbCrLf & _
           "Error Number: " & Err.Number & vbCrLf & _
           "Error Source: XLS_SecureWrkBk" & vbCrLf & _
           "Error Description: " & Err.Description & _
           Switch(Erl = 0, "", Erl <> 0, vbCrLf & "Line No: " & Erl) _
           , vbOKOnly + vbCritical, "An Error has Occured!"
    oExcel.DisplayAlerts = True
    oExcel.ScreenUpdating = True
    oExcel.Visible = True   'Make excel visible to the user
    Resume Error_Handler_Exit
End Sub
