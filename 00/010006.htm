If by "Complete" you mean the Percent Complete field, then the following formula should work. Note that the current date starts at the beginning of a work day (i.e. 8:00 AM) while a deadline is normally at the end of a workday (i.e. 5:00 PM), so you might want to add an extra day to get the true number of calendar days between the current date and the deadline.   Dim R As Range
  On Error Resume Next
  Set wsAll = ThisWorkbook.Worksheets("ALL")
  If Err<>0 Then
    ' no ALL sheet, so create one
    Set wsAll = ThisWorkbook.Worksheets.Add
    wsAll.Name = "ALL"
  Else
    wsAll.UsedRange.ClearContents ' lose previous content
  End If
  On Error Goto 0
  For Each WS In ThisWorkbook.Worksheets
    If WS.Name="ALL" Then
       ' do nothing
    Else
      If wsAll.Range("A1")="" Then
        ' no headings yet - copy all
        WS.Range("A1").CurrentRegion.Copy wsAll.Range("A1")
      Else
        ' don't copy the headings
        WS.Range("A1").CurrentRegion.Offset(1).Copy wsAll.Cells(wsAll.Rows.Count,1).End(xlUp).Offset(1)
      End If
    End If
  Next
End Sub
