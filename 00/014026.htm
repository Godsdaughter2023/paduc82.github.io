How do I color fill only the top half of a horizontal range of - Microsoft Community

emerogork2 say:

How do I color fill only the top half of a horizontal range of cells?

That is it, top half one color, bottom half another color.

Lets say cells C30:Y30 for starters....

hansv mvp say:

The following version should add two rectangular shapes to the range C30:Y30. As far as I can tell, it should work in all recent versions of Excel:

Sub Fill2Colors()
    ' based on code by Héctor Miguel
    Dim sL As Single
    Dim sT As Single
    Dim sW As Single
    Dim sH As Single
    Dim sN As String
    Const vbGrey = &H808080
    With Range("C30:Y30")
        sL = .Left
        sT = .Top
        sW = .Width
        sH = .Height
        sN = .Address(False, False)
        With ActiveSheet.Shapes
            On Error Resume Next
            .Range(Array(sN & "1", sN & "2", "grp" & sN)).Delete
            On Error GoTo 0
            With .AddShape(msoShapeRectangle, sL, sT, sW, sH / 2)
                .Name = sN & "1"
                .Line.Visible = msoFalse
                .Fill.Visible = msoTrue
                .Fill.Transparency = 0.75
                .Fill.ForeColor.RGB = vbBlue
            End With
            With .AddShape(msoShapeRectangle, sL, sT + sH / 2, sW, sH / 2)
                .Name = sN & "2"
                .Line.Visible = msoFalse
                .Fill.Visible = msoTrue
                .Fill.Transparency = 0.75
                .Fill.ForeColor.RGB = vbGrey
            End With
            With .Range(Array(sN & "1", sN & "2")).Group
                .Name = "Grp" & sN
                .Placement = xlMoveAndSize
            End With
        End With
    End With
End Sub
