Ahh, This seems to be a limitation on that type of form generation. I just tested it and it creates a tablular form with equal sized controls. There is a limit to the width of a form at 22 inches. If the number of controls hits the 22 inch limit, then it ignores the rest of the fields.

Method #1

  1. Split the function using any operator and test if the content is a range.
  2. Apply a test on each part to detect a valid "Range" using a function like "IsRange"
  3. Add the sheetname if the name is not already there like .Value2 = "='" & shName & "'!" & rng.Address

Formula : $B22-3+S6

myArray(1) = $B22 'IsRange($B22) = True

myArray(2) = "-"

myArray(3) = 3

Public Function IsRange(ByVal sRangeAddress As String) As Boolean
Dim TestRange As Range
IsRange = True
On Error Resume Next
Set TestRange = Range(sRangeAddress)
If Err.Number <> 0 Then
IsRange = False
End If
Err.Clear
On Error GoTo 0
Set TestRange = Nothing
End Function

Method #2

  1. Find all precedent cells that is included in the formula.
  2. Find a way to match a value in .FormulaArray like $B22 with the value of the precedent cell that is displayed: 'Feuil1'!$B$22.

Using a function like:

Sub FindPrecedents()
    ' written by Bill Manville
    ' With edits from PaulS
    ' this procedure finds the cells which are the direct precedents of the active cell
    Dim rLast As Range, iLinkNum As Integer, iArrowNum As Integer
    Dim stMsg As String
    Dim bNewArrow As Boolean
    Application.ScreenUpdating = False
    ActiveCell.ShowPrecedents
    Set rLast = ActiveCell
    iArrowNum = 1
    iLinkNum = 1
    bNewArrow = True
    Do
        Do
            Application.Goto rLast
            On Error Resume Next
            ActiveCell.NavigateArrow TowardPrecedent:=True, ArrowNumber:=iArrowNum, LinkNumber:=iLinkNum
            If Err.Number > 0 Then Exit Do
            On Error GoTo 0
            If rLast.Address(external:=True) = ActiveCell.Address(external:=True) Then Exit Do
            bNewArrow = False
            If rLast.Worksheet.Parent.Name = ActiveCell.Worksheet.Parent.Name Then
                If rLast.Worksheet.Name = ActiveCell.Parent.Name Then
                    ' local
                    stMsg = stMsg & vbNewLine & Selection.Address
                Else
                    stMsg = stMsg & vbNewLine & "'" & Selection.Parent.Name & "'!" & Selection.Address
                End If
            Else
                ' external
                stMsg = stMsg & vbNewLine & Selection.Address(external:=True)
            End If
            iLinkNum = iLinkNum + 1  ' try another link
        Loop
        If bNewArrow Then Exit Do
        iLinkNum = 1
        bNewArrow = True
        iArrowNum = iArrowNum + 1  'try another arrow
    Loop
    rLast.Parent.ClearArrows
    Application.Goto rLast
    MsgBox "Precedents are" & stMsg
    Exit Sub
End Sub

Table2 is shown on the same sheet below but can be wherever you want

1 - Format your Countries list as a Table and name it i.e. Table2 > Data tab > Get Data > From Table/Range

2 - Once in the Power Query editor > Home tab > Advance Editor > select all > copy/paste the following M code:

EDITED: Code updated (added step in bold) for a more accurate match in case of Country names like:

  • Netherlands
  • Netherlands Antilles

let
    Source = Excel.CurrentWorkbook(){[Name="Table2"]}[Content],
    #"Changed Type" = Table.TransformColumnTypes(Source,{{"Countries", type text}}),
    #"Removed Blank Rows" = Table.SelectRows(#"Changed Type", each
 not List.IsEmpty(List.RemoveMatchingItems(Record.FieldValues(_), {"", null}))),
    #"Trimmed Text" = Table.TransformColumns(#"Removed Blank Rows",{{"Countries", Text.Trim, type text}}),
    #"Removed Duplicates" = Table.Distinct(#"Trimmed Text"),
    #"Added Custom" = Table.AddColumn(#"Removed Duplicates", "Length", each
 Text.Length([Countries]), Int64.Type),
    #"Sorted Rows" = Table.Sort(#"Added Custom",{{"Length", Order.Descending}}),
    #"Removed Other Columns" = Table.SelectColumns(#"Sorted Rows",{"Countries"})
in
    #"Removed Other Columns"

3 - Close & Load To… > Connection only

4 - Format your Destination list as a Table and name it i.e. Table1 > Data tab > Get Data > From Table/Range

5 - Once in the Power Query editor > Home tab > Advance Editor > select all > copy/paste the following M code:

let

   Source = Excel.CurrentWorkbook(){[Name="Table1"]}[Content],
   #"Changed Type" = Table.TransformColumnTypes(Source,{{"Destination", type text}}),
   #"Trimmed Text" = Table.TransformColumns(#"Changed Type",{{"Destination", Text.Trim, type text}}),
   Table2Buffered = Table.Buffer(Table2),
   AddedColumn = Table.AddColumn(#"Trimmed Text", "MatchCountry",
       (i)=> Table.SelectRows(Table2Buffered,
          (j) => Text.StartsWith(i[Destination], j[Countries], Comparer.OrdinalIgnoreCase)
         )
   ),
   // In case > 1 record match keep the 1st
   FirstRow = Table.TransformColumns(AddedColumn, {"MatchCountry", each Table.FirstN(_,1)}),
   #"Removed Other Columns" = Table.SelectColumns(FirstRow,{"MatchCountry"}),
   #"Expanded Match" = Table.ExpandTableColumn(#"Removed Other Columns", "MatchCountry",
      {"Countries"}, {"Country"}),
   #"Changed Type1" = Table.TransformColumnTypes(#"Expanded Match",{{"Country", type text}})
in
   #"Changed Type1"
