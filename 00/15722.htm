ken sheridan say: As it happens, I have recently put together a little demonstration file for another user, which does what you are asking.  You'll find the file as gcreed.accdb in my public databases folder at:

https://onedrive.live.com/?cid=44CC60D7FEA42912&id=44CC60D7FEA42912!169

Note that if you are using an earlier version of Access you might find that the colour of some form objects such as buttons shows incorrectly and you will need to  amend the form design accordingly.  

If you have difficulty opening the link, copy the link (NB, not the link location) and paste it into your browser's address bar.

Click on the Transfers button to open the relevant form.  The code in the Click event of the Confirm Transfer button  inserts two rows into the Transfers table by executing two INSERT INTO statements within a Transaction.  Before doing so it also checks that the quantity to be transferred is currently in the 'from' location (warehouse in this case) is sufficient:

Private Sub cmdConfirm_Click()

    On Error GoTo Err_Handler
    
    Const MESSAGE_TEXT = "Insufficient items currently in stock in selected warehouse."
    Dim intStockLevel As Integer
    Dim strSQL As String
    Dim strCriteria As String
    Dim lngTransferID As Long
    
    ' get next TransferID value
    lngTransferID = Nz(DMax("TransferID", "Transfers"), 0) + 1
    
    strCriteria = "ItemID = " & Me.cboItem & " And WarehouseID = " & Me.cboTransferFrom
    intStockLevel = DLookup("TotalQuantity", "qryWarehouseItems", strCriteria)
    
    If intStockLevel - Me.txtQuantity < 0 Then
        MsgBox MESSAGE_TEXT, vbExclamation, "Invalid Operation"
        Exit Sub
    End If
    
    DBEngine.BeginTrans
        
    ' insert row into Transfers for warehouse from which item is transferred,
    ' inserting a negative quantity
    strSQL = "INSERT INTO Transfers(ItemID,WareHouseID,TransferID,TransferDate,QuantityTransferred) " & _
        "VALUES(" & Me.cboItem & "," & Me.cboTransferFrom & "," & lngTransferID & ",#" & _
        Format(Me.txtTransferDate, "yyyy-mm-dd") & "#,-" & Me.txtQuantity & ")"
        
    CurrentDb.Execute strSQL, dbFailOnError
    
    ' insert row into Transfers for warehouse to which item is transferred,
    ' inserting a positive quantity
    strSQL = "INSERT INTO Transfers(ItemID,WareHouseID,TransferID,TransferDate,QuantityTransferred) " & _
        "VALUES(" & Me.cboItem & "," & Me.cboTransferTo & "," & lngTransferID & ",#" & _
        Format(Me.txtTransferDate, "yyyy-mm-dd") & "#," & Me.txtQuantity & ")"
        
    CurrentDb.Execute strSQL, dbFailOnError
        
    DBEngine.CommitTrans
    
    MsgBox Me.txtQuantity & " items transferred.", vbInformation, "Transfer Confirmed"
    
Exit_Here:
    Exit Sub
    
Err_Handler:
    MsgBox Err.Description, vbExclamation, "Error"
    DBEngine.Rollback
    Resume Exit_Here

End Sub
