Declare x and y as Long if they should be dates, or as Double if they should be date+time values. Don't apply a format.

 

Sub Myfilter()
    Dim x As Double
    Dim y As Double
    'Output is the sheet the user sees
    x = Sheets("Output").DTPicker1.Value '
    y = Sheets("Output").DTPicker2.Value
    'display x, y and vessel in cells to check format and values
    Sheets("output").Range("K5").Value = x
    Sheets("output").Range("K6").Value = y
    ' All_info is the table with all the imported data
    ' and column 11 contains the date and time in  ("dd/mm/yyyy hh:mm")
    Sheets("Info_sheet").ListObjects("All_info").Range.AutoFilter Field:=11
    Sheets("Info_sheet").ListObjects("All_info").Range.AutoFilter Field:=11, _
        Criteria1:=">=" & x, Operator:=xlAnd, Criteria2:="<=" & y
End Sub
