hansv mvp say:
Unfortunately, setting ScreenUpdating to False is far from perfect. You generally won't see changes within a worksheet while it is in force, but showing, hiding and switching worksheets will still cause the screen to flicker.

andreas killer say:

Am 19.01.2011 22:14, schrieb Ordnance1:

Can anyone tell me why with Application.ScreenUpdating = False set when the
code runs you see it cycle through each worksheet?

Try the windows API instead.

Andreas.

Option Explicit

Private Declare Function LockWindowUpdate Lib "user32" ( _
    ByVal hwndLock As Long) As Long
Private Declare Function FindWindow Lib "user32" Alias "FindWindowA" ( _
    ByVal lpClassName As String, ByVal lpWindowName As String) As Long

Const WelcomePage = "Macros"

Sub HideAllSheets()
  Dim hWnd As Long
  Dim ws As Worksheet
  'Find the excel window
  hWnd = FindWindow(vbNullString, Application.Caption)
  'Lock the window update
  LockWindowUpdate hWnd
  'Do your work
  Worksheets(WelcomePage).Visible = xlSheetVisible
  For Each ws In ThisWorkbook.Worksheets
    If Not ws.Name = WelcomePage Then ws.Visible = xlSheetVeryHidden
  Next ws
  Worksheets(WelcomePage).Activate
  'Unlock the window update
  LockWindowUpdate False
End Sub

Option Explicit

Private Declare Function LockWindowUpdate Lib "user32" ( _
    ByVal hwndLock As Long) As Long
Private Declare Function FindWindow Lib "user32" Alias "FindWindowA" ( _
    ByVal lpClassName As String, ByVal lpWindowName As String) As Long

Const WelcomePage = "Macros"

Sub HideAllSheets()
  Dim hWnd As Long
  Dim ws As Worksheet
  'Find the excel window
  hWnd = FindWindow(vbNullString, Application.Caption)
  'Lock the window update
  LockWindowUpdate hWnd
  'Do your work
  Worksheets(WelcomePage).Visible = xlSheetVisible
  For Each ws In ThisWorkbook.Worksheets
    If Not ws.Name = WelcomePage Then ws.Visible = xlSheetVeryHidden
  Next ws
  Worksheets(WelcomePage).Activate
  'Unlock the window update
  LockWindowUpdate False
End Sub
