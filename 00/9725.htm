Function fncIsInDelimitedList( _

                TheList As String, _

    ' and a value, with optional arguments specifying the delimiter
    ' and whether partial-match logic is in effect.
    ' Returns True if the value is found in the delimited list,
    ' False if not.
    '
    ' Arguments:
    '
    '   TheValue -
    '       This is the value to be looked for in the list.
    '       Non-text values will be converted to text for comparison.
    '
    '   TheList -
    '       This is a string containing a delimited list of values to
    '       be searched for the value.
    '
    '   Delim -  (Optional)
    '       This is a string specifying the delimiter used in the list.
    '       Default value is a comma (",").
    '
    '   PartialMatch -  (Optional)
    '       This is a string boolean value specifying whether partial matches
    '       should be accepted.  If PartialMatch = True, then the function
    '       will return True if any of the list items is contained in TheValue.
    '       If PartialMatch = False, then the function will return True only
    '       if one of the list items exactly matches TheValue.

    ' Copyright ¬© Dirk Goldgar, 2012
    ' Permission: You may use this function in your applications and
    '       distribute it freely, provided that the copyright notice
    '       remains unchanged.

    On Error GoTo Err_Handler

    Static strOriginalList As String
    Static strList As String
    Static astrListValues() As String
    Static strOriginalDelim As String
    Static strDelim As String
   
    Dim strValue As String
    Dim strDelimPadRight As String
    Dim strDelimPadLeft As String
    Dim I As Long

        Exit Function

    ' If the list (or delimiter) has changed, parse to set up a new list.
    If (TheList <> strOriginalList) Or (Delim <> strOriginalDelim) Then
   
        ' Make working copies of the list and the delimiter.
        strOriginalList = TheList
        strList = TheList
        strOriginalDelim = Delim
        strDelim = Delim
       
        astrListValues = Split(strOriginalList, strDelim)
       
        ' Remove any spaces that may surround the delimited values.
        For I = LBound(astrListValues) To UBound(astrListValues)
            astrListValues(I) = Trim$(astrListValues(I))
        Next I
   
    End If

    ' Okay, now we have a list of values parsed into the array, astrListValues.
    ' Is the value we were passed in the list.
   
    For I = LBound(astrListValues) To UBound(astrListValues)
        If PartialMatch Then
            If TheValue Like "*" & astrListValues(I) & "*" Then
                fncIsInDelimitedList = True
                Exit Function
            End If
        End If
        If TheValue = astrListValues(I) Then
            fncIsInDelimitedList = True
            Exit Function
        End If
    Next I
   
Exit_Point:
    Exit Function
   
Err_Handler:
    fncIsInDelimitedList = CVErr(Err.Number)
    Resume Exit_Point

End Function

 

Function fncIsSelectedInListbox( _
                pListboxRef As String, _
                pValue As Variant) _
            As Variant

    ' Accepts a string that can be interpreted as a reference to a list box,
    ' and a value.  Returns True if the value is selected in the list box,
    ' False if not.  Note that it's the bound column of the list box that is
    ' tested for the value, not necessarily the displayed column.
    '
    ' Arguments:
    '
    '   pListboxRef -
    '       This is a string of the form "FormName!ListboxName", or
    '       "FormName!SubformName!ListboxName",
    '       or "Formname!SubformName!SubSubFormName!ListboxName", etc.
    '       Note that subform names must be provided as the names of the
    '       subform *controls* on their parent forms, as in normal
    '       form/subform references.
    '
    '   pValue -
    '       This is the value that will be sought in the selected items
    '       of the list box.

    '       distribute it freely, provided that the copyright notice
    '       remains unchanged.

    On Error GoTo Err_Handler
   
    Dim frm As Access.Form
    Dim lst As Access.ListBox
    Dim astrObjectName() As String
   
    Dim varRow As Variant
    Dim strValue As String
    Dim I As Integer
   
    fncIsSelectedInListbox = False
   
    If IsNull(pValue) Then Exit Function
   
    astrObjectName = Split(pListboxRef, "!")
    I = LBound(astrObjectName)
    Set frm = Forms(astrObjectName(I))
    I = I + 1

        Set frm = frm(astrObjectName(I)).Form
        I = I + 1
    Wend

        If lst.ItemData(varRow) = strValue Then
            fncIsSelectedInListbox = True
            Exit For
        End If
    Next varRow
   
Exit_Point:
    Set lst = Nothing
    Set frm = Nothing
    Exit Function
   
Err_Handler:
    fncIsSelectedInListbox = CVErr(Err.Number)
    Resume Exit_Point
   
End Function

dirk goldgar say:

I should probably add that this approach to filtering -- having the query call VBA functions to evaluate criteria for every record in the query -- will inevitably be *extremely* slow compared to building the criteria into a SQL string and then executing it.  The main reason for that is that indexes can't be used, plus the overhead of calling into VBA.

Even though I went to the trouble to build the functions, mostly as an intellectual exercise (although they could come in handy for quick queries of relatively small datasets), I suspect you will find the performance too slow for the large set of records you say you have.  A much better approach is to use code to build a SQL statement in a string variable, embedding the criteria values the user has selected directly into the SQL string, and then assigning that string to the RecordSource property of a form, or to the RowSource property of a list box.  I can provide example code for this process, too.

dirk goldgar say:

I don't know how much this will help you, but here is a function I use in in a filter form in one of my applications.  It's somewhat abstract, because the same form may be used to filter either of two different tables, with different field names corresponding to the logical entities to be filtered.  Also, it doesn't use any list boxes for criteria, only combo boxes, text boxes, and option groups.  But it may give you an idea of how one can build up a filter string from various controls on a form.

 

'------ start of code ------
Public Const Q As String = """"
Public Const QQ As String = Q & Q
Public Const JET_DATE As String = "\#mm\/dd\/yyyy\#"

Public Function FilterString() As String
   
    ' This function assembles and returns a filter string from the
    ' values of the various criteria controls on this form.  If no
    ' filter criteria are specified, then a zero-length string is
    ' returned. The names of the fields to be filtered are taken from
    ' the .Tag properties of the criteria controls, which may have
    ' been set in this form's open event, based on the OpenArgs.
    ' If those field names are not appropriate, the calling procedure
    ' will have to transform them in the returned filter string before
    ' applying it.
   
    On Error GoTo Err_Handler
   
    Dim strFilter As String
    Dim strConj As String
    Dim strFieldName As String
   
    If Me.optAndOr = Me.optAnd.OptionValue Then
        strConj = " AND "
    Else
        strConj = " OR "
    End If
   
    ' Build date filter, if any.
    strFieldName = Me.txtDateFrom.Tag
    If Not IsNull(Me.txtDateFrom) And Not IsNull(Me.txtDateTo) Then
        ' A full date range was specified.
        strFilter = strFilter & strConj & _
            "(" & strFieldName & " Between " & fncSQLDate(Me.txtDateFrom) & _
            " And " & fncSQLDate(Me.txtDateTo) & ")"
    ElseIf Not IsNull(Me.txtDateFrom) Then
        strFilter = strFilter & strConj & _
            "(" & strFieldName & " >= " & fncSQLDate(Me.txtDateFrom) & ")"
    ElseIf Not IsNull(Me.txtDateTo) Then
        strFilter = strFilter & strConj & _
            "(" & strFieldName & " <= " & fncSQLDate(Me.txtDateTo) & ")"
    End If
   
    ' Build client filter, if any.
    With Me.cboClient
        If Nz(.Value, "(all clients)") <> "(all clients)" Then
            strFilter = strFilter & strConj & _
                "(" & .Tag & " = " & fncSQLText(.Value) & ")"
        End If
    End With
   
    ' Build exec filter, if any.
    With Me.cboExec
        If Nz(.Value, "(all execs)") <> "(all execs)" Then
            strFilter = strFilter & strConj & _
                "(" & .Tag & " = " & fncSQLText(.Value) & ")"
        End If
    End With
   
    ' Build include/exclude filter, if any.
    With Me.optIncludeExclude
        Select Case .Value
            Case Me.optIncludeExcludeAll.OptionValue
                ' no filter
            Case Me.optIncludeExcludeIncluded.OptionValue
                strFilter = strFilter & strConj & _
                    "(" & .Tag & " = 'Include')"
            Case Me.optIncludeExcludeExcluded.OptionValue
                strFilter = strFilter & strConj & _
                    "(" & .Tag & " = 'Exclude')"
        End Select
    End With
   
    ' Build category filter, if any.
    With Me.cboCategory
        If Nz(.Value, "(all categories)") <> "(all categories)" Then
            strFilter = strFilter & strConj & _
                "(" & .Tag & " = " & fncSQLText(.Value) & ")"
        End If
    End With
   
    ' Build project filter, if any.
    With Me.cboProject
        If Nz(.Value, "(all projects)") <> "(all projects)" Then
            strFilter = strFilter & strConj & _
                "(" & .Tag & " = " & fncSQLText(.Value) & ")"
        End If
    End With
   
    ' The filter string is complete.  Trim off the leading
    ' conjunction and return it.
    If Len(strFilter) > 0 Then
        FilterString = Mid$(strFilter, Len(strConj) + 1)
    End If

Exit_Point:
    Exit Function
   
Err_Handler:
    DisplayAndLogError "FilterString", Err.Number, Err.Description
    Resume Exit_Point

End Function

Public Function fncSQLDate(ArgValue As Variant) As String

    If Len(ArgValue & "") = 0 Then
        fncSQLDate = "Null"
    Else
        fncSQLDate = Format(ArgValue, JET_DATE)
    End If

End Function


Public Function fncSQLText(ArgValue As Variant) As String

    If IsNull(ArgValue) Then
        fncSQLText = "Null"
    Else
        fncSQLText = _
            Q & _
            Replace(ArgValue, Q, QQ, , , vbBinaryCompare) & _
            Q
    End If

End Function

hansv mvp say:

Let's say the column header "Correct Answer Text" is in A1.

In A2, enter the formula

=LOOKUP("(X)*",A2:G2)

Fill down as far as needed. If there is no cell that begins with (X), the formula will return #N/A ("not available"). If you'd rather leave column A blank in that situation, use

=IFERROR(LOOKUP("(X)*",A2:G2),"")
