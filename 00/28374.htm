rich007a say:

OK, I think I've got it.  Might be worth starting again from scratch.  Note the names of the controls have changed, please read the comments at the top.

 

Option Explicit
'
' Userform has the following controls:
' - ComboBox called "cbShelf"
' - ComboBox called "cbCols"
' - ComboBox called "cbLevels"
' - TextBox called "TextBox1"
' - CommandButton called "cbOK" (OPTIONAL)
' When the textbox is changed the contents of the text box is written to the
' cell who's column is indicated in the Cols combobox and who's row is indicated
' in the Levels combobox, relative to the range specified in.
'
Private lnRowTop As Long, lnRowBot As Long, lnColL As Long, lnColR As Long
 
Private Sub UserForm_Initialize()
    Me.cbShelf.AddItem "Shelf 1"
    Me.cbShelf.AddItem "Shelf 2"
    Me.cbShelf.AddItem "Shelf 3"
   
    Me.cbShelf.ListIndex = 0
End Sub
 
Private Sub cbShelf_Change()
    Dim i As Long
    Dim strColL As String, strColR As String
   
    Select Case Me.cbShelf.Value
        Case "Shelf 1"
            strColL = "B"
            strColR = "E"
            lnRowTop = 7
            lnRowBot = 11
        Case "Shelf 2"
            strColL = "F"
            strColR = "I"
            lnRowTop = 12
            lnRowBot = 16
        Case "Shelf 3"
            strColL = "J"
            strColR = "M"
            lnRowTop = 17
            lnRowBot = 21
    End Select
   
    lnColL = Range(strColL & ":" & strColL).Column
    lnColR = Range(strColR & ":" & strColR).Column
   
     'COLUMNS:
     'Remove all entries from the cb:
    For i = Me.cbCols.ListCount - 1 To 0 Step -1
        Me.cbCols.RemoveItem i
    Next i
    
     'Add new entries to the cb:
    For i = 1 To lnColR - lnColL + 1
        Me.cbCols.AddItem "Column " & i
    Next i
    Me.cbCols.ListIndex = 0
   
     'ROWS:
     'Remove all entries from the cb:
    For i = Me.cbLevels.ListCount - 1 To 0 Step -1
        Me.cbLevels.RemoveItem i
    Next i
   
     'Add new entries to the cb:
    For i = 1 To lnRowBot - lnRowTop + 1
        Me.cbLevels.AddItem "Level " & i
    Next i
    Me.cbLevels.ListIndex = 0
End Sub
 
Private Sub cbCols_Change()
    Call UpdateTextbox
End Sub
 
Private Sub cbLevels_Change()
    Call UpdateTextbox
End Sub
 
Private Sub UpdateTextbox()
    Dim lnRow As Long, lnCol As Long
   
    If Me.cbCols.Value <> "" And Me.cbLevels.Value <> "" Then
        lnRow = lnRowTop + Me.cbLevels.ListIndex
        lnCol = lnColL + Me.cbCols.ListIndex
        Me.TextBox1.Text = Cells(lnRow, lnCol).Value
    End If
End Sub
 
Private Sub TextBox1_Change() 'or cbOK_Click()
    Dim lnRow As Long, lnCol As Long
   
    If Me.cbCols.Value <> "" And Me.cbLevels.Value <> "" Then
        lnRow = lnRowTop + Me.cbLevels.ListIndex
        lnCol = lnColL + Me.cbCols.ListIndex
        Cells(lnRow, lnCol).Value = Me.TextBox1.Text
    End If
End Sub

Are CDate() and DateValue() interchangeable

Original title: Dates

Are CDate() and DateValue() interchangeable?

andreas killer say:

Not exactly - guess which one of these doesn't work?

That depends on the language, in europe DateValue uses a Day/Month/Year interpretation, in america it's Month/Day/Year. And there are other countries that use other masks.

So be very carefully if you use strings with CDate / DateValue. If I have to read dates from other countries, I use always my own string conversation routine.

But the best way, if possible, is to use DateSerial.

Andreas.
