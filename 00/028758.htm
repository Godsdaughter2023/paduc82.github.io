Celia,

Also, you will need to set appWD to Nothing before Word will actually stop running (if you look in taskmanager WORD.EXE will remain even after you have used appWD.Quit):

 

    Set appWD = Nothing

 

What is not clear is whether you have declared your appWD object.  If you are using Early Binding, your code should look like this:


Sub DoingStuffWithWord()

' Early Binding - must set a reference to 'Microsoft Word Object Library' via VBE >Tools > References..

'

    Dim appWD As Word.Application

    Set appWD = CreateObject("Word.Application")
   
    'DO STUFF WITH WORD, CREATE DOCUMENT, ETC...
   
    appWd.ActiveDocument.ExportAsFixedFormat _
        OutputFileName:= "C:\Users\Celia\Documents\TEST4.pdf", _
        ExportFormat:=wdExportFormatPDF, _
        OpenAfterExport:=True, _
        OptimizeFor:=wdExportOptimizeForPrint, _
        Range:=wdExportAllDocument, _
        From:=1, _
        To:=1, _
        Item:=wdExportDocumentContent, _
        IncludeDocProps:=True, _
        KeepIRM:=True, _
        CreateBookmarks:=wdExportCreateNoBookmarks, _
        DocStructureTags:=True, _
        BitmapMissingFonts:=True,  _
        UseISO19005_1:=False
    
    appWD.ActiveDocument.Close SaveChanges:=False
    appWD.Quit
    Set appWD = Nothing
End Sub

OR with late binding:

Sub DoingStuffWithWord()

' Late Binding - (no need to set a reference)

'

    Dim appWD As Object

    Set appWD = CreateObject("Word.Application")
   
    'DO STUFF WITH WORD, CREATE DOCUMENT, ETC...
   
    appWd.ActiveDocument.ExportAsFixedFormat _
        OutputFileName:="C:\Users\Celia\Documents\TEST4.pdf", _
        ExportFormat:=17, _
        OpenAfterExport:=True, _
        OptimizeFor:=0, _
        Range:=0, _
        From:=1, _
        To:=1, _
        Item:=0, _
        IncludeDocProps:=True, _
        KeepIRM:=True, _
        CreateBookmarks:=0, _
        DocStructureTags:=True, _
        BitmapMissingFonts:=True, _
        UseISO19005_1:=False
    
    appWD.ActiveDocument.Close SaveChanges:=False
    appWD.Quit
    Set appWD = Nothing
End Sub

e.j. gun say:

Hans,

A slightly more intuitive approach might be to use the fill color of the cell, since this is very visual and does not require the user to translate color to a number.  Also, the code below corrects a minor issue where your code did not go through all the points in the Series.

Sub ColorPoints()

    Dim cht as Chart

    Dim ser as Series

    Dim pnt as Point

    Dim i as Long, j as Long

    Dim rng as Range

    Set cht = ActiveChart

    Set ser = cht.SeriesCollection(1)

    Set Rng = ActiveSheet.Range("A1:A8") ' Each of these cells has a different color

    j = 0

    For i = 1 to ser.Points.Count

        j = j + 1

        Set pnt = ser.Points(i)

        pnt.MarkerBackgroundColor = rng(j).Interior.Color ' Cycle through available colors

        If (j > rng.Count) Then j = 0

    Next i

End Sub

Step 0: as a precaution, make a backup copy of the workbook.

    Set rng = ActiveSheet.Range("A1:A8")

    Set rng = ActiveSheet.Range("C2:C15")
