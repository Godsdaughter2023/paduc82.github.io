hansv mvp say:

Here is a VBA procedure you can use:

Sub FindRanges(rng As Range)
    Dim r As Long
    Dim n As Long
    Dim r0 As Long
    If rng.Columns.Count > 1 Then Exit Sub
    Application.ScreenUpdating = False
    rng.Offset(0, 1).ClearContents
    n = rng.Rows.Count
    r0 = 1
    For r = 2 To n
        If rng(r).Value > rng(r - 1).Value + 1 Then
            If r - 1 = r0 Then
                rng(r0).Offset(0, 1).Value = rng(r0).Value
            Else
                rng(r0).Offset(0, 1).Value = rng(r0).Value & "-" & rng(r - 1).Value
            End If
            r0 = r
        End If
    Next r
    ' Last one
    If n = r0 Then
        rng(r0).Offset(0, 1).Value = rng(r0).Value
    Else
        rng(r0).Offset(0, 1).Value = rng(r0).Value & "-" & rng(n).Value
    End If
    Application.ScreenUpdating = True
End Sub

Let's say your zip codes are in A2:A7043. Run the following macro to fill B2:B7043:

Sub Test()
    Call FindRanges("A2:A7043")
End Sub
