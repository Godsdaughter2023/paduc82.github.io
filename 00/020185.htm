To loop through all cells, and hide columns of cells who's node is not checked, I cobbled together this:

Private Sub TreeView1_NodeCheck(ByVal Node As MSComctlLib.Node)
    Dim rng As Range, c As Range
    Set rng = Sheets("Quic View").Range("AA7:MX7")
    If Not Node.Parent Is Nothing Then '(only act on child nodes)
         'Loop through each cell:
        For Each c In rng
             'If cell's value matches combination of Parent and Child node then:
            If c.Value = Node.Text & " " & Node.Parent.Text Then
                c.EntireColumn.Hidden = Not Node.Checked 'Hide/show cell's column
            End If
        Next c
    End If
End Sub


I took your code and made it work for just the first TreeView.  Note how I added a .tag to each node in the Initialize event (below); all parents are .tag=1, all children are .tag=2 (if you had further levels, they would be .tag=3, etc).  This was the only way I could figure out how to keep track of which level a node was in, apart from testing If Not Node.Parent Is Nothing Then..., but that wouldn't work if you had multiple levels of children.


Option Explicit
Private UpdateParentCheck As Boolean

Private Sub TreeView1_NodeCheck(ByVal Node As MSComctlLib.Node)
'
    Dim rng As Range, c As Range, n As MSComctlLib.Node
   
    Application.ScreenUpdating = False
   
     'Set range of cells to be tested:
    Set rng = Sheets("Quic View").Range("AA7:MX7")
   
    If Node.Tag = 2 Then 'node is level 2
         'Loop through each cell:
        For Each c In rng
             'If cell's value matches combination of Parent and Child node then:
            If c.Value = Node.Text & " " & Node.Parent.Text Then
                c.EntireColumn.Hidden = Not Node.Checked 'Hide/show cell's column
            End If
        Next c
           
        If UpdateParentCheck Then
             'If the node was checked
            If Node.Checked Then
                 'Temporarily check parent
                Node.Parent.Checked = True
                 'Loop through each node...
                For Each n In Me.TreeView1.Nodes
                    If n.Tag = 2 Then
                         'And for each child of the parent:
                        If n.Parent = Node.Parent And Not n.Checked Then
                             'If any are unchecked, uncheck the parent
                            Node.Parent.Checked = False
                            Exit For
                        End If
                    End If
                Next n

            Else 'node was unchecked:
                Node.Parent.Checked = False 'uncheck the parent
            End If
        End If
       
    Else 'Node.Tag = 1 (i.e. parent node)
         'Set all child nodes to match checked status of parent:
        For Each n In Me.TreeView1.Nodes
            If n.Tag = 2 Then
                If n.Parent = Node Then
                     'Set child checked state to match parent:
                    n.Checked = Node.Checked
                    UpdateParentCheck = False
                    TreeView1_NodeCheck n 'Call the NodeCheck macro
                    UpdateParentCheck = True
                End If
            End If
        Next n
    End If
    
    Application.ScreenUpdating = True
End Sub

Private Sub UserForm_Initialize()
'
     'Load all nodes from a sheet into a tree, remove existing nodes in tree
    Dim Contents1 As String
    Dim Contents2 As String
    Dim Level1 As String
    Dim Level2 As String
    Dim Stack1() As Node
    Dim Stack2() As Node
   
    Dim Line1 As Long
    Dim Line2 As Long
    
     'Read the sheet line by line
'    On Error GoTo ExitPoint1
    ReDim Stack1(1 To 64) As Node
    ReDim Stack2(1 To 64) As Node
   
    Line1 = 1
    Line2 = 1
 
     'TreeView1
    Do While Not IsEmpty(Sheet1.Cells(Line1, 1))
         'Get the contents of the node
        Contents1 = Sheet1.Cells(Line1, 1)
        Level1 = Sheet1.Cells(Line1, 2)
        Line1 = Line1 + 1
       
        If Level1 > UBound(Stack1) Then
            ReDim Preserve Stack1(1 To UBound(Stack1) * 2) As Node
        End If
       
        If Level1 = 1 Then
             'This is a root node
            Set Stack1(Level1) = Me.TreeView1.Nodes.Add(Text:=Contents1)
            Stack1(Level1).Expanded = True
            Stack1(Level1).Tag = 1   'Add tag for easy tracking of level
        Else
             'This is a child node
            Set Stack1(Level1) = Me.TreeView1.Nodes.Add(Relative:=Stack1(Level1 - 1), _
                RelationShip:=tvwChild, Text:=Contents1)
            Stack1(Level1).Expanded = True
            Stack1(Level1).Tag = 2   'Add tag for easy tracking of level
        End If
        Stack1(Level1).Checked = True
    Loop
   
     'Enable Updating of parent check status:
    UpdateParentCheck = True
 
ExitPoint1:
End Sub

don cuthbert say:

When an upgrade to a new build fails, these steps may enable it to succeed: 

  1. Completely uninstall any non-Microsoft anti-virus software. 

  2. Perform a clean boot before upgrading. 

  3. Disconnect any un-needed peripherals, such as devices connected via USB. 

  4. Disable all network adapters after the build has downloaded but before it Restarts, or before running setup.exe, if you are upgrading via a mounted ISO or Windows 10 install media

Sometimes disconnecting the Internet, such as unplugging the Ethernet cable, will enable a stalled upgrade to complete. 

When Windows Update downloads a new build, it puts the main ESD file in a hidden folder at C:\$WINDOWS.~BT\Sources\Install.esd.  You can use this file to create a bootable USB or an ISO file for the new build, so that you don't have to keep re-downloading it.  If you are very short of space, this file may get deleted, but usually it remains.
