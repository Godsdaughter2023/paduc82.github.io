Function CompareFiles(ByVal FileName1 As String, ByVal FileName2 As String, _
    Optional ByVal Compare As VbCompareMethod = vbBinaryCompare, _
    Optional ByVal Size As Long) As Boolean
  'Compare the data of two files
  Dim FLen1 As Long, FLen2 As Long
  Dim FNum1 As Integer, FNum2 As Integer
  Dim FData1() As Byte, FData2() As Byte
  Dim FText1 As String, FText2 As String
  Dim i As Long
  On Error GoTo ExitPoint
  FLen1 = FileLen(FileName1)
  FLen2 = FileLen(FileName2)
  If FLen1 <> FLen2 Then Exit Function
  If Size <= 0 Then Size = FLen1
  FNum1 = FreeFile
  Open FileName1 For Binary Access Read As #FNum1
  FNum2 = FreeFile
  Open FileName2 For Binary Access Read As #FNum2
  For i = 1 To FLen1 Step Size
    FData1() = InputB(Size, #FNum1)
    FData2() = InputB(Size, #FNum2)
    If Compare = vbTextCompare Then
      FText1 = StrConv(FData1, vbUnicode)
      FText2 = StrConv(FData2, vbUnicode)
      If StrComp(FText1, FText2, vbTextCompare) <> 0 Then GoTo ExitPoint
    Else
      If StrComp(FData1, FData2, vbBinaryCompare) <> 0 Then GoTo ExitPoint
    End If
  Next
  CompareFiles = True
ExitPoint:
  Close #FNum1
  Close #FNum2
End Function

hansv mvp say:

You could run this macro:

Sub Transform()
    Dim wshS As Worksheet
    Dim wshT As Worksheet
    Dim s As Long
    Dim m As Long
    Dim c As Long
    Dim n As Long
    Dim t As Long
    Application.ScreenUpdating = False
    Set wshS = ActiveSheet
    m = wshS.Cells(wshS.Rows.Count, 1).End(xlUp).Row
    n = wshS.Cells(1, wshS.Columns.Count).End(xlToLeft).Column
    Set wshT = Worksheets.Add(After:=wshS)
    wshT.Cells(1, 1).Value = "Product"
    wshT.Cells(1, 2).Value = "Date"
    wshT.Cells(1, 3).Value = "Type"
    wshT.Cells(1, 4).Value = "Value"
    t = 1
    For c = 3 To n
        For s = 3 To m
            t = t + 1
            wshT.Cells(t, 1).Value = wshS.Cells(s, 1).Value
            wshT.Cells(t, 2).Value = wshS.Cells(1, c).Value
            wshT.Cells(t, 3).Value = wshS.Cells(2, c).Value
            wshT.Cells(t, 4).Value = wshS.Cells(s, c).Value
        Next s
    Next c
    wshT.Range("B2:B" & t).NumberFormat = wshS.Cells(1, 3).NumberFormat
    wshT.Range("A1:D1").EntireColumn.AutoFit
    Application.ScreenUpdating = True
End Sub

vijay a. verma say:

Put following Array Formula in C15 and drag right and down -

=IFERROR(INDEX($B$4:$B$12,SMALL(IF(C$4:C$12="",ROW($C$4:$C$12)-ROW($C$4)+1),ROWS($1:1))),"")

Note - Array Formula is not entered by pressing ENTER after entering your formula but by pressing CTRL+SHIFT+ENTER. If you are copying and pasting this formula, take F2 after pasting and CTRL+SHIFT+ENTER. This will put { } brackets around the formula which you can see in Formula Bar. If you edit again, you will have to do CTRL+SHIFT+ENTER again. Don't put { } manually.
