Try this small macro

Sub delete_Me()
Dim LastRow As Long, x As Long
LastRow = Cells(Rows.Count, "A").End(xlUp).Row
For x = LastRow To 1 Step -1
    If UCase(Cells(x, 1).Value) = "108US" And _

Next
End Sub

File saved with Date and Time in file name:   yx_yyyy mm dd hh mm ss.xlsx

Note that if you use date and time then must be in yyyy mm dd hh mm ss format with year first and use 2 digits for months, days, hours, minutes and seconds otherwise they will not correctly sort in ascending order.

The workbooks do not have to be saved with the date and time. You can use any numbering method you like so long as the number is formatted with the same number of characters and smaller numbers are padded out with leading zeros. eg you could use 0001 to 9999 if you like but all must be 4 characters in length so workbook name is yx_0001.xlsx

Sub GetLatestFileMatch()
    Dim strPath As String
    Dim strFileFilter As String
    Dim strNextFile As String
    Dim arrFiles() As Variant
    Dim k As Long
    Dim j As Long
    Dim strTemp As String
    Dim strMostRecentFile As String
   
    strPath = ThisWorkbook.Path & "\" & "FilterTest\"
    strFileFilter = "yx_*.xlsx"
   
    k = 1   'Initialize to one
    ReDim arrFiles(1 To k)  'Initialise array with one element so that ReDim Preserve will work
   
    strNextFile = Dir(strPath & strFileFilter)
   
    Do
        If strNextFile <> "" Then
            ReDim Preserve arrFiles(1 To k)  'Note number of elements does not increment on first iteration
            arrFiles(k) = strNextFile
            k = k + 1
        Else
            Exit Do
        End If
        strNextFile = Dir()
        'Next line is safety to prevent eternal loop during testing
        'Edit 100 to a number greater than likely number of files
        'When finished testing then the line can be deleted
        If k > 100 Then Exit Do
    Loop
   
    'Following for testing only. Delete when no longer required
    'In the VBA Editor use Ctrl and K to view the immediate window with results
    'Select all results and delete before re-running the code.
    For k = LBound(arrFiles) To UBound(arrFiles)
        Debug.Print arrFiles(k)
    Next k
   
    'Sort the files in ascending order so can find last file
    For k = LBound(arrFiles) To UBound(arrFiles) - 1
        For j = LBound(arrFiles) To UBound(arrFiles) - 1
            If arrFiles(j) > arrFiles(j + 1) Then
                strTemp = arrFiles(j)
                arrFiles(j) = arrFiles(j + 1)
                arrFiles(j + 1) = strTemp
            End If
        Next j
    Next k
           
    'Following assigns the last file in the array to a string variable
    strMostRecentFile = arrFiles(UBound(arrFiles))
      
    MsgBox "Most recent file is " & strMostRecentFile

End Sub
