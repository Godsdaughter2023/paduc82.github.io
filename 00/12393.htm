Comment out the line:

   .Send

and add the line:

   .Display

then step through the code until you get there.  .Display displays the created message (you might need to click on it in the task bar to bring it in front of Excel/VBA).   When I commented out everything other than the code below, it displayed an e-mail message that with the correct e-mail address:

Sub Emailer()
'Working in 2000-2010
    Dim FileExtStr As String
    Dim FileFormatNum As Long
    Dim Sourcewb As Workbook
    Dim Destwb As Workbook
    Dim TempFilePath As String
    Dim TempFileName As String
    Dim OutApp As Object
    Dim OutMail As Object
    Dim emailname As String
   
    emailname = Worksheets("PARAMETERS").Range("$B$48").Value

    Set OutApp = CreateObject("Outlook.Application")
    Set OutMail = OutApp.CreateItem(0)

    On Error Resume Next
    With OutMail
        .to = emailname
      '   .CC = ""
      '  .BCC = ""
        .Subject = "Time spread warning from Super Spreads"
        .Body = "A threshhold has been meet to generate this email.  Certain timespreads may be out of line."
        '.Attachments.Add Destwb.FullName
        'You can add other files also like this
        '.Attachments.Add ("C:\test.txt")
        .Display
        '.Send   'or use .Display
    End With
    On Error GoTo 0

    Set OutMail = Nothing
    Set OutApp = Nothing
End Sub
