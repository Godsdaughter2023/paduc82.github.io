andreas killer say:

In addition:

You can draw small rectangles between the points in the chart and color them as already shown.

In that way the chart looks like as it has gradient lines.

I don't have added you special color coding, the code below shows just the basics.

Andreas.

Option Explicit

Sub Test()
  Const LineWidth = 2
  Dim C As Chart
  Dim S As Series
  Dim Pt1 As Point, Pt2 As Point
  Dim i As Long
  Dim Sh As Shape
  Dim Dist As Double
  Dim X As Double, Y As Double, A As Double, W As Double

  Set C = ActiveSheet.ChartObjects(1).Chart
 
  'Delete all shapes in the chart that we have created
  For Each Sh In C.Shapes
    Sh.Delete
  Next
 
  Set S = C.SeriesCollection(1)
  'We don't need any format or the line
  S.ClearFormats
  S.Format.Line.Visible = msoFalse

  Set Pt2 = S.Points(1)
  'A circle as marker looks nice
  Pt2.MarkerStyle = xlMarkerStyleCircle
  Pt2.MarkerSize = 8

  For i = 2 To S.Points.Count
    Set Pt1 = Pt2
    Set Pt2 = S.Points(i)
    'All markers have to be equal
    Pt2.MarkerStyle = Pt1.MarkerStyle
    Pt2.MarkerSize = Pt1.MarkerSize

    'Calculate the size of the rectangle
    X = Pt2.Left - Pt1.Left
    Y = Pt2.Top - Pt1.Top
    W = Sqr(X ^ 2 + Y ^ 2) - Pt1.MarkerSize
    A = ArcTanG2(X, Y)
   
    'Create it in the middle of the chart (otherwise it might be cut off)
    Set Sh = C.Shapes.AddShape(msoShapeRectangle, _
      C.Parent.Width / 2 - W / 2, C.Parent.Height / 2 - LineWidth / 2, W, LineWidth)
    With Sh
      'Rotate and move it in the middle of the points
      .Line.Visible = msoFalse
      .Rotation = A
      .Top = Pt1.Top + Y / 2 - .Height / 2
      .Left = Pt1.Left + X / 2 - .Width / 2

      'Create the colors
      .Fill.TwoColorGradient msoGradientVertical, 1
      '.Fill.GradientStops(1).Color.RGB = RGB(255, 0, 0) <= doesn't work in XL2010!?
      .Fill.GradientStops.Insert vbRed, 0 'this does!
      .Fill.GradientStops(2).Color.RGB = vbGreen
    End With
  Next
End Sub

Function ArcTanG2(ByVal X As Double, ByVal Y As Double) As Double
  'Gibt den Arcus-Tangens von Y/X in Grad in allen Quadranten zurück
  ArcTanG2 = Grad(ArcTan2(X, Y))
End Function

Function ArcTan2(ByVal X As Double, ByVal Y As Double) As Double
  'Gibt den Arcus-Tangens von Y/X in allen Quadranten zurück
  If X = 0 And Y = 0 Then
    ArcTan2 = 0
  Else
    If X = 0 Then X = 1E-16
    ArcTan2 = Atn(Y / X) - IIf(X < 0, IIf(Y > 0, -Pi, Pi), 0)
  End If
End Function

Function Grad(ByVal X As Double) As Double
  'Wandelt Radiant in Grad
  Grad = X * 180 / Pi
End Function

Function Pi() As Double
  'Gibt die Zahl Pi zurück
  Pi = Atn(1) * 4
End Function
