=((L5>0)*L5+(AE5>0)*AE5+(AW5>0)*AW5+(BQ5>0)*BQ5+(CK5>0)*CK5)/((L5>0)*1+(AE5>0)*1+(AW5>0)*1+(BQ5>0)*1+(CK5>0)*1)
