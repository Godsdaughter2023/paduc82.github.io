I don't need general find and replace help.

How do I do this?

john sr wilson say:

Might be tricky because PPT includes spaces in words and you may not need spaces underlined. Try this but it is just code of the top of my head and not tested. It should underline the words but it will not underline a run of words including spaces.

Sub to_UL_some()
   Dim osld As Slide
   Dim oshp As Shape
   Dim R As Long
   Dim otr2 As TextRange2
   Dim otr2Strip As TextRange2
   For Each osld In ActivePresentation.Slides
      For Each oshp In osld.NotesPage.Shapes
         If oshp.Type = msoPlaceholder Then
            If oshp.PlaceholderFormat.Type = 2 Then
               If oshp.TextFrame2.HasText Then
                  For R = 1 To oshp.TextFrame2.TextRange.Words.Count
                     Set otr2 = oshp.TextFrame2.TextRange.Words(R)
                     Select Case otr2.Characters(otr2.Characters.Count)
                    Case Is = " "
                        Set otr2Strip = otr2.Characters(1, otr2.Characters.Count - 1)
                     Case Else
                        Set otr2Strip = otr2
                     End Select
                     Select Case UCase(otr2Strip)
                     Case Is = "WORD", "THIS", "SAMPLE"   'these are the search words
                        'add more seperated by comma - use upper case
                        With otr2Strip
                           If .Font.Bold Then
                              .Font.Bold = False
                              .Font.UnderlineStyle = msoUnderlineSingleLine
                           End If
                        End With
                     End Select
                  Next R
               End If
               End If
            End If
         Next oshp
      Next osld
   End Sub
