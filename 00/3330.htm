ossiemac say:

A 'fresh start' perspective from this community is sought.

I am interpreting the above that you only need to be pointed in the right direction.

The code below loops through one table and finds the matching column header in another table and appends the data to the bottom of each of the matching columns.

Data from Table2 in Sheet2 is appended to the bottom of Table1 in Sheet1

A message box displays a list of any columns not found in the destination table.

There is an example workbook at the following link.

http://1drv.ms/1yVTVJr

Code below. Edit Worksheet names and Table names towards the top of the code.

In the example, the data in the Source worksheet (Sheet2) has the column Id's incorporated so that you will be able to see that they have appended to the correct column in the Destination (Sheet1)

If the data in the Source is not in a Table then you could loop through the column headers in a similar manner and then search for the matching column in the Destination table. Alternatively make the Source a table also.

Run the sub MatchTableColumns.

Sub MatchTableColumns()
    Dim ws1 As Worksheet
    Dim ws2 As Worksheet
    Dim lstObj1 As ListObject
    Dim lstObj2 As ListObject
    Dim lstCol As ListColumn
    Dim rngCol As Range
    Dim rngDestin As Range
    Dim lngLastRow As Long
    Dim strMissCols As String
   
    Set ws1 = Worksheets("Sheet1")      'Destination sheet
   
    Set ws2 = Worksheets("Sheet2")      'Source sheet
   
    Set lstObj1 = ws1.ListObjects("Table1")     'Destination table
   
    Set lstObj2 = ws2.ListObjects("Table2")     'Source table
   
    'Call UDF to find the last row (Plus 1) of the Destination range
    lngLastRow = LastRowOrCol(ws1, True, lstObj1.Range) + 1
   
    With lstObj2
        For Each lstCol In lstObj2.ListColumns  'Each column in Table2
           
            'Call UDF to find matching column header in Table1
            Set rngCol = FindCol(lstObj1, lstCol.Name)
           
            If Not rngCol Is Nothing Then   'Not nothing then Col Header found
                With lstObj1.Parent     'The worksheet containing Table1
                   
                    'Destination is Next blank row in required column in Table1
                    Set rngDestin = .Cells(lngLastRow, rngCol.Column)
                   
                    'Copy the column from Table2 to bottom of Table1
                    lstCol.DataBodyRange.Copy Destination:=rngDestin
                End With
            Else
                'Create string of any column not found in destination
                strMissCols = strMissCols & vbCrLf & lstCol.Name
            End If
        Next lstCol
    End With
   
    If strMissCols <> "" Then   'Missing column/s
        'Insert meaningful comment before list of missing columns
        strMissCols = "Columns not found in destination:" & strMissCols
        MsgBox strMissCols
    End If
       
End Sub


Function FindCol(lstObj As ListObject, strName As String) As Range

    Set FindCol = Nothing
    With lstObj.HeaderRowRange
        Set FindCol = .Find(What:=strName, _
                        LookIn:=xlFormulas, _
                        LookAt:=xlWhole, _
                        SearchOrder:=xlByColumns, _
                        SearchDirection:=xlNext, _
                        MatchCase:=False)
    End With
   
End Function

Function LastRowOrCol(ws As Worksheet, bolRowCol As Boolean, Optional rng As Range) As Long
    'Finds the last used row or column in a worksheet
    'First parameter is Worksheet
    'Second parameter is True for Last Row or False for last Column
    'Third parameter is optional. Use to find the last row or column in a specific range
   
    Dim lngRowCol As Long
    Dim rngToFind As Range
   
    If rng Is Nothing Then
        Set rng = ws.Cells
    End If
   
    If bolRowCol Then
        lngRowCol = xlByRows
    Else
        lngRowCol = xlByColumns
    End If
   
    With ws
        Set rngToFind = rng.Find(What:="*", _
                LookIn:=xlFormulas, _
                LookAt:=xlPart, _
                SearchOrder:=lngRowCol, _
                SearchDirection:=xlPrevious, _
                MatchCase:=False)
    End With
   
    If Not rngToFind Is Nothing Then
        If bolRowCol Then
            LastRowOrCol = rngToFind.Row
        Else
            LastRowOrCol = rngToFind.Column
        End If
    End If
   
End Function
