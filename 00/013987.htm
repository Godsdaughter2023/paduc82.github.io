gord dibben say:

How about assign a macro to each checkbox to toggle row color on/off?

With your workbook open hit Alt + F11 to go to Visual Basic Editor.
CTRL + r to open Project Explorer if you don't see it.
Select your workbook/project and right-click>insert>module.
Paste the macro into that module.
Alt + q to return to the Excel window.

With your worksheet active, F5>Special>Objects>OK

Right-click on one of the selected objects taking care to leave all selected.

Assign Macro>Toggle_Row_Color

Sub Toggle_Row_Color()
    Dim ws As Worksheet
    Dim chk As CheckBox
    Dim lColChk As Long
    Dim lRow As Long
    Dim rngD As Range

    Set ws = ActiveSheet
    Set chk = ws.CheckBoxes(Application.Caller)
    lRow = chk.TopLeftCell.Row
    lColChk = chk.TopLeftCell.Column
    Set rngD = ws.Cells(lRow, lColChk)

    Select Case chk.Value
        Case 1                                           'box is checked
            rngD.EntireRow.Interior.ColorIndex = 8
        Case Else                                        'box is not checked
            rngD.EntireRow.Interior.ColorIndex = xlNone
    End Select
End Sub

    oTbl.LeftPadding = 0
 Next
