lindasueb say:

After much struggling, and contacting Microsoft Chat (who couldn't help me; they referred me to Pro Support), I finally got it myself! I'm posting in case someone else needs it.

Private Sub cmd2SpellCheck_Click()
   Dim WdApp As Object, WdDoc As Object, WdRng As Object, WdSavedOption As Boolean
   Dim sText$
   Static spellCheckInProgress As Boolean
  
   sText$ = Trim(TextBox1.Text)
   If Len(sText$) = 0 Then Exit Sub
  
'===>To prevent opening the Word document when it is already open:
   If spellCheckInProgress Then Exit Sub Else spellCheckInProgress = True
     
   Set WdApp = CreateObject("Word.Application")
   WdApp.Visible = False
   WdApp.AutomationSecurity = msoAutomationSecurityForceDisable
  
   Application.DisplayAlerts = False
   WdApp.DisplayAlerts = False
   Set WdDoc = WdApp.Documents.Open(gControlPath$ & "\Spell.docm")
   WdApp.DisplayAlerts = True
   Application.DisplayAlerts = True
  
   DoEvents
   WdSavedOption = WdApp.Options.CheckSpellingAsYouType
   WdApp.Options.CheckSpellingAsYouType = True
  
   Set WdRng = WdDoc.Range
   WdRng.Text = sText$
  
   If WdRng.spellingErrors.Count = 0 Then
      gMsgAndBorders "No spelling errors found.", vbBlue
      GoTo CleanUp
   End If
  
   Application.ScreenUpdating = True
  
   WdApp.Visible = True
   WdApp.Activate
   WdDoc.Select
   WdRng.CheckSpelling
   TextBox1.Text = Trim(Application.WorksheetFunction.Clean(WdRng.Text))
CleanUp:
   Set WdRng = Nothing
   WdApp.Options.CheckSpellingAsYouType = WdSavedOption
  
   Do Until WdApp.AddIns.Count = 0
      WdApp.AddIns.Unload RemoveFromList:=False
   Loop
   WdDoc.Close savechanges:=False
   WdApp.Quit
   Set WdDoc = Nothing
   Set WdApp = Nothing
  
   DoEvents
   Application.DisplayAlerts = True
   Application.ScreenUpdating = True
   TextBox1.SetFocus
   Application.ScreenUpdating = False
   spellCheckInProgress = False
End Sub
