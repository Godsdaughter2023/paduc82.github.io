andreas killer say:

how to work around this so that the scrollbar is always at the maximum "scrolldown" position for the textbox?

Make a new file, add a Userform, add 2 text boxes, add the code below and run the form.

Andreas.

Private Sub UserForm_Initialize()
  Dim Item
  Dim i As Integer
  Dim s As String
  For Each Item In Array(Me.TextBox1, Me.TextBox2)
    With Item
      .MultiLine = True
      .ScrollBars = fmScrollBarsBoth
      .WordWrap = True
      .Locked = True
      .EnterFieldBehavior = fmEnterFieldBehaviorRecallSelection

      For i = 1 To Rnd * 15 + 10
        s = s & String(Rnd * 20 + 1, Chr(64 + i)) & vbCrLf
      Next
      .Value = s
      .SelStart = Len(.Value)
      'Let the textbox scroll
      .SetFocus
      DoEvents
    End With
  Next

  Me.TextBox1.SetFocus
End Sub
