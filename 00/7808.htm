Sub LINEST_in_VBA_Test_2a()

Dim output As Variant, x As Variant, y As Variant, z As Variant
   
Set x = Range("c5:C18")
Dim XArr()
XArr = XArray(x.Value, 4)

Set y = Range("D5:D18")

output = Application.LinEst(y, XArr, 1, 1)

   'The following code puts the output onto the current sheet
  
   For I = 1 To UBound(output, 1)
  
   If I = 1 Then Cells(43 + I, 13).Value = "Slope & Intercept"
   If I = 2 Then Cells(43 + I, 13).Value = "+ or - "
   If I = 3 Then Cells(43 + I, 13).Value = "r squared & s(y)"
   If I = 4 Then Cells(43 + I, 13).Value = "F & df"

   For J = 1 To UBound(output, 2)
  
        Cells(42 + I + 1, 13 + J).Value = output(I, J)
     
   Next J
   Next I

Function XArray(InArr, PolyPower)

' taken from Tushar Mehta's article http://www.tushar-mehta.com/publish_train/data_analysis/16.htm

    Dim XArr()

    Dim I As Long, J As Long

    ReDim XArr(LBound(InArr) To UBound(InArr), 1 To PolyPower)

    For I = LBound(XArr, 1) To UBound(XArr, 1)

        XArr(I, 1) = InArr(I, LBound(InArr, 2))

        For J = 1 To PolyPower

            XArr(I, J) = XArr(I, 1) ^ J

            Next J

        Next I

    XArray = XArr

    End Function

greg maxey say:

Of course the type of dropdown you use will make a difference but assuming you are using a Content Control dropdown with the title "DD" you could insert this code in the project "ThisDocument" module.  The code is triggered when you make your selection and exit the content control.

Private Sub Document_ContentControlOnExit(ByVal ContentControl As ContentControl, Cancel As Boolean)
Dim oRng As Range
  Select Case ContentControl.Title
    Case "DD"
      Select Case ContentControl.Range.Text
        Case "does"
          'Using a bookmark named bmRange
          Set oRng = ActiveDocument.Bookmarks("bmRange").Range
          Set oRng = ThisDocument.AttachedTemplate.BuildingBlockEntries("does").Insert(oRng, True)
          ActiveDocument.Bookmarks.Add "bmRange", oRng
          'Using a rich text content control named "ccRange"
          ThisDocument.AttachedTemplate.BuildingBlockEntries("does").Insert _
                       ActiveDocument.SelectContentControlsByTitle("ccRange").Item(1).Range, True
          'Using a table Cell e.g., document table 1, row 1, column 2
          Set oRng = ActiveDocument.Tables(1).Cell(1, 2).Range
          oRng.Text = vbNullString
          oRng.Collapse wdCollapseStart
          ThisDocument.AttachedTemplate.BuildingBlockEntries("does").Insert oRng, True
        Case "does not"
          'Using a bookmark named bmRange
          Set oRng = ActiveDocument.Bookmarks("bmRange").Range
          Set oRng = ThisDocument.AttachedTemplate.BuildingBlockEntries("does_not").Insert(oRng, True)
          ActiveDocument.Bookmarks.Add "bmRange", oRng
          'Using a rich text content control named "ccRange"
          ThisDocument.AttachedTemplate.BuildingBlockEntries("does_not").Insert _
                       ActiveDocument.SelectContentControlsByTitle("ccRange").Item(1).Range, True
          'Using a table Cell e.g., document table 1, row 1, column 2
          Set oRng = ActiveDocument.Tables(1).Cell(1, 2).Range
          oRng.Text = vbNullString
          oRng.Collapse wdCollapseStart
          ThisDocument.AttachedTemplate.BuildingBlockEntries("does_not").Insert oRng, True
        Case Else
         'Using a bookmark named bmRange
          Set oRng = ActiveDocument.Bookmarks("bmRange").Range
          oRng.Text = vbNullString
          ActiveDocument.Bookmarks.Add "bmRange", oRng
          'Using a rich text content control named "ccRange"
          ActiveDocument.SelectContentControlsByTitle("ccRange").Item(1).Range.Text = vbNullString
          'Using a table Cell e.g., document table 1, row 1, column 2
          Set oRng = ActiveDocument.Tables(1).Cell(1, 2).Range
          oRng.Text = vbNullString
      End Select
  End Select
lbl_Exit:
  Exit Sub
End Sub

Since you already have buildingblocks for your "does" and "does not" conditions and assuming they are named as shown in the code i.e., does and does_not, you might as well use them.

The only thing left is to define the range where these should be inserted.  That can  be either a bookmark named bmRange, a rich text CC name ccRange or a table cell.

rick song msft say:

Hello JonRibbens,

 

Based on my experience and test results, the internal emails with the same domain will show “dkim=none (message not signed)” in the mail headers. It means that it is a normal behavior the internal emails within the same domain will not be signed by DKIM, because DomainKeys Identified Mail (DKIM) with Office 365 is to ensure that destination email systems trust messages sent outbound from your custom domain. If you send a message to an external recipient, you will see the message is signed in the mail header.
