scottgem say:

No you don't. This is a job for a query! As a general rule we do NOT store calculated values. Especially in a case like this where the count can change.

SELECT [AuditCasenumber], Count([AuditCaseNumber) As Distinct_Count

FROM tablename

GROUP BY [AuditCasenumber]

ORDER BY [AuditCasenumber];

Will give you what you need. The results should look like:

auditcasenumber Distinct Count
1A 2
1B 1
1C 1
1D 2

    Dim intLenB4Last As Integer     'length before last concatenation
    Set DB = CurrentDb
    Set RS = DB.OpenRecordset(pstrSQL)

'======For ADO uncomment next two lines=====

    'Dim rs As New ADODB.Recordset

        adOpenKeyset, adLockOptimistic
    
    Dim strConcat As String 'build return string
   
    With RS
        If Not .EOF Then
            .MoveFirst
            Do While Not .EOF
                intCount = intCount + 1
                intLenB4Last = Len(strConcat)
                strConcat = strConcat & _
                .Fields(0) & pstrDelim
                strLastValue = .Fields(0)
                .MoveNext
               
               
            Loop
        End If
        .Close
    End With
    Set RS = Nothing
'====== uncomment next line for DAO ========
    Set DB = Nothing
    If Len(strConcat) > 0 Then
        strConcat = Left(strConcat, _
            Len(strConcat) - Len(pstrDelim))
        If Len(pstrLastDelim) > 0 And intCount > 1 Then
            strConcat = Left(strConcat, intLenB4Last - Len(pstrDelim)) & pstrLastDelim & strLastValue
        End If
    End If
    If Len(strConcat) > 0 Then
        Concatenate = strConcat
     Else
        Concatenate = Null
    End If
End Function
