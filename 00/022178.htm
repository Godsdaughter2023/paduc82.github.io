Function CountConsecutiveZeros(ByVal Horizontal As Range, ByVal Start As Long, ByVal Size As Long) As Long
  Dim Data, i As Long
  'Just to be sure:
  '  Return the result if we run out of the limits of the given cells
  On Error GoTo ExitPoint
  'Read in all values
  Data = Horizontal.Value
  'Count all zeros inside this limit
  For i = Start To Start + Size - 1
    If Data(1, i) = 0 Then
      CountConsecutiveZeros = CountConsecutiveZeros + 1
    Else
      If Data(1, i) = 100 Then Exit For
    End If
  Next
  'Zeros found?
  If CountConsecutiveZeros > 0 Then
    'Go further till the end of the given cells and we have zeros
    For i = i To UBound(Data, 2)
      If Data(1, i) = 0 Then
        CountConsecutiveZeros = CountConsecutiveZeros + 1
      Else
        Exit For
      End If
    Next
  End If
ExitPoint:
End Function
