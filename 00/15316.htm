granitestatecolin say:

OK, I finally solved this. I'm surprised at the fairly absurd solutions of "just create a new profile and start over." Might as well tell me to reformat my hard drive and reinstall Windows, instead of just this simple process that completely fixed the problem:


1. From Home tab in the Outlook 2013 Ribbon Bar (other places in other versions of Outlook), click on the Address Book button (all the way to the right).


2. In the Address Book (note that this is NOT the People/Contacts page of Outlook, but rather the separate little mini-app called "Address Book" reached as noted in #1) click on Tools, then Options...


3. Select Custom.


4. If the Contacts folder does not appear in the list, click the Add.. button below the list of contact folders. Find your contact folder in the list and add it. You may wish to use the buttons on the right to move it to the top of the list.


5. If this is your main contact list, then under "When opening the address, show this address list first:" select the contact folder you just added.


6. Click OK.


7. Confirm your contacts now appear when you click on Check Names from a new email.


Worked for me.

Please try the code below.

Then do Insert > Module.

Sub ExtractFromOutlookMsgs()
'Add a reference to "Microsoft Outlook xx.0 Object Library" via VBE > Tools > References...
'
     'Outlook variables:
    Dim olApp As Outlook.Application
    Dim olSelectedFolder As Outlook.Folder
    Dim olMsg As Outlook.MailItem
   
     'Excel variables:
    Dim wb As Workbook
    Dim ws As Worksheet
    Dim strBody As String
    Dim vBody As Variant, v As Variant, v2 As Variant
    Dim r As Long, c As Long, lnTest As Long
   
     'Add new sheet to this workbook:
    Set wb = ThisWorkbook
    Set ws = wb.Worksheets.Add
   
     'Add column titles to worksheet:
    r = 1
    ws.Cells(r, 1).Value = "Order Confirmation"
    ws.Cells(r, 2).Value = "Payment Date"
    ws.Cells(r, 3).Value = "Total Sale"
    ws.Cells(r, 4).Value = "Name"
    ws.Cells(r, 5).Value = "Paypal Email"
    ws.Cells(r, 6).Value = "Item number"
    ws.Cells(r, 7).Value = "Item price"
   
     'Get Outlook App:
    Set olApp = GetObject(, "Outlook.Application")
   
     'Loop through each message in the selected Outlook folder:
    Set olSelectedFolder = olApp.ActiveExplorer.CurrentFolder
    For Each olMsg In olSelectedFolder.Items
        c = 1
        r = r + 1
        If Left(olMsg.Subject, Len("Order Confirmation ")) = "Order Confirmation " Then
             'Extract data from subject:
            v2 = Split(olMsg.Subject, Chr(124), 2)
            ws.Cells(r, 1).Value = Trim(v2(UBound(v2)))
           
             'Condition the body text:
            strBody = olMsg.Body
            strBody = Replace(strBody, Chr(160), "")
            strBody = Replace(strBody, Chr(10), Chr(13))
            strBody = Replace(strBody, Chr(13) & " ", Chr(13))
            Do
                strBody = Replace(strBody, Chr(13) & Chr(13), Chr(13))
            Loop While InStr(strBody, Chr(13) & Chr(13)) > 0
           
             'Split body text into an array of paragraphs:
            vBody = Split(strBody, Chr(13))
            For Each v In vBody
                If Trim(v) <> vbNullString And InStr(v, ":") > 0 Then
                     'Split paragraphs by ":"
                    v2 = Split(Trim(v), ":", 2)
                    Select Case Trim(v2(LBound(v2)))
                        Case "Payment Date"
                            c = 2
                            ws.Cells(r, c).Value = Trim(v2(UBound(v2)))
                           
                        Case "Total Sale"
                            c = 3
                            ws.Cells(r, c).Value = Trim(v2(UBound(v2)))
                           
                        Case "Name"
                            c = 4
                            ws.Cells(r, c).Value = Trim(v2(UBound(v2)))
                           
                        Case "Paypal Email"
                            c = 5
                            ws.Cells(r, c).Value = Trim(v2(UBound(v2)))
                           
                        Case "Item number"
                            c = 6
                            If ws.Cells(r, c).Value <> "" Then r = r + 1 'Start new row for each item
                            ws.Cells(r, c).Value = Trim(v2(UBound(v2)))
                           
                        Case "Item price"
                            c = 7
                            If ws.Cells(r, c).Value <> "" Then r = r + 1 'Start new row for each item
                            ws.Cells(r, c).Value = Trim(v2(UBound(v2)))
                           
                    End Select
                End If
            Next v
        End If
    Next olMsg
    ws.Columns.AutoFit
   
     'Fill in the blank cells with the contents of the cell above,
     '(use if you need to sort or filter columns, uncomment if required):

You may need to add a reference to Microsoft Excel 14.0 Object Library also.

''''''''''''''''''''''''''''''''''''''''
Sub CopyToExcel()
Dim xlApp As New Excel.Application
xlApp.Visible = True
Dim xlWB As Excel.Workbook
Dim xlSheet As Excel.Worksheet
Dim olItem As Outlook.MailItem
Dim vText As Variant
Dim sText As String
Dim vItem As Variant
Dim oRng As Range
Dim i As Long
Dim rCount As Long

If Application.ActiveExplorer.Selection.Count = 0 Then
    MsgBox "No Items selected!", vbCritical, "Error"
    Exit Sub
End If
'Open the workbook to input the data
Set xlWB = xlApp.Workbooks.Add
Set xlSheet = xlWB.Sheets("Sheet1")
'Process each selected record
For Each olItem In Application.ActiveExplorer.Selection
    sText = olItem.Body
    vText = Split(sText, Chr(13))
    'Find the next empty line of the worksheet
    xlSheet.Range("A1") = "Total Price"
    xlSheet.Range("B1") = "Payment Date"
    xlSheet.Range("C1") = "Total Sale"
    xlSheet.Range("D1") = "Name"
    xlSheet.Range("E1") = "PayPal Email"
    xlSheet.Range("F1") = "Item Number"
    xlSheet.Range("G1") = "Item Price"
    xlSheet.Range("H1") = "Item Number"
    xlSheet.Range("I1") = "Item Price"
    xlSheet.Range("J1") = "Item Number"
    xlSheet.Range("K1") = "Item Price"
    xlSheet.Range("L1") = "Item Number"
    xlSheet.Range("M1") = "Item Price"
    xlSheet.Range("N1") = "Item Number"
    xlSheet.Range("O1") = "Item Price"
    rCount = xlSheet.Range("A" & xlSheet.Rows.Count).End(xlUp).Row
    rCount = rCount + 1
    xlSheet.Range("A" & rCount) = Right(olItem.Subject, Len(olItem.Subject) - InStr(olItem.Subject, "$"))
    'Check each line of text in the message body
    For i = UBound(vText) To 0 Step -1
        If InStr(1, vText(i), "Payment Date:") > 0 Then
            vItem = Split(vText(i), Chr(58))
            xlSheet.Range("B" & rCount) = Trim(vItem(1)) & ":" & Trim(vItem(2)) & ":" & Trim(vItem(3))
        ElseIf InStr(1, vText(i), "Total Sale:") > 0 Then
            vItem = Split(vText(i), Chr(58))
            xlSheet.Range("C" & rCount) = Trim(vItem(1))
        ElseIf InStr(1, vText(i), "Name:") > 0 Then
            vItem = Split(vText(i), Chr(58))
            xlSheet.Range("D" & rCount) = Trim(vItem(1))
        ElseIf InStr(1, vText(i), "PayPal Email:") > 0 Then
            vItem = Split(vText(i), Chr(58))
            xlSheet.Range("E" & rCount) = Trim(vItem(1))
        ElseIf InStr(1, vText(i), "Item number:") > 0 Then
            vItem = Split(vText(i), Chr(58))
            xlSheet.Range("F" & rCount) = Trim(vItem(1))
        ElseIf InStr(1, vText(i), "Item price:") > 0 Then
            vItem = Split(vText(i), Chr(58))
            xlSheet.Range("G" & rCount) = Trim(vItem(1))
        End If
    Next i
Next olItem
Set xlApp = Nothing
Set xlWB = Nothing
Set xlSheet = Nothing
Set olItem = Nothing
End Sub
