Private Sub BUT_EMAIL_SEND_Click()
If Me.CUR_EMAIL_STATUS = 1 Then
    If MsgBox("DO YOU REALLY WANT TO RESEND IT ?", 292, "HEY THERE . . EMAIL HAS ALREADY BEEN SENT !") = 7 Then Exit Sub
End If
Me.CUR_EMAIL_STATUS = 0

If Len(Me.BS_EMAIL_ACCOUNTS) = 0 Or IsNull(Me.BS_EMAIL_ACCOUNTS) Then
    If MsgBox("NEED A VALID EMAIL ADDRESS - CONTINUE ANYWAY?", 292, "CAUTION - CANT SEND EMAIL") = 7 Then Exit Sub
End If

H_ALINES = Me.TOT_LINES
If H_ALINES = 0 Then
    MsgBox "THERE ARE NO ALLOCATIONS AGAINS INVOICES ", 292, "CAUTION - CANT SEND EMAIL"
    Exit Sub
End If

' SEND TO
HE_SENDTO = Me.BS_EMAIL_ACCOUNTS & ""
HE_SUBJECT = "PAYMENT ADVICE FROM KINETIC CREATIONS / DUCKIES TO " & Me.BS_NAME
H_NEWLINE = Chr$(13) & Chr$(10)      <<<<<<<<<<  used to add line feed into the text  <<<<<<<<<<<<

HE_MESSAGE = HE_MESSAGE & "AMOUNT PAID :  " & Format$(Me.BKE_EXPENSE, "$#,##0.00;-$#,##0.00;0") & H_NEWLINE
HE_MESSAGE = HE_MESSAGE & "PAYMENT METHOD :  " & Me.TY_MONEY & H_NEWLINE
HE_MESSAGE = HE_MESSAGE & "BANK LODGEMENT No. :  " & Me.BKE_LODGEMENT & H_NEWLINE & H_NEWLINE
HE_MESSAGE = HE_MESSAGE & "PAYMENT WAS ALLOCATED TO THE FOLLOWING INVOICE(S)" & H_NEWLINE & H_NEWLINE

H_ALINE = 1
HE_ALLOC_LINE = ""

LOOP_ALLOC:
    HE_ALLOC_LINE = "INVOICE - " & Me.LIST_ALLOCATIONS.Column(1, H_ALINE)
    HE_ALLOC_LINE = HE_ALLOC_LINE & " ON " & Format$(Me.LIST_ALLOCATIONS.Column(2, H_ALINE), "D MMM YYYY")
    HE_ALLOC_LINE = HE_ALLOC_LINE & " AMOUNT : " & Format$(Me.LIST_ALLOCATIONS.Column(3, H_ALINE), "$#,##0.00;-$#,##0.00;0")
    HE_ALLOC_LINE = HE_ALLOC_LINE & "   PAID : " & Format$(Me.LIST_ALLOCATIONS.Column(4, H_ALINE), "$#,##0.00;-$#,##0.00;0")

    HE_MESSAGE = HE_MESSAGE & HE_ALLOC_LINE & H_NEWLINE
    H_ALINE = H_ALINE + 1
    If H_ALINE <= H_ALINES Then GoTo LOOP_ALLOC


    HE_MESSAGE = HE_MESSAGE & H_NEWLINE ' ADD EXTRA LINE
'END LOOP


' ADD SIGNATURE
If Not IsNull(Me.BKE_COMMENT) Then HE_MESSAGE = HE_MESSAGE & "COMMENT:  " & Me.BKE_COMMENT & H_NEWLINE & H_NEWLINE
HE_MESSAGE = HE_MESSAGE & "HAVE A NICE DAY" & H_NEWLINE & "PAUL" & H_NEWLINE & "ACCOUNTS PAYABLE"
   
If HPREVIEW = True Then
    HPREVIEW = False
    If MsgBox(HE_MESSAGE & H_NEWLINE & H_NEWLINE & "* * * * * * * * * * * * * * * * SEND EMAIL ??", 292, HE_SUBJECT) = 7 Then Exit Sub
End If
Me.BUT_CLOSE.SetFocus

On Error GoTo EMAIL_ERROR

DoCmd.SendObject acSendNoObject, , , HE_SENDTO, , , HE_SUBJECT, HE_MESSAGE

Exit Sub

EMAIL_ERROR:

Me.CUR_EMAIL_STATUS = -1
Exit Sub
