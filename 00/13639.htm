daniel pineault say:

Emil,

To answer your direct question regarding running a query in code and retrieving a value in VBA, you'd do something along the lines of

Private Sub MediaLista_Click()
    On Error GoTo Error_Handler
    Dim rs                    As DAO.Recordset
    Dim strSQL                As String

    'If nothing is selected end sub
    If IsNull(Me.MediaLista) = True Then
        Exit Sub
    End If

    'Create SQL query to find the link in column HYPERLÄNK in table LISTA_MEDIA that has a value in column ID_MEDIA that corresponds with the value saved in MediaListaVal
    strSQL = "SELECT HYPERLÄNK FROM [LISTA_MEDIA] WHERE ID_MEDIA=" & Me.MediaListaVal

    'Run the query
    Set rs = CurrentDb.OpenRecordset(strSQL, dbOpenSnapshot)
    If rs.RecordCount <> 0 Then
        'Set the controlsource for Bild to the value in MediaLink
        MediaLista.ControlSource = Nz(rs![HYPERLÄNK], "")
    Else
        'Set the controlsource for Bild to the value in MediaLink
        MediaLista.ControlSource = ""
    End If

Error_Handler_Exit:
    On Error Resume Next
    If Not rs Is Nothing Then
        rs.Close
        Set rs = Nothing
    End If
    Exit Sub

Error_Handler:
    MsgBox "The following error has occured" & vbCrLf & vbCrLf & _
           "Error Number: " & Err.Number & vbCrLf & _
           "Error Source: MediaLista_Click" & vbCrLf & _
           "Error Description: " & Err.Description & _
           Switch(Erl = 0, "", Erl <> 0, vbCrLf & "Line No: " & Erl) _
           , vbOKOnly + vbCritical, "An Error has Occured!"
    Resume Error_Handler_Exit
End Sub

That said, in this instance, I fully agree with Ken, that it would make more sense to simply use the DLookUp() function by doing something like

Private Sub MediaLista_Click()
    On Error GoTo Error_Handler

    'If nothing is selected end sub
    If IsNull(Me.MediaLista) = True Then
        Exit Sub
    End If

    MediaLista.ControlSource = Nz(DLookup("HYPERLÄNK", "LISTA_MEDIA", "ID_MEDIA=" & Me.MediaListaVal), "")

Error_Handler_Exit:
    On Error Resume Next
    Exit Sub

Error_Handler:
    MsgBox "The following error has occured" & vbCrLf & vbCrLf & _
           "Error Number: " & Err.Number & vbCrLf & _
           "Error Source: MediaLista_Click" & vbCrLf & _
           "Error Description: " & Err.Description & _
           Switch(Erl = 0, "", Erl <> 0, vbCrLf & "Line No: " & Erl) _
           , vbOKOnly + vbCritical, "An Error has Occured!"
    Resume Error_Handler_Exit
End Sub

=IFERROR(INDEX($C$2:$C$20,MATCH(0,IF(MONTH($B$2:$B$20)<>1,"",COUNTIF(D$1:$D1,$C$2:$C$20)),0)),"")

gord dibben say:

Would like to automatically update cell with current time and date after another cell is updated.

Example: when I update cell F5, I would like for cell G5 to automatically populate with current date and time.

Copy/paste to sheet module.

Private Sub Worksheet_Change(ByVal Target As Range)
    If Target.Cells.Count > 1 Then Exit Sub
    If Intersect(Target, Me.Range("F5")) Is Nothing Then Exit Sub
    On Error GoTo CleanUp
    Application.EnableEvents = False
       With Range("G5")
            .Value = Format(Now, "hh:mm:ss am/pm")
      End With
CleanUp:
    Application.EnableEvents = True
End Sub
