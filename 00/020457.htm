Well, what works for me here is:
  1. Go into File> Open
  2. Type Command+F, then type the criteria such as title If the found list is too long refine it further by:
    1. Click the + next to the Save button in the upper right corner of the dialog just below the Spotlight field

    4. Enter the Criteria for that Property

    Option Explicit
    Option Compare Text

    Private Sub Workbook_SheetChange(ByVal Sh As Object, ByVal Target As Range)
      Const DefNumberFormat = """INV-""0""/yy"""
      Dim HighestNumber

      Application.EnableEvents = False
      'Which sheet?
      Select Case Sh.Name
        Case "Sheet1", "Sheet2", "Sheet3"
          'More then one cell?
          If Target.Count > 1 Then GoTo ExitPoint
          'Calculate the highest invoice number
          HighestNumber = Application.Max( _
            Sheets("Sheet1").Range("S:S"), _
            Sheets("Sheet2").Range("S:S"), _
            Sheets("Sheet3").Range("S:S"))
          'Which column?
          Select Case Target.Column
            Case Columns("Q").Column
              If IsEmpty(Target.Value) Then
                'Clear the invoice number
                Target.Offset(, 2).ClearContents
              Else
                With Target.Offset(, 2)
                  'Only when column S is empty
                  If Not IsEmpty(.Value) Then GoTo ExitPoint
                  'Next number
                  .Value = HighestNumber + 1
                  .NumberFormat = Replace(DefNumberFormat, "yy", Format(Date, "yy"))
                End With
              End If
            Case Columns("S").Column
              With Target
                'Only if used
                If IsEmpty(.Value) Then GoTo ExitPoint
                .NumberFormat = Replace(DefNumberFormat, "yy", Format(Date, "yy"))
                'Must be the largest number!
                If .Value <> HighestNumber Then
                  MsgBox "Invalid Invoice Number", vbExclamation, "Invalid Entry"
                  Target.ClearContents
                End If
              End With
          End Select
      End Select
    ExitPoint:
      Application.EnableEvents = True
    End Sub
