rick rothstein, mvp - excel say:

Give this macro a try...

Sub SeparateOddEvenRows()
  Dim R As Long, C As Long, Xo As Long, Xe As Long
  Dim Data As Variant, Odd As Variant, Even As Variant
  Data = Range("A3").CurrentRegion
  ReDim Odd(1 To (UBound(Data) + 1) / 2, 1 To UBound(Data, 2))
  ReDim Even(1 To (UBound(Data) + 1) / 2, 1 To UBound(Data, 2))
  For R = 1 To UBound(Data)
    If R Mod 2 Then
      Xo = Xo + 1
    Else
      Xe = Xe + 1
    End If
    For C = 1 To UBound(Data, 2)
      If R Mod 2 Then
        Odd(Xo, C) = Data(R, C)
      Else
        Even(Xe, C) = Data(R, C)
      End If
    Next
  Next
  Cells(3, Columns.Count).End(xlToLeft).Offset(, 2).Resize(UBound(Odd), UBound(Odd, 2)) = Odd
  Cells(3, Columns.Count).End(xlToLeft).Offset(, 2).Resize(UBound(Even), UBound(Even, 2)) = Even
End Sub

Option Explicit

Sub ertert()
Dim x, i&, j&
With Sheets("Counter Totals")
    x = .Range("A2:ED" & .Cells(Rows.Count, 1).End(xlUp).Row).Value

For i = 1 To UBound(x)
    If (x(i, 1)) = "Game" Then j = j + 1
    If (IsNumeric(x(i, 1))) * (Len(x(i, 1))) Then
        With Sheets("Game" & x(i, 1)).Columns(1).SpecialCells(2)
            .Areas(j)(.Areas(j).Count + 1, 1).Resize(, 134).Value = Application.Index(x, i, 0)
        End With
    End If
Next i
