Carlos, I see that you found BeforeUpdate, however, the first problem you encountered with AfterUpdate is that you added the Cancel argument which isn't expected for this event.  Furthermore, AfterUpdate means that the record has already been saved, so validation is relatively pointless at this stage.

You also might want to check out VBA Help on using the Like operator, particularly how it regards case sensitivity. 

For my purposes, I have created two modules and three functions to manage using Like and case sensitivity.  See the following:

Module 1:

Option Explicit
'=====================================================================================================================
'   Module:         LibMod_StringSearch_IsLikeIgnoreCase
'   Type:           Standard
'   Version:        1.1
'   Created:        01/01/1992 00:00    By: RMD ©
'
'   Description:    Library wrapper for the VBA Operator: Like. This module works together with the library module
'                   LibMod_StringSearch_IsLikeMatchCase to produce a single function that handles case sensitivity.
'
'=====================================================================================================================

Private Property Get MyModule() As String
    'Version:   1.0
    'Created:   01/13/2009 07:34    By: RMD ©

Function IsLike(SearchString As Variant, Pattern As String, Optional MatchCase As Boolean) As Boolean
    'Version:   1.0
    'Created:   01/01/1992 00:00    By: RMD ©
   
    On Error GoTo Err_Process
   
    Dim blnReturn As Boolean
   
    If (MatchCase) Then
        blnReturn = IsLikeMatchCase(SearchString, Pattern)
    Else
        blnReturn = IsLikeIgnoreCase(SearchString, Pattern)
    End If
   
Exit_Process:
    IsLike = blnReturn
    Exit Function

    Select Case Err
    Case Else
        Msgbox_Error "IsLike", MyModule
    End Select

   
End Function

Function IsLikeIgnoreCase(SearchString As Variant, Pattern As String) As Boolean
    'Version:   1.0
    'Created:   01/01/1992 00:00    By: RMD ©
   
    On Error GoTo Err_Process
   
    Dim blnReturn As Boolean
   
    If (Not IsMissing(SearchString)) Then
        If (Nz(SearchString, "") <> "") Then
            If (Pattern <> "") Then
                blnReturn = SearchString Like Pattern
            End If
        End If
    End If
   
Exit_Process:
    IsLikeIgnoreCase = blnReturn
    Exit Function

    Select Case Err
    'Case 93
        'Invalid Pattern String
    Case Else
        Msgbox_Error "IsLikeIgnoreCase", MyModule
    End Select

   
End Function

Private Sub Msgbox_Error(Optional ProcName As String, Optional ModuleName As String)
   
    Dim strMsg As String

    If (ModuleName <> "") Then
        strMsg = strMsg & vbCrLf & "Module:" & vbTab & vbTab & ModuleName
    End If
    If (ProcName <> "") Then
        strMsg = strMsg & vbCrLf & "Procedure:" & vbTab & ProcName
    End If
    strMsg = strMsg & vbCrLf & vbCrLf & Err.Description

   
End Sub
