macropod say:

The use of custom building blocks is one very good way. For that, you might simply insert the building block at the current insertion point, or use the ActiveX check box to trigger the updating of a bookmarked range.

 

As an example of conditionally adding/deleting content, suppose you have an ActiveX checkbox named CheckBox1 and a bookmark named MyBookmark:

 

Private Sub CheckBox1_Click()
Dim StrTxt As String, StrBkMk As String
StrBkMk = "MyBookmark"
If Me.CheckBox1.Value = True Then
  StrTxt = "Some text"
Else
  StrTxt = vbNullString
End If
Call UpdateBookmark(StrBkMk, StrTxt)
End Sub

 

Sub UpdateBookmark(StrBkMk As String, StrTxt As String)
Dim BkMkRng As Range
With ActiveDocument
  If .Bookmarks.Exists(StrBkMk) Then
    Set BkMkRng = .Bookmarks(StrBkMk).Range
    BkMkRng.Text = StrTxt
    .Bookmarks.Add StrBkMk, BkMkRng
  End If
End With
Set BkMkRng = Nothing
End Sub

greg maxey say:

Paul mentions BuildingBlocks.


Assume you have:


1) A checkbox "CheckBox1"

2) A bookmark "bmBBTarget"

3) A buildingblock "BB Text" saved in the template attached to the document


Private Sub CheckBox1_Click()
Dim strBBName As String, strBMName As String
  strBMName = "bmBBTarget"
  strBBName = "BB Text"
  UpdateBookmark CheckBox1, strBMName, strBBName
End Sub
Sub UpdateBookmark(ByRef oCheck As Object, strBMName As String, strBBName As String)
Dim oBMRng As Range
  With ActiveDocument
    If .Bookmarks.Exists(strBMName) Then
      Set oBMRng = .Bookmarks(strBMName).Range
      If oCheck.Value Then
        'General method
        'Set oBMRng = ActiveDocument.AttachedTemplate.BuildingBlockEntries(strBBName).Insert(Where:=oBMRng, RichText:=True)
        'Explicit method
        Set oBMRng = ActiveDocument.AttachedTemplate.BuildingBlockTypes(wdTypeAutoText).Categories("General"). _
                     BuildingBlocks(strBBName).Insert(Where:=oBMRng, RichText:=True)
      Else
        oBMRng.Text = vbNullString
      End If
      .Bookmarks.Add strBMName, oBMRng
    End If
   End With
   Set oBMRng = Nothing
 End Sub
