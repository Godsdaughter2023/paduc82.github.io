# root local folder of the synced library, please change it to the folder on your computer.

$rooturl = "c:\users\daizys\sharepoint\tx";                              

function recursedirectory([string]$root) {   

    if ($root -eq "")
   {
       return;                                                                                         
    } 

    $items = get-childitem $root; 

    foreach ($item in $items)
