Private Sub CbAdd_Click()

    Dim LastRow As Long
    Dim lstObj As ListObject
    Dim bolNoRecords As Boolean
    Dim rngTable As Range
    Dim rngNewRecord As Range
    Dim strTemp As String
   
    'Check to make sure all required fields are completed
    'Use Trim in case user inadvertantly enters a space in otherwise empty field (OssieMac's edit)
    If Trim(Me.TxtbFirstName) = "" Or Trim(Me.TxtbLastName) = "" Then
        MsgBox "A first and last name must be provided to create a new client.", vbExclamation + vbOKOnly, "Information Missing"
        Exit Sub
    End If
   
    With Sheets("Clients")
        .Select     'Probably superfluous because already selected when "Add New Client" button clicked.
       
        'Assigning to a properly declared variable provides the intellisense dropdowns
        'when coding to help with the syntax.
        Set lstObj = .ListObjects("Table2")
       
        With lstObj
            'Method of ShowAllData for Tables (List Objects) without using On Error Resume Next
            If .ShowAutoFilter Then         'Test if AutoFilter is on
                .AutoFilter.ShowAllData
            End If
        End With
       
        bolNoRecords = False        'Initialize but probably superfluous
        With lstObj.DataBodyRange
            If .Rows.Count = 1 Then     'Only one row of records
                If .Cells(1, 1) = "" And .Cells(1, 2) = "" Then    'Test if first row is empty
                    bolNoRecords = True     'True indicates no existing records
                End If
            End If
        End With
       
        If bolNoRecords = False Then    'One or more records exist so add a row for new record
            Set rngTable = lstObj.Range 'Assign the existing table range to a range variable
       
            With rngTable
                'Resize rngTable to the new size required for the table (eg. Rows count + 1)
                Set rngTable = .Resize(.Rows.Count + 1, .Columns.Count)
            End With
           
            lstObj.Resize rngTable  'Resize the table to fit the new resized range
        End If
       
        'Assign first cell of the last row of the DataBodyRange to a range variable
        With lstObj.DataBodyRange
            Set rngNewRecord = .Cells(.Rows.Count, 1)  'First cell of last (added) row
        End With
       
        With rngNewRecord
            .Value = WorksheetFunction.Proper(Me.TxtbLastName.Text) 'Assign first value to the rngNewRecord
            .Offset(, 1) = WorksheetFunction.Proper(Me.TxtbFirstName.Text)  'Offset to remaining columns of table (Same row)
            .Offset(, 2) = WorksheetFunction.Proper(Me.TxtbFirstName.Text) & _
                            " " & WorksheetFunction.Proper(Me.TxtbLastName.Text)    'Concatenate here instead of using formula
            .Offset(, 3) = Me.TxtbEmail.Text
            .Offset(, 4) = Me.TxtbPhone1.Text
            .Offset(, 5) = Me.TxtbPhone2.Text
            .Offset(, 6) = Me.TxtbHomeAdd.Text
            .Offset(, 7) = Me.TxtbBusinessAdd.Text
        End With
        .Columns.AutoFit        'Optional
    End With
   
    'Hide Userform
    Unload Me
   
    'Sort the table on Last Name column
    With lstObj
        .Sort.SortFields.Clear
        .Sort.SortFields.Add Key:=.ListColumns(1).Range, _
                            SortOn:=xlSortOnValues, _
                            Order:=xlAscending, _
                            DataOption:=xlSortNormal
    End With
                                   
    With lstObj.Sort
        .Header = xlYes
        .MatchCase = False
        .Orientation = xlTopToBottom
        .SortMethod = xlPinYin
        .Apply
    End With
    
End Sub
