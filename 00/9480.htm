Sub Splitter2()
 ' Based on a Macro created 16-08-98 by Doug Robbins
 ' to save each letter created by a
 ' mailmerge as a separate file.
 Dim Mask As String
 Dim Letters As Long
 Dim Counter As Long
 Dim docName As String
 Dim oDoc As Document
 Dim oNewDoc As Document
 Set oDoc = ActiveDocument
 oDoc.Save
 Selection.EndKey Unit:=wdStory
 Letters = Selection.Information(wdActiveEndSectionNumber)
 Mask = "ddMMyy"
 Selection.HomeKey Unit:=wdStory
 Counter = 1
 While Counter < Letters
 docName = "CA" & " " & LTrim$(Str$(Counter)) & ".doc"
 oDoc.Sections.First.Range.Cut
 Set oNewDoc = Documents.Add
 'Documents are based on the Normal template
 'To use an alternative template follow the link.
 With Selection
 .Paste
 .EndKey Unit:=wdStory
 .MoveLeft Unit:=wdCharacter, Count:=1
 .Delete Unit:=wdCharacter, Count:=1
 End With
 oNewDoc.SaveAs FileName:=docName, _
 FileFormat:=wdFormatDocument, _
 AddToRecentFiles:=False
 ActiveWindow.Close
 Counter = Counter + 1
 Wend

I had ignored the possibility of multiple  bracketed expressions in an individual line of the cell expression.

That said, your amendment would be subject to similar problems if there were more than two bracketed expressions in a given line of the cell text.

To overcome the problem of multiple bracketed expressions, I have introduced a simple Instr function loop:

'=========>>
Public Function DeleteBracketData(r)
    Dim arrSplit As Variant
    Dim s As String, sStr As String
    Dim iPos As Long, jPos As Long
    Dim i  As Long, j As Long

    arrSplit = Split(r, Chr(10))
    For i = LBound(arrSplit) To UBound(arrSplit)
        s = arrSplit(i)
        iPos = InStr(1, s, "(")
        jPos = InStr(1, s, ")")
        Do Until iPos = 0
           s = Left(s, iPos - 1) & Mid(s, jPos + 1, 200)
           iPos = InStr(1, s, "(")
        jPos = InStr(1, s, ")")
        Loop
        arrSplit(i) = s
    Next i
    DeleteBracketData = Join(arrSplit, vbNewLine)
End Function
