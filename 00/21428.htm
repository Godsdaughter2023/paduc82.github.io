=LEFT(REPLACE(A2,1,FIND("/",A2),""),FIND("/",REPLACE(A2,1,FIND("/",A2),""))-1)&"-"&TEXT(LEFT(A2,FIND("/",A2)-1)*30,"mmm")&"-"&RIGHT(A2,4)
