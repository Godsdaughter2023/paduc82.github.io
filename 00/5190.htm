graham mayor say:
The short answer is that you cannot simply do this. You would have to open each document and record the count. You can do that with a macro.

http://www.gmayor.com/installing_macro.htm

Put the documents you wish to count in a separate folder and run the macro. The macro will fail if the documents are password protected. The document names with their individual and total counts are recorded in a new document.

Sub BatchCount()
Dim strFilename As String
Dim strPath As String
Dim oDoc As Document
Dim oLog As Document
Dim fDialog As FileDialog
Dim icounter As Long
Set fDialog = Application.FileDialog(msoFileDialogFolderPicker)
With fDialog
    .Title = "Select folder and click OK"
    .AllowMultiSelect = False
    .InitialView = msoFileDialogViewList
    If .Show <> -1 Then
        MsgBox "Cancelled By User", , _
               "List Folder Contents"
        Exit Sub
    End If
    strPath = fDialog.SelectedItems.Item(1)
    If Right(strPath, 1) <> "\" _
       Then strPath = strPath + "\"
End With

If Documents.Count > 0 Then
    Documents.Close SaveChanges:=wdPromptToSaveChanges
End If

Set oLog = Documents.Add

If Left(strPath, 1) = Chr(34) Then
    strPath = Mid(strPath, 2, Len(strPath) - 2)
End If

strFilename = Dir$(strPath & "*.doc")
icounter = 0
While Len(strFilename) <> 0
    WordBasic.DisableAutoMacros 1
    Set oDoc = Documents.Open(strPath & strFilename)
    icounter = icounter + oDoc.Words.Count
    oLog.Range.InsertAfter oDoc.Name & vbTab & oDoc.Words.Count & vbCr
    oDoc.Close SaveChanges:=wdDoNotSaveChanges
    strFilename = Dir$()
    WordBasic.DisableAutoMacros 0
Wend
oLog.Range.InsertAfter "Total words " & icounter
End Sub
