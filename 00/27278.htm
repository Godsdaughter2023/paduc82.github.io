doug robbins - mvp office apps & services (word) say:

With the data arranged the way that you describe on Sheet 1 of a Workbook, if you run a macro with the following code, it will populate Sheet2 with the necessary records so that you can merge the data on that sheet into a Directory type mail merge main document in which you have a 1 row table with the merge fields in the cells (you only need 3 columns) so that when you execute that merge, you would get what you want.

Sub MakeTicketData()
Dim shtsource As Worksheet, shtdestination As Worksheet
Dim i As Long, j As Long, n As Long
Dim strStudent As String
Dim strTeacher As String
Dim lngTickets As Long
Set shtsource = Sheets(1)
Set shtdestination = Sheets(2)
With shtdestination.Range("A1")
    .Offset(0, 0) = "Student Name"
    .Offset(0, 1) = "Teacher Name"
    .Offset(0, 2) = "Ticket Number"
End With
With shtsource.Range("A1")
    r = 1
    For i = 1 To .CurrentRegion.Rows.Count
        strStudent = .Offset(i, 0)
        strTeacher = .Offset(i, 1)
        lngTickets = .Offset(i, 2)
        n = 1
        For j = 1 To lngTickets
            With shtdestination.Range("A1")
                .Offset(r, 0) = strStudent
                .Offset(r, 1) = strTeacher
                .Offset(r, 2) = n & " of " & lngTickets
            End With
            n = n + 1
            r = r + 1
        Next j
    Next i
End With
End Sub
