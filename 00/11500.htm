Sub SaveActiveDocAsPDF()

On Error Resume Next

Dim saveName As String

saveName = ActiveDocument.FullName

saveName = Left(saveName, Len(saveName) - 5) & ".pdf"

ActiveDocument.SaveAs fileName:=saveName, FileFormat:=wdFormatPDF
