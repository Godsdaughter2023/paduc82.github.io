Option Compare Database
Option Explicit

Sub SplitWords()
Dim myDb As Database
Dim myRst As Recordset
Dim Arr As Variant
Dim i As Integer

DoCmd.SetWarnings False
DoCmd.RunSQL ("Delete * From TABLE1")
DoCmd.SetWarnings True

Set myDb = CurrentDb
Set myRst = myDb.OpenRecordset("TABLE1")

Arr = Split(DLookup("Label", "PIX PRO ORIGINE", "Id=14817"))

On Error Resume Next
For i = 0 To UBound(Arr)
    myRst.AddNew
    myRst("wd") = Arr(i)
    myRst.Update
Next
On Error GoTo 0

myRst.Close
myDb.Close
Set myRst = Nothing
Set myDb = Nothing
End Sub
