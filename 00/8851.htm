mike h.. say:

Hi,

 

Try this. Select the range which can be a single or multiple row(s)/column(s) and run the code.

 

EDIT.. No of iterations reduced.

 

Sub Random_Sort()
Dim v As Variant, t As Variant, x As Long
Dim RwCell1 As Long, RwCell2 As Long, ColCell1 As Long, ColCell2 As Long
Dim Rw As Long, Col As Long, Iterations As Long
v = Selection.Value
Rw = Selection.Rows.Count
Col = Selection.Columns.Count
Iterations = Rw * Col
For x = Iterations To 1 Step -1

    RwCell1 = Int(1 + Rw * Rnd)
    RwCell2 = Int(1 + Rw * Rnd)
    ColCell1 = Int(1 + Col * Rnd)
    ColCell2 = Int(1 + Col * Rnd)
    If Selection.Rows.Count = 1 Then
        If ColCell1 <> ColCell2 Then
        t = v(RwCell1, ColCell1)
        v(RwCell1, ColCell1) = v(RwCell2, ColCell2)
        v(RwCell2, ColCell2) = t
    End If
    ElseIf Selection.Columns.Count = 1 Then
        If RwCell1 <> RwCell2 Then
        t = v(RwCell1, ColCell1)
        v(RwCell1, ColCell1) = v(RwCell2, ColCell2)
        v(RwCell2, ColCell2) = t
    End If
    Else
        If RwCell1 <> RwCell2 And ColCell1 <> ColCell2 Then
        t = v(RwCell1, ColCell1)
        v(RwCell1, ColCell1) = v(RwCell2, ColCell2)
        v(RwCell2, ColCell2) = t
    End If
    End If
Next x
Selection.Value = v
End Sub

andreas killer say:
Why not generate some random numbers and just exchange the rows / columns?

Okay, it's not so fast, but if you like this idea, you can read all values in a 2 dimensional array and do the same.

Andreas.


Option Explicit

Sub Test()
  Dim Values As Range
  Dim i As Long, j As Long, k As Long
  Dim x As Long, y As Long, r As Long, c As Long
  Dim SortOrientation As XlSortOrientation
 
  SortOrientation = xlSortRows
 
  'Cells selected?
  If Not TypeOf Selection Is Range Then Exit Sub
  'Did not work with areas
  If Selection.Areas.Count > 1 Then Exit Sub
 
  'Get the dimension of the selected cells
  x = Selection.Row
  y = Selection.Column
  r = Selection.Rows.Count
  c = Selection.Columns.Count
 
  'Initialize
  Randomize
  Application.ScreenUpdating = False
 
  'Sort by rows or columns?
  If SortOrientation = xlSortRows Then
    For i = 1 To r
      'Get a unique random row number from 1..r
      j = RandomUnique(1, r)
      'Did we have to change the rows?
      If i <> j Then
        'Make sure we get the lower row first
        If j > i Then
          k = i
        Else
          k = j
          j = i
        End If
        'Exchange the rows
        Cells(y + k - 1, x).Resize(1, c).Cut
        Cells(y + j - 1, x).Resize(1, c).Insert xlDown
        Cells(y + j - 1, x).Resize(1, c).Cut
        Cells(y + k - 1, x).Resize(1, c).Insert xlDown
      End If
    Next
  Else
    'Same as above, just for columns
    For i = 1 To c
      j = RandomUnique(1, c)
      If i <> j Then
        If j > i Then
          k = i
        Else
          k = j
          j = i
        End If
        Cells(y, x + k - 1).Resize(r, 1).Cut
        Cells(y, x + j - 1).Resize(r, 1).Insert xlToRight
        Cells(y, x + j - 1).Resize(r, 1).Cut
        Cells(y, x + k - 1).Resize(r, 1).Insert xlToRight
      End If
    Next
  End If
 
  Application.ScreenUpdating = True
End Sub

Private Function RandomUnique(ByVal Lo As Long, ByVal hi As Long, _
    Optional Reset As Boolean = False) As Long
  Static Dict As Object 'Dictionary
  'Init dictionary if necessary
  If Dict Is Nothing Then Set Dict = CreateObject("Scripting.Dictionary")
  'Remove all used numbers if forced from outside
  If Reset Then Dict.RemoveAll
  Do
    'Get a random number
    RandomUnique = Int((hi - Lo + 1) * Rnd) + Lo
    'Already used?
  Loop Until Not Dict.Exists(RandomUnique)
  'Remember it
  Dict.Add RandomUnique, 0
  'Automatic reset if all numbers used
  If Dict.Count > hi - Lo Then Dict.RemoveAll
End Function

graham mayor say:
You can certainly call an image file based on the selection from the combo box, or instead you could create a Word add-in template to provide the macro/userform and have the table in the body of the add-in. You can insert the signatures in the last column table as in-line graphics.

This can then be used to populate a multicolumn combobox, with the first four columns of the table. However you cannot store graphics in a combobox so you would have to read the table again to grab the appropriate in-line graphic from the Signature column of the table.

Assuming a simple userform with a combobox and a pair of command buttons coded as follows

Option Explicit
Private Sub CommandButton1_Click()
Me.Hide
Me.Tag = 1
End Sub

Private Sub CommandButton2_Click()
Me.Hide
Me.Tag = 0
End Sub

you could populate the combobox and extract the image using the following code

Option Explicit
Sub Main()
Dim oTable As Table
Dim i As Long, j As Long, m As Long, n As Long
Dim myArray() As Variant
Dim oData As Range
Dim oCell As Range
Dim oRow As Row
Dim oSig As Range
    Set oTable = ThisDocument.Tables(1)
    i = oTable.Rows.Count
    j = oTable.Columns.Count
    ReDim myArray(i - 1, j - 1)
    For n = 0 To j - 1
        For m = 1 To i - 1
            Set oData = oTable.Cell(m + 1, n + 1).Range
            oData.End = oData.End - 1
            If oData.Text <> "" Then
                myArray(m, n) = oData.Text
            Else
                myArray(m, n) = " "
            End If
        Next m
    Next n
    With UserForm1
        With .ComboBox1
            .ColumnCount = j
            .Width = 192
            .ColumnWidths = "190 pt;0 pt;0 pt;0 pt"
            .List = myArray
        End With
        Erase myArray
        .ComboBox1.ListIndex = -1
        .Show
        If .Tag = 0 Then GoTo Cleanup
        For i = 2 To ThisDocument.Tables(1).Rows.Count
            'Skip the table header row
            Set oRow = ThisDocument.Tables(1).Rows(i)
            'compare the selection in the userform with the matching value in the table
            If InStr(1, oRow.Cells(1).Range.Text, Trim(.ComboBox1.Column(0))) > 0 Then
                Set oCell = oRow.Cells(5).Range        'the table cell with the graphic
                oCell.End = oCell.End - 1        'strip the end of cell marker from the range
                'Set the location of the signature graphic - here at the end of the document
                Set oSig = ActiveDocument.Range
                oSig.Start = oSig.End
                oSig.FormattedText = oCell.FormattedText
                Exit For
            End If
        Next i
    End With
Cleanup:
    Unload UserForm1
    Set oTable = Nothing
    Set oRow = Nothing
    Set oSig = Nothing
End Sub

Save in the Word startup folder then run the macro from the document.

You can download the example from - http://www.gmayor.com/Userform_ComboBox.html
