chip pearson say:

You can do it with a bit of VBA.


Sub AAAA()
    Dim R As Range
    Dim Dest As Range
    Dim N As Long
   
    Set R = Worksheets("Sheet1").Range("A2") '<<< START OF DATA
    Set Dest = Worksheets("Sheet2").Range("A2") '<<< DESTINATION
   
    Do Until R.Value = vbNullString
        N = 5
        Do
            Dest(1, 1).Value = R(1, 1).Value
            Dest(1, 2).Value = R(1, 2).Value
            Dest(1, 3).Value = R(1, 3).Value
            Dest(1, 4).Value = R(1, 4).Value
            Dest(1, 5).Value = R(1, N).Value
            Set Dest = Dest(2, 1)
            N = N + 1
        Loop Until R(1, N).Value = vbNullString
        Set R = R(2, 1)
    Loop
End Sub

Change the first line of code marked with '<<< to the sheet and cell in which the original data begins. Change the second line marked with '<<< to the first destination cell where the new data is to be written.
