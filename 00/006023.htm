The Label Options in Word 2007 (and every other version) contains Avery 8160 and 5160 definitions.

First make sure you have the Label Vendors dropdown set to Avery US Letter.

You do not need Data Validation on cells that you only write to with VBA. Any validation that is required is done with the VBA code before writing to the worksheet so to alleviate any problems that it can cause, remove any data validation from columns J:R as follows:

  1. Select entire columns J:R
  2. Select Data validation
  3. If you get a message re extending the data validation then answer No.
  4. On the Data validation dialog, select "Clear all" (at the bottom left of dialog)
  5. Click OK to finish

Then replace "Private Sub SaveData()" with the following code. Because Completion can now be numeric or text, the control values that can be numeric or text need to be tested for text before attempting to use CDbl function.

Private Sub SaveData()
  'Save the data from the controls into current row
  Dim C As MSForms.Control
  Dim varValue As Variant    'Must be variant to accept different types of data
 
    For Each C In Me.Controls
        If C.Tag <> "" Then
            'Get the value from the control
            varValue = C.Value
            Select Case C.Tag
                Case "A", "C", "G", "H", "I", "J", "M", "P"
                    'These are a direct copy of their values to the worksheet
                    Worksheets("Data").Range(C.Tag & ThisRow.Row) = varValue
           
                Case "B", "D"
                    'These values must be valid dates so convert text date to date value
                    If IsDate(varValue) Then
                        Worksheets("Data").Range(C.Tag & ThisRow.Row) = CDate(varValue)
                    End If
               
                Case "K", "L", "N", "O", "Q", "R"
                    If WorksheetFunction.IsText(C.Tag) Then   'Tests if "Complete" selected from DropDown
                        Worksheets("Data").Range(C.Tag & ThisRow.Row) = varValue
                    Else
                        'If not text thenthis value is a double number so convert text number to double number
                        Worksheets("Data").Range(C.Tag & ThisRow.Row) = CDbl(varValue)
                    End If
            End Select
           
            'Insert Vlookup formula for the Employee Number
            Worksheets("Data").Range("F" & ThisRow.Row).FormulaR1C1 _
                    = "=IFERROR(VLOOKUP(RC[1],dropdown!C2:C3,2,FALSE),"""")"
           
            'Insert lookup formula for Department.
            'Note: To insert line breaks in a formula that is enclosed in double quotes
            '1. At the required break point, insert one additional double quote, ampersand, space and underscore
            '2. Press Enter to move remainder of line of code to next line
            '3. Insert additional double quote and space at start of second line. (additional space is optional)
            Worksheets("Data").Range("E" & ThisRow.Row).FormulaR1C1 = _
                    "=IF(ISNUMBER(SEARCH(""Huang"",RC[-2])),""Electrical""," & _
                    " IF(ISNUMBER(SEARCH(""Mao"",RC[-2]))," & _
                    " ""Moulding"",IF(ISNUMBER(SEARCH(""Ding"",RC[-2])),""Paintwork"","""")))"
        End If
    Next C
    Saved = True
End Sub
