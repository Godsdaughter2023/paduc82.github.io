I've played with the countif function but I'm struggling on how to piece it together. Any help would be vastly appreciated. Thank you!

jeeped say:

Typically you would assign a 1 for Yes and 0 for No as Excel interprets a 0 as False and anything that is not False is True. However, if it makes more sense to you to use a 1 (duplicate) and a 2 (unique) then that's the way to go.

 

In Sheet2!B2 use this formula,

 

=IF(COUNTIF(Sheet1!A:A,A2),1,2)

 

Fill down as required to reference all of the values in Sheet2's column A.

daniel pineault say:

You'd have to do something along the lines of (untested aircode):

Private Sub EMAIL_ROC_Click()
Dim varItem             As Variant  'Selected items
Dim strBody             As String   'String to use as e-mail body
Dim selEMAIL            As String   'string used for email To:
Dim subEMAIL            As String   'string used for email subject
Dim db                  As Database
Dim rs                  As Recordset
Dim SQL                 As String
Dim sEmailAddresses     As String

On Error GoTo Error_Handler
    Set db = CurrentDb()
   
    'Loop through the ItemsSelected in the list box and build your e-mail body content
    With Me.ROC_LIST
        For Each varItem In .ItemsSelected
            If Not IsNull(varItem) Then
                strBody = strBody & "ROC# " & Me.ROC_LIST.Column(1, varItem) & "" & Chr(13) & Chr(10)
                strBody = strBody & "MN: " & Me.ROC_LIST.Column(4, varItem) & " SN: " & Me.ROC_LIST.Column(6, varItem) & " issued to " & Me.ROC_LIST.Column(7, varItem) & " on TCN# " & Me.ROC_LIST.Column(9, varItem) & "" & Chr(13) & Chr(10) & Chr(13) & Chr(10)
            End If
        Next
    End With
    subEMAIL = subEMAIL & " & Me.ROC_LIST.Column(4) & " & " & Me.ROC_LIST.Column(7) & """
   
    'Build your Query
    SQL = "SELECT [NAMES].EMAIL, [NAMES].NAMES " & _
          "FROM NAMES " & _
          "WHERE [NAMES].NAMES = """ & Me.ROC_LIST.Column(8) & """;"
    Set rs = db.OpenRecordset(SQL, dbOpenSnapshot)
    'Build a listing of email addresses based on the query results
    If rs.RecordCount <> 0 Then
        With rs
            .MoveFirst
            Do While Not .EOF 'Loop through all the records returned by the query
                sEmailAddresses = sEmailAddresses & ![Email] & ";"
                .MoveNext
            Loop
        End With
    End If
   
    'Send the e-mail itself
    DoCmd.SendObject acSendNoObject, , , sEmailAddresses, , , subEMAIL, strBody, True

 
Error_Handler_Exit:
    On Error Resume Next
    rs.Close
    Set rs = Nothing
    Set db = Nothing
    Exit Function

Error_Handler:
    MsgBox "The following error has occured" & vbCrLf & vbCrLf & _
           "Error Number: " & Err.Number & vbCrLf & _
           "Error Source: EMAIL_ROC_Click" & vbCrLf & _
           "Error Description: " & Err.Description, vbCritical, _
           "An Error has Occured!"
    Resume Error_Handler_Exit
End Sub

Always add in error handling and indent your code to simplify your life.

ken sheridan say: Are you committed to using a macro rather than VBA. i.e. is this a web database?   If not, the following code is from one of the form's in my online FindRecord demo:

Private Sub cboGotoContact_AfterUpdate()

    Const MESSAGETEXT = "No matching record"
    Dim ctrl As Control
    
    Set ctrl = Me.ActiveControl
    
    If Not IsNull(ctrl) Then
        If ctrl = 0 Then
            ' go to new record and move focus to FirstName control
            DoCmd.GoToRecord acForm, Me.Name, acNewRec
            Me.FirstName.SetFocus
        Else
            With Me.RecordsetClone
                .FindFirst "ContactID = " & ctrl
                If Not .NoMatch Then
                    ' go to record by synchronizing bookmarks
                    Me.Bookmark = .Bookmark
                Else
                    MsgBox MESSAGETEXT, vbInformation, "Warning"
                End If
            End With
        End If
    End If
    
End Sub

This would work with any combo box in the form where the BoundColumn is ContactID, which is the key of the form's recordset.  The above code allows for going to an empty new record, and the Control's RowSource property adds an extra row to the list for this by means of a UNION operation, but if you simply wanted to go to a record you can dispense with the UNION operation and shorten the code to:

Private Sub cboGotoContact_AfterUpdate()

    Const MESSAGETEXT = "No matching record"
    Dim ctrl As Control
    
    Set ctrl = Me.ActiveControl
    
    If Not IsNull(ctrl) Then
        With Me.RecordsetClone
            .FindFirst "ContactID = " & ctrl
            If Not .NoMatch Then
                ' go to record by synchronizing bookmarks
                Me.Bookmark = .Bookmark
            Else
                MsgBox MESSAGETEXT, vbInformation, "Warning"
            End If
        End With
    End If
    
End Sub

Even better, to avoid having to duplicate this code in the AfterUpdate event procedure of each 'search' combo box in the form, you could wrap it in a function in the form's module:

Private Function GotoToRecord()

    Const MESSAGETEXT = "No matching record"
    Dim ctrl As Control
    
    Set ctrl = Me.ActiveControl
    
    If Not IsNull(ctrl) Then
        With Me.RecordsetClone
            .FindFirst "ContactID = " & ctrl
            If Not .NoMatch Then
                ' go to record by synchronizing bookmarks
                Me.Bookmark = .Bookmark
            Else
                MsgBox MESSAGETEXT, vbInformation, "Warning"
            End If
        End With
    End If

End Function

Then just call the function by entering the following as the AfterUpdate event property in the properties sheet of each control:

=GotoToRecord()
