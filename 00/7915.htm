Sub AutoF_SplitData_NewWBks()
'May 18, 2015
Const sC1 As String = "A"   '<<  filter1 in col A
Const sC2 As String = "C"   '<<  filter2 in col C
Const shN As String = "MASTER"   '<<  source sheet name
Dim wb As Workbook, wb1 As Workbook
Set wb = ThisWorkbook
Dim fPath
fPath = ThisWorkbook.Path & "\"

Dim ws As Worksheet
Set ws = wb.Sheets(shN)
Dim r As Long, c As Long, x As Long, y As Long, r1 As Long, r2 As Long, L As Long
Dim fName
Application.ScreenUpdating = False
ws.AutoFilterMode = False
r = ws.Cells(Rows.Count, sC1).End(xlUp).Row
c = ws.Cells(1, Columns.Count).End(xlToLeft).Column
ws.Range(sC1 & "2:" & sC1 & r).Copy: ws.Cells(1, c + 1).PasteSpecial xlValues: Application.CutCopyMode = False
ws.Range(sC2 & "2:" & sC2 & r).Copy: ws.Cells(1, c + 2).PasteSpecial xlValues: Application.CutCopyMode = False
ws.Cells(1, c + 1).Resize(r).RemoveDuplicates Columns:=1, Header:=xlNo
ws.Cells(1, c + 2).Resize(r).RemoveDuplicates Columns:=1, Header:=xlNo
r1 = ws.Cells(Rows.Count, c + 1).End(xlUp).Row
r2 = ws.Cells(Rows.Count, c + 2).End(xlUp).Row
ws.Cells(1, c + 1).Resize(r1).Sort key1:=ws.Cells(1, c + 1), Header:=xlNo
ws.Cells(1, c + 2).Resize(r2).Sort key1:=ws.Cells(1, c + 2), Header:=xlNo

For x = 1 To r1
For y = 1 To r2
With ws.Range("A1").Resize(r, c)
.AutoFilter field:=ws.Cells(1, sC1).Column, Criteria1:=ws.Cells(x, c + 1)
.AutoFilter field:=ws.Cells(1, sC2).Column, Criteria1:=ws.Cells(y, c + 2)
End With
Set wb1 = Workbooks.Add(xlWorksheet)
fName = ws.Cells(x, c + 1) & "_" & ws.Cells(y, c + 2)
ws.Range("A1").Resize(r, c).SpecialCells(xlCellTypeVisible).Copy
Cells(1, "A").PasteSpecial Paste:=xlPasteFormats
Cells(1, "A").PasteSpecial Paste:=xlPasteValues
Application.CutCopyMode = False

ActiveSheet.UsedRange.EntireColumn.AutoFit

L = Cells(Rows.Count, "A").End(xlUp).Row
On Error Resume Next
Kill (fPath & fName & ".xlsx")
wb1.SaveAs fPath & fName, 51
wb1.Close False
If L = 1 Then Kill (fPath & fName & ".xlsx")
Next y
Next x
ws.AutoFilterMode = False

ws.Cells(1, c + 1).Resize(r1).ClearContents
ws.Cells(1, c + 2).Resize(r2).ClearContents
Application.ScreenUpdating = True
On Error GoTo 0
End Sub
