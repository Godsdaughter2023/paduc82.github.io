Function SnapCascade(WindowTitle As String) As Boolean
 
  Dim try As Long
  Dim hSnap As Long
  Dim hMainMenu As Long
  Dim hMenu As Long
  Dim MenuID As Long
  Dim MenuItemNum As Long
 
  ' Get snapshot viewer handle
  try = 0
  hSnap = 0&
  Do While hSnap = 0&
    hSnap = FnFindWindowLike(WindowTitle, 0&)
    PauseApp 1000
    try = try + 1
    If try > 10 Then
      SnapCascade = False
      Exit Function
    End If
  Loop
 
  ' Select Window - Cascade menu item
  hMainMenu = GetMenu(hSnap)
  MenuGetItemNum hMainMenu, "&Window", MenuItemNum
  hMenu = GetSubMenu(hMainMenu, MenuItemNum)
  MenuGetItemNum hMenu, "&Cascade", MenuItemNum
  MenuID = GetMenuItemID(hMenu, MenuItemNum)
  PostMessage hSnap, WM_COMMAND, MenuID, 0&
 
  SnapCascade = True


End Function

You get a total of 5 desktop licenses to share with up to 4 additional people in your household. Or you can use all 5 licenses yourself. From your PC ..........

with IPAQ cradled

click on Start, Control Panel

Double-click on System

Click on Hardware, then Device Manager

Click on '+' to left of Universal Serial Bus Controllers (Portable Devices in VISTA)

Click on '+' to left of Windows CE USB Devices (or Windows Mobile Device in VISTA)
