jay freedman say:

I don't have an answer for you, only some informed guesses.

The loss of the entries from the generated index is probably just a consequence of the removal of the XE fields from the body of the document.

If I wanted to do that -- selectively remove all XE fields containing a specific range of entries -- I'd have to build a Replace expression with a wildcard in the Find What box and an empty Replace With box, or the equivalent in a macro. In fact, trying to do it in the UI would take at least two separate passes in order to remove some of the terms beginning with "Me" and not others. Doing it with a macro, it would need to examine each found entry to determine whether it comes before or after the break point. All of this is highly unlikely.

Is the document in question in Word 97-2003 (*.doc) format or in .docx format? Since it originated with a Word 2003 user, I'll guess it was .doc. A lot of the metadata in a .doc file is in tables in the file header, with pointers to the document body. If that's the case for XE fields, some kind of corruption could cause the program to stop reading the XE table at a certain point, and the damage would show up like this. I just don't know whether that's the situation.

An old "trick" that may help is to start a new blank document (I'd recommend using .docx format for this); in the damaged document, insert a new paragraph mark at the very end, copy everything have to rebuild the headers and footers in at least the last section in the new document.

Sub Example_GetGLYPHSET()
  Dim GS As GLYPHSET
  Dim i As Long, j As Long, r As Long
  Dim Data(32 To 65535, 1 To 4)
 
  'Clear previous results
  ActiveSheet.UsedRange.Offset(1).Clear
 
  'Get the number of rows from the array
  r = UBound(Data) - LBound(Data) + 1
 
  'Get the font from cell A1
  GS = GetGLYPHSET(Range("A1").Font)
  'Apply the same font to the output range
  Range("A1").Copy
  Range("A1:D" & r + 1).PasteSpecial xlPasteFormats
  Application.CutCopyMode = False
  'Write the headings
  Range("A1:D1") = Array("Nr", "Unicode", "Char", "Supported")
  'Create the output
  For i = LBound(Data) To UBound(Data)
    Data(i, 1) = i
    Data(i, 2) = "U+" & Right$("0000" & Hex(i), 4)
    Data(i, 3) = ChrW$(i)
    Data(i, 4) = IIf(HuntGLYPHSET(i, GS), "y", "")
  Next
  'Write it to the sheet
  Range("A2").Resize(r, 4) = Data
End Sub

Function GetGLYPHSET(ByVal F As Font) As GLYPHSET
  'http://msdn.microsoft.com/en-us/library/windows/desktop/dd144887%28v=vs.85%29.aspx
  Dim Data() As Byte, i As Long, j As Long
  Dim Result As Long, Count As Long
  Dim GS As GLYPHSET
  Dim hDC As Long

  'Some of the propertys might not exists in the Font object
  On Error Resume Next
  Dim fSize As Integer, fBold As Integer
  Dim FName As String
  Dim fItalic As Boolean, fUnderline As Boolean, fStrikethrough As Boolean
  Dim hFont As Long, hOldFont As Long
  fSize = -MulDiv(F.Size, GetDeviceCaps(hDC, LOGPIXELSY), 72)
  FName = F.Name
  fBold = IIf(F.Bold, FW_BOLD, FW_NORMAL)
  fItalic = F.Italic
  fUnderline = F.Underline <> xlUnderlineStyleNone
  fStrikethrough = F.Strikethrough
  On Error GoTo 0

  'Create a device context compatible to the screen
  hDC = CreateCompatibleDC(GetDC(0&))
  If hDC = 0 Then Exit Function
  'Create font object
  hFont = CreateFont(fSize, 0, 0, 0, fBold, fItalic, fUnderline, fStrikethrough, _
    DEFAULT_CHARSET, OUT_DEFAULT_PRECIS, CLIP_DEFAULT_PRECIS, DEFAULT_QUALITY, _
    DEFAULT_PITCH, FName)
  If hFont = 0 Then Exit Function
  'Load the font into the device context, storing the original font object
  hOldFont = SelectObject(hDC, hFont)

  'Get the size of the GLYPHSET structure required to store the information
  Result = GetFontUnicodeRanges(hDC, ByVal 0&)
  If Result = 0 Then Exit Function
  ReDim Data(0 To (Result - 1)) As Byte
  'Get the data and the number of bytes written
  Result = GetFontUnicodeRanges(hDC, Data(LBound(Data)))
  If Result = 0 Then Exit Function

  'Coyp the first 4 LONG into the GLYPHSET structure
  i = LBound(Data)
  Count = 4 * 4
  CopyMemory GS, Data(i), Count
  i = i + Count
  With GS
    'Did the font support unicode?
    If .cRanges > 0 Then
      'Create the array of ranges and copy the data into there
      ReDim .aRANGE(0 To (.cRanges - 1)) As WCRANGE
      j = LBound(.aRANGE)
      Count = .cRanges
      Do While Count > 0
        CopyMemory .aRANGE(j), Data(i), Len(.aRANGE(j))
        i = i + Len(.aRANGE(j))
        j = j + 1
        Count = Count - 1
      Loop
    End If
  End With
  'Destroy the created objects and return results
  DeleteObject hFont
  DeleteDC hDC
  GetGLYPHSET = GS
End Function

Function HuntGLYPHSET(ByVal CharCode As Long, ByRef GS As GLYPHSET) As Boolean
  'Hunt through the GLYPHSET, True if the Unicode character is supported
  Dim i As Long, Low As Long
  With GS
    CharCode = CharCode + 1
    For i = 0 To .cRanges - 1
      With .aRANGE(i)
        If .wcLow < 0 Then
          Low = .wcLow And &HFFFF&
        Else
          Low = .wcLow
        End If
        If CharCode > Low Then
          If CharCode <= Low + .cGlyphs Then
            HuntGLYPHSET = True
            Exit Function
          End If
        End If
      End With
    Next
  End With
End Function
