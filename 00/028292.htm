=DATE(YEAR(I2)-TRIM(LEFT(H2,FIND(",",H2)-1)), MONTH(I2)-TRIM(MID(H2, FIND(",",H2)+1, FIND(",", H2, FIND(",",H2)+1) - FIND(",",H2)-1)), DAY(I2)-TRIM(RIGHT(SUBSTITUTE(H2,",",REPT(" ",255)),255)))
