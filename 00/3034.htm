norman david jones say:

Hi Roopesh,

AFAIK this is not possible.

However, you could probably use the worksheet change event to obtain the required results.

By way of example,  right-click the worksheet's tab | select the View Code  option from the context menu and paste the following code:

'=========>>

Option Explicit

'--------->>
Private Sub Worksheet_Change(ByVal Target As Range)
    Dim Rng As Range, Rng2 As Range, rCell As Range
    Dim RngList As Range
    Dim Res As Variant, Res2 As Boolean
    Dim vVal As Variant
    Const sAddress As String = "A1:A10"                             ' << === See (1) below     
    Const sList As String = "C1:C20"                                    ' << === See (2) below              

    With Me
        Set Rng = .Range(sAddress)
        Set RngList = .Range(sList)
    End With

    Set Rng2 = Intersect(Rng, Target)

    If Not Rng2 Is Nothing Then
    On Error GoTo XIT
    Application.EnableEvents = False
        For Each rCell In Rng2.Cells
            With rCell
                Res = Application.Match(.Value, RngList, 0)     
                Res2 = .Value < 100 And .Value > 1                  ' << === See (3) below 
                If Not IsError(Res) Or Res2 Then
                    '\\ entry valid, do nothing
                Else
                    vVal = .Value
                   Application.Undo
                    MsgBox vVal & " is an Invalid value!"
                    Exit For
                End If
            End With
        Next rCell
    End If
XIT:
Application.EnableEvents = True
End Sub
'<<=========

(1) Change A1:A10 to the address of your data validation cells

(2) Change C1:C20 to the address of your data validation list

(3) Amend the conditions to accord with the condition(s) of your data validation formula.

Alt-Q to return to the worksheet.

Without knowledge of your intended data validation conditions, this is merely intended to illustrate a possible approach which you can adapt to accord with your specific requirements by amending points (1), (2) and (3).

===

Regards,

Norman

ken sheridan say: The best way to do this is to store the path to the image file in a text column in the table and set the Picture property of an image control to the path at runtime.  In a form this is done in its Current event procedure, in a report in the detail section's print event procedure.

The following is the code for a form's module, which does this, and includes a button to open the common dialogue to browse to an image file and insert its path into the ImagePath field, a button to delete the current image from the record (this doesn't delete the image file of course) and one to open a report  The table also has an ImageTitle field for captioning the image:

''''module starts''''
Option Compare Database
Option Explicit

Private Sub cmdAddImage_Click()

   On Error GoTo Err_Handler
  
   Dim OpenDlg As New BrowseForFileClass
   Dim strPath As String
   Dim strAdditionalTypes As String, strFileList As String

   ' grab position of cursor
   GrabCursor
  
   ' amend the following line for whatever file
   ' types you wish to use
   strFileList = "*.bmp; *.jpg"
   strAdditionalTypes = "Image Files (" & strFileList & ") |" & strFileList
  
   ' force form to Dirty
   Me.ImageTitle = Me.ImageTitle
  
   ' open common 'file open' dialogue and get path to selected file
   OpenDlg.DialogTitle = "Select Image File"
   OpenDlg.AdditionalTypes = strAdditionalTypes
   strPath = OpenDlg.GetFileSpec
   Set OpenDlg = Nothing

   ' if file selected then set Picture property of Image control
   ' to path to file and enable ImageTitle control
   If Len(strPath) > 0 Then
       Me.ImagePath = strPath
       Me.Image1.Picture = strPath
       Me.Image1.Visible = True
   End If

  
Exit_here:
   ' reset cursor position
   ReturnCursor
   Exit Sub
  
Err_Handler:
   Select Case Err.Number
       Case 2001
       Resume
       Case Else
       MsgBox Err.Number, vbExclamation, "Error"
       Resume Exit_here
   End Select

End Sub

Private Sub cmdDeleteImage_Click()

   Me.ImagePath = Null
   Me.Image1.Visible = False

End Sub

Private Sub cmdReport_Click()

   On Error Resume Next
   DoCmd.OpenReport "rptImages", acViewPreview
  
End Sub

   GrabCursor
  
   If Not IsNull(Me.ImagePath) Then
       Me.Image1.Visible = True

   Dim pt As POINTAPI

   lngCursorY = pt.Y
