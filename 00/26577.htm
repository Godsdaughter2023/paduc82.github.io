Ok, run the following macro. It will add a paragraph break every 150 words.

Sub Macro1()
Dim oRng As Range
On Error Resume Next
    Set oRng = ActiveDocument.Range
    oRng.End = ActiveDocument.Range.Start
    Do While oRng.End < ActiveDocument.Range.End
        oRng.MoveEnd wdWord, 150
        oRng.InsertAfter vbCr
    Loop
End Sub
