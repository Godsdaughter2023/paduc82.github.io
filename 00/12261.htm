This one goes from the bottom up...

Sub InsertTransposeRow_R2()
 Dim TempRow As Long
 Dim FirstRw As Long
 Dim LastRw  As Long
 Dim N       As Long
 Dim InsertSize As Long

 FirstRw = 3
 LastRw = Cells(Rows.Count, 1).End(xlUp).Row
 InsertSize = 13

 For N = LastRw To FirstRw Step -1
  TempRow = N + InsertSize
  Range(Rows(N), Rows(N + InsertSize - 1)).Insert shift:=xlDown
  Cells(TempRow, "I").Resize(1, 14).Copy
  Cells(TempRow, "H").Resize(14, 1).PasteSpecial Paste:=xlPasteAll, Transpose:=True
  Cells(TempRow, "I").Resize(1, 14).Clear
 Next 'N
 Application.CutCopyMode = False
 Columns("H:H").WrapText = False
 Columns("H:H").AutoFit
 Range("A1").Select
End Sub
