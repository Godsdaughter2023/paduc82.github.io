    varOpenSave = False                         'Set Dialog to Save As mode
    
    sRptFileName = GetFileName(, varDialogTitle, varOpenSave)   'Get the path to output to.
    sPath = ParceFilePath(sRptFileName)                         'Parce the entered file name out of the string. The file name used will be programmed.

    Set rs = db.OpenRecordset(pLoopController, dbOpenDynaset)
            
    rs.MoveLast
    rs.MoveFirst
                    
    c = rs.RecordCount      'Store the record count
    i = 0                           'Set the counter to 0
            
    'size the array variables. Since Array element indexes start with 0, size upper limit of the array is 1 less than the record count
    ReDim sRefNbr(c - 1)
    ReDim sRecipientName(c - 1)
           
    'loop through controller and export data to a separate tab in the datafile workbook
    Do While i <= c - 1
        With rs
            'add recordset data to the array variables
            sRefNbr(i) = !RefNbr
            sRecipiantName(i) = !Recipient
            
            Forms!fMyForm!lblProcessNbr.Caption = "Process " & i + 1 & " of " & c
            Forms!fMyForm!lblProgress.Caption = "Exporting " & sRecipientName(i) & " to datafile"
            Call objPctMeter.PctMeter(((i + 1) / c) * 100, 100)
            Call objSleep.sSleep(2000)
            
            Forms!fMyForm.Repaint
            
            'create path and file name
            sFileName = sPath & sRefNbr(i) & " DataFile " & sRptDate & ".xlsx"
                                
            'build a sql string filtered on the broker name and deal package
            sExportQueryString = BuildExportQuery(sRefNbr(i), sRecipiantName(i))
            
            'insert sql string into a predetermined query
            Call BuildQuery(pExportQuery, sExportQueryString)
            
            'export to excel
            DoCmd.TransferSpreadsheet acExport, acSpreadsheetTypeExcel12Xml, pExportQuery, sFileName, True, sRecipiantName(i)
            
            'move to next record and repeat for all records in the loop controller
            .MoveNext
            
            'increment counter
            i = i + 1
        End With
    Loop

FINISH:

    Forms!fDPAdmin!baselbl.Caption = ""
    Call objPctMeter.PctMeter((0 / 2) * 100, 100)
    Call objSleep.sSleep(1000)
    Forms!fDPAdmin!lblProgress.Caption = "Completed"
    Call objSleep.sSleep(1000)
    Forms!fDPAdmin!lblProgress.Caption = ""
    
    ExportDataFiles= sFileName
    Exit Function
    
Error_Handler:
    'set error flag to 1, reset progress data on updater form
    intErrorFlag = 1
    strErrMessage = "Error number " & Err.Number & ": " & Err.Description
    MsgBox strErrMessage, vbCritical + vbOKOnly, "Error!!"
    sFileName = "Error"
    Resume FINISH

End Function
