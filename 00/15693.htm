jllatham say:

I took your code and modified it to do what I think you want it to do.  Test it, making sure you do so with a backup copy of your real data file just in case.

I also used the built in Application.FileDialog() as opposed to the API calls.

 

Function BrowseFolder(Title As String, _
  Optional InitialFolder As String = vbNullString, _
  Optional InitialView As Office.MsoFileDialogView = _
  msoFileDialogViewList) As String

  Dim V As Variant
  Dim InitFolder As String
  With Application.FileDialog(msoFileDialogFolderPicker)
    .Title = Title
    .InitialView = InitialView
    If Len(InitialFolder) > 0 Then
      If Dir(InitialFolder, vbDirectory) <> vbNullString Then
        InitFolder = InitialFolder
        If Right(InitFolder, 1) <> "\" Then
          InitFolder = InitFolder & "\"
        End If
        .InitialFileName = InitFolder
      End If
    End If
    .Show
    On Error Resume Next
    Err.Clear
    V = .SelectedItems(1)
    If Err.Number <> 0 Then
      V = vbNullString
    End If
  End With
  BrowseFolder = CStr(V)
End Function

Sub DirCheck()
  Dim FName As String
  Dim DispAlert As Boolean
  Dim i As Integer
 
  i = 1
 
'Browse for save location returns null string if cancelled,
'or full path to the folder without final \ path separator.
  FName = BrowseFolder(Title:="Select Save Location", InitialFolder:="Desktop")

  If FName = vbNullString Then
    MsgBox ("Process terminated")
    Exit Sub
  End If

'Ask user if they want to overwrite any existing files
  If MsgBox("Would you like to overwrite any existing files in the save location?", _
   vbYesNo) = vbYes Then
    DispAlert = False
  Else
    DispAlert = True
  End If
    
  If DispAlert = True Then
    'Do not overwrite existing files in folder.
    'Determine if file exists, if not save, otherwise append with (i)
    Do While Dir$(FName & "\Budget for " & Cells(6, 2).Text & _
     "(" & CStr(i) & ").xls") <> ""
      'need a new name!
      i = i + 1
    Loop
    Application.DisplayAlerts = False
    ActiveWorkbook.SaveAs _
    Filename:=FName & "\Budget for " & Cells(6, 2).Text & _
     "(" & CStr(i) & ").xls", _
     FileFormat:=56 ' 56=Excel 2003 file format, matches with ".xls"
    Application.DisplayAlerts = True
  Else
    'Save each new workbook and overwrite existing file.
    Application.DisplayAlerts = DispAlert
    ActiveWorkbook.SaveAs _
    Filename:=FName & "\Budget for " & Cells(6, 2).Text & ".xls", _
     FileFormat:=56 ' 56=Excel 2003 file format, matches with ".xls"
    Application.DisplayAlerts = True
   End If
    
End Sub

Sub ReplaceLettersWithAUTOTEXT()

' Replace text globally with Autotext Macro

' Macro created 21/06/2004 by Graham Mayor

' Revised 28/08/2004 with additions suggested by Greg Maxey

' Revised 01/08/2007 with modifications for Word 2007

' Revised 27/08/2007 with further modifications suggested by Greg Maxey

Dim findText As String

Dim ReplaceText As String

Dim strWildcards As String

Dim bWild As Boolean

Dim sQuery As String

Dim sType As String

Start:

findText = InputBox("Enter the text string you want to find", "Find")

If findText = "" Then

sQuery = MsgBox("You have not entered any text to find" & vbCr & "Or you have selected 'Cancel" & vbCr & _

"Select OK to re-try or Cancel to quit", vbOKCancel, "Find")

If sQuery = vbOK Then

GoTo Start

Else

Exit Sub

End If

End If

strWildcards = MsgBox("Use Wildcards", vbYesNo, "Find")

If strWildcards = 6 Then bWild = True Else bWild = False

GetInput:

On Error GoTo Oops 'Handle incorrect AutoText request

'Create a scratch pad

Documents.Add

If Val(Application.Version) > 11 Then

'Word 2007/2010 - Use the Building Blocks Organizer

Dialogs(GetDialog).Show

sType = "Building Blocks"

Else

'Not Word 2007 - Use the Autotext dialog.

Dialogs(wdDialogEditAutoText).Show

sType = "Autotext"

End If

'Cut the inserted entry to the clipboard

Selection.HomeKey Unit:=wdStory, Extend:=wdExtend

Selection.Cut

'crumple up scratch pad :-)

ActiveDocument.Close SaveChanges:=wdDoNotSaveChanges

'Replace found text with the clipboard contents.

With Selection.Find

.ClearFormatting

.Replacement.ClearFormatting

.Text = findText

.Replacement.Text = "^c"

.Forward = True

.Wrap = wdFindContinue

.Format = False

.MatchCase = True

.MatchWholeWord = False

.MatchWildcards = bWild

.MatchSoundsLike = False

.MatchAllWordForms = False

.Execute Replace:=wdReplaceAll

End With

End

Oops: 'Error handler

ActiveDocument.Close SaveChanges:=wdDoNotSaveChanges

sQuery = MsgBox("Select 'OK' to reselect the " & sType & " entry then click 'Insert'" & vbCr & vbCr & _

"Click 'Cancel' to exit", vbOKCancel, sType)

If sQuery = vbOK Then

Resume GetInput

End If

End Sub

    It shouldn’t really matter if you have another AV app installed, because Windows Defender isn’t supposed to be turned off via Group Policy when a third-party AV app gets installed, as I’ve explained in this thread:

    https://answers.microsoft.com/en-us/protect/forum/protect_defender-protect_start/windows-defender-is-off-i-cant-turn-it-on/c1ef0d9b-7940-4108-9578-de2c996b75ea

    However, it is possible that some errant AV/Antimalware app might be using Group Policy for this purpose – so kindly let us know if there’s any possibility of this being associated with the installation of a legitimate third-party app. Otherwise, on a home PC, we would strongly suspect that the Group Policy switch was thrown by malware that’s somehow managed to escalate its privilege-level to the rank of “administrator” – and scanning with this set of malware-removal tools is consequently recommended:

    Kaspersky Virus Removal Tool:

    http://support.kaspersky.com/viruses/kvrt2015

    Emsisoft Emergency Kit:

    http://www.emsisoft.com/en/software/eek/

    Malwarebytes Anti-Malware (free version only):

    https://www.malwarebytes.org/antimalware/

    Eset Online Scanner:

    http://www.eset.com/us/online-scanner/

    Some other trusted third-party malware-removal tools are listed here:

    https://answers.microsoft.com/en-us/protect/wiki/protect_other-protect_scanning/list-of-malware-removal-tools/d824b9af-ebd8-4c47-94e2-8ee6c544c100

    Then, when you're confident that the system is clean, attempt to turn Windows Defender back on: Note that there isn’t any Local Group Policy Editor in the Home edition of Windows 10, so local administrators often need to either edit the registry directly – or run this REG command in order to turn Defender on (right-click on the Start button, select Command Prompt (Admin), and then copy, paste, and run (enter) this command line):

    REG DELETE "HKLM\SOFTWARE\Policies\Microsoft\Windows Defender" /v DisableAntiSpyware
