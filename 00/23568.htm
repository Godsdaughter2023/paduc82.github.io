jp ronse say:

Progress bar

Hi All,

I was looking, yet a long time, for a nice solution to show the progress of a long routine (macro). When do you know when "it" hangs or is in an indefinite loop or is still continuing to process?

As usually, Google gives thousands of hits, one better then the other, but they could not charm me. Most of them were using a form what I would like to avoid. Yes, I know about Application.StatusBar="Hello World" and I used it before.

A year ago, I started with an add-in to speed up tasks that I had to perform on a regular base but I had no idea how long the macro would take to complete. One time, I had to walk through 1000 cells, another time it could be 100,000 cells.

Using Application.StatusBar is of course an easy way, but I had more requirements:
- Easy to implement in existing macro's just by adding of few statements;
- An update about each second;
- Show a progress bar;
- Show the estimated remaining time to complete

Finally, I came to the code hereafter. (With the help of Google.)

Sub StatusBar is the long routine
Herein:
- InitProgressBar: first step to initialize
- If glngStep = 0 Then InitProgressBar (UpTo): second step in initialization progress
- If x >= WorksheetFunction.Min(glngStep, UpTo) Then ProgressBar x, UpTo, glngRemainingTime, glngStep: update statusbar, show progress
- Exit_Sub:
   ResetProgressBar: reset the statusbar

It will look as this:



I hope you will find it useful. And of course, feedback is welcome.

With kind regards,

JP Ronse

Private Declare Function timeGetTime Lib "winmm.dll" () As Long

Public glngTimer              As Long        ''' Progress bar variable
Public glngStep               As Long        ''' Progress bar variable
Public glngRemainingTime      As Long        ''' Progress bar variable
Public gblnDisplayStatusBar   As Boolean     ''' Progress bar variable


Sub StatusBar()
   Dim x               As Long, UpTo As Long
   Dim MyTimer         As Double
  
   With Application
      .EnableCancelKey = xlInterrupt    ''' Should be xlErrorHandler
      .DisplayStatusBar = True
      .StatusBar = False
   End With
  
   UpTo = 500
   Application.DisplayStatusBar = True
   InitProgressBar
  
   For x = 1 To UpTo                       ''' the loop we have to go through
     
      MyTimer = Timer                      ''' just waisting time
      Do                                   ''' just waisting time
      Loop While Timer - MyTimer < 0.03    ''' just waisting time
     
     
      If glngStep = 0 Then InitProgressBar (UpTo)
     
     
      ''' update progress
      If x >= WorksheetFunction.Min(glngStep, UpTo) Then ProgressBar x, UpTo, glngRemainingTime, glngStep
        
     DoEvents
   Next x
  
Exit_Sub:
   ResetProgressBar
              
End Sub

Sub ProgressBar(Handled, ToHandle, RemainingTime, Step)
   Const BLOCK_CHAR As Long = 9605 '&H2585         ''' character used as progress character
   Const EMPTY_BLOCK As Long = 9641 '&H2610
   Static InitStep  As Long
  
   Dim strProgress  As String
   Dim dblProgress  As Double
  
   On Error GoTo Error_ProgressBar
  
   If InitStep = 0 Then InitStep = Step
  
   RemainingTime = RemainingTime - 1
   dblProgress = Handled / ToHandle
   strProgress = " " & Handled & "/" & ToHandle & " - " & Format(dblProgress, "Percent") & " - Remaining sec(s): " & RemainingTime
  
   With Application
      Select Case CDbl(dblProgress)
         Case Is >= 1: .StatusBar = False: InitStep = 0
         Case Is >= 0.9: .StatusBar = String$(17, ChrW(BLOCK_CHAR)) & String$(1, "_") & strProgress
         Case Is >= 0.8: .StatusBar = String$(16, ChrW(BLOCK_CHAR)) & String$(2, "_") & strProgress
         Case Is >= 0.7: .StatusBar = String$(14, ChrW(BLOCK_CHAR)) & String$(4, "_") & strProgress
         Case Is >= 0.6: .StatusBar = String$(12, ChrW(BLOCK_CHAR)) & String$(6, "_") & strProgress
         Case Is >= 0.5: .StatusBar = String$(10, ChrW(BLOCK_CHAR)) & String$(8, "_") & strProgress
         Case Is >= 0.4: .StatusBar = String$(8, ChrW(BLOCK_CHAR)) & String$(10, "_") & strProgress
         Case Is >= 0.3: .StatusBar = String$(6, ChrW(BLOCK_CHAR)) & String$(12, "_") & strProgress
         Case Is >= 0.2: .StatusBar = String$(4, ChrW(BLOCK_CHAR)) & String$(14, "_") & strProgress
         Case Is >= 0.1: .StatusBar = String$(2, ChrW(BLOCK_CHAR)) & String$(16, "_") & strProgress
         Case Is >= 0: .StatusBar = String$(1, ChrW(BLOCK_CHAR)) & String$(17, "_") & strProgress
         Case Else: .StatusBar = False: InitStep = 0
      End Select
   End With
   Step = Step + InitStep
   DoEvents
  
   Exit Sub
Error_ProgressBar:
   Application.StatusBar = False
   InitStep = 0
End Sub

Sub ResetProgressBar()
   glngTimer = 0
   glngStep = 0
   glngRemainingTime = 0
   With Application
      .StatusBar = False
      .DisplayStatusBar = gblnDisplayStatusBar
   End With
End Sub

Sub InitProgressBar(Optional ToHandle)
   If IsMissing(ToHandle) Then
      glngTimer = timeGetTime
      gblnDisplayStatusBar = Application.DisplayStatusBar
      Application.DisplayStatusBar = True
   Else
      glngTimer = timeGetTime - glngTimer
      glngRemainingTime = Int(CDec((glngTimer * ToHandle) / 1000)) + 1
      glngStep = Int(ToHandle / glngRemainingTime) + 1
   End If
End Sub

Private Declare Function timeGetTime Lib "winmm.dll" () As Long

Public Const PROG_BAR_INTERVAL   As Integer = 1000    ''' Progress bar constant
Public gblnDisplayStatusBar      As Boolean           ''' Progress bar variable
Public glngStartTimer            As Long              ''' Progress bar variable
Public glngNewTime               As Long              ''' Progress bar variable
Public glngToHandle              As Long              ''' Progress bar varialble
Public glngHandled               As Long              ''' Progress bar variable

Sub ProgressBar(Handled, ToHandle, NewTime, Optional PartNumb, Optional NumbLoops)
   Const BLOCK_CHAR           As Long = 9605    '''&H2585         character used as progress character
   Const EMPTY_BLOCK          As Long = 9641    '''&H2610         character to show the length of the progressbar
  
   Dim strProgress            As String
   Dim dblProgress            As Double
   Dim lngRemainingTime       As Long
   Static lngElapsedTime      As Long
   Static lngPass             As Long
  
   On Error GoTo Error_ProgressBar
  
   ''' calculate the progress already made
   dblProgress = Handled / ToHandle
  
   ''' do not display a progressbar when 75% of the progress has been done in 1 pass.
   If lngPass = 0 Then
      If dblProgress >= 0.75 Then
         Exit Sub
      Else
         lngPass = 1
      End If
   End If
  
   lngElapsedTime = lngElapsedTime + (NewTime - glngStartTimer)
   glngStartTimer = NewTime
   lngRemainingTime = Round(((ToHandle - Handled) * lngElapsedTime / Handled) / 1000, 0)
  
   strProgress = " " & Handled & "/" & ToHandle & " - " & Round(dblProgress, 3) * 100 & "% - Remaining sec(s): " & lngRemainingTime
  
   If IsMissing(PartNumb) = False And IsMissing(NumbLoops) = False Then strProgress = strProgress & " - Loop: " & PartNumb & "/" & NumbLoops
  
   With Application
      Select Case CDbl(dblProgress)
         Case Is >= 1: .StatusBar = String$(18, ChrW(BLOCK_CHAR)) & strProgress: lngElapsedTime = 0: lngPass = 0
         Case Is >= 0.9: .StatusBar = String$(17, ChrW(BLOCK_CHAR)) & String$(1, "_") & strProgress
         Case Is >= 0.8: .StatusBar = String$(16, ChrW(BLOCK_CHAR)) & String$(2, "_") & strProgress
         Case Is >= 0.7: .StatusBar = String$(14, ChrW(BLOCK_CHAR)) & String$(4, "_") & strProgress
         Case Is >= 0.6: .StatusBar = String$(12, ChrW(BLOCK_CHAR)) & String$(6, "_") & strProgress
         Case Is >= 0.5: .StatusBar = String$(10, ChrW(BLOCK_CHAR)) & String$(8, "_") & strProgress
         Case Is >= 0.4: .StatusBar = String$(8, ChrW(BLOCK_CHAR)) & String$(10, "_") & strProgress
         Case Is >= 0.3: .StatusBar = String$(6, ChrW(BLOCK_CHAR)) & String$(12, "_") & strProgress
         Case Is >= 0.2: .StatusBar = String$(4, ChrW(BLOCK_CHAR)) & String$(14, "_") & strProgress
         Case Is >= 0.1: .StatusBar = String$(2, ChrW(BLOCK_CHAR)) & String$(16, "_") & strProgress
         Case Is >= 0: .StatusBar = String$(1, ChrW(BLOCK_CHAR)) & String$(17, "_") & strProgress
         Case Else: .StatusBar = False
      End Select
   End With
  
   DoEvents
  
   Exit Sub
Exit_ProgressBar:
   With Application
      .StatusBar = False
   End With
Error_ProgressBar:
   Resume Exit_ProgressBar
End Sub

Sub ResetProgressBar()
   glngStartTimer = 0
   glngToHandle = 0
   glngHandled = 0
  
   With Application
      .StatusBar = False
      .DisplayStatusBar = gblnDisplayStatusBar
   End With
End Sub

Sub InitProgressBar(Optional ToHandle)
   If IsMissing(ToHandle) Then
      glngStartTimer = timeGetTime
      glngHandled = 0
      glngToHandle = 0
     
      With Application
         gblnDisplayStatusBar = .DisplayStatusBar
         .DisplayStatusBar = True
         .StatusBar = False
      End With
   Else
      glngToHandle = ToHandle
   End If
End Sub

jay freedman say:
Most likely you had (and still have) Tracked Changes turned on, and the bar shows where the changes (probably only the typing/pasting of text) occurred.

On the Review tab, if the Track Changes button is turned on (showing a yellow-orange highlight), click it to turn it off. Then click the bottom half of the Accept button on the same tab, and click Accept All Changes in Document. That should make the bar disappear.

The default keyboard shortcut for Track Changes is Ctrl+Shift+E, which is easily pressed by mistake when you mean to press just Shift+E. That's probably how the problem started. For more information and hints, read http://www.shaunakelly.com/word/sharing/howtrackchangesworks.html.
