john sr wilson say:

As I said earlier PPT does not support full screen from the GUI for online video.

When you play a You tube video the normal Full Screen button is also disabled.

If you add the video via a Shockwave flash object you can enable the Full Screen button. This might give you what you need.

You can do this all manually but the code below automates it.

Make sure you have the URL to the You tube Clip

Open the vb editor ALT f11

Insert >> Module and paste in this code

'Start

Sub insertUTube()
    Dim osld   As Slide
    Dim oshp   As Shape
    Dim sngW   As Single
    Dim sngH   As Single
    Dim SR     As ShapeRange
    Dim uText As String
    sngH = 180
    sngW = 240
    
    Set osld = ActiveWindow.Selection.SlideRange(1)
    If Not osld Is Nothing Then
    uText = InputBox("URL here")
        If LCase(uText) Like "*watch?*" Then
            uText = Replace(uText, "watch?v=", "v/")
            uText = uText & "&autoplay=1"
            Set oshp = osld.Shapes.AddOLEObject(Left:=0, Top:=0, Width:=sngW, Height:=sngH, ClassName:="ShockwaveFlash.ShockwaveFlash.1", Link:=msoFalse)

            If Err.Number = -2147467259 Then
                MsgBox "Unable to locate the Shockwave Flash ActiveX control. " & vbCrLf & "You may need to reinstall it.", vbCritical
                Exit Sub
            End If
        End If
        With oshp.OLEFormat.Object
            .Movie = uText
            .Playing = "True"
            .FrameNum = -1
            .AllowFullScreen = True
        End With
        oshp.Select
        Set SR = ActiveWindow.Selection.ShapeRange
        SR.Align msoAlignCenters, True
        SR.Align msoAlignMiddles, True
    Else
        MsgBox "That doesn't appear to be a valid You Tube URL.", vbExclamation
    End If
End Sub

'End

Run the macro (VIEW Tab) > Enter the URL and you should get a flash object where the You Tube Full screen is enabled.

If you are not sure with code make sure you do this on a presentation that is not important!
