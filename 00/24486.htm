Hi,

try this sample,

select a range and run the code...

 

Sub Excel_To_Word()

'25Jan2013
If MsgBox("select a range ?", vbOKCancel) = vbCancel Then Exit Sub

Application.ScreenUpdating = False
Dim wdApp As Object
Set wdApp = CreateObject("Word.Application")
Dim wdDoc
Set wdDoc = wdApp.Documents.Add
wdApp.Visible = False
Selection.Copy
wdDoc.Content.PasteSpecial Link:=False, DataType:=20, Placement:=wdInLine, _
DisplayAsIcon:=False
Application.CutCopyMode = False
Application.DisplayAlerts = False
wdDoc.SaveAs (ThisWorkbook.Path & "\myWord File.docx")
Application.DisplayAlerts = True
wdDoc.Close
wdApp.Quit
Application.ScreenUpdating = True
MsgBox "done.."

End Sub
