'---------------------------------------------------
'Modify the nominated file by setting byte 4 to $03.
'Created for JD's .MP3 files.
'8.9.2015 FNL

Set oFSO = CreateObject("Scripting.Filesystemobject")
If WScript.Arguments.Count = 0 Then
    MsgBox "No file name supplied. Program terminated.", 48
    WScript.Quit
End If

sOldName = WScript.Arguments(0)
Set oFile = oFSO.GetFile(sOldName)
iSize = oFile.Size
p = InStrRev(sOldName, ".")
sNewName = Left(oFile.Path, p-1) & "(Modified)" & Mid(sOldName, p)

If MsgBox("Modifying """ & sOldName & """." & vbLf _
    & "This process can take 30 seconds or more." & vbLf & vbLf _
    & "Please click OK to start", 1) <> 1 Then WScript.Quit
    
Set oFile = oFSO.OpenTextFile(sOldName)
sData = oFile.Read(iSize)
oFile.Close

Set oFile = oFSO.CreateTextFile(sNewName, True)
oFile.Write(Left(sData, 3))
oFile.Write(Chr(3))
oFile.Write(Mid(sData, 5))
oFile.Close

MsgBox "Created """ & sNewName & """", 0
