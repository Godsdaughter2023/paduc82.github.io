Hi,

On your active worksheets, add the follwowing column heading

FOLDER FILE Last saved LINK

Insert a userform, add a label named label1, a lisbox named list1 and a command button named CommandButton1

Make a ref to Microsoft Scripting Runtime library

Add the code below to the user form:

Private Sub CommandButton1_Click()
   
    Dim nDirs As Double, nfiles As Double, lSize As Double
    Dim sDir As String, sSrchString As String
    Range("A2").Activate
    sDir = InputBox("Please enter the Directory to search", "FileSytemObjects example", "C:\Temp\")
    sSrchString = InputBox("Please enter the file name to search", "FileSytemObjects example", "*.mp3")
    Application.ScreenUpdating = False
    MousePointer = fmMousePointerHourGlass
    Application.ScreenUpdating = False
    Label1.Caption = "Searching " & vbCrLf & UCase(sDir) & "..."
    lSize = FindFile(sDir, sSrchString, nDirs, nfiles)
    Application.ScreenUpdating = True
    MousePointer = fmMousePointerDefault
   
    MsgBox Str(nfiles) & " files found in " & Str(nDirs) & " directories", vbInformation
    MsgBox "Total size= " & lSize & " bytes"
    Application.ScreenUpdating = True
   
   
End Sub

Private Function FindFile(ByVal sfol As String, sfile As String, nDirs As Double _
                        , nfiles As Double) As Double
    Dim FileName As String
    Dim tfld As Folder
    Dim f As File
    Set fld = fso.GetFolder(sfol)
    On Error Resume Next
    FileName = Dir(fso.BuildPath(fld.Path, sfile), vbNormal Or vbHidden Or vbSystem Or vbReadOnly)
   
    While Len(FileName) <> 0
        FindFile = FindFile + FileLen(fso.BuildPath(fld.Path, FileName))
        nfiles = nfiles + 1
        Set f = fso.GetFile(fso.BuildPath(fld.Path, FileName))
        ActiveCell.Value = fld.Path
        ActiveCell.Offset(0, 1).Value = FileName
        ActiveCell.Offset(0, 2).Value = f.DateLastModified
        ActiveSheet.Hyperlinks.Add ActiveCell.Offset(0, 3), fso.BuildPath(fld.Path, FileName), , , "Open"
     
        ActiveCell.Offset(1, 0).Activate
        List1.AddItem fso.BuildPath(fld.Path, FileName)
       
        FileName = Dir()
            DoEvents
    Wend
   
    Label1 = "Searching " & vbCrLf & fld.Path & "..."
    nDirs = nDirs + 1
    If fld.SubFolders.Count > 0 Then
        For Each tfld In fld.SubFolders
        DoEvents
        FindFile = FindFile + FindFile(tfld.Path, sfile, nDirs, nfiles)
        Next
    End If
   
End Function

wendy_jones say:
Hello, my name is Wendy Jones, I am an Independent Advisor.

First of all I suggest to do an Office online repair as show the following link:

https://support.office.com/en-gb/article/repair...

If the issue still persist you need to clear Office keys existing before with these steps:

Remove product key
1. Locate the one in the path where the "ospp.vbs" file is located, usually it is:
1. C: \ Program Files (x86) \ Microsoft Office \ Office16
2. <C: \ Program Files \ Microsoft Office \ Office16>
2. Open a command console by specifying with CLIK RIGHT "Run as Administrator" and execute the command cd <C: \ Program Files \ Microsoft Office \ Office16> (or C: \ Program Files (x86) \ Microsoft Office \ Office16 depending on the path where the "ospp.vbs" file is located).
3. Execute the following command in the console: cscript ospp.vbs / dstatus
4. * Take note of the results of "product key", these would be 5 uppercase characters, for example 2PK9C
5. Execute the following command in the console: cscript ospp.vbs / unpkey: 2PK9C
 
Edit the record to delete cached credentials. To do this, follow these steps:
1. Click Start >> Run, type regedit, and then click OK.
2. In the Registry Editor, locate the following registry subkey: HKEY_CURRENT_USER \ Software \ Microsoft \ Office \ 16.0 \ Common \ Identity \ Identities
3. Remove all keys listed within this record.

Restart the system and try the activation again.

Let me know if the problem was solved with these steps. Good luck!!!
