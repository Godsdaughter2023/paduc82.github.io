jay freedman say:

You're misunderstanding the information that Paul gave you. When the Find box contains ^t^p, that will match only the combination of a tab character followed by a paragraph mark. It will not match a tab character followed by a digit or any other character.

Similarly, ^w^p will match any white space followed by a paragraph mark, but not white space followed by a digit.

macropod say:

You could do this with Find/Replace where:

Find = ^t^p

Replace = ^p

If there could be other white-space content in addition to the tabs, you could use a Find/Replace where:

Find = ^w^p

Replace = ^p

Private Sub App_SheetSelectionChange(ByVal Sh As Object, ByVal Target As Range)
  If Abs(StrComp(DestWB.Name, ThisWorkbook.Name, vbTextCompare)) <> DestToSource Then
    If IsArray(LastValue) And Target.Count = 1 Then
      'Store matrix
      Target.Resize(UBound(LastValue), UBound(LastValue, 2)) = LastValue
    Else
      'Store the value
      Target = LastValue
    End If
    'Remove the marker
    Application.CutCopyMode = False
  Else
    'Get the value
    LastValue = Target
    'Show that we work
    Target.Copy
  End If
End Sub

andreas killer say:

Am 10.09.2010 10:36, schrieb stefi2:

In other words, I want to copy a value from the second workbook to a cell in the first workbook only by clicking on the appropriate cells.

Store the code below in the code module "ThisWorkBook" of the second workbook that holds the values, save, close and reopen that workbook. (This force Workbook_Open to run)

If you want to change the direction (copy values from first workbook to second workbook) call the macro ChangeClickCopyDirection. If you want to have this direction by default place the line
  DestToSource = 1
in the sub Workbook_Open.

And perhaps you will find the macros DisableClickCopy/EnableClickCopy helpful.

Andreas.

Option Explicit

Dim WithEvents App As Application
Dim DestWB As Workbook
Dim LastValue As Variant
Dim DestToSource As Integer

Private Sub App_SheetSelectionChange(ByVal Sh As Object, ByVal Target As Range)
  If Abs(StrComp(DestWB.Name, ThisWorkbook.Name, vbTextCompare)) <> DestToSource Then
    'Store the value
    Target = LastValue
  Else
    'Get the value
    LastValue = Target
  End If
End Sub

Private Sub App_WorkbookActivate(ByVal Wb As Workbook)
  Set DestWB = Wb
End Sub

Private Sub Workbook_Open()
  Set App = Application
  Set DestWB = ThisWorkbook
End Sub

Sub ChangeClickCopyDirection()
  DestToSource = Abs(DestToSource - 1)
End Sub

Sub DisableClickCopy()
  Set App = Nothing
End Sub

Sub EnableClickCopy()
  Workbook_Open
End Sub
