Andreas.

Sub Test()
  Dim Data As Variant   '2D array
  Dim Result As Variant '1D array
  Dim i As Long, r As Long
  Dim m As Double

  'Read the 2D array from the sheet
  Data = Range("A1").CurrentRegion

  'Search the min. value in the first column
  m = 1E+308
  For i = 1 To UBound(Data)
    If Data(i, 1) < m Then
      'Save the min. value
      m = Data(i, 1)
      'remember this row
      r = i
    End If
  Next

  'Create the 1D array
  ReDim Result(1 To UBound(Data, 2))

  'Copy the data from the row
  For i = 1 To UBound(Data, 2)
    Result(i) = Data(r, i)
  Next
End Sub
