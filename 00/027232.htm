Dim NextRow As Long
Const strWorkbookname As String = "C:\Path\WorkBook.xlsx"
    On Error Resume Next
    Set xlApp = GetObject(, "Excel.Application")
    If Err Then
        Set xlApp = CreateObject("Excel.Application")
    End If
    On Error GoTo 0
    Set xlBook = xlApp.workbooks.Open(Filename:=strWorkbookname)
    xlApp.Visible = True

    For j = 1 To ActiveDocument.Tables.Count
        For Each oTable In ActiveDocument.Tables(j).Tables
            NextRow = xlBook.sheets(1).Range("A" & xlBook.sheets(1).Rows.Count).End(-4162).Row + 1

            'Column A - ID
            Set oRng = oTable.Range.Cells(3).Range
            oRng.End = oRng.End - 1
            strID = oRng.Text
            xlBook.sheets(1).Range("A" & NextRow) = strID

            'Column B - Salutation
            Set oRng = oTable.Range.Cells(2).Range
            oRng.End = oRng.End - 1
            strName = oRng.Text
            vName = Split(strName, Chr(46))
            xlBook.sheets(1).Range("B" & NextRow) = vName(0)

            'Column C - Name
            For i = 1 To UBound(vName)
                xlBook.sheets(1).Range("C" & NextRow) = xlBook.sheets(1).Range("C" & NextRow) & vName(i)
                If i < UBound(vName) Then xlBook.sheets(1).Range("C" & NextRow) = xlBook.sheets(1).Range("C" & NextRow) & Chr(46)
            Next i

            'Column D - Full Address
            Set oRng = oTable.Range.Cells(5).Range
            oRng.End = oRng.End - 1
            strAddress = oRng.Text
            xlBook.sheets(1).Range("D" & NextRow) = strName & Chr(32) & strID & Chr(32)
            vAddress = Split(strAddress, Chr(13))
            For i = 0 To UBound(vAddress)
                xlBook.sheets(1).Range("D" & NextRow) = xlBook.sheets(1).Range("D" & NextRow) & vbCrLf & vAddress(i)
            Next i

            'Columns E & F email and phone
            strEmail = vbNullString
            For i = 0 To UBound(vAddress)
                If Left(UCase(vAddress(i)), 4) = "TEL." Then
                    strPhone = Trim(Right(vAddress(i), Len(vAddress(i)) - 4))
                    vPhone = Split(strPhone, Chr(32))
                End If
                If Left(UCase(vAddress(i)), 7) = "E-MAIL:" Then
                    strEmail = Trim(Right(vAddress(i), Len(vAddress(i)) - 7))
                End If
                If InStr(1, strPhone, Chr(32)) > 0 Then
                    strPhone = ""
                    For k = 1 To UBound(vPhone)
                        strPhone = strPhone & Chr(32) & vPhone(k)
                    Next k
                    xlBook.sheets(1).Range("E" & NextRow) = Trim(strPhone)
                    xlBook.sheets(1).Range("H" & NextRow) = vPhone(0)
                Else
                    xlBook.sheets(1).Range("E" & NextRow) = Trim(strPhone)
                End If
                xlBook.sheets(1).Range("F" & NextRow) = Replace(strEmail, "*", "")
            Next i

            'Column G - Full Name
            xlBook.sheets(1).Range("G" & NextRow) = strName

        Next oTable
    Next j
End Sub

ronrosenfeld say:
On Tue, 22 Feb 2011 21:32:44 +0000, BrettM WinTeam wrote:
 
>Hi friends, I need a formula to calculate dates in a year avoiding weekends. I?m trying to schedule visits for a clients every 90 days avoiding weekends in one year.
>Thanks
 
If day 90 falls on a weekend (or holiday), do you want the scheduled day to be the day before?  or after?  the weekend or holiday day?
 
With your date in A1, to use the day before (e.g. Friday):  =WORKDAY(A1+91,-1)
        to use the day after (e.g. Monday):    =WORKDAY(A1+89,1)
 
Note that the WORKDAY function has an optional Holidays argument which you could incorporate into the above function.
 

hansv mvp say:

If you have Excel 2007 or later, the WORKDAY function is built in, for older versions you need to activate the Analysis ToolPak add-in in Tools | Add-Ins...

With the starting date in A1, the formula =WORKDAY(A1,90) will return the date 90 days later excluding weekend days.

If you create a range containing public holidays and name it Holidays, you can even use =WORKDAY(A1,90,Holidays) to exclude holidays too.
