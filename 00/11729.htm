greg maxey say:

Try:

Sub SuperS()
Dim oRng As Word.Range
  Set oRng = ActiveDocument.Range
  With oRng.Find
    .Text = "([0-9]{1,2})([dhnrst]{2})"
    .Forward = True
    .Wrap = wdFindStop
    .Format = True
    .MatchCase = False
    .MatchWholeWord = False
    .MatchAllWordForms = False
    .MatchSoundsLike = False
    .MatchWildcards = True
    While .Execute
      Do While IsNumeric(oRng.Characters.First)
        oRng.MoveStart wdCharacter, 1
      Loop
      oRng.Font.Superscript = True
      oRng.Collapse wdCollapseEnd
    Wend
  End With
End Sub
