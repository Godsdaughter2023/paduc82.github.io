andreas killer say:

IMHO to complicated for a formula, try the UDF below.

It is not clear which result your want, the code below works for both.

Andreas.

Option Explicit

#Const Version = 2

#If Version = 1 Then

Function MyReplace(ByVal Expression As String) As String
  '123@A45 234@B5 7@C6 => 123AA45 234BB5 7CC6
  Dim i As Long
  For i = Len(Expression) - 1 To 1 Step -1
    If Mid$(Expression, i, 1) = "@" Then
      Mid$(Expression, i, 1) = Mid$(Expression, i + 1, 1)
    End If
  Next
  MyReplace = Expression
End Function

#Else

Function MyReplace(ByVal Expression As String) As String
  '123@A45 234@B5 7@C6 => 123_AA45 234_BB5 7_CC6
  Dim i As Long
  Expression = Replace(Expression, "@", "@@")
  For i = Len(Expression) - 1 To 1 Step -1
    If Mid$(Expression, i, 2) = "@@" Then
      Mid$(Expression, i, 2) = "_" & Mid$(Expression, i + 2, 1)
    End If
  Next
  MyReplace = Expression
End Function

#End If

This is not something that you can do readily with mail merge, but is easy enough with a macro. If you have a data source that looks like

and you have a document with a table e.g.

then you can read the data into an array and use that to populate the table according to the grading to produce a result like

Option Explicit
Const strWorkbook As String = "C:\Path\Forums\Students.xlsx" 'The name and path of the data source
Const strSheet As String = "Sheet1"        'The name of the worksheet
Const strPath As String = "C:\Path\Documents\" ' The path to save the documents
Const strDoc As String = "C:\Path\Forums\Student Merge.docx" 'The name and path of the merge document

Sub MergeDoc()
'A Word example macro Graham Mayor - www.gmayor.com
Dim Arr() As Variant
Dim i As Long, j As Long
Dim oDoc As Document
Dim oTable As Table
    Arr = xlFillArray(strWorkbook, strSheet)
    For i = 0 To UBound(Arr, 2)
        Set oDoc = Documents.Open(FileName:=strDoc, AddToRecentfiles:=False)
        Set oTable = oDoc.Tables(1)
        FillBM "bmStudent", CStr(Arr(0, i))
        For j = 1 To 5
            Select Case CStr(Arr(j, i))
                Case "A"
                    oTable.Rows(j).Cells(2).Shading.BackgroundPatternColor = wdColorYellow
                Case "B"
                    oTable.Rows(j).Cells(3).Shading.BackgroundPatternColor = wdColorYellow
                Case "C"
                    oTable.Rows(j).Cells(4).Shading.BackgroundPatternColor = wdColorYellow
                Case "D"
                    oTable.Rows(j).Cells(5).Shading.BackgroundPatternColor = wdColorYellow
                Case Else
                    oTable.Rows(j).Cells(6).Shading.BackgroundPatternColor = wdColorYellow
            End Select
        Next j
        oDoc.SaveAs2 FileName:=strPath & CStr(Arr(0, i)) & ".docx", AddToRecentfiles:=False
        oDoc.Close 0
    Next i
lbl_Exit:
    Exit Sub
End Sub

Private Function xlFillArray(strWorkbook As String, _
                             strWorksheetName As String) As Variant
'An Office VBA function Graham Mayor - www.gmayor.com
Dim RS As Object
Dim CN As Object
Dim iRows As Long

    strWorksheetName = strWorksheetName & "$]"
    Set CN = CreateObject("ADODB.Connection")
    CN.Open ConnectionString:="Provider=Microsoft.ACE.OLEDB.12.0;" & _
                              "Data Source=" & strWorkbook & ";" & _
                              "Extended Properties=""Excel 12.0 Xml;HDR=YES"";"

    Set RS = CreateObject("ADODB.Recordset")
    RS.Open "SELECT * FROM [" & strWorksheetName, CN, 2, 1

    With RS
        .MoveLast
        iRows = .RecordCount
        .MoveFirst
    End With
    xlFillArray = RS.GetRows(iRows)
    If RS.State = 1 Then RS.Close
    Set RS = Nothing
    If CN.State = 1 Then CN.Close
    Set CN = Nothing
lbl_Exit:
    Exit Function
End Function

Private Sub FillBM(strBMName As String, strValue As String)
'An Office VBA function Graham Mayor - www.gmayor.com
Dim oRng As Range
    With ActiveDocument
        On Error GoTo lbl_Exit
        Set oRng = .Bookmarks(strBMName).Range
        oRng.Text = strValue
        oRng.Bookmarks.Add strBMName
    End With
lbl_Exit:
    Set oRng = Nothing
    Exit Sub
End Sub
