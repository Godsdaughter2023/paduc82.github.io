I don't have enough information as to your pivot tables but try this to get strarted.

Sub Split_Companies()
    Dim d As Long, s As Long, dict As New Scripting.Dictionary
    dict.CompareMode = TextCompare
    With Sheets("DataSheet")
        For d = 2 To .Cells(Rows.Count, "B").End(xlUp).Row
            If CBool(Len(Trim(.Cells(d, "B").Value))) Then
                If Not dict.Exists(Trim(.Cells(d, "B").Value)) Then
                    dict.Add Key:=Trim(.Cells(d, "B").Value), Item:=vbNullString
                End If
            End If
        Next d
        For d = LBound(dict.Keys) To UBound(dict.Keys)
            Application.DisplayAlerts = False
            For s = 1 To Sheets.Count
                If LCase(Sheets(s).Name) = LCase(dict.Keys(d)) Then
                    Sheets(dict.Keys(d)).Delete
                    Exit For
                End If
            Next s
            Application.DisplayAlerts = True
            .Copy After:=Sheets(Sheets.Count)
            Sheets(Sheets.Count).Name = dict.Keys(d)
            With Sheets(dict.Keys(d)).Cells(1, 1).CurrentRegion
                .AutoFilter
                .AutoFilter field:=2, Criteria1:="<>" & dict.Keys(d)
                If Application.Subtotal(103, .Columns("B")) > 1 Then _
                    .Offset(1, 0).EntireRow.Delete
                .AutoFilter
            End With
        Next d
    End With
    dict.RemoveAll
    Set dict = Nothing
End Sub

While still in the VBE, choose Tools, References. Locate Microsoft Scripting Runtime in the Available References: list and put a check beside it then click OK in the top right. When you are back at the code sheet, tap Alt+Q to return to your worksheet then Alt+F8 to Run the macro.

Addendum: this removed worksheets with similar names so you might want to test it on a copy of your original.

Sub SplitData_newWBKS()

Const sht As String = "Sheet1" '<<< Sheet Name, change

Set myWB = ThisWorkbook
Dim ws As Worksheet
Set ws = myWB.Sheets(sht)

r1 = ws.Cells(Rows.Count, "B").End(xlUp).Row
Dim myPath
myPath = myWB.Path & "\"
Application.ScreenUpdating = False
ws.Range("B1:B" & r1).Copy Range(Hc & "1")
ws.Range(Hc & "1:" & Hc & r1).RemoveDuplicates Columns:=1, Header:=xlYes
r2 = ws.Cells(Rows.Count, Hc).End(xlUp).Row
For x = 2 To r2
Set newWB = Workbooks.Add(xlWorksheet)
With ws
.AutoFilterMode = False
.Range("B1").Resize(r1).AutoFilter field:=1, Criteria1:=ws.Cells(x, Hc)
.Range("A1").CurrentRegion.SpecialCells(xlCellTypeVisible).Copy Range("A1")
.AutoFilterMode = False
End With
ActiveSheet.UsedRange.EntireColumn.AutoFit
Application.DisplayAlerts = False
newWB.SaveAs Filename:=myPath & ws.Cells(x, Hc).Value & "-" & ws.Range("C2:C" & r1).SpecialCells(xlCellTypeVisible)(1) & ".xlsx"
Application.DisplayAlerts = True
newWB.Close False
Next
ws.Range(Hc & "1:" & Hc & r2).Clear
ActiveWorkbook.Save
Application.ScreenUpdating = True
MsgBox "done"
End Sub
