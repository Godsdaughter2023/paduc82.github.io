bernie deitrick say:

To start at row 2, and go to the bottom of column M, you need to have a loop:

Sub datesexcelvba2()
    Dim myApp As Outlook.Application
    Dim mymail As Outlook.MailItem
    Dim x As Long
    
    Set myApp = New Outlook.Application
    
    'Assuming your data starts in row 2
    For x = 2 To Cells(Rows.Count, 13).End(xlUp).Row
        If Cells(x, 13).Value = "Attention Needed" Then
            
            Set mymail = myApp.CreateItem(olMailItem)
            mymail.To = Cells(x, "B").Value ' if the email address is stored onthe same row, column B
            
            With mymail
                .Subject = "Testing Testing"
                .Body = "Hey it works!!!"
                .Send
            End With
            
        End If
    Next x
    
    Set myApp = Nothing
    Set mymail = Nothing
End Sub
