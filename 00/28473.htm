lars-Åke aspelin say:

Notes:

- All formulas in the merged cells, except in the top left cell of each merged area, will be destroyed.

- Empty cells will not be merged.

- Horizontal alignment comes first, Vertical alignment comes next.

- If you don't need vertical alignment, some parts of the macro (se below) can be omitted.

- All merged areas are rectangular, i.e. no areas with more than four corners, like "L-shaped" areas, can be done.

Try this macro:

Sub merge_cells_with_same_content()
    
  If Selection Is Nothing Then
     Exit Sub
  End If
    
' Remove all merging in the selected area
    For Each c In Selection
       Set ma = c.MergeArea
       If c.MergeCells Then
         v = c.Value
         ma.UnMerge
         For Each c2 In ma
            c2.Value = v
         Next
       End If
    Next
     
' Merge horizontally adjacent cells with same values
     For Each c In Selection
       If Not c.MergeCells And c.Value <> "" Then
           k = 1
          While c.Offset(0, k).Value = c.Value And Not Intersect(Selection, c.Offset(0, k)) Is Nothing
            k = k + 1
          Wend
          If k > 1 Then
             Application.DisplayAlerts = False
             c.Resize(1, k).Merge
             Application.DisplayAlerts = True
             c.HorizontalAlignment = xlCenter
             c.VerticalAlignment = xlCenter
          End If
       End If
    Next
    
'Merge vertically adjacent mergeareas with same values (skip this section if you don't need vertical merging)
    Ready = False
    While Not Ready
      Ready = True
      For Each c In Selection
        Set ma1 = c.MergeArea
        If c.MergeCells Then
          Set ma2 = c.Offset(1, 0).MergeArea
          If c.Offset(1, 0).MergeCells Then
            If c.Value <> "" And c.Value = c.Offset(1, 0).Value And ma1.Columns.Count = ma2.Columns.Count Then
              ma1.UnMerge
              ma2.UnMerge
              Set ma = Union(ma1, ma2)
              Application.DisplayAlerts = False
              ma.Merge
              Application.DisplayAlerts = True
              Ready = False
            End If
          End If
        End If
      Next
    Wend
    
' Merge vertically adjacent and not yet merged cells with same values (skip this if you don't need vertical alignment)
    For Each c In Selection
       If Not c.MergeCells And c.Value <> "" Then
          Set ma = c
           k = 1
           While Not c.Offset(k, 0).MergeCells And c.Offset(k, 0).Value = c.Value And Not Intersect(Selection, c.Offset(k, 0)) Is Nothing
            k = k + 1
          Wend
          If k > 1 Then
             Application.DisplayAlerts = False
             c.Resize(k, 1).Merge
             Application.DisplayAlerts = True
             c.HorizontalAlignment = xlCenter
             c.VerticalAlignment = xlCenter
          End If
       End If
    Next
   
End Sub

Hope this helps / Lars-Åke
