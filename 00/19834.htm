Try the following code:

 

Sub DelRows()
    Dim WS As Worksheet
    Dim N As Long
    Dim DelRow As Boolean
    Dim LastRow As Long
    Set WS = Worksheets("Sheet1") '<<< CHANGE
    With WS
        LastRow = .Cells(.Rows.Count, "A").End(xlUp).Row
        For N = LastRow To 1 Step -1
            DelRow = False
            If .Cells(N, "A") <> vbNullString Then
                If IsNumeric(Left(.Cells(N, "A"), 1)) = True Then
                    DelRow = True
                End If
            End If
            If .Cells(N, "B") = vbNullString Then
                DelRow = True
            End If
            If DelRow = True Then
                .Rows(N).Delete
            End If
        Next N
    End With
End Sub
