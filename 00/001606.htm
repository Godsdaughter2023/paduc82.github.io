'---
Sub ShowPrinterSelection()

 Dim bChoice As Boolean
 bChoice = Application.Dialogs(xlDialogPrinterSetup).Show(ActivePrinter)

 If Not bChoice Then
    MsgBox "User cancelled"
 Else

 End If
End Sub
