Private Sub Worksheet_Change(ByVal Target As Range)
    If Not Intersect(Target, Me.Range("A2:A10")) Is Nothing Then
        Application.EnableEvents = False
        Dim a As Range
        For Each a In Intersect(Target, Me.Range("A2:A10"))
            Select Case a
                Case 1
                    Target.Offset(0, 1).Resize(1, 4).Formula = "=AA1"
                Case 2
                    Target.Offset(0, 1).Resize(1, 4).Formula = "=AB1"
                Case 3
                    Target.Offset(0, 1).Resize(1, 4).Formula = "=AC1"
                Case Else
                    'do nothing
            End Select
        Next a
    End If
Fallthrough:
    Application.EnableEvents = True
End Sub

Sub mcr_Populate_Import_List()

    For Each a In Range("A2:A" & Cells(Rows.Count, 1).End(xlUp).Row)
