Sub Take4()
    Dim Rng As Range, StartCell  As Range, ProcessRange As Range

You might like to take a look at DatabaseBasics.zip in my public databases folder at:

 

https://onedrive.live.com/?cid=44CC60D7FEA42912&id=44CC60D7FEA42912!169

 

Note that if you are using an earlier version of Access you might find that the colour of some form objects such as buttons shows incorrectly and you will need to amend the form design accordingly. 

 

If you have difficulty opening the link copy its text (NB, not the link location) and paste it into your browser's address bar.

 

In this little demo file, which illustrates some of the basic principles in designing a simple relational database, take a look at the option: '6. Entering data via a form/subforms'.  This form uses correlated combo boxes to allow a country to be first selected, then a region within the selected country and finally a city within the selected region. The important thing to note is that only the CityID is entered via a bound control and stored in the table.  If the region and country were also stored in columns in the table this would introduce  redundancies as city determines region, and region determines country, so the table would not be normalized to Third Normal Form, and consequently open to the risk of update anomalies.

 

The relevant code in the form's module which populates the unbound controls and allows new country, region and city values not currently represented in the database to be entered via them is as follows.  The code is extensively commented to explain how it works:

 

Option Compare Database

Option Explicit

 

 

Private Sub cboCity_NotInList(NewData As String, Response As Integer)

 

    Const MESSAGE_TEXT = "No region selected."

    Dim ctrl As Control

    Dim strSQL As String, strMessage As String

   

    Set ctrl = Me.ActiveControl

   

    ' abort insertion of new region if no region selected

    If IsNull(Me.cboRegion) Then

        MsgBox MESSAGE_TEXT, vbExclamation, "Invalid Operation"

        Response = acDataErrContinue

        ctrl.Undo

        Exit Sub

    End If

       

    ' subject to user confirmation insert a new row into Cities table,

    ' inserting NewData value into City column and value of cboRegion

    ' control into RegionID column

    strMessage = "Add " & NewData & " to list?"

   

    strSQL = "INSERT INTO Cities(City,RegionID) VALUES(""" & _

            NewData & """," & Me.cboRegion & ")"

  

    If MsgBox(strMessage, vbYesNo + vbQuestion) = vbYes Then

        CurrentDb.Execute strSQL, dbFailOnError

        Response = acDataErrAdded

    Else

        Response = acDataErrContinue

        ctrl.Undo

    End If

 

End Sub

 

Private Sub cboCountry_AfterUpdate()

 

    ' if Country is updated then erase current values

    ' for Region and City and requery combo boxes

    ' to show regions of selected country and empty list

    ' of cities (pending selection of a region)

    Me!cboRegion = Null

    If Not Nz(Me!cboCity) = 0 Then

        Me!cboCity = Null

    End If

    Me!cboRegion.Requery

    Me!cboRegion.Requery

 

End Sub

 

 

Private Sub cboCountry_NotInList(NewData As String, Response As Integer)

 

    Dim ctrl As Control

    Dim strSQL As String, strMessage As String

   

    Set ctrl = Me.ActiveControl

   

    ' subject to user confirmation insert a new row into Countries table,

    ' inserting NewData value into Country column

    strMessage = "Add " & NewData & " to list?"

   

    strSQL = "INSERT INTO Countries(Country) VALUES(""" & _

            NewData & """)"

  

    If MsgBox(strMessage, vbYesNo + vbQuestion) = vbYes Then

        CurrentDb.Execute strSQL, dbFailOnError

        Response = acDataErrAdded

    Else

        Response = acDataErrContinue

        ctrl.Undo

    End If

   

End Sub

 

 

Private Sub cboRegion_AfterUpdate()

 

    ' if Region is updated then erase current values

    ' for City combo box.

    ' requery city combo box to restrict list

    ' to cities from selected region

    If Not Nz(Me!cboCity) = 0 Then

        Me!cboCity = Null

    End If

    Me!cboCity.Requery

   

End Sub

 

Private Sub cboRegion_NotInList(NewData As String, Response As Integer)

 

    Const MESSAGE_TEXT = "No country selected."

    Dim ctrl As Control

    Dim strSQL As String, strMessage As String

   

    Set ctrl = Me.ActiveControl

   

    ' abort insertion of new region if no country selected

    If IsNull(Me.cboCountry) Then

        MsgBox MESSAGE_TEXT, vbExclamation, "Invalid Operation"

        Response = acDataErrContinue

        ctrl.Undo

        Exit Sub

    End If

       

    ' subject to user confirmation insert a new row into Regions table,

    ' inserting NewData value into Region column and value of cboCountry

    ' control into CountryID column

    strMessage = "Add " & NewData & " to list?"

   

    strSQL = "INSERT INTO Regions(Region,CountryID) VALUES(""" & _

            NewData & """," & Me.cboCountry & ")"

  

    If MsgBox(strMessage, vbYesNo + vbQuestion) = vbYes Then

        CurrentDb.Execute strSQL, dbFailOnError

        Response = acDataErrAdded

    Else

        Response = acDataErrContinue

        ctrl.Undo

    End If

 

End Sub

 

 

Private Sub Form_Current()

 

    ' if new record set unbound combo boxes to Null,

    ' for existing record call functions to return

    ' region and country for current value of CityID

    If Me.NewRecord Then

        Me!cboCountry = Null

        Me!cboRegion = Null

    Else

        Me!cboRegion = GetRegion(Me!cboCity)

        Me!cboCountry = GetCountry(Me!cboRegion)

    End If

   

    ' requery unbound combo boxes to list

    ' only regions and cities for current city, if any

    Me!cboRegion.Requery

    Me!cboCity.Requery

   

End Sub

 

Private Function GetCountry(varRegionID)

 

    ' get CountryID for current value of Region combo box

    If Not IsNull(varRegionID) Then

        GetCountry = DLookup("CountryID", "Regions", "RegionID = " & varRegionID)

    End If

 

End Function

 

Private Function GetRegion(varCityID)

 

    ' get RegionID for current value of CityID field

    If Not IsNull(varCityID) Then

        GetRegion = DLookup("RegionID", "Cities", "CityID = " & varCityID)

    End If

   

End Function
