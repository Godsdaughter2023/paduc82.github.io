Sub CreateOrderSheets()
'************ Code Start **********
'Much of this code was originally written by Dev Ashish
'It is not to be altered or distributed,
'except as part of an application.
'You are free to use it in any application,
'provided the copyright notice is left unchanged.

'Code Courtesy of

    Dim objXL As Object
    Dim objActiveWkb As Object
    Dim ws As Object
    Dim boolXL As Boolean
    Dim db As DAO.Database
    Dim rs As DAO.Recordset
    Dim intRow As Integer
    Dim intF As Integer
    Dim strCompany As String
    Set db = CurrentDb
    Set rs = db.OpenRecordset("CompanyOrders")
    ' SQL View of query CompanyOrders ### note the ORDER BY Company ###
    'SELECT Customers.Company, Customers.[Last Name], Orders.[Order ID],
    'Products.[Product Name], [Order Details].[Unit Price], [Order Details].Quantity,
    '[Order Details].Discount, CCur([Quantity]*[Unit Price]*(1-[Discount])) AS [Extended Price]
    'FROM Products INNER JOIN (Customers INNER JOIN (Orders INNER JOIN [Order Details]
    'ON Orders.[Order ID] = [Order Details].[Order ID]) ON Customers.ID = Orders.[Customer ID])
    'ON Products.ID = [Order Details].[Product ID]
    'ORDER BY Customers.Company, Products.[Product Name];

    If fIsAppRunning("Excel") Then
        Set objXL = GetObject(, "Excel.Application")
        boolXL = False
    Else
        Set objXL = CreateObject("Excel.Application")
        boolXL = True
    End If

    objXL.Application.workbooks.Add
    Set objActiveWkb = objXL.Application.ActiveWorkBook
    With objActiveWkb
        Do Until rs.EOF
            If strCompany <> rs.Fields("Company") Then
                intRow = 1
                strCompany = rs.Fields("Company")
                Set ws = .Sheets.Add(After:=.Sheets(.Sheets.Count))
                ws.Name = strCompany
                For intF = 1 To rs.Fields.Count - 1
                    ws.cells(intRow, intF) = rs.Fields(intF).Name
                Next
                intRow = intRow + 1
            End If
            For intF = 1 To rs.Fields.Count - 1
                ws.cells(intRow, intF) = rs.Fields(intF).Value
            Next
            intRow = intRow + 1
            rs.MoveNext
        Loop

    End With
    rs.Close
    Set rs = Nothing
    Set db = Nothing
    objXL.Visible = True

    Set objActiveWkb = Nothing: Set objXL = Nothing
End Sub

Can't we use sumproduct with one column and one row? - Microsoft Community

rasoul khoshravan say:

Can't we use sumproduct with one column and one row?

I know how to use sumproduct between two rows or two columns.

I was wondering if it would be possible to use it with one row and one column with same size, as below image. It gives an answer (90), but not the expected one (32).

It  sums all values in the column(4+5+6=15) and multiplies it with sum of all values in the row: (1+2+3=6). 6*15=90. In other words it behaves like this: (1+2+3)*(4+5+6)=90

Changing the formula into sumproduct((A4:C4),(E4:E6)) gives #value error message.

tasos_k say:

Hi,

try the MMULT function:

by default

[one row] x [one column]

samples

=MMULT(A1:C1,A2:A4)

=MMULT(A1:E1,F1:F5)

A1:C1=1-2-3

A2:A4= 4-5-6

returns

1*4+2*5+3*6 = 4+10+18 = 32

xxxxxxxxxxxxxxx

NOTE

1) data in two rows

using TRANSOPE function

and MMULT as Array function

sample

A1:C1=1-2-3

A2:C2=4-5-6

Array Formula

=MMULT(A1:C1,TRANSPOSE(A2:C2))

result = 32

2) data in two columns

sample

A1:A3=1-2-3

B1:B3=4-5-6

Array formula

=MMULT(TRANSPOSE(A1:A3),B1:B3)

result = 32

lz. say:

Hi

SUMPRODUCT(A4:C4,E4:E6) actually generates the following array: {4;8;12,5;10;15,6;12;18}.

Values of row 4 are first multiplied by the 1st value of your column => 1*4;2*4;3*4, then by the 2nd (1*5;2*5;3*5)... so SUM({4;8;12,5;10;15,6;12;18}) => 90

To get the expected 32 try:

=SUMPRODUCT(A4:C4,TRANSPOSE(E4:E6)) and validate with Ctrl+Shift+Enter
