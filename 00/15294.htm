Option Explicit

Dim ColorsInitialized As Boolean

Private Sub Workbook_Open()
    ColorsInitialized = False
End Sub

Private Sub Workbook_SheetCalculate(ByVal Sh As Object)
    Dim ChartSheets() As String
    Dim chartRange As Range
    Dim colOffset As Long
    Dim i As Long
'
    If (Not ColorsInitialized) Then
        Call InitSeriesColors(0)
        ColorsInitialized = True
    End If

' List of worksheets to monitor for data range changes (assumes each has a "ProdChart" chart on it!)
    ChartSheets = Split("EX1,EX2,TF191,Sheet33", ",")
'
' See if a sheet with a production chart has changed, if so, update the chart's data range
'
    For i = LBound(ChartSheets) To UBound(ChartSheets)
        If (Sh.Name = ChartSheets(i)) Then  ' One of the target sheets changed, reset data range
            colOffset = 3 ' Start of first data range
            Call SetChartDataRange(Sh.Name, "ProdChart", colOffset)
            colOffset = 39 ' Start of second data range
            Call SetChartDataRange(Sh.Name, "MoyProdChart", colOffset)
        End If
    Next i
'
End Sub


'
' SetChartDataRange Macro
'
' wsName    = Name of worksheet with chart on it
' chartName = Name of chart on worksheet
' colOffset = Starting column of data for that chart
'
Private Sub SetChartDataRange(wsName As String, chartName As String, colOffset As Long)
    Dim i As Long
    Dim lastRow As Long, lastColumn As Long
    Dim dataRange As Range
    Dim lastCell As Variant

    On Error GoTo NoChart
   
' Find the extent of the data
    With ActiveWorkbook.Worksheets(wsName)
        lastCell = Last(3, .Range("D5:AH36").Offset(0, colOffset - 3))
        Set dataRange = ActiveWorkbook.Worksheets(wsName).Range( _
                    ActiveWorkbook.Worksheets(wsName).Cells(5, 3 + colOffset - 3), _
                    ActiveWorkbook.Worksheets(wsName).Cells(Range(lastCell).Row, Range(lastCell).Column))
        With .ChartObjects(chartName).Chart
            For i = 1 To .SeriesCollection.Count
                .SeriesCollection(1).Delete
            Next i
            For i = 1 To Range(lastCell).Row - 5
                .SeriesCollection.NewSeries
                If (i = 1) Then
                    .SeriesCollection(i).XValues = "='" & wsName & "'!" & dataRange.Offset(0, 1).Resize(1, dataRange.Columns.Count - 1).Address
                End If
                .SeriesCollection(i).Name = "='" & wsName & "'!" & dataRange.Offset(i, 0).Resize(1, 1).Address
                .SeriesCollection(i).Values = "='" & wsName & "'!" & dataRange.Offset(i, 1).Resize(1, dataRange.Columns.Count - 1).Address
                With .SeriesCollection(i).Format.Fill
                    .Visible = msoTrue
                    .ForeColor.RGB = SeriesColors(i)
                    .Transparency = 0
                    .Solid
                End With
            Next i
        End With
    End With
    Exit Sub
'
NoChart:
    MsgBox "Feuille " & wsName & " n'a pas un tableau nommé " & chartName, vbOKOnly, "Erreur"
    Exit Sub
End Sub

Private Function Last(choice As Long, rng As Range) As Variant
'Ron de Bruin, 5 May 2008
' (modified slightly, changing xlFormulas to xlValues in "LookIn")
' 1 = last row
' 2 = last column
' 3 = last cell
    Dim lrw As Long
    Dim lcol As Long

    Select Case choice

    Case 1:
        On Error Resume Next
        Last = rng.Find(what:="*", _
                        After:=rng.Cells(1), _
                        Lookat:=xlPart, _
                        LookIn:=xlValues, _
                        searchorder:=xlByRows, _
                        searchdirection:=xlPrevious, _
                        MatchCase:=False).Row
        On Error GoTo 0

    Case 2:
        On Error Resume Next
        Last = rng.Find(what:="*", _
                        After:=rng.Cells(1), _
                        Lookat:=xlPart, _

                        searchorder:=xlByColumns, _

End Function
