Private Declare PtrSafe Function CopyIcon Lib "user32" (ByVal hIcon As LongPtr) As Long
Private Declare PtrSafe Function LoadCursorFromFile Lib "user32" Alias "LoadCursorFromFileA" (ByVal lpFileName As String) As Long
Private Declare PtrSafe Function SetSystemCursor Lib "user32" (ByVal hcur As LongPtr, ByVal id As LongPtr) As Long
Private Declare PtrSafe Function GetCursor Lib "user32" () As Long

Private Const normalCursorID As Long = 32512

Private currentIco As Long
Private tmpCursor As Long
Private newCursor As Long

After starting the slide show I move the mouse by one pixel. That way no matter what the cursor icon was before it will change to a normal arrow (in most cases anyway). Only after that I will copy the icon and change the cursor.

If you use a hyperlink in your slideshow and the mouse cursor will end up on the hyperlink this will not work, as you will copy the pointer/hand icon instead of the normal arrow. You can try to move the mouse to one of the corners of the slide, where you are sure there are no hyperlinks.

This is the adapted code:

Private Declare PtrSafe Function CopyIcon Lib "user32" (ByVal hIcon As LongPtr) As Long
Private Declare PtrSafe Function LoadCursorFromFile Lib "user32" Alias "LoadCursorFromFileA" (ByVal lpFileName As String) As Long
Private Declare PtrSafe Function SetSystemCursor Lib "user32" (ByVal hcur As LongPtr, ByVal id As LongPtr) As Long
Private Declare PtrSafe Function GetCursor Lib "user32" () As Long

Private Declare PtrSafe Function SetCursorPos Lib "user32" (ByVal x As Long, ByVal y As Long) As Long
Private Declare PtrSafe Function GetCursorPos Lib "user32" (lpPoint As POINTAPI) As Long


Private Const normalCursorID As Long = 32512

Private currentIco As Long
Private tmpCursor As Long
Private newCursor As Long

Private Type POINTAPI
  x As Long
  y As Long
End Type

Public Sub changeCursor(cursorColor As String)
    Dim strPath As String
    Dim myCP As POINTAPI
    

    strPath = "C:\tmp\cursor.cur"
    
    'get the current cursor position
    GetCursorPos myCP
    'now move the cursor by one pixel
     SetCursorPos (myCP.x + 1), (myCP.y + 1)
    
    'only now, after moving the mouse we can copy the current mouse cursor to a temporary mouse cursor (for reset after slide show ended)
    currentIco = GetCursor()
    tmpCursor = CopyIcon(currentIco)
        
    newCursor = LoadCursorFromFile(strPath)
    Call SetSystemCursor(newCursor, normalCursorID)
End Sub

Public Sub ResetCursor()
   'reset to previously save cursor icon
    Call SetSystemCursor(tmpCursor, normalCursorID)
End Sub
