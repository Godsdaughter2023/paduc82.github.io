I'm not sure you can do this directly in Excel, but in Word:
 
    Sub RandomWord()
   
        Const MaxWordLen    As Long = 7
        Dim ndx             As Long
       
        Dim TestWord        As String
        Dim PossWords       As SpellingSuggestions
       
        Do
            TestWord = ""
            For ndx = 1 To Int(Rnd * MaxWordLen) + 1
                TestWord = TestWord & Chr(Int(Rnd * 26 + Asc("a")))
            Next
           
            Set PossWords = GetSpellingSuggestions(TestWord)
        Loop Until PossWords.Count > 0
       
        Debug.Print PossWords(Int(Rnd * PossWords.Count) + 1)
       
    End Sub
To do this in Excel you would need to instantiate a Word object, and open a document. Here's a variation that will give you 100 random correctly-spelled words in Excel
 
    Function RandomWords(HowMany As Long) As String()
   
        Dim WordApp         As Object
       
        Const MaxWordLen    As Long = 7
        Dim ndx             As Long
        Dim WordNum         As Long
       
        Dim Words()         As String
        ReDim Words(1 To HowMany)
        WordNum = LBound(Words)
       
        Dim TestWord        As String
        Dim PossWords       As Object 'SpellingSuggestions
       
        Set WordApp = CreateObject("Word.Application")
        WordApp.Documents.Add
       
        Do
            TestWord = ""
            For ndx = 1 To Int(Rnd * MaxWordLen) + 1
                TestWord = TestWord & Chr(Int(Rnd * 26 + Asc("a")))
            Next
            On Error Resume Next
            Set PossWords = WordApp.GetSpellingSuggestions(TestWord)
            If PossWords.Count > 0 Then
                Words(WordNum) = PossWords(Int(Rnd * PossWords.Count) + 1)
                WordNum = WordNum + 1
            End If
           
        Loop Until WordNum > UBound(Words)
       
        WordApp.Quit
        Set WordApp = Nothing
       
        RandomWords = Words
       
    End Function
   
    Sub Test()
        Dim Words() As String
        Words = RandomWords(100)
    End Sub

Private Sub Worksheet_Change(ByVal Target As Range)
If Target.Address = "$A$1" Then
Application.EnableEvents = False
On Error Resume Next

    xlCellTypeBlanks)
    If Not rngTemp Is Nothing Then
    rngTemp.Formula = "=IF(B" & rngTemp.Row & "=$A$1,""x"","""")"

sheeloo say:

Yes, if you want to retain the previous 'x' then you need VBA

Try the following;

Right-click on the sheet name at the bottom of the Excel window

Choose View Code

Paste the following code in the code window (VB Editor) which opens up

Private Sub Worksheet_Change(ByVal Target As Range)

If Not Intersect(Range("A1"), Target) Is Nothing Then

searchItem = ActiveSheet.Range("A1")

lrow = ActiveSheet.Range("B" & ActiveSheet.Rows.Count).End(xlUp).Row

For i = 2 To lrow

If ActiveSheet.Cells(i, 2) = searchItem Then

ActiveSheet.Cells(i, 1) = "x"

End If

Next

End If

End Sub

Close the code window

Now whenever you enter any value in A1 it will mark 'x' in Col A for the rows matching the value in A1 in Col B
