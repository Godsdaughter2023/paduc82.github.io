do this, and you should be able to set up as a pop3 setting and not delete your messages on the server, but only on the device:

1. open accounts

7. name your mailbox

9. user name (use your full email address)
10. password (use your password)
11. server address - enter "pop3.hotmail.com" or "pop3.live.com"
12. secure connection - select SSL
13. port - 995
14. Server type - smtp
15. user name (use your full email address)
16. password (use your password)
17. server address - enter "smtp.hotmail.com" or "smtp.live.com"
18. secure connection - select TLS
19. port - 25
20. Authentication - Plain password
21. Done

now, open your email and it should start to sync your email.

test on a couple of email to see if it is deleting only on your device now and leaving all the mesages on the server.

Function col_name_from_num(col As Variant) As String
' change a number into a letter style column name
  Dim cnfn_num As Integer
  Dim cnfn_num1 As Integer
  Dim cnfn_num2 As Integer
  col_name_from_num = ""
  cnfn_num = 0 + col
  alphalist = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
  If cnfn_num > 26 Then
     cnfn_num1 = cnfn_num / 26
     cnfn_num2 = cnfn_num - (cnfn_num1 * 26)
     If cnfn_num2 < 1 Then
       cnfn_num1 = cnfn_num1 - 1
       cnfn_num2 = cnfn_num2 + 26
       End If
     col_name_from_num = Mid(alphalist, cnfn_num1, 1) & _
      Mid(alphalist, cnfn_num2, 1)
  Else
     col_name_from_num = Mid(alphalist, cnfn_num, 1)
     End If
End Function

Function colors_t(index As Variant, row As Variant) As Integer

  Dim colix As Integer
  colors_t = 1

End Function
