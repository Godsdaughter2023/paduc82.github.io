graham mayor say:

Although directed primarily at Word, there are some macros at http://www.gmayor.com/Macrobutton.htm which demonstrate how to draw data from the Outlook addressbook using vba, though the available fields are limited and you cannot add custom fields to the list.

An alternative is to address Outlook directly through the Outlook object library. This time you can access the four User fields (which you could use for the additional field information). The essential code structure would be similar to.

Dim olApp As New Outlook.Application
Dim oNS As Outlook.NameSpace
Dim oContacts As Object
Dim oContactItem As Outlook.ContactItem
Dim bStarted As Boolean
Dim i As Long
On Error Resume Next
Set olApp = GetObject(, "Outlook.Application")
If Err <> 0 Then
    Set olApp = CreateObject("Outlook.Application")
    bStarted = True
End If
Set oNS = olApp.GetNamespace("MAPI")
For i = 1 To oNS.Folders.Count
    Set oContacts = oNS.Folders(i).Folders("Contacts")
    For Each oContactItem In oContacts.Items
        'search for a company name
        If oContactItem.CompanyName = "Name of Company" Then
            'if found enter the value from User4
            Selection.TypeText oContactItem.User4
             GoTo CleanUp
             Exit For
        End If
    Next
Next i
MsgBox "Not found"
CleanUp:
If bStarted Then olApp.Quit
Set oContactItem = Nothing
Set oContacts = Nothing
Set oNS = Nothing
Set olApp = Nothing

Sub Combine()
    Dim J As Integer

    On Error Resume Next
    Sheets(1).Select
    Worksheets.Add ' add a sheet in first place
    Sheets(1).Name = "Combined"

    ' copy headings
    Sheets(2).Activate
    Range("A1").EntireRow.Select
    Selection.Copy Destination:=Sheets(1).Range("A1")

    ' work through sheets
    For J = 2 To Sheets.Count ' from sheet 2 to last sheet
        Sheets(J).Activate ' make the sheet active
        Range("A1").Select
        Selection.CurrentRegion.Select ' select all cells in this sheets

        ' select all lines except title
        Selection.Offset(1, 0).Resize(Selection.Rows.Count - 1).Select

        ' copy cells selected in the new sheet on last line
        Selection.Copy Destination:=Sheets(1).Range("A65536").End(xlUp)(2)
    Next
End Sub

hansv mvp say:

Try this version. It avoids selecting sheets and ranges.

 

Sub Combine()
    Dim wsh1 As Worksheet
    Dim wsh2 As Worksheet
    Dim rng As Range

    On Error Resume Next
    Set wsh1 = Worksheets.Add(Before:=Worksheets(1))
    wsh1.Name = "Combined"

    ' copy headings
    Worksheets(2).Range("A1").EntireRow.Copy Destination:=wsh1.Range("A1")

    ' work through sheets
    For Each wsh2 In Worksheets
        Select Case wsh2.Name
            Case "D212", "D231", "D261", "D270", "D288", "Total DDF"
                Set rng = wsh2.Range("A1").CurrentRegion
                rng.Offset(1).Resize(rng.Rows.Count - 1).Copy _
                    Destination:=wsh1.Range("A" & wsh1.Rows.Count).End(xlUp).Offset(2)
        End Select
    Next wsh2
End Sub
