hansv mvp say:

Make a backup first!

Then test this macro on a folder with a few files only, to see if it does what you want, before running it on the "real" folder.

 

Sub ProcessFolder()
    ' Include trailing backslash in paths
    Const strSourcePath = "C:\SourceFiles\"
    Const strTargetPath = "C:\TargetFiles\"
    Dim strFile As String
    Dim wbk As Workbook
    Dim wsh As Worksheet
    Application.ScreenUpdating = False
    ' Get first file name
    strFile = Dir(strSourcePath & "*.xls")
    ' Loop
    Do While strFile <> ""
        ' Open workbook
        Set wbk = Workbooks.Open(Filename:=strSourcePath & strFile, AddToMRU:=False)
        ' Refer to a sheet
        Set wsh = wbk.Worksheets(1)
        ' *** Edit worksheet - this is just an example! Modify before running! ***
        wsh.Range("A1") = Date
        ' Save workbook
        wbk.SaveAs Filename:=strTargetPath & strFile, AddToMRU:=False
        ' Close
        wbk.Close SaveChanges:=False
        ' On to the next file
        strFile = Dir
    Loop
    Application.ScreenUpdating = True
End Sub
