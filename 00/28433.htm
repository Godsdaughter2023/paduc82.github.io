ken sheridan say:

If it is the age in years you want, then you can use this function:

 

Public Function GetAge(varDoB As Variant, Optional varAgeAt As Variant) As Variant

 

    If IsMissing(varAgeAt) Then varAgeAt = VBA.Date

   

    GetAge = DateDiff("yyyy", varDoB, varAgeAt) - _

        IIf(Format(varAgeAt, "mmdd") < Format(varDoB, "mmdd"), 1, 0)

 

End Function

 

and call it in the query like this:

    Count Number Of Rows Of All Files In Folder

    Count Number Of Rows Of All Files In Folder
    I am looking for a method of accessing all files within a folder determing the number of rows within these files that contain data.

    The file could ideally look at the current folder it's in and collect the requested data.

    Many thanks

     

    Nick

    andreas killer say:
    Sub Test()
      Dim WB As Workbook
      Dim Count As Long
      Dim FName As String
     
      'Speed up execution
      With Application
        .EnableEvents = False
        .ScreenUpdating = False
      End With
     
      'Find the first file
      FName = Dir(ThisWorkbook.Path & "\*.xls")
      'File found?
      Do While FName <> ""
        If FName <> ThisWorkbook.Name Then
          'Open it
          Set WB = Workbooks.Open(ThisWorkbook.Path & "\" & FName, False, True)
          'Count the rows
          Count = Count + WB.ActiveSheet.Range("A" & Rows.Count).End(xlUp).Row
          'Close it
          WB.Close
        End If
        'Find next file
        FName = Dir
      Loop
     
      With Application
        .EnableEvents = True
        .ScreenUpdating = True
      End With
     
      MsgBox Count
    End Sub
