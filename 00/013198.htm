Function StrToNum(rng As Range) As String
Dim x As Long
For x = 1 To Len(rng.Text)
    Select Case UCase(Mid(rng.Text, x, 1))
        Case "A" To "C"
            StrToNum = StrToNum & 2
        Case "D" To "F"
            StrToNum = StrToNum & 3
        Case "G" To "I"
            StrToNum = StrToNum & 4
         Case "J" To "L"
            StrToNum = StrToNum & 5
        Case "M" To "O"
            StrToNum = StrToNum & 6
        Case "P" To "S"
            StrToNum = StrToNum & 7
        Case "T" To "V"
            StrToNum = StrToNum & 8
        Case "W" To "Z"
            StrToNum = StrToNum & 9
        Case Else
       
        End Select
Next
End Function
