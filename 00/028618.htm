    1. Insert some text and a picture.

    2. Make sure the text is 'at the back'.

    3. Format the picture so that the wrapping style is either Tight or Through. Continue to keep the picture selected.

    4. On the Format menu select 'Wrap Text' and then 'Edit Wrap points'.

    hansv mvp say:

    You'd have a problem in December: you'd get an error since whatMonth would be 13, and MonthName(13) doesn't exist.

    Moreover, Dir isn't very suitable to loop through subfolders. Try this instead:

    Sub DSD_mail()
        Dim mgrDSD As Variant, filePath As String, subPath As String
        Dim d As Date, fso As Object, fld As Object, sfl As Object
        d = DateSerial(Year(Date), Month(Date) + 1, 1)
        filePath = "C:\Users\user1\DSD\" & Format(d, "yyyymm mmmm") & "\Proofs"
        'Debug.Print filePath
        Set fso = CreateObject("Scripting.FileSystemObject")
        Set fld = fso.GetFolder(filePath)
        For Each sfl In fld.SubFolders
            mgrDSD = sfl.Name
            subPath = sfl.Path & "\"
            'Debug.Print mgrDSD, subPath
            Call Open_mail(mgrDSD, subPath, Month(d))
        Next sfl
    End Sub
