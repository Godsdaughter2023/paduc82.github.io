Sub Test()
    Dim MasterRange As Range
    Set MasterRange = Range("A1:B3")
    
    Set TargetCell = Range("D1")
    TargetCell.Select
    
    iRow = 1
    For iRow = 1 To MasterRange.Rows.Count
         cName = MasterRange.Cells(iRow, 1)
         cValues = MasterRange.Cells(iRow, 2)
         
         Dim Array1() As String
         Array1 = Split(cValues, ",")
         For nLine = 0 To UBound(Array1)
             ActiveCell.Value = cName
             ActiveCell.Offset(0, 1).Select
             ActiveCell.Value = Array1(nLine)
             ActiveCell.Offset(1, -1).Select
    
         Next
    Next
End Sub

Sub Test2()

                ActiveCell.Value = lcase(iCell.Address(False, False))
                ActiveCell.Offset(1, -1).Select
            Next
         Next
    Next
End Sub

Function fugit(mystart, myend)
  mymin = Left(myend, Len(myend) - 3) - Left(mystart, Len(mystart) - 3)
  mysec = Right(myend, 2) - Right(mystart, 2)
  If mysec < 0 Then
    mymin = mymin - 1
    mysec = mysec + 60
  End If
  fugit = TimeValue("0:" & mymin & ":" & mysec)
End Function

ken sheridan say: Base the subform on a query which references the check box as a parameter in its WHERE clause like this:

WHERE (Forms!PaymentSummary!excludeClositem = TRUE AND DRStatus = "Open")
OR Forms!PaymentSummary!excludeClositem = FALSE

In the AfterUpdate event procedure of the check box requery the subform control with:

Me.F_Q_OrderStatussubform.Requery

Note that F_Q_OrderStatussubform is the name of the subform control, i.e. the control in the parent form which houses the subform.  This might or might not also be the name of the source form object.
