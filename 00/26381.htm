Sub GetGoogleMaps()
Dim r As Long
Dim strURL As String
r = 2 ' change in your code if the address is in some other row.

strURL = strURL & Replace(Range("A" & r), " ", "+")

strURL = strURL & "+" & Replace(Range("C" & r), " ", "+")
