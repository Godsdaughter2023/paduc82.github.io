Option Explicit
Function ConcatenateMolecules(ByVal Molecule As String, ParamArray Molecules()) As String
  Dim i As Integer
  'Create space for the analyzer
  Dim Numbers, Elements
  Numbers = Array()
  Elements = Array()
  'Analyze each molecule and combine the elements
  AnalyzeMolecule Molecule, Elements, Numbers
  For i = 0 To UBound(Molecules)
    AnalyzeMolecule Molecules(i), Elements, Numbers
  Next
  'Create space for the output
  ReDim Result(0 To UBound(Elements))
  For i = 0 To UBound(Elements)
    If Numbers(i) = 1 Then
      Result(i) = Elements(i)
    Else
      Result(i) = Elements(i) & CStr(Numbers(i))
    End If
  Next
  ConcatenateMolecules = Join(Result, "")
End Function
Sub AnalyzeMolecule(ByVal S As String, ByRef Elements, ByRef Numbers)
  Dim Digit As String, Element As String, Number As String
  Dim i As Integer, j As Integer
  Dim Last As Boolean
  'Parse the string
  For i = 1 To Len(S)
    Digit = Mid$(S, i, 1)
    'Are we at the start of an element?
    Select Case Digit
      Case "A" To "Z"
        'Do we have a previous Element?
GrabLast:
        If Len(Element) > 0 Then
          For j = LBound(Elements) To UBound(Elements)
            If StrComp(Element, Elements(j), vbTextCompare) = 0 Then
            End If
          Next
          'Found?
          If j > UBound(Elements) Then
            'No, add a slot
            j = UBound(Elements) + 1
            ReDim Preserve Elements(0 To j)
            ReDim Preserve Numbers(0 To j)
            'Store the element
            Elements(j) = Element
          End If
          'Add the amount of atoms
          If Number = "" Then
            Numbers(j) = Numbers(j) + 1
          Else
            Numbers(j) = Numbers(j) + CInt(Number)
          End If
        End If
        'Done?
        If i > Len(S) Then Exit Sub
        'Prepare for the next round
        Element = Digit
        Number = ""
      Case "a" To "z"
        Element = Element & Digit
      Case "0" To "9"
        Number = Number & Digit
    End Select
  Next
  'Grab the last one
  GoTo GrabLast
End Sub

Function ChemConcat(a, b) As String
    Dim aArr, bArr
    Dim x, y, C, H, O, N
    
    'Get value of C
    x = Mid(a, InStr(a, "C") + 1, InStr(a, "H") - 1 - InStr(a, "C"))
    If Not IsNumeric(x) Then x = 1
    y = Mid(b, InStr(b, "C") + 1, InStr(b, "H") - 1 - InStr(b, "C"))
    If Not IsNumeric(y) Then y = 1
    C = CLng(x) + CLng(y)
    
    'Get value of H
    x = Mid(a, InStr(a, "H") + 1, InStr(a, "O") - 1 - InStr(a, "H"))
    If Not IsNumeric(x) Then x = 1
    y = Mid(b, InStr(b, "H") + 1, InStr(b, "O") - 1 - InStr(b, "H"))
    If Not IsNumeric(y) Then y = 1
    H = CLng(x) + CLng(y)
    
    'Get value of O
    x = Mid(a, InStr(a, "O") + 1, InStr(a, "N") - 1 - InStr(a, "O"))
    If Not IsNumeric(x) Then x = 1
    y = Mid(b, InStr(b, "O") + 1, InStr(b, "N") - 1 - InStr(b, "O"))
    If Not IsNumeric(y) Then y = 1
    O = CLng(x) + CLng(y)
    
    'Get value of N
    x = Mid(a, InStr(a, "N") + 1, Len(a))
    If Not IsNumeric(x) Then x = 1
    y = Mid(b, InStr(b, "N") + 1, Len(b))
    If Not IsNumeric(y) Then y = 1
    N = CLng(x) + CLng(y)
    
    'Give the result
    ChemConcat = "C" & C & "H" & H & "O" & O & "N" & N
End Function

At the end of the install process, if your Mac is connected to the internet, the AuotUpdate app should launch and install the updates. From the error log it appears that perhaps the updates were not installed. Office will not run without both Mac OS and Office updates having been installed. 

Make sure your Mac has a working internet connection before checking for updates. Install Mac OS X updates followed by Office updates.

 

Mac OS X updates:

To get Mac OS updates, go to the App store on your dock, and then click the Updates button to reveal available updates.

 

Office Updates

To get the update:

1. Open any Office application:

3. Make sure your Mac has a working internet connection

4. Go to the Help menu and choose Check for Updates

5. Let AutoUpdate check for and install updates

6. Restart your Mac
