Sub Splitter()
    Dim Rng As Range, Cell As Range
    Dim Lr As Long, i As Long
    Dim SArr, TArr, Arr
    Dim Ws As Worksheet
    Set Ws = ActiveSheet
    Lr = Ws.Range("A" & Rows.Count).End(xlUp).Row
    ReDim SArr(1 To Lr - 1, 1 To 1)
    ReDim TArr(1 To Lr - 1, 1 To 1)
    SArr = Ws.Range("A2:A" & Lr)
    For i = 1 To Lr - 1
        Arr = Split(SArr(i, 1), "\")
        TArr(i, 1) = Arr(0) & "\" & Arr(1)
    Next i
    Ws.Range("A2:A" & Lr) = TArr
End Sub

andreas killer say:
No problem, try the code below.

Andreas. 

Sub Test()
  Dim FirstAddress As String
  Dim Where As Range, R As Range
  Dim i As Long, j As Long
  
  'Search in the whole sheet, customize if necessary
  Set Where = Cells
  
  Set R = Where.Find("*\*")
  If R Is Nothing Then Exit Sub
  Application.ScreenUpdating = False
  FirstAddress = R.Address
  Do
    With R
      'We must have 2 backslashes
      i = InStr(.Value, "\")
      j = InStrRev(.Value, "\")
      If i <> j Then .Value = Left$(.Value, j - 1)
    End With
    Set R = Where.FindNext(R)
    If R Is Nothing Then Exit Do
  Loop Until R.Address = FirstAddress
End Sub
