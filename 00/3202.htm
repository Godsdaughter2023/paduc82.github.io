You can try the code below. Unfortunately I always forget to ask something and did not ask which columns you want to sum on each sheet so if you let me know then I will edit the code for you. (This won't stop you from testing the code.)

 

You said you are new to code so here are the guidelines to install.

 

If you have not already done so then within Options, enable macros. See Help for how to do this. (Disable all with notification will be OK.)

 

Select the Developer ribbon (If not already displayed then see help for how to display.)

You will see a default Sub and End sub.  For the moment place a single quote in front of these 2 lines to make them comments. (They should turn green)

 

Copy the all of the code below and paste under the 2 commented lines. (There is actually 2 routines; a Sub and a UDF (User Defined Function)

 

Check the name of the default sub that was created and if it matches the name of the first sub you just pasted then you can delete the old default Sub and End sub. If it does not match then edit the sub name that you pasted in to match the name that was created by default.

 

Close the VBA editor (Cross with red background top right)

Private Sub CommandButton1_Click()
    Dim lngLastCol As Long
    Dim wsDestin As Worksheet
    Dim rngCel
    Dim lngTotalRow As Long
   
    Application.ScreenUpdating = False
   
    lngLastCol = LastRowOrCol(Me, False)    'False parameter to find last used column
   
    With Me     'Me is the generic name for a worksheet when code is in the Worksheet's module
        .AutoFilterMode = False     'Turn off AutoFilter (if on)
       
        'Copy the first column to a column 10 columns to right of last used column
        .Range(.Cells(1, "A"), .Cells(.Rows.Count, "A").End(xlUp)).Copy _
                Destination:=.Cells(1, lngLastCol + 10)
       
        'Remove duplicates from the pasted new range
        .Columns(lngLastCol + 10).RemoveDuplicates Columns:=1, Header:=xlYes
       
        'Apply AutoFilter to the Source data
        .Range(.Cells(1, "A"), .Cells(1, lngLastCol)).AutoFilter
       
        'Loop through the unique list of names created from column A data
        For Each rngCel In .Range(.Cells(2, lngLastCol + 10), _
                .Cells(.Rows.Count, lngLastCol + 10).End(xlUp))
            'Test if a worksheet matching name exists
            Set wsDestin = Nothing  'Clear the worksheet variable
           
            On Error Resume Next    'Code errors if name does not exist
            Set wsDestin = Worksheets(rngCel.Value)
            On Error GoTo 0
           
            If wsDestin Is Nothing Then     'If nothing then sheet does not already exist
                Set wsDestin = Worksheets.Add(After:=Worksheets(Worksheets.Count))
               
                'Next lines re-names the added worksheet. _
                'If name contains invalid characters then user gets MsgBox saying not re-named.
                On Error Resume Next
                wsDestin.Name = rngCel.Value
                If Err.Number > 0 Then
                    MsgBox "Worksheet name " & rngCel.Value & _
                            " possibly contains invalid characters." & vbCrLf _
                            & "This worksheet not re-named."
                End If
                On Error GoTo 0
               
                'Return to Source data sheet. (Adding sheet makes new sheet the ActiveSheet)
                .Activate
            Else
                wsDestin.Cells.Clear    'If worksheet exist then clear existing data
            End If
           
            'Set the Filter of first column to match the name
            .AutoFilter.Range.AutoFilter Field:=1, Criteria1:=rngCel.Value
           
            'Copy the header and visible data cells and paste to the destination worksheet
            .AutoFilter.Range.Cells.SpecialCells(xlCellTypeVisible).Copy _
                        Destination:=wsDestin.Cells(1, 1)
                   
           
            With wsDestin
                '****************************************************************
                'Find last row of data plus 1 for Totals row
                lngTotalRow = LastRowOrCol(wsDestin, True) + 1
               
                'Insert Sum of column "O"
                .Cells(lngTotalRow, "O") = WorksheetFunction.Sum(.Range(.Cells(2, "O"), _
                    .Cells(lngTotalRow - 1, "O")))
               
                'Insert "Total:" in column to left and format cells
                .Cells(lngTotalRow, "N").Value = "Total:"
                .Cells(lngTotalRow, "N").HorizontalAlignment = xlRight
                .Range(.Cells(lngTotalRow, "N"), .Cells(lngTotalRow, "O")).Font.Bold = True
                '*****************************************************************
               
                'Set Bold text for the column headers and AutoFit the columns.
                .Range(.Cells(1, 1), .Cells(1, .Columns.Count).End(xlToLeft)).Font.Bold = True
                .Columns.AutoFit
            End With
           
        Next rngCel
        .AutoFilterMode = False
        .Columns(lngLastCol + 10).Clear
           
    End With
   
    Application.ScreenUpdating = True

End Sub


Function LastRowOrCol(ws As Worksheet, bolRowCol As Boolean, Optional rng As Range) As Long
    'Finds the last used row or column in a worksheet
    'First parameter is Worksheet
    'Second parameter is True for Last Row or False for last Column
   
    Dim lngRowCol As Long
    Dim rngToFind As Range
   
    If rng Is Nothing Then
        Set rng = ws.Cells
    End If
   
    If bolRowCol Then
        lngRowCol = xlByRows
    Else
        lngRowCol = xlByColumns
    End If
   
    With ws
        Set rngToFind = rng.Find(What:="*", _
                LookIn:=xlFormulas, _
                LookAt:=xlPart, _
                SearchOrder:=lngRowCol, _
                SearchDirection:=xlPrevious, _
                MatchCase:=False)
    End With
   
    If Not rngToFind Is Nothing Then
        If bolRowCol Then
            LastRowOrCol = rngToFind.Row
        Else
            LastRowOrCol = rngToFind.Column
        End If
    End If
   
End Function
