Sub InteriorColorMatch()
Dim r As Range, r1 As Range
For Each r In [K8:K37]
For Each r1 In [A3:G7]
If Day(r.Value) = r1.Value Then
r1.Interior.ColorIndex = r.Offset(, 1).Interior.ColorIndex
GoTo mynext
End If
Next
mynext:
Next
End Sub
