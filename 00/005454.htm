daniel g. grau say:
Take a look at this macro. It does work and appears to do what you want. Works only with footnotes.

macropod say:

The following macro processes footnotes AND endnotes, turning them into ordinary text but retaining their internal formatting (eg bold/italic/underline) and referencing style (eg superscripted roman numerals). Footnotes are turned into a second series of (unlinked) endnotes. Note comments in the code regarding optionally enclosing the endnote and footnote references with { } and [ ], respectively.

Sub UnLinkNotes()
Application.ScreenUpdating = False
Dim Fld As Field, nRng As Range, eNote As Endnote, fNote As Footnote, nRef As String
With ActiveDocument
  'Convert any endnote/footnote cross-reference fields to text
  For Each Fld In .Fields
    If Fld.Type = wdFieldNoteRef Then Fld.Unlink
  Next
  For Each eNote In .Endnotes
    With eNote
      With .Reference.Characters.First
        'The next two lines insert a brace after the endnote. Two lines
        'further down insert a brace before the endnote.
        'This is useful if you want to be able to re-create the endnotes later on.
        .InsertAfter "}"
        .Characters.Last.Font.Superscript = True
        .Collapse wdCollapseStart
       .InsertCrossReference wdRefTypeEndnote, wdEndnoteNumberFormatted, eNote.Index
        nRef = .Characters.First.Fields(1).Result
        .Characters.First.Fields(1).Unlink
        'The next two lines insert a brace before the endnote. Two lines
        'further up insert a brace after the endnote.
        'This is useful if you want to be able to re-create the endnotes later on.
        .InsertBefore "}"
        .Characters.First.Font.Superscript = True
      End With
      .Range.Cut
    End With
    'To position converted Endnotes at the end of the document, use:
    Set nRng = .Range
    'To position converted Endnotes at the end of the Section, use:
    'Set nRng = eNote.Range.Sections.First.Range
    'To let the code decide, based on whatever's in the document, use:
    'If .Range.EndnoteOptions.Location = wdEndOfSection Then
    '  Set nRng = eNote.Range.Sections.First.Range
    'Else
    '  Set nRng = .Range
    'End If
    With nRng
      .Collapse wdCollapseEnd
      .End = .End - 1
      If .Characters.Last <> Chr(12) Then .InsertAfter vbCr
      .InsertAfter nRef & " "
      With .Paragraphs.Last.Range
        .Style = "Endnote Text"
        .Words.First.Style = "Endnote Reference"
      End With
      .Collapse wdCollapseEnd
      .Paste
      If .Characters.Last.Next = Chr(12) Then .InsertAfter vbCr
    End With
  Next
  For Each eNote In .Endnotes
    eNote.Delete
  Next
  For Each fNote In .Footnotes
    With fNote
      With .Reference.Characters.First
        'The next two lines insert a square bracket after the footnote. Two lines
        'further down insert a square bracket before the footnote.
        'This is useful if you want to be able to re-create the footnotes later on.
        .InsertAfter "]"
        .Characters.Last.Font.Superscript = True
        .Collapse wdCollapseStart
        .InsertCrossReference wdRefTypeFootnote, wdFootnoteNumberFormatted, fNote.Index
        nRef = .Characters.First.Fields(1).Result
        .Characters.First.Fields(1).Unlink
        'The next two lines insert a square bracket before the footnote. Two lines
        'further up insert a square bracket after the footnote.
        'This is useful if you want to be able to re-create the footnotes later on.
        .InsertBefore "["
        .Characters.First.Font.Superscript = True
      End With
      .Range.Cut
    End With
    Set nRng = .Range
    With nRng
      .Collapse wdCollapseEnd
      .End = .End - 1
      If .Characters.Last <> Chr(12) Then .InsertAfter vbCr
      .InsertAfter nRef & " "
      With .Paragraphs.Last.Range
        .Style = "Footnote Text"
        .Words.First.Style = "Footnote Reference"
      End With
      .Collapse wdCollapseEnd
      .Paste
      If .Characters.Last = Chr(12) Then .InsertAfter vbCr
    End With
  Next
  For Each fNote In .Footnotes
    fNote.Delete
  Next
End With
Set nRng = Nothing
Application.ScreenUpdating = True
End Sub
