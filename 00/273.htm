bernie deitrick say:

You could select the cells manually, then run this:

Sub SpellCheckSpecificCells()
    Dim C As Range
    For Each C In Selection
        If Not Application.CheckSpelling(C.Value) Then
            MsgBox "There is a word in " & C.Address(False, False) & " that is spelled incorrectly."
        End If
    Next C
End Sub

or you can specify the range (this example checks 7 cells)

Sub SpellCheckSpecificCells()
    Dim C As Range
    For Each C In Range("A9:A11,A24,B5:B7")
        If Not Application.CheckSpelling(C.Value) Then
            MsgBox "There is a word in " & C.Address(False, False) & " that is spelled incorrectly."
        End If
    Next C
End Sub
