Function DecimalToDMS(ByVal Value As Double, Optional ByVal Places) As String
  'Convert a Decimal into a DMS: 12.58225 => 12° 34' 56.1"
  Dim i As Long, s As Integer
  Dim Result(0 To 2) As String
  s = Sgn(Value)
  Value = Abs(Value)
  i = Int(Value)
  If i <> 0 Then DecimalToDMS = i * s & "° "
  Value = (Value - i) * 60
  i = Int(Value)
  If i <> 0 Then DecimalToDMS = DecimalToDMS & i & "' "
  Value = (Value - i) * 60
  Value = WorksheetFunction.Round(Value, IIf(IsMissing(Places), 0, Places))
  If Value <> 0 Then DecimalToDMS = DecimalToDMS & Value & """"
  If DecimalToDMS = "" Then DecimalToDMS = "0°"
End Function
Function DMSToDecimal(ByVal DMS As String) As Double
  'Convert as DMS into a Decimal: 12° 34' 56.1" => 12.58225
  Dim i As Long, j As Long, s As Integer, Value
  DMS = Replace(DMS, " ", "")
  If Left$(DMS, 1) = "-" Then
    s = -1
    DMS = Mid$(DMS, 2)
  Else
    s = 1
  End If
  For i = 1 To Len(DMS)
    Select Case Mid$(DMS, i, 1)
      Case "°"
        Value = CDbl(Mid(DMS, j + 1, i - j - 1))
        DMSToDecimal = DMSToDecimal + Value * s
        j = i
      Case "'"
        Value = CDbl(Mid(DMS, j + 1, i - j - 1)) / 60
        DMSToDecimal = DMSToDecimal + Value * s
        j = i
      Case """"
        Value = CDbl(Mid(DMS, j + 1, i - j - 1)) / 3600
        DMSToDecimal = DMSToDecimal + Value * s
        j = i
      Case "0" To "9", "+", Application.International(xlDecimalSeparator)
        'Okay
      Case Else
        DMSToDecimal = CVErr(xlErrNum)
        Exit Function
    End Select
  Next
End Function

In this case we assume that cell A1 is filled with a year (2011) either manually or thru Data Validation.  Cell B1 is filled via Data Validation from a list from F1 thru either F52 or F53.

 

Insert the following macro in the worksheet code area:

 

Private Sub Worksheet_Change(ByVal Target As Range)
Set t = Target
If Not Intersect(t, Range("A1")) Is Nothing Then
    Dim I As Long, th As String
    thu = "Thursday"
    I = Range("A1").Value
    d1 = Format(DateSerial(I, 1, 1), "dddd")
    d2 = Format(DateSerial(I, 12, 31), "dddd")
    If d1 = thu Or d2 = thu Then
        n = 53
    Else
        n = 52
    End If
        With Range("B1").Validation
            .Delete
            .Add Type:=xlValidateList, AlertStyle:=xlValidAlertStop, Operator:= _
            xlBetween, Formula1:="=$F$1:$F$" & n
            .IgnoreBlank = True
            .InCellDropdown = True
            .InputTitle = ""
            .ErrorTitle = ""
            .InputMessage = ""
            .ErrorMessage = ""
            .ShowInput = True
            .ShowError = True
        End With
End If
End Sub

 

The macro will run when the value in A1 is entered.  The macro determines if the year is a 52-week-year or a 53-week-year.  It then sets the validation list for cell B1 to either F1:F52 or F1:F53 respectively.

As with traditional PBX systems and with Enterprise Voice, calls between users in your organization are handled internally within Cloud PBX, never going to the PSTN. This applies to calls between users in your organization located in different geographical areas and calls to federated partners, removing long-distance costs on these internal calls. To enable calls between your users and people outside of your organization, Cloud PBX can be connected to the PSTN in one of two ways:

1. By using Skype for Business software on-premises to connect with your existing telephony infrastructure and configure on-premises PSTN connectivity, allowing for the customer’s existing numbers and PSTN service to be used by Skype for Business Cloud PBX capability.

2. By using Cloud PBX with PSTN calling and assigning phone numbers to those Office 365 users using the Skype for Business admin center. You can get phone numbers from Office 365 or transfer phone numbers over to Office 365 from another service provider or carrier.
