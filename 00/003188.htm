Function IncDateSession(ByVal StartDate As Date, ByVal Sessions As Integer, _
    Optional ByVal HoliDays As Range = Nothing) As Date
  Dim Dates, I As Long, J As Long
  If HoliDays Is Nothing Then
    'No holidays
    Dates = Array()
  Else
    'Read in the values
    Dates = Intersect(HoliDays, HoliDays.Parent.UsedRange)
  End If
  Do While Sessions > 1
    'Increment date depending on weekday
    Select Case Weekday(StartDate)
      Case vbMonday To vbThursday
        StartDate = StartDate + 2
      Case vbFriday To vbSaturday
        StartDate = StartDate + 3
      Case vbSunday
        'Return Startdate
        Exit Do
    End Select
    'Test for holidays
    For I = 1 To UBound(Dates)
      For J = 1 To UBound(Dates, 2)
        If Dates(I, J) = StartDate Then
          Sessions = Sessions + 1
          Exit For
        End If
      Next
    Next
    Sessions = Sessions - 1
  Loop
  IncDateSession = StartDate
End Function

My function just supports the MWF and TTS rotation, that's correct. I made this so you can see how a simple rotation goes. I thought that this would be easier for you to understand and expand, because I saw that you are familiar with VBA.

If I've understand your problem correctly, you would do a MWF and TTS rotation if the StartDate is on a MTWTF and you would have a SS rotation if the StartDate is on a SS?

Function IncDateSession(ByVal StartDate As Date, _
    Optional ByVal HoliDays As Range = Nothing) As Date
  Dim Dates, I As Long, J As Long
  Dim Sessions As Integer, StartWeekDay As Integer
  If HoliDays Is Nothing Then
    'No holidays
    Dates = Array()
  Else
    'Read in the values
    Dates = Intersect(HoliDays, HoliDays.Parent.UsedRange)
  End If
  '
  'Get weekday to determine the rotation typ and number of sessions
  StartWeekDay = Weekday(StartDate)
  Select Case StartWeekDay
    Case vbMonday To vbFriday
      'MWF or TTS rotation
      Sessions = 12
    Case Else
      'SS rotation
      Sessions = 8
  End Select
  '
  'Iterate the end date
  Do While Sessions > 1
    'Increment date depending on weekday and rotation typ
    Select Case StartWeekDay
      Case vbMonday To vbFriday
        'MWF or TTS rotation
        Select Case Weekday(StartDate)
          Case vbMonday To vbThursday
            StartDate = StartDate + 2
          Case vbFriday To vbSaturday
            StartDate = StartDate + 3
        End Select
      Case Else
        'SS rotation
        Select Case Weekday(StartDate)
          Case vbSaturday
            StartDate = StartDate + 1
          Case vbSunday
            StartDate = StartDate + 6
        End Select
    End Select
    'Test for holidays
    For I = 1 To UBound(Dates)
      For J = 1 To UBound(Dates, 2)
        If Dates(I, J) = StartDate Then
          Sessions = Sessions + 1
          Exit For
        End If
      Next
    Next
    Sessions = Sessions - 1
  Loop
  IncDateSession = StartDate
End Function
