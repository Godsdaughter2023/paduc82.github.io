Hi Doug,

Try something along the lines of:

Sub ResetForm_Click()
Dim ws As Worksheet
With Worksheets("Start")
  .Range("C3:C7").ClearContents
  .Range("D9").ClearContents
  .Range("D11").ClearContents
  .Range("D13").ClearContents
End With
With Worksheets("Product")
  .Range("G6:G7").ClearContents
  .Range("G13:G14").ClearContents
  .Range("G22:G24").ClearContents
End With
For Each ws In ThisWorkbook.Worksheets
  If ws.Name <> "Welcome" Then _
    If ws.Name <> "Start" Then _
    If ws.Name <> "Product" Then ws.Visible = xlSheetHidden
Next
End Sub
