hansv mvp say:

Create a list of keywords in column K, starting in K1, and add the corresponding categories in the cells next to it in column L.

Code:

Sub Search()
    Const KeyCol = "K"
    Const Col = "D"
    Dim r As Long
    Dim m As Long
    Dim n As Long
    Dim rng As Range
    Dim cel As Range
    Dim s As String
    Dim WordToFind As String
    Dim WordToWrite As String
   
    ' Range to search
    n = Range(Col & Rows.Count).End(xlUp).Row
    Set rng = Range(Col & "1:" & Col & n)
   
    Application.ScreenUpdating = False
    ' Loop through the keywords
    m = Range(KeyCol & Rows.Count).End(xlUp).Row
    For r = 1 To m
        WordToFind = Range(KeyCol & r).Value
        WordToWrite = Range(KeyCol & r).Offset(0, 1).Value
        ' Find all occurrences of WordToFind
        With rng
            Set cel = .Find(What:=WordToFind, LookAt:=xlPart, MatchCase:=False)
            If Not cel Is Nothing Then
                s = cel.Address
                Do
                    If cel.Offset(0, 1).Value = "" Then
                        cel.Offset(0, 1).Value = WordToWrite
                    End If
                    Set cel = .FindNext(After:=cel)
                Loop Until cel.Address = s
            End If
        End With
    Next r
    Application.ScreenUpdating = True
End Sub
