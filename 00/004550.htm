tom ogilvy say:

this worked for me.

Sub abc()
Dim i As Long, cell As Range
i = 1
For Each cell In Range("A2:A21")

  If Application.CountIf(Range("F:F"), cell.Value) = 0 Then
     Cells(i, "F").Value = cell.Value
     Cells(i, "G").Value = Evaluate("=MAX(IF(A:A=""" & cell.Value & """,B:B))")
     i = i + 1
  End If
Next
End Sub

Results in columns F and G

Adjust A2:A10 to reflect your range.

See the snip below. this code loops down column A and outputs the name and max values to columns C & D. See the line in bold, this is a list of your names which you can add to as required but keep the syntax the same.

Sub somesub()
Dim Lastrow As Long, MyRange As Range
Dim c As Range, y As Long
Dim MyMax, S As String, v
Dim x As Long
y = 1
Lastrow = Cells(Rows.Count, "A").End(xlUp).Row
S = "Bob,Ken,Bill,Mike"
v = Split(S, ",")
Set MyRange = Range("A1:A" & Lastrow)
For x = LBound(v) To UBound(v)
MyMax = 0
For Each c In MyRange
    If c.Value = v(x) Then
        If c.Offset(, 1) > MyMax Then MyMax = c.Offset(, 1)
    End If
Next
Cells(y, 3) = v(x)
Cells(y, 4) = MyMax
y = y + 1
Next
End Sub
