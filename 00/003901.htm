vijay a. verma say:
Use the below macro 

Sub StringToTime()
    Dim Rng As Range
    Dim Arr
    Dim r As Long, i As Long
    Dim Ws As Worksheet
    Set Ws = ActiveSheet
    Set Rng = Ws.Range("A2:A" & Ws.Range("A" & Rows.Count).End(xlUp).Row)
    r = Rng.Rows.Count
    ReDim Arr(1 To r, 1 To 1)
    Arr = Rng
    Ws.Columns("A").NumberFormat = "hh:mm am/pm"
    For i = 1 To r
        Arr(i, 1) = Format(Mid(Arr(i, 1), 9, 4), "00\:00")
    Next i
    Rng = Arr
End Sub

andreas killer say:
Please never use SELECT, SELECTION, ACTIVECELL, it is slow and error prone. Always refer to the objects directly.

Try the code below.

Andreas. 

Sub Test()
  Dim Where As Range, This As Range
  Dim All As Range
  Dim HypCount As Long, JamaCount As Long
  'Refer to the used cells in column A
  Set Where = Range("A1", Range("A" & Rows.Count).End(xlUp))
  For Each This In Where
    If This.Hyperlinks.Count > 0 Then
      'Count the hyperlink
      HypCount = HypCount + 1
      If InStr(1, This.Hyperlinks(1).Address, "jama", vbTextCompare) > 0 Then
        JamaCount = JamaCount + 1
      End If
    Else
      'Remember the row
      If All Is Nothing Then
        Set All = This
      Else
        Set All = Union(All, This)
      End If
    End If
  Next
  'Show the result
  MsgBox _
    HypCount & " hyperlinks found" & vbCrLf & _
    JamaCount & " contains jama hyperlinks", vbInformation
  'Found some rows to delete?
  If Not All Is Nothing Then
    If MsgBox("Delete other rows?", vbOKCancel Or vbQuestion) = vbCancel Then Exit Sub
    'Delete the rows
    All.EntireRow.Delete
  End If
End Sub
