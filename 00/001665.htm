ossiemac say:

Reply edited since initial post. I just remembered that must work backwards in a loop to delete rows so best to use 2 loops and delete the rows after the loop to copy.

Sub CopyColumnP()
    Dim wsSource As Worksheet
    Dim wsDestin As Worksheet
    Dim lngDestinRow As Long
    Dim rngSource As Range
    Dim rngCel As Range
    Dim lngRows As Long
    Dim r As Long
   
    Set wsSource = Sheets("KING")     'Edit "Sheet1" to your source sheet name
    Set wsDestin = Sheets("THEY")
   
    With wsSource
        'Following line assumes column headers in Source worksheet so starts at row2
        Set rngSource = .Range(.Cells(2, "P"), .Cells(.Rows.Count, "P").End(xlUp))
    End With
   
    For Each rngCel In rngSource
        If rngCel.Value = "Bought" Then
            With wsDestin
                'Following line assumes column headers in Destination worksheet
                lngDestinRow = .Cells(.Rows.Count, "P").End(xlUp).Offset(1, 0).Row
                rngCel.EntireRow.Copy Destination:=wsDestin.Cells(lngDestinRow, "A")
            End With
        End If
    Next rngCel
   
    lngRows = rngSource.Rows.Count
   
    With rngSource     'Note that rngSource is like a mini worksheet with only one column and rows start at 1.
        For r = lngRows To 1 Step -1
            If .Cells(r, 1) = "Bought" Then
                .Cells(r, 1).EntireRow.Delete
            End If
        Next r
    End With
       
 End Sub
