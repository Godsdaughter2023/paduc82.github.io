Private Sub CommandButton1_Click()
On Error Resume Next
Dim sC48 As String
Dim sZ50 As String
Dim sZ51 As String
  sC48 = Range("C48").Formula
  sZ50 = Range("Z50").Formula
  sZ51 = Range("Z51").Formula
  Rows("48:51").Insert Shift:=xlDown
  Range("C48:C51").Merge
  Range("D48:M51").Merge
  Range("N48:Q51").Merge
  Range("S48:U51").Merge
  Range("V48:X51").Merge
  Range("Z48:AA49").Merge
  Range("Z50:AA50").Merge
  Range("Z51:AA51").Merge
  Range("C48").Formula = sC48

  Range("Z50").NumberFormat = "[$-14809]d mmm yyyy;@"
