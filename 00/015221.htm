Try this formula:

=INDEX($A$2:$D$98,(MATCH($H$4 & $G$2 & $H$3,$A$2:$A$98 & $B$2:$B$98 & $D$2:$D$98,0)),3)

in the formula display area.  If you don't get it right, you'll see a #VALUE! error in the cell with the formula.

Alternatively, try this function:

 

Function WorkingHours(ByVal StartDate As Date, ByVal EndDate As Date) As Double
    Dim w As Long
    Dim d As Date
    w = Int(EndDate - StartDate) \ 7
    WorkingHours = 48 * w
    StartDate = StartDate + 7 * w
    For d = Int(StartDate) To Int(EndDate) - 1
        Select Case Weekday(d)
            Case 1 To 4, 7
                WorkingHours = WorkingHours + 8.5
            Case 5
                WorkingHours = WorkingHours + 5.5
        End Select
    Next d
    WorkingHours = WorkingHours + 24 * (EndDate - Int(EndDate) - StartDate + Int(StartDate))
    WorkingHours = Round(WorkingHours, 6)
End Function

graham mayor say:
I am not convinced we have enough information to perform this calculation as we don't know what the start and end times are of the working day, so the first and last day calculations are something of a guess.

Working on the premise that the start and end times are midnight, you could use something like the following (I have used constants rather than values fro a userform but the principle is the same) If the start and end times of the working day are are something different e.g. 08:30 and 15:00 the calculations for the first and last days would have to be changed to reflect that.

The calculations also assume Saturday and Sunday as working days - and we are not even going to consider public holidays.

Sub Macro1()
Const StartDateTime As Date = "03-07-2012 8:30:00"
Const EndTimeDate As Date = "16-07-2012 15:00:00"
Dim iHours As Single
Dim iFirstDayHours As Single
Dim iLastDayHours As Single
Dim iWholeDays As Long
Dim i As Long
Dim iDate As Date
iHours = 0
iFirstDayHours = DateDiff("h", StartDateTime, Format(StartDateTime, "dd-MM-yyyy") & " 23:59:59")
If iFirstDayHours > 5.5 Then
    GetHours StartDateTime, iHours
End If
iLastDayHours = DateDiff("h", Format(EndTimeDate, _
                                     "dd-MM-yyyy") & " 00:00:01", _
                                     EndTimeDate)
If iLastDayHours > 5.5 Then
    GetHours StartDateTime, iHours
End If
iWholeDays = DateDiff("d", Format(StartDateTime, _
                                  "dd/MM/yyyy"), _
                                  Format(EndTimeDate, "dd/MM/yyyy"))
For i = 1 To iWholeDays - 1
    iDate = DateAdd("d", i, Format(StartDateTime, _
                                   "dd/MM/yyyy"))
    GetHours iDate, iHours
Next i
MsgBox iHours
End Sub

Private Function GetHours(iCalc As Date, iHours As Single)
    Select Case WeekDay(iCalc)
        Case Is = 1, 2, 3, 4, 7
            iHours = iHours + 8.5
        Case Is = 5
            iHours = iHours + 5.5
        Case Else
    End Select
GetHours = iHours
End Function
