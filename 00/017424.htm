Sub CreateCalendarMonth()
  Dim TopLeftCell As Range
  Dim RowsperDay As Long, ColumnsPerDay As Long
  Dim R As Long, C As Long
  Dim WeeksHorizontal As Boolean, SuppressDaysNotInThisMonth As Boolean
  Dim StartDay As Date, ThisDay As Date

  'The start day can be any date, the calendar is created for this month
  StartDay = Date
  'Number of rows per day, must be >= 1
  RowsperDay = 5
  'Number of columns per day, must be >= 1
  ColumnsPerDay = 1
  'If True days of Weeks are created horizontal, otherwise vertical
  WeeksHorizontal = False
  'If True days that are not within this month are not written
  SuppressDaysNotInThisMonth = True
  'The top left cell of the first day
  Set TopLeftCell = Range("B4")

  'There is no need to change anything below this line!

  'Get the date of the first day in the first week of the month
  ThisDay = FirstDayOfWeek(DateSerial(Year(StartDay), Month(StartDay), 1))

  If Not WeeksHorizontal Then Exchange RowsperDay, ColumnsPerDay
  For R = 0 To 6
    For C = 0 To 6
      If SuppressDaysNotInThisMonth Then
        If Month(ThisDay) <> Month(StartDay) Then GoTo SkipDay
      End If
      TopLeftCell.Offset( _
        IIf(WeeksHorizontal, R * RowsperDay, C * ColumnsPerDay), _
        IIf(WeeksHorizontal, C * ColumnsPerDay, R * RowsperDay)) = ThisDay
SkipDay:
      ThisDay = ThisDay + 1
    Next
    If Month(ThisDay) > Month(StartDay) Then Exit For
  Next
End Sub

Private Function FirstDayOfWeek(ByVal DateTime As Date) As Date
  'Returns the 1. day of the week of DateTime
  FirstDayOfWeek = DateTime - Weekday(DateTime, vbUseSystemDayOfWeek) + 1
End Function

Private Sub Exchange(ByRef A, ByRef B)
  Dim C
  C = A
  A = B
  B = C
End Sub
