No, a solution would be that a workbook could be added with all default styles in it. But I have a work-around:

Sub ResetDefaultStyles()
   Dim lngCount As Long, lngBuiltInStyles As Long, lngStyles As Long
   Dim lngCalc As Long
   Dim awb As Workbook
   Dim wb As Workbook
   Dim strMsg As String
   Dim sh As Worksheet
     
   On Error GoTo Error_ResetDefaultStyles
   
   With Application
      .EnableCancelKey = xlErrorHandler
      .ScreenUpdating = False
      lngCalc = .Calculation
      .Calculation = xlCalculationManual
      .DisplayAlerts = False
   End With
   
   Set awb = ActiveWorkbook
   lngStyles = awb.Styles.Count
      
     
   ''' delete all custom styles
   With ActiveWorkbook
      For lngCount = .Styles.Count To 1 Step -1
         gdblHandled = gdblHandled + 1
         If Not .Styles(lngCount).BuiltIn Then
            .Styles(lngCount).Delete
         Else
            lngBuiltInStyles = lngBuiltInStyles + 1
         End If
NextStyle:
       
      Next lngCount
   End With
   
   ''' import builtin styles
   Set awb = ActiveWorkbook
   Set wb = Workbooks.Add
   awb.Styles.Merge Workbook:=wb
   wb.Close: Set wb = Nothing
   
   ''' check for the Hyperlink & Followed Hyperlink styles
   DefaultHyperlink
   
   
Exit_ResetDefaultStyles:
   If lngStyles = lngBuiltInStyles Then
      strMsg = "No custom styles in " & ActiveWorkbook.Name
   ElseIf ActiveWorkbook.Styles.Count = lngBuiltInStyles Then
      strMsg = "All " & lngStyles - lngBuiltInStyles & " custom styles deleted."
   Else
      strMsg = lngStyles - ActiveWorkbook.Styles.Count & " custom styles deleted." & vbCr & ActiveWorkbook.Styles.Count - lngBuiltInStyles & " custom styles could not be deleted."
   End If
   MsgBox prompt:=strMsg, Buttons:=vbOKOnly, Title:=JPE_APP_TITLE
Exit_ResetDefaultStyles2:
   With Application
      .ScreenUpdating = True
      .Calculation = lngCalc
      .DisplayAlerts = True
   End With
 
Exit Sub
   
   
Error_ResetDefaultStyles:
   Select Case Err.Number
      Case 18
         MsgBox "User escape request.", Title:=ThisWorkbook.VBProject.Description
         Resume Exit_ResetDefaultStyles
      Case 91
         RefreshRibbon
         Resume
      Case 1004
         If TestProtection Then
            MsgBox prompt:="This option does not work on protected sheets or protected workbooks.", Title:=JPE_APP_TITLE
            Resume Exit_ResetDefaultStyles2
         Else
            Resume NextStyle
         End If
      Case Else
         MsgBox "Error " & Err.Number & " (" & Err.Description & ") in procedure ResetDefaultStyles of Module modWorkbook", Title:=ThisWorkbook.VBProject.Name
         Resume Exit_ResetDefaultStyles2
   End Select
End Sub



Sub DefaultHyperlink()
   Dim strStyle As String
   
   On Error Resume Next
   strStyle = ActiveWorkbook.Styles("Hyperlink").Name
   If Len(strStyle) > 0 Then
      With ActiveWorkbook.Styles("Hyperlink").Interior
          .Pattern = xlNone
          .TintAndShade = 0
          .PatternTintAndShade = 0
      End With
      With ActiveWorkbook.Styles("Hyperlink")
        .IncludeNumber = False
        .IncludeFont = True
        .IncludeAlignment = False
        .IncludeBorder = False
        .IncludePatterns = False
        .IncludeProtection = False
      End With
   End If
   strStyle = ActiveWorkbook.Styles("Followed Hyperlink").Name
   If Len(strStyle) > 0 Then
      With ActiveWorkbook.Styles("Followed Hyperlink").Interior
          .Pattern = xlNone
          .TintAndShade = 0
          .PatternTintAndShade = 0
      End With
      With ActiveWorkbook.Styles("Followed Hyperlink")
        .IncludeNumber = False
        .IncludeFont = True
        .IncludeAlignment = False
        .IncludeBorder = False
        .IncludePatterns = False
        .IncludeProtection = False
      End With
   End If
   
End Sub
