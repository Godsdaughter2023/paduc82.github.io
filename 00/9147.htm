mike middleton say:

Two suggestions:

(A) TIME B3:B86 is your X variable, and VAL C3:C86 is your Y variable. For a quadratic fit, i.e., polynomial of order 2, select three horizontal cells, and your array-entered function should be

={LINEST(C3:C86,B3:B86^{1,2},TRUE,TRUE)}

I think your worksheet has the X and Y references reversed, and the LINEST function argument Y^{1,2,3} instead of Y^{1,2} would yield a cubic fit instead of a quadratic fit.

(B) Your TIME data is actually date-time data, and each serial value used by LINEST is 42752 plus a decimal fraction. To avoid numerical calculation problems, subtract 42752 from each date-time to obtain the time-only portion of the serial number to be used as the TIME data. Then plot the new VAL vs. TIME data, add a polynomial trendline of order 2, and compare the result with LINEST using the revised TIME data.

When I make the changes above, the LINEST result matches the trendline result.

bernie deitrick say:

I have assumed that your values start in cell A2, and that you have B, C, and D empty. I have further assumed that the numbers like 11-121212 can have, at most, 3 digits before the - 

Enter this formula in cell B2

=TRIM(MID(SUBSTITUTE(TRIM(LEFT(A2,FIND(" ",A2,FIND("-",A2))))," ",REPT(" ",FIND("-",A2))),(FIND("-",A2)-LEN(SUBSTITUTE(LEFT(A2,FIND("-",A2))," ",""))+1)*FIND("-",A2)-3,FIND("-",A2)))

and this in C2

=TRIM(LEFT(A2,FIND(B2,A2)-1))

and this in D2

=TRIM(SUBSTITUTE(SUBSTITUTE(A2,B2,""),C2,""))


Then copy down to match your list.

Function custom_parse(ByRef str As String, ByVal res_i As Integer, ByVal res_num As Integer)
    ' This sub function will take a single restriction string
    ' ex: AAA NA BBB, BBB W/ CCC, AAA MH (BBB OR CCC) etc.
    ' And divide into 3 variables. One group of W/'s counts as one,
    ' Logical operator counts as one, and (...) counts as one as well.
    ' This is so the W/'s and the () logics can be processed as a group

    ' If by the end of this parsing, there aren't THREE variables (checked when split)
    ' Then something went wrong (didn't get grouped properly, too many ops/variables, etc.)
    

    Dim b1, b2, w1, w2 As Integer
    Dim ors, wis As String
    Dim tarr() As String
    Dim str_original As String
    
    str_original = str

    b1 = InStr(1, str, "(")
    b2 = InStr(1, str, ")")
    If (b1 <> 0 And b2 <> 0) Then
        ors = Mid(str, b1 + 1, b2 - b1 - 1)
        str = Replace(str, "(" & ors & ")", "ANDOR")
    End If

     
    w1 = InStr(1, str, "W/") 'Find first W/
    w2 = InStrRev(str, "W/") 'find last W/
    If (w1 <> 0 And w2 <> 0) Then
        wis = Mid(str, w1 - 4, w2 + 5 - (w1 - 4) + 1)
        str = Replace(str, wis, "WITHS")
    End If
     
     
    ' Split variables into 3. If the var was neither W/'s nor (),
    ' then it'll be parsed as a single sales code as it normally would
    tarr() = Split(str, " ")
    
    '-------------------------------
    ' ERROR: if there aren't 3 variables. Indicates that W/ or () didn't get group properly, or too many variables
    ' "#" will indicate DO NOT MAP/CALCUALTE
    If UBound(tarr) <> 2 Then
        Sheets(res_sheet).Cells(r_row + res_i, r_col + (sheet_i * 3)).value = "#" & str_original
        Sheets(raw_sheet).Cells(r_row + res_i, r_col + (sheet_i * 3)).value = "#" & str_original
        'Sheets(res_sheet).Cells(r_row + res_i, r_col + (sheet_i * 3)).value = "!" & str_original
        'Sheets(raw_sheet).Cells(r_row + res_i, r_col + (sheet_i * 3)).value = "!" & str_original
        'Sheets(raw_sheet).Cells(r_row + res_i + res_bad_const + res_num, r_col + (sheet_i * 3)).value = "!"
        GoTo endCustomParse
    End If
    ' -------------------------------
    
    ' Flagging so code can determine type of group
    For i = 0 To 2
        If tarr(i) = "ANDOR" Then
            tarr(i) = "@" & ors         ' () group
        ElseIf tarr(i) = "WITHS" Then
            tarr(i) = "/" & wis         ' W/ group
        Else: tarr(i) = "+" & tarr(i)   ' Regular var/op
        End If
    Next i
    
    ' Fill in parsed restrictions (nicely split it up & type-flagged)
    Sheets(res_sheet).Range(Cells(r_row + res_i, r_col + (sheet_i * 3)).Address(False, False) & ":" _
        & Cells(r_row + res_i, r_col + (sheet_i * 3) + 2).Address(False, False)).value _
        = tarr()

endCustomParse:
End Function
