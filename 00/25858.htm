Sub Combine()
Dim oTbl As Word.Table
Dim oRow As Word.Row
Dim oRng As Word.Range
Dim oRngDup As Word.Range
Set oTbl = Selection.Tables(1)
For Each oRow In oTbl.Rows

  oRng.MoveStart wdCell, 1
