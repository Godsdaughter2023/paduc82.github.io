Option Explicit
Public Sub CreateOutlookApptz()
   Sheets("CalendarItems").Select
    On Error GoTo Err_Execute
     
    Dim olApp As Outlook.Application
    Dim olMeet As Outlook.AppointmentItem
    Dim blnCreated As Boolean
    Dim olNs As Outlook.Namespace
    Dim CalFolder As Outlook.MAPIFolder
    Dim subFolder As Outlook.MAPIFolder
    Dim arrCal As String
     
    Dim i As Long
     
    On Error Resume Next
    Set olApp = Outlook.Application
     
    If olApp Is Nothing Then
        Set olApp = Outlook.Application
         blnCreated = True
        Err.Clear
    Else
        blnCreated = False
    End If
     
    On Error GoTo 0
     
    Set olNs = olApp.GetNamespace("MAPI")
    Set CalFolder = olNs.GetDefaultFolder(olFolderCalendar)
         
    i = 2
    Do Until Trim(Cells(i, 1).Value) = ""
    arrCal = Cells(i, 1).Value
    Set subFolder = CalFolder.Folders(arrCal)
     
    Set olMeet = subFolder.Items.Add(olAppointmentItem)
          
    'MsgBox subFolder, vbOKCancel, "Folder Name"
 
    With olMeet
     
    'Define calendar item properties
        .Start = Cells(i, 9) + Cells(i, 10)     '+ TimeValue("9:00:00")
        .End = Cells(i, 11) + Cells(i, 12)       '+TimeValue("10:00:00")
        .Subject = Cells(i, 2)
        .Location = Cells(i, 3)
        .Body = Cells(i, 4)
        .BusyStatus = olBusy
        .ReminderMinutesBeforeStart = Cells(i, 16)
        .ReminderSet = True
        .Categories = Cells(i, 5)
        
        .MeetingStatus = olMeeting
        
        .RequiredAttendees = Cells(i, 14)
    
        
       

        .Save
        .Send
      
        
        

     
    End With
                 
        i = i + 1
        Loop
    Set olMeet = Nothing
    Set olApp = Nothing
     
    Exit Sub
     
Err_Execute:
    MsgBox "An error occurred - Exporting items to Calendar."
     
End Sub
