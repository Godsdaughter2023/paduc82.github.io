' Paste this into ThisOutlookSession and enable macros,

' manually replace your@email below, then save and restart Outlook

' your@email might need to be your account name if it is not the default (your email)

' make sure to keep the structure in tact. You mileage may vary. Do as you like. Don't blame me.

Private WithEvents objinspectors As Outlook.Inspectors
Private WithEvents objexplorer As Outlook.Explorer
Private WithEvents NewItem As Outlook.MailItem

Private Sub Application_Startup()
    Set objinspectors = Application.Inspectors
    Set objexplorer = Application.ActiveExplorer
End Sub

Private Sub objexplorer_InlineResponse(ByVal Item As Object)
    If TypeName(Item) = "MailItem" Then
        Item.SendUsingAccount = Application.GetNamespace("MAPI").Accounts("your@email")
        Item.SentOnBehalfOfName = ""
        Item.Save
        Set NewItem = Item 

        'Tested this again with refactored code and it seems it can be set once
    End If
End Sub

Private Sub objinspectors_NewInspector(ByVal Inspector As Inspector)
  If TypeName(Inspector.CurrentItem) = "MailItem" Then
        If Inspector.CurrentItem.SenderEmailAddress = "" Then
            Inspector.CurrentItem.SendUsingAccount = _
                Application.GetNamespace("MAPI").Accounts("your@email")
        End If
        Item.SentOnBehalfOfName = ""
        Item.Save

        Set NewItem = Inspector.CurrentItem
        'Tested this again with refactored code and it seems it can be set once
  End If
End Sub

You have to change one of the declarations from

Private WithEvents NewItem As Outlook.MailItem

to

Private NewItem As Redemption.SafeMailItem

You have to select the Redemption reference on the Tools menu. I also had to create the My updated routine which uses Redemption and corrects the bug follows:

Private Sub objinspectors_NewInspector(ByVal inspec As Inspector)
    If TypeName(inspec.CurrentItem) = "MailItem" Then
        Dim theItem As MailItem

            Dim redItem As Redemption.SafeMailItem

            redItem.Item = theItem
            Set redItem.SendUsingAccount = _
                Application.GetNamespace("MAPI").Accounts("your@email")

            theItem.Save

            'Tested this again with refactored code and it seems it can be set once
        End If

End Sub

Private Sub objexplorer_InlineResponse(ByVal Item As Object)
    If TypeName(Item) = "MailItem" Then

        Item.Save
        Set NewItem = Nothing
        Set NewItem = redItem
        'Tested this again with refactored?? code and it seems it can be set once
    End If
End Sub

But this definitely seems to do the job!

The only issue is that when you define the your@email account, Outlook wants a PST file to be used by the account - it can't/won't use the OST file that my Outlook.com account uses. I just gave it a new PST to write to. But that means that the sent email appears in the Sent folder of the PST file, not the Sent folder of the OST.

To fix that, I added the following routine:

Private Sub Application_ItemSend(ByVal Item As Object, Cancel As Boolean)

    If Not (Item Is Nothing) Then
        If TypeName(Item) = "MailItem" Then
            Dim mItem As MailItem
            Set mItem = Item
            Dim snd As String
            snd = LCase$(mItem.SenderEmailAddress)
            If snd = "your@email" Then
                    Dim recip As Recipient
                    Set recip = mItem.Recipients.Add(snd)
                    recip.Type = olBCC
                    recip.Resolve
            End If
        End If
    End If

End Sub
