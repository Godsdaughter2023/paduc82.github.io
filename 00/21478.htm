    bKelRank --
SELECT [QUALITY HOLD TABLE Chart by Orig&Shift Query].SHIFT, [QUALITY HOLD TABLE Chart by Orig&Shift Query].[Issue Originator], [QUALITY HOLD TABLE Chart by Orig&Shift Query].[CountOfHOLD#], Sum(IIf(XX.[CountOfHOLD#]>=[QUALITY HOLD TABLE Chart by Orig&Shift Query].[CountOfHOLD#],1,0)) AS Rank
FROM [QUALITY HOLD TABLE Chart by Orig&Shift Query], [QUALITY HOLD TABLE Chart by Orig&Shift Query] AS XX
WHERE (((XX.[SHIFT])=[QUALITY HOLD TABLE Chart by Orig&Shift Query].[SHIFT]) AND ((([XX].[CountOfHOLD#]))>=[QUALITY HOLD TABLE Chart by Orig&Shift Query].[CountOfHOLD#]))
GROUP BY [QUALITY HOLD TABLE Chart by Orig&Shift Query].SHIFT, [QUALITY HOLD TABLE Chart by Orig&Shift Query].[Issue Originator], [QUALITY HOLD TABLE Chart by Orig&Shift Query].[CountOfHOLD#]
ORDER BY [QUALITY HOLD TABLE Chart by Orig&Shift Query].SHIFT, [QUALITY HOLD TABLE Chart by Orig&Shift Query].[CountOfHOLD#] DESC;
