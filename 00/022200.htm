andreas killer say:

Jordan,

Insert the code below into a regular module.
Switch to the worksheet to be printed.
Press Alt-F8 and execute sub Main.

Andreas.

Option Explicit

Sub Main()
  Dim This As SlicerItem, Item As SlicerItem
  Dim Ws As Worksheet
  Dim Path As String
 
  'Refer to the worksheet to print out
  Set Ws = ThisWorkbook.ActiveSheet
 
  'Get the path to the users desktop
  Path = CreateObject("Wscript.Shell").SpecialFolders("Desktop" & vbNullChar) & "\Dashboard\"
  'Create if not exists
  If Dir(Path, vbDirectory) = "" Then MkDir Path
 
  'Screen off
  Application.ScreenUpdating = False
 
  'Access the slicer cache
  With ThisWorkbook.SlicerCaches("Slicer_Name")
    'For each item
    For Each This In .SlicerItems
      'Clear all filters
      .ClearManualFilter
      'Set the filter for this item
      For Each Item In .SlicerItems
        Item.Selected = Item.Value = This.Value
      Next
      'Print as PDF
      DoEvents
      Ws.ExportAsFixedFormat xlTypePDF, Path & ValidFileName(This.Value, "_") & ".pdf"
    Next
    .ClearManualFilter
  End With
End Sub

Private Function ValidFileName(ByVal FName As String, _
    Optional ByVal ReplaceChar As String = "") As String
  'Return a filename without invalid chars
  'Avoid CON, PRN, AUX, NUL, COM1 to COM9, LPT1 to LPT9 as filename
  Const InvalidChars = "\/:*?""<>|"
  Dim i As Integer, p As Long
  Dim Digit As String
  For i = 1 To Len(InvalidChars)
    Digit = Mid$(InvalidChars, i, 1)
    p = InStr(FName, Digit)
    Do While p > 0
      Mid$(FName, p, 1) = vbNullChar
      p = InStr(FName, Digit)
    Loop
  Next
  For i = 1 To 31
    Digit = Chr$(i)
    p = InStr(FName, Digit)
    Do While p > 0
      Mid$(FName, p, 1) = vbNullChar
      p = InStr(FName, Digit)
    Loop
  Next
  ValidFileName = Replace(FName, vbNullChar, ReplaceChar)
End Function

gary's student say:

You can use the =TODAY() function to make a real-time daily counter:

In column A enter the project name

In column B enter the project start date

In column C enter the project completion date once the project is complete

In D2 enter:

=IF(B2="","",IF(C2="",TODAY()-B2,C2-B2)) and copy down.
