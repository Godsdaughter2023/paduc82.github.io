hansv mvp say:

Try this macro:


Sub CreateFolder()
    ' Parent folder with trailing backslash
    Const strParent = "I:\Projects\"
    Dim strFolder As String
    strFolder = InputBox("Enter the name of the folder to create under " & strParent)
    If strFolder = "" Then
        MsgBox "You didn't provide a valid name!", vbCritical
    ElseIf Dir(strParent & strFolder, vbDirectory) <> "" Then
        MsgBox "The folder " & strParent & strFolder & " already exists!", vbInformation
    Else
        MkDir strParent & strFolder
        MsgBox "The folder " & strParent & strFolder & " has been created.", vbInformation
    End If
End Sub
