tintombstone say:

Not really following the borloop thing

But this will cycle through all rows on Sheet1 of the currentworkbook, starting at row 4, which is where your data seems to start

Create a new workbook based on a template (You put in the full path and name of the template)

Copy one row of data into the new workbook then save back into the folder you specify

Sub ExportRows()

Dim newBook As Workbook
Dim currBook As Workbook
Dim currSheet As Worksheet
Dim rCnt As Integer, cCnt As Integer
Dim dogNum As Integer, savePath As String, TemplatePath As String

    savePath = "C:\Sample\" ' Set as needed
    TemplatePath = "C:\Sample\sample.xltx" ' Set as needed
   
    Set currBook = ActiveWorkbook
    Set currSheet = currBook.Sheets("Sheet1")
    Application.ScreenUpdating = False

    For rCnt = 3 To currSheet.Range("A1").CurrentRegion.Rows.Count
        dogNum = dogNum + 1
        Application.StatusBar = "Creating: " & savePath & "dog_c" & dogNum & ".xlsx"
        Set newBook = Workbooks.Add(TemplatePath)
        With newBook.Sheets("Sheet1")
            For cCnt = 1 To 7
                .Cells(2, cCnt).Value = currSheet.Cells(rCnt, cCnt).Value
            Next cCnt
        End With
        newBook.SaveAs savePath & "dog_c" & dogNum & ".xlsx"
        newBook.Close
    Next
   
    Application.StatusBar = False

End Sub

    In your table design, set the field Id (Primary Key) to:

    Data type: AutoNumber

    New values: Random

    Now, when you retrieve the data and wish to display the custom autonumber, use a query where you apply the custom format using a fancy expression:

    SELECT Chr(65+Abs(Val(Left(Right([Id],8),2)))/4) & Format(Right([Id],6),"000000") AS CustomId, *
    FROM YourTable;
