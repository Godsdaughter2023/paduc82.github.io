Try the code below, which I found (and modified slightly) here:

http://www.mrexcel.com/forum/excel-questions/312792-visual-basic-applications-controlling-acrobat-distiller.html

 

'===== BEGIN CODE =====

Option Explicit

Public MyPdf_Machine As PdfDistiller
'
Sub test_it()     ' This is just for testing the function below...
    Dim i As Integer
    i = Print_PDF_Sheets("testPDF.pdf")
End Sub

Function Print_PDF_Sheets(sPDF_File_Name As String) As Integer
'must add reference to Acrobat Distiller
'In Printer Setting "Acrobat PDF" Printing Preferences, you must uncheck Rely on system fonts or it throws an error
'This function returns 1 ot 0 based on success or failure
'This function prints the Print Area of the activesheet to the Specified PDF File Name
'Note: The PDF will appear in the same directory as the calling workbook

Dim sSet_Adobe_Printer As String
Dim sTemp_PS_FileName As String
Dim Puter As String, Got_Case As Boolean
Dim iTryPort As Integer
Dim sOld_Printer As String
Dim sD_Path As String

If UCase(Right(sPDF_File_Name, 4)) = ".XLS" Then sPDF_File_Name = Left(sPDF_File_Name, Len(sPDF_File_Name) - 4) & ".PDF"
If UCase(Right(sPDF_File_Name, 4)) <> ".PDF" Then MsgBox "Bad File Name": Stop

If sD_Path = "" Then sD_Path = ThisWorkbook.Path & "\"

sOld_Printer = Application.ActivePrinter

On Error GoTo Active_Printer_Trap:

Puter = Environ("computername"): Got_Case = False
'we know what the port number is on some computers, trying for it the easy way.
Select Case UCase(Puter)
Case "wd001": sSet_Adobe_Printer = "Adobe PDF on Ne05:": Got_Case = True
Case "wd002": sSet_Adobe_Printer = "Adobe PDF on Ne06:": Got_Case = True
Case "wd00031": sSet_Adobe_Printer = "Adobe PDF on Ne05:": Got_Case = True
Case "wl001": sSet_Adobe_Printer = "Adobe PDF on Ne06:": Got_Case = True
Case "WDKR03", "WLKR03": sSet_Adobe_Printer = "Adobe PDF on Ne05:": Got_Case = True
End Select

Set_Printer:
If Application.ActivePrinter <> sSet_Adobe_Printer Then Application.ActivePrinter = sSet_Adobe_Printer
On Error GoTo 0

'Ready to make a PDF File
Application.DisplayAlerts = False

If MyPdf_Machine Is Nothing Then Set MyPdf_Machine = New PdfDistiller

'If you put the tmp.ps file in a different directory, the Directory MUST exist or you get no file
sTemp_PS_FileName = sD_Path & "tmp.ps"
ActiveSheet.PrintOut Copies:=1, Preview:=False, Collate:=True, printtofile:=True, prtofilename:=sTemp_PS_FileName

'FileToPDF will return 1 on success
If (InStr(1, sPDF_File_Name, "\") = 0) Then sPDF_File_Name = sD_Path & sPDF_File_Name
Print_PDF_Sheets = MyPdf_Machine.FileToPDF(sTemp_PS_FileName, sPDF_File_Name, "")

'if 1 then success and delete the log file and the temp PS File
If Print_PDF_Sheets = 1 Then
Kill (Left(sPDF_File_Name, Len(sPDF_File_Name) - 3) & "log")
Kill (sTemp_PS_FileName)
End If

Application.DisplayAlerts = True

'Note: Changing the Active Printer back and forth makes things very slow ... on my machines
If sOld_Printer <> Application.ActivePrinter Then Application.ActivePrinter = sOld_Printer

Exit Function

'******* Error Traps ************
Active_Printer_Trap:
'Trying to find the printer port.
If Err.Number = 1004 Then
Err.Clear
If iTryPort <= 0 Then iTryPort = 1 Else iTryPort = iTryPort + 1
sSet_Adobe_Printer = "Adobe PDF on Ne0" & iTryPort & ":"
If iTryPort >= 10 Then
MsgBox "Can't PDF it"
Exit Function
End If
Resume Set_Printer:
End If

'Sometimes resetting the distiller will keep things running smoothly
If Err.Number = 462 Then
Set MyPdf_Machine = Nothing
If MyPdf_Machine Is Nothing Then Set MyPdf_Machine = New PdfDistiller
Err.Clear
Stop 'because this can end up in a loop I don't want to deal with yet.
Resume
End If

Stop 'oops, unhandled error
Resume

End Function
'===== END CODE =====

The most likely explanation is that the same Style is used throughout the document & that Style is defined to Automatically Update. There could be more involved, but that's the simplest scenario. Private Sub Worksheet_Change(ByVal Target As Range)

An interesting aspect with merged cells is that if there is a value in the Merged Cells then the Target address in the Change Event code is simply one cell (first cell of merged range) but if the value is deleted and the merged cells are blank then then Target address becomes the entire merged range. I edited the code to test for an intersect with the merged range so it does not matter if the target is one cell or entire merged range.
