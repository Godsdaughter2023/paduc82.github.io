jay freedman say:

> I see that this macro does not cater for sub folders - possible?

Yes, it's possible, although it adds quite a bit of complexity to the code.

By the way, one caveat: Running the macro will change the Date Modified of every document in all the folders and subfolders that it touches to the current date and time, even if there were no changes in the document's attached template. If the folders/subfolders contain any documents for which you know the date must not change, I'd advise moving them temporarily to a different area of the disk before running the macro.

Here's the update...

Sub MultiFile_StyleUpdate()
    Dim strFile As String
    Dim strPath As String
    Dim ListOfFolders As Variant
    Dim curPath As Variant
    Dim curDoc As Document
    
    ' call function to display dialog for user to choose top-level folder
    strPath = SelectRootFolder
    If strPath = "" Then Exit Sub
    
    ' call function to get an array of all subfolders of root folder
    ' their subfolders, etc.
    ListOfFolders = funcGetSubfolders(strPath)
    
    ' process top folder and each of its subfolders, sub-subfolders, etc.
    For Each curPath In ListOfFolders
        strFile = Dir$(curPath & "*.doc") ' also finds *.docx, *.docm
        Do
            On Error GoTo skipfile ' need to skip over docs that won't open
            Set curDoc = Documents.Open(FileName:=curPath & strFile)
            ' make the styles in curDoc the same as those in its
            ' attached template
            curDoc.UpdateStyles
            
            curDoc.Close SaveChanges:=wdSaveChanges
skipfile:
            On Error GoTo 0
            ' look for next file in this folder
            strFile = Dir
        Loop Until strFile = ""
    Next curPath
End Sub

Private Function SelectRootFolder() As String
    With Application.FileDialog(msoFileDialogFolderPicker)
        .AllowMultiSelect = False
        .InitialFileName = "C:\"   ' <== change this if you want to start elsewhere
        .Title = "Select Starting Folder"
        If .Show <> 0 Then
            SelectRootFolder = .SelectedItems(1)
        Else
            SelectRootFolder = ""  ' this will cause the main macro to quit
        End If
    End With
End Function

' The next two functions are from the article at
' http://www.word.mvps.org/FAQs/MacrosVBA/ReadFoldersIntoArray.htm
' which includes comments to explain their operation.
Public Function funcGetSubfolders(ByVal FolderToRead As String) As Variant
    Dim AllSubFolders(0) As Variant
    
    On Error Resume Next
    System.Cursor = wdCursorWait
    
    If (Right$(FolderToRead, 1) <> "\") Then
        FolderToRead = FolderToRead & "\"
    End If
    
    AllSubFolders(0) = FolderToRead
    funcGetSubfolders = funcGetAllSubfolders(AllSubFolders)
    
    System.Cursor = wdCursorNormal
    On Error GoTo 0
End Function

Private Function funcGetAllSubfolders(ByVal AllSubFoldersArray As Variant) As Variant
    Dim Counter As Integer
    Dim CurFolderName As String
    Dim SubFolderName As String
    Dim SubFolderList() As String
    
    On Error Resume Next
    
    CurFolderName = CStr(AllSubFoldersArray(UBound(AllSubFoldersArray)))
    
    ReDim SubFolderList(0)
    SubFolderName = Dir$(CurFolderName, vbDirectory)
    Do While Len(SubFolderName) <> 0
        If SubFolderName <> "." And SubFolderName <> ".." Then
            If (GetAttr(CurFolderName & SubFolderName) _
                    And vbDirectory) = vbDirectory Then
                ReDim Preserve SubFolderList(UBound(SubFolderList) + 1)
                SubFolderList(UBound(SubFolderList)) = SubFolderName
            End If
        End If
        SubFolderName = Dir$()
    Loop
    
    If UBound(SubFolderList) > 0 Then
        WordBasic.SortArray SubFolderList()
    End If
    
    For Counter = 1 To UBound(SubFolderList)
        ReDim Preserve AllSubFoldersArray(UBound(AllSubFoldersArray) + 1)
    
        AllSubFoldersArray(UBound(AllSubFoldersArray)) = _
                CurFolderName & SubFolderList(Counter) & "\"
    
        AllSubFoldersArray = funcGetAllSubfolders(AllSubFoldersArray)
    Next Counter
    
    funcGetAllSubfolders = AllSubFoldersArray
    On Error GoTo 0
End Function

jay freedman say:

That's not too difficult.

First, in the MultiFile_StyleUpdate macro, just after the Dim curDoc line, add these lines:

    Dim strNewTemplate As String
    
    ' call function to select optional new template
    strNewTemplate = GetNewTemplate()

In the same macro, after the line that starts with Set curDoc = Documents.Open, add these lines:

                ' if a new template was selected, attach it to the document
                If strNewTemplate <> "" Then
                    curDoc.AttachedTemplate = strNewTemplate
                End If

Finally, add this function to the end of the code:

Private Function GetNewTemplate() As String
    With Application.FileDialog(msoFileDialogOpen)
        .AllowMultiSelect = False
        .InitialFileName = Options.DefaultFilePath(wdUserTemplatesPath)   ' <== change this if you want to start elsewhere
        .Title = "Select Optional New Template"
        If .Show <> 0 Then
            GetNewTemplate = .SelectedItems(1)
        Else
            GetNewTemplate = ""  ' this will cause the main macro to quit
        End If
    End With
End Function

When the Select Optional New Template pops up at the beginning of a run, if you want to leave the documents' template as is, just cancel the dialog. To attach a different template, select it in the dialog and click Open.

Your symptom certainly indicates that Windows Defender is sensing another

If you decide you want keep your third party, then set the periodic scanning

rich007a say:

You've pretty much said everything you need to know....

From Excel pres Alt+F11 to open the VBE.  Then do Tools > References...    then scroll down to the "Microsoft PowerPoint 11.0 Object Library" and check the box.  OK.

That's all that you need to do.

In your macro you can now do something like:

Option Explicit

Sub DoStuffWithPowerPointFromExcel()
    Dim PPT As PowerPoint.Application
    Dim pptPres As PowerPoint.Presentation
    Dim pptSld As PowerPoint.Slide
   
    Dim bWeStartedPpt As Boolean
   
     'Get PPT if it's running:
    On Error Resume Next
    Set PPT = GetObject(, "PowerPoint.Application")
    If PPT Is Nothing Then
        On Error GoTo CleanUp
         'PPT wasn't running, start it from code:
        Set PPT = CreateObject("PowerPoint.Application")
        bWeStartedPpt = True
    End If
    On Error GoTo CleanUp
   
     'Now you can use PPT from Excel, e.g.:
    '==========================================================
    Set pptPres = PPT.Presentations.Open("C:/temp/my_pres.pptx")
    Set pptSld = pptPres.Slides(1) 'The first slide in the pres
   
    ActiveSheet.Range("A1:B10").Copy 'Copy from Excel
    pptSld.Select
    PPT.ActiveWindow.View.Paste 'Paste into slide
   
    pptPres.Save
    pptPres.Close
    '==========================================================
   
CleanUp:
    If bWeStartedPpt Then
          'If we started PPT from code, then close it:
        If Not PPT Is Nothing Then PPT.Quit
    End If
    Set PPT = Nothing
End Sub

 

Hope that helps.
