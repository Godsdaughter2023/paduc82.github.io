doug robbins - mvp office apps & services (word) say:

The following should insert the data into an Excel Workbook

Sub BatchExtracttoExcel()

Dim strPath As String
Dim strFile As String
Dim FD As FileDialog
Dim Doc As Document
Dim rng As Range
Dim rngCompany As Range
Dim rngAddress As Range
Dim rngContact As Range
Dim rngEmail As Range
Dim rngPhone As Range
Dim rngFax As Range
Set FD = Application.FileDialog(msoFileDialogFolderPicker)
With FD
    .Title = "Select the folder that contains the documents."
    If .Show = -1 Then
        strFolder = .SelectedItems(1) & "\"
    Else
        MsgBox "You did not select a folder."
        Exit Sub
    End If
End With
Dim xlapp As Object
Dim xlbook As Object
Dim xlsheet As Object
Dim i As Long
On Error Resume Next
Set xlapp = GetObject(, "Excel.Application")
If Err Then
    bstartApp = True
    Set xlapp = CreateObject("Excel.Application")
End If
On Error GoTo 0
Set xlbook = xlapp.Workbooks.Add
Set xlsheet = xlbook.Worksheets(1)
With xlsheet.Range("A1")
    .Offset(0, 0) = "Company"
    .Offset(0, 1) = "Address"
    .Offset(0, 2) = "Contact"
    .Offset(0, 3) = "E-mail"
    .Offset(0, 4) = "Phone"
    .Offset(0, 5) = "Fax"
End With
i = 1
strFile = Dir$(strFolder & "*.docx")
While strFile <> ""
    Set Doc = Documents.Open(strFolder & strFile)
    With Doc
        With Selection
            .HomeKey wdStory
            With .Find
                .ClearFormatting
                .Text = " [0-9]{1,2}, [0-9]{4}^13"
                .Forward = True
                .Wrap = wdFindContinue
                .MatchWildcards = True
                .Execute
            End With
            Set rng = .Range
            With rng
                .Start = rng.End
                .End = ActiveDocument.Range.End
                .End = rng.Start + InStr(rng, "Reference") - 2
                .Text = Replace(.Text, vbCr & vbCr, vbCr)
                .Text = Replace(.Text, vbCr & vbCr, vbCr)
                Set rngCompany = .Paragraphs(2).Range
                Set rngAddress = .Paragraphs(3).Range
                rngAddress.End = .Paragraphs(4).Range.End - 1
                rngAddress.Text = Replace(rngAddress.Text, vbCr, ", ")
                Set rngContact = .Paragraphs(4).Range
                With rngContact
                    .Start = .Start + InStr(.Text, ":") + 1
                    .Text = Replace(.Text, vbTab, "")
                End With
                Set rngEmail = .Paragraphs(5).Range
                With rngEmail
                    .Start = .Start + InStr(.Text, ":") + 1
                    .Text = Replace(.Text, vbTab, "")
                End With
                Set rngPhone = .Paragraphs(6).Range
                With rngPhone
                    .Start = .Start + InStr(.Text, ":") + 1
                    .Text = Replace(.Text, vbTab, "")
                End With
                Set rngFax = .Paragraphs(7).Range
                With rngFax
                    .Start = .Start + InStr(.Text, ":") + 1
                    .Text = Replace(.Text, vbTab & vbTab, "")
                End With
            End With
        End With
    End With
    With xlsheet.Range("A1")
        .Offset(i, 0) = rngCompany
        .Offset(i, 1) = rngAddress
        .Offset(i, 2) = rngContact
        .Offset(i, 3) = rngEmail
        .Offset(i, 4) = rngPhone
        .Offset(i, 5) = rngFax
    End With
    i = i + 1
    Doc.Close wdDoNotSaveChanges
    strFile = Dir$()
Wend
xlbook.Activate

End Sub
