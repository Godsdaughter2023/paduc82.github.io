You can test if the fileexists, and if it does, promtp for another filename:

Sub SaveWorkbook()
 
Dim varWorkbookName As String
Dim sFileExtension As String
Dim fs

Set fs = CreateObject("Scripting.FileSystemObject")

MsgBox "Please select where you want to save this file. If you fail to select a location it will be stored in the My Documents on this computer and you will not have access to it from other computers."
 
MsgBox "Also PLEASE DO NOT CHANGE THE 'SAVE AS TYPE'." & vbLf & vbLf & _
    "If it is not saved as a Macro Enabled file, it WILL NOT WORK."
    Application.EnableEvents = False
    Do
     varWorkbookName = Application.GetSaveAsFilename(InitialFileName:=Worksheets("Master Sheet").[AJ1].Value & ", " & Worksheets("Master Sheet").[DC2].Value & " " & Worksheets("Master Sheet").[DC3].Value & ", " & "Employee Tracker" & ".xlsm", _
    filefilter:="Excel Macro Enabled Workbook (*.xlsm), *.xlsm)", _
    FilterIndex:=1)
     If varWorkbookName <> "False" Then
        sFileExtension = CreateObject("Scripting.FileSystemObject").GetExtensionName(varWorkbookName)
        varWorkbookName = Left(varWorkbookName, Len(varWorkbookName) - Len(sFileExtension)) & "xlsm"
        fe = fs.fileexists(varWorkbookName)
        If Not fe Then
         ActiveWorkbook.SaveAs Filename:=varWorkbookName, FileFormat:=52
        Else
         MsgBox ("File exists!" & vbLf & vbLf & "Pick another filename")
        End If
    End If
    Loop Until fe = False
    Application.EnableEvents = True
 
End Sub
