Private Sub Worksheet_Change(ByVal Target As Range)
Dim rngDV As Range
Dim oldVal As String
Dim newVal As String
Dim lUsed As Long
If Target.Count > 1 Then GoTo exitHandler

On Error GoTo exitHandler

If rngDV Is Nothing Then GoTo exitHandler

If Intersect(Target, rngDV) Is Nothing Then
   'do nothing
Else
  Application.EnableEvents = False
  newVal = Target.Value
  Application.Undo
  oldVal = Target.Value
  Target.Value = newVal
  If Target.Column = 7 Or 8 Then
    If oldVal = "" Then
      'do nothing
      Else
      If newVal = "" Then
      'do nothing
      Else
        lUsed = InStr(1, oldVal, newVal)
        If lUsed > 0 Then
            If Right(oldVal, Len(newVal)) = newVal Then
                Target.Value = Left(oldVal, Len(oldVal) - Len(newVal) - 2)
            Else
                Target.Value = Replace(oldVal, newVal & ", ", "")
            End If
        Else
            Target.Value = oldVal _
              & ", " & newVal
        End If
       
      End If
    End If
  End If
End If

exitHandler:

Private Sub Worksheet_Change(ByVal Target As Range)
  Dim N As Name
  Dim R As Range
  On Error GoTo ErrorHandler
  'Check all names
  For Each N In ThisWorkbook.Names
    'A change in the same column?
    If Intersect(Target, N.RefersToRange.EntireColumn) Is Nothing Then GoTo Skip
    'Get the top cell
    Set R = N.RefersToRange.Cells(1)
    'Resize to the last cell
    Set R = Range(R, R.Offset(Rows.Count - R.Row).End(xlUp))
    'Store the new range
    N.RefersTo = "=" & R.Address
Skip:
  Next
  Exit Sub
ErrorHandler:
  'Skip the name if an error occurs
  Resume Skip
End Sub
