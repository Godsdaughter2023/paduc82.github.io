Check if the file is already open before you try to open the file exclusively.

Andreas.

Function IsOpen(ByVal FileName As String) As Boolean
  'True if FileName is open
  Dim slot As Integer
  If Dir(FileName) <> vbNullString Then
    On Error Resume Next
    slot = FreeFile
    Open FileName For Binary Access Read Lock Read As #slot
    IsOpen = Err.Number <> 0
    Close #slot
  Else
    IsOpen = False
  End If
End Function
