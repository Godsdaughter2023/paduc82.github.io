The code below will unprotect all worksheet with one entry of the password provided that all worksheets have the same password. (Your original code would do this but it aborted if a password was incorrect.)

However, with this code, if any sheet does not have the same password then instead of aborting, the InputBox is displayed again with a modified message telling you for which sheet the password is incorrect and requeststs that you re-enter it and the code continues.

Note that there is a sub plus a Function.

Code modified since initial post to include close workbook after 3 failed password attempts.

 

 

The Public declaration must be at the top of a standard module before any subs

Public passwordCount As Long    'Will be initialized to zero on workbook open

Sub UnrotectAll()
 'assumes that all sheets in the workbook
 'have the same non-empty password or
 'that they are not protected at all.
 '
 Dim anyWS As Worksheet
 Dim thePassword As String
 Dim strPrompt As String
 
 strPrompt = "Enter Password for sheets in the workbook: "
 
 thePassword = EnterPassword(strPrompt)
 
 If thePassword = "" Then
    MsgBox "User cancelled or did not enter a value. Processing terminated"
    Exit Sub
 End If
 
 On Error Resume Next
 passwordCount = 0  'Ensure initialized
 For Each anyWS In ThisWorkbook.Worksheets
    If anyWS.ProtectContents Then
        anyWS.Unprotect Password:=thePassword
        If Err.Number <> 0 Then
            Do
                Err.Clear
                strPrompt = "The password you entered for sheet " & anyWS.Name & _
                            " is incorrect." & vbCrLf & "Re-Enter password."
                passwordCount = passwordCount + 1
               
                'Close the workbook after 3 unsuccessfull attempts
                If passwordCount >= 3 Then ThisWorkbook.Close SaveChanges:=False
               
                If thePassword = "" Then
                    MsgBox "User cancelled or did not enter a value." _
                            & vbCrLf & "Processing terminated"
                    Exit Sub
                End If

                thePassword = EnterPassword(strPrompt)
                'Retry for current worksheet
                anyWS.Unprotect Password:=thePassword
            Loop While Err.Number > 0
        End If
    End If
 Next
 On Error GoTo 0
 MsgBox "Worksheets are now unprotected.", _
 vbOKOnly + vbInformation, "Task Completed"
 End Sub


Function EnterPassword(strPrompt) As String
    EnterPassword = InputBox(strPrompt, "Enter Password", "")
End Function
