jesse petre say:

While this won't FORCE you to assign a category, it should help you get in the habit of doing so.

 

  1. Create a new rule.
  2. Choose "Apply rule on messages I send" and click Next
  3. On the "Select Condition" dialogue, don't check anything and click next. Outlook will warn you that this will apply to all outgoing email, click past the warning.
  4. On the "Select Action(s)" dialouge check "Defer Delivery by a number of minutes"  and set it to 120 (the max allowed...or whatever you want.) Click Next
  5. On the "Select Exceptions" screen, check "except if assigned to any category".

Now, if you don't assign a category to an outgoing email it will sit in your Outbox until you either:

  1. Wait 2 hours
  2. Assign a category to it and click "Send' again

*****************************

** BIG GIANT WARNING ***

*****************************

If you CLOSE Outlook with messages in the Outbox, they WILL NOT send until you come back and re-open Outlook. So if you use this method, ALWAYS CHECK your Outbox before shutting down for the day.

norwoodada say: Hi,  You can use the following macro to the task:

Press Alt + F11 keys to enter VBA eidtor window, and then copy the following macro to the ThisOutlookSession project.

     
Public WithEvents objExplorer As Outlook.Explorer
Public strPrompt As String, nWarning As Integer
Private Sub Application_Startup()
    Set objExplorer = Outlook.Application.ActiveExplorer
End Sub
Private Sub objExplorer_SelectionChange()
    Dim objItem As Object
    Dim strItemType As String
 
    On Error GoTo ErrorHandler
 
    Set objItem = objExplorer.Selection.Item(1)
    strItemType = Replace(TypeName(objItem), "Item", "")
 
    If objItem.Categories = "" Then
       strPrompt = "You haven't assigned a category to the " & strItemType & " - " & objItem.Subject & "!" & vbCrLf & vbCrLf & "Do you want to categorize it now?"
       nWarning = MsgBox(strPrompt, vbExclamation + vbYesNo, "Warning: No Category!")
       If nWarning = vbYes Then
          objItem.ShowCategoriesDialog
       End If
    End If
 
ErrorHandler:
    Exit Sub
End Sub
'Force you to assign category to the outgoing items before sending
Private Sub Application_ItemSend(ByVal objItem As Object, Cancel As Boolean)
    If objItem.Categories = "" Then
       strPrompt = "You haven't assigned a category to current item!" & vbCrLf & vbCrLf & "Do you want to categorize it now?"
       nWarning = MsgBox(strPrompt, vbExclamation + vbYesNo, "Warning: No Category!")
       If nWarning = vbYes Then
          objItem.ShowCategoriesDialog
       End If
    End If
End Sub
