hansv mvp say:

Here you go. Change the constants at the beginning as needed. I have assumed that you will run the macro on the active sheet.

Sub ConcatColor()
    Const lngColumn = 1 ' Column A
    Const lngColor = vbRed ' Color to look for
    Dim lngRow As Long
    Dim lngLastRow As Long
    Dim blnFlag As Boolean
    Dim strResult As String
    Application.ScreenUpdating = False
    lngLastRow = Cells(Rows.Count, lngColumn).End(xlUp).Row
    For lngRow = lngLastRow To 1 Step -1
        If Cells(lngRow, lngColumn).Interior.Color = lngColor Then
            If blnFlag Then
                strResult = Cells(lngRow, lngColumn).Value & "," & strResult
            Else
                strResult = Cells(lngRow, lngColumn).Value
                blnFlag = True
            End If
        ElseIf blnFlag Then
            Cells(lngRow, lngColumn).Value = strResult
            blnFlag = False
        End If
    Next lngRow
    Application.ScreenUpdating = True
End Sub
