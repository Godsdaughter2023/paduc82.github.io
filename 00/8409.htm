hansv mvp say:

See if this does what you want.

 

Sub Copy_Data_To_Word()
    ' Path to save the documents in - modify as needed
    Const sFolder = "C:\Reminders\"
    Dim oWA As Object
    Dim oWD As Object
    Dim BMRange As Object
    Dim bStart As Boolean
    Dim r As Long
    Dim m As Long
   
    On Error Resume Next
    Set oWA = GetObject(Class:="Word.Application")
    If oWA Is Nothing Then
        Set oWA = CreateObject(Class:="Word.Application")
        If oWA Is Nothing Then
            MsgBox "Failed to start Word", vbCritical
            Exit Sub
        Else
            bStart = True
        End If
    End If

    On Error GoTo ErrHandler
    m = Cells(Rows.Count, 1).End(xlUp).Row
    For r = 2 To m
        Set oWD = oWA.Documents.Add("C:\Business\MISC\Expiration Reports\Expiration report.dot")
        Set BMRange = oWD.Bookmarks("ExpDate").Range
        BMRange.Text = Cells(r, 1)
        oWD.Bookmarks.Add "ExpDate", BMRange
        Set BMRange = oWD.Bookmarks("RemindDate").Range
        BMRange.Text = Cells(r, 2)
        oWD.Bookmarks.Add "RemindDate", BMRange
        Set BMRange = oWD.Bookmarks("Company").Range
        BMRange.Text = Cells(r, 3)
        oWD.Bookmarks.Add "Company", BMRange
        Set BMRange = oWD.Bookmarks("Code").Range
        BMRange.Text = Cells(r, 4)
        oWD.Bookmarks.Add "Code", BMRange
        ' Code for saving the document
        oWD.SaveAs sFolder & Cells(r, 3) & Format(Cells(r, 2), "yyyymmdd") & ".doc"
        oWD.Close
    Next r

ExitHandler:
    ' Releasing objects etc
    On Error Resume Next
    If Not oWA Is Nothing And bStart Then
        oWA.Quit SaveChanges:=False
    End If
    Exit Sub

ErrHandler:
    MsgBox Err.Description, vbExclamation
    Resume ExitHandler
End Sub

Installing 2016 doesn't affect your 2011 installation. Unless you manually deleted it, your Office 2011 install is right where it was before you installed Office 2016.

When your license expires, the Office install still stays on your Mac. The Office applications will become viewers. You will be able to open and look at documents, workbooks and presentations, but you won't be able to edit or save files. Outlook will go into Offline mode.
