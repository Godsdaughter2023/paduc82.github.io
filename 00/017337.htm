Function ConvertDegMinSec(DegMinSeconds As Double) As Double
'INPUT: angle as degrees.minutesseconds
'     Where: degrees is a 1, 2 or 3 digit angle
'            minutes is a 1 or 2 digit number for the actual minutes of the angle: <=59
'             if seconds are to be used must have leading 0 for minutes < 10, as 10.09
'             for 10 degrees 9 minutes.
'            seconds are actual seconds with any decimal portion <=60, as
'             3075 for 30.75 seconds
'OUTPUT: Decimal representation of the input angle.
'Example:
' Input:  10.303075  (10° 30' 30.75")
' Output: 10.50513
'
  Dim DegreeValue As Integer
  Dim MinutesValue As Double
  Dim SecondsValue As Double
 
  DegreeValue = Fix(DegMinSeconds) ' integer without rounding
  MinutesValue = Fix((DegMinSeconds - DegreeValue) * 100) / 60
  SecondsValue = (DegMinSeconds - (DegreeValue + (MinutesValue * 0.6))) * 100 / 60
  ConvertDegMinSec = DegreeValue + MinutesValue + SecondsValue
End Function
