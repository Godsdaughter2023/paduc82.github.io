andreas killer say:

The code below can fail or return wrong results on different conditions:

a) All rows are filtered

b) Returned string from Range.Address is longer then 255 chars
c) No filter is active
d) ...

A failsafe solution is IMHO only possible without names for this.

Andreas.

Sub TestRun()
  Dim All As Range, This As Range

  With Sheet6
    Set All = GetAutoFilterRange(True, .Cells(1, 1).Parent)
    Set This = Intersect(All, .Columns("J"))
    .Names("CheckBoxControlsT").RefersTo = "=" & This.Address(External:=True)
    Set This = Intersect(All, .Columns("C"))
    .Names("BondAmountsT").RefersTo = "=" & This.Address(External:=True)
  End With
End Sub

Function GetAutoFilterRange( _
    Optional WithOutHeader As Boolean = True, _
    Optional ByVal Ws As Worksheet = Nothing) As Range
  'Returns the visible range of an autofilter
  Dim R As Range, Temp(), i As Long, j As Long
  If Ws Is Nothing Then Set Ws = ActiveSheet
  'No filter, return nothing
  If Not Ws.AutoFilterMode Then Exit Function
  'Get the whole range from the filter
  Set R = Ws.AutoFilter.Range
  'Remove headings?
  If WithOutHeader Then
    If R.Rows.Count = 1 Then Exit Function
    Set R = R.Offset(1, 0).Resize(R.Rows.Count - 1, R.Columns.Count)
  End If
  'Is the filter activ?
  If Ws.FilterMode Then
    'Error's off, we get an error if no cells are visible
    On Error Resume Next
    Set GetAutoFilterRange = R.SpecialCells(xlCellTypeVisible)
  Else
    'Whole range
    Set GetAutoFilterRange = R
  End If
End Function

Open administrative Command Prompt and type following commands one-by-one followed by Enter key.

net stop wuauserv

net stop msiserver
Ren C:\Windows\SoftwareDistribution SoftwareDistribution.old
Ren C:\Windows\System32\catroot2 Catroot2.old

net start bits
net start msiserver

Close Command Prompt and see if Windows Updates works then
