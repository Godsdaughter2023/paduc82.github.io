I have workbook that contains worksheets with several cells that contain Data Validation input messages. The workbook requires translation into other language. bernie deitrick, excel mvp 2000-2010 say:

Here is a full solution.

In the workbook with the DV, insert a standard codemodule and copy the code below into it. Then run the first macro "GetMessages".

All the messages will be listed on the sheet "Messages", with the current messages in B to E and room for translations in columns F to I.

Put the translations in the table, and run the second macro "UpdateDVMessages".

If you ever need to revert to the old messages, copy columns B to E over columns F to I and run the second macro again.

Sub GetMessages()
'
    Dim i As Integer
    Dim myM As Worksheet
    Dim myC As Range
    Dim myRow As Long
    Dim myCalc As XlCalculation

    With Application
        .DisplayAlerts = False
        .EnableEvents = False
        myCalc = .Calculation
        .Calculation = xlCalculationManual
    End With


    Set myM = Worksheets.Add(before:=Worksheets(1))
    myM.Name = "Messages"
    myM.Cells(1, 1).Value = "Address"
    myM.Cells(1, 2).Value = "Existing Input Title"
    myM.Cells(1, 3).Value = "Existing InputMessage"
    myM.Cells(1, 4).Value = "Existing Error Title"
    myM.Cells(1, 5).Value = "Existing Error Message"
    myM.Cells(1, 6).Value = "Translated Input Title"
    myM.Cells(1, 7).Value = "Translated InputMessage"
    myM.Cells(1, 8).Value = "Translated Error Title"
    myM.Cells(1, 9).Value = "Translated Error Message"
    myM.Rows(1).Cells.WrapText = True

    On Error GoTo NoValidation
    For i = 2 To Worksheets.Count
        For Each myC In Worksheets(i).Cells.SpecialCells(xlCellTypeAllValidation)
            myRow = myM.Cells(Rows.Count, 1).End(xlUp).Row + 1
            myM.Cells(myRow, 1).Value = myC.Address(False, False, xlA1, True)
            If myC.Validation.ShowInput Then
                myM.Cells(myRow, 2).Value = myC.Validation.InputTitle
                myM.Cells(myRow, 3).Value = myC.Validation.InputMessage
            End If
            If myC.Validation.ShowError Then
                myM.Cells(myRow, 4).Value = myC.Validation.ErrorTitle
                myM.Cells(myRow, 5).Value = myC.Validation.ErrorMessage
            End If
        Next myC
NoValidation:
    Next i

    With myM.Cells
        .ColumnWidth = 16.43
        .EntireRow.AutoFit
    End With

    With myM.Range("A1").CurrentRegion
        .Borders(xlDiagonalDown).LineStyle = xlNone
        .Borders(xlDiagonalUp).LineStyle = xlNone
        With .Borders(xlEdgeLeft)
            .LineStyle = xlContinuous
            .Weight = xlThin
            .ColorIndex = xlAutomatic
        End With
        With .Borders(xlEdgeTop)
            .LineStyle = xlContinuous
            .Weight = xlThin
            .ColorIndex = xlAutomatic
        End With
        With .Borders(xlEdgeBottom)
            .LineStyle = xlContinuous
            .Weight = xlThin
            .ColorIndex = xlAutomatic
        End With
        With .Borders(xlEdgeRight)
            .LineStyle = xlContinuous
            .Weight = xlThin
            .ColorIndex = xlAutomatic
        End With
        With .Borders(xlInsideVertical)
            .LineStyle = xlContinuous
            .Weight = xlThin
            .ColorIndex = xlAutomatic
        End With
        With .Borders(xlInsideHorizontal)
            .LineStyle = xlContinuous
            .Weight = xlThin
            .ColorIndex = xlAutomatic
        End With
    End With
   
    With Application
        .DisplayAlerts = True
        .EnableEvents = True
        .Calculation = myCalc
    End With
End Sub


Sub UpdateDVMessages()
    Dim myM As Worksheet
    Dim i As Long
    Set myM = Worksheets("Messages")
    For i = 2 To myM.Cells(Rows.Count, 1).End(xlUp).Row
        With Range(myM.Cells(i, 1).Value)
            If myM.Cells(i, 2).Value <> "" Then .Validation.InputTitle = myM.Cells(i, 6).Value
            If myM.Cells(i, 3).Value <> "" Then .Validation.InputMessage = myM.Cells(i, 7).Value
            If myM.Cells(i, 4).Value <> "" Then .Validation.ErrorTitle = myM.Cells(i, 8).Value
            If myM.Cells(i, 5).Value <> "" Then .Validation.ErrorMessage = myM.Cells(i, 9).Value
        End With
    Next i
End Sub

jeeped say:

Oh-oh-oh , how hard we try to save users from themselves.

The following event macro belongs in the ThisWorkbook module.

Private Sub Workbook_BeforeSave(ByVal SaveAsUI As Boolean, Cancel As Boolean)

    Dim sFN As String
    If SaveAsUI Then
        Cancel = True
        sFN = Application.GetSaveAsFilename(, _
           fileFilter:="Excel Macro-Enabled Workbooks (*.xlsm),*.xlsm")
        Debug.Print sFN
        If sFN <> "False" Then _
            ThisWorkbook.SaveAs sFN, xlXMLSpreadsheet
    End If
End Sub


Don't put this into a new macro-enabled template before saving it to a location and filename first. The SaveAsUI parameter means that the SaveAs dialog is intended to open and this will not allow you to save as a template. However, if it is already saved as a template you can just tap Ctrl+S and the changes will be saved.

The same will occur for the user. When they try to initially save, it will bring up a Save As dialog and force the save to be .XLSM. If they want to save afterwards, they can just save but if they try to Save As again, it will force the XLSM format again.
