alan wu msft say:
Hi Yasser m.abdel moneim,

Given the situation, you can try assigning the user for the “send on behalf” permission of the distribution group (DG) in your Exchange 2010 (on premise).

Based on my test, there isn’t a choice of the user synced from Office 365 while editing the “group delegation” (send on behalf of permission) of the DG. Public Function OpenOutlook(Optional strWhere As String = "Drafts") As Boolean
    Dim objMAPINamespace As Object
    On Error Resume Next
    
    'Get handle to Outlook if already open
    Set objOutlook = GetObject(, "Outlook.Application")
    
    'Create new Outlook object if not open
    If Err.Number = 429 Then
        Set objOutlook = CreateObject("Outlook.application")
    End If
    
    On Error GoTo 0
    
    Set objMAPINamespace = objOutlook.GetNamespace("MAPI")
    If objOutlook.ActiveExplorer Is Nothing Then
        If strWhere = "Drafts" Then
            ' Drafts folder
            objOutlook.Explorers.Add _
                    (objMAPINamespace.GetDefaultFolder(16), 0).Activate
        Else
            ' Inbox folder
            objOutlook.Explorers.Add _
                    (objMAPINamespace.GetDefaultFolder(6), 0).Activate
        End If
    Else
        If strWhere = "Drafts" Then
            ' Drafts folder
            Set objOutlook.ActiveExplorer.CurrentFolder = objMAPINamespace.GetDefaultFolder(16)
        Else
            ' Inbox folder
            Set objOutlook.ActiveExplorer.CurrentFolder = objMAPINamespace.GetDefaultFolder(6)
        End If
        'Make Outlook visible
        objOutlook.ActiveExplorer.Display
    End If
    OpenOutlook = True
End Function

I'm not sure what error message you are getting. Do you have 4 values? Have you attempted to msgbox Tests(0)?

Below is some code  I used with Excel (not Access) to conditionally format bars to red or green.

Sub ChartGreenRed(cht As Chart, Optional intSeries As Integer = 1, _
        Optional intTarget As Integer = 2, _
        Optional intAboveBelow As Integer = 1, _
        Optional lngGoodColor As Long = vbGreen, _
        Optional lngBadColor As Long = vbRed)

'---------------------------------------------------------------------------------------
' Procedure : ChartGreenRed
' Author    : ushookomd
' Date      : 11/8/2012
' Purpose   : Chart color coding bars to green or red based on there relation to a target
' cht is a chart object (required)
' intSeries is the SeriesCollection number of the actual data bars to color (default is 1 or first series)
' int Target is the SeriesCollection number of the target values (default is 2 or second series)
' intAboveBelow is 1 if the green values are higher than target or -1 if green values are lower than target (default is 1 - Higher is better)
' ChartGreenRed cht, 1, 2, -1 (will color the 1st series based on 2nd series where lower is better/green
'---------------------------------------------------------------------------------------

    Dim intPoints As Integer
    Dim arPointsA                                        'array to store the point values for first series
    Dim arPointsB                                        'array to store the point values for second series
    arPointsA = cht.SeriesCollection(intSeries).Values   'assign the series values to the array
    arPointsB = cht.SeriesCollection(intTarget).Values   'assign the series values to the array
    For intPoints = 1 To UBound(arPointsA)               'loop through each value of the array of points
        If intAboveBelow * arPointsA(intPoints) >= intAboveBelow * arPointsB(intPoints) Then   'check for values above or below target
            'need to set the color of the bar to green else red
            cht.SeriesCollection(1).Points(intPoints).Interior.Color = lngGoodColor      'assign the point a color of green
         Else
            cht.SeriesCollection(1).Points(intPoints).Interior.Color = lngBadColor        'assign the point a color of red
        End If
    Next intPoints
   
End Sub
