    doug robbins - mvp office apps & services (word) say:

    A macro containing the following code will create a new document that contains the text from the page of the active document on which the selection is located and then attach that document to an email message:

    Dim i As Long
    Dim strRecipients As String
    Dim arrRecipients As Variant
    Dim rngPage As Range
    Dim DocSource As Document
    Dim strMessage As String
    Dim strSubject As String
    Dim bStarted As Boolean
    Dim oOutlookApp As Object
    Dim oItem As Object
    Dim DocTarget As Document
    Dim strfname As String
    Set DocSource = ActiveDocument
    Set rngPage = DocSource.Bookmarks("\page").Range
    Set DocTarget = Documents.Add
    DocTarget.Range.FormattedText = rngPage.FormattedText
    DocTarget.Save
    strfname = DocTarget.fullname
    strRecipients = InputBox("Enter the e-mail addresses to which the page is to be sent, with each one separated by a semi-colon (;)")
    arrRecipients = Split(strRecipients, ";")
    strMessage = InputBox("Enter the text for the covering e-mail message.")
    strSubject = InputBox("Enter the subject for the e-mail message.")
    On Error Resume Next
    Set oOutlookApp = GetObject(, "Outlook.Application")
    If Err <> 0 Then
        Set oOutlookApp = CreateObject("Outlook.Application")
        bStarted = True
    End If
    On Error GoTo ErrMsg
    Set oItem = oOutlookApp.CreateItem(0)
    With oItem
        For i = LBound(arrRecipients) To UBound(arrRecipients)
            .Recipients.Add arrRecipients(i)
        Next i
        .Subject = strSubject
        .Body = strMessage
        .Attachments.Add strfname
        .Send
    End With
    Set oItem = Nothing
    DocTarget.Close wdDoNotSaveChanges
    Set oItem = Nothing
    If bStarted Then
        oOutlookApp.Quit
    End If
    Set oOutlookApp = Nothing
    Exit Sub
    ErrMsg:
    MsgBox Err.Number & vbCr & Err.Description

    You could put a macro button field to run the macro on each page of the document, or add a button to the Quick Access Toolbar to run it.
