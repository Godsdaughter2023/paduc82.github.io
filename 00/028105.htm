OK - what does the hyperlink editor say the address is? The macro can only check valid Windows file paths. Sub ChkHypLnks()
Dim wksHypLnks, wksBadHypLnks As Worksheet
Dim curHypLnk As Hyperlink
Dim curFile As String
Dim iBadHypLnks As Integer

    Set wksHypLnks = ActiveSheet
    Set wksBadHypLnks = ActiveWorkbook.Worksheets.Add

    wksBadHypLnks.Name = "BadHypLnks" _
                         & Right(wksBadHypLnks.Name, Len(wksBadHypLnks.Name) - 5)

    For Each curHypLnk In wksHypLnks.Hyperlinks
        If Dir(curHypLnk.Address) = "" Then
            If Dir(ActiveWorkbook.Path & Chr(92) & curHypLnk.Address) = "" Then
                iBadHypLnks = iBadHypLnks + 1
                wksBadHypLnks.Cells(iBadHypLnks, 1) = curHypLnk.Address
            End If
        End If
    Next curHypLnk
    Application.DisplayAlerts = False
    If iBadHypLnks < 1 Then wksBadHypLnks.Delete
    Application.DisplayAlerts = True
End Sub

To do the same in SQL then I would suggest the following, replacing the word NameOfQry  with the actual query name.

SELECT DCount("CallKey","[NameOfQry]","CallKey <= '" & [Transactions].[CallKey] & "'") AS ID, CallRecord.CallDate, Transactions.CallKey, Transactions.TransNum, Transactions.Type, Transactions.PartID, Transactions.DOB, Transactions.Data, Transactions.Error
FROM CallRecord LEFT JOIN Transactions ON CallRecord.CallKey = Transactions.CallKey
WHERE (((CallRecord.CallDate) Between DLookUp("[T_TEMP DATA]![Start Date]","[T_TEMP DATA]") And DLookUp("[T_TEMP DATA]![End Date]","[T_TEMP DATA]")+1) AND ((Transactions.TransNum)=1) AND ((Transactions.Type)<>"CSR Transfer") AND ((Transactions.Error)<>"unable to retrieve"))
