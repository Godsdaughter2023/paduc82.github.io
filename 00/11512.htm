=IF(ISBLANK(B2),HLOOKUP(1E+99,$A2:B2,1)+((INDEX(B2:$ZZ2,,MATCH(TRUE,INDEX(ISNUMBER(B2:$ZZ2),0),0))-HLOOKUP(1E+99,$A2:B2,1))/(MATCH(TRUE,INDEX(ISNUMBER(B2:$ZZ2),0),0)+COLUMN()-MATCH(1E+99, $A2:B2)-1)*(COLUMN(B2)-MATCH(1E+99,$A2:B2))),B2)
