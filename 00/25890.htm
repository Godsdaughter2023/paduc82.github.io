A short test please, make a new file, add a userform, paste the code below into that form and run it. Can you see a textbox with coloured chars?

Andreas.

 

Private Sub UserForm_Initialize()
  Dim RTF_Font As String
  Dim RTF_Color As String
  Dim RTF_Text As String
  Dim C As Control
  Dim i As Integer
  Set C = Me.Controls.Add("Msrtedit.AAMSREdit.1")
  C.Height = 20
 
  RTF_Font = "{\rtf1\ansi\ansicpg1252\deff0\deflang1031" & _
    "{\fonttbl{\f0\fswiss\fprq2\fcharset0 Tahoma;}}"
  RTF_Color = "{\colortbl "
  For i = 0 To 15
    RTF_Color = RTF_Color & _
      ";\red" & (QBColor(i) And &HFF) & _
      "\green" & (QBColor(i) \ 256 And &HFF) & _
      "\blue" & (QBColor(i) \ 65536 And &HFF)
  Next
  RTF_Color = RTF_Color & ";}"
  RTF_Text = "\uc1\pard\cf1\f0\fs16 "
  For i = 0 To 15
    RTF_Text = RTF_Text & "\cf" & i & Chr(65 + i)
  Next
  RTF_Text = RTF_Text & "}"
 
  C.TextRTF = RTF_Font & RTF_Color & RTF_Text
End Sub
