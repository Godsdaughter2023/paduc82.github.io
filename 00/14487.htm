john sr wilson say:

I'm really not sure WHY your link text needs to include the slide number. Maybe it could say "Explanation of ...." or something similar.

If you want the number and it to update when slides are added or removed you would currently need to write code.

Here's a very untested start. If you test it make sure it's on a copy.

Sub chexLink()
Dim slideNum As Long
Dim ohl As Hyperlink
Dim osld As Slide
Dim slideID() As String
Dim W As Long
Dim strText As String
For Each osld In ActivePresentation.Slides
For Each ohl In osld.Hyperlinks
If ohl.SubAddress <> "" Then
slideID() = Split(ohl.SubAddress, ",")
If Not isCustomShow(ohl) Then
If Not isNavLink(ohl) Then
slideNum = ActivePresentation.Slides.FindBySlideID(CLng(slideID(0))).SlideIndex
Select Case ohl.Type
Case Is = msoHyperlinkRange
strText = ohl.TextToDisplay
strText = Replace(strText, getCurrNum(ohl.Parent.Parent), CStr(slideNum))
ohl.TextToDisplay = strText
ohl.SubAddress = slideID(0) & "," & slideNum & "," & slideID(2)
Case Is = msoHyperlinkShape
strText = ohl.Parent.Parent.TextFrame.TextRange.Text
strText = Replace(strText, getCurrNum(ohl.Parent.Parent.TextFrame.TextRange), CStr(slideNum))
ohl.Parent.Parent.TextFrame.TextRange.Text = strText
ohl.SubAddress = slideID(0) & "," & slideNum & "," & slideID(2)
End Select
End If
End If
End If
Next
Next
End Sub


Function isCustomShow(ohl As Hyperlink) As Boolean
On Error Resume Next
Err.Clear
Select Case ohl.ShowAndReturn
Case Is = -1, 0
isCustomShow = True
End Select
If Err <> 0 Then isCustomShow = False
End Function


Function isNavLink(ohl As Hyperlink) As Boolean
Dim slideID() As String
slideID() = Split(ohl.SubAddress, ",")
If slideID(0) = "-1" Then isNavLink = True
End Function


Function getCurrNum(otr As TextRange) As Long
Dim L As Long
For L = otr.Words.Count To 1 Step -1
If IsNumeric(otr.Words(L)) Then
getCurrNum = CLng(otr.Words(L))
Exit Function
End If
Next L
End Function
