Sub Parse_Strings()

Sub Permutate(x As String, y As String)
    Dim I As Long, j As Long, p As Long
    Dim tmp As String, yn As Integer
    j = Len(y)
    If j < 2 Then
        For p = 1 To Len(x & y)
            tmp = tmp & Mid(x & y, p, 1) & " "
        Next
        tmp = RTrim(tmp)

        For p = Len(tmp) - 1 To 2 Step -2
            yn = Int((2 * Rnd) + 0)
            If Len(tmp) < 20 Then
            If yn = 1 Then
                tmp = Left(tmp, p) & Mid(tmp, p)
            Else
                tmp = Left(tmp, p - 1) & Mid(tmp, p + 1)
            End If
            End If
        Next

Cells(currentrow, 1) = tmp
Cells(currentrow, 1).TextToColumns Destination:=Cells(currentrow, 1), _
