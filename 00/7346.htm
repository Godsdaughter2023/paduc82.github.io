andreas killer say:
Option Explicit

Public Enum RibbonStates
  rsMinimized
  rsVisible
End Enum

Property Get RibbonState(ByVal State As RibbonStates) As Boolean
  If State = rsMinimized Then
    RibbonState = CommandBars("Ribbon").Controls(1).Height < 100
  Else
    RibbonState = Application.ExecuteExcel4Macro("Get.ToolBar(7,""Ribbon"")")
  End If
End Property

Property Let RibbonState(ByVal State As RibbonStates, ByVal Value As Boolean)
  If State = rsMinimized Then
    If Value Then
      If Not RibbonState(State) Then CommandBars.ExecuteMso "MinimizeRibbon"
    Else
      If RibbonState(State) Then CommandBars.ExecuteMso "MinimizeRibbon"
    End If
  Else
    If Value Then
      Application.ExecuteExcel4Macro "Show.ToolBar(""Ribbon"", True)"
    Else
      Application.ExecuteExcel4Macro "Show.ToolBar(""Ribbon"", False)"
    End If
  End If
End Property

Private Sub Example_RibbonState()
  Dim State As RibbonStates
  'Minimized?
  If RibbonState(rsMinimized) Then
    'Maximize
    RibbonState(rsMinimized) = False
  End If
  'Hidden?
  If Not RibbonState(rsVisible) Then
    'Show
    RibbonState(rsVisible) = True
  End If
  'Toggle a random state
  State = Rnd
  RibbonState(State) = Not RibbonState(State)
End Sub
