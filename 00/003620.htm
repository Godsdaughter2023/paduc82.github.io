End Sub

' Modified not to require Microsoft Word 9.0 Object Library ref
'----------------------------------------------------
Public Function SpellCheck(ByRef strBuffer As String) As String
    'Dim objWord As Word.Application
    Dim objWord As Object
    'Dim WordDoc As Word.Document
    Dim WordDoc As Object
    Dim strlen As Integer
    Dim lLanguageID As String
    'Dim strOriginalText As String
    
    'strOriginalText = strBuffer
    
    On Error GoTo WordAutoError
        'Set objWord = New Word.Application
        Set objWord = CreateObject("Word.Application")
        ' try setting it invisible first, THEN adding the document
        objWord.Visible = False
        Set WordDoc = objWord.Documents.Add
        'objWord.Visible = False
        objWord.WindowState = 2 ' wdWindowStateMinimize

93tid say:
VoLTE means Voice over LTE or Voice over 4G. So far 4G services are still mainly used just for data and whenever you make a voice call over the network (not via Skype, Whatsapp, Viber, imo or any other similar VoIP app) it is established via a 3G (or sometimes 2G) connection.
