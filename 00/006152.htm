Private Sub Detail_Print(Cancel As Integer, PrintCount As Integer)
    If Me.Controls("Event").Text Like "*@@@@@@*" Then
        Me.Controls("Event").BackColor = RGB(247, 252, 28)  'Yellow
    ElseIf Me.Controls("Event").Text Like "*@@@@@@*" Then
        Me.Controls("Event").BackColor = RGB(245, 55, 55)   'Red
    Else
        Me.Controls("Event").BackColor = RGB(255, 255, 255) 'White
    End If
End Sub
