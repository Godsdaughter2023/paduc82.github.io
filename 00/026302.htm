dhookom say:

A crosstab would require first creating a union query:

SELECT patId, "mob"&"_"&Visit AS TheHeading, mob AS TheValue
FROM tblGarriePowers
WHERE Visit <>0
UNION ALL
SELECT patId, "sc"&"_"&Visit, sc
FROM tblGarriePowers
WHERE Visit <>0
UNION ALL
SELECT patId, "ua"&"_"&Visit, ua
FROM tblGarriePowers
WHERE Visit <>0
UNION ALL
SELECT patId, "pd"&"_"&Visit, pd
FROM tblGarriePowers
WHERE Visit <>0
UNION ALL
SELECT patId, "ad"&"_"&Visit, ad
FROM tblGarriePowers
WHERE Visit <>0
UNION ALL SELECT patId, "yht"&"_"&Visit, yht
FROM tblGarriePowers
WHERE Visit <>0;

TRANSFORM First(quniGarriePowers.TheValue) AS FirstOfTheValue
SELECT quniGarriePowers.patId
FROM quniGarriePowers
GROUP BY quniGarriePowers.patId
PIVOT quniGarriePowers.TheHeading;

This won't be in the desired column order but that could be handled using the column heading property:

TRANSFORM First(quniGarriePowers.TheValue) AS FirstOfTheValue
SELECT quniGarriePowers.patId
FROM quniGarriePowers
GROUP BY quniGarriePowers.patId
PIVOT quniGarriePowers.TheHeading In ("mob_1","sc_1","ua_1","pd_1","ad_1","yht_1","mob_2","sc_2","ua_2","pd_2","ad_2","yht_2","mob_3","sc_3","ua_3","pd_3","ad_3","yht_3");
