Sub TheWeeBitSmarterSingleQuoteReplacer()

   'hopefully a bit smarter but probably still not perfect

    Dim doc As Word.Document
    Dim rng As Word.Range
    Set doc = ActiveDocument
    Set rng = doc.Content

    Word.Options.AutoFormatAsYouTypeReplaceQuotes = True
    With rng.Find
        .ClearFormatting
        .Forward = True
        .Format = False
        .Text = "'"
        .Wrap = wdFindStop
        .Execute
        Do While .Found
            rng.MoveStart Word.WdUnits.wdCharacter, -1
            rng.MoveEnd Word.WdUnits.wdCharacter, 1
            If Asc(rng.Characters.First.Text) = 32 And Asc(rng.Characters.Last.Text) > 32 Then
                rng.MoveStart Word.WdUnits.wdCharacter, 1
                rng.MoveEnd Word.WdUnits.wdCharacter, -1
                'replace single straight with left double curly quote
                rng.Text = ChrW(8220)
            ElseIf Asc(rng.Characters.First.Text) > 32 And Asc(rng.Characters.Last.Text) = 32 Then
                rng.MoveStart Word.WdUnits.wdCharacter, 1
                rng.MoveEnd Word.WdUnits.wdCharacter, -1
                'replace single straight with right double curly quote
                rng.Text = ChrW(8221)
            ElseIf Asc(rng.Characters.First.Text) > 32 And Asc(rng.Characters.Last.Text) > 32 Then
                If Asc(rng.Characters.Last.Text) = 46 Then
                    'the last character is a period, test if it's end of sentence
                    rng.MoveEnd Word.wdCharacter, 1
                    If Asc(rng.Characters.Last.Text) <= 32 Then
                        'it is an end of sentence and the close quote is misplaced, so fix it
                        rng.MoveStart Word.WdUnits.wdCharacter, 1
                        rng.MoveEnd Word.WdUnits.wdCharacter, -2
                        rng.Delete
                        rng.Move Word.WdUnits.wdCharacter, 1
                        rng.Text = ChrW(8221)
                    End If
                Else
                    'probably an apostrophe
                    rng.MoveStart Word.WdUnits.wdCharacter, 1
                    rng.MoveEnd Word.WdUnits.wdCharacter, -1
                    'replace single straight with right single curly quote
                    rng.Text = ChrW(8217)
                End If
            End If
            rng.Collapse Word.WdCollapseDirection.wdCollapseEnd
           .Execute
        Loop
    End With
End Sub
