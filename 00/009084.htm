rick rothstein, mvp - excel say:

Okay, see if the following macro does what you want. Note that it assumes you are running it with a blank worksheet active. Also note that you will have to change the path and filename to the actual name and location for your text file in the bolded line of code.

Sub ProcessDerivedNOCs()
  Dim X As Long, Z As Long, FileNum As Long
  Dim FieldNames As Variant, RecordFields As Variant
  Dim TotalFile As String, PathAndFileName As String
  Dim Records() As String, Fields() As String
  PathAndFileName = "c:\temp\TestTextFile.txt"
  FieldNames = Array("Reference Number", "Returning ABA", "Process Date", _
                     "Settle Date", "Amount", "Debit or Credit", "Change Code", _
                     "Original Trace No", "New Trace No", "Effective Entry Date", _
                     "Company Name", "Individual Name", "Account Number", _
                     "SEC Code", "Derived By ABA", "Derived By", "Change Field 1")
  Range("A1:Q1") = FieldNames
  Columns("A:Q").AutoFit
  FileNum = FreeFile
  Open PathAndFileName For Binary As #FileNum
    TotalFile = Space(LOF(FileNum))
    Get #FileNum, , TotalFile
  Close #FileNum
  TotalFile = Replace(Mid(TotalFile, InStr(TotalFile, "Reference Number:")), vbNewLine, "   ")
  Records = Split(TotalFile, String(111, "_"))
  ReDim RecordFields(1 To UBound(Records) + 1, 1 To 17)
  For X = 0 To UBound(Records) - 1
    Fields = Split(Records(X), ":")
    For Z = 1 To UBound(Fields)
      RecordFields(X + 1, Z) = Trim(Left(Fields(Z), InStr(Fields(Z), "   ") - 1))
      If InStr(" 1 2 8 9 13 15 17 ", " " & Z & " ") Then
        RecordFields(X + 1, Z) = "'" & Trim(RecordFields(X + 1, Z))
      End If
    Next
  Next
  Range("A:A,B:B,H:H,I:I,M:M,O:O,Q:Q").EntireColumn.NumberFormat = "@"
  Range("A2").Resize(UBound(RecordFields), 17) = RecordFields
  Columns("A:Q").AutoFit
End Sub

NOTE: This code assumes that for the fields on the same line that there will be at least 3 contiguous space characters between those fields and then none of the data itself will contain 3 consecutive spaces within them. Also, as established earlier, it is assumed that exactly 111 underline characters appear between each record.
