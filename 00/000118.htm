    --- schnipp ---
    Option Explicit

    Public WithEvents OptionButton As MSForms.OptionButton

    Private Sub OptionButton_Click()
      'Enable the commandbutton

    End Sub
    --- schnapp ---

    And add the code below to your userform.

    Andreas.

    Option Explicit

    Dim MyOptionButtons As New Collection

    Private Sub UserForm_Initialize()
      Dim C As Control
      Dim MOB As MyOptionButton
      'Disable the commandbutton
      Me.CommandButton1.Enabled = False
      'Visit all controls
      For Each C In Me.Controls
        'Is it an optionbutton?
        If TypeOf C Is MSForms.OptionButton Then
          'Yes, initialize a new instance of our class
          Set MOB = New MyOptionButton
          'Set a reference to this optionbutton
          Set MOB.OptionButton = C
          'Keep it alive in our collection
          MyOptionButtons.Add MOB
        End If
      Next
    End Sub
