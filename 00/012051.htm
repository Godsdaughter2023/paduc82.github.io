jllatham say:

Disclaimer:  this process may be slow!  How fast it runs is determined by the number of raw data entries to be examined and by the number of common strings and their individual pieces have to be compared to all of the raw data entries.

To help speed things up, I've turned off ScreenUpdating which means while the processing is going on you are not going to see anything happening on the worksheet.  In some cases the screen may even go gray -- don't worry, just be patient, have faith.

You'll need a sheet in the workbook to put your list of common+string entries.  That sheet must be selected when you run the macro and right now it expects the list in column A starting at row 2.  It will use columns B, C and D for the results.  It automatically clears old results when the macro starts.

You will need to check/edit the assigned values to the first 4 Const values to get them to point to your raw data sheet and the proper columns to use on it.  Here is a link to a working example workbook that you can download also with this same code in it.

http://www.mediafire.com/view/raf9o73587cm2wb/MatchupCommonStringsExample.xlsm

This code goes into a regular code module, and instructions for putting it into one of those can be found here; it has video and written instructions:

http://www.contextures.com/xlvba01.html#videoreg

Be sure to save the workbook as a macro enabled workbook (type .xlsm or .xlsb) after adding the code to it to preserve the code.

The code:

Sub MatchCommonStrings()
  'change these to match with the name and layout of the
  'worksheet with the data to be examined
  '******** change these to match your raw data sheet *******
  Const rawDataSheetName = "Sheet1"
  Const rawDataIDNumberColumn = "A"
  Const rawDataTextColumn = "B" ' the list we will examine
  Const rawDataFirstRow = 2 ' assumes labels in row 1
 
  'change these to change layout of the results worksheet
  'this is the sheet where you enter your common strings list
  'and where the results will appear...
  '***************
  ' this sheet must be selected when you start the macro
  '***************
  Const csColumn = "A" ' column with your common strings list
  Const csSeparator = "+" ' separate strings with this character
  Const echoColumn = "B" ' column to put 'current' common string into
  Const numColumn = "C" ' column for the ID number
  Const titleColumn = "D" ' column for potential matchups
  Const csFirstDataRow = 2 ' assumes row 1 for labels of columns MUST be 2 or greater!
 
  Dim rdWS As Worksheet ' will point to the raw data sheet
  Dim rawDataList As Range ' list of raw data entries to examine
  Dim anyRDEntry As Range ' any individual raw data text entry
  Dim commonStringsList As Range ' entire list of common strings
  Dim anyCS As Range ' individual common string entry
  Dim lastRow As Long ' can be last row, may be next row
  Dim testTitle As String ' will hold UPPERCASE of each raw title
  'this is to hold the individual common string entries during
  'the searches
  Dim indCS As Variant    'indCS = individual common strings
  Dim indPtr As Integer   'a loop pointer
 
  'set reference to the raw data sheet
  Set rdWS = ThisWorkbook.Worksheets(rawDataSheetName)
  'set reference to entire list of text entries on raw data sheet
  lastRow = rdWS.Range(rawDataTextColumn & Rows.Count).End(xlUp).Row
  If lastRow < 2 Then
    MsgBox "No raw data to work with. Quitting.", vbOKOnly, "No Raw Data"
    Set rdWS = Nothing
    Exit Sub
  End If
  Set rawDataList = rdWS.Range(rawDataTextColumn & rawDataFirstRow & _
   ":" & rawDataTextColumn & lastRow)
  'set reference to list of key words to look for
  lastRow = ActiveSheet.Range(csColumn & Rows.Count).End(xlUp).Row
  If lastRow < csFirstDataRow Then
    MsgBox "No key words list to use. Quitting.", vbOKOnly, "No Common String List"
    Set rdWS = Nothing
    Set rawDataList = Nothing
    Exit Sub
  End If
  Set commonStringsList = ActiveSheet.Range(csColumn & csFirstDataRow & _
   ":" & csColumn & lastRow)
  'we should clear out any old data here in columns B, C and D
  With ActiveSheet
    .Range(echoColumn & ":" & echoColumn).ClearContents
    .Range(echoColumn & csFirstDataRow - 1) = "Common String"
    .Range(numColumn & ":" & numColumn).ClearContents
    .Range(numColumn & csFirstDataRow - 1) = "Number"
    .Range(titleColumn & ":" & titleColumn).ClearContents
    .Range(titleColumn & csFirstDataRow - 1) = "Title"
  End With
  'this will speed things up at the expense of NOT displaying
  'the results while it is working, so the screen may remain
  'empty looking (or even go gray) while this processing takes place
  Application.ScreenUpdating = False
  For Each anyCS In commonStringsList
    If Not IsEmpty(anyCS) Then
      'UCase converts to all UPPERCASE for non-case-sensitive testing later
      indCS = Split(UCase(anyCS), csSeparator)
      'put common string into column B at next unused row of B, C & D
      lastRow = Range(titleColumn & Rows.Count).End(xlUp).Row + 1 ' next row
      Range(echoColumn & lastRow) = anyCS
      'work through all raw data titles
      For Each anyRDEntry In rawDataList
        If Not IsEmpty(anyRDEntry) Then
          testTitle = UCase(anyRDEntry)
          For indPtr = LBound(indCS) To UBound(indCS)
            If InStr(testTitle, indCS(indPtr)) > 0 Then
              'found a possible, show it
              lastRow = Range(titleColumn & Rows.Count).End(xlUp).Row + 1 ' next row
              'don't forget the associated number!
              Range(numColumn & lastRow) = rdWS.Range(rawDataIDNumberColumn & anyRDEntry.Row)
              'show the original raw title/employer entry
              Range(titleColumn & lastRow) = anyRDEntry
              'no need to examine any more move on to next raw data entry
              Exit For
            End If ' end test for match of common string to raw data entry
          Next ' end indPtr loop
        End If ' end test for empty raw title entries
      Next ' end anyRDEntry loop
    End If ' end test for empty common string cell
  Next ' end anyCS loop
'release assigned resources back to the system
  Set rawDataList = Nothing
  Set commonStringsList = Nothing
  Set rdWS = Nothing
'announce job finished
'when you click [OK] the results will appear on screen
  MsgBox "The matchups have been processed.", _
   vbOKOnly + vbInformation, "Task Completed"
End Sub

or
