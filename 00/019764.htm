Try it like this. Change the worksheet name in bold to the correct one

Sub formatcells()
Dim Lastrow As Long, x As Long
Dim wshT As Worksheet
Set wshT = Sheets("Sheet1")
Lastrow = wshT.Cells(Rows.Count, "B").End(xlUp).Row
For x = 17 To Lastrow Step 8
With wshT.Range("B" & x)
    .NumberFormat = "0000000000000"
    .HorizontalAlignment = xlLeft
End With
With wshT.Range("B" & x + 1)
    .NumberFormat = "m/d/yyyy"
    .HorizontalAlignment = xlLeft
End With
Next
End Sub
