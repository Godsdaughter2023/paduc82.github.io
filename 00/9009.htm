Sub Demo()

' Turn Off Screen Updating

Application.ScreenUpdating = False

Dim bView As Boolean, bHidd As Boolean, bMove As Boolean, Para As Paragraph, Rng As Range

' Store the field code display state and ensure field codes are displayed

bView = ActiveWindow.View.ShowFieldCodes

bHidd = ActiveWindow.View.ShowHiddenText

ActiveWindow.View.ShowFieldCodes = True

ActiveWindow.View.ShowHiddenText = True

With ActiveDocument

  ' Loop through all paragraphs

  For Each Para In .Paragraphs

    With Para.Range

      If InStr(.Text, "born") > 0 Then

        If InStr(.Text, "died") > 0 Then

          Set Rng = .Duplicate

          With Rng

            .Start = .Start + InStr(.Text, "born")

            .End = .Start + InStrRev(.Text, "died") - 1

            ' If "married" occurs between the "born" and "died" text, that indicates

            '  the presence of other content to be moved to the end of the paragraph

            If InStr(.Text, "married") > 0 Then

              bMove = True

              ' Get the range for the intervening text

              .Start = .Start + InStr(.Text, "married") - 1

              .MoveStart wdWord, -1

              .MoveStartWhile " ", -1

              If .Characters.First.Previous = " " Then .Start = .Start - 1

              .MoveEnd wdWord, -1

              .MoveEndWhile " ", -1

              If .Characters.Last = " " Then .End = .End - 1

            Else

              bMove = False

            End If

          End With

          If bMove = True Then

            With Para.Range

              ' Move our intervening range to follow the death text

              .End = .End - 1

              .Collapse wdCollapseEnd

              .FormattedText = Rng.FormattedText

              Rng.Delete

            End With

          End If

        End If

      End If

    End With

  Next

End With

ActiveWindow.View.ShowFieldCodes = bView

ActiveWindow.View.ShowHiddenText = bHidd

' Restore Screen Updating

Application.ScreenUpdating = True

End Sub
