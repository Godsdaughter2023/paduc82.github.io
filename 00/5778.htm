bill manville say:

The 'not answered' bit means that you can't use a standard MsgBox.
You will have to create a userform and display it non-modally.

So, first build a userform frmUpdateCheck containing the message and Yes and No buttons cmdYes and cmdNo.

In the module of the userform add

Private Sub cmdYes_Click()
  Replied "Yes"
End Sub

Private Sub cmdNo_Click()
  Replied "No"
End Sub

Private Sub UserForm_QueryClose(Cancel As Integer, CloseMode As Integer)
  If CloseMode <> 1 Then
    Replied "No"
  End If
End Sub

Private Sub UserForm_QueryClose(Cancel As Integer, CloseMode As Integer)
  If CloseMode <> 1 Then
    Replied "No"
  End If
End Sub

In a standard module add the following.



Dim dtAsk As Date
Dim dtClose As Date

Sub Ask()
' run this to start the sequence of asking
  dtClose = Now + TimeValue("0:1:0")
  Application.OnTime dtClose, "RunApples"
  frmUpdateCheck.Show False
End Sub

Sub RunApples()
  dtClose = 0
  Unload frmUpdateCheck
  Apples
End Sub

Sub Replied(Answer As String)
  If Answer = "Yes" Then
    ' ask again in 5 minutes
    dtAsk = Now + TimeValue("0:5:0")
    Application.OnTime dtAsk, "Ask"
    Apples
  ElseIf dtClose <> 0 Then
    Application.OnTime dtClose, "RunApples", schedule:=False
    dtClose = 0
  End If
  Unload frmUpdateCheck
End Sub

Sub StopAsking()
' call this from Workbook_BeforeClose and to stop the asking sequence
  If dtAsk <> 0 Then Application.OnTime dtAsk, "Ask", schedule:=False
  dtAsk = 0
  If dtClose <> 0 Then Application.OnTime dtClose, "RunApples", schedule:=False
  dtClose = 0
  Unload frmUpdateCheck
End Sub
