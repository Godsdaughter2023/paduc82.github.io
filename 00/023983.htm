Hello Dave Peters,

Pls check below code. This covers columns A to K, keeping complete rows & retaining column width.

Sub Extract_Data()

    Dim FilterCriteria
    Dim CurrentFileName As String, CurrentSheetName As String
    Dim NewFileName As String
    CurrentFileName = ActiveWorkbook.Name
    On Error Resume Next
    CurrentSheetName = ActiveSheet.Name
    ActiveSheet.AutoFilterMode = False
    On Error GoTo 0
    Range("A1:K1").AutoFilter
    FilterCriteria = InputBox("Enter Code")
    If FilterCriteria = "" Then Exit Sub
    FilterCriteria = Replace(FilterCriteria, "*", "")
    FilterCriteria = "*" & FilterCriteria & "*"
    Range("A1:A1").AutoFilter field:=1, Criteria1:=FilterCriteria
    
    ActiveSheet.UsedRange.Columns("A:K").SpecialCells(xlCellTypeVisible).Copy
    Sheets.Add before:=Worksheets(1)
    Range("A1").PasteSpecial Paste:=xlPasteColumnWidths
    Range("A1").PasteSpecial Paste:=xlPasteValues
    Range("A1").PasteSpecial Paste:=xlPasteFormats
    Application.CutCopyMode = False
    Range("A1").Select
    
    Dim n As Long
    Application.DisplayAlerts = False
    For n = ThisWorkbook.Sheets.count To 2 Step -1
    ThisWorkbook.Sheets(n).Delete
    Next
    Application.DisplayAlerts = True
    
    ActiveSheet.Name = CurrentSheetName
    ThisWorkbook.Save
    
End Sub
