daniel pineault say:

i use vba because I can use RegEx to perform validation.

 

Function ValidateEmail(sEmail As Variant) As Boolean
On Error GoTo Error_Handler
    Dim oRegEx As Object
   
    If IsNull(sEmail) = True Or Trim(sEmail) = vbNullString Then
        ValidateEmail = True
        GoTo Error_Handler_Exit
    End If
    Set oRegEx = CreateObject("vbscript.regexp")
    'oRegEx.Pattern = "[_a-zA-Z0-9-""'\/]+(\.[_a-zA-Z0-9-""'\/]+)*@[a-zA-Z0-9-]+(\.[a-zA-Z0-9-]+)*\.(([0-9]{1,3})|([a-zA-Z]{2,3})|(aero|coop|info|museum|name))"
    oRegEx.Pattern = "^[\w-]+(\.[\w-]+)*@([A-Za-z0-9-]+(\.[A-Za-z0-9-]+)*?\.[A-Za-z]{2,6}|(\d{1,3}\.){3}\d{1,3})(:\d{4})?$"
    ValidateEmail = oRegEx.Test(sEmail)

Error_Handler_Exit:
    On Error Resume Next
    Set oRegEx = Nothing
    Exit Function

Error_Handler:
    MsgBox "YourErrrorHandler"
    Resume Error_Handler_Exit
End Function

 

and then in the before update event of the e-mail control, I use something along the lines of

    If ValidateEmail(Me.Email) = False Then
        MsgBox "The inputted e-mail does not seem to be a valid e-mail address.", vbCritical + vbOKOnly, "Invalid Entry"
        Cancel = True
    End If

Sub InsertRow()
    Dim SWs As Worksheet, TWs As Worksheet
    Dim r As Long, LastRow As Long
    Set SWs = Worksheets("Quote")
    Set TWs = Worksheets("Invoice")
    Application.ScreenUpdating = False
    LastRow = Application.Match("Subtotal:", Columns("H"), 0)
    r = ActiveCell.Row
    If r <= 16 Or r > LastRow - 1 Then
        MsgBox "The row can not be inserted in this position"
        Exit Sub
    End If
    SWs.Rows(r).Copy
    SWs.Rows(r).Insert Shift:=xlDown
    TWs.Rows(r).Copy
    TWs.Rows(r).Insert Shift:=xlDown
    Application.CutCopyMode = False
    Application.ScreenUpdating = True
End Sub
Sub DeleteRow()
    Dim SWs As Worksheet, TWs As Worksheet
    Dim r As Long, LastRow As Long
    Set SWs = Worksheets("Quote")
    Set TWs = Worksheets("Invoice")
    Application.ScreenUpdating = False
    LastRow = Application.Match("Subtotal:", Columns("H"), 0)
    r = ActiveCell.Row
    If r <= 16 Or r > LastRow - 1 Then
        MsgBox "The row can not be deleted in this position"
        Exit Sub
    End If
    SWs.Rows(r).Delete
    TWs.Rows(r).Delete
    Application.ScreenUpdating = True
End Sub
