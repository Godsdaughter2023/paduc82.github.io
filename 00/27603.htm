andreas killer say:
The important thing when you compare strings is to ignore spelling, the easiest way is to use the line 
Option Compare Text

at the top of the module.

Please never use SELECT, SELECTION, ACTIVECELL, it is slow and error prone. Always refer to the objects directly.

Means if you have the sheet object use WITH and refer to the sheet. That is faster and save as to use SELECT and switch to the sheet.

Andreas. 

Option Explicit
Option Compare Text
Sub NameTabsToMatchDates()
  Dim xSh As Worksheet
  For Each xSh In Worksheets
    Select Case xSh.Name
      Case "Sheet2", "Sheet16"
        'Excluded
      Case Else
        'Call the code and pass the sheet object as argument
        RunCode xSh
    End Select
  Next
End Sub
Sub RunCode(ByVal Sh As Worksheet)
  'Refer to the sheet
  With Sh
    'Note: The dot in front of Range means we refer to the WITH statement
    .Range("b5").ClearContents
    .Range("b8:b21").ClearContents
    .Range("f8:f21").ClearContents
    .Range("c26").ClearContents
    .Range("b31:c31").ClearContents
    .Range("c33:c34").ClearContents
    With .Range("d2")
      If .Value = "" Then Exit Sub
      Sh.Name = VBA.Left(.Value, 10)
    End With
  End With
End Sub

Sub multiply()

Dim factor As Double
Set Rng = Range("E16:E45")
