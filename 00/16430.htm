Try this:

 

Private Sub OpenClose_Click()
    Dim i As Long
    Dm Filename As String
    Dim CellRange As String
    Dim wbkCur As Workbook
    Dim wbkNew As Workbook
    Set wbkCur = Workbooks("function book")
    For i = 1 To 2
