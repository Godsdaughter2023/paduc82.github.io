dave lett say:

Hi,

This should be most of what you're looking for

Dim oRng As Range
Dim oRngDestination As Range
Dim oDocSource As Document
Dim oDocDestination As Document

Set oDocSource = ActiveDocument
Set oDocDestination = Documents.Add

oDocSource.Activate
'''you might need to "clear"
'''any previous search parameters
'''I re-use this routine:
'''http://word.mvps.org/FAQs/MacrosVBA/ClearFind.htm
With Selection
    .HomeKey unit:=wdStory
    With .Find
        .ClearFormatting
        .text = "QUESTION*^13"
        .MatchWildcards = True
        Do While .Execute
            Set oRng = Selection.Range
            With oRng
                .Collapse Direction:=wdCollapseEnd
                Do While InStr(1, LCase(.text), "question") = 0
                    .MoveEnd unit:=wdParagraph, Count:=1
                    If .End = oDocSource.Range.End Then
                        Exit Do
                    End If
                Loop
                If .Paragraphs.Last.Range <> oDocSource.Paragraphs.Last.Range Then
                    .MoveEnd unit:=wdParagraph, Count:=-1
                End If
                .Copy
                Set oRngDestination = oDocDestination.Range
                oRngDestination.Collapse Direction:=wdCollapseEnd
                oRngDestination.PasteAndFormat wdFormatOriginalFormatting
            End With
        Loop
    End With
End With

hansv mvp say:

You could use

Private Sub CommandButton1_Click()
    Dim objWord As Object
    Dim objDoc As Object
    Dim f As Boolean
    On Error Resume Next
    Set objWord = GetObject(Class:="Word.Application")
    If objWord Is Nothing Then
        Set objWord = CreateObject(Class:="Word.Application")
        f = True
    End If
    On Error GoTo ErrHandler
    Set objDoc = objWord.Documents.Add
    objDoc.Content.Text = Me.TextBox1.Text
    objDoc.PrintOut Background:=False
ExitHandler:
    On Error Resume Next
    objDoc.Close SaveChanges:=False
    If f And Not objWord Is Nothing Then
        objWord.Quit SaveChanges:=False
    End If
    Exit Sub
ErrHandler:
    MsgBox Err.Description, vbExclamation
    Resume ExitHandler
End Sub

where CommandButton1 is a command button on the userform, and TextBox1 is the text box whose contents you want to print.
