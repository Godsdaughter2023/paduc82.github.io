ossiemac say:

Hello again Chet,

This example uses a different methodology as follows;

1.       The range of data is copied to a new added worksheet.

2.       Advanced filter is used to create a copy of the unique values to separate range.

3.       Using CountA function, the instances of each unique value is inserted in a separate column.

4.       The 2 columns are then sorted on the Count column in Descending order.

5.       Now we have a unique range that:

·         Does not have any hidden rows

·         Has the highest frequency of the values at the top.

·         Can be copied and Paste Special -> Transpose as is.

6.       The code then counts the available columns from the destination point.

7.       If the range to be copied is larger than the destination then the code truncates the range to be copied down to a size that will fit.

8.       I have commented out the line that deletes the temporary worksheet (see code between asterisk lines) so that you can see how the data is copied to the temporary worksheet. (Remove the leading single quote for your production workbook.)

9.       The method that I have used of assigning the new worksheet to a variable when the sheet is added allows that variable to be used throughout the code to address the worksheet and we never need to know the name of the worksheet from creation to deletion.

Sub TransposeUniqueCells()
    Dim wsTemp As Worksheet
    Dim rngTemp As Range
    Dim lngMaxRow As Long
    Dim lngMaxDestin As Long
    Dim rngSort As Range
    Dim rngUnique As Range
    Dim lngCellsToCopy As Long
    Dim r As Long
   
    Application.ScreenUpdating = False      'Optional code
   
    Set wsTemp = Sheets.Add(After:=Sheets(Sheets.Count))
   
    With Sheets("Source")
        On Error Resume Next
        .ShowAllData
        On Error GoTo 0
   
        .Range(.Cells(1, "A"), _
            .Cells(.Rows.Count, "A").End(xlUp)).Copy _
            Destination:=wsTemp.Cells(1, 1)
    End With
   
    With wsTemp
      
        Set rngTemp = .Range(.Cells(1, "A"), .Cells(.Rows.Count, "A").End(xlUp))
       
        rngTemp.AdvancedFilter Action:=xlFilterCopy, _
                CopyToRange:=.Cells(1, "E"), _
                Unique:=True
       
        Set rngUnique = .Range(.Cells(1, "E"), .Cells(Rows.Count, "E").End(xlUp))
         .Cells(1, "F") = "Counts"
       
        For r = 2 To rngUnique.Rows.Count
            .Cells(r, "F") = WorksheetFunction.CountIf(rngTemp, .Cells(r, "E"))
        Next r
       
        Set rngSort = .Range(Cells(1, "E"), .Cells(.Rows.Count, "F").End(xlUp))
       
        rngSort.Sort Key1:=.Range("F2"), _
                    Order1:=xlDescending, _
                    Key2:=Range("E2"), _
                    Order2:=xlAscending, _
                    Header:=xlYes, _
                    OrderCustom:=1, _
                    MatchCase:=False, _
                    Orientation:=xlTopToBottom, _
                    DataOption1:=xlSortNormal, _
                    DataOption2:=xlSortNormal
       
        Set rngUnique = .Range(.Cells(2, "E"), _
                        .Cells(.Rows.Count, "E").End(xlUp)) _
                        .SpecialCells(xlCellTypeVisible)
                       
    End With
   
    With Sheets("Summary")
        'Number of cells to copy
        lngCellsToCopy = rngUnique.Cells.Count
       
        'Count available columns in destination
        lngMaxDestin = .Range(.Cells(6, "J"), _
                    .Cells(6, .Columns.Count)).Cells.Count
                   
        'Test if cells to copy > destination columns available
        If lngCellsToCopy > lngMaxDestin Then
            'Range to copy too large for destination
            'Reduce rngUnique to fit the destination
            With wsTemp
                'lngMaxDestin + 1 is because starts at row 2 _
                 and need extra row to lngMaxDesin value
                Set rngUnique = .Range(.Cells(2, "E"), _
                        .Cells(lngMaxDestin + 1, "E")) _
                        .SpecialCells(xlCellTypeVisible)
                       
                MsgBox "Range copied will be truncated to fit available columns."
               
            End With
