graham mayor say:
You could use a macro in Outlook. The following will create a new contact for each recipient in a selected message and copy the list created to the clipboard. You can paste that into (e.g.) a Word document for reference.

Sub AddContacts()
Dim olContact As Outlook.ContactItem
Dim olMsg As Outlook.MailItem
Dim oName As Recipient
Dim strName As String
Dim i As Long
Dim dFname As DataObject
Set dFname = New DataObject
strName = ""
On Error GoTo UserCancelled:
Set olMsg = ActiveExplorer.Selection.item(1)
For i = 1 To olMsg.Recipients.Count
    Set oName = olMsg.Recipients(i)
    strName = strName & oName & vbTab & oName.Address
    If i < olMsg.Recipients.Count Then strName = strName & vbCr
    Set olContact = CreateItem(olContactItem)
    With olContact
        .FullName = oName
        .FileAs = oName
        .Email1Address = oName.Address
        .Save
    End With
Next i
dFname.SetText strName
dFname.PutInClipboard
Exit Sub
UserCancelled:
Set olContact = Nothing
Set dFname = Nothing
End Sub
