This one worked for me

DOMDocument50 xmldoc = new DOMDocument50();
 XmlNamespaceManager nm = GetNamespaceManager();
 xmldoc.setProperty("SelectionNamespaces", "xmlns:my='http://schemas.microsoft.com/office/infopath/2003/myXSD/2010-02-08T11-28-21'     xmlns:xsi='http://www.w3.org/2001/XMLSchema-instance'");

 xmldoc.async= false;
 xmldoc.validateOnParse = false;
 xmldoc.loadXML(xmlNode.OuterXml);
 domNode = (IXMLDOMNode)xmldoc;
