i have a date in a cell. if it is older than 7 years from the current date i need it to turn red
ashish mathur say:

Hi,

 

Try this

 

1. Select the cell (say B4) with the date and go to Format > Conditional Formatting > Formula Is

2. In the formula bar there, enter this formula =B4<=DATE(YEAR(TODAY())-7,MONTH(TODAY()),DAY(TODAY()))

3. Click on Format and select any colour

4. Click on OK

 

The DATE(YEAR(TODAY())-7,MONTH(TODAY()),DAY(TODAY())) can also be replaces with EDATE(TODAY(),-84) but for this you will have to install the Analysis Toolpak from Tools > Addins

Sub Initialize()

  MyCollection.Add 13
  MyCollection.Add 16
  Initialized = True
End Sub

Sub Quiz()
Dim Inum As Integer
Dim MyAnswer As String
  If Initialized = False Then
    Initialize
  End If
  If AllDone = True Then
    ActivePresentation.SlideShowWindow.View.GotoSlide (5)
    Exit Sub
  End If
  MyAnswer = InputBox(Prompt:="Question?", Title:="Question group name")
  If MyAnswer = "Answer" Then
    MsgBox MyCollection.Count
    If MyCollection.Count > 1 Then
      Inum = Int(MyCollection.Count * rnd + 1)
      ActivePresentation.SlideShowWindow.View.GotoSlide (MyCollection(Inum))
      MyCollection.Remove (Inum)
    Else
      ActivePresentation.SlideShowWindow.View.GotoSlide (MyCollection(1))
      MyCollection.Remove (1)
      AllDone = True
    End If
  Else
    MsgBox "There is more hint"
    ActivePresentation.SlideShowWindow.View.GotoSlide 7
  End If
End Sub
