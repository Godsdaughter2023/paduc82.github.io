Use an array formula, entered using Ctrl-Shift-Enter, based on your original data, like

=SLOPE(IF(A2:A200>0.02,IF(A2:A200<0.05,B2:B200)),IF(A2:A200>0.02,IF(A2:A200<0.05,A2:A200)))

Dim iRow

Sub ListFiles()
    iRow = 13
    Call ListMyFiles(Range("C7"), Range("C8"))
End Sub

Sub ListMyFiles(mySourcePath, IncludeSubfolders)
    Dim d As Date
    d = Date
    Sheet1.Rows(iRow & ":" & Sheet1.Rows.Count).ClearContents
    Set MyObject = CreateObject("Scripting.FileSystemObject")
    Set mySource = MyObject.GetFolder(mySourcePath)
    On Error Resume Next
    For Each myfile In mySource.Files
        If d - myfile.DateLastModified < 365 Then
            iCol = 2
            Cells(iRow, iCol).Value = myfile.Path
            iCol = iCol + 1
            Cells(iRow, iCol).Value = myfile.Name
            iCol = iCol + 1

            iCol = iCol + 1

            iRow = iRow + 1
        End If
    Next
    If IncludeSubfolders Then
        For Each mySubFolder In mySource.SubFolders
