=SUMPRODUCT((Account=$A81)*(Company=INDIRECT($B$3) Col)*(INDEX('YTD TB'!$G$4:$AU$1139,,MATCH(C$4,'YTD TB'!$G$3:$AU$3,0))-IF(MONTH(C$4)>1,INDEX('YTD TB'!$G$4:$AU$1139,,MATCH(EOMONTH(C$4,-1),'YTD TB'!$G$3:$AU$3,0)))))

'YTD TB' $G$4:$AU$1139 = raw data

'YTD TB' $G$3:$AU$3 = Months header for raw data
