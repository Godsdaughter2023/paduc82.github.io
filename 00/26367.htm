Sub GetString()
    Dim InString As String

    msg = "Do You Want to Add a Sheet Y/N" & Chr(13) _
            & "If No, Column A Will be Overwritten"

    ans = MsgBox(msg, vbQuestion + vbYesNoCancel)
    Select Case ans
        Case vbYes
            Sheets.Add
        Case vbNo
            GoTo carryon
        Case vbCancel
            Cancel = True
            Exit Sub
    End Select
carryon:
    InString = InputBox("Enter text to permute:")
    If Len(InString) < 2 Then Exit Sub
    If Len(InString) > 8 Then                                  '9 digits for xl2007 and later
        MsgBox "Too many permutations!"
        Exit Sub
    Else
        StartTime = Timer
        ActiveSheet.Columns(1).Clear
        CurrentRow = 1
        Call GetPermutation("", InString)
    End If
    MsgBox Timer - StartTime
End Sub

Sub GetPermutation(x As String, y As String)
'   The source of this algorithm is unknown

    Dim i As Integer, j As Integer
    j = Len(y)
    If j < 2 Then
        Cells(CurrentRow, 1) = x & y
        CurrentRow = CurrentRow + 1
    Else

        For i = 1 To j
            Call GetPermutation(x + Mid(y, i, 1), _
                    Left(y, i - 1) + Right(y, j - i))
        Next
    End If
End Sub

I took the challenge to improve it to 10 digits and speed it up.

Following code goes up to 10 characters and does it in less than 30 seconds.

Dim CurrentRow
Dim Permut As Variant
Dim CurrentCol
'GetString and GetPermutations must be run from a workbook, not add-in
'Copy both including Dim CurrentRow
'Max 8 digits excel 2003
''' Gord Dibbon posted

'  James Cone wrote:
'  Posted by Myrna Larson
'  July 25, 2000
'  Microsoft.Public.Excel.Misc
'  Subject:  Combin

Sub GetString()
   Dim InString As String

   msg = "Do You Want to Add a Sheet Y/N" & Chr(13) _
      & "If No, Column A Will be Overwritten"

   ans = MsgBox(msg, vbQuestion + vbYesNoCancel)
   Select Case ans
      Case vbYes
         Sheets.Add
      Case vbNo
         GoTo carryon
      Case vbCancel
         Cancel = True
         Exit Sub
   End Select
carryon:
   Application.ScreenUpdating = False
   InString = InputBox("Enter text to permute:")
   If Len(InString) < 2 Then Exit Sub
   If Len(InString) > 10 Then
      MsgBox "Too many permutations!"
      Exit Sub
   Else
      StartTime = Timer
      Permut = Empty
      ActiveSheet.Columns(1).Clear
      CurrentRow = 1: CurrentCol = 1
      With WorksheetFunction
         ReDim Permut(1 To .Min(65535, .Permut(Len(InString), Len(InString))), _
            1 To .Max(1, .Permut(Len(InString), Len(InString)) \ 65535 + 1))
      End With
      
      Call GetPermutation("", InString)
   End If
   
   Range("$A$1").Resize(UBound(Permut, 1), UBound(Permut, 2)).Select
   Selection = Permut
   MsgBox Timer - StartTime
   Application.ScreenUpdating = True
End Sub

Sub GetPermutation(x As String, y As String)
   '   The source of this algorithm is unknown
      
   Dim i As Integer, j As Integer
   If CurrentRow > 65535 Then
      CurrentRow = 1: CurrentCol = CurrentCol + 1
   End If
   
   j = Len(y)
   If j < 2 Then
      Permut(CurrentRow, CurrentCol) = x & y
      CurrentRow = CurrentRow + 1
   Else

      For i = 1 To j
         Call GetPermutation(x + Mid(y, i, 1), _
            Left(y, i - 1) + Right(y, j - i))
      Next
   End If
End Sub

#1 - Open your source contact folder (the one with "This Computer Only" in the description

#2 - Press CTRL+A to select all the items
