Option Explicit

Private Const PROCESS_QUERY_INFORMATION = 1024
Private Const PROCESS_VM_READ = 16
Private Const PROCESS_TERMINATE = &H1
Private Const MAX_PATH = 260

Private Declare Function EnumProcesses Lib "PSAPI.DLL" ( _
    ByRef lpidProcess As Long, ByVal CB As Long, ByRef cbNeeded As Long) As Long
Private Declare Function OpenProcess Lib "Kernel32.dll" ( _
    ByVal dwDesiredAccessas As Long, ByVal bInheritHandle As Long, _
    ByVal dwProcId As Long) As Long
Private Declare Function TerminateProcess Lib "kernel32" ( _
    ByVal hProcess As Long, ByVal uExitCode As Long) As Long
Private Declare Function CloseHandle Lib "Kernel32.dll" ( _
    ByVal Handle As Long) As Long
Private Declare Function EnumProcessModules Lib "PSAPI.DLL" ( _
    ByVal hProcess As Long, ByRef lphModule As Long, ByVal CB As Long, _
    ByRef cbNeeded As Long) As Long
Private Declare Function GetModuleFileNameExA Lib "PSAPI.DLL" ( _
    ByVal hProcess As Long, ByVal hModule As Long, ByVal ModuleName As String, _
    ByVal nSize As Long) As Long
Private Declare Function GetExitCodeProcess Lib "kernel32" ( _
    ByVal hProcess As Long, lpExitCode As Long) As Long

Sub KillProcess(ByVal ExeName As String)
  'http://msdn.microsoft.com/en-us/library/windows/desktop/ms682623%28v=vs.85%29.aspx
  Dim ProcessIDs() As Long
  Dim Modules(0 To 200) As Long
  Dim cbNeeded As Long, i As Long, Result As Long
  Dim hProcess As Long
  Dim ProcessName As String
 
  'Get a list of all running processes
  ReDim ProcessIDs(0 To 1023) As Long
  If EnumProcesses(ProcessIDs(0), 1024 * 4, cbNeeded) = 0 Then Exit Sub
  'Loop thru each process
  For i = 0 To cbNeeded / 4
    'Get a handle to the process
    hProcess = OpenProcess(PROCESS_QUERY_INFORMATION Or PROCESS_VM_READ, 0, ProcessIDs(i))
    If hProcess = 0 Then GoTo NextProcess
    'Get an array of the module handles for the specified process
    If EnumProcessModules(hProcess, Modules(0), UBound(Modules), cbNeeded) = 0 Then _
      GoTo NextProcess
    'Get the name of the process
    ProcessName = Space(MAX_PATH + 1)
    Result = GetModuleFileNameExA(hProcess, Modules(0), ProcessName, MAX_PATH)
    ProcessName = Left$(ProcessName, Result)

    'Found?
    If InStr(1, ProcessName, ExeName, vbTextCompare) > 0 Then
      'Close the process
      CloseHandle hProcess
      'Try to terminate
      hProcess = OpenProcess(PROCESS_TERMINATE, 0&, ProcessIDs(i))
      If hProcess = 0 Then GoTo NextProcess
      TerminateProcess hProcess, 0&
      CloseHandle hProcess
      Exit Sub
    End If

NextProcess:
    If hProcess <> 0 Then CloseHandle hProcess
  Next
End Sub

Use this UDF in VBA. Pass to it the cell or value you are looking up, the range to look in and which column to return.

It will remove any duplicates.

So for your example, enter this in a cell:

=MYVLOOKUP(B2,'PAYMENT METHOD'!A2:C5,3)

Function MYVLOOKUP(pValue As String, pWorkRng As Range, pIndex As Long)

Dim arr() As String
Dim x As Long
Dim rng As Range
Dim xResult As String

xResult = ""
For Each rng In pWorkRng
    If rng = pValue Then
            If x = 0 Then
                ReDim Preserve arr(x)
                arr(x) = rng.Offset(0, pIndex - 1)
                x = x + 1
            Else
                If UBound(Filter(arr, rng.Offset(0, pIndex - 1))) <> 0 Then 'not already in array
                    ReDim Preserve arr(x)
                    arr(x) = rng.Offset(0, pIndex - 1)
                    x = x + 1
                End If
            End If
    End If
Next

For x = LBound(arr) To UBound(arr)
    xResult = xResult & ", " & arr(x)
Next x

xResult = Right(xResult, Len(xResult) - 2)
MYVLOOKUP = xResult

End Function
