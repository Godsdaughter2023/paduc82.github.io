e.j. gun say:

The code below presents one approach.  Create a new folder named "Temp" in your main Outlook folder (same one as your Inbox).  Then select a few emails from your Inbox.  They will be copied, moved to the Temp folder, converted to plain text, and opened.

You can link this Sub to a button.  Then select emails, push the button, and Voilà!

Hope this helps,

Eric

Private Sub OpenAsPlainText()
    Dim i As Long
    '
    Dim myNameSpace As Outlook.NameSpace
    '
    Dim oApp As New Outlook.Application
    Dim oExp As Outlook.Explorer
    Dim oSel As Outlook.Selection   ' You need a selection object for getting the selection.
    Dim oItem As Object   ' You don't know the type yet...
    Dim nItem As Outlook.MailItem
    Dim myFolder As Outlook.Folder
    Dim myNewFolder As Outlook.Folder
   
    Set oExp = oApp.ActiveExplorer  ' Get the ActiveExplorer.
    Set oSel = oExp.Selection       ' Get the selection.
    Set myNameSpace = oApp.GetNamespace("MAPI")
    Set myNewFolder = myNameSpace.GetDefaultFolder(olFolderInbox).Parent.Folders("Temp")
   
    For i = 1 To oSel.Count         ' Loop through all the currently .selected items
        Set oItem = oSel.Item(i)    ' Get a selected item.
        If (oItem.Class = olMail) Then
            Set nItem = oItem.Copy
            nItem.Move myNewFolder
            nItem.BodyFormat = olFormatPlain
            nItem.Display
        End If
    Next i
End Sub
