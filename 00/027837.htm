ossiemac say:

The following code works for me in Win 7, Office 2010.

I suspect that your columns may not all be the same width on each sheet and you might be positioning based on the top and left. The following code gets the top and left position of the rows and columns on each worksheet.

Sub InsertPicture()
   
    Dim strFileName As String
    Dim ws As Worksheet
    Dim dblLeft As Double
    Dim dblTop As Double
    Dim dblHeight As Double
    Dim shpPic As Shape
   
    strFileName = ThisWorkbook.Path & "\My Picture.jpg"
    For Each ws In Worksheets
        With ws
            dblLeft = .Columns("M").Left    '
            dblTop = .Rows(1).Top
            dblHeight = .Rows(10).Top - dblTop
           
            'Note: -1 parameters in following for Width and Height _
             are to maintain the aspect ratio.
           Set shpPic = .Shapes.AddPicture _
                    (Filename:=strFileName, _
                    LinkToFile:=msoFalse, _
                    SaveWithDocument:=msoTrue, _
                    Left:=dblLeft, _
                    Top:=dblTop, _
                    Width:=-1, _
                    Height:=-1)
           
            Application.ScreenUpdating = True
            'If re-sizing only re-set the Height or the Width (Not both) to retain aspect ratio
            'Otherwise set AspectRatio to false to allow both Width and Height to be set
            With shpPic
                '.LockAspectRatio = msoFalse 'Use this line if want to set both Width and Height
                .Height = dblHeight
            End With
        End With
    Next ws
   
End Sub
