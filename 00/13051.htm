bill manville say:

The elements are:

- identifying the files to be processed - presumably all files in a folder?
- creating a worksheet for the copied data  Dim WS As Worksheet

- copying the data.

- identifying the files to be processed, e.g.:
Sub ProcessFiles()
  Dim stFolder As String
  Dim stFile As String
  Dim WB As Workbook
  stFolder = ThisWorkbook.Path & "\Returns"
  stFile = Dir(stFolder & "\*.xls*)
  Do Until stFile = "" ' no more
    Set WB = Workbooks.Open(stFolder & "\" & stFile)
    ProcessFile WB
    WB.Close False
    stFile = Dir()  ' next one
  Loop
End Sub

Sub ProcessFile(FromWB As Workbook)
'- creating a worksheet for the copied data 

  Dim WS As Worksheet
  Dim R As Range

  ' add a worksheet
  Set WS =  ThisWorkbook.Worksheets.Add
  WS.Name = Left(FromWB.Name,InStr(FromWB.Name,".")-1)  ' set tab name from filename

  ' suppose the tables are to be pasted side by side
  Set R = WS.Range("A1")

  '- copy the data - change sheet name and range names
  FromWB.Sheets("Whatever").Range("FirstTable").Copy
  ' assuming you don't want formulas copied, just values and formats.
  R.PasteSpecial xlValues
  R.PasteSpecial xlFormats 
  ' name the copied data

  R.CurrentRegion.Name = "'" & WS.Name & "'!FirstTable"

  ' move right and leave a column empty
  Set R = R.End(xlToRight).Offset(,2)

  ' repeat for other ranges

  FromWB.Sheets("Whatever").Range("SecondTable").Copy
  R.PasteSpecial xlValues
  R.PasteSpecial xlFormats 
  R.CurrentRegion.Name = "'" & WS.Name & "'!SecondTable"

  ' move right and leave a column empty
  Set R = R.End(xlToRight).Offset(,2)
' etc

End Sub

Hope this gives you sufficient of a start to complete the task - if not come back with what you have done and what is needed to be different.

bill manville say:

Assuming you have a cell named CurMonth containing the month name and CurYear containing the year and these cells are in Sheet1 of the workbook containing the macro:

Sub ProcessFiles()
Dim stFolder As String
   Dim stFile As String
   Dim WB As Workbook

   Dim stFileGen As String
   stFileGen = "_" & ThisWorkbook.Sheets("Sheet1").Range("CurMonth").Value & "_" & ThisWorkbook.Sheets("Sheet1").Range("CurYear").Value

   stFolder = ThisWorkbook.Path
   stFile = Dir(stFolder & "\*" & stFileGen & ".xlsm")
   Do Until stFile = "" ' no more
      If stFile<>ThisWorkbook.Name Then
       Set WB = Workbooks.Open(stFolder & "\" &stFile)
       ProcessFile WB
       WB.Close False
      End If
      stFile = Dir()  ' next one
   Loop
End Sub

and (assuming X in the filename does not contain a  "_")

   WS.Name = Left(FromWB.Name, InStr(FromWB.Name, "_") - 1) ' set tab name from filename before date
