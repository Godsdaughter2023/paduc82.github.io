You can disable the screen with the API LockWindowUpdate from the windows kernel.

Andreas.

Option Explicit

Private Declare Function FindWindow Lib "user32" Alias "FindWindowA" ( _
    ByVal lpClassName As String, ByVal lpWindowName As String) As Long
Private Declare Function LockWindowUpdate Lib "user32" (ByVal hWndLock As Long) As Long
Private Declare Function UpdateWindow Lib "user32" (ByVal hWnd As Long) As Long

Sub Example_ScreenUpdating()
  ScreenUpdating(FindWindowHandle(Application)) = True 'or False
End Sub

Property Let ScreenUpdating(Optional ByVal hWnd As Long, ByVal State As Boolean)
  If Not State Then
    LockWindowUpdate hWnd
  Else
    LockWindowUpdate False
    UpdateWindow hWnd
  End If
End Property

Function FindWindowHandle(ByVal App As Object, Optional ByVal Caption As String) As Long
  If App Is Nothing Then
    FindWindowHandle = FindWindow(vbNullString, Caption)
  Else
    On Error Resume Next
    Select Case App.Name
      Case "Microsoft Access"
        'Caption = App.Name
        FindWindowHandle = FindWindow("OMAIN", Caption)
      Case "Microsoft Excel"
        'Caption = App.Caption
        FindWindowHandle = FindWindow("XLMAIN", Caption)
      Case "Microsoft PowerPoint"
        Select Case Val(Application.Version)
          Case 8
            FindWindowHandle = FindWindow("PP97FrameClass", Caption)
          Case 9 To 12
            'Caption = App.Caption & " - [" & App.ActiveWindow.Caption & "]"
            FindWindowHandle = FindWindow("PP" & Val(Application.Version) & "FrameClass", _
              Caption)
          Case Else
            'Caption = App.Caption
            FindWindowHandle = FindWindow("PPTFrameClass", Caption)
        End Select
      Case "Microsoft Word"
        'Caption = App.ActiveWindow.Caption & " - " & App.Caption
        FindWindowHandle = FindWindow("OPUSAPP", Caption)
      Case "Outlook"
        'Caption = Application.ActiveExplorer.Caption
        FindWindowHandle = FindWindow("rctrl_renwnd32", Caption)
      Case Else
        'Userform
        'Caption = App.Caption
        If Val(Application.Version) >= 9 Then
          FindWindowHandle = FindWindow("ThunderDFrame", Caption)
        Else
          FindWindowHandle = FindWindow("ThunderXFrame", Caption)
        End If
    End Select
  End If
End Function

Try the following on a subset of the msg files copied to a new folder. You will see two prompts for folders - the first is the folder containing the msg files, the second the folder to store the attachments. I have added the current time and date in minutes and seconds to the extracted filenames, but as the process is fairly fast, with several items processed a second, I wonder whether this is the best option? Existing files of the same name will be overwritten without warning or backup.

Don't use the desktop folder or a sub folder of the desktop for either folder locations as I have no error trapped the use of desktop and it will crash.

Allow the macro to run until you see the completed message. With luck it should do the job during your lunch hour :)

Sub SaveAttachmentsFromMSG()
Dim olItem As MailItem
Dim SH As Shell32.Shell
Dim FldrIn As Shell32.Folder
Dim FldrOut As Shell32.Folder
Dim strFilesFldr As String
Dim strSaveFldr As String
Dim olAttach As Attachment
Dim strFilename As String

On Error GoTo Cleanup
Set SH = New Shell32.Shell

Set FldrIn = SH.BrowseForFolder(0, "Select the folder containing the msg files", &H400)
If FldrIn Is Nothing Then Exit Sub
strFilesFldr = FldrIn.Items.Item.Path & "\"

Set FldrOut = SH.BrowseForFolder(0, "Select the folder to store the attachments", &H400)
If FldrOut Is Nothing Then Exit Sub
strSaveFldr = FldrOut.Items.Item.Path & "\"

strFilename = Dir$(strFilesFldr & "*.msg")

While Len(strFilename) <> 0
