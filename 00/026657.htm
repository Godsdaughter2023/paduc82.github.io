graham mayor say:

Assuming a dropdown list content control titled "Path", containing at least the values A, B & C then put the following code in the ThisDocument module of the document and save it as macro enabled. CreateFolders will create the named folder, based on the selection from the content control, when you leave the content control. If the folder exists the function does nothing.

Option Explicit

Private Sub Document_ContentControlOnExit(ByVal ContentControl As ContentControl, Cancel As Boolean)
    If ContentControl.Title = "Path" Then
        Select Case ContentControl.Range.Text
            Case "A": CreateFolders ("C:\Path A\")
            Case "B": CreateFolders ("C:\Path B\")
            Case "C": CreateFolders ("C:\Path C\")
            Case Else: CreateFolders ("C:\Path D\") 'The Default Path
        End Select
    End If
End Sub

Private Sub CreateFolders(strPath As String)
'A Graham Mayor/Greg Maxey AddIn Utility Macro
Dim oFSO As Object
Dim lngPathSep As Long
Dim lngPS As Long
    If Right(strPath, 1) <> "\" Then strPath = strPath & "\"
    lngPathSep = InStr(3, strPath, "\")
    If lngPathSep = 0 Then GoTo lbl_Exit
    Set oFSO = CreateObject("Scripting.FileSystemObject")
    Do
        lngPS = lngPathSep
        lngPathSep = InStr(lngPS + 1, strPath, "\")
        If lngPathSep = 0 Then Exit Do
        If Len(Dir(Left(strPath, lngPathSep), vbDirectory)) = 0 Then Exit Do
    Loop
    Do Until lngPathSep = 0
        If Not oFSO.FolderExists(Left(strPath, lngPathSep)) Then
            oFSO.CreateFolder Left(strPath, lngPathSep)
        End If
        lngPS = lngPathSep
        lngPathSep = InStr(lngPS + 1, strPath, "\")
    Loop
lbl_Exit:
    Set oFSO = Nothing
    Exit Sub
End Sub
