The following code will select the worksheet by its name irrespective of the order of the sheets.

 

Private Sub Workbook_Open()
    Dim ws As Worksheet
    Dim strMonth As String
   
    For Each ws In Sheets
        ws.Tab.ColorIndex = -4142
    Next
   
    strMonth = Format(Date, "mmm")
   
    Sheets(strMonth).Select
    ActiveSheet.Tab.ColorIndex = 3
End Sub
