Sub AddScore(oShp As Shape)
Dim score As Long
Dim amount As Long
    With ActivePresentation.Designs(1).SlideMaster.Shapes("ScoreBox")
    With .TextFrame.TextRange
    If .Text = "" Then .Text = "0"

    End With

    End With
