jeffgoodman say: Hi there

I am pretty sure I have got this to work with the below code.

I made the following changes which are bolded:
  1. Commented out the "Report Start" line because this didn't contribute to the code.
  2. Changed While to Loop Until (I was experimenting trying to stop the intermittent run time error, not sure if this does it or not)
  3. Repeated the loop after setting the frm object.  I am fairly certain this has prevented the run time error.
  4. Reused the lst object to call the webpage's populatereporttype function.  I got this from viewing the source code on the page and finding the onchange event for the "SelectReport" object.
  5. Kill the objects at the end of the code
Sub Open_IE_New()
Dim ie As Object
Dim frm As Object
Dim lst As Object
    Set ie = CreateObject("internetexplorer.application")
    ie.Visible = True
    'ie.navigate "Report Start"
    ie.navigate "http://ets.aeso.ca/ets_web/docroot/Market/Reports/HistoricalReportsStart.html"
    'While ie.busy
    '    DoEvents
    'Wend
    Do
    Loop Until Not (ie.busy)
    'Pass variables to the webpage
    Set frm = ie.document.frames("report_nav")
    Do
    Loop Until Not (ie.busy)
    Set lst = frm.document.getElementById("SelectReport")
    'lst.selectedindex = 10
    'Do
    'Loop Until Not (ie.busy)
    lst.Value = "Market/Reports/HistoricalPoolPriceReportServlet"
    Set lst = frm.Function("populateReportType(document.reportForm, getCurrReportObj())")
    lst
    
Set ie = Nothing
Set frm = Nothing
Set lst = Nothing
    
    'ie.document.getElementById("SelectFormat").Value ="csv"
    'then execute the ‘OK Button’
End Sub
