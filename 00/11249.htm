Private Sub Export_Click()
    If Me.Dirty Then
        Me.Dirty = False
    End If

    If Me.AddedToOutlook = True Then
        DoCmd.GoToRecord , , acNext
        Exit Sub
    Else
        Dim olApp As Object
        Dim olappt As Object
        End If

            Set olApp = CreateObject("Outlook.Application")
        Else
            Set olApp = GetObject(, "Outlook.Application")
        End If

        Set olappt = olApp.CreateItem(1)
  
        With olappt

        .End = Nz(Me.Date, "") & " " & Nz(Me.End_Time, "")
        .Subject = Nz(Me.Instructors, "") & " " & Nz(Me.Activity, "")

        End With

    Me.AddedToOutlook = True
    
    If Me.Dirty Then

    End If
    DoCmd.GoToRecord , , acNext

Function isAppThere(appName) As Boolean
    On Error Resume Next

    Dim objApp As Object

    isAppThere = True

    Set objApp = GetObject(, appName)
    If Err.Number <> 0 Then isAppThere = False

End Function

daniel pineault say:

I'd probably do something more like

Private Sub Export_Click()
    On Error GoTo Error_Handler
    Dim olApp                 As Object
    Dim olappt                As Object

    If Me.Dirty Then Me.Dirty = False

    If isAppThere("Outlook.Application") = False Then
        Set olApp = CreateObject("Outlook.Application")
    Else
        Set olApp = GetObject(, "Outlook.Application")
    End If

    With Me.Recordset
        If .RecordCount <> 0 Then
            .MoveFirst
            Do While Not .EOF
                Set olappt = olApp.CreateItem(1)
                olappt.Start = Nz(Me.Date, "") & " " & Nz(Me.Start_Time, "")
                olappt.End = Nz(Me.Date, "") & " " & Nz(Me.End_Time, "")
                olappt.Subject = Nz(Me.Instructors, "") & " " & Nz(Me.Activity, "")
                olappt.Body = Nz(Me.Group_Name, "")
                olappt.ReminderSet = False
                olappt.Save
                Set olappt = Nothing
                .AddedToOutlook = True
                .MoveNext
            Loop
        End If
    End With

    If Me.Dirty Then Me.Dirty = False

Error_Handler_Exit:
    On Error Resume Next
    Set olappt = Nothing
    Set olApp = Nothing
    Exit Sub

Error_Handler:
    MsgBox "The following error has occured" & vbCrLf & vbCrLf & _
           "Error Number: " & Err.Number & vbCrLf & _
           "Error Source: Export_Click" & vbCrLf & _
           "Error Description: " & Err.Description _
           , vbOKOnly + vbCritical, "An Error has Occured!"
    Resume Error_Handler_Exit
End Sub
