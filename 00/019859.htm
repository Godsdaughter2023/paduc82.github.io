bernie deitrick say:

Select the files of interest when the dialog appears - or re-write to work on all files in a directory...

Sub AutoFitSelectedFiles()
    Dim FileArray As Variant
    Dim i As Integer
    Dim wkBk As Workbook
    Dim wkSht As Worksheet

    FileArray = Application.GetOpenFilename(MultiSelect:=True)
    If IsArray(FileArray) Then
        Application.ScreenUpdating = False
        For i = LBound(FileArray) To UBound(FileArray)
            Application.ScreenUpdating = False
            Set wkBk = Workbooks.Open(FileArray(i))
            For Each wkSht In wkBk.Worksheets

                wkSht.Cells.EntireColumn.AutoFit
                wkSht.Cells.EntireRow.AutoFit
            Next wkSht
            wkBk.Close True
        Next i
        Application.ScreenUpdating = True
    Else
        MsgBox "You clicked cancel"
    End If
End Sub

Change the path to your folder

Sub AutoFitSelectedDirectory()
    Dim strFile As String
    Dim i As Integer
    Dim wkBk As Workbook
    Dim wkSht As Worksheet
    Dim strPath As String
    Dim lngR As Long

    strPath = "C:\Documents\Excel\"

    strFile = Dir(strPath & "*.xls*")
    Application.ScreenUpdating = False
    While strFile <> ""
        Application.ScreenUpdating = False
        Set wkBk = Workbooks.Open(strPath & strFile)
        For Each wkSht In wkBk.Worksheets
            wkSht.Cells.EntireColumn.AutoFit
            wkSht.Cells.EntireRow.AutoFit
        Next wkSht
        With ThisWorkbook.Worksheets(1)
            lngR = .Cells(.Rows.Count, "A").End(xlUp).Row + 1
            .Cells(lngR, "A").Value = Date
            .Cells(lngR, "B").Value = Time
            .Cells(lngR, "C").Value = strPath & strFile
        End With
        wkBk.Close True
        strFile = Dir()
    Wend
    Application.ScreenUpdating = True
End Sub
