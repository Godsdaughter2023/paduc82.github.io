macropod say:

Word doesn't have a column-based table-splitting function. You have two options:

1. Rotate the table and its contents, so you can continue using the row-split method (though that probably becomes unnecessary once you've done the roatation); or

2. Use a macro like the following (note the comments in the code re column-1 replication):

Sub TableSplit()
Application.ScreenUpdating = False
Dim oLeftMargin As Single, oRightMargin As Single, oGutter As Single
Dim oPageWidth As Single, oPrintWidth As Single, oTableWidth As Single
Dim oTable As Table, oCounter As Long, i As Long, j As Long
With ActiveDocument
  If .Tables.Count = 0 Then Exit Sub
  oCounter = 1
Restart:
  For j = oCounter To .Tables.Count
    Set oTable = .Tables(j)
    With .Sections(oTable.Range.Information(wdActiveEndSectionNumber)).PageSetup
      oLeftMargin = .LeftMargin
      oRightMargin = .RightMargin
      oGutter = .Gutter
      oPageWidth = .PageWidth
    End With
    oPrintWidth = oPageWidth - oLeftMargin - oRightMargin - oGutter
    With .Tables(j)
      .AllowAutoFit = False
      .PreferredWidth = 0
    End With
    oTableWidth = 0
    If oTable.Columns.Count < 3 Then GoTo SkipTable
    For i = 1 To oTable.Columns.Count
      oTableWidth = oTableWidth + oTable.Columns(i).Width
      If oTableWidth > oPrintWidth + 1 Then
        oTableWidth = oTableWidth - oTable.Columns(i).Width
        oTable.Columns(i).Select
        With Selection
          .MoveRight Unit:=wdCharacter, Count:=oTable.Columns.Count - i + 1, Extend:=wdExtend
          .Copy
          .Columns.Delete
          .MoveDown Unit:=wdLine, Count:=oTable.Rows.Count
          .InsertParagraph
          .MoveDown Unit:=wdLine, Count:=1
          .Paste
        End With
        oCounter = j
        'Delete the next 6 lines (marked '*) if column 1 is not to be replicated
        oTable.Columns(1).Select '*
        With Selection '*
          .Copy '*
          .MoveDown Unit:=wdLine, Count:=2 '*
          .Paste '*
        End With '*
        GoTo Restart
      End If
    Next
SkipTable:
  Next
End With
Application.ScreenUpdating = True
End Sub
