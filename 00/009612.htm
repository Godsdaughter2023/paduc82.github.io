Try the below macro which will create a copy of the active workbook with all values and no formulas..The sheet names also would be same as the activeworkbook

 

Sub CopyActiveWorkbookValues()
'Jacob Skaria 2004/3/21
Dim intTemp As Integer, wbSource As Workbook, wbTarget As Workbook

Application.ScreenUpdating = False

Set wbSource = ActiveWorkbook
Set wbTarget = Workbooks.Add

wbTarget.Sheets.Add Count:=(wbSource.Sheets.Count - wbTarget.Sheets.Count)

For intTemp = 1 To wbSource.Sheets.Count
wbTarget.Sheets(intTemp).Name = "_" & wbSource.Sheets(intTemp).Name
wbSource.Sheets(intTemp).UsedRange.Copy
wbTarget.Sheets(intTemp).Range("A1").PasteSpecial xlValues
Application.CutCopyMode = False
Next

For intTemp = 1 To wbTarget.Sheets.Count
wbTarget.Sheets(intTemp).Name = Mid(wbTarget.Sheets(intTemp).Name, 2)
Next


'You can save the new workbook
'wbTarget.SaveAs "<filename>"

Application.ScreenUpdating = True

End Sub

Sub CopySelSheetsToNewWB()

Dim intTemp As Integer, wbSource As Workbook, wbTarget As Workbook

ReDim arrNames(0)

Set wbSource = ActiveWorkbook
intSelCount = wbSource.Windows(1).SelectedSheets.Count
ReDim arrNames(intSelCount)

For intTemp = 1 To intSelCount
arrNames(intTemp) = wbSource.Windows(1).SelectedSheets(intTemp).Name

wbSource.Sheets(1).Select

Set wbTarget = Workbooks.Add

With the selection in a date control, go to the Developer tab of the ribbon and then click on Properties in the Controls section and then select the required date format

d       one digit day
dd     two digit day

ddd   three character abbreviation for day of the week
dddd name of day of the week

similarly for M
