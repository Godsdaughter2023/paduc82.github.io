Private Sub Worksheet_Change(ByVal Target As Range)
    If Intersect(Target, Range("A5,H5")) Is Nothing Then Exit Sub
    If UCase(Range("H5")) = "UNC" Then Exit Sub
    
    If UCase(Range("A5")) = "UNC" Then
        Application.EnableEvents = False
        Range("H5").Value = "UNC"
        Application.EnableEvents = True
    End If
End Sub

Private Sub Worksheet_Change(ByVal Target As Range)
    If Not Intersect(Target, Range("A5")) Is Nothing Then
        If UCase(Target.Value) = "UNC" Then

            Else
            With Range("A5")

                   Range("H5").Value = ""

               End If
            End With
        End If
    End If
End Sub

ossiemac say:

Copy the code below into a workbook of its own.

The code allows you to select the required workbook from the File Dialog. Clicking Cancel in the file dialog will terminate the code without processing.

See comment at bottom of code re saving the workbook with deleted rows.

Ensure you have backups of your workbooks before running the code in case it does not work as anticipated.

Your comment: "Because column A has the same data on every sheet it will delete the same rows on every sheet per workbook." All worksheets except "Instruction" are selected soit will do exactly as your comment. If you have any worksheets where the rows are out of order then it will delete the incorrect rows on those worksheets.  If this could be a problem then it is easy enough to re-code so that it tests the cells in column A on every individual worksheet but code might take longer to run.

Sub DeleteRows()

Dim fd As FileDialog
Dim strFilename As String
Dim wb As Workbook
Dim arrShts()
Dim ws As Worksheet
Dim i As Long
Dim lngLastRow As Long

Set fd = Application.FileDialog(msoFileDialogFilePicker)

With fd
  .AllowMultiSelect = False
  If .Show Then
    strFilename = .SelectedItems(1)
  Else
    MsgBox "User cancelled file selection." _
        & vbCrLf & "Processing Terminated."
        Exit Sub
  End If
End With

Set fd = Nothing

Workbooks.Open Filename:=strFilename

strFilename = Mid(strFilename, InStrRev(strFilename, "\") + 1)

Set wb = Workbooks(strFilename)

wb.Activate

With wb
 
  ReDim arrShts(1 To .Sheets.Count - 1)

  For Each ws In .Worksheets
    If ws.Name <> "Instruction" Then
      i = i + 1
      arrShts(i) = ws.Name
    End If
  Next ws
 
  'Select all worksheets except Instruction
  .Worksheets(arrShts).Select
   
    With .ActiveSheet
      lngLastRow = .Cells(.Rows.Count, "A").End(xlUp).Row
      'Work backwards when deleting rows
      'To 2 assumes column headers
      For i = lngLastRow To 2 Step -1
        If .Cells(i, "A") = 1 Then
          .Rows(i).Select
          Selection.Delete
        End If
      Next i
      .Cells(1, 1).Select
    End With
End With

'wb.Save  'Optional. Don't use until sure code is doing as expected.
'wb.Close 'Optional. Don't use until sure code is doing as expected.

'Cleanup object variables
Set ws = Nothing
Set wb = Nothing

End Sub
