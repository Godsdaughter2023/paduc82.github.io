Sub RenameFiles()
    Dim fso, fso_folder, fso_file
    Dim sTargetFolder As String, sSubFolderName As String
    Dim sFileName As String, sTruncSubFldr As String
    Dim sOldName As String, sNewName As String
    Dim sFileExt As String, sFileNameNoExt As String
    Dim sFileToRename As String
   
    'INPUTS:
    sFileToRename = "data.txt"                                         'Enter file to be renamed here
    sTargetFolder = "C:\TESTING\"                                      'Enter parent folder path here
   
    'PROCESS:
    Set fso = CreateObject("Scripting.FileSystemObject")
    Set fso_folder = fso.GetFolder(sTargetFolder)                      'Set the FSO Folder (directory) object.
                             
    For Each fso_folder In fso_folder.SubFolders
        sSubFolderName = fso_folder.Name
        sTruncSubFldr = "_" & Right$(sSubFolderName, 3)                'Format the subfolder addition...
       
        For Each fso_file In fso_folder.Files
            If sFileName = sFileToRename Then                          'Only rename files called this
               
                sOldName = fso_file                                    'The original full path & filename
                sFileName = fso_file.Name                              'Just the file name+ext, e.g. "data.txt"
                sFileNameNoExt = Left(sFileName, Len(sFileName) - 4)   'Just the file name,     e.g. "data"
                sFileExt = Right(sFileName, 4)                         'Just the extension,     e.g. ".txt"
               
                sNewName = sTargetFolder & sSubFolderName & "\" & sFileNameNoExt & sTruncSubFldr & sFileExt
               
                Name sOldName As sNewName                              '(Move and) rename file.
           
            End If
        Next fso_file
    Next fso_folder
End Sub
