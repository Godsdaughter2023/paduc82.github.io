'Code shown below...

Sub CopyMultipleSelectionsZ()
'Jim Cone - Portland Oregon USA - February 2018
 On Error GoTo Refused
 Application.EnableCancelKey = xlErrorHandler
 Dim NextRow As Long
 Dim TopRow   As Long
 Dim RowTotal  As Long
 Dim DataCount  As Long
 Dim rngAll As Variant
 Dim rngPart As Excel.Range
 Dim WSsheet  As Excel.Worksheet

 Set rngAll = Excel.Selection
 If VBA.TypeName(rngAll) <> "Range" Then
   VBA.MsgBox " Select some cells.", , "Copy Selections"
   GoTo CleanUp
 ElseIf rngAll.Areas.Count < 2 Then
   GoTo CleanUp
 End If
 Set WSsheet = rngAll.Parent

 On Error Resume Next
 NextRow = WSsheet.UsedRange.Rows(WSsheet.UsedRange.Rows.Count).Row + 2
 If VBA.Err.Number <> 0 Then
   GoTo Refused
 Else
   On Error GoTo Refused
   TopRow = NextRow - 1 'save location
   For Each rngPart In rngAll.Areas
     RowTotal = RowTotal + rngPart.Rows.Count + 1
    'There must be at least two rows with data
     DataCount = DataCount + Application.WorksheetFunction.CountA(rngPart)
   Next 'rngPart
   If RowTotal + NextRow >= WSsheet.Rows.Count Then
     VBA.MsgBox " Not enough rows.", , "Copy Selections"
     GoTo CleanUp
   ElseIf DataCount < 2 Then
     GoTo CleanUp
   End If
 End If

 Application.ScreenUpdating = False
 For Each rngPart In rngAll.Areas
   rngPart.Copy Destination:=WSsheet.Cells(NextRow, 1)
  'End(xlup) fails if some selections are (partially) blank
   NextRow = NextRow + rngPart.Rows.Count + 1
 Next 'rngPart
'Add a marker and show copied data at top of window.
 WSsheet.Cells(TopRow, 1).Borders(xlEdgeBottom).Weight = xlMedium
 ActiveWindow.ScrollRow = TopRow
 ActiveWindow.ScrollColumn = 1

CleanUp:
 Set rngAll = Nothing
 Set WSsheet = Nothing
 Application.ScreenUpdating = True
 Exit Sub

Refused:
 Beep
 Resume CleanUp
End Sub
