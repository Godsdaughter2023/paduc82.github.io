hansv mvp say:

Here is a macro you can use. Note the comments - one line is optional.


Sub CombineData()
    Dim lngRow As Long
    Dim lngLastRow As Long
    Dim lngCol As Long
    ' Speed up macro by hiding what we're doing
    Application.ScreenUpdating = False
    ' Determine last row
    lngLastRow = Cells(Rows.Count, 2).End(xlUp).Row
    ' Loop backwards
    For lngRow = lngLastRow To 3 Step -1
        ' Adjusted time value same as row above?
        If Cells(lngRow, 3).Value = Cells(lngRow - 1, 3).Value Then
            ' Loop through data columns
            For lngCol = 4 To 23 ' D to W
                ' Do we have a value to copy?
                If Cells(lngRow, lngCol).Value <> "" And Cells(lngRow - 1, lngCol).Value = "" Then
                    ' If so, copy it
                    Cells(lngRow - 1, lngCol).Value = Cells(lngRow, lngCol).Value
                End If
            Next lngCol
            ' Optional: delete row
            Cells(lngRow, 1).EntireRow.Delete
        End If
    Next lngRow
    ' Show the result
    Application.ScreenUpdating = True
End Sub
