tasos_k say:

Hi,

scenario is....

you search for an ID number or a name (not unique)

in column A (in Source sheet named: Transposed Data)

then copy entire row into the last empty cell (in column A)

in Target sheet (named Search Results)

 

if so, try this code...

(make a copy, before you run the code)

 

 

Sub abcd()
Dim ws1 As Worksheet, ws2 As Worksheet
Dim r As Range, r2 As Long
Set ws1 = Sheets("Transposed Data")   '<<< source sheet, check sht name
Set ws2 = Sheets("Search Results")    '<<< target sheet, check sht name
Dim X As String
X = InputBox("select ID number or name", , "1234")
If X = "" Then Exit Sub
Application.ScreenUpdating = False
For Each r In ws1.Range("A:A").SpecialCells(xlCellTypeConstants)
If r.Value = X Then
r2 = ws2.Cells(Rows.Count, "A").End(xlUp).Row + 1
ws1.Rows(r.Row).Copy
ws2.Cells(r2, "A").PasteSpecial xlValues

Application.CutCopyMode = False
End If
Next
Application.ScreenUpdating = True
End Sub
