graham mayor say:
The following Outlook macro will copy the currently selected message to Word and list the BCCs each on a new line at the start. You can print the resulting document.


Sub OpenWord()
Dim wdApp As Word.Application
Dim wdDoc As Word.Document
Dim bStarted As Boolean
Dim olMsg As MailItem
Dim vBCC As Variant
Dim i As Long
On Error Resume Next
Set olMsg = ActiveExplorer.Selection.item(1)
vBCC = Split(olMsg.BCC, ";")

On Error Resume Next
Set wdApp = GetObject(, "Word.Application")
If Err Then
    Set wdApp = CreateObject("Word.Application")
    bStarted = True
End If
Set wdDoc = wdApp.Documents.Add
wdDoc.Range.InsertAfter "Subject: " & olMsg.Subject & vbCr
wdDoc.Range.InsertAfter "BCC: "
For i = 0 To UBound(vBCC)
    wdDoc.Range.InsertAfter vBCC(i) & vbCr
Next i
wdDoc.Range.InsertAfter olMsg.Body
wdApp.Visible = True
wdApp.Activate

Set wdDoc = Nothing
Set wdApp = Nothing
End Sub
