My goal is to create a concise list of all slides with a one or two line note on how to present them. I can type the slide titles, obviously, but I would like the computer to take care of this rather simple task.

john sr wilson say:

You can use a macro to get this.

Copy the code below

In PPt press ALT f11

In the new window INSERT > Module

Paste in the code

Run the code with f5 or from the Run menu (Run Sub / User Form)

'Code

Sub getTitles()
Dim osld As Slide
Dim sReport As String
Dim iFilenum As Integer
Dim sFilePath As String
For Each osld In ActivePresentation.Slides
    If osld.Shapes.HasTitle Then
        If osld.Shapes.Title.TextFrame.HasText Then
            sReport = sReport & "Slide  " & osld.SlideIndex & ": " _
                      & osld.Shapes.Title.TextFrame.TextRange & vbCrLf
        Else:
            sReport = sReport & "Slide  " & osld. _
                      SlideIndex & ": No title text" & vbCrLf
        End If
    Else
        sReport = sReport & "Slide  " & osld. _
                  SlideIndex & ": No title" & vbCrLf
    End If
Next osld
iFilenum = FreeFile
sFilePath = Environ("TEMP") & "\data.txt"
Open sFilePath For Output As iFilenum
Print #iFilenum, sReport
Close iFilenum
Call Shell("Notepad.exe " & sFilePath, vbNormalFocus)
End Sub
