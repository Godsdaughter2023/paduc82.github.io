rich michaels - mvp (greatcirclelearning.com) say:

Hi MBK999,

This will work:

Sub InsertPictures()
    'Written by Richard V Michaels, Word MVP
    Dim MyPath As String
    Dim MyScript As String
    Dim MyFiles As String
    Dim MySplit As Variant
    Dim N As Long
    Dim fName As String
    On Error Resume Next
    
    MyPath = MacScript("return (path to pictures folder) as String")
    MyScript = _
    "set applescript's text item delimiters to "","" " & vbNewLine & _
               "set theFiles to (choose file of type " & _
             " {""png"",""jpg"",""pdf"",""gif"",""bmp""} " & _
               "with prompt ""Use the Apple key Command + Click to select mulitples"" default location alias """ & _
               MyPath & """ multiple selections allowed true) as string" & vbNewLine & _
               "set applescript's text item delimiters to """" " & vbNewLine & _
               "return theFiles"

    MyFiles = MacScript(MyScript)
    On Error GoTo 0

    If MyFiles <> "" Then
        Dim rng As Word.Range
        Set rng = Selection.Range
        MySplit = Split(MyFiles, ",")
        For N = LBound(MySplit) To UBound(MySplit)
            fName = Right(MySplit(N), Len(MySplit(N)) - InStrRev(MySplit(N), Application.PathSeparator, , 1))
            rng.InsertParagraph
            rng.MoveStart Unit:=Word.WdUnits.wdParagraph, Count:=1
            rng.InlineShapes.AddPicture MySplit(N)
            rng.MoveEnd Unit:=Word.WdUnits.wdParagraph, Count:=1
            rng.Collapse Word.WdCollapseDirection.wdCollapseEnd
        Next N
    End If
End Sub
