Sub FindFromList()
Dim xlSheet As Worksheet
Dim LastRow As Integer
Dim strFind() As String
Dim i As Long, j As Long
Const ColLetter As String = "C"        'The column to search
Const strList As String = "worda|wordb|wordc|wordd"        'enter words to find in lower case each separated by "|"
    Set xlSheet = ActiveWorkbook.Sheets(1)

    For j = LastRow To 1 Step -1
        For i = 0 To UBound(strFind)

                xlSheet.Rows(j).Delete
            End If
        Next i
    Next j
End Sub

  3. Go to startup tab.
