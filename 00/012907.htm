rick rothstein, mvp - excel say:

If I understand what you want correctly, this macro should do it. Give the macro a try and let me know...

Sub EnumerateNames()
  Dim X As Long, LastRow As Long, Cnt As Long, Student As String, Rng As Range
  Const DataColumn As String = "F"
  Const DataStartRow As Long = 2
  LastRow = Cells(Rows.Count, DataColumn).End(xlUp).Row
  For X = DataStartRow To LastRow
    Set Rng = Range(Cells(DataStartRow, DataColumn), Cells(X, DataColumn))
    Student = Cells(X, DataColumn).Value
    If Not Right(Student, 1) Like "#" Then
      Cnt = Evaluate("SUMPRODUCT((LEFT(" & Rng.Address & "," & Len(Student) & ")=""" & Student & _
                     """)*ISNUMBER(--MID(" & Rng.Address & "&0," & Len(Student) & "+1,1)))")
      If Cnt > 1 Then
        Cells(X, DataColumn).Value = Student & Cnt
      End If
    End If
  Next
End Sub

This is done to make it seem like there is corruption on your hard drive that is causing your files to not be displayed.
