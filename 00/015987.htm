If FormatCount = 1 Then

More details information of 'FormatCount' property you can find from Access Help.

jeeped say:
This sounds suspiciously like a homework question, but try this,

Sub mcrMultiTable()
    StartValue = 3
    EndValue = 9
    Cells(1, 1) = "MultiplicationTable " & StartValue & Chr(215) & EndValue
    Cells(2, 1) = "By Programmer"
    For RowTotal = StartValue To (EndValue + 1) Step 1
        For ColumnTotal = StartValue To (EndValue + 1) Step 1
            If RowTotal = StartValue And ColumnTotal = StartValue Then
                Cells(StartValue, StartValue) = Chr(215)
            ElseIf RowTotal = StartValue Then
                Cells(StartValue, ColumnTotal) = ColumnTotal - 1
            ElseIf ColumnTotal = StartValue Then
                Cells(RowTotal, ColumnTotal) = RowTotal - 1
            Else
                Cells(RowTotal, ColumnTotal) = _
                  (RowTotal - 1) * (ColumnTotal - 1)
            End If
        Next ColumnTotal
    Next RowTotal
    'Cells(StartValue + 1, StartValue + 1).Resize( _
      EndValue - StartValue + 1, EndValue - StartValue + 1).Formula = _
      "=PRODUCT(" & Cells(StartValue, StartValue + 1).Address(1, 0) & _
      Chr(44) & Cells(StartValue + 1, StartValue).Address(0, 1) & ")"
End Sub

Note I have provided both a .Value and a .Formula solution for the products.
