Sub mcrRowColorChange()
    Dim rDataSet As Range
    Set rDataSet = Range(ActiveCell, Cells.SpecialCells(xlLastCell))
    With rDataSet.Interior
        .Pattern = xlSolid
        .PatternColorIndex = xlAutomatic
        .ThemeColor = xlThemeColorLight1
        .TintAndShade = 0.799981688894314
        .PatternTintAndShade = 0
        For Each c In Intersect(ActiveCell.EntireColumn, rDataSet)
            If c <> c.Offset(-1, 0) Then _
                Range(c, c.End(xlToRight)).Interior.ThemeColor = xlThemeColorLight2
        Next c
    End With
End Sub
