First Page

First Page

Primary Header (appears on all pages except the first page of a Section)

Sub InsertReferees(Control As IRibbonControl)
Dim fd As FileDialog
Dim strFile As String
Set fd = Application.FileDialog(msoFileDialogFilePicker)
