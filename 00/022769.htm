john sr wilson say:

On a Mac this might work (not foolproof) 

Sub reapplyNotes()
Dim shapetofind As Shape
Dim shapeMaster As Shape
Dim rayType() As String
Dim T As Long
rayType = Split("Slide Image,Notes,Date,Slide Number ,Header,Footer", ",")
Dim osld As Slide
For T = 0 To 5
For Each osld In ActivePresentation.Slides
Set shapetofind = getshape(osld.NotesPage, rayType(T))
Set shapeMaster = getshape(ActivePresentation.NotesMaster, rayType(T))
If Not shapetofind Is Nothing Then
shapetofind.Left = shapeMaster.Left
shapetofind.Top = shapeMaster.Top
shapetofind.Height = shapeMaster.Height
shapetofind.Width = shapeMaster.Width
shapeMaster.PickUp
shapetofind.Apply
End If
Next
Next
End Sub



Function getshape(obj As Object, strType As String) As Shape
Dim oshp As Shape
For Each oshp In obj.Shapes
If oshp.Type = msoPlaceholder Then
If oshp.Name Like strType & "*" Then
Set getshape = oshp
Exit Function
End If
End If
Next
End Function

=LEFT(A2,FIND("_",A2,1)-1) & RIGHT(A2,LEN(A2)-FIND("_",A2,1))
