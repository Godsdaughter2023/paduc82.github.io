jllatham say:

Here is a macro that should do the work for you very nicely - it did for me.  You just need a second sheet available for the results to be placed onto.

Change the values associated with the Const (constant) values declared in it to match your worksheet names and what columns are used.  There's a link in the code to a page that will tell you how to put the code into your workbook.

 

Here is the code:

Sub ExpandData()
'see this web page for instructions on how to
'put this code into a Regular Code Module in
'your workbook so you can use it:
' http://www.contextures.com/xlvba01.html#Regular
'If you are using excel 2007/2010, be sure to
'save your workbook as type .xlsm

  'change these constants as required
  'to match your worksheet names and layouts
  '
  'the worksheet with the list to be expanded
  Const dataWSName = "Sheet1"
  'these define where the data is
  ' where the first entry to be expanded is at (row)
  Const dataFirstRow = 2
  ' where the Land Use Types are entered
  Const dataLUTCol = "A"
  ' where the count is at
  Const dataCountCol = "B"
  ' where the "Richness" is at
  Const dataRichCol = "C"
  ' where the Priority is at
  Const dataPriorityCol = "D"
 
  'now for a sheet we can use to expand the data to
  Const destWSName = "Sheet2" ' cannot be same as dataWSName
'*** End of user definable constants ***
  Dim sourceWS As Worksheet
  Dim countsRange As Range
  Dim anyCount As Range
  Dim destWS As Worksheet
  Dim anyLastRow As Long
  Dim rowLoop As Long
  Dim TypeUse As String
  Dim Richness As Single
  Dim Priority As Single
 
  Set sourceWS = Worksheets(dataWSName)
  anyLastRow = _
   sourceWS.Range(dataCountCol & Rows.Count).End(xlUp).Row
  If anyLastRow < dataFirstRow Then
    Exit Sub ' no work to do!
  End If
  Set countsRange = sourceWS.Range(dataCountCol & dataFirstRow & _
   ":" & dataCountCol & anyLastRow)
 
  Application.ScreenUpdating = False
  Set destWS = Worksheets(destWSName)
  destWS.Cells.ClearContents ' clear any older results
  For Each anyCount In countsRange
    If Val(anyCount) > 0 Then
      TypeUse = sourceWS.Range(dataLUTCol & anyCount.Row)
      Richness = sourceWS.Range(dataRichCol & anyCount.Row)
      Priority = sourceWS.Range(dataPriorityCol & anyCount.Row)
      For rowLoop = 1 To Val(anyCount)
        anyLastRow = _
         destWS.Range(dataCountCol & Rows.Count).End(xlUp).Row + 1
        destWS.Range(dataLUTCol & anyLastRow) = TypeUse
        destWS.Range(dataCountCol & anyLastRow) = 1
        destWS.Range(dataRichCol & anyLastRow) = Richness
        destWS.Range(dataPriorityCol & anyLastRow) = Priority
      Next
    End If
  Next
  'housekeeping and cleanup
  Set countsRange = Nothing
  Set sourceWS = Nothing
  Set destWS = Nothing
  MsgBox "All done"
 
End Sub
