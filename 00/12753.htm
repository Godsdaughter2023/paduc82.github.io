jay freedman say:

Doug's version of the AddPicture call is functionally identical to mine, except for passing the arguments by position rather than by name.

The reason your version "did not work" is that you missed the point of the line continuation underscores that made all three of my lines into one statement (http://www.word.mvps.org/FAQs/MacrosVBA/_AtEndOfLine.htm). Then, when you put the LinkToFile and Range arguments outside the parentheses and made separate statements with them, VBA assumed that you were simply assigning values to (undeclared) variables -- and those had no effect on the AddPicture call or on oImage. In order to have a linked picture, the LinkToFile argument in the AddPicture call must be set to True.

macropod say:

The following code will process a whole folder of documents, copying formfield results from the document the macro is in to the corresponding formfields in all documents in the selected folder.

Sub UpdateDocuments()
Application.ScreenUpdating = False
Dim strFolder As String, strFile As String, strDocNm As String
Dim wdDocSrc As Document, wdDocTgt As Document, i As Long, StrBkMk As String
Set wdDocSrc = ActiveDocument
strDocNm = wdDocSrc.FullName
strFolder = GetFolder
If strFolder = "" Then Exit Sub
strFile = Dir(strFolder & "\*.doc", vbNormal)
While strFile <> ""
  If strFolder & "\" & strFile <> strDocNm Then
    Set wdDocTgt = Documents.Open(FileName:=strFolder & "\" & strFile, AddToRecentFiles:=False, Visible:=False)
    With wdDocTgt
      For i = 1 To wdDocSrc.FormFields.Count
        StrBkMk = wdDocSrc.FormFields(i).Name
        If .Bookmarks.Exists(StrBkMk) Then
          .FormFields(StrBkMk).Result = wdDocSrc.FormFields(i).Result
        End If
      Next
      .Close SaveChanges:=True
    End With
  End If
  strFile = Dir()
Wend
Set wdDocSrc = Nothing: Set wdDocTgt = Nothing
Application.ScreenUpdating = True
End Sub

Function GetFolder() As String
Dim oFolder As Object
GetFolder = ""
Set oFolder = CreateObject("Shell.Application").BrowseForFolder(0, "Choose a folder", 0)
If (Not oFolder Is Nothing) Then GetFolder = oFolder.Items.Item.Path
Set oFolder = Nothing
End Function
