    Option Explicit

    Private Sub Example_FormatSubTotals()
      Dim Data(0 To 10, 1 To 5)
      Dim i As Long, j As Long
      Dim TopLeftCell As Range
     
      If MsgBox("This example erases all data in the active sheet! Continue?", _
        vbOKCancel + vbDefaultButton2, "Example_FormatSubTotals") = vbCancel Then Exit Sub
      Cells.Clear
      Cells.ClearOutline
     
      Set TopLeftCell = Range("B2")
     
      'Create some example data
      For j = 1 To UBound(Data, 2)
        Data(0, j) = "Header " & j
      Next
      For j = 1 To UBound(Data, 2) \ 2
        For i = 1 To UBound(Data)
          Data(i, j) = String(j, Chr(65 + j * Rnd))
        Next
      Next
      For j = j To UBound(Data, 2)
        For i = 1 To UBound(Data)
          Data(i, j) = Int(UBound(Data) * Rnd) + 1
        Next
      Next
      TopLeftCell.Resize(UBound(Data) + 1, UBound(Data, 2)) = Data
      If MsgBox("Data without subtotals. Continue?", vbOKCancel) = vbCancel Then Exit Sub
     
      'Create some subtotals
      Application.ScreenUpdating = False
      With TopLeftCell.CurrentRegion
        .RemoveSubtotal
        For j = 1 To .Columns.Count \ 2
          .Subtotal j, xlSum, Array(.Columns.Count - j + 1), False
        Next
        .EntireColumn.AutoFit
      End With
      Application.ScreenUpdating = True
      If MsgBox("Data before formatting. Continue?", vbOKCancel) = vbCancel Then Exit Sub
     
      'Format them
      FormatSubTotals TopLeftCell.CurrentRegion, Weight:=xlMedium, ColorIndex:=5
    End Sub

    Sub FormatSubTotals(ByVal Where As Range, _
        Optional ByVal Header As XlYesNoGuess = xlYes, _
        Optional ByVal LineStyle As XlLineStyle = xlContinuous, _
        Optional ByVal Weight As XlBorderWeight = xlThin, _
        Optional ByVal ColorIndex As XlColorIndex = xlColorIndexAutomatic, _
        Optional ByVal Color As Variant)
      Dim All As Range, R As Range, B As Range, T As Range
      Dim FirstAddress As String
      Dim i As Integer
      Dim SaveScreenUpdating As Boolean
     
      'Search for the formulas from the top to the bottom in each row
      Set R = Where.Find("=SUBTOTAL(", LookIn:=xlFormulas, LookAt:=xlPart, SearchORder:=xlByRows)
      If R Is Nothing Then Exit Sub
     
      'Screen off, runs faster
      SaveScreenUpdating = Application.ScreenUpdating
      Application.ScreenUpdating = False

      'Border all, show all
      Where.Font.ColorIndex = xlColorIndexAutomatic
      If IsMissing(Color) Then
        Where.BorderAround LineStyle, Weight, ColorIndex
      Else
        Where.BorderAround LineStyle, Weight, Color:=Color
      End If
      Color = Where.Borders(xlEdgeTop).Color
      For i = xlInsideVertical To xlInsideHorizontal
        With Where.Borders(i)
          .LineStyle = xlContinuous
          .Weight = Weight
          .Color = Color
        End With
      Next
     
      'Set the top row as limit
      Set All = Where.Rows(IIf(Header = xlNo, 1, 2))
      FirstAddress = R.Address
      Do
        'Get the "..Total" cell on the left
        Set B = R.Offset(, Where.Column - R.Column)
        If IsEmpty(B) Then Set B = B.End(xlToRight)
        'Just to be sure
        If Not Intersect(B, Where) Is Nothing Then
          'Get the top cell within our limit
          Set T = Intersect(All, B.EntireColumn, Where.Resize(B.Row - 1))
          Set T = T.Areas(T.Areas.Count)
          Set T = T.Cells(T.Cells.Count)
          '"Hide" the contents if more then one row
          If B.Row - T.Row > 0 Then
            With Where.Parent.Range(T, B.Offset(-1))
              .Font.Color = .Interior.Color
            End With
          End If
          'Remove the border inside
          With Where.Parent.Range(B, T)
            .Borders(xlInsideHorizontal).LineStyle = xlLineStyleNone
          End With
          With Where.Parent.Range(B, R)
            .Borders(xlInsideVertical).LineStyle = xlLineStyleNone
          End With
          'Add the horizontal cells to our limit
          Set All = Union(All, Range(B, R).Offset(1))
        End If
        'Find the next formula
        Set R = Where.FindNext(R)
        If R Is Nothing Then Exit Do
      Loop Until R.Address = FirstAddress

      Application.ScreenUpdating = SaveScreenUpdating
    End Sub

    doug robbins - mvp office apps & services (word) say:

    The following code in the ThisOutlookSession object will do it:

    Private Sub Application_ItemSend(ByVal Item As Object, Cancel As Boolean)

    Dim msg As MailItem

    Dim strGreeting

    Dim hrs As Long

    Dim mins As Long

    Dim lngTime As Long

    hrs = Val(Left(Format(Time, "HH:mm"), 2))

    mins = Val(Right(Format(Time, "HH:mm"), 2))

    lngTime = 60 * hrs + mins

    Select Case lngTime

    Case Is <= 720

        strGreeting = "Good Morning"

    Case Is > 960

        strGreeting = "Good Evening"

    Case Else

        strGreeting = "Good Afternoon"

    End Select

    Set msg = Application.ActiveInspector.CurrentItem

    msg.Body = strGreeting & vbCr & msg.Body

    End Sub
