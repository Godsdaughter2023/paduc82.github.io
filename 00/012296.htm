graham mayor say:
The short answer is no, but you could format the text you want to hide with a specific character style and then use a macro to toggle the hidden font parameter of that style. e.g. assuming a style called 'Hidden'


Sub ToggleHidden()
Dim oStyle As Style
    ActiveWindow.View.ShowHiddenText = False
    For Each oStyle In ActiveDocument.Styles
        If oStyle.NameLocal = "Hidden" Then
            With oStyle.Font
                .Hidden = Not .Hidden
            End With
            Exit For
        End If
    Next oStyle
End Sub

*********************************************************************************************************

Option Base 1
Sub Deleting_DTD_declaration_in_XML_File()

    Dim Text, Corrected_Text, Dir1, Filename1, Filename2 As String
    Dim Less_Than_Sign(2), Greater_Than_Sign(2), Number_of_left_or_right_chevron, Text_size As Long
    Dim File_number As Integer
    
    File_number = 1
    Number_of_left_or_right_chevron = 1
             
    Call Import_Data_from_XML_file(Dir1, Filename1, Text, Text_size)
    Call Read_File(Less_Than_Sign, Greater_Than_Sign, Number_of_left_or_right_chevron, Text, Text_size)
    Call Open_File(Dir1, Filename1, Filename2, File_number)
    Call Modifying_Text(Text, Less_Than_Sign, Greater_Than_Sign, Corrected_Text)
    Call Corrected_Text_File(Corrected_Text, File_number)
   
End Sub
Sub Import_Data_from_XML_file(Dir1, Filename1, Text, Text_size)

    Dir1 = "C:\path1\"
    Filename1 = "Data.xml"
        
    'read file content
    Text = CreateObject("scripting.filesystemobject").opentextfile(Dir1 + Filename1).readall()
    Text_size = Len(Text)

End Sub
Sub Read_File(Less_Than_Sign, Greater_Than_Sign, Number_of_left_or_right_chevron, Text, Text_size)

Dim I_Character As String

    'Less_Than_Sign, <
    'Greater_Than_Sign, >
    
    
For I = 1 To Text_size
    I_Character = Mid(Text, I, 1) ' Get one character.
    
    If Number_of_left_or_right_chevron = 3 Then Exit For
    
    If I_Character = Chr(60) Then
         Less_Than_Sign(Number_of_left_or_right_chevron) = I
    ElseIf I_Character = Chr(62) Then
        Greater_Than_Sign(Number_of_left_or_right_chevron) = I
        Number_of_left_or_right_chevron = Number_of_left_or_right_chevron + 1
    End If
Next
End Sub
Sub Open_File(Dir1, Filename1, Filename2, File_number)
    
    Filename2 = Left(Filename1, (Len(Filename1) - 4)) + "_corr.xml"
    Open Dir1 + Filename2 For Output As #File_number
    
End Sub

Sub Modifying_Text(Text, Less_Than_Sign, Greater_Than_Sign, Corrected_Text)
    Dim Temp As String
    
    Temp = Left(Text, Greater_Than_Sign(1))
    Corrected_Text = Left(Text, Greater_Than_Sign(1)) + vbLf + Mid(Text, Greater_Than_Sign(2) + 1)
    
End Sub
Sub Corrected_Text_File(Corrected_Text, File_number)
    
    Print #File_number, Corrected_Text
    Close #File_number
    
End Sub
