jan karel pieterse say:

I assumed you used buttons from the forms toolbar, all of them tied to the sub called StartStopPause ...

 

The code below keeps score of the status of each cell in the cell to its immediate right (if that is not what you want, change the .offset(,1) to something else).

 

This code goes in a normal module...

 

Option Explicit

 

Dim mdNextTime As Double
Dim mcTimers As Collection
Dim mdLastTime As Double

 

Sub StartStopPause()
    Dim sAddress As String
    If mcTimers Is Nothing Then
        Set mcTimers = New Collection
        mdNextTime = Now
        mdLastTime = Now
        Application.OnTime mdNextTime, "UpdateTimers"
    End If
    sAddress = ActiveSheet.Buttons(Application.Caller).TopLeftCell.Address
    Select Case ActiveSheet.Buttons(Application.Caller).Caption
    Case "Start"
        If Not IsIn(mcTimers, sAddress) Then
            mcTimers.Add sAddress
            If Range(sAddress).Offset(, 1).Value = "Stopped" Or Len(Range(sAddress).Offset(, 1).Value) = 0 Then
                Range(sAddress).Value = 0
            End If
            Range(sAddress).Offset(, 1).Value = "Running"
        End If
    Case "Stop"
        If IsIn(mcTimers, sAddress) Then
            RemoveCell sAddress
            Range(sAddress).Offset(, 1).Value = "Stopped"
        End If
    Case "Pause"
        If IsIn(mcTimers, sAddress) Then
            RemoveCell sAddress
            Range(sAddress).Offset(, 1).Value = "Paused"
        End If
    End Select
    If mcTimers.Count = 0 And mdNextTime <> 0 Then
        On Error Resume Next
        'Unschedule timer as there are no cells active.
        Application.OnTime mdNextTime, "UpdateTimers", , False
        Set mcTimers = Nothing
    End If
End Sub

 

Sub UpdateTimers()
    Dim vAddress As Variant
    mdNextTime = Now + TimeValue("00:00:01")
    For Each vAddress In mcTimers
        With Range(vAddress)
            .Value = .Value + (Now - mdLastTime)
        End With
    Next
    mdLastTime = Now
    Application.OnTime mdNextTime, "UpdateTimers"
End Sub

 

Function IsIn(ByRef colCollection As Collection, ByVal sAddress As String) As Boolean
'-------------------------------------------------------------------------
' Procedure : IsIn Function Created by Jan Karel Pieterse
' Company   : JKP Application Development Services (c) 2006
' Author    : Jan Karel Pieterse
' Created   : 11-03-2013
' Purpose   : Determines if object/item is in collection
'-------------------------------------------------------------------------
    Dim vMember As Variant
    On Error Resume Next
    For Each vMember In colCollection
        Err.Clear
        If vMember = sAddress Then
            If Err.Number = 0 Then
                IsIn = True
                Exit For
            End If
        End If
    Next vMember
    Err.Clear
End Function

    Next
End Sub

jay freedman say:

If you haven't assigned names to the content controls, they don't have any names. To use them as you indicated in your first post, you need to assign names to at least the No box in Question 1 and the three boxes in Question 2. To do that:

  • Click in the Yes box in Question 1, and then click the Properties button on the Developer ribbon (just to the right of the content controls).
  • In the Properties dialog that opens, click in the Tag box and enter a name. I'll suggest Q1Yes for the name.
  • Click OK.
  • Open the Properties dialogs of the other two content controls in Question 1 and the three content controls in Question 2, one at a time, and assign their names in the Tag box. I suggest Q1No, Q2NA, Q2Yes, Q2No, and Q2NA.

If your form has other sets of questions that should behave in a similar way, they also need names so the macro can work with them.

Using the names I listed above, you can just copy/paste the following code into the ThisDocument module of the form's template, saving the template as macro-enabled with a .dotm extension. (You could save the form as a .docm document, but then you'd run into problems with new forms overwriting older ones.)

Private Sub Document_ContentControlOnExit(ByVal ExitCC As ContentControl, Cancel As Boolean)
    ' If Question 1 is answered with No, then Question 2 should be marked NA
    ' and all three boxes locked. If Question 1 is then changed to Yes,
    ' Question 2 should be unlocked and all boxes cleared.
    
    Dim otherCC As ContentControl
    
    Select Case ExitCC.Tag
        Case "Q1Yes"
            If ExitCC.Checked Then
                GetCCbyTag("Q1No").Checked = False
                GetCCbyTag("Q1NA").Checked = False
                Set otherCC = GetCCbyTag("Q2Yes")
                With otherCC
                    .LockContents = False
                    .Checked = False
                End With
                Set otherCC = GetCCbyTag("Q2No")
                With otherCC
                    .LockContents = False
                    .Checked = False
                End With
                Set otherCC = GetCCbyTag("Q2NA")
                With otherCC
                    .LockContents = False
                    .Checked = False
                End With
            End If
        
        Case "Q1No"
            If ExitCC.Checked Then
                GetCCbyTag("Q1Yes").Checked = False
                GetCCbyTag("Q1NA").Checked = False
                Set otherCC = GetCCbyTag("Q2Yes")
                With otherCC
                    .LockContents = False
                    .Checked = False
                    .LockContents = True
                End With
                Set otherCC = GetCCbyTag("Q2No")
                With otherCC
                    .LockContents = False
                    .Checked = False
                    .LockContents = True
                End With
                Set otherCC = GetCCbyTag("Q2NA")
                With otherCC
                    .LockContents = False
                    .Checked = True
                    .LockContents = True
                End With
            End If
        
        Case Else ' do nothing
    End Select
End Sub

Private Function GetCCbyTag(strTag As String) As ContentControl
    ' Notes: (1) The argument of the SelectContentControlsByTag function is
    ' case-sensitive. If the capitalization of the name passed into strTag
    ' is incorrect, no content control will be found. (2) Tag values are
    ' not necessarily unique. If the document contains more than one
    ' content control with the same Tag value, this function will return
    ' only the first one.
    
    Dim CCs As ContentControls
    Dim CC As ContentControl
    
    Set CC = Nothing
    Set CCs = ActiveDocument.SelectContentControlsByTag(strTag)
    If CCs.Count > 0 Then
        Set CC = CCs(1)
    End If
    
    If CC Is Nothing Then
        MsgBox "A content control with Tag value " & strTag & " was not found", _
            vbOKOnly + vbCritical, "Error"
    End If
    
    Set GetCCbyTag = CC
End Function
