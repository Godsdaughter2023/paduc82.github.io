You can execute the below macro to print out the specific pages.

 

Sub PrintMacro()

' PrintMacro Macro
    ActiveWindow.SelectedSheets.PrintOut From:=4, To:=4, Copies:=1, Collate _
        :=True, IgnorePrintAreas:=False
    ActiveWindow.SelectedSheets.PrintOut From:=45, To:=45, Copies:=1, Collate _
        :=True, IgnorePrintAreas:=False
    ActiveWindow.SelectedSheets.PrintOut From:=67, To:=67, Copies:=1, Collate _

End Sub
