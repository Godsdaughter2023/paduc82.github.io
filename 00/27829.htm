graham mayor say:

If your worksheet has the common names in column 1 and the scientific names in column 2 then  you could use a macro to perform the lookup. If you attach the macro to a keyboard shortcut or a button on the ribbon or QAT (Quick Access Toolbar) the lookup is but a click away. - see http://www.gmayor.com/installing_macro.htm


You will need to change the bold text to suit your requirement.


The following function is fast. It works by loading the worksheet into an array in memory which is then searched in column 1 for the typed word. The function is not case sensitive, and will put the corresponding scientific name from column 2 after the common name, italicised and in brackets e.g. michaelmas daisy (Aster amellus)

The only proviso is that it performs the search on the paragraph containing the common name, so if there are multiple plants in the paragraph, the process will not work as intended. Beyond that just type the common name and run the macro.


Option Explicit
Private RS As Object
Private CN As Object
Private iRows As Long
Private sResult As String
Private oRng As Range
Const strWorkbook As String = "C:\Path\WorkbookName.xlsx"        'The path of the workbook
Const strSheet As String = "Sheet1"        'The name of the worksheet

Private Function xlFillArray(strWorkbook As String, _
                             strWorksheetName As String) As Variant
    strWorksheetName = strWorksheetName & "$]"
    Set CN = CreateObject("ADODB.Connection")
    CN.Open ConnectionString:="Provider=Microsoft.ACE.OLEDB.12.0;" & _
                              "Data Source=" & strWorkbook & ";" & _
                              "Extended Properties=""Excel 12.0 Xml;HDR=YES"";"

    Set RS = CreateObject("ADODB.Recordset")
    RS.Open "SELECT * FROM [" & strWorksheetName, CN, 2, 1

    With RS
        .MoveLast
        iRows = .RecordCount
        .MoveFirst
    End With
    xlFillArray = RS.GetRows(iRows)
    If RS.State = 1 Then RS.Close
    Set RS = Nothing
    If CN.State = 1 Then CN.Close
    Set CN = Nothing
lbl_Exit:
    Exit Function
End Function

Function GetArrayValue(strItem As String) As String
Dim arr() As Variant
    arr = xlFillArray(strWorkbook, strSheet)
    For iRows = 0 To UBound(arr, 2)
        sResult = arr(0, iRows)
        If InStr(1, sResult, strItem) > 0 Then GetArrayValue = arr(1, iRows)
    Next iRows
lbl_Exit:
    Exit Function
End Function

Sub GetScientificName()
Dim arr() As Variant
Dim bFound As Boolean
Dim sFindText As String
Dim sReplaceText As String
Dim bBold As Boolean
Dim bItalic As Boolean
    'sFindText = Trim(Selection.Range.Text)
    arr = xlFillArray(strWorkbook, strSheet)
    For iRows = 0 To UBound(arr, 2)
        sFindText = arr(0, iRows)
        sReplaceText = arr(1, iRows)
        Set oRng = Selection.Paragraphs(1).Range
        With oRng.Find
            .ClearFormatting
            .Replacement.ClearFormatting
            Do While .Execute(FindText:=sFindText, _
                              MatchWholeWord:=True, _
                              Forward:=True, _
                              Wrap:=wdFindStop) = True
                oRng.Collapse wdCollapseEnd
                oRng.Text = " (" & sReplaceText & ") "
                oRng.Font.Italic = True
                oRng.Font.Bold = False
                oRng.Collapse 0
                oRng.Font.Italic = bItalic
                oRng.Font.Bold = bBold
                oRng.Select
                bFound = True
                Exit Do
            Loop
        End With
    Next
    If Not bFound Then
        MsgBox "Scientific name not found. Check your spelling."
    End If
lbl_Exit:
    Set oRng = Nothing
    Exit Sub
End Sub
