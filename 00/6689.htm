This works for me.

Place both procedures in a Sheet object and it lists the file name in column A hyperlinked to the file, and the full path in column B

 

'This is the bit that does the listing

Sub ListFilesInFolder(SourceFolderName As String, IncludeSubfolders As Boolean)
Dim FSO As Scripting.FileSystemObject 'Microsoft Scripting Runtime reference must be ticked
Dim SourceFolder As Scripting.Folder, SubFolder As Scripting.Folder
Dim FileItem As Scripting.File
Dim r As Long
    Set FSO = New Scripting.FileSystemObject
    Set SourceFolder = FSO.GetFolder(SourceFolderName)
    r = Range("A65536").End(xlUp).Row + 1
    For Each FileItem In SourceFolder.Files
        On Error Resume Next
        Cells(r, 1).Formula = FileItem.Name
        'Create a hyperlink to the file
         ActiveSheet.Hyperlinks.Add anchor:=Cells(r, 1), Address:=FileItem.Path, TextToDisplay:=FileItem.Name
            Cells(r, 2).Formula = FileItem.Path
        Set objShell = CreateObject("Shell.application")
        Set objfolder = objShell.Namespace(SourceFolder.Path)
        Set objfolderitem = objfolder.ParseName(FileItem.Name)
        r = r + 1
    Next FileItem
    If IncludeSubfolders Then
        For Each SubFolder In SourceFolder.SubFolders
            FldNum = FldNum + 1
            ListFilesInFolder SubFolder.Path, True
        Next SubFolder
    End If
    Set FileItem = Nothing
    Set SourceFolder = Nothing
    Set FSO = Nothing
    ActiveWorkbook.Saved = True
End Sub
