Option Explicit

'This example requires one listbox in the left upper corner of a userform
'The active sheet should have some headers in the first row and some data below:
'
'Nr Surename Forename City
'1  Duck     Donald   Entenhausen
'2  Bugs     Bunny
'3  Killer   Andreas  Hannover

#Const AutoUpdate = True
'If AutoUpdate is True every textbox will update the data in sheet immediately
'This requires a class module named EditField too.
'
'Public WithEvents TextBox As MSForms.TextBox
'
'Private Sub TextBox_Change()
'  'Update the cell correspond to this textbox
'  Dim Value
'  If IsNumeric(TextBox) Then
'    Value = CDbl(TextBox)
'  Else
'    Value = TextBox
'  End If
'  Cells(ActiveCell.Row, CLng(TextBox.Tag)) = Value
'End Sub


#If AutoUpdate Then
Dim EditFields As New Collection
#End If

Private Sub ListBox1_Change()
  Dim Y As Long
  Dim C As Control
  'Select row in sheet which correspond to our entry
  Y = ListBox1.ListIndex + 2
  Cells(Y, 1).EntireRow.Select
  'Fill our Textboxes
  For Each C In Me.Controls
    If C.Tag <> "" Then C = Cells(Y, CLng(C.Tag))
  Next
End Sub

Private Sub UserForm_Initialize()
  Dim R As Range
  Dim X As Single, Y As Single
  Dim C As Control
  'Create fields for each header on the right side of the listbox
  With ListBox1
    X = .Left + .Width + 3
    Y = 3
  End With
  'Walk trough the first row
  For Each R In Range(Cells(1, 1), Cells(1, Columns.Count).End(xlToLeft))
    'Add description
    Set C = Me.Controls.Add("Forms.Label.1")
    C.Caption = R
    C.AutoSize = True
    C.Left = X
    C.Top = Y
    Y = Y + C.Height + 3
    'Add edit field
    Set C = Me.Controls.Add("Forms.TextBox.1")
    'Store the column in the tag field
    C.Tag = R.Column
#If AutoUpdate Then
    'Enable edit field
    Dim EF As EditField
    Set EF = New EditField
    Set EF.TextBox = C
    EditFields.Add EF
#End If
    C.Left = X
    C.Top = Y
    Y = Y + C.Height + 3
  Next
  'Add scrollbar to userform if necessary
  If Y > Me.InsideHeight Then
    Me.ScrollBars = fmScrollBarsVertical
    Me.ScrollHeight = Y
  End If
  '
  'Read the values from column A into the listbox
  For Each R In Range(Range("A2"), Range("A" & Rows.Count).End(xlUp))
    ListBox1.AddItem R
  Next
  'Listbox should search if multiple keys are pressed
  ListBox1.MatchEntry = fmMatchEntryComplete
  'Select entry which is selected in the sheet
  '(this fires the change event of the listbox and the record is loaded into the textboxes)
  ListBox1.ListIndex = ActiveCell.Row - 2
End Sub

To run the PowerShell script downloaded from this link normally, please make sure:
1. The following paths to SDK in the script file are collect for your computer:
Add-Type -Path "c:\Program Files\Common Files\microsoft shared\Web Server Extensions\16\ISAPI\Microsoft.SharePoint.Client.dll"

2. You have imported this module with command below:
Import-Module C:\test\GetSPOListItems9.psm1 -Verbose

Meanwhile, based on my test, we can use the following command to list all unique file types in a document library:
Get-SPOListItems -Username *** Email address is removed for privacy *** -Url https://contoso.sharepoint.com/sites/test/ -AdminPassword Pass -ListTitle Documents -IncludeAllProperties $true -Recursive | select -Property File_x0020_Type -Unique

 

Note: It will take a while to finish running the command.
