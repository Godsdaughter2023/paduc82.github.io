graham mayor say:

Use the following custom process with http://www.gmayor.com/document_batch_processes.htm which will handle the file/folder processing. Test with a single document.

Function MoveCol(oDoc As Document) As Boolean
Dim oTable As Table
Dim oCell1 As Range, oCell2 As Range
Dim i As Long
    On Error GoTo err_handler
    For Each oTable In oDoc.Tables
        If oTable.Columns.Count > 2 Then
            For i = 2 To oTable.Rows.Count
                Set oCell1 = oTable.Rows(i).Cells(2).Range
                oCell1.End = oCell1.End - 1
                Set oCell2 = oTable.Rows(i).Cells(3).Range
                oCell2.End = oCell2.End - 1
                oCell2.FormattedText = oCell1.FormattedText
                oCell1.Text = ""
            Next i
        End If
    Next oTable
    MoveCol = True
lbl_Exit:
    Exit Function
err_handler:
    MoveCol = False
    Resume lbl_Exit
End Function
