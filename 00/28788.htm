Enter below formula in cell D2 & copy down:

=IFERROR(VALUE(IF(COUNTIF(D$1:D1,D1)=IFERROR(VLOOKUP(D1,$A$2:$B$100,2,FALSE),""),"",IF(COUNTIF(D$1:D1,D1)<IFERROR(VLOOKUP(D1,$A$2:$B$100,2,FALSE),0),D1,INDEX($A$2:$A$100,MATCH(0,INDEX(COUNTIF(D$1:D1,$A$2:$A$100&""),0,0),0))&""))),"")
