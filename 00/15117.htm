bernie deitrick say:

You could keep a record in the same row - say, starting in column AA:

Private Sub Worksheet_Change(ByVal Target As Range)

    Dim c As Range

    Dim lngC As Long

   If Intersect(Target,Range("G2:G18")) Is Nothing Then Exit Sub

   'Turn off events to keep out of loops

    Application.EnableEvents = False

    For Each c In Intersect(Target,Range("G2:G18"))
        lngC = Cells(c.Row, Columns.Count).End(xlToLeft).Column + 1
        If lngC < 28 Then lngC = 27
        Cells(c.Row, lngC).Value = c.Value
    Next c

    'Turn events back on to get ready for the next change
    Application.EnableEvents = True

End Sub

vijay a. verma say:

You can do this through following VBA Macro. This macro copies the data in a worksheet named "Results" and Sheet1 has first row as Header.

1. Make a backup of your workbook. 
2. Open your workbook and ALT+F11
3. Locate your Workbook name in Project Explorer Window
4. Right click on your workbook name > Insert > Module 
5. Copy paste the Macro code given and change the bold line as per your requirement
6. Go back to your Workbook and ALT+F8 to display Macro Window
7. Run your Macro from here
8. Delete you Macro if the Macro was needed to be run only once.
9. Otherwise save your file as .xlsm if you intend to reuse Macro again.

'*** Macro Starts

Sub TransMail()
    Dim SWs As Worksheet, TWs As Worksheet, NewSWs As Worksheet
    Dim Cell As Range, Rng As Range
    Dim i As Long, j As Long
    Set SWs = Worksheets("Sheet1")
    SWs.Copy After:=SWs
    Set NewSWs = ActiveSheet
    NewSWs.Cells.Sort Key1:=NewSWs.Range("A1"), Header:=xlYes
    Set Rng = NewSWs.Range("A2:A" & NewSWs.Range("A" & Rows.Count).End(xlUp).Row)
    i = 1
    Application.ScreenUpdating = False
    Application.DisplayAlerts = False
    On Error Resume Next
    Worksheets("Results").Delete
    On Error GoTo 0
    Set TWs = Worksheets.Add(After:=NewSWs)
    TWs.Name = "Results"
    TWs.Range("A1") = SWs.Range("A1")
    For Each Cell In Rng
        If Cell = Cell.Offset(-1, 0) Then
            j = j + 2
            TWs.Cells(i, 1) = Cell
            TWs.Cells(1, j) = SWs.Range("B1")
            TWs.Cells(i, j) = Cell.Offset(0, 1)
            TWs.Cells(1, j + 1) = SWs.Range("C1")
            TWs.Cells(i, j + 1) = Cell.Offset(0, 2)
            Else
            j = 2
            i = i + 1
            TWs.Cells(i, 1) = Cell
            TWs.Cells(1, j) = SWs.Range("B1")
            TWs.Cells(i, j) = Cell.Offset(0, 1)
            TWs.Cells(1, j + 1) = SWs.Range("C1")
            TWs.Cells(i, j + 1) = Cell.Offset(0, 2)
        End If
    Next Cell
    NewSWs.Delete
    Application.ScreenUpdating = True
    Application.DisplayAlerts = True
End Sub
