HOSTNAME                   POINTS TO                               TTL(SEC)             TYPE
      www                xxxxxxx.sharepoint.com                     3600               CNAME

Extract comments to new doc Macro - Microsoft Community

allisonbarlow say:

Extract comments to new doc Macro

I am trying to run a macro in a word doc.  I have enabled the macros and when I go to view the macros there is nothing there.  I've added comments to my doc (coded it) and want to extract those comments into another doc.  I saw on youtube how to do this.  Problem is I don't have any macros.  Help!!!

doug robbins - mvp office apps & services (word) say:

Use the following code in a macro:

Sub exportCommentsandChanges()

Dim xlApp As Object

Dim xlWB As Object

Dim xlSheet As Object

Dim i As Integer, HeadingRow As Integer

Dim objPara As Paragraph

Dim objComment As Comment

Dim strSection As String

Dim strTemp

Dim myRange As Range

Dim oDoc As Document

Dim oTable As Table

Dim oRow As Row

Dim oCol As Column

Dim oRange As Range

Dim oRevision As Revision

Dim strText As String

Dim n As Long

Dim Title As String

Dim j As Integer

 

Set xlApp = CreateObject("Excel.Application")

xlApp.Visible = True

Set xlWB = xlApp.Workbooks.Add 'create a new excel sheet

Set xlSheet = xlWB.Worksheets(1)

With xlSheet

    .Name = "Comments"

' Create Heading for the excel

    HeadingRow = 1

    .Cells(HeadingRow, 1).Formula = "Comment ID"

    .Cells(HeadingRow, 2).Formula = "Page"

    .Cells(HeadingRow, 3).Formula = "Comment text"

    .Cells(HeadingRow, 4).Formula = "Reviewer"

    .Cells(HeadingRow, 5).Formula = "Comment Date"

    strSection = "preamble" 'all sections before "1." will be labeled as "preamble"

    strTemp = "preamble"

    If ActiveDocument.Comments.Count = 0 Then

        MsgBox ("There are no comments made in this document.")

    End If

    For i = 1 To ActiveDocument.Comments.Count

        Set myRange = ActiveDocument.Comments(i).Scope

        .Cells(i + HeadingRow, 1).Formula = ActiveDocument.Comments(i).Index

        .Cells(i + HeadingRow, 2).Formula = ActiveDocument.Comments(i).Reference.Information(wdActiveEndAdjustedPageNumber)

        .Cells(i + HeadingRow, 3).Formula = ActiveDocument.Comments(i).Range

        .Cells(i + HeadingRow, 4).Formula = ActiveDocument.Comments(i).Author

        .Cells(i + HeadingRow, 5).Formula = Format(ActiveDocument.Comments(i).Date, "dd/MM/yyyy")

    Next i

  ' Let the user know that the excel document is created

   ' Switch to the Excel workbook

   .Visible = True

End With

Set xlSheet = xlWB.Worksheets.Add

With xlSheet

    .Name = "Revisions"

    HeadingRow = 1

    .Cells(HeadingRow, 1).Formula = "Page"

    .Cells(HeadingRow, 2).Formula = "Line"

    .Cells(HeadingRow, 3).Formula = "Type"

    .Cells(HeadingRow, 4).Formula = "What has been inserted or deleted"

    .Cells(HeadingRow, 5).Formula = "Reviewer"

    .Cells(HeadingRow, 6).Formula = "Comment Date"

    'Get info from each tracked change (insertion/deletion) from oDoc and insert in table

    For Each oRevision In ActiveDocument.Revisions

        Select Case oRevision.Type

            'Only include insertions and deletions

            Case wdRevisionInsert, wdRevisionDelete

                'In case of footnote/endnote references (appear as Chr(2)),

                'insert "[footnote reference]"/"[endnote reference]"

                With oRevision

                    'Get the changed text

                    strText = .Range.Text

                    Set oRange = .Range

                    Do While InStr(1, oRange.Text, Chr(2)) > 0

                        'Find each Chr(2) in strText and replace by appropriate text

                        i = InStr(1, strText, Chr(2))

                        If oRange.Footnotes.Count = 1 Then

                            strText = Replace(Expression:=strText, _

                                    Find:=Chr(2), Replace:="[footnote reference]", _

                                    Start:=1, Count:=1)

                            'To keep track of replace, adjust oRange to start after i

                            oRange.Start = oRange.Start + i

                        ElseIf oRange.Endnotes.Count = 1 Then

                            strText = Replace(Expression:=strText, _

                                    Find:=Chr(2), Replace:="[endnote reference]", _

                                    Start:=1, Count:=1)

                            'To keep track of replace, adjust oRange to start after i

                            oRange.Start = oRange.Start + i

                        End If

                   Loop

                End With

                  'Add 1 to counter

                n = n + 1

                'Add row to table

                'Insert data in cells in oRow

                    'Page number

                    .Cells(n + HeadingRow, 1).Formula = _

                        oRevision.Range.Information(wdActiveEndPageNumber)

                    'Line number - start of revision

                    .Cells(n + HeadingRow, 2).Formula = _

                        oRevision.Range.Information(wdFirstCharacterLineNumber)

                    'Type of revision

                    If oRevision.Type = wdRevisionInsert Then

                        .Cells(n + HeadingRow, 3).Formula = "Inserted"

                        'Apply automatic color (black on white)

                    Else

                        .Cells(n + HeadingRow, 3).Formula = "Deleted"

                        'Apply red color

                    End If

                    'The inserted/deleted text

                    .Cells(n + HeadingRow, 4).Formula = strText

                    'The author

                    .Cells(n + HeadingRow, 5).Formula = oRevision.Author

                    'The revision date

                    .Cells(n + HeadingRow, 6).Formula = Format(oRevision.Date, "mm-dd-yyyy")

        End Select

    Next oRevision

End With

If n = 0 Then

    MsgBox ("There are no insertions or deletions made in this document."), vbOKOnly, Title

End If

Set oDoc = Nothing

Set xlWB = Nothing

Set xlApp = Nothing

Set oTable = Nothing

Set oRow = Nothing

Set oRange = Nothing

End Sub

To see what to do with the code, see the following page of Graham Mayor’s website

 

http://www.gmayor.com/installing_macro.htm

The metered connection tries to save your data plan, but you can turn it off following these steps:
Open the start menu and select Settings.
Click the Network and Internet link.
Click the Manage known network link under your existing connection.
Choose the network you're currently using from the list of available ones.
Click Properties.
Find the Metered connection section and toggle the switch to Off
