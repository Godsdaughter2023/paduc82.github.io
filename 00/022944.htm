Give this macro a try (change the indicated line as needed)...

Sub DeleteColumns()
  Dim ColumnsToDelete As String, V As Variant
  ColumnsToDelete = "Inventory,Sales,Address"
  On Error Resume Next
  For Each V In Split(ColumnsToDelete, ",")
    Rows(1).Find(Trim(V), , , xlWhole, , , False).EntireColumn.Delete
  Next
  On Error GoTo 0
End Sub
