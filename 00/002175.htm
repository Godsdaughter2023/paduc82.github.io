Try this code which assumes the Employee column is column A. The code will test all the names in Col A and copy the data to new worksheets. the worksheet will be given the name of the employee.

Sub Copy_Data()
 Dim r As Range, LastRow As Long, ws As Worksheet
 Dim v As Variant, s As String, LastRow1 As Long
 Dim src As Worksheet
 'Change name to the worksheet with the data on
 Set src = Sheets("Sheet1")
 LastRow = src.Cells(Cells.Rows.Count, "A").End(xlUp).Row
 For Each r In src.Range("A2:A" & LastRow)
    
         On Error Resume Next
         Set ws = Sheets(CStr(r.Value))
         On Error GoTo 0
         If ws Is Nothing Then

Sub Demo()
    Dim Col As Collection               'Item, Key, Before, After
    Dim Emp                             'Employees
    Const MainWB As String = "Main"     'Workbook with Data
    Const Name = 0: Const WB = 1
    Const sTab As String = "Emp - #"    'Tab name for each Employee
    Set Col = New Collection
   
    Worksheets(MainWB).Activate
    'Turn Filter Mode off for now
    ActiveSheet.AutoFilterMode = False
   
    'Get Unique Names, & make sheet names:  {Name, SheetName}
    On Error Resume Next
    For Each Emp In [A1].CurrentRegion.Columns(1).Cells
        If Emp.Text <> vbNullString Then Col.Add _
            Array(Emp.Text, Replace(sTab, "#", Emp.Text)), Emp.Text
    Next Emp
    Col.Remove (1) 'Remove Header
   
    'Delete sheets that may already exist for each Employee
    Application.DisplayAlerts = False
    For Each Emp In Col
        ThisWorkbook.Sheets(Emp(WB)).Delete
    Next Emp
  
    'Make seperate sheets for each Employee
    For Each Emp In Col
        ThisWorkbook.Sheets.Add _
            (After:=Worksheets(Worksheets.Count)).Name = Emp(WB)
    Next Emp
   
    'Copy data to each workbook
    Worksheets(MainWB).Activate
    With Range("A1").CurrentRegion
        If Not ActiveSheet.AutoFilterMode Then .AutoFilter
        For Each Emp In Col
            .AutoFilter Field:=1, Criteria1:=Emp(Name)
            .SpecialCells(xlCellTypeVisible).Copy _
                Worksheets(Emp(WB)).Range("A1")
            Worksheets(Emp(WB)).UsedRange.EntireColumn.AutoFit
        Next Emp
    End With
   
    Worksheets(MainWB).Activate
    ActiveSheet.AutoFilterMode = False
End Sub
