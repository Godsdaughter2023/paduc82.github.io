greg maxey say: Sub ScratchMacro()
'A basic Word macro coded by Greg Maxey
Dim oRng As Word.Range
Dim oCC As ContentControl
  For Each oCC In ActiveDocument.Range.ContentControls
    Select Case oCC.Type
      Case 8
       
        If oCC.Tag = "No" Then
          'If the CC tag property is set to No use this.
          If oCC.Checked Then
            oCC.Range.Font.ColorIndex = wdBrightGreen
          Else
            oCC.Range.Font.ColorIndex = wdAuto
          End If
        Else
          'Otherwise ensure the CC and the word "No" is separated by a single space and use.
          Set oRng = oCC.Range
          With oRng
            .Collapse wdCollapseEnd
            .Move wdWord, 2
            .MoveEnd wdWord, 1
            If oRng.Text = "No" Then
              If oCC.Checked Then
                oCC.Range.Font.ColorIndex = wdBrightGreen
              Else
                oCC.Range.Font.ColorIndex = wdAuto
              End If
            End If
          End With
        End If
      Case 0, 1
        If Not oCC.ShowingPlaceholderText Then
          oCC.Range.Font.ColorIndex = wdBrightGreen
        Else
          oCC.Range.Style = "Placeholder Text"
        End If
    End Select
  Next oCC
lbl_Exit:
  Exit Sub
End Sub
