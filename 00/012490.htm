Assuming that the Score column contains the values 10, 20, 30 etc, you could use a series of nested If...then...Else field constructions that check for the value of the field and insert the appropriate number of tab spaces for each value

{ IF { MERGEFIELD Score } = 10 "[tab]{ MERGEFIELD Score }" {IF { MERGEFIELD Score } = 20 "[tab][tab]{ MERGEFIELD Score }" { IF { MERGEFIELD Score } = 30 "[tab][tab][tab]{ MERGEFIELD Score }" ...etc..."[tab][tab][tab][tab][tab][tab][tab][tab][tab][tab]{ MERGEFILD Score }"}}}}}}}}}

Press the Tab key to replace each [tab] in the above.

Sub CoffeeFilter()
    Dim N As Long, i As Long, s As String, j As Long
    N = Cells(Rows.Count, "A").End(xlUp).Row
    For i = 1 To N
        s = Cells(i, 1).Value
        For j = 1 To 5
            If InStr(1, s, "Text" & j) > 0 Then
                Cells(i, 1).EntireRow.Hidden = True
            End If
        Next j
    Next i
End Sub
