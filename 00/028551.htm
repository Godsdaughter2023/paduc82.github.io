tasos_k say:

Hi,

try this code in order to avoid empty rows

in .csv file

 

convert data from active sheet in a .csv file.

(export .csv file in same path with .xlsm file)

 

(save as your workbook with extension .xlsm - macros enabled)

 

 

Sub ADODB_SaveAsCSV()
'Nov 20 2013

'non empty rows in csv file
Dim flag As Boolean
Const myDelim As String = "," 
Dim WS As Worksheet
Set WS = ActiveSheet
Dim r As Long, c As Long, i As Long, j As Long
r = WS.Cells.Find(What:="*", SearchOrder:=xlByRows, SearchDirection:=xlPrevious).Row
c = WS.Cells.Find(What:="*", SearchOrder:=xlByColumns, SearchDirection:=xlPrevious).Column
Dim myPath As String
myPath = ThisWorkbook.Path
Dim myFile As String
myFile = myPath & "\" & ActiveSheet.Name & ".csv"

Dim obj As Object
Set obj = CreateObject("ADODB.Stream")
obj.Type = 2
obj.Charset = "utf-8"
obj.Open
Dim v() As Variant
ReDim v(1 To c)
For i = 1 To r
flag = False
For j = 1 To c
If Not IsEmpty(WS.Cells(i, j)) Then flag = True
v(j) = WS.Cells(i, j)
Next
If flag = True Then
obj.writetext Join(v, myDelim), 1
End If
Next
MsgBox "file is ready"
End Sub

 

 

Note 

i assume that rows, are entirely empty,
e.g. no formulas that return blank or zero
