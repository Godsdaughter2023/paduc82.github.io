Option 1: ONLY for NON-Contiguous multiple columns

Sub ColorDuplicates1()
'for NON-Contiguous multiple columns

On Error GoTo ErrHandler
Application.ScreenUpdating = False

Sheets("Sheet1").Activate
Dim rngData As Range, rng1 As Range, rng2 As Range, rng3 As Range, rngArea As Range, cell As Range, lColor As Long, lFirstRow As Long
Set rng1 = Range("D5:D" & ActiveSheet.Range("D" & Rows.count).End(xlUp).Row)
Set rng2 = Range("F5:F" & ActiveSheet.Range("F" & Rows.count).End(xlUp).Row)
Set rng3 = Range("H5:H" & ActiveSheet.Range("H" & Rows.count).End(xlUp).Row)

Set rngData = Union(rng1, rng2, rng3)
rngData.Interior.ColorIndex = xlNone
lFirstRow = rngData.Row

lColor = 3
For Each rngArea In rngData.Areas
    For Each cell In rngArea
        If Application.CountIf(rng1, cell) + Application.CountIf(rng2, cell) + Application.CountIf(rng3, cell) > 1 Then
            If Application.CountIf(rngArea.Resize(cell.Row - lFirstRow + 1), cell) = 1 Then
                cell.Interior.ColorIndex = lColor
                lColor = lColor + 1
                If lColor > 56 Then
                    MsgBox "Max ColorIndex values reached"
                    GoTo ErrHandler
                End If
            Else
                cell.Interior.ColorIndex = rngData.Find(What:=cell.Value, After:=rng3.Cells(rng3.Cells.count)).Interior.ColorIndex
            End If
        End If
    Next cell
Next rngArea
     
ErrHandler:
    Application.ScreenUpdating = True

End Sub

Here is a macro you can use:

Sub HighlightDup()
    Dim rngCheck As Range
    Dim rngArea As Range
    Dim rngCell As Range
    Dim strValue As String
    Dim colValues As New Collection
    Dim lngColor As Long
    Dim lngMaxColor As Long
    Dim lngCount As Long
    Set rngCheck = Range("D5:D14,F5:F14,H5:I14")
    rngCheck.Interior.ColorIndex = xlColorIndexNone
    lngMaxColor = 2
    For Each rngCell In rngCheck
        strValue = CStr(rngCell.Value)
        If strValue <> "" Then
            lngCount = 0
            For Each rngArea In rngCheck.Areas
                lngCount = lngCount + Application.CountIf(rngArea, strValue)
            Next rngArea
            If lngCount > 1 Then
                lngColor = 0
                On Error Resume Next
                lngColor = colValues(strValue)
                On Error GoTo 0
                If lngColor = 0 Then
                    lngMaxColor = lngMaxColor + 1
                    If lngMaxColor > 56 Then
                        MsgBox "Too many different duplicate values!", vbExclamation
                        Exit Sub
                    End If
                    rngCell.Interior.ColorIndex = lngMaxColor
                    colValues.Add Item:=lngMaxColor, Key:=strValue
                Else
                    rngCell.Interior.ColorIndex = lngColor
                End If
            End If
        End If
    Next rngCell
End Sub

How do I get MS Word 2007 to lock a graphic? - Microsoft Community

aug dog say:

How do I get MS Word 2007 to lock a graphic?

How do I get MS Word 2007 to lock a graphic so that it does not move with the text?  I already set the text to run over it.

jay freedman say:

From your description, I gather than the graphic's text wrapping is set to Behind text.

To make it stay in one position on the page, select the graphic (you may have to use the Home > Select > Select Object command to get it), go to the Picture Tools | Format tab, click the Position button, and choose More Layout Options at the bottom of the menu.

In the Advanced Layout dialog box, change the vertical absolute position's reference point from Paragraph to Margin.

Click OK. If you want to change the position of the graphic once you see it in place, reopen the dialog and change the number.

You could alternatively change the reference point to Page, but then you would have to add the size of the top margin area to the numerical position (for example, with a 1" margin, the value shown above would have to be changed to 1.64" relative to Page).

Note: If the paragraph to which the graphic was initially anchored moves to another page because of editing, the picture will also move to the same page. A graphic must always be on the same page as the anchor point. You can drag the anchor to a new position if necessary, or cut/paste the graphic.

anastasiya ilgova say:

Hi all, after 2 days of searching a solution I finally did it - One Drive is in System Tray after reinstall. The only solution that helped me was here: https://answers.microsoft.com/en-us/msoffice/forum/msoffice_onedrivefb-mso_win10-mso_o365b/onedrive-doesnt-start-in-windows-10/b5c35700-de74-4a06-a912-55c9779224a6

1. Check Registry.

    a)  Press (Windows key) + R to open the Run box.

    b)  Type regedit and click OK

    c)  Navigate to HKEY_LOCAL_MACHINE\Software\Policies\Microsoft\Windows\OneDrive. Make sure      DisableFileSyncNGSC is set to "0". (It is also OK if the registry doesn’t exist,)

2. Check Group Policy settings.

    a)  Press (Windows key) + R to open the Run box.

    b)  Type gpedit.msc and click OK.

    c)  In the Local Group Policy Editor, in the folder list under Local Computer Policy, navigate to Local Computer Policy > Computer Configuration > Administrative Templates > Windows Components > OneDrive > Prevent the usage of OneDrive for file storage. Make sure it is set to "Not configured" or "Disabled".

3. Uninstall the OneDrive sync client

    a)  Right click the Start menu on your Windows then select Run.

    b)  Enter cmd then click OK

    c)  Type in the following command and hit Enter to uninstall OneDrive.
For 32-bit Windows 10: %SystemRoot%\System32\OneDriveSetup.exe /uninstall
For 64-bit Windows 10: %SystemRoot%\SysWOW64\OneDriveSetup.exe /uninstall

4. Install the OneDrive sync client

    a)  Right click the Start menu on your Windows then select Run.

    b)  Enter cmd then click OK

    c)  Type in the following command and hit Enter to uninstall OneDrive.
For 32-bit Windows 10: %SystemRoot%\System32\OneDriveSetup.exe /install
For 64-bit Windows 10: %SystemRoot%\SysWOW64\OneDriveSetup.exe /install

After step 4 I was able to see One Drive in my system tray after install....!!!
