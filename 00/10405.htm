Thanks to anyone who stared at this in disbelief :)  but I think I have solved it.  In case anyone ever needs it, here's the code.

Sub Expand()

'

' Expand Macro

'

' Keyboard Shortcut: Option+Cmd+e (Ctrl+e?)

'Allow only one Expanded cell at a time

Dim ExpdAddr As String, ExpdAddr2 As String, ExpdAddrOld As String


    Worksheets("Worksheet").Activate

    ExpdAddr = ActiveCell.Address  'Cell to Expand

    If IsEmpty(Range("G1").Value) = False Then 'Expanded Cell Exists

        'Shrink the Expanded Cell before Epanding a new one

        ExpdAddrOld = Range("G1").Value 'Prviously expanded cell saved in G1

        Range(ExpdAddrOld).Select

        Selection.UnMerge

        ActiveCell.EntireRow.Offset(1, 0).Delete

        Range(ExpdAddrOld).Select

        Selection.RowHeight = 25

        Selection.ColumnWidth = 25

   End If

        'Expand Cell for viewing

        Range("G1").Value = ExpdAddr 'New Expanded Field Flag Set

        Range(ExpdAddr).Select

        ActiveCell.EntireRow.Offset(1, 0).Insert

        ActiveCell.Offset(1, 0).Select

        ExpdAddr2 = ActiveCell.Address

        Range(ExpdAddr & ":" & ExpdAddr2).Select

        Selection.Merge

        With Selection  'Format for Reading

            .HorizontalAlignment = xlLeft

            .VerticalAlignment = xlCenter

            .WrapText = True

            .Orientation = 0

            .AddIndent = False

            .IndentLevel = 0

            .ShrinkToFit = False

            .ReadingOrder = xlContext

            .ColumnWidth = 50

            .RowHeight = 400

        End With

       

End Sub

Sub Shrink()

'

' Shrink Macro

'

' Keyboard Shortcut: Option+Cmd+s  (Ctrl s?)

'

    Dim ExpdAddr As String, ExpdAddr2 As String


    Worksheets("Worksheet").Activate

    If IsEmpty(Range("G1").Value) = True Then  'Dont do anything if the G1 Expanded Flag is not set

        MsgBox "No Expanded Field"

   Else

        ExpdAddr = Range("G1").Value

        Range(ExpdAddr).Select

        Selection.UnMerge

        ActiveCell.EntireRow.Offset(1, 0).Delete

        Range(ExpdAddr).Select

        Selection.RowHeight = 25

        Selection.ColumnWidth = 25

        Range("G1").ClearContents

    End If

End Sub

Dim strFilter As String

steve rindsberg say:

You mentioned that you saved the slide master.  That makes me wonder if you're opening and changing the template your presentation is based on and saving IT.

That'll only affect the template (ie, the master/s) and any new presentations based on it, and in some cases, presentations that you re-apply the template to.

To change the current presentation, you need to do View | Slide Master and make your changes there.  You may need to change the main master (the thumbnail at the very top in Slide Master view) or the individual layout that the current slide is based on.  Or both.

Finally, for things like text formatting, if the text on the individual slide has been reformatted, PowerPoint figures that you overrode the master or layout formatting for a reason and won't update the slide's formatting when you change the master unless you use the REAPPLY command in the Layout area of the ribbon.
