garsularpit say:

vba hyperlink a series of terms to bookmarks

I am trying to hyperlink a series of string terms found in a document to bookmarks in the same document.  The search terms found in a selection would be non-bold and some have apostrophes.  The string terms from "FndString" below would hyperlink to the Bookmarks in "BkmString".  For example, "TermOne" hyperlinks to the bookmark name of "TermOne_Details", "TermTwo's" hyperlinks to the bookmark name of "TermTwo's_Details", and so forth.  Also, "TermOne" is the display text of the Bookmark name of "TermOne_Details", "TermTwo's" is the display text of the Bookmark name of "TermTwo's_Details", etc.. 

Any help is appreciated.

Sub HyperlinkToBookmark()

Dim r As range
Dim rDup As range
Dim FndString As Variant
Dim BkmString As Variant
Dim i As Long
Set r = Selection.range
Set rDup = Selection.range
FndString = Split("TermOne,TermTwo's," _
& "TermThree," _
& "Termfour,", ",")

BkmString = Split("TermOne_Details,TermTwo's_Details," _
& "TermThree_Details," _
& "TermFour_Details,", ",")

For i = LBound(FndString) To UBound(FndString)
With r.Find
        .ClearFormatting
        .Replacement.ClearFormatting
        .Text = ""
        .Replacement.Text = ""
        .Forward = True
        .Wrap = wdFindContinue
        .format = True
        .MatchCase = False
        .MatchWholeWord = True
        .MatchWildcards = True
        .MatchSoundsLike = False
        .MatchAllWordForms = False
    End With
    Selection.Find.Execute
    With ActiveDocument.Bookmarks
        .DefaultSorting = wdSortByLocation
        .ShowHidden = False
    End With
      With r.Find
        .ClearFormatting
        .Replacement.ClearFormatting
        .Text = FndString(i)
        .Replacement.Text = ""
        .Font.Bold = False
        .Forward = True
        .Wrap = wdFindContinue
        .format = False
        .MatchCase = False
        .MatchWholeWord = True
        .MatchWildcards = True
        .MatchSoundsLike = False
        .MatchAllWordForms = False
    End With
        Selection.Find.Execute
    Do While Selection.Find.Execute
    ActiveDocument.Hyperlinks.Add Anchor:=r, SubAddress:=BkmString(i), target:=BkmString(i), ScreenTip:="", TextToDisplay:=FndString(i)
    Loop
Next
rDup.Select
msgBox "Done"

End Sub
