Welcome to Microsoft Answers Forum!

1. The behavior you are experiencing is by design. The 2007 Microsoft Office system intentionally sets a Browser Flags registry value for its file types to force the documents to open within the Office application window and not inside the Internet Explorer window.

2. Certain editing features are available only when the document is open outside the Web browser.

3. Additionally, the Office Ribbon may not work as expected in Internet Explorer. opulent say:

Eg: if 3, 31, 48 (or any combination up to 5 numbers) have been drawn in the same draw, 3 times in different draws. How can i identify them over this period?

What would be the easiest way to resolve this query?

Thank you in advance.

andreas killer say:
What would be the easiest way to resolve this query?

This is a very huge work, it may take several hours to execute, it is not easy and it is so.... excuse me, silly to do that.

Never mind, it can be done and it might be a good example for other reasonable purposes.

I assume that your numbers are stored in column A:E

To determine if a combination of your 3 numbers can be found is relatively easy, we can search for any number, store the row of that number. After all rows are found for the first number, we can search this rows for the second number. And so on until the last number. If any number isn't found in this process the combination can not be found.

Execute Sub Example_SearchCombinations() for testing this.

The next one (create combinations of 2, 3, or 4 numbers) is harder to do.

Execute Sub Example_Combinations() for testing this.

The whole process is a combination of all this. First we must build a unique list of all numbers, then create all 4,3,2 combinations of that list and search each combination in each row.

I suggest that you start Sub Main(), go to bed and see if it is finished morning.

Andreas.

Option Explicit

Sub Example_SearchCombinations()
  Dim InputRange As Range, R As Range
  'Locate the numbers
  Set InputRange = Range(Range("A1"), Range("E" & Rows.Count).End(xlUp))
  'Search for 3 numbers
  Set R = SearchCombinations(InputRange, Array(3, 31, 48))
  If R Is Nothing Then
    MsgBox "Combination not found."
  Else
    R.Select
  End If
End Sub

Function SearchCombinations(ByVal R As Range, Numbers) As Range
  'Search numbers in R, return found rows that include all numbers
  Dim NumPos As Range, C As Range
  Dim FirstAddress As String
  Dim I As Integer
  'Search each number
  For I = LBound(Numbers) To UBound(Numbers)
    'Search it
    Set C = R.Find(Numbers(I), LookIn:=xlValues, LookAt:=xlWhole)
    If C Is Nothing Then
      'combination is not found, abort
      Set SearchCombinations = Nothing
      Exit Function
    Else
      'Save first position
      FirstAddress = C.Address
      'Find all positions
      Set NumPos = Nothing
      Do
        'Remember the position of each number
        If NumPos Is Nothing Then
          Set NumPos = C
        Else
          Set NumPos = Union(C, NumPos)
        End If
        'Search next position
        Set C = R.FindNext(C)
      Loop Until C.Address = FirstAddress
      If SearchCombinations Is Nothing Then
        'First number, store all found rows
        Set SearchCombinations = Intersect(NumPos.EntireRow, R)
      Else
        'Store remaining rows
        Set SearchCombinations = Intersect(NumPos.EntireRow, SearchCombinations)
      End If
      'Reduce the search range to our rows for speed up
      Set R = SearchCombinations
    End If
  Next
End Function

Sub Example_Combinations()
  Dim C As Collection, Item
  Set C = Combinations(Array("A", "B", "C", "D"), 3)
  For Each Item In C
    Debug.Print Join(Item, "")
  Next
End Sub

Function Combinations(Arr, ByVal Count As Integer) As Collection
  'Return all possible combinations of count elements in Arr as arrays in a collection
  Dim Index() As Long
  Dim Result()
  Dim I As Long, J As Integer
  'Create a collection
  Set Combinations = New Collection
  'Return empty collection if Count exceed the bound's of Arr
  If LBound(Arr) + Count - 1 > UBound(Arr) Or Count < 1 Then Exit Function
  'Setup space for index and result array
  ReDim Index(1 To Count) As Long
  ReDim Result(1 To Count)
  'Setup index for first combination
  For I = 1 To Count
    Index(I) = LBound(Arr) + I - 1
  Next
  Do
    'Build combination
    For I = 1 To Count
      Result(I) = Arr(Index(I))
    Next
    'Save it
    Combinations.Add Result
    'Get next index
    I = Count
    J = 0
    Do
      'Max. position for this index reached?
      If Index(I) = UBound(Arr) - Count + I Then
        J = J + 1
        'Move index before up one step in next loop
        I = I - 1
        If I < 1 Then Exit Function
      Else
        'Increment position
        Index(I) = Index(I) + 1
        'Setup next indices
        For J = 1 To J
          I = I + 1
          Index(I) = Index(I - 1) + 1
        Next
        Exit Do
      End If
    Loop
  Loop
End Function

Sub Main()
  Dim InputRange As Range, R As Range, A As Range
  Dim Numbers As Object 'Dictionary
  Dim NumItems, Item
  Dim Combination As Collection
  Dim I As Integer
  Dim Count As Long
  Dim OutputRange As Range
  'Locate the numbers
  Set InputRange = Range(Range("A1"), Range("E" & Rows.Count).End(xlUp))
  'Result goes left and down from this cell
  Set OutputRange = Range("G1")
  OutputRange.Resize(1, 5) = Array("Count", "Nr1", "Nr2", "Nr3", "Nr4")
  'Build a unique list of numbers
  Set Numbers = CreateObject("Scripting.Dictionary")
  For Each R In InputRange
    If Not Numbers.Exists(R.Value) Then Numbers.Add R.Value, R.Value
  Next
  'Build array of unique numbers
  NumItems = Numbers.Items
  'Build combinations from 4 downto 2
  For I = 4 To 2 Step -1
    'Get possible combinations
    Set Combination = Combinations(NumItems, I)
    'Search for each
    For Each Item In Combination
      Set R = SearchCombinations(InputRange, Item)
      If R Is Nothing Then
        'MsgBox "Combination not found."
      Else
        'Count found rows
        Count = 0
        For Each A In R.Areas
          Count = Count + A.Rows.Count
        Next
        If Count > 1 Then
          Set OutputRange = OutputRange.Offset(1, 0)
          'Store result in sheet
          OutputRange = Count
          OutputRange.Offset(0, 1).Resize(1, UBound(Item)) = Item
        End If
      End If
    Next
  Next
End Sub
