gord dibben say:

Try this on a copy of the sheet.   Worked for me with your example date.

Sub ReOrganize()
'JBeaucaire  (11/4/2009)
'Turns row data into columnar data
' a  1   2   3
' b  7   8
' c  4   5   6   9
'into
' a  1
' a  2
' a  3
' b  7
' b  8
' c  4
' c  5
' c  6
' c  9
Dim LR As Long, i As Long, r As Long, c As Long, v As Long
Application.ScreenUpdating = False
LR = Range("A" & Rows.Count).End(xlUp).Row
i = 1

Do Until Range("A" & i) = ""
    If Range("C" & i) <> "" Then
        c = Cells(i, Columns.Count).End(xlToLeft).Column
        v = i
        For r = 3 To c
            i = i + 1
            Rows(i).Insert xlShiftDown
            Range("A" & i) = Range("A" & i - 1)
            Range("B" & i) = Cells(v, r)
        Next r
    End If
    i = i + 1
Loop

Range("C1", Cells(Rows.Count, Columns.Count)).ClearContents
Application.ScreenUpdating = True
End Sub

-----------------------------------
<Configuration>

       <ExcludeApp ID=”Outlook” />
    </Product>
  </Add>
  <Updates Enabled="TRUE" Channel="Current" />
  <Display Level="None" AcceptEULA="TRUE" />
</Configuration>
