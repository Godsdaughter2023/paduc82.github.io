Sub ReplaceText()
 
    Dim oSld As Slide
    Dim oShp As Shape
    Dim oTxtRng As TextRange
    Dim oTmpRng As TextRange
    Dim strFind As String, strReplace As String
    Set oSld = Application.ActivePresentation.Slides(2)
    strFind = "    "
    strReplace = " "
    For Each oShp In oSld.Shapes
        Set oTxtRng = oShp.TextFrame.TextRange
        Set oTmpRng = oTxtRng.Replace(FindWhat:=strFind, Replacewhat:=strReplace, WholeWords:=False)
        Do While Not oTmpRng Is Nothing
            Set oTxtRng = oTxtRng.Characters(oTmpRng.Start + oTmpRng.Length, oTxtRng.Length)
            Set oTmpRng = oTxtRng.Replace(FindWhat:=strFind, Replacewhat:=strReplace, WholeWords:=False)
        Loop
    Next oShp


End Sub

Sub ReplaceText()
    Dim oSld As Slide

    Dim oTxtRng As TextRange

I'm not aware of any specified limit (http://support.microsoft.com/kb/211489) but, as you can see, a great deal of processing can be required for complex tables, especially if they've been left with the AutoFit attribute set to True.
