andrea jones - all about resources say:

You could create this macro to set a custom document property called AppendixCount containing the number of paragraphs formatted with that style

 

Sub countappends()
    Dim intCount As Integer
    Dim styleCount As Integer
    Dim intParaCount As Integer
    intCount = 1
    styleCount = 0
    With ThisDocument
        Do
            If .Paragraphs(Index:=intCount).Style = "Heading 7" Then
            styleCount = styleCount + 1
            End If
            intCount = intCount + 1
            intParaCount = .Paragraphs.Count
        Loop Until intCount = intParaCount
With ActiveDocument.CustomDocumentProperties
    .Add Name:="AppendixCount", _
        LinkToContent:=False, _
        Type:=msoPropertyTypeNumber, _
        Value:=styleCount
    End With
    End With
End Sub
