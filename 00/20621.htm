a) You can't compare arrays in that way, you have to compare each item in each array.

b) Well, not really. We can use a FOR loop, but the result is not faster, only shorter written (and more complicated to read for a human).

But we can make it much faster if we separate the FOR loops and use a dictionary to store the row number of one array associated with an unique key build from the columns. After that you have to "walk" through the 2nd array only once to find the rows.

And you image shows that the data is contiguous, so you can use CurrentRegion to find the cells. Much simpler.

Andreas.

Public Sub Plan_Into_RawData()
 
  '------------------------ Read Excel ranges into Arrays -----------------
 
  Dim rngPlan As Range, rngRawData As Range
  Dim arrPlan As Variant, arrRawData As Variant
 
  Set rngPlan = Sheet1.Range("A1").CurrentRegion
  arrPlan = rngPlan.Value
 
  Set rngRawData = Sheet2.Range("A1").CurrentRegion
  arrRawData = rngRawData.Value

  '------------------------ Compare arrays, write amounts ---------------

  Dim i As Long, j As Long
  Dim Dict As Object 'Scripting.Dictionary
  Dim Key As String
  Dim Arr(1 To 5), Item
 
  'Create a dictionary for faster access
  Set Dict = CreateObject("Scripting.Dictionary")
  Dict.CompareMode = vbTextCompare

  For i = LBound(arrRawData, 1) + 1 To UBound(arrRawData, 1)
    'Build the key from the columns
    j = 0
    For Each Item In Array(1, 2, 3, 4, 5)
      j = j + 1
      Arr(j) = arrRawData(i, Item)
    Next
    Key = Join(Arr, ";")
    'Be sure it's unique!
    If Dict.Exists(Key) Then
      MsgBox "Duplicate data in row " & i
      Exit Sub
    End If
    'Store the key and the row number
    Dict.Add Key, i
  Next

  For i = LBound(arrPlan, 1) + 1 To UBound(arrPlan, 1)
    'Build the key from the columns
    j = 0
    For Each Item In Array(1, 6, 7, 8, 10)
      j = j + 1
      Arr(j) = arrPlan(i, Item)
    Next
    Key = Join(Arr, ";")
    'Search in the dictionary
    If Not Dict.Exists(Key) Then
      If MsgBox(Key & " not found. Abort?", vbOKCancel) = vbCancel Then Exit Sub
    End If
    'Get the row number
    j = Dict.Item(Key)
    
    'Update the data
    arrRawData(j, 12) = arrPlan(i, 6)
    arrRawData(j + 1, 12) = arrPlan(i, 7)
    arrRawData(j + 2, 12) = arrPlan(i, 8)
    arrRawData(j + 3, 12) = arrPlan(i, 9)
    arrRawData(j + 4, 12) = arrPlan(i, 10)
    arrRawData(j + 5, 12) = arrPlan(i, 11)
    arrRawData(j + 6, 12) = arrPlan(i, 12)
    arrRawData(j + 7, 12) = arrPlan(i, 13)
    arrRawData(j + 8, 12) = arrPlan(i, 14)
    arrRawData(j + 9, 12) = arrPlan(i, 15)
    arrRawData(j + 10, 12) = arrPlan(i, 16)
    arrRawData(j + 11, 12) = arrPlan(i, 17)
  Next
 
  '------------------------ Write Arrays back to the sheet -----------------
 
  rngRawData.Value = arrRawData
End Sub
