Sincerely,

 

Melanie

tasos_k say: Hi Melanie,
I assume that you need cut and paste.
try this code..
(change Sheet1 name as needed)

Sub myCode2()
Dim Ws As Worksheet, newWs As Worksheet
Application.ScreenUpdating = False
Set Ws = Sheets("Sheet1")
Set newWs = Worksheets.Add(Before:=Worksheets(1))
Ws.Columns("A:A").Cut Destination:=newWs.Range("C1")
Ws.Columns("M:M").Cut Destination:=newWs.Range("D1")
Ws.Columns("R:R").Cut Destination:=newWs.Range("E1")
Ws.Columns("O:O").Cut Destination:=newWs.Range("F1")
Ws.Columns("C:C").Cut Destination:=Ws.Range("H1")
Ws.Columns("T:T").Cut Destination:=Ws.Range("I1")
Ws.Columns("X:X").Cut Destination:=Ws.Range("K1")
Ws.Columns("Y:Y").Cut Destination:=Ws.Range("L1")
Ws.Columns("V:V").Cut Destination:=Ws.Range("M1")
Application.ScreenUpdating = True
End Sub
