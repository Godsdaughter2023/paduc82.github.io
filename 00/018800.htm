mike h.. say:

Hi,

Try this which inserts the row and fills the blank.

Sub Insert_FillBlanks1()
Dim Area As Range, LastRow As Long
Dim MyColumn As String, x As Long
MyColumn = "A"
For x = Cells(Rows.Count, MyColumn).End(xlUp).Row To 2 Step -1
    If Cells(x - 1, MyColumn) <> Cells(x, MyColumn) Then Rows(x).Insert
Next x
On Error Resume Next
  LastRow = Cells(Cells.Rows.Count, MyColumn).End(xlUp).Row
  For Each Area In Range(MyColumn & "1:" & MyColumn & LastRow). _
                   SpecialCells(xlCellTypeBlanks).Areas
    Area.Value = Area(1).Offset(-1).Value
  Next
End Sub
