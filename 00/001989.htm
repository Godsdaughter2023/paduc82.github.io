Something along the lines of...

On a Command Button...

Private Sub cmdSendToExcel_Click()

        Dim dbs As DAO.Database
        Dim qryDef As DAO.QueryDef
        Dim strSQL As String
        Dim strWhere As String
        Dim lngLen As Long
        Set dbs = CurrentDb

    strSQL = "SQL goes here..."

    'Text field example. Use Like to find anywhere in the field.
    If Not IsNull(Me.ControlOnForm) Then
        strWhere = strWhere & "([FieldFromRecordSource] = """ & Me.ControlOnForm & """) AND "
    End If

    'Yes/No field and combo example. If combo is blank or contains "ALL", we do nothing.
    If Me.ControlOnForm = True Then
        strWhere = strWhere & "([FieldFromRecordSource] = True) AND "
    ElseIf Me.cboRailCar = 0 Then
        strWhere = strWhere & "([FieldFromRecordSource] = False) AND "
    End If

    'Number field example. Do not add the extra quotes.
    If Not IsNull(Me.ControlOnForm) Then
        strWhere = strWhere & "([FieldFromRecordSource] = " & Me.ControlOnForm & ") AND "
    End If

    lngLen = Len(strWhere) - 5

    If lngLen <= 0 Then
        strQueryWhere = Left$(strQueryWhere, lngLen)
        strSQL = strSQL & " WHERE " & strQueryWhere & strOrderBy
        Set qryDef = dbs.CreateQueryDef("qryExportToExcel", strSQL)
        'DoCmd.OpenQuery qryDef.Name
        qryDef.Close
        Set qryDef = Nothing
        DoEvents
        Call SendToExcel("qryExportToExcel", "OpenLifts")
        DoEvents
        DoCmd.DeleteObject acQuery, "qryExportToExcel"
    Else
        strWhere = Left$(strWhere, lngLen)
        strSQL = strSQL & " WHERE " & strQueryWhere & " AND " & strWhere & strOrderBy
        Set qryDef = dbs.CreateQueryDef("qryExportToExcel", strSQL)
        'DoCmd.OpenQuery qryDef.Name
        qryDef.Close
        Set qryDef = Nothing
        DoEvents
        Call SendToExcel("qryExportToExcel", "OpenLifts")
        DoEvents
        DoCmd.DeleteObject acQuery, "qryExportToExcel"
    End If

        dbs.Close
        Set dbs = Nothing
End Sub
