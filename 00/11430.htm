andreas killer say:
You have to look how many lines that start with a blank follows each other. Based on this result you can parse the lines.

Andreas.

Sub Test()
  Dim fso As Object 'Scripting.FileSystemObject
  Dim ts As Object 'Scripting.TextStream
  Dim Contents
  Dim i As Long, j As Long, f As Long, t As Long, b As Integer
 
  'Read the file
  Set fso = CreateObject("Scripting.FileSystemObject")
  Set ts = fso.OpenTextFile("c:\test.txt")
  Contents = ts.ReadAll
  ts.Close
 
  'Split the string into lines
  Contents = Split(Contents, vbCrLf)
 
  'Find the first useable line
  For i = 0 To UBound(Contents)
    If Left(Contents(i), 1) = "-" Then Exit For
  Next
  f = i + 1
  'Find the last useable line
  For j = UBound(Contents) To 0 Step -1
    If Len(Contents(j)) > 0 Then Exit For
  Next
  t = j - i
 
  'Parse anything between
  For i = f To t
    'Count the following lines that start with a blank
    b = 0
    For j = i + 1 To t
      If Left(Contents(j), 1) = " " Then
        b = b + 1
      Else
        Exit For
      End If
    Next
    Select Case b
      Case 0
        GoSub Parse1Line
      Case 1
        If Len(Contents(i + 1)) > 13 Then
          GoSub Parse2LinesA
        Else
          GoSub Parse2LinesB
        End If
        i = i + 1
      Case 2
        GoSub Parse3Lines
        i = i + 2
    End Select
  Next
  Exit Sub
 
Parse1Line:
  'One line with all
  Debug.Print "Name", Trim(Left(Contents(i), 13))
  Debug.Print "Path", Trim(Mid(Contents(i), 14, 32))
  Debug.Print "Description", Trim(Mid(Contents(i), 46))
  Return
Parse2LinesA:
  'First line contains Name and Path
  Debug.Print "Name", Trim(Left(Contents(i), 13))
  Debug.Print "Path", Trim(Mid(Contents(i), 14))
  Debug.Print "Description", Trim(Mid(Contents(i + 1), 46))
  Return
Parse2LinesB:
  'Second line contains Path and Description
  Debug.Print "Name", Trim(Left(Contents(i), 13))
  Debug.Print "Path", Trim(Mid(Contents(i + 1), 14, 32))
  Debug.Print "Description", Trim(Mid(Contents(i + 1), 46))
  Return
Parse3Lines:
  'Second line contains Path and Description
  Debug.Print "Name", Trim(Left(Contents(i), 13))
  Debug.Print "Path", Trim(Mid(Contents(i + 1), 14))
  Debug.Print "Description", Trim(Mid(Contents(i + 2), 46))
  Return
End Sub

The NetShareEnum solution is much more better. I have refined and shorten the code from the link that you've posted, the code below returns all informations for all levels in a collection filled with zero based arrays.

The Sub Example_NetShare should do what you want, the Name is in Item(0), the Path is in Item(6) and the description is in Item(2).

Private Declare Function NetShareEnum Lib "Netapi32.dll" ( _
    ByVal ServerName As Long, ByVal Level As Long, ByVal BufPtr As Long, ByVal _
    PrefMaxLen As Long, ByRef EntriesRead As Long, ByRef TotalEntries As Long, _
    ByRef Resume_Handle As Long) As Long

Private Sub Example_NetShare()
  Dim Item, Level As Long

  'http://msdn.microsoft.com/en-gb/library/windows/desktop/bb525387%28v=vs.85%29.aspx
  '
  'Remarks: If you call NetShare with a server name:
  '
  'For interactive users (users who are logged on locally to the machine), no special _
    group membership is required to execute the NetShareEnum function.
  'For non-interactive users, Administrator, Power User, Print Operator, or Server _
    Operator group membership is required to successfully execute the NetShareEnum _
  function at levels 2, 502, and 503. No special group membership is required for level _
    or level 1 calls.

  Level = 2
  Debug.Print
  Debug.Print "Level " & Level
  For Each Item In NetShare("", Level)
    Debug.Print Join(Item)
  Next

  '  Debug.Print
  '  For Each Item In NetShare("\\server", 1)
  '    Debug.Print Join(Item)
  '  Next
End Sub

Function NetShare(ByVal Server As String, Optional ByVal Level As Long = 1) As Collection
  Dim Result As Long, Count As Long, Total As Long
  Dim Buf As Long, ResumeHandle As Long, i As Long, n As Long, Ofs As Long
  Dim This(), Size As Long

  Set NetShare = New Collection

  'Setup the number of items in the structure
  Select Case Level
    Case 0:    n = 1
    Case 1:    n = 3
    Case 2:    n = 8
    Case 502:  n = 10
    Case 503:  n = 11
    Case Else: Exit Function
  End Select
  'Calculate the size of the structure
  Size = n * 4
  'Create space for the output
  ReDim This(0 To n - 1)

  NetShareEnum StrPtr(Server), Level, VarPtr(Buf), -1, Count, Total, ResumeHandle
  For i = 0 To Count - 1
    'Note: Any item in the structure is a pointer to the value (DWORD) or to the _
      pointer of a string (LPWSTR)!

    'name
    Ofs = 0: n = 0
    This(n) = ReadStrPtr(ReadLongPtr(Buf + Size * i + Ofs))
    If Level = 0 Then GoTo AddThis

    'type
    Ofs = Ofs + 4: n = n + 1
    This(n) = ReadLongPtr(Buf + Size * i + Ofs)
    'remark
    Ofs = Ofs + 4: n = n + 1
    This(n) = ReadStrPtr(ReadLongPtr(Buf + Size * i + Ofs))
    If Level = 1 Then GoTo AddThis

    'permissions
    Ofs = Ofs + 4: n = n + 1
    This(n) = ReadLongPtr(Buf + Size * i + Ofs)
    'max_uses
    Ofs = Ofs + 4: n = n + 1
    This(n) = ReadLongPtr(Buf + Size * i + Ofs)
    'current_uses
    Ofs = Ofs + 4: n = n + 1
    This(n) = ReadLongPtr(Buf + Size * i + Ofs)
    'path
    Ofs = Ofs + 4: n = n + 1
    This(n) = ReadStrPtr(ReadLongPtr(Buf + Size * i + Ofs))
    'passwd
    Ofs = Ofs + 4: n = n + 1
    This(n) = ReadStrPtr(ReadLongPtr(Buf + Size * i + Ofs))
    If Level = 2 Then GoTo AddThis

    If Level = 502 Then
      'reserved
      Ofs = Ofs + 4: n = n + 1
      This(n) = ReadLongPtr(Buf + Size * i + Ofs)
      'security_descriptor
      Ofs = Ofs + 4: n = n + 1
      This(n) = ReadLongPtr(Buf + Size * i + Ofs)
    Else 'Level = 503
      'servername
      Ofs = Ofs + 4: n = n + 1
      This(n) = ReadStrPtr(ReadLongPtr(Buf + Size * i + Ofs))
      'reserved
      Ofs = Ofs + 4: n = n + 1
      This(n) = ReadLongPtr(Buf + Size * i + Ofs)
      'security_descriptor
      Ofs = Ofs + 4: n = n + 1
      This(n) = ReadLongPtr(Buf + Size * i + Ofs)
    End If

AddThis:
    NetShare.Add This
  Next
  If Buf <> 0 Then NetApiBufferFree Buf
End Function

Private Function ReadLongPtr(ByVal Pointer As Long) As Long
  'Reads a long from a pointer
  CopyMemory ReadLongPtr, ByVal Pointer, 4
End Function

Private Sub Example_ReadStrPtr()
  Dim S As String, L As Long
  S = "Andreas Killer"
  L = StrPtr(S)
  Debug.Print ReadStrPtr(L)
End Sub

Private Function ReadStrPtr(ByVal Pointer As Long) As String
  'Reads a string from a pointer
  Dim Length As Long, Buf() As Byte
  If Pointer = 0 Then Exit Function
  Length = lstrlenW(Pointer) * 2
  If Length = 0 Then Exit Function
  ReDim Buf(Length - 1) As Byte
  CopyMemory Buf(0), ByVal Pointer, Length
  ReadStrPtr = Buf
End Function
