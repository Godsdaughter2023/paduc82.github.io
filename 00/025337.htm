ossiemac say:

Hello Leonard,

What about just selecting the data in the column then Copy -> select a destination -> Paste Special -> Tranpose.

Note that the destination cannot overlap the source with this action. If Data starts in cell A1 then A1 cannot be the destination; select b1 for the destination. Alternatively paste to a separate worksheet.

Following code will loop through the data and will paste to a separate worksheet. See my comments in the code where you might need to edit the code.

Bold part in comments edited since initial posting.

Sub CopyAndTranspose()

    Dim wsSource As Worksheet
    Dim wsDestin As Worksheet
    Dim lngLastRow As Long
    Dim rngDestin As Range
    Dim r As Long
   
    Set wsSource = Worksheets("Sheet1")     'Edit "Sheet1" to source sheet name
   
    Set wsDestin = Worksheets("Sheet2")     'Edit "Sheet2" to destination sheet name
   
    With wsSource
        'Next line assumes data is in column A otherwise Edit "A" to required column
        lngLastRow = .Cells(.Rows.Count, "A").End(xlUp).Row
    End With
   
    With wsDestin
        Set rngDestin = .Cells(.Rows.Count, "A").End(xlUp).Offset(1, 0)
    End With
   
    With wsSource
        'Following assumes column headers on source sheet and
        'data starts at row 2. (Edit 2 to 1 if data starts on row 1)
        For r = 2 To lngLastRow Step 7      'Edit 7 to number of rows of data
            .Range(.Cells(r, "A"), .Cells(r + 6, "A")).Copy     'Edit 6 to one less than number of row of data
            rngDestin.PasteSpecial Transpose:=True
            Set rngDestin = rngDestin.Offset(1, 0)
        Next r
    End With
   
End Sub

Try as follows:

  • Alt-F11 to open the VBA editor
  • Alt-IM to insert a new code module
  • In the new module, paste the following code

'=========>>
Option Explicit

'--------->>
Public Sub Tester()
    Dim WB As Workbook
    Dim srcSH As Worksheet, destSH As Worksheet
    Dim Rng As Range
    Dim arrIn As Variant, arrOut() As Variant
    Dim LRow As Long
    Dim i As Long, j As Long, k As Long, iBound As Long
    Const iDataRows As Long = 7

    Set WB = ActiveWorkbook

    With WB
        Set srcSH = .Sheets("Sheet1")                                      '<<=== Change to suit
        Set destSH = .Sheets("Sheet2")                                    '<<=== Change to suit
    End With

    With srcSH
        LRow = LastRow(srcSH, .Columns("A:A"))
        Set Rng = .Range("A1:A" & LRow)
    End With

    arrIn = Rng.Value
    iBound = UBound(arrIn, 1)
    ReDim arrOut(1 To (iBound \ 7), 1 To iDataRows)

    For i = 1 To LRow Step iDataRows
        k = k + 1
        For j = 1 To iDataRows
            arrOut(k, j) = arrIn(i + j - 1, 1)
        Next j
    Next i

    With destSH
        .UsedRange.ClearContents
        .Range("A1:G1").Value = Array("Date", "Time", "Type", "Number", _
                                      "Duration", "Charged", "Cost")
        .Range("A2").Resize(k, 7).Value = arrOut
    End With
End Sub

'--------->>
Public Function LastRow(SH As Worksheet, _
                        Optional Rng As Range)
    If Rng Is Nothing Then
        Set Rng = SH.Cells
    End If

    On Error Resume Next
    LastRow = Rng.Find(What:="*", _
                       after:=Rng.Cells(1), _
                       Lookat:=xlPart, _
                       LookIn:=xlFormulas, _
                       SearchOrder:=xlByRows, _
                       SearchDirection:=xlPrevious, _
                       MatchCase:=False).Row
    On Error GoTo 0
End Function

Is there a way to do what I want to do, without printing each sheet out individually?

jllatham say:

Here is the code I promised.  It's written so that it could be placed into any workbook or even into your PERSONAL workbook.  To place it into a Regular Code Module follow the instructions here:

http://www.contextures.com/xlvba01.html#videoreg

Sub ExportSheetsToPDFFiles()
'Exports all sheets in the active workbook
'to individual .pdf files
'
'Files are named using the workbook's name
'plus _WS_ and a 4-digit number indicating
'the order of the output file from the
'order of the sheets within the workbook
'
'files are created in the same folder with
'the source workbook itself
'
  Dim anyWB As Workbook
  Dim anyWS As Worksheet
  Dim theName As String
  Dim newName As String
  Dim sheetCounter As Integer
 
  Set anyWB = ActiveWorkbook
  theName = Left(anyWB.FullName, InStrRev(anyWB.FullName, ".") - 1)
 
  For Each anyWS In anyWB.Worksheets
    sheetCounter = sheetCounter + 1
    newName = theName & "_WS_" & Format(sheetCounter, "0000") & ".pdf"
    anyWS.ExportAsFixedFormat Type:=xlTypePDF, _
     Filename:=newName, Quality:=xlQualityStandard, _
     IncludeDocProperties:=True, IgnorePrintAreas:=False, _
     OpenAfterPublish:=False
  Next ' end anyWS loop
  Set anyWB = Nothing
  MsgBox "Export to separate .pdf files completed", _
   vbOKOnly + vbInformation, "Task Completed"
End Sub
