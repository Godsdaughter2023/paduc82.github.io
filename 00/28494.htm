Abcd Efgh     -->     abcd efgh

macropod say:

There are numerous ways of handling this. One would be to create a Find/Replace list in another document that you could use with a Word macro to automate the replacement. The following macro does just that, using a document holding numerous paragraphs of tab-separated Find/Replace strings as the source for a large-scale Find/Replace operation.

Sub BulkFindReplace()
Application.ScreenUpdating = False
Dim FRDoc As Document, FRList, j As Long
 'Load the strings from the reference doc into a text string to be used as an array.
Set FRDoc = Documents.Open("C:\Users\" & Environ("Username") & "\Documents\FindReplaceList.doc", ReadOnly:=True, Addtorecentfiles:=False, Visible:=False)
FRList = FRDoc.Range.FormattedText
FRDoc.Close False
Set FRDoc = Nothing
With ActiveDocument.Range.Find
  .ClearFormatting
  .Replacement.ClearFormatting
  .MatchWholeWord = True
  .MatchCase = True
   'Process each word from the Check List. Tab-delimited strings are assumed, formatted as:
   'Find text <Tab> Replace text
  For j = 0 To UBound(Split(FRList, vbCr)) - 1
    .Text = Split(Split(FRList, vbCr)(j), vbTab)(0)
    .Replacement.Text = Split(Split(FRList, vbCr)(j), vbTab)(1)
    .Execute Replace:=wdReplaceAll
  Next
End With
Application.ScreenUpdating = True
End Sub

The above macro assumes the Find/Replace list is held in a document named FindReplaceList.doc in your Documents folder.

For PC macro installation & usage instructions, see: http://www.gmayor.com/installing_macro.htm

andreas killer say:
You need a UDF for that, try the function below and use the "Nth" argument to get the next date.

Andreas.


Function VLookupA(ByVal Value, ByVal Matrix As Range, _
    ByVal SearchColumn, ByVal ReturnColumn, _
    Optional ByVal Nth, Optional ByVal Compare) As Variant
  'Search the SearchColumn of Matrix for Value by Compare mode, _
    return the Nth value of ReturnColumn
  'Compare:
  '    1 search the next <= Value
  '    0 search exact Value, ignore spelling
  '   -1 search the next element >= Value
  ' TRUE search exact Value, binary compare
  Dim j As Long, p As Long
  Dim Search As Range
  Dim BinaryCompare As Boolean
  'Install an errorhandler for specific errors
  On Error GoTo ExitPoint
  'Be sure Nth is >= 1
  If IsMissing(Nth) Then Nth = 1 Else If Nth < 1 Then Nth = 1
  'Special handling for Compare as described above
  If Not IsMissing(Compare) Then
    If VarType(Compare) = vbBoolean Then
      If Compare Then BinaryCompare = True
    End If
  End If
  'Get the search column
  VLookupA = CVErr(xlErrRef)
  'Search the headers if column arguments are not numeric
  If Not IsNumeric(SearchColumn) Then
    SearchColumn = WorksheetFunction.Match(SearchColumn, Matrix.Rows(1), 0)
  End If
  If Not IsNumeric(ReturnColumn) Then
    ReturnColumn = WorksheetFunction.Match(ReturnColumn, Matrix.Rows(1), 0)
  End If
  Set Search = Matrix.Columns(SearchColumn)
  'Assume failure
  VLookupA = CVErr(xlErrNA)
  If BinaryCompare Then
    Do
      'Find an exact match
      j = WorksheetFunction.Match(Value, Search, 0)
      'Calculate the absolute position
      p = p + j
      'Check the spelling
      If StrComp(Value, WorksheetFunction.Index(Matrix, p, SearchColumn), _
          vbBinaryCompare) = 0 Then
        'Count it
        Nth = Nth - 1
        'Finished?
        If Nth = 0 Then Exit Do
      End If
      'Reduce the next range to the cells below
      Set Search = Search.Resize(Search.Rows.Count - j, 1).Offset(j, 0)
    Loop
  Else
    Do
      'Find a match
      j = WorksheetFunction.Match(Value, Search, Compare)
      'Calculate the absolute position
      p = p + j
      'Count it
      Nth = Nth - 1
      'Finished?
      If Nth = 0 Then Exit Do
      'Reduce the next range to the cells below
      Set Search = Search.Resize(Search.Rows.Count - j, 1).Offset(j, 0)
    Loop
  End If
  'Return the value from ReturnColumn
  VLookupA = CVErr(xlErrRef)
  VLookupA = WorksheetFunction.Index(Matrix, p, ReturnColumn)
ExitPoint:
End Function
