Public Function CcpMouseUp(ByRef C As Object, ByVal Button As Integer, _
    ByVal Shift As Integer) As Boolean
  'Emuliert das Rechtsklick-Kontextmenü für Eingabefelder
  'True wenn der User einen Eintrag gewählt hat
  Dim CcpMenubar As CommandBar, CcpItem As CommandBarControl
  Dim MenuExists As Boolean, I As Long
  Dim CName As String
   'Das Menü nur anzeigen bei Rechtsklick ohne Shift
  If Button <> vbKeyRButton Or Shift <> 0 Then Exit Function
   'Haben wir das temporäre Menü schon erzeugt?
  For Each CcpMenubar In CommandBars
    If CcpMenubar.Name = CcpMenuName Then
      MenuExists = True
      Exit For
    End If
  Next

#If EnableUndo Then
  'Unser Undo initiallisieren
  If CcpUndoStack Is Nothing Then
    Set CcpUndoStack = CreateObject("Scripting.Dictionary")
    CcpUndoStack.CompareMode = vbTextCompare
  End If
#End If
   'Eine Referenz für die externen Routinen erzeugen
  Set CcpControl = C
   If Not MenuExists Then
    'Temporäres Menü anlegen
    Set CcpMenubar = CommandBars.Add(Name:=CcpMenuName, Position:=msoBarPopup, Temporary:=True)
    CcpMenubar.Enabled = True
     Set CcpItem = CommandBars(CcpMenuName).Controls.Add(msoControlButton)
    With CcpItem
      .Caption = CcpCutName
      .OnAction = "CcpCut"
      .FaceId = 21
    End With
     Set CcpItem = CommandBars(CcpMenuName).Controls.Add(msoControlButton)
    With CcpItem
      .Caption = CcpCopyName
      .OnAction = "CcpCopy"
      .FaceId = 19
    End With
     Set CcpItem = CommandBars(CcpMenuName).Controls.Add(msoControlButton)
    With CcpItem
      .Caption = CcpPasteName
      .OnAction = "CcpPaste"
      .FaceId = 22
    End With

#If EnableUndo Then
    Set CcpItem = CommandBars(CcpMenuName).Controls.Add(msoControlButton)
    With CcpItem
      .BeginGroup = True
      .Caption = CcpUndoName
      .OnAction = "CcpUndo"
      .FaceId = 128
    End With
     Set CcpItem = CommandBars(CcpMenuName).Controls.Add(msoControlButton)
    With CcpItem
      .Caption = CcpRedoName
      .OnAction = "CcpRedo"
      .FaceId = 129
    End With
#End If
  End If
   'Ausschneiden verbieten wenn kein Text markiert
  'oder Box gesperrt ist
  Set CcpItem = CommandBars(CcpMenuName).Controls(Replace(CcpCutName, "&", ""))
  CcpItem.Enabled = C.SelLength > 0 And C.Enabled
   'Kopieren verbieten wenn kein Text markiert

  CcpItem.Enabled = C.SelLength > 0
   'Einfügen verbieten wenn kein Text in der Zwischenablage
  'oder Box gesperrt ist
  Set CcpItem = CommandBars(CcpMenuName).Controls(Replace(CcpPasteName, "&", ""))
  CcpItem.Enabled = C.CanPaste And C.Enabled

#If EnableUndo Then
  With CcpUndoStack
    Set CcpItem = CommandBars(CcpMenuName).Controls(Replace(CcpUndoName, "&", ""))
    CName = C.Parent.Name & "!" & C.Name

Private Sub CountCrLf(ByVal S As String, ByRef SelStart As Long, ByRef SelLength As Long)
  'Excel-Bug: In einer Textbox zählt ein vbCrLf (2 Zeichen) nur als 1 Zeichen!
  Dim I As Long, J As Long
  I = 1
  Do While I <= SelStart + 1 + SelLength
    If Mid(S, I, 2) = vbCrLf Then
      If I <= SelStart Then
        SelStart = SelStart + 1
      Else
        SelLength = SelLength + 1
      End If
    End If
    I = I + 1
  Loop
End Sub

Private Sub CcpCut()
  'Schneidet einen Text aus
  Dim SelSrt As Long, SelLen As Long
  'CcpControl.Cut funktioniert nicht in Tabellen!
  With CcpControl
    'Irgendwas zu tun?
    If .SelLength = 0 Then Exit Sub
    'Positionen korrigieren
    SelSrt = .SelStart
    SelLen = .SelLength
    CountCrLf .Value, SelSrt, SelLen
    'Text in Zwischenablage kopieren
    SetClip Mid(.Value, SelSrt + 1, SelLen)
    'Text aus Box entfernen
    .Value = strDelete(.Value, SelSrt + 1, SelSrt + SelLen)
  End With
  CcpResult = True
End Sub

Private Sub CcpCopy()

Private Sub CcpPaste()
  'Fügt einen Text ein
  Dim S As String, I As Long, SelSrt As Long, SelLen As Long
  Dim RemoveCC As Boolean
  '.Paste entfernt keine vbCrLF in Comboboxen!
  '.Paste reagiert nicht auf .Cut bei Zellen!
  '.Paste funktioniert nicht in Tabellen!
  '  With CcpControl
  '    If .CanPaste Then .Paste
  '    Exit Sub
  '  End With
  With CcpControl
    S = GetClip
    'Müssen wir vbCrLF entfernen?
    Select Case TypeName(CcpControl)
      Case "ComboBox"
        'Immer, Comboboxen haben kein Multiline
        RemoveCC = True
      Case "TextBox"
        RemoveCC = Not CcpControl.MultiLine
    End Select
    If RemoveCC Then
      'Alle Steuerzeichen entfernen
      I = 1
      Do While I <= Len(S)
        If Asc(Mid(S, I, 1)) < 32 Then
          S = strDelete(S, I, I)
        Else
          I = I + 1
        End If
      Loop
      If Len(S) = 0 Then Exit Sub
    End If
     'Positionen korrigieren
    SelSrt = .SelStart
    SelLen = .SelLength
    CountCrLf .Value, SelSrt, SelLen
     'Ggf. markierten Text entfernen
    If SelLen > 0 Then .Value = strDelete(.Value, SelSrt + 1, SelSrt + SelLen)
    'Text einfügen
    .Value = strInsert(.Value, S, SelSrt)
     'Wurde eine Zelle kopiert?
    Select Case Application.CutCopyMode
      Case False
        'Weder im Ausschneide- noch im Kopiermodus
      Case xlCopy
        'Im Kopiermodus
        Application.CutCopyMode = False
      Case xlCut
        'Im Ausschneidemodus
        Selection.ClearContents
        Application.CutCopyMode = False
    End Select
  End With
  CcpResult = True
End Sub

Public Sub CcpAddUndo(C As Object, Optional ByRef UndoStack)
  Dim CName As String
  'Unseren Undostck initiallisieren, oder einen externen übernehmen
  If Not IsMissing(UndoStack) Then

Public Sub CcpClearUndo(Optional ByRef UndoStack As Object = Nothing)

  If Not UndoStack Is Nothing Then Set CcpUndoStack = UndoStack
  If CcpUndoStack Is Nothing Then Exit Sub

Private Sub CcpUndo()
  'Box auf Anfangswert zurückstellen
  Dim S As String, CName As String
  With CcpUndoStack
    CName = CcpControl.Parent.Name & "!" & CcpControl.Name
    If .Exists(CName) Then
      S = .Item(CName)
      If CBool(Left(S, 1)) Then
        .Item(CName) = "0" & CcpControl.Value
        CcpControl.Value = Mid(S, 2)
      End If
    End If
  End With
End Sub

Private Function GetClip() As String

Private Function strInsert(ByVal S As String, ByVal SubStr As String, ByVal Pos As Long, _
    Optional ByVal Before As Boolean = False) As String
  'Fügt SubStr nach der Stelle Pos in S ein, wenn Pos negativ beginnt die Zählung von hinten
  If Pos < 0 Then
    If Abs(Pos) > Len(S) Then
      Pos = -Len(S)
      Before = False
    End If
    strInsert = Mid(S, 1, Len(S) + Pos + CDbl(Before)) & SubStr & _
      Mid(S, Len(S) + Pos + 1 + CDbl(Before))
  Else
    If Pos = 0 Then Before = False
    strInsert = Mid(S, 1, Pos + CDbl(Before)) & SubStr & Mid(S, Pos + 1 + CDbl(Before))
  End If
End Function

Private Function strDelete(ByVal S As String, ByVal FromPos As Long, _
    Optional ByVal ToPos As Long = 0, Optional ByVal Between As Boolean = False) As String
  'Löscht aus S den Teil von FromPos bis ToPos, wenn Between dann von FromPos+1 bis ToPos-1
  'Positionen korrigieren
  FromPos = FromPos - CDbl(Between)
  ToPos = ToPos + CDbl(Between)
  If FromPos <= 0 Then FromPos = 1
  'Wenn der Bereich außerhalb S liegt, S ganz zurückgeben
  If FromPos > Len(S) Then
    strDelete = S
    Exit Function
  End If
  If ToPos >= Len(S) Or ToPos <= 0 Then
    'Das Ende abschneiden
    strDelete = Mid(S, 1, FromPos - 1)
  Else
    'Teilbereiche kopieren
    strDelete = Mid(S, 1, FromPos - 1) & Mid(S, ToPos + 1)
  End If
End Function
