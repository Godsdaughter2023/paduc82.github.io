Private Sub Worksheet_SelectionChange(ByVal Target As Range)
  Dim Order
  Static Last
  Dim i As Integer
  Order = Array("A5", "D9", "C1", "A10", "G13", "E12", "H5")
  
  'If this cell one of ours?
  For i = 0 To UBound(Order)
    If StrComp(Order(i), Target.Address(0, 0), vbTextCompare) = 0 Then
      'Yes, remember and done
      Last = Order(i)
      Exit Sub
    End If
  Next
  'No, get the next cell after the last one
  For i = 0 To UBound(Order)
    If StrComp(Order(i), Last, vbTextCompare) = 0 Then
      i = (i + 1) Mod UBound(Order)
      Last = Order(i)
      Exit For
    End If
  Next
  'If not found use the first one
  If i > UBound(Order) Then Last = Order(0)
  'Select that cell
  Application.EnableEvents = False
  Range(Last).Select
  Application.EnableEvents = True
End Sub

If you want the horizontal axis spaced equally use a line chart.
