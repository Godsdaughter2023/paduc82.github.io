I did some looking around and found this on one of the fourms:

BEFORE connecting your 30gb zune, Open the (up to date) Zune software, click Settings, Click Device, Click update, and THEN connect your zune to the pc. Wait about 10 seconds, and click the back arrow on the upper left corner. 

A Full screen update window will appear, and seconds later you will get the prompt to perform the update.

Use only one (always filled) column to find the last row, then copy all values into that row.

Sub SummurizeSheets()
  Dim Ws As Worksheet
  Dim Base As Range, Dest As Range, Source As Range
  Application.Calculation = xlCalculationManual
  Application.EnableEvents = False
  For Each Ws In ThisWorkbook.Worksheets
    If Ws.Name <> "Master_Sheet" And Ws.Name <> "Monthly_Report" And Ws.Name <> "Default" Then
      'Find next empty slot
      Set Dest = Worksheets("Monthly_Report").Range("F" & Rows.Count).End(xlUp).Offset(1, 0)
      Set Base = Dest
      'Copy each value
      For Each Source In Ws.Range("B14,C14,B2,B4,D4,E4,A9,C9,C12,D21,C23,D32,G12,H21,G23,H32")
        Dest.Value = Source.Value
        Set Dest = Dest.Offset(, 1)
      Next
      'Be sure 1st cell is filled!
      If IsEmpty(Base) Then Base = "<missing>"
    End If
  Next
  Application.Calculation = xlCalculationAutomatic
  Application.EnableEvents = True
End Sub
