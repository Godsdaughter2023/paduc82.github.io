Here is a macro you can run anytime you want to refresh the list of serial numbers in column E.  How long it takes to run depends on the total number of entries in each list.

Here are instructions for putting the code into a regular code module in the workbook:

http://www.contextures.com/xlvba01.html#Regular

Here is the code.  To use it after it's in the workbook, from the [View] tab use the drop-down list at the right end of the tab and choose {View Macros}, highlight the GetSerialNumbers macro in the list and click the [Run] button...
and wait... it will let you know when it is finished.

You can tailor the code for your workbook by making appropriate changes to the Const values set up near the beginning of the code.  I've set them up based on your description above, so if the sheet names and column identifiers are correct, and your two lists start on row 2 on each sheet, it should be all good to go.

Sub GetSerialNumbers()
'change the values of these constants to match
'the worksheet names and column use in your workbook
Const s1Name = "Sheet1"
Const s1InvCol = "A" ' invoice # column
Const s1CustCol = "B" ' customer ID column
Const s1ProdCol = "C" ' product code column
Const s1SNCol = "E" ' column to place s/numbers into
Const s1FirstDataRow = 2 ' assumes first data entry on row 2

Const s2Name = "Sheet2"
Const s2InvCol = "A" ' invoice # col
Const s2ProdCol = "B" ' product code col
Const s2SNCol = "C" ' serial number column
Const s2FirstDataRow = 2 ' entries start on row 2
'end of user definable constant values

  Dim ws1 As Worksheet
  Dim s1InvList As Range
  Dim anyS1InvEntry As Range
  Dim s1Offset2Cust As Integer
  Dim s1Offset2Prod As Integer
  Dim s1Offset2SN As Integer
 
  Dim ws2 As Worksheet
  Dim s2InvList As Range
  Dim anyS2InvEntry As Range
  Dim s2Offset2Prod As Integer
  Dim s2Offset2SN As Integer
 
  Dim snListing As String
 
  Set ws1 = ThisWorkbook.Worksheets(s1Name)
  Set s1InvList = ws1.Range(s1InvCol & s1FirstDataRow & ":" _
   & ws1.Range(s1InvCol & Rows.Count).End(xlUp).Address)
  s1Offset2Cust = Range(s1CustCol & 1).Column - Range(s1InvCol & 1).Column
  s1Offset2Prod = Range(s1ProdCol & 1).Column - Range(s1InvCol & 1).Column
  s1Offset2SN = Range(s1SNCol & 1).Column - Range(s1InvCol & 1).Column
 
  Set ws2 = ThisWorkbook.Worksheets(s2Name)
  Set s2InvList = ws2.Range(s2InvCol & s2FirstDataRow & ":" _
   & ws2.Range(s2InvCol & Rows.Count).End(xlUp).Address)
  s2Offset2Prod = Range(s2ProdCol & 1).Column - Range(s2InvCol & 1).Column
  s2Offset2SN = Range(s2SNCol & 1).Column - Range(s2InvCol & 1).Column
 
  'finally ready to go to work after so much set up
  'work through every non-blank invoice entry on Sheet1
  For Each anyS1InvEntry In s1InvList
    snListing = "" ' reset to empty
    If Not IsEmpty(anyS1InvEntry) Then
      'test against each entry in the list on Sheet2
      For Each anyS2InvEntry In s2InvList
        DoEvents ' give OS time to do background work
        If anyS1InvEntry = anyS2InvEntry Then
          'invoice numbers matched, does product match?
          If anyS1InvEntry.Offset(0, s1Offset2Prod) = _
           anyS2InvEntry.Offset(0, s2Offset2Prod) Then
            'Invoice number and product ID match, get serial number
            snListing = snListing & anyS2InvEntry.Offset(0, s2Offset2SN) & ","
          End If
        End If
      Next ' end s2InvList loop
    End If
    If snListing <> "" Then
      'remove the extra "," from snListing string
      anyS1InvEntry.Offset(0, s1Offset2SN) = Left(snListing, Len(snListing) - 1)
    End If
  Next ' end s1InvList loop
  'do housekeeping of system resources
  Set s1InvList = Nothing
  Set s2InvList = Nothing
  Set ws1 = Nothing
  Set ws2 = Nothing
  'announce completion
  MsgBox "Serial Number Listings rebuilt.", vbOKOnly + vbInformation, "Task Done"
End Sub

Option Explicit
Sub ConsolidateSerialNums()
  Dim wsData As Worksheet
  Dim wsReport As Worksheet
  Dim rData As Range
  Dim rReport As Range
  Dim c As Range
  Dim i As Long
  Dim sSerialNums As String
  Dim sFind As String, sFirstAddress As String
 
Set wsData = Worksheets("Sheet2")
With wsData
    Set rData = .Range("A2", .Cells(.Rows.Count, "A").End(xlUp)) _
        .Resize(columnsize:=3)
End With

Set wsReport = Worksheets("Sheet1")
With wsReport
    Set rReport = .Range("A2", .Cells(.Rows.Count, "A").End(xlUp)) _
        .Resize(columnsize:=5)
End With

For i = 1 To rReport.Rows.Count
    sFind = rReport(i, 1)
    sSerialNums = ""
    With rData.Columns(1)
        Set c = .Find(what:=sFind, after:=rData.End(xlDown), _
            LookIn:=xlValues, lookat:=xlWhole, _
            searchdirection:=xlNext, MatchCase:=False)
        If Not c Is Nothing Then
            sFirstAddress = c.Address
            If c(1, 2) = rReport(i, 3) Then
                sSerialNums = ", " & c(1, 3)
            End If
           
            Do
                Set c = .FindNext(c)
                If Not c Is Nothing Then
                    If c.Address <> sFirstAddress Then
                        If c(1, 2) = rReport(i, 3) Then
                            sSerialNums = sSerialNums & ", " & c(1, 3)
                        End If
                    End If
                End If
            Loop Until c Is Nothing Or sFirstAddress = c.Address
        End If
    End With
    rReport(i, 5) = Mid(sSerialNums, 3)
Next i
End Sub
