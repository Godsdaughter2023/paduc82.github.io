You can use the Find method to search upwards from the last cell. The first value encountered would be the last used cell in the column.

Sub AAA()
    Dim WS As Worksheet
    Dim LastCell As Range
    Dim R As Range
    Set WS = ActiveSheet
    With WS
        Set LastCell = .Columns(1).Find(what:="*", _
            after:=.Cells(.Rows.Count, "A"), _
            LookIn:=xlFormulas, _
            lookat:=xlPart, searchdirection:=xlPrevious)
    End With
    Debug.Print LastCell.Address
End Sub
