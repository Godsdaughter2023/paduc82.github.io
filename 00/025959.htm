If you just want to take a look, you have to do two things

try this sample

 

Sub AutoFilter1()
Const myCol As String = "C"    '<<< dates in column C, change
Dim r As Long
r = ActiveSheet.Cells(Rows.Count, myCol).End(xlUp).Row
Dim dt1 As String, dt2 As String
dt1 = "15/05/2011" '<< May 15 2011
dt2 = "20/05/2011"  '<< May 20 2011
ActiveSheet.AutoFilterMode = False
Range(myCol & "1:" & myCol & r).AutoFilter Field:=1, _
Criteria1:=">=" & Format(dt1, "mm/dd/yyyy"), Operator:=xlAnd, _
Criteria2:="<=" & Format(dt2, "mm/dd/yyyy")  '<<< NOTE: don't change format, mm/dd/yyyy
MsgBox "next.."
ActiveSheet.AutoFilterMode = False
End Sub

Sub AutoFilter2()
Const myCol As String = "C" '<<<dates in column A, change
Dim ws As Worksheet
Set ws = Sheets("Sheet1") '<<< data in sheet1, change
Dim r As Long
r = ws.Cells(Rows.Count, myCol).End(xlUp).Row
Dim dt1 As String, dt2 As String
dt1 = "15/05/2011" '<< May 15 2011
dt2 = "20/05/2011" '<< May 20 2011
ws.AutoFilterMode = False
ws.Range(myCol & "1:" & myCol & r).AutoFilter Field:=1, _
Criteria1:=">=" & Format(dt1, "mm/dd/yyyy"), Operator:=xlAnd, _
Criteria2:="<=" & Format(dt2, "mm/dd/yyyy")
ws.Range(myCol & "1:" & myCol & r).Copy
Sheets.Add
Range("A1").PasteSpecial xlPasteValues
Range("A:A").NumberFormat = "dd/mm/yyyy"
Application.CutCopyMode = False
ws.AutoFilterMode = False
End Sub
