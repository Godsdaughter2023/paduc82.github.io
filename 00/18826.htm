jay freedman say:
As Stefan said, Word sorts tables by considering each row as a single piece, and sorting them by the contents of a particular column.

The way to get this as you want it is to convert the original table to plain text, sort that, and convert it back to a table:
  1. Click in the table. On the Table Tools | Layout tab, click Convert to Text.
  2. In the conversion dialog, select 'Paragraph marks' and click OK. You'll get all the cell contents as text, one cell per paragraph. That text will be selected automatically.
  3. On the Home tab, click the Sort button in the Paragraph group. (It's the one with A over Z and a downward arrow.) Leave the settings as is and click OK.
  4. On the Insert tab, click the Table button and click Convert Text to Table.
  5. In the conversion dialog, select 'Paragraphs' and set the number of columns to the original number. Click OK.

The sorted items will go across the row and then down. There is no simple way to get them to go down and then across, other than copying the table into Excel and using the transpose command.

  • A 27-character PIN

  • B and 8
  • G and 6
  • U and V
  • VV and W
  • Q and O
  • The letter O and the number 0 (zero)
  • The letter L and the number 1

If none of the above resolved the issue, you have two options:

Dim appOutlook As Outlook.Application
Dim nmsNameSpace As Outlook.NameSpace
Dim fldCalendar As Outlook.MAPIFolder
Dim itemCalendar As Outlook.AppointmentItem
Dim ItemsCal As Outlook.Items
Set appOutlook = GetObject(, "Outlook.Application")
If Err <> 0 Then
    'Outlook wasn't running, start it from code
    Set appOutlook = CreateObject("Outlook.Application")
    'bStarted = True
End If
Set nmsNameSpace = appOutlook.GetNamespace("MAPI")
Set fldCalendar = nmsNameSpace.GetDefaultFolder(olFolderCalendar)
'Set fldCalendar = "StuffFromAccess"
Set itemCalendar = appOutlook.CreateItem(olAppointmentItem)
With itemCalendar
    .Subject = theSubject
    .Body = theBody & TriggeredByItemNo & " (Posted automatically on " & Format(Now, "m/d/yy") & ".)"
    .Start = StartDate  '(Date + 6) & " " & TimeSerial(8, 30, 0)
    .Categories = "From Access"
    If StartTime <> "" Then
        .Start = StartDate & " " & StartTime
        Else
        .Start = StartDate
        IsThisAnAllDayEvent = True
    End If
    If EndDate <> "" Then
        .End = EndDate
        Else
        If EndTime <> "" Then
        .End = EndDate & " " & EndTime '& TimeSerial(10, 30, 0)
        End If
    End If
    .AllDayEvent = IsThisAnAllDayEvent
    
    '.ReminderSet = vReminderSet ' False
    If ReoccuringType <> "" Then
        With .GetRecurrencePattern
            .RecurrenceType = theRecurringNum '"OlRecurs" & ReoccuringType  'e.g., olRecursDaily
            If Not IsNull(ReoccuringInterval) Then
                .Interval = ReoccuringInterval
                Else
                .Interval = 1
            End If
            If NumOfDaysOfWarning <> 0 Then
                .PatternStartDate = StartDate
                .PatternEndDate = StartDate - NumOfDaysOfWarning
            End If
        End With
   End If
   If Not IsNull(theDurationInMinutes) Then
   .Duration = theDurationInMinutes ' To create a duration of one day for an all-day event, set the duration equal to 1440.  There are 1440 minutes in a day.
   Else
   .Duration = 60
   End If
   If theLocation <> "" Then
   .Location = theLocation
   End If
    .Save
End With
'appOutlook.Quit
Set itemCalendar = Nothing
Set fldCalendar = Nothing
Set nmsNameSpace = Nothing
