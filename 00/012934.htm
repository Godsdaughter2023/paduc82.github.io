Sub HL_fix()
    Dim osld As Slide
    Dim oHl As Hyperlink
    Dim otr As TextRange
    For Each osld In ActivePresentation.Slides
        For Each oHl In osld.Hyperlinks
            On Error Resume Next
            Debug.Print TypeName(oHl.Parent.Parent)
