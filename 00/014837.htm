Once you get a working formula you can make your final CONCATENATE formula a lot shorter using this UDF

=IF($E145="regular";ConCatRange($I145:$I194);"bad formula")

Copy the UDF to a module.

Function ConCatRange(CellBlock As Range) As String
'for non-contiguous cells  =ccr((a1:a10,c4,c6,e1:e5))
'note double parens
Dim Cell As Range
Dim sbuf As String
    For Each Cell In CellBlock
    On Error GoTo fred
    If Len(Cell.Text) > 0 Then sbuf = sbuf & Cell.Text & " ."   'delimiter of choice
        Next
    ConCatRange = Left(sbuf, Len(sbuf) - 2)
fred:
End Function
