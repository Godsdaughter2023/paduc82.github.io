greg maxey say:

You can do it with VBA easy enough:

ub ScratchMacro()
'A basic Word macro coded by Greg Maxey
Dim oRng As Word.Range
Dim oRngP As Word.Range
Set oRng = ActiveDocument.Range
With oRng.Find
  .Text = "\[*\]"
  .MatchWildcards = True
  While .Execute
    Set oRngP = oRng.Duplicate
    oRng.Collapse wdCollapseEnd
    With oRngP.Find
     .Text = ","
     .Replacement.Text = "qqq"
     .Execute Replace:=wdReplaceAll
    End With
  Wend
End With
   
End Sub
