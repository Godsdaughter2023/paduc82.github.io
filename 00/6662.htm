You can't turn off Excel warnings without using VBA.  But, you can check for the file and then delete it (but using VBA). 

 

Function KillifFileExists(strFileNameAndPath)

    If Dir(strFileNameAndPath) <> vbNullstring Then

        Kill strFileNameAndPath

    End If

Function

 

 

If you put that into a standard module (don't name the module the same as this function, use a different name like basFileProcs)

 

Then you can call that function from your macro using the RunCode action and you can pass the file name and path, including extension in the Arguments area.

Sub CopyRange()
  Dim myInput As Variant
  Dim testAddress As Range
 
  myInput = InputBox(prompt:="Enter address to copy selection to:", Title:="my title")
  On Error Resume Next
  Set testAddress = Range(myInput)
  If Err <> 0 Then
    MsgBox "Please enter a valid cell address.", vbOKOnly + vbExclamation, "Try Again!"
    Err.Clear
    On Error GoTo 0
    Exit Sub
  End If
  On Error GoTo 0
  'if you get here, it's a valid address.
  Selection.TextToColumns Destination:=Range(myInput), DataType:=xlDelimited, _
         TextQualifier:=xlDoubleQuote, ConsecutiveDelimiter:=False, Tab:=True, _
         Semicolon:=False, Comma:=False, Space:=False, Other:=True, OtherChar _
         :="-", FieldInfo:=Array(1, 1), TrailingMinusNumbers:=True
End Sub

Sub QuickTextToColumns()
'will function on all cells in column A
'beginning at row 6 and continuing down until
'an empty cell is found:
' empty=nothing typed into it and no formula in it.
'
  Dim workRange As Range
  Dim destCell As Range
 
  Set destCell = Range("A6")
  Set workRange = Range("A6" & ":" & _
   Range("A6").End(xlDown).Address)
  Application.ScreenUpdating = False ' improves performance
  With workRange
    .TextToColumns Destination:=destCell, DataType:=xlDelimited, _
     TextQualifier:=xlDoubleQuote, ConsecutiveDelimiter:=False, Tab:=True, _
     Semicolon:=False, Comma:=False, Space:=False, Other:=True, OtherChar:="-", _
     FieldInfo:=Array(1, 1), TrailingMinusNumbers:=True
  End With
End Sub
