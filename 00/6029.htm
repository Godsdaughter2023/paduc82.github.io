 Exit Sub

 t = y

 Set dcn = dc2

For Each y In x.Value2

 v(k, 1) = dco.Keys(k - 1)
 v(k, 2) = dco.Items(k - 1)
 Next k
 
qsortd v, 1, N

 v(k, 3) = v(k, 1) * v(k, 2) + v(IIf(k = N, N, k + 1), 3)
 If v(k, 3) > t Then dcn.Add Key:="+" & _

 x = v(j, 1)

 If p Then
 RE.Pattern = "\" & s & "(?=(\+|$))"
 If RE.Execute(y).Count < v(j, 2) Then
 u = dco(y) + x

 recsoln y & s
 ElseIf u < t - TOL Then

 c = CDec(c + 1)

 Format(c)
 End If
 End If
 End If
 Next j

 Next k

 MsgBox Prompt:="all combinations exhausted", _
 Title:="No Solution"

Private Function recsoln(Optional s As String)
 Const OUTPUTWSN As String = "findsums solutions" 'modify to taste
 
Static r As Range
 Dim ws As Worksheet
 
If s = "" And r Is Nothing Then
 On Error Resume Next

 Set r = Worksheets.Add.Range("A1")

 End If

Private Sub qsortd(v As Variant, lft As Long, rgt As Long)
 'ad hoc quicksort subroutine
 'translated from Aho, Weinberger & Kernighan,
 '"The Awk Programming Language", page 161
 
Dim j As Long, pvt As Long
 
If (lft >= rgt) Then Exit Sub
 swap2 v, lft, lft + Int((rgt - lft + 1) * Rnd)
 pvt = lft
 For j = lft + 1 To rgt
 If v(j, 1) > v(lft, 1) Then
 pvt = pvt + 1
 swap2 v, pvt, j
 End If
 Next j
 
swap2 v, lft, pvt
 
qsortd v, lft, pvt - 1
 qsortd v, pvt + 1, rgt
 End Sub
 
Private Sub swap2(v As Variant, i As Long, j As Long)
 'modified version of the swap procedure from
 'translated from Aho, Weinberger & Kernighan,
 '"The Awk Programming Language", page 161
 
Dim t As Variant, k As Long
 
For k = LBound(v, 2) To UBound(v, 2)
 t = v(i, k)
 v(i, k) = v(j, k)
 v(j, k) = t
 Next k
 End Sub
 
Private Sub swapo(a As Object, b As Object)
 Dim t As Object
 
Set t = a
 Set a = b
 Set b = t
 End Sub

don cuthbert say:

You could try a more thorough disk cleanup if you are content to not have to restore your prior OS.  Are you on an Activated install of Windows 10 Build 10240?

First create a Recovery drive by running recoverydrive.exe from within Build 10240.  Check the box to include System files.  You need a USB flash drive of 4GB or larger to hold the Windows system image to restore Windows 10.  This will give you the option for a clean install.

Then launch an Administrator Command prompt.  Run cleanmgr /sageset.  If you select all options it will free the most space, but you will lose the ability to rollback to the previous OS and the Reset option if you select everything.  (But since you have a recovery disk flash drive you should be OK).  You can read the description of each choice.  Your choices will not be processed until you re-run cleanmgr as described below.

After making your "sageset" choices, run cleanmgr /sagerun.  This will process a thorough disk cleanup, based on your selections made when you ran "sageset".  You can also choose to delete all but the most recent Restore point, on the second page of disk cleanup, which will also free more space.

If this is not enough, there is also a way to install Windows 10 in Compact mode using DISM, which is designed especially for tablets and small SSD's.  It would be easiest to do with a clean install, but I am considering trying it with my existing SSD, which is usually almost full too.  It involves making an image, repartitioning the hard disk and then re-applying the Windows image in compact mode.  (This is different than just compressing drive C: and is new with Windows 10).
