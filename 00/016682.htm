graham mayor say:

Frankly Microsoft have ruined the new document dialog in the latest versions of Office. The last version that made any sense was 2010, but it is possible to force recent versions of Word to work as you require, to avoid the problem you are having, and that is to use the document as a template. This will create a new unnamed document, containing the content of the original document for you to work with.

The following macro will allow you to select a document and use it as a template http://www.gmayor.com/installing_macro.htm   The macro name (FileNew) will cause it to intercept the built FileNew command, which is accessed by adding the New command to the QAT (Quick Access Toolbar) from the all commands group.

Alternatively you could rename the macro and add the macro to the QAT (Quick Access Toolbar) or the ribbon and use that button when you  want to create a new document from an existing document.

Option Explicit

Sub FileNew()
Dim strPath As String
    strPath = BrowseForFile(strTitle:="Select the document to use as a template")
    If strPath = "" Then GoTo lbl_Exit
    Documents.Add Template:=strPath
lbl_Exit:
    Exit Sub
End Sub

Private Function BrowseForFile(Optional strTitle As String, Optional bExcel As Boolean) As String
'Graham Mayor
'strTitle is the title of the dialog box
'Set bExcel value to True to filter the dialog to show Excel files
'The default is to show Word files
Dim fDialog As FileDialog
    On Error GoTo err_handler
    Set fDialog = Application.FileDialog(msoFileDialogFilePicker)
    With fDialog
        .Title = strTitle
        .AllowMultiSelect = False
        .Filters.Clear
        If bExcel Then
            .Filters.Add "Excel workbooks", "*.xls,*.xlsx,*.xlsm"
        Else
            .Filters.Add "Word documents", "*.doc,*.docx,*.docm"
        End If
        .InitialView = msoFileDialogViewList
        If .Show <> -1 Then GoTo err_handler:
        BrowseForFile = fDialog.SelectedItems.Item(1)
    End With
lbl_Exit:
    Exit Function
err_handler:
    BrowseForFile = vbNullString
    Resume lbl_Exit
End Function
