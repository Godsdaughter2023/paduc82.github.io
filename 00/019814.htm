graham mayor say:
You haven't said what the values are in your combo box, but assuming that one of the values contains the word 'bank' then you could use something like

Private Sub cmdCreateRecord_Click()
Dim r As Long
Dim strSheet As String
    If InStr(1, LCase(Me.Type.Value), "bank") > 0 Then
        strSheet = "Bank Loans"
    Else
        strSheet = "Personal Loans"
    End If
    With Worksheets(strSheet)
        r = .Range("A" & .Rows.Count).End(xlUp).Row + 1
        .Range("A" & r) = Me.cboType
        .Range("B" & r) = Me.cboBorrower
        .Range("C" & r) = Me.cboLender
        .Range("D" & r) = Me.cboCCY
        .Range("E" & r) = Me.txtAmount
        .Range("F" & r) = Me.dtpStartDate
        .Range("G" & r) = Me.dtpEndDate
    End With
End Sub
