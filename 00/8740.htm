tom van stiphout say:

Create a new query and select both tables. Make sure they are joined on ProductID.

Click the Sigma button to turn it into a Totals query.

For a first taste of how this works, select only SaleID and in the second column write:

Total: Price * Quantity

Then design again and add the date field and take out the SaleID. Run again and you will see totals per day.

All that aside: you seem to have a database design problem. Do your product prices ever change? Probably so. Then this approach is not going to work. You will have to store the Price as it was on the day of the sale, typically in an InvoiceDetails table, or perhaps OrderDetails.

accessnewb117 say:

Hi tom, 

thanks for reply. the database im creating is more for learning purposes than real life application. im still encountering issues though as while your solution is a starting point, it doesn't refine the total cost by sale id. is there a way to combine the two totals of sale id 7 to get one result as opposed to two?

ken sheridan say:
You appear not to have followed Tom's directions as regards clicking on the upper case Sigma icon in the ribbon.  The query in SQL view would look like this:

SELECT [Sale ID], SUM([Price]*[Quantity]) AS TotalPrice
FROM [Products] INNER JOIN [Sales And Products]
ON [Products].[Product ID] = [Sales And Products].[Product ID]
GROUP BY [Sale ID];

If you open the query designer and switch to SQL view, and paste the above in, being sure to change the table and column names where necessary so that they exactly match your actual object names, then switch back to design view, you'll  see how it is designed in the latter.

To aggregate the sales by date you'll need to include the relevant sale date column in the query in place of the Sale ID column.  If the sale date column is in a separate Sales table, as I'd expect. you'll first need to include that table in the query also, joined to the Sales And Products table on Sale ID.

graham mayor say:

The following should work. Change the Template Name and Path where indicated as appropriate:

http://www.gmayor.com/installing_macro.htm   

Option Explicit

Sub BatchProcess()
Dim strFilename As String
Dim strPath As String
Dim strNewPath As String
Dim oDoc As Document
Dim fDialog As FileDialog
Const strTemplate as String = "D:\Path\Template Name.dotx"

    Set fDialog = Application.FileDialog(msoFileDialogFolderPicker)
    With fDialog
        .Title = "Select folder and click OK"
        .AllowMultiSelect = False
        .InitialView = msoFileDialogViewList
        If .Show <> -1 Then
            MsgBox "Cancelled By User", , _
                   "List Folder Contents"
            Exit Sub
        End If
        strPath = fDialog.SelectedItems.Item(1) & "\"
        strNewPath = strPath & "Updated\"
        If Not FolderExists(strNewPath) Then
            CreateFolders (strNewPath)
        End If
    End With
    strFilename = Dir$(strPath & "*.doc")
    While Len(strFilename) <> 0
        WordBasic.DisableAutoMacros 1
        Set oDoc = Documents.Open(strPath & strFilename)
        With oDoc
            .UpdateStylesOnOpen = True
            .AttachedTemplate = strTemplate
            .UpdateStylesOnOpen = False
            .SaveAs Filename:=strNewPath & strFilename
            .Close SaveChanges:=wdDoNotSaveChanges
        End With
        strFilename = Dir$()
        WordBasic.DisableAutoMacros 0
    Wend
    Set oDoc = Nothing
End Sub

Private Function FolderExists(ByVal PathName As String) As Boolean
Dim lngAttr As Long
    On Error GoTo NoFolder
    lngAttr = GetAttr(PathName)
    If (lngAttr And vbDirectory) = vbDirectory Then
        FolderExists = True
    End If
NoFolder:
End Function

Private Function CreateFolders(strPath As String)
Dim strTempPath As String
Dim lngPath As Long
Dim vPath As Variant
    vPath = Split(strPath, "\")
    strPath = vPath(0) & "\"
    For lngPath = 1 To UBound(vPath)
        strPath = strPath & vPath(lngPath) & "\"
        If Not FolderExists(strPath) Then MkDir strPath
    Next lngPath
End Function

If the tech simply copied the Favorites folder, Windows 7 is unhappy that it's not using its own Favorites folder.

 

This procedure has helped a few others with this problem:

Let's work first with the Public profile.

Right-click and rename the Favorites folder at C:\Users\Public\Favorites to Favorites.bak then restart the computer and see if Windows has created a new Favorites folder.

 

If Windows created a new Favorites folder, open both of these folders C:\Users\Public\Favorites and C:\Users\Public\Favorites.bak and resize them so you can see both. Select a number of Favorites (not folders or sub-folders) in the bak folder and copy them to the new Favorites folder.

 

Restart again and see if the issue is resolved.

rich007a say:

OK, try this, again it's completely not tested.

 

Sub UpdateMasterQuoteLog()
    Dim wbLog As Workbook, wbMast As Workbook
    Dim wsLog As Worksheet, wsMast As Worksheet
    Dim rngMast As Range, rngLog As Range
    Dim strPath As String
    Dim lnTopRow As Long, lnBotRow As Long
   
     'QUOTE_LOG:
    strPath = "C:\temp\Quote_Log.xls"  '#### UPDATE AS REQUIRED ####
    Set wbLog = Workbooks.Open(strPath)
    Set wsLog = wbLog.Worksheets("Sheet Name") '#### EDIT SHEET NAME ####
    lnBotRow = wsLog.Range("A" & wsLog.Rows.Count).End(xlUp).Row 'last row in log
    Set rngLog = wsLog.Range("A" & lnBotRow).Resize(1, 18) 'Cols A:R, row of last non-empty cell in col A
   
     'MASTER_QUOTE_LOG:
    Set wbMast = ThisWorkbook
    Set wsMast = wbMast.Sheets("Sheet Name") '#### EDIT SHEET NAME ####
    Set rngMast = wsMast.Range("A" & wsMast.Rows.Count).End(xlUp).Resize(1, 18) 'Cols A:R, row of last non-empty cell in col A
   
    If rngLog(1, 1).Value + rngLog(1, 10).Value < rngMast(1, 1).Value + rngMast(1, 10).Value Then
        MsgBox "The last entry in the Master_Quote_Log is dated after the last entry in the Quote_Log!", vbCritical
        GoTo ExitPoint
    ElseIf rngLog(1, 1).Value + rngLog(1, 10).Value = rngMast(1, 1).Value + rngMast(1, 10).Value Then
        MsgBox "The Master_Quote_Log is already up-to-date.", vbInformation
        GoTo ExitPoint
    ElseIf rngLog(1, 1).Row = 1 Then 'special case of only one row in the Quote_Log file...
        rngMast = rngMast.Offset(1, 0) 'move down one row in the Master, there is only one row to copy.
    Else
        Set rngLog = rngLog.Offset(-1, 0) 'move up one row
        Do
            If rngLog(1, 1).Value + rngLog(1, 10).Value <= rngMast(1, 1).Value + rngMast(1, 10).Value Then 'dates&times match
                lnTopRow = rngLog(1, 1).Row + 1  'move back down one row
                Set rngLog = rngLog.Offset(1, 0).Resize(lnBotRow - lnTopRow + 1, 18)
                Set rngMast = rngMast.Offset(1, 0).Resize(lnBotRow - lnTopRow + 1, 18) 'move down one, and resize the range in the Master
                Exit Do
            Else
                Set rngLog = rngLog.Offset(-1, 0) 'move up 1 row
            End If
        Loop While rngLog(1, 1).Row > 1
    End If
   
     'Copy data from row in Quote_Log.xls to row in Master_Quote_Log.xls:
    rngMast.Value = rngLog.Value

'    wbMast.Save 'uncomment when debug complete
   
ExitPoint:
    wbLog.Close SaveChanges:=False
End Sub

Sub Matrix_Correl_1column()

'tested

Dim N As String

Dim Tx As Long, Ty As Long

Dim ws1 As Worksheet, ws2 As Worksheet

N = InputBox("entries per month", , "3")

If N = "" Then Exit Sub

Set ws1 = Sheets("Sheet1")'source data

Set ws2 = Worksheets.Add(before:=Worksheets(1))

Application.ScreenUpdating = False

For i = 1 To 12

ws2.Cells(1, i + 1) = i & "-2011"

ws2.Cells(i + 1, 1) = i & "-2010"

Next i

Tx = 0

Ty = 0

For i = 1 To ws1.Cells(ws1.Rows.Count, "B").End(xlUp).Row / 2 Step N

Tx = Tx + 1

For j = 12 * N + 1 To ws1.Cells(ws1.Rows.Count, "B").End(xlUp).Row Step N

Ty = Ty + 1

ws2.Cells(Tx + 1, Ty + 1) = Round(WorksheetFunction.Correl _

(ws1.Range(ws1.Cells(i, "B"), ws1.Cells(i + N - 1, "B")), _

ws1.Range(ws1.Cells(j, "B"), ws1.Cells(j + N - 1, "B"))), 2)

Next j

Ty = 0

Next i

Application.ScreenUpdating = True

End Sub
