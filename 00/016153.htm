teylyn say:

Hello,

you're leaving out the ForeColor when setting the text box font and you're not checking for the correct color on the selected cell.

This corrected code works for me:

Sub test()
Set objAny = Selection
    'Turn font black if it is currently red
    If TypeName(objAny) = "Range" And objAny.Font.Color = 255 Then
       objAny.Font.ColorIndex = xlAutomatic
    'Turns font red if any color other than itself is being used
    ElseIf TypeName(objAny) = "Range" Then
      objAny.Font.Color = -16776961
    'Same process but for Textbox font colors
    ElseIf TypeName(objAny) = "TextBox" And objAny.ShapeRange.TextFrame2.TextRange.Font.Fill.ForeColor = RGB(255, 0, 0) Then
         objAny.ShapeRange.TextFrame2.TextRange.Font.Fill.ForeColor.RGB = RGB(0, 0, 0)
    ElseIf TypeName(objAny) = "TextBox" Then
    With objAny.ShapeRange.TextFrame2.TextRange.Font.Fill
      .Solid
      .Transparency = 0
      .ForeColor.RGB = RGB(255, 0, 0)
    End With
   End If
End Sub
