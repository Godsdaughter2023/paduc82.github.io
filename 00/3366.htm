Try these versions:

 

Function DaysInside(DocID As String) As Long
    Dim dbs As DAO.Database
    Dim rst As DAO.Recordset
    Dim f As Boolean
    Dim d As Date
    Set dbs = CurrentDb
    Set rst = dbs.OpenRecordset("SELECT * FROM 04_DocMovt WHERE Doc=" & _
        DocID & " ORDER BY RecordNo", dbOpenForwardOnly)
    Do While Not rst.EOF
        Select Case rst!Decision
        Case "Referred Back", "Forwarded"
            If Not f Then
                d = rst!ReceivedOn
                f = True
            End If
        Case "Sent Out of Department"
            If f Then
                DaysInside = DaysInside + rst!ReceivedOn - d
                f = False
            End If
        Case "Approved", "Rejected"
            If f Then
                DaysInside = DaysInside + rst!ReceivedOn - d
                f = False
            End If
            Exit Do
        End Select
        rst.MoveNext
    Loop
    If f Then
        DaysInside = DaysInside + Date - d
    End If
    rst.Close
End Function

Assuming that MovtID sorts the records in the correct order, you could use these custom functions. I have used Movt as name for the table.

 

Function DaysInside(DocID As Long) As Long
    Dim dbs As DAO.Database
    Dim rst As DAO.Recordset
    Dim f As Boolean
    Dim d As Date
    Set dbs = CurrentDb
    Set rst = dbs.OpenRecordset("SELECT * FROM Movt WHERE DocID=" & DocID & _
        " ORDER BY MovtID", _
        dbOpenForwardOnly)
    Do While Not rst.EOF
        Select Case rst!DocMvt
        Case "TransferIn"
            d = rst!MovtDt
            f = True
        Case "TransferOut"
            If f Then
                DaysInside = DaysInside + rst!MovtDt - d
            End If
            f = False
        Case "Finalized"
            If f Then
                DaysInside = DaysInside + rst!MovtDt - d
            End If
            f = False
            Exit Do
        End Select
        rst.MoveNext
    Loop
    If f Then
        DaysInside = DaysInside + Date - d
    End If
    rst.Close
End Function

    Dim dbs As DAO.Database

Outlook doesn't really have an address book.  All your contacts are kept in your Contacts folder.  What appears as "Address Book" is just a view of those contacts in your Contacts folder that have electronic addresses (email or fax).
