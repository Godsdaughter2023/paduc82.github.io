For that amount of data we need some power, otherwise the macro runs "forever".

Copy the code below into a regular module and run sub Main. If you want to test the code before you use it in your file, make a new file and run sub Scenario.

Option Explicit

Sub Main()
  Dim CensusA, CensusC, ReportA, ReportO
  Dim i As Long, j As Long
  Dim Dict As Object 'Scripting.Dictionary
  Dim Changed As Boolean

  'Read in all data
  With Sheets("Census")
    i = .Range("C" & Rows.Count).End(xlUp).Row
    CensusA = .Range("A2").Resize(i - 1).Value
    CensusC = .Range("C2").Resize(i - 1).Value
  End With
  With Sheets("Report")
    i = .Range("A" & Rows.Count).End(xlUp).Row
    ReportA = .Range("A2").Resize(i - 1).Value
    ReportO = .Range("O2").Resize(i - 1).Value
  End With

  'Use a dictionary for faster access
  Set Dict = CreateObject("Scripting.Dictionary")
  For i = 1 To UBound(CensusC)
    If Dict.Exists(CensusC(i, 1)) Then
      Sheets("Census").Select
      Range("C" & i).Select
      MsgBox "Duplicate UID found => Aborted!"
      Exit Sub
    End If
    Dict.Add CensusC(i, 1), i
  Next

  'Compare the UIDs
  For i = 1 To UBound(ReportA)
    If Dict.Exists(ReportA(i, 1)) Then
      j = Dict(ReportA(i, 1))
      If CensusA(j, 1) <> ReportO(i, 1) Then
        'Store the group number into report data
        ReportO(i, 1) = CensusA(j, 1)
        Changed = True
      End If
    Else
      'The UID was not found in report.
    End If
  Next

  'Changes made?
  If Changed Then
    'Write the data back into report
    With Sheets("Report")
      .Range("O2").Resize(UBound(ReportO)) = ReportO
    End With
  End If
End Sub

    .Range("C1") = "UID"

    Group(i, 1) = Int(10000000 * Rnd)

    .Range("A1") = "UID"
    .Range("A2").Resize(Count) = UID
    .Range("O1") = "Group"
    .Range("O2").Resize(Count) = Group
  End With
End Sub
