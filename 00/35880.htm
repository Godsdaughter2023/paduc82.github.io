CommandLine.vbs

'--------------------------------------------------
'Report the full command line for an active program
'12.2.2009 FNL
'--------------------------------------------------
If WScript.Arguments.Count = 0 Then
    sName = InputBox("Name of executable: ", "Exe Locator")
    if sName = "" then WScript.Quit
Else
    sName = WScript.Arguments(0)
End If

if lcase(right(sName, 4)) <> ".exe" then sName = sName & ".exe"

Set oWMIService = GetObject("winmgmts:\\.\root\CIMV2") 
Set cItems = oWMIService.ExecQuery("SELECT * FROM Win32_Process where Name = '" & sName & "'") 

If cItems.Count = 0 Then
    WScript.Echo "Executable """ & sName & """ not found."
Else
    For Each oItem In cItems 
        WScript.Echo "***", oitem.Handle, oItem.CommandLine
    Next
End If
