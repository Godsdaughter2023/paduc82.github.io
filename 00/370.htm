    nRows = t.Rows.Count
    ncols = t.Columns.Count
    Dim neRows As Integer, neCols As Integer 'excel
    Set shex = wbex.Worksheets(1)
    DetermineUsedRange wbex.Worksheets("Extracted"), raex
    If raex Is Nothing Then

        neCols = 0
    Else

Sub DetermineUsedRange(ByRef xs As Excel.Worksheet, ByRef theRng As
Excel.Range)
Dim nFirstRow As Integer, nFirstCol As Integer, _
    nlastrow As Integer, nlastcol As Integer
On Error GoTo handleError
nFirstRow = xs.Cells.Find(What:="*", _
      SearchDirection:=xlNext, _
      SearchOrder:=xlByRows).Row
nFirstCol = xs.Cells.Find(What:="*", _
      SearchDirection:=xlNext, _
      SearchOrder:=xlByColumns).Column
nlastrow = xs.Cells.Find(What:="*", _
      SearchDirection:=xlPrevious, _
      SearchOrder:=xlByRows).Row
nlastcol = xs.Cells.Find(What:="*", _
      SearchDirection:=xlPrevious, _
      SearchOrder:=xlByColumns).Column
Set theRng = xs.Range(xs.Cells(nFirstRow, nFirstCol), _
    xs.Cells(nlastrow, nlastcol))
Exit Sub
handleError:
End Sub

brian tillman [outlook mvp 2007-2019] say:
You cannot erase those. Every default data file must have a certain complement of folders, including Inbox, Outbox, Drafts, Sent Items, Deleted Items, Calendar, Contacts, Notes, Tasks, and Journal. Since IMAP cannot handle non-mail folders, Outlook still creates them, but marks them "This computer only" so you know they don't sync. What you can do is copy the contents of the PST data file calendar, contacts, etc. to the IMAP versions of those folders so all your data is in a single data file.

Just keep in mind that if you delete that IMAP account for any reason (like, say, to remove and re-add the account because of some problem), you'll lose the contents of those folders permanently. This is one reason why some people still have a PST data file to hold their calendar, contacts, and other non-mail folders, making a backup (copy) of that data file periodically. You can even change the default data file to be the PST so that the calendar and contacts folders there are the ones that contain active events. Your mail will still flow through the IMAP folders.

Hi, I think the best way to accomplish what you want to do is, insert MacroButtons or images with the forward and back arrows, into your header and then assign a macro to the click action of the buttons or images.

The code for the next page macro is:

Selection.GoTo What:=wdGoToPage, Which:=wdGoToNext

The code for the previous page macro is:

Selection.GoTo What:=wdGoToPage, Which:=wdGoToPrevious

There is also the commands:  wdGoToLast and wdGoToFirst   available to you
