re:  Matching lists

The range in the formula should extend to the bottom of the "old" list.
So, in the picture, if the list in column B ended at row 3000 the formula should be...
  =If(CountIf($B$2:$B$3000, F2)> 0, "", "NEW")

Note:  the dollar signs "$" are required.  They fix the range size so it remains unchanged as you fill the formula down.
            F2, however, increments as the formula is filled down.

Open the mail and run this to see if it standardizes the email addresses.

 

Sub AddressSMTPRecipients()

    Dim emailAddress As String
    Dim olObj As MailItem
    Dim I As Long
    Dim Report As String
         
    On Error Resume Next
    Set olObj = Application.ActiveInspector.currentItem
    If olObj Is Nothing Then Set olObj = ActiveExplorer.Selection.Item(1)
    On Error GoTo 0
    
    'Debug.Print " - "
   
    For I = 1 To olObj.Recipients.Count
   
        On Error Resume Next
        ' error if not on Exchange
        emailAddress = olObj.Recipients.Item(I).AddressEntry.GetExchangeUser.PrimarySmtpAddress
        On Error GoTo 0
       
        If emailAddress <> "" Then
            'Debug.Print "Exchange Address: " & emailAddress
        Else
            emailAddress = olObj.Recipients.Item(I).Address
            'Debug.Print "Internet Address: " & emailAddress
        End If
               
        'Debug.Print "Email Address is " & emailAddress
       
        Call AddToReport(Report, emailAddress)
       
        emailAddress = ""
       
    Next I
   
    Call CreateReportAsEmail("List of recipient addresses in " & olObj.Subject, Report)
   
    Set olObj = Nothing
    
End Sub

 

 

Private Function AddToReport(Report As String, FieldValue)    
    AddToReport = ""
    AddToReport = FieldValue & vbCrLf
    Report = Report & AddToReport   
End Function

 

 

' VBA SubRoutine which displays a report inside an email
' Programming by Greg Thatcher, http://www.GregThatcher.com

Public Sub CreateReportAsEmail(Title As String, Report As String)
    On Error GoTo On_Error
   
    Dim Session As Outlook.Namespace
    Dim mail As MailItem
    Dim MyAddress As AddressEntry
    Dim Inbox As Outlook.folder
   
    Set Session = Application.Session
    Set Inbox = Session.GetDefaultFolder(olFolderInbox)
    Set mail = Inbox.items.Add("IPM.Mail")
   
    mail.Subject = Title
    mail.Body = Report
   
    mail.Save
    mail.Display
   
Exiting:
    Set Session = Nothing
    Set Inbox = Nothing
    Set mail = Nothing
    Exit Sub

On_Error:
    MsgBox "error=" & Err.Number & " " & Err.Description
    Resume Exiting

End Sub
