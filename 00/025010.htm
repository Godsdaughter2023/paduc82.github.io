=VALUE(LEFT(RIGHT(A1,17),2)&MID(A1,FIND(" ",A1)+1,3)&LEFT(RIGHT(A1,13),4))

=MONTH(1&MID(A1,FIND(" ",A1)+1,3))&"/"&VALUE(LEFT(RIGHT(A1,17),2))&"/"&LEFT(RIGHT(A1,13),4)
