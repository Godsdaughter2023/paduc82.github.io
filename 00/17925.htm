Sub Waterfall_via_ColumnnStacked()
'
Dim ws As Worksheet
Set ws = ActiveSheet 'data in active sheet
'
Dim rng As Range, cel As Range
Set rng = ws.Range("A1:D8") ' range/source data
Set cel = [A10] 'chart/top left cell
Dim N As Integer
N = rng.Rows.Count - 1
'
'delete old chart from active sheet
On Error Resume Next
ws.ChartObjects(1).Delete
'
Dim obj As ChartObject
Set obj = ActiveSheet.ChartObjects.Add(cel.Left, cel.Top, 300, 200) 'chart size/ width=300, height=200
Dim c As Chart
Set c = obj.Chart
'
With c
.ChartType = xlColumnStacked
.SetSourceData Source:=rng
End With
'
With c.FullSeriesCollection(1) 'column val
.Format.Fill.Visible = msoFalse
.Format.Line.Visible = msoFalse
.Points(N).Format.Fill.Visible = msoTrue
.Points(N).Format.Line.Visible = msoTrue
.Points(N).Format.Fill.ForeColor.RGB = RGB(169, 169, 169)
.Points(N).Format.Line.ForeColor.RGB = RGB(169, 169, 169)
End With
'
With c.FullSeriesCollection(2) ' column down
.Format.Fill.ForeColor.RGB = vbRed
.Format.Line.ForeColor.RGB = vbRed
End With
'
With c.FullSeriesCollection(3) 'column up
.Format.Fill.ForeColor.RGB = RGB(0, 176, 80)
.Format.Line.ForeColor.RGB = RGB(0, 176, 80)
.Points(1).Format.Fill.ForeColor.RGB = RGB(169, 169, 169)
.Points(1).Format.Line.ForeColor.RGB = RGB(169, 169, 169)
End With
'
'add labels in Start/End
c.FullSeriesCollection(1).Points(N).Select 'End
c.SetElement msoElementDataLabelCenter
c.FullSeriesCollection(3).Points(1).Select 'Start
c.SetElement msoElementDataLabelCenter
'
'add legend
c.SetElement (msoElementLegendBottom)
c.Legend.LegendEntries(1).Delete ' delete legend <val>, first entry
'
'format chart/plot area
c.ChartArea.Interior.Color = RGB(240, 255, 255)
c.PlotArea.Interior.Color = RGB(240, 255, 255)
cel.Select
End Sub
