bernie deitrick say:

Easy enough:

Sub SaveSheetAsFile2()
    Dim strFName As String
    Dim strFolderPath As String
    Dim WB As Workbook
    
    Set WB = ActiveWorkbook
    On Error GoTo NoSelectedFolder
    
    With Application.FileDialog(msoFileDialogFolderPicker)
        .InitialFileName = "S:\5. Certificates in cust order\"
        .Show
        strFolderPath = .SelectedItems(1)
    End With
    strFName = Range("K5").Value
    'Use the name of the sheet you want to make into the file
    Worksheets("CofC").Copy
    'Optional, to convert to values
    'With ActiveSheet.UsedRange
    '    .Value = .Value
    'End With
    ActiveWorkbook.SaveAs strFolderPath & "\" & strFName, FileFormat:=51  'regular .xlsx
    'Close the new file
    WB.Close False
    Exit Sub
    
NoSelectedFolder:
    MsgBox "There was a problem with selecting the folder - or you cancelled."
End Sub
