No, since Outlook for Mac is using EWS, automapping doesn’t work for it. Given the situation, I recommend you open the full access mailbox manually:

Through a lot of research, I've resolved the issue.




Firstly I would suggest enabling Office setup logging by adding this registry value as your problem may differ to mine.




To enable logging, you can just run the following command from Start -> Run




reg add HKLM\SOFTWARE\Microsoft\ClickToRun\OverRide /v LogLevel /t REG_DWORD /d 3




Then run the Office setup. A ULS log file should be created both in the %temp% folder and the %windir%\temp folder.  The file name will be: <machinename>-<date>-<time>.log




In regards my specific issue, it was to do with the proxy server at work and Windows not resetting the proxy address even though I disabled it in Internet Explorer.




I simply ran the following command in command-prompt to reset the Windows proxy not just IE:




netsh winhttp reset proxy




If your issue is that it's not picking up the proxy details, first set it in Internet Explorer then run the following command in command-prompt:




netsh winhttp import proxy source=ie

Option Explicit

Public Sub Tester()

    Dim SH As Worksheet

    Dim rCol As Range, Rng As Range

    Dim bHide As Boolean, bFound As Boolean

    Dim Btn As Button

    Const sCaption = "Hide Activities"



    Set SH = ActiveSheet

    Set Rng = SH.Range("K:NK")

    Set Btn = SH.Buttons(Application.Caller)



    bHide = Btn.Caption = sCaption

    On Error GoTo XIT

    Application.ScreenUpdating = False

    For Each rCol In Rng.Columns

        bFound = xInColumn(rCol)

        rCol.Hidden = bHide * bFound

    Next rCol

    With Btn

        .Caption = IIf(bHide, "Un" & sCaption, sCaption)

        With .Characters(Start:=1, Length:=Len(Btn.Caption)).Font

            .ColorIndex = IIf(bHide, 4, 3)

        End With

    End With

XIT:

    Application.ScreenUpdating = True

End Sub



'--------->>

Public Function xInColumn(myCol As Range) As Boolean

      On Error Resume Next

    xInColumn = CBool(myCol.Find(What:="x", _

                                 after:=myCol.Cells(1), _

                                 Lookat:=xlPart, _

                                 LookIn:=xlFormulas, _

                                 SearchOrder:=xlByRows, _

                                 SearchDirection:=xlPrevious, _

                                 MatchCase:=False).Row)

    On Error GoTo 0

End Function

'The following Sub simply hides the columns




Sub HideColumns()

    Dim ws As Worksheet

    Dim c As Long

    

    Set ws = ActiveSheet    'Button clicked will always be the ActiveSheet

    

    Application.ScreenUpdating = False

    With ws

        For c = .Columns("K").Column To Columns("NL").Column

            If WorksheetFunction.CountIf(.Range(.Cells(2, c), _

                    .Cells(.Rows.Count, c).End(xlUp)), "X") = 0 Then

                    

                .Columns(c).Hidden = True

            End If

        Next c

    End With

    

    Application.ScreenUpdating = True

End Sub










'The following sub will hide the columns or

'Unhide previously hidden columns




Sub HideUnHideColumns()

    Dim ws As Worksheet

    Dim c As Long

    Dim shpButton As Shape

    

    Set ws = ActiveSheet    'Button clicked will always be the ActiveSheet

    

    Set shpButton = ws.Shapes(Application.Caller)

    

    Application.ScreenUpdating = False

    

    If shpButton.TextFrame.Characters.Text = "Hide Columns" Then

       

        With ws

            For c = .Columns("K").Column To Columns("NL").Column

                If WorksheetFunction.CountIf(.Range(.Cells(2, c), _

                        .Cells(.Rows.Count, c).End(xlUp)), "X") = 0 Then

                        

                    .Columns(c).Hidden = True

                End If

            Next c

        End With

        shpButton.TextFrame.Characters.Text = "UnHide columns"

        

    Else

        ws.Columns.Hidden = False

        shpButton.TextFrame.Characters.Text = "Hide Columns"

        

    End If

    

    Application.ScreenUpdating = True

End Sub
