Sub ADiocese1()
Dim i As Long, p As Long
With ActiveDocument.MailMerge
  .Destination = wdSendToNewDocument
  For i = 1 To recs_2_merge
    With .DataSource
      .FirstRecord = i
      .ActiveRecord = i
      .LastRecord = i
      p = .DataFields("PAGE")
    End With
    .Execute Pause:=False
    With ActiveDocument
      .PrintOut Range:=wdPrintRangeOfPages, Pages:="p" & p
      .Close False
    End With
  Next
End With
End Sub

Remove an alias from your account

Remove an alias from your account

  1. In Outlook.com, click your user name in the upper-right corner.

  2. Click Account settings. You might have to sign in again.

  3. Click Overview, and then click Add or change aliases.

  4. Click Remove under the alias.

  5. Click Remove alias.
