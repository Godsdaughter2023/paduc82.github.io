Both features are dependent on the display of the Status Bar. The page info is automatic but the inclusion of word count is individually controlled. In Word> Preferences> View make sure the boxes are checked for:

ossiemac say:

Another code example. I thought that it should be possible to find the last button on the worksheet irrespective of its' index number or when created so I did a little testing. The following code addresses the Top and Left properties of the button to determine which is the last button on the worksheet.

 

Button positions are tested across the worksheet and down. The Top of the button is the first priority and then the left position is the second priority for buttons across the worksheet with the same top position.

 

Sub ReturnLastButtonOnSheet()
    'This is the last button on the worksheet _
     irrespective of its' name, index number or when created.
     'Based on Top property and Left property of button
    Dim shp As Shape
    Dim strShapeName As String
    Dim dblTop As Double
    Dim dblLeft As Double
   
    For Each shp In ActiveSheet.Shapes
        If shp.Type = msoFormControl Then
            If shp.FormControlType = xlButtonControl Then
                If shp.Top > dblTop Then
                    dblTop = shp.Top    'Increment dblTop
                    dblLeft = shp.Left
                    strShapeName = shp.Name
                Else
                    If shp.Top = dblTop Then
                        If shp.Left > dblLeft Then
                            dblLeft = shp.Left
                            strShapeName = shp.Name
                        End If
                    End If
                End If
            End If
        End If
    Next shp
   
    MsgBox "Last button on worksheet is: " & strShapeName
   
End Sub

Sub UpdateLinks()

Dim ExcelFile
Dim exl As Object

Set exl = CreateObject("Excel.Application")
ExcelFile = exl.Application.GetOpenFilename()
If ExcelFile = False Then Exit Sub

For i = 1 To ActivePresentation.Slides.Count
  For j = 1 To ActivePresentation.Slides(i).Shapes.Count
    With ActivePresentation.Slides(i).Shapes(j)
        If .Type = msoChart Then
            With .LinkFormat
'                Debug.Print .SourceFullName

                .SourceFullName = ExcelFile
                .AutoUpdate = ppUpdateOptionManual
            End With
        End If
    End With
  Next j
Next i


Set exl = Nothing
End Sub
