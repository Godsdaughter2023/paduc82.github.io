cjlim say: Try checking the following as they all affect Caller ID.

The most common causes of caller ID problems are:

- duplicate entries in your contacts - you appear to have checked this, however I wonder if your check included searching for the same number in different formats? Try searching for a PART of the number as if for example you search for +44 123 456 7890, it won't find +441234567890, whereas if you search for 7890 it would.

Having different formats is a problem for caller ID.

- using your international dial code instead of "+", eg: using 00441234567890 (wrong) instead or +441234567890 (right).

- spaces in numbers (some phones, some firware versions only), eg: +44 123 456 7890 (wrong) instead or +441234567890 (right).

- MS Outlook putting "(0)" in number - if you define a country for a contact in MS Outlook it will automatically add a country code and insert a "(0)" in the number, eg: +441234567890 becomes +44 (0) 1234567890. You need to remove the "(0)"s

- there has been one report of 9 and 10 digit numbers causing problems, one doesn't work with SMS the other with phone calls. I forget which way around it was ..

Option Explicit

''//  25 May 2003 //
''// Amended Ivan F Moala

Private Declare Function GetActiveWindow Lib "user32" () As Long
Private Declare Function FindWindowEx Lib "user32" Alias "FindWindowExA" ( _
    ByVal hWnd1 As Long, ByVal hWnd2 As Long, ByVal lpsz1 As String, _
    ByVal lpsz2 As String) As Long
Private Declare Function SendMessage Lib "user32" Alias "SendMessageA" ( _
    ByVal hWnd As Long, ByVal wMsg As Long, ByVal wParam As Long, lParam As Any) As Long
Private Declare Function SetTimer Lib "user32" ( _
    ByVal hWnd As Long, ByVal nIDEvent As Long, ByVal uElapse As Long, _
    ByVal lpTimerFunc As Long) As Long
Private Declare Function KillTimer Lib "user32" ( _
    ByVal hWnd As Long, ByVal nIDEvent As Long) As Long
Private Declare Function GetForegroundWindow Lib "user32" () As Long

Private Const nIDE As Long = &H100
Private Const EM_SETPASSWORDCHAR = &HCC
Private hdlEditBox As Long

Private Function TimerFunc(ByVal hWnd As Long, ByVal wMsg As Long, ByVal nEvent As Long, _
    ByVal nSecs As Long) As Long
  Dim hdlwndAct As Long
  '// Do we have a handle to the EditBox
  If hdlEditBox > 0 Then Exit Function
  '// Get the handle to the ActiveWindow
  hdlwndAct = GetActiveWindow()
  '// Get the Editbox handle
  hdlEditBox = FindWindowEx(hdlwndAct, 0, "Edit", "")
  '// Set the password character for the InputBox
  SendMessage hdlEditBox, EM_SETPASSWORDCHAR, Asc("*"), ByVal 0
End Function

Public Function InputBoxPwd(Prompt As String, Optional Title As String, _
    Optional Default As String, Optional XPos As Long, Optional YPos As Long, _
    Optional Helpfile As String, Optional Context As Long) As String
  Dim Fgrndhdl As Long
  '// Initialize
  hdlEditBox = 0
  Fgrndhdl = GetForegroundWindow
  '// Windows-Timer
  SetTimer Fgrndhdl, nIDE, 100, AddressOf TimerFunc
  '// Main InputBox
  If XPos Then
    InputBoxPwd = InputBox(Prompt, Title, Default, XPos, YPos, Helpfile, Context)
  Else
    InputBoxPwd = InputBox(Prompt, Title, Default, , , Helpfile, Context)
  End If
  '// Kill the correct Timer
  KillTimer Fgrndhdl, nIDE
End Function

Public Sub Example_InputBoxPwd()
  Dim x As String
  x = InputBoxPwd("Please enter password")
  If x = vbNullString Then
    MsgBox "User Cancelled"
  Else
    MsgBox "User entered " & x
  End If
End Sub
