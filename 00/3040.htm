Const UNLOCKABLE = 438

VBA macro to add / rename Sent Date+Time to Outlook attachment

I got this VBA for a macro from

It lets you manually rename each attachment in an outlook (2010) email message. Works nicely. I was hoping to prefix it with the "sent" date+time. My idea is to do the rename, then remove and save the attachment with kopf outlook attachment remover. I have found VBA that will save attachments with date etc, but the advantage of that nice little kopf program is that you can still click on the link to view the file and even reintroduce it.

I suppose a simpler routine might be to use the existing VBA below, then run some extra VBA that added a date+time to all attachments.

Any ideas/ VBA solutions greatly appreciated.  

Sub RenameAttachments()

Dim olkItem As Outlook.MailItem, olkAttachment As Outlook.Attachment

 Dim strFolder As String, strFilename As String

 Dim strExtension As String, lngExtIndex As Long, strBaseFilename As String

 Dim FileChanges() As String, Cnt As Long, i As Long

 If Application.ActiveInspector Is Nothing Then

  MsgBox "Please run this when the message you want to check is active. Exiting."

  Exit Sub

 Else

  Set olkItem = Application.ActiveInspector.CurrentItem

 End If

 If olkItem.Attachments.Count = 0 Then

  MsgBox "No attachments found?"

 Else

  strFolder = Environ("tmp") & "\"

  Cnt = 0

  ReDim FileChanges(1, Cnt) 'array to store filename changes. 0=index, 1=newname'

  For Each olkAttachment In olkItem.Attachments

   strFilename = olkAttachment.FileName

   lngExtIndex = InStrRev(strFilename, ".")

   strBaseFilename = Left(strFilename, lngExtIndex - 1)

   strExtension = Mid(strFilename, lngExtIndex)

   strFilename = InputBox("What shall we rename the attachment to?", _

    "Rename Attachment", strBaseFilename)

   If Len(strFilename) > 0 And strFilename & strExtension <> olkAttachment.FileName Then

    ReDim Preserve FileChanges(1, Cnt)

    FileChanges(0, Cnt) = olkAttachment.Index

    FileChanges(1, Cnt) = strFilename & strExtension

    Cnt = Cnt + 1

   End If

  Next

  If Cnt > 0 Then

   For i = Cnt - 1 To 0 Step -1

    For Each olkAttachment In olkItem.Attachments

     If olkAttachment.Index = FileChanges(0, i) Then

      olkAttachment.SaveAsFile strFolder & FileChanges(1, i)

      olkAttachment.Delete

      olkItem.Attachments.Add strFolder & FileChanges(1, i)

      Exit For

     End If

    Next

   Next

  End If

 End If

 Set olkItem = Nothing

 Set olkAttachment = Nothing

End Sub
