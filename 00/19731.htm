norman david jones say:

Hi Lyn_750,

Hi there, please can anyone help me:  In Excel I want to show a number e.g. 3,550,000 as 3 million, Five hundred AND fifty thousand Rand.  I don't want any cents. I know how to put the Macro into the spreadsheet, I just need the correct VB code. Many thanks.

The And is very important in front of any thousands.  Sorry this is very urgent if possible.

See the code suggested at:

            http://www.meadinkent.co.uk/xlnumberstext.htm

In order not to show any cents and to add the final word Rand, replace the code of the WordNum function with:

'=========>>
Function WordNum(MyNumber As Double) As String
    Dim DecimalPosition As Integer, ValNo As Variant, StrNo As String
    Dim NumStr As String, n As Integer, Temp1 As String, Temp2 As String
    ' This macro was written by Chris Mead - www.MeadInKent.co.uk

    If Abs(MyNumber) > 999999999 Then
        WordNum = "Value too large"
        Exit Function
    End If

    SetNums

    ' String representation of amount (excl decimals)
    NumStr = Right("000000000" & Trim(Str(Int(Abs(MyNumber)))), 9)
    ValNo = Array(0, Val(Mid(NumStr, 1, 3)), Val(Mid(NumStr, 4, 3)), Val(Mid(NumStr, 7, 3)))

    For n = 3 To 1 Step -1    'analyse the absolute number as 3 sets of 3 digits
        StrNo = Format(ValNo(n), "000")

        If ValNo(n) > 0 Then
            Temp1 = GetTens(Val(Right(StrNo, 2)))
            If Left(StrNo, 1) <> "0" Then
                Temp2 = Numbers(Val(Left(StrNo, 1))) & " hundred"
                If Temp1 <> "" Then Temp2 = Temp2 & " and "
            Else
                Temp2 = ""
            End If

            If n = 3 Then
                If Temp2 = "" And ValNo(1) + ValNo(2) > 0 Then Temp2 = "and "
                WordNum = Trim(Temp2 & Temp1)
            End If
            If n = 2 Then WordNum = Trim(Temp2 & Temp1 & " thousand " & WordNum)
            If n = 1 Then WordNum = Trim(Temp2 & Temp1 & " million " & WordNum)

        End If
    Next n

    NumStr = Trim(Str(Abs(MyNumber)))

    If Len(WordNum) = 0 Or Left(WordNum, 2) = " p" Then
        WordNum = "Zero" & WordNum
    End If
    WordNum = WordNum & " Rand"
End Function
'<<=========

The other procedures remain unaltered.

===

Sub Main()
  Dim rptWS As Worksheet
  Const filesType = "*.xlsb" ' change to type you want to detect/process
 
  Dim colFiles As New Collection
  Dim vFile As Variant
  Dim lastRow As Long ' to loop through files found to process
  Dim rowPtr As Long
  Dim folderChosen As String
  Dim fullName As String
  Dim filenameOnly As String
  Dim budgetWB As Workbook
  Dim budgetWS As Worksheet

  If folderChosen = vbNullString Then
    'user cancelled
    Exit Sub
  End If
  If Right(folderChosen, 1) <> Application.PathSeparator Then
    folderChosen = folderChosen & Application.PathSeparator
  End If

  With rptWS
    .Cells.ClearContents
    .Cells(1, 1) = "Files To Be Processed"
    .Cells(1, 2) = "Status"
  End With
 
  RecursiveDir colFiles, folderChosen, filesType, True
 
  For Each vFile In colFiles
    rptWS.Range("A" & Rows.Count).End(xlUp).Offset(1, 0) = vFile
  Next vFile
 
  lastRow = rptWS.Range("A" & Rows.Count).End(xlUp).Row
  If lastRow < 2 Then
    'no files found, quit
    MsgBox "No budget files identified.", vbOKOnly + vbInformation, "Halting Process"
    Set rptWS = Nothing
    Exit Sub
  End If
  For rowPtr = 2 To lastRow
    fullName = rptWS.Range("A" & rowPtr)
    filenameOnly = Right(fullName, Len(fullName) - InStrRev(fullName, Application.PathSeparator))
    Set budgetWB = Workbooks.Open(fullName, False, False)
    'probably will fail because macros not enabled in the opened workbook!!!!
    Application.Run filenameOnly & "!" & "create_import_sheet"
    budgetWB.Close True
    Set budgetWB = Nothing
  Next ' end rowPtr loop
 
 
  Set rptWS = Nothing
End Sub
