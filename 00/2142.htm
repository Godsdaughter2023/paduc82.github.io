bettertoday say:

Thanks a lot for providing me with this very valuable code example.

I have taken this example and updated it for my requirements. Worked as a charm!

Just in case anyone might have the same issue, here's the code I've written (I also added a screenshot of the code view in case this forum's text formatting (automatic text wrapping, for example) may be confusing to readers):


' / ========================================================================================= \
' | -- For updating all events of a given location, set this function's parameters to the
' | -- required values, then run this function by clicking its name and hitting the [F5] key.
' | -----------------------------------------------------------------------------------------
' | -- parameter #1: Location (calendar event must match this location to be updated).
' | -- parameter #2: Category to be assigned to all events of the specified location.
' | -- parameter #3: Category to search for. Optional; if empty, all events of a given
' | --               location will be replaced regardless of the currently assigned category.
' \ ========================================================================================= /
Public Sub SetCountryEventColour()
  SetCountryEventColour_ "France", "French Holiday", "Holiday"
End Sub


' / ========================================================================================= \
' | -- This function handles the actual category update.
' \ ========================================================================================= /
Public Sub SetCountryEventColour_(location$, newCategories$, Optional oldCategories$ = "")
  Dim Items As Outlook.Items
  Dim ns As Outlook.NameSpace
  Dim i As Object
  Dim ai As AppointmentItem
 
  On Error GoTo ErrorOccured

  Set ns = Application.GetNamespace("MAPI")
  Set Items = ns.GetDefaultFolder(olFolderCalendar).Items
 
  If Items Is Nothing Then
    MsgBox "No items in calendar", vbExclamation Or vbOKOnly, "Warning"
  Else
    On Error GoTo ErrorOccuredInLoop
 
    For Each i In Items
      If TypeOf i Is AppointmentItem Then
        Set ai = i
        
        If ai.AllDayEvent And ai.location = location Then
          If oldCategories <> "" And ai.Categories <> oldCategories Then GoTo SkipToNextItem
          ai.Categories = newCategories
          ai.Save
        End If
      End If
      
SkipToNextItem:
    Next

    MsgBox "Successfully updated categories on calendar items.", vbInformation Or vbOKOnly
  End If

QuitAfterError:
  On Error GoTo 0
  Exit Sub

ErrorOccured:
  MsgBox Err.Description, vbCritical Or vbOKCancel, "Error"
  Resume QuitAfterError

ErrorOccuredInLoop:
  If MsgBox(Err.Description + vbNewLine + vbNewLine + "Continue?", vbCritical Or vbOKCancel, "Error") = vbOK Then Resume SkipToNextItem
  Resume QuitAfterError
End Sub

ossiemac say:

Try the following and note the comments.

Private Sub Workbook_BeforeSave(ByVal SaveAsUI As Boolean, Cancel As Boolean)

     Application.ScreenUpdating = False
     Dim rngCondFormat As Range
     Dim rng As Range
    
     'Attempting to Reference SpecialCells(xlCellTypeAllFormatConditions)
     'produces an error if no cells meet the criteria so need to test
     'using the following routine.
     On Error Resume Next
     Set rngCondFormat = ActiveSheet.UsedRange.SpecialCells(xlCellTypeAllFormatConditions)
     On Error GoTo 0
    
     If Not rngCondFormat Is Nothing Then   'Not nothing then is something (Valid)
        For Each rng In rngCondFormat
            'Use DisplayFormat to return Conditional Format color.
            If rng.DisplayFormat.Interior.ColorIndex = 3 Then
                MsgBox ("Cell with color fill found in cell " & rng.Address(0, 0))
                Cancel = True
                Application.ScreenUpdating = True
                Exit Sub
            End If
        Next rng
    End If
     Application.ScreenUpdating = True
 End Sub
