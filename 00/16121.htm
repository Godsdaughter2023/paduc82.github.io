jllatham say:

The code below works with your sample workbook.  The number of rows is irrelevant: the code figures out how many based on the entries in column A.  The results will be placed on a new sheet that will be named SplitDataSheet and placed right behind the original

See initial comments in the code for getting it into your workbook.  Once it is in your workbook, save the file as a Macro Enabled Workbook (type .xlsm or .xlsb) to preserve the code.

To run the macro (named SplitDataIntoRows) select the sheet with the 'raw' data on it and use the {Macros} icon on either the [View] or [Developer] tab in the workbook.

Sub SplitDataIntoRows()
'this goes into a Regular Code module.  Instructions to put it
'into one can be found on this web page:
' http://www.contextures.com/xlvba01.html#videoreg
'
'the page with the data to be split must be selected when
'you start this macro.
'
'results will be placed on a sheet named [SplitDataSheet]
'which will be created if needed.
'
  Dim splitChar As String
  Const eLastCol = "A"
  Dim eLast As String ' employee last name
  Const ePhoneCol = "D"
  Const eTotalCol = "L"
  Dim eTotal As Integer ' total attending (is text, convert to #)
  Const cFirstCol = "E"
  Dim cFirst As Variant
  Const cLastCol = "F"
  Dim cLast As Variant
  Const cAgeCol = "G"
  Dim cAge As Variant ' ages, as text
  Const cGenderCol = "H"
  Dim cGender As Variant
  Const cRelCol = "I"
  Dim cRel As Variant
  Const cNeedsCol = "J"
  Dim cNeeds As Variant
  Const cSNeedCol = "K"
  Dim cSNeed As Variant
 
  Dim dataWS As Worksheet
  Dim sdWS As Worksheet
  Dim lastSrcRow As Long
  Dim nextDestRow As Long
  Dim cellColor As Long
  Dim srcRowPtr As Long
  Dim formatRange As Range
  Dim eInfo1 As Range
  Dim eInfo2 As Range
  Dim LC As Integer
 
  splitChar = Chr$(10)
  Application.ScreenUpdating = False
  Set dataWS = ActiveSheet
  'need a 'SplitDataSheet' worksheet
  On Error Resume Next
  Set sdWS = ThisWorkbook.Worksheets("SplitDataSheet")
  If Err <> 0 Then
    'assume sheet does not exist, create it
    Err.Clear
    ThisWorkbook.Worksheets.Add after:=ActiveSheet
    Set sdWS = ActiveSheet
    sdWS.Name = "SplitDataSheet"
  End If
  sdWS.Cells.Clear
  dataWS.Activate
  lastSrcRow = dataWS.Range(eLastCol & Rows.Count).End(xlUp).Row
  'copy titles row
  dataWS.Range(eLastCol & "1:" & eTotalCol & 1).Copy
  sdWS.Range("A1").PasteSpecial xlPasteAll
  For srcRowPtr = 2 To lastSrcRow
    'is this a row to be split or not?
    'if col E entry has Chr$(10) in it, yes, otherwise no
    If InStr(dataWS.Range(cFirstCol & srcRowPtr), splitChar) = 0 Then
      'not a row to be split, just copy it
      dataWS.Range("A" & srcRowPtr).EntireRow.Copy
      nextDestRow = sdWS.Range("A" & Rows.Count).End(xlUp).Row + 1
      sdWS.Range("A" & nextDestRow).PasteSpecial xlPasteAll
      With sdWS.Range(eTotalCol & nextDestRow)
        .NumberFormat = "General"
        .Value = .Value * 1
      End With
      With sdWS.Range(cAgeCol & nextDestRow)
        .NumberFormat = "General"
        .Value = .Value * 1
      End With
    Else
      'this is a row that must be split
      'split children's first names
      cFirst = Split(dataWS.Range(cFirstCol & srcRowPtr), splitChar)
      cLast = Split(dataWS.Range(cLastCol & srcRowPtr), splitChar)
      cAge = Split(dataWS.Range(cAgeCol & srcRowPtr), splitChar)
      cGender = Split(dataWS.Range(cGenderCol & srcRowPtr), splitChar)
      cRel = Split(dataWS.Range(cRelCol & srcRowPtr), splitChar)
      cNeeds = Split(dataWS.Range(cNeedsCol & srcRowPtr), splitChar)
      cSNeed = Split(dataWS.Range(cSNeedCol & srcRowPtr), splitChar)
      Set formatRange = dataWS.Range(eLastCol & srcRowPtr & ":" & eTotalCol & srcRowPtr)
      Set eInfo1 = dataWS.Range(eLastCol & srcRowPtr & ":" & ePhoneCol & srcRowPtr)
      Set eInfo2 = dataWS.Range(eTotalCol & srcRowPtr)
      For LC = LBound(cFirst) To UBound(cFirst)
        nextDestRow = sdWS.Range("A" & Rows.Count).End(xlUp).Row + 1
        formatRange.Copy
        sdWS.Range("A" & nextDestRow).PasteSpecial xlPasteFormats
        eInfo1.Copy
        sdWS.Range("A" & nextDestRow).PasteSpecial xlPasteAll
        'only display total attending in first entry!
        If LC = LBound(cFirst) Then
          eInfo2.Copy
          sdWS.Range(eTotalCol & nextDestRow).PasteSpecial xlPasteAll
          With sdWS.Range(eTotalCol & nextDestRow)
            .NumberFormat = "General"
            .Value = .Value * 1
          End With
        End If
        sdWS.Range(cFirstCol & nextDestRow) = cFirst(LC)
        sdWS.Range(cLastCol & nextDestRow) = cLast(LC)
        With sdWS.Range(cAgeCol & nextDestRow)
          .NumberFormat = "General"
          .Value = Val(cAge(LC))
        End With
        sdWS.Range(cGenderCol & nextDestRow) = cGender(LC)
        sdWS.Range(cRelCol & nextDestRow) = cRel(LC)
        sdWS.Range(cNeedsCol & nextDestRow) = cNeeds(LC)
        sdWS.Range(cSNeedCol & nextDestRow) = cSNeed(LC)
      Next ' end LC loop
    End If
  Next ' end srcRowPtr loop
  'set column widths
  sdWS.Columns(eLastCol & ":" & eTotalCol).AutoFit
  'cleanup assigned resources
  If Err <> 0 Then
    Err.Clear
  End If
  On Error GoTo 0
  Set formatRange = Nothing
  Set eInfo1 = Nothing
  Set eInfo2 = Nothing
  Set dataWS = Nothing
  'announce job finished
  MsgBox "Data has been split to worksheet " & sdWS.Name, _
   vbOKOnly + vbInformation, "Task Completed"
  Set sdWS = Nothing
End Sub
