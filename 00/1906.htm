I do not understand why you want to use the field, it is not necessary.

Word has a collection for all templates:

 

Sub Test1()
  Dim aTemplate As Template
  For Each aTemplate In Templates
     MsgBox aTemplate.Name & " - " & aTemplate.Path

But if your template is not in that collection, you can search for it at different locations. I suggest to do so, because some of the pathes may be empty:

 

Sub Test2()
  Dim Path As String
  Dim TemplateName As String
  Dim i As Integer


  TemplateName = "test.dot"
 
  For i = 1 To 3
    'Try a path
    Select Case i
      Case 1
        Path = Options.DefaultFilePath(wdWorkgroupTemplatesPath)
      Case 2
        Path = Options.DefaultFilePath(wdUserTemplatesPath)
      Case 3
        Path = Options.DefaultFilePath(wdCurrentFolderPath)
    End Select
    'Is the template there?
    If Len(Dir(Path & "\" & TemplateName)) > 0 Then Exit For
  Next
  If i = 4 Then
    MsgBox "Template not found"
    Exit Sub
  End If
 
  Documents.Open Path & "\" & TemplateName
End Sub
