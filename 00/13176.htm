graham mayor say:
Without knowing exactly what is in your template it's all a bit hit and miss, but at its simplest you would need something like the following. Add a button to the Calendar Tools ribbon to call the macro. Select the appointment in the Calendar view and click the button which will then appear. Put the name and path of the template where indicated. This will add the date and time of the appointment and a short text.

Public Sub CreateEmailFromAppt()
Dim olMailItem As Outlook.MailItem
Dim olMail As Outlook.MailItem
Dim olAppt As Outlook.AppointmentItem
Dim strAppt As String
Dim olInspector As Outlook.Inspector
Dim wdDoc As Object
Dim wdRange As Object

    Set olAppt = ActiveExplorer.Selection.Item(1)
    Set olMail = CreateItemFromTemplate("D:\Path\Template.oft")

    strAppt = "Your appointment is confirmed for " & _
              Format(olAppt.Start, "Long Date") & " at " & _
              Format(olAppt.Start, "hh:mm am/pm") & vbCr & vbCr & _
              "Please see the map below which gives our location:" & vbCr & vbCr

    With olMail
        .Display

        Set olInspector = .GetInspector
        Set wdDoc = olInspector.WordEditor
        Set wdRange = wdDoc.Range(0, wdDoc.Characters.Count)
        wdRange.InsertBefore strAppt
    End With
End Sub
