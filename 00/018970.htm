Since the problem is related to search feature, please try rebuilding the Spotlight index on your Mac. Here are the steps:

1. Choose Apple menu > System Preferences, then click Spotlight.

2. Click the Privacy tab.

3. Drag the folder or disk that you want to index again to the list of locations that Spotlight is prevented from searching. Or click the Add (+) button and select the folder or disk to add.

Hi,

Try the revised code below:

 

Sub PortMacro()
' PortMacro Macro
' Keyboard Shortcut: Ctrl+g
'
    Dim appWD As Word.Application
    Dim i As Long, FinalRow As Long
   
     'File open dialog box (exit sub if user cancels):
    If Application.Dialogs(xlDialogOpen).Show = False Then Exit Sub
   
     'Disable screen updating in Excel:
    Application.ScreenUpdating = False
   
     'Initialize and declare appWD object for opening Word. Use a predefined template:
    Set appWD = CreateObject("Word.Application.8")
    appWD.Documents.Add Template:="C:\test.dotx"
   
     'Format borders of cells for transposed data:
    With Range("O1:P5")
        .HorizontalAlignment = xlLeft
        .VerticalAlignment = xlBottom
        With .Borders
            .LineStyle = xlContinuous
            .Color = -1003520    'Note you could use:    .Color = RGB(0, 176, 240)
            .Weight = xlThick
        End With
    End With
   
     'Variable i to increment to end of the column:
    FinalRow = Range("A600").End(xlUp).Row
    For i = 2 To FinalRow
         'Show message in Excel's status bar to indicate progress:
        Application.StatusBar = "Processing record " & i - 1 & " of " & FinalRow - 1
       
         'Copy incremental data from col A and paste it as heading in Word:
        Range("A" & i).Copy
        appWD.Selection.EndKey Unit:=wdStory
        appWD.Selection.TypeParagraph
        appWD.Selection.PasteSpecial DataType:=wdPasteText 'THIS WILL PASTE THE HEADING AS TEXT, NOT A SINGLE CELL TABLE
        appWD.Selection.EndKey Unit:=wdStory
        appWD.Selection.TypeParagraph
       
         'Select headers and incremental data, transpose paste them on a blank column:
        Range("B1:F1,B" & i & ":F" & i).Copy
        Range("O1").PasteSpecial Paste:=xlPasteValuesAndNumberFormats, Transpose:=True
       
         'Copy formatted data and paste it to word. Every increment on new page:
        Range("O1:P5").Copy
        appWD.Selection.Paste
        If i < FinalRow Then appWD.Selection.InsertBreak Type:=wdPageBreak
    Next i
   
    appWD.Visible = True
    appWD.Activate
   

    Range("O1:P5").Clear
    Range("A1").Select
    Application.ScreenUpdating = True
    Application.StatusBar = False
End Sub
