wsuds say:

Since this is actually a system error (either in the operating system or the scheduler itself) and not an error in the scheduled task, one thing I've been able to do that helps, is to reset the "Running" flag in the .job file. This way the scheduler will now know that the task is not running and go ahead and run it at the next scheduled run time. I have a vb script to do this.

Right now it only resets the job files you tell it to, but I will be modifying it to check all .job files in the windows\tasks folder on a regular basis, check to see if the task is actually running or not, and if not, reset the “Running” flag.

Here’s the current vb script. You can drag and drop the job file of the stuck job onto it and it will reset the flag.

SetJobToIdle.vbs

---------------------------

option explicit

Const ForReading = 1, ForWriting = 2, ForAppending = 8

Const TristateFalse = 0

dim fso 'File System Object

dim ii 'Loop Index

dim jobfile 'Job File Name

on error goto 0

'

' Create a file system object.

'

Set fso = CreateObject("Scripting.FileSystemObject")

'

' Just drag and drop one or more files onto this vbs script.

'

for ii = 0 to WScript.Arguments.Count-1

    jobfile = WScript.Arguments.Item(ii)

    SetJobToIdle jobfile

next

Set fso = Nothing

wscript.quit

 

sub SetJobToIdle(jobfile)

    dim gfo 'Get File Object

    dim tso 'Text Stream Object

    dim inpline 'Input Line

    '

    ' Job File fixed portion layout and values example:

    '

    ' 01 05 Product Version 0501 = version 5.1

    ' 01 00 File Version 0001 = 1

    ' 67 6b uuid

    ' b2 74 uuid0

    ' e3 3f uuid1

    ' aa 42 uuid2

    ' aa 3a uuid3

    ' 44 e4 uuid4

    ' 51 b5 uuid5

    ' d5 00 uuid6

    ' 46 00 uuid7

    ' 4c 01 app name len offset 014c = 332

    ' 00 00 trigger offset

    ' 00 00 error retry count

    ' 3c 00 error retry interval 003c

    ' 0a 00 Idle Deadline 000a

    ' 20 00 Idle Wait 0020

    ' 00 00 Priority

    ' 00 14 ...

    ' 73 0f Max Run Time

    ' 3a 01 ...

    ' 00 c0 Exit Code c000

    ' 00 13 ...

    ' 04 00 Status 0004

    ' c0 00 ...

    ' 80 21 db 07 flags

    ' db 07 year 07db = 2011

    ' 09 00 month 0009 = 9

    ' 02 00 week day 0002 = Tuesday

    ' 0d 00 day 000d = 13

    ' 0f 00 hour 000f = 15

    ' 39 00 minute 0039 = 57

    ' 05 00 second 0005 = 5

    ' 10 03 milli seconds 0310 = 784

    ' 01 00 running instance count 0001 = 1 = Running

    '

    ' Get the job file.

    '

    set gfo = fso.GetFile(jobfile)

    '

    ' Open the job file for reading.

    '

    set tso = gfo.OpenAsTextStream(ForReading, TristateFalse)

    '

    ' Read in the whole job file.

    '

    inpline = tso.Read(gfo.Size)

    '

    ' Echo out the current value of the 69th byte.

    '

    wscript.echo asc(mid(inpline,69,1))

    '

    ' Set the 69th byte to 0 (i.e., idle).

    '

    inpline = mid(inpline,1,68) & chr(00) & mid(inpline,70)

    '

    ' Close the job file.

    '

    tso.Close

    '

    ' Open the job file for writing.

    '

    set tso = gfo.OpenAsTextStream(ForWriting, TristateFalse)

    '

    ' Write out the modified contents to the job file.

    '

    tso.Write inpline

    '

    ' Close the job file.

    '

    tso.Close

    '

    ' Set our objects to nothing.

    '

    set gfo = Nothing

    set tso = Nothing

end sub
