per jessen say:

This macro will do it for you.

 

If you haven't worked with macros before, this should get you started.

Open the VBA editor (ALT+F11)

Insert > Module

Paste the code below into the white code sheet.

Now you can run the macro. Just remember to use a copy of your workbook for testing (Just in case).

 

Sub CustomSort()
Dim TempSh As Worksheet
Dim TargetSh As Worksheet
Dim SortRng As Range
Dim CopyRng As Range
Dim DestCell As Range
Dim LastRow As Long

Application.ScreenUpdating = False

Set TargetSh = Worksheets("Sheet1")
Set TempSh = Worksheets.Add
Set DestCell = TargetSh.Range("A2")
LastRow = TargetSh.Range("A1").End(xlDown).Row

TargetSh.Cells.Copy TempSh.Range("A1")
Set SortRng = TempSh.Range("A1:AA" & LastRow) 'Change AA to Last column with data
Set CopyRng = TempSh.Range("A2:AA" & LastRow) 'Change AA to Last column with data
TargetSh.Range("A2:AA" & LastRow).ClearContents 'Change AA to Last column with data

TempSh.Sort.SortFields.Add Key:=Range("F2:F" & LastRow), _
    SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
TempSh.Sort.SortFields.Add Key:=Range("A2:A" & LastRow), _
    SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
With TempSh.Sort
    .SetRange SortRng
    .Header = xlYes
    .MatchCase = False
    .Orientation = xlTopToBottom
    .SortMethod = xlPinYin
    .Apply
End With

Do
    SortRng.AutoFilter Field:=1, Criteria1:= _
            SortRng(1, 1).Offset(1, 0).Value
    Set CopyRng = SortRng.Offset(1, 0).SpecialCells(xlCellTypeVisible)
    SortRng.AutoFilter
    CopyRng.Copy DestCell
    CopyRng.EntireRow.Delete
    Set DestCell = DestCell.End(xlDown).Offset(1, 0)
Loop Until TempSh.Range("A2").Value = ""

Application.DisplayAlerts = False
TempSh.Delete
Application.DisplayAlerts = True
Application.ScreenUpdating = True
End Sub
