andreas killer say:

At first copy your MyList and MyList2 arrays into this sub, customize the path "Z:\MyList.xlsx" and execute it once to create the Excel file.

Sub CreateXLSX()
  Dim MyList As Variant
  Dim MyList2 As Variant
 
  MyList = Array("x", "y", "z")
  MyList2 = Array("x <ex>", "y <why>", "z <zee>")
 
  Dim xlApp As Object ' Excel.Application
  Dim xlWb As Object 'Excel.Workbook
  Set xlApp = CreateObject("Excel.Application")
  Set xlWb = xlApp.Workbooks.Add
 
  With xlWb.ActiveSheet.Range("A1")
    .Resize(UBound(MyList) + 1, 1) = xlApp.WorksheetFunction.Transpose(MyList)
    .Offset(, 1).Resize(UBound(MyList2) + 1, 1) = xlApp.WorksheetFunction.Transpose(MyList2)
  End With
  xlWb.SaveAs "Z:\MyList.xlsx"
  xlApp.Quit
End Sub

After that you can use the macro below, after you've cusotmized the path "Z:\MyList.xlsx"

Note that the lists are imported only once per session, means if you make a change in MyList.xlsx you have to close and reopen word. This speed up the macro if you call it more then one time in Word.

Andreas.

Sub Listsvalidate()
  Static MyList
  Dim r As Range
  Dim i As Long
  Dim xlApp As Object 'Excel.Application
  Dim xlWb As Object 'Excel.Workbook

  If IsEmpty(MyList) Then
    Set xlApp = CreateObject("Excel.Application")
    Set xlWb = xlApp.Workbooks.Open("Z:\MyList.xlsx", False, True)
    MyList = xlWb.ActiveSheet.Range("A1").CurrentRegion
    xlWb.Close
    xlApp.Quit
  End If

  For i = 1 To UBound(MyList)
    Set r = ActiveDocument.Range
    With r.Find
      .Text = MyList(i, 1)
      .MatchCase = True
      .MatchWholeWord = True
      .Replacement.Text = MyList(i, 2)
      .Execute Replace:=wdReplaceAll
    End With
  Next
End Sub
