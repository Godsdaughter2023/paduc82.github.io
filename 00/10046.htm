Private Sub email_Click()
    Dim appOutLook As Outlook.Application
    Dim MailOutLook As Outlook.MailItem
    Dim strpath As String
    Dim strfile As String
    Dim varName As Variant
    Dim varCC As Variant
    Dim varSubject As Variant
    Dim varcontact As Variant
    Dim varGroup As Variant
    Dim varBody As Variant
    
    strpath = "PATH NAME TO EXPORT TO"
    strfile = Format(me.source of file name) & ".PDF"
    varName = Me.email source
    varCC = Me.mail to be cc'd
    varSubject = "subject here"
    varcontact = Me.name to send to
    varGroup = Me.name of group booking
    varBody = "<P><font face=calibri color=Black style= font-size:14.5px> Dear " & varcontact & " </p<p>rest of content here</p>"

        Set MailOutLook = appOutLook.CreateItem(olMailItem)

        With MailOutLook
            .BodyFormat = olFormatRichText
            .To = varName
            .cc = varCC
            
            .Subject = varSubject
            .HTMLBody = varBody
            .Attachments.Add ("file path and name here if more than 1 attachment required")
            .Attachments.Add (strpath & strfile)
            .Display
         End With
         
End Sub
