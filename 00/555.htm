Okay,

data starts in row 2 (in row 1 are headers)

 

Sub abcd2()
Const FirstSht As String = "xyz"    '<<< source sheet name
Dim r As Long, t As Long
r = Sheets(FirstSht).Cells(Rows.Count, "B").End(xlUp).Row

t = 2
For Each rr In Sheets(FirstSht).Range("B2:B" & r).SpecialCells(xlCellTypeVisible)
Sheets(t).Range("H3").Value = rr.Value
Sheets(t).Range("H8").Value = rr.Offset(, 1).Value
If t = 21 Then Exit Sub
t = t + 1
Next
End Sub
