You will see that the top and left of the image is set to the top and left of the destination cell PLUS 1 POINT. The plus 1 point is to ensure that the TopLeftCell of the image is in the correct cell. Positioning is done to 2 decimal places and it is possible for the system to make a correction that will make the top and/or left marginally in the adjacent cells and then if at a later time you want to enhance the code and the TopLeftCell is required for the VBA code then it could be the wrong cell. Incrementing the top and left by one point is not noticeable to the user and it ensures the TopLeftCell for the image is always correct.

Following is the code that is in the uploaded workbook. (for the benefit of others who might see this thread.

Sub InsertScreenShot()   'Insert picture at selected location
    Dim strImagePath As String
    Dim strPrompt As String
    Dim strTitle As String
    Dim rngImageDestin As Range
    Dim wsForImage As Worksheet
    Dim dblLeft As Double
    Dim dblTop As Double
    Dim strFullImageName As String  'Includes the full path and name of the image file
    Dim shpImage As Shape
   
   
   
    'Following line: Edit if required to your path to Screen Shots
    'Also edit .JPG if required to your image type
    strImagePath = Environ("USERPROFILE") & "\Documents\Snipping Tool\"     'Edit as required
   
    'Alternative example to previous line without using Environ function
    'strImagePath = "C:\Users\Ossie\Documents\Snipping Tool\"
   
    strPrompt = "Select a single cell in column F where image is to be inserted." & vbCrLf & _
                "or Cancel to terminate processing."
   
    strTitle = "Select Image Destination"
   
    Do
        On Error Resume Next
        'Next line: Suppress message if User Clicks OK without selecting a cell
        'and InputBox remains displayed for the User to Select or Cancel.
        Application.DisplayAlerts = False
       
        Set rngImageDestin = Nothing    'Prevent retaining previous selection if code loops due to incorrect selection.
       
        Set rngImageDestin = Application.InputBox(Prompt:=strPrompt, Title:=strTitle, Type:=8)
       
        Err.Clear
       
        If Not rngImageDestin Is Nothing Then
            If rngImageDestin.Cells.Count = 1 Then
                If rngImageDestin.EntireColumn.Address = "$F:$F" Then
                    With rngImageDestin
                        'Following 2 lines in case User changed worksheets to select Image Destination
                        Set wsForImage = rngImageDestin.Parent
                        wsForImage.Select
                       
                        dblLeft = .Left + 1 'Plus 1 ensures that position will not overlap to cell on left
                        dblTop = .Top + 1   'Plus 1 ensures that position will not overlap to cell above
                    End With
                    Exit Do
                Else
                    strPrompt = "Select a single cell in column F ONLY."
                End If
                   
            Else
                strPrompt = "Select ONLY a single in column F where image is to be inserted." & vbCrLf & _
                            "or Cancel to terminate processing."
            End If
        Else
            MsgBox "User cancelled without selecting Image destination." & vbCrLf & _
                    "Processing terminated."
                   
            GoTo EndProcessing
        End If
    Loop
   
   
    Application.DisplayAlerts = True
   
    With Application.FileDialog(msoFileDialogOpen) '
        .Filters.Clear
        .Filters.Add "Screen Shot Images", "*.jpg"      'Edit jpg if not jpg files
        .InitialFileName = strImagePath
        .Title = "Select File"
        If .Show Then
            strFullImageName = .SelectedItems(1)
        Else
            MsgBox "No file selected." & vbCrLf & _
                    "Procesing terminated.", vbCritical
            GoTo EndProcessing
        End If
    End With
   
    'Setting Width and Height to -1 retains picture current dimensions
    'and locks the Aspect Ratio (eg. Locks Ratio of Height to Width)
    'Image is saved with the workbook (NOT as a link to the image)
    Set shpImage = wsForImage.Shapes.AddPicture(Filename:=strFullImageName, _
                                LinkToFile:=msoFalse, _
                                SaveWithDocument:=msoCTrue, _
                                Left:=dblLeft, _
                                Top:=dblTop, _
                                Width:=-1, _
                                Height:=-1)
                               
EndProcessing:
    Application.DisplayAlerts = True    'Ensure alerts turned on again
   
End Sub

bernard liengme say:

In the screenshot columns A and B hold the original data while D and E hold the data for the chart.

The latter may be generated 'manly' or by the VBA sub shown below

best wishes

Sub MakeStep()
 ' Let original data begin in A1
 ' Let generated data begin in D1
 InRow = 1: InTime = "A": InAcc = "B"
 OutRow = 1: OutTime = "D": OutAcc = "E"
 Cells(OutRow, OutTime) = Cells(InRow, InTime)
 Cells(OutRow, OutAcc) = Cells(InRow, InAcc)
 LastRow = Cells(Rows.Count, InTime).End(xlUp).Row
 k = 2
 For j = InRow + 1 To LastRow
   Cells(k, OutTime) = Cells(j, InTime)
   Cells(k, OutAcc) = Cells(j, InAcc)
   k = k + 1
   Cells(k, OutTime) = Cells(j + 1, InTime)
   Cells(k, OutAcc) = Cells(j, InAcc)
   k = k + 1
 Next j
 Cells(k - 1, OutTime).Clear
 Cells(k - 1, OutAcc).Clear
 
 'Make Chart
 Range(OutTime & OutRow).Select
 ActiveSheet.Shapes.AddChart2(240, xlXYScatterLinesNoMarkers).Select
 ActiveChart.SetSourceData Source:=Range("Sheet2!$D$1:$E$14")
End Sub
