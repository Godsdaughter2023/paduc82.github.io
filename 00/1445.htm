Sub colour()

Dim r As Range
Dim lastrow As Long
Dim lastcol As Long


lastrow = Cells(Cells.Rows.Count, "a").End(xlUp).Rows
 lastcol = Cells(Cells.Columns.Count).End(xlToLeft).Column


For Each r In Range("a2:a" & lastrow)
If r.Offset(0, 8).Value = Range("i1").Value Then

r.Resize(, r.Columns.Count + lastcol).Interior.ColorIndex = 8
Else
r.Resize(, r.Columns.Count + lastcol).Interior.Pattern = xlNone



End If

Next r


Range("a1").Select


End Sub
