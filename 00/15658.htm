   Dim SW As Long
   Dim SH As Long
   Dim SldCount As Long
   Dim imageCNT As Long
   SldCount = ActivePresentation.Slides.Count
   Call killem
   SW = ActivePresentation.PageSetup.SlideWidth
   SH = ActivePresentation.PageSetup.SlideHeight
   Call ActivePresentation.SaveCopyAs(Environ("USERPROFILE") & "\Desktop\Slides", ppSaveAsJPG)
   For L = 1 To ActivePresentation.Slides.Count - 1
      Select Case L
      Case Is < SldCount - 2
         imageCNT = 3
      Case Is < SldCount - 1
         imageCNT = 2
      Case Else
         imageCNT = 1
      End Select
      For c = 1 To imageCNT
         With ActivePresentation.Slides(L).Shapes.AddPicture _
         (Environ("USERPROFILE") & "\Desktop\Slides\Slide" & L + (c) & ".jpg", msoFalse, msoTrue, 10, 10)
            .Left = 0 + SW / 4 * (c - 1)
            .Top = SH - SH / 4
            .Width = SW / 4
            .Height = SH / 4
            .Tags.Add "TYPE", "DELETE"
         End With
      Next c
   Next L
End Sub



Sub killem()
Dim osld As Slide
Dim L As Long
For Each osld In ActivePresentation.Slides
For L = osld.Shapes.Count To 1 Step -1
If osld.Shapes(L).Tags("TYPE") = "DELETE" Then osld.Shapes(L).Delete
Next L
Next osld
End Sub
