Sub Bubbles()
  Const Size = 100
  Dim Ws As Worksheet
  Dim Sh As Shape
  Dim Base As Range
  Dim Pool As Collection
  Dim t As Double, l As Double, w As Double, h As Double, A As Double, R As Double, b As Double
  Dim i As Long
  Dim s As Double, n As Double
  Dim p
  Dim Step As MSForms.ScrollBar
  Dim Delay As MSForms.ScrollBar
  Dim Count As MSForms.ScrollBar

  Randomize
Restart:
  Set Pool = New Collection
  Set Ws = ActiveSheet
  For Each Sh In Ws.Shapes
    If Sh.Type <> msoOLEControlObject Then Sh.Delete
  Next

  Set Step = Ws.OLEObjects(1).Object
  Set Delay = Ws.OLEObjects(2).Object
  Delay.Min = 0
  Set Count = Ws.OLEObjects(3).Object

  Set Base = ActiveWindow.VisibleRange
  Set Base = Base.Resize(Base.Rows.Count - 3, Base.Columns.Count - 3).Offset(1, 1)

  With Base
    R = .Left + .Width
    b = .Top + .Height
    For i = 1 To Count.Value
      l = Random(R - Size * 2, .Left + Size, False)
      t = Random(b - Size * 2, .Top + Size, False)
      w = Random(Size, Size / 4, False)
      h = Random(Size, Size / 4, False)
      Set Sh = Ws.Shapes.AddShape(msoShapeOval, l, t, w, h)
      Sh.Rotation = Random(360, 0)
      Pool.Add Sh
    Next
    l = .Left
    t = .Top
    i = Count.Value

    On Error GoTo Errorhandler
    Application.EnableCancelKey = xlErrorHandler
    Do
      If i <> Count.Value Then GoTo Restart
      For Each Sh In Pool
        With Sh
Again:
          p = PointOnCircle(Step.Value, .Rotation, .Left, .Top)
          If p(0) <= l Then
            A = Normalize(.Rotation)
            If A < 90 Then
              .Rotation = -A
            Else
              .Rotation = 180 - A
            End If
            GoTo Again
          End If
          If p(0) + .Width >= R Then
            A = Normalize(.Rotation)
            If A > 90 Then
              .Rotation = -A
            Else
              .Rotation = 180 - A
            End If
            GoTo Again
          End If
          If p(1) <= t Then
            .Rotation = (180 - Normalize(.Rotation))
            GoTo Again
          End If
          If p(1) + .Height >= b Then
            .Rotation = -Normalize(.Rotation)
            GoTo Again
          End If
          .Left = p(0)
          .Top = p(1)
        End With
      Next
      n = MicroTimer
      Do
        DoEvents
      Loop Until MicroTimer - n > Delay.Value / 1000
    Loop
  End With

Errorhandler:
End Sub

Private Function MicroTimer() As Double
  'return containing seconds uses Windows API calls to the high resolution timer
  Dim Ticks As Currency
  Static Frequency As Currency
  MicroTimer = 0
  ' get frequency
  If Frequency = 0 Then
    QueryPerformanceFrequency Frequency
    If Frequency = 0 Then Exit Function
  End If
  ' get ticks
  QueryPerformanceCounter Ticks
  ' calc seconds
  MicroTimer = Ticks / Frequency
End Function

Private Function Random(Optional ByVal Hi = 0, Optional ByVal Lo = 0, _
    Optional AsInteger As Boolean = True)
  'Generates a random number in the range Hi..Lo
  If AsInteger Then
    Random = Int((Hi - Lo + 1) * Rnd + Lo)
  Else
    Random = (Hi - Lo + 1) * Rnd + Lo
  End If
End Function

Private Function Normalize(ByVal A As Double) As Double
  Do While A > 180
    A = A - 180
  Loop
  Do While A < -180
    A = A + 180
  Loop
  Normalize = A
End Function
