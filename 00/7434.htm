Sub ExpandRange()
  Dim X As Long, Y As Long, Z As Long, Num1 As Long, Num2 As Long, OutStr As String, Prefix As String, Ranges() As String, Parts() As String
  Dim Cell As Range
  For Each Cell In Selection
    OutStr = ""
    Ranges = Split(Replace(Replace(Cell.Value, " ", ""), "-", ""), ",")
    For X = 0 To UBound(Ranges)
      Prefix = ""
      For Y = 1 To Len(Ranges(X))
        If IsNumeric(Mid(Ranges(X), Y, 1)) Then
          Prefix = Left(Ranges(X), Y - 1)
          If Len(Prefix) Then
            Parts = Split(Ranges(X), Prefix)
            Select Case UBound(Parts)
              Case 1
                OutStr = OutStr & " " & Ranges(X)
                Exit For
              Case 2
                Num1 = Parts(1)
                Num2 = Parts(2)
                If Len(Num1) > 0 And Not Num1 Like "*[!0-9]*" And Len(Num2) > 0 And Not Num2 Like "*[!0-9]*" Then
                  For Z = 0 To Num2 - Num1
                    OutStr = OutStr & " " & Prefix & (Num1 + Z)
                  Next
                End If
                Exit For
            End Select
          End If
        End If
      Next
    Next
    If Len(OutStr) Then
      If InStr(Trim(OutStr), " ") Then Cell.Value = Trim(OutStr)
      Cell.TextToColumns Cell, xlDelimited, xlTextQualifierNone, True, False, False, False, True
    End If
  Next
End Sub

And, if you want a straight macro, you might want to consider this one...

Sub ExpandRange()
  Dim X As Long, Z As Long, Cell As Range, CellValue As String, Prefix As String, Num1 As Long, Num2 As Long, Parts() As String
  For Each Cell In Selection
    CellValue = Replace(Cell.Value, "-", "")
    If Len(CellValue) Then
      For X = 1 To Len(CellValue)
        If IsNumeric(Mid(CellValue, X, 1)) Then
          Prefix = Left(CellValue, X - 1)
          If Len(Prefix) Then
            Parts = Split(CellValue, Prefix)
            If UBound(Parts) = 2 Then
              Num1 = Parts(1)
              Num2 = Parts(2)
              If Len(Num1) > 0 And Not Num1 Like "*[!0-9]*" And Len(Num2) > 0 And Not Num2 Like "*[!0-9]*" Then
                For Z = 0 To Num2 - Num1
                  Cell.Offset(0, Z).Value = Prefix & (Num1 + Z)
                Next
              End If
            End If
            Exit For
          End If
        End If
      Next
    End If
  Next
End Sub

This macro will only process correctly stated ranges (case sensitive prefixes are the same and the first number is smaller than the second number)... any incorrectly formed text will be left as is. This macro will over-write the cell with the original text (as you asked about in your initial message) and continue across from there.

Back at your worksheet, select all the cells that you want to process, then press Alt+F8 to bring up the macro dialog box, select ExpandRange and Run it.

john mcghie say:

Hi Michael:

Word is not a design tool. But there is no design tool that normal office knowledge workers know how to use. I have produced major documents direct to press from Word: it's more capable than it looks, if you know it well.

I think you may have misunderstood my explanation of headers and footers.  A left page footer will appear on every left page to the end of the section, or the end of the document. As will a right one.

If your document has only one header and footer, the footer will indeed run right through the document and appear on every page. The running header or running footer is the only way to achieve that.

If you want a special header on the first page, you need to enable "Different first page".  If you do, the document will have two footers and you will need to copy one into the other (or cross-reference them).

If you want alternating headers and footers suitable for double-sided impression, you will need to enable "different odd and even".  If you do, the document then has three footers: copy one into the other two (then remember to move the page number so it appears on the outside margin).

To create a custom cover page, simply add a section break on the first page of the document. A Section acts as a document within a document.  This will fence off the headers and footers in the rest of the document.

Then hit three page breaks before the new section break (this makes it possible for you to see and edit the three headers and three footers in the first section). Go into each of the three headers and footers and disable the "Same as Previous" command. If you don't disable "same as previous", any changes you make to the headers or footers in the first section will copy throughout the document.

After unlinking the Same As Previous you can then delete the headers and footers of the first section.  Then delete the three page breaks you added so you could see the headers and footers.  The headers and footers will remain in the document, you just won't be able to see them. I tell you this so you will understand if you get some strange effects from "same as previous" later.  If you don't unlink them, they will appear in the second section as "same as previous" even though they do not appear in the first section.

I do serve PC clients from my Mac.  But I do it by running Parallels, Windows 8.1, and Word 2013.  Word for Mac 2011 is four versions behind: it is no longer suited for commercial service.
