macropod say:

You could use a macro like the following. Simply select the destination cells in Word, then the top-left source cell in Excel and run the macro. All the inserted data will respect your table cell formatting.

Sub AddTableData()
' Note: A reference to the Microsoft Excel # Object Library
' is required, set via Tools|References in the Word VBE.
Dim r As Long, c As Long, x As Long, y As Long
Dim xlApp As Excel.Application, TblCell As Cell
Set xlApp = GetObject(, "Excel.Application")
If Selection.Information(wdWithInTable) = True Then
  r = Selection.Cells(1).RowIndex
  c = Selection.Cells(1).ColumnIndex
  For Each TblCell In Selection.Cells
    With TblCell
      x = .RowIndex - r
      y = .ColumnIndex - c
      .Range.Text = xlApp.ActiveCell.Offset(x, y).Text
    End With
  Next
End If
End Sub

Note: A reference to the Microsoft Excel # Object Library is required, set via Tools|References in the Word VBE.
