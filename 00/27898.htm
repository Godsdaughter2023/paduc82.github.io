jay freedman say:

The brute force method is unnecessary. A quicker way to find the paragraph number is shown in http://www.word.mvps.org/FAQs/MacrosVBA/GetIndexNoOfPara.htm.

This macro should do what you need, with a few caveats: It assumes the list of words is in column A of the first sheet in the current workbook. The line numbers it finds start at 1 at the top of each page in the Word document. An empty paragraph mark (a "blank line") counts as both a line and a paragraph. If you've restarted the page numbers, the wdActiveEndAdjustedPageNumber parameter will cause the macro to get the modified number (i.e., what a {PAGE} field would show); if you want the absolute page number, replace that parameter with wdActiveEndPageNumber.

Sub LocateWords()
    Dim shtIn As Worksheet
    Dim shtOut As Worksheet
    Dim oWord As Word.Application
    Dim WordWasNotRunning As Boolean
    Dim oDoc As Word.Document
    Dim oRange As Word.Range
    Dim LastRow As Long     ' last row with data in shtIn
    Dim lgCurrRow As Long   ' current row in shtIn
    Dim lgOutputRow As Long ' current row in shtOut
    Dim lgPage As Long
    Dim lgLine As Long
    Dim lgPara As Long
    
    On Error Resume Next

    Set oWord = GetObject(, "Word.Application")
    If Err Then
        Set oWord = New Word.Application
        WordWasNotRunning = True
    End If
    
    On Error GoTo Err_Handler
    
    oWord.Visible = True
    oWord.Activate
    Set oDoc = oWord.Documents.Open( _
        "<path>\<filename>.docx")       ' <= modify as needed
    
    Set shtIn = ThisWorkbook.Worksheets(1)
    If ThisWorkbook.Worksheets.Count < 2 Then
        ThisWorkbook.Worksheets.Add After:=shtIn
    End If
    Set shtOut = ThisWorkbook.Worksheets(2)
    
    LastRow = shtIn.UsedRange.Rows( _
        shtIn.UsedRange.Rows.Count).Row
    
    For lgCurrRow = 2 To LastRow
        Set oRange = oDoc.Range
        With oRange.Find
            .Text = shtIn.Cells(lgCurrRow, 1).Text
            .MatchCase = False
            .MatchWholeWord = True
            While oRange.Find.Execute = True
                oRange.Select
                lgPage = oWord.Selection.Information( _
                    wdActiveEndAdjustedPageNumber)
                lgLine = oWord.Selection.Information( _
                    wdFirstCharacterLineNumber)
                lgPara = oDoc.Range(0, _
                    oWord.Selection.Paragraphs(1).Range.End) _
                    .Paragraphs.Count
                
                lgOutputRow = lgOutputRow + 1
                shtOut.Cells(lgOutputRow, 1).Value = .Text
                shtOut.Cells(lgOutputRow, 2).Value = lgPage
                shtOut.Cells(lgOutputRow, 3).Value = lgLine
                shtOut.Cells(lgOutputRow, 4).Value = lgPara
                
                oRange.Collapse wdCollapseEnd
            Wend
        End With
    Next lgCurrRow
    
    If WordWasNotRunning Then
        oWord.Quit
    End If
    
    'Make sure you release object references.
    
    Set oWord = Nothing
    Set oDoc = Nothing
    
    'quit
    Exit Sub
    
Err_Handler:
    MsgBox "Word caused a problem. " & Err.Description, _
        vbCritical, "Error: " & Err.Number
    If WordWasNotRunning Then
        oWord.Quit
    End If
End Sub
