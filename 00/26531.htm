danielco say:

Have a look at the following file :

http://www.filedropper.com/customdatalabels

Ask for what is not clear for you. Mainly, there is a macro which is triggered when the dropdown llist in D2 is changed : 

Private Sub Worksheet_Change(ByVal Target As Range)

Dim c1, c2, c3, i As Integer, States, x As Integer

If Target.Address <> "$D$2" Then Exit Sub

States = Array("Mass", "VT", "NH")

c1 = Array(255, 0, 255)

c2 = Array(0, 255, 255)

c3 = Array(255, 0, 0)

x = Application.Match([D2], States, 0) - 1

[D2].Interior.Color = RGB(c1(x), c2(x), c3(x))

With ActiveSheet.ChartObjects(1).Chart.SeriesCollection(1)

    .DataLabels.Delete

    For i = 6 To Range("A65536").End(xlUp).Row

        If Cells(i, 1) = [D2] Then

            .Points(i - 5).ApplyDataLabels

            .Points(i - 5).DataLabel.Format.Fill.ForeColor.RGB = RGB(c1(x), c2(x), c3(x))

            .Points(i - 5).DataLabel.Text = Cells(i, 1)

        End If

    Next i

End With

End Sub

ENTRY.Shapes("btn_AnyNum_pls").TextFrame.Characters.Text = ChrW(&25BA)

After another 10mins of hunting, I found that ChrW code needs to have an H for Hexidecimal after the &

ENTRY.Shapes("btn_AnyNum_pls").TextFrame.Characters.Text = ChrW(&H25BA)

Now works as required
