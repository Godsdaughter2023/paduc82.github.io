    and stick it in another column in the same row

Option Explicit

Sub Demo()
    Dim fso As Object 'FileSystemObject
    Dim fldStart As Object 'Folder
    Dim fld As Object 'Folder
    Dim fl As Object 'File
    Dim Mask As String
    
Application.ScreenUpdating = False
Dim newWS As Worksheet
Set newWS = Sheets.Add(before:=Sheets(1))
    
    Set fso = CreateObject("scripting.FileSystemObject") ' late binding

L = myWB.Sheets(1).Cells(Rows.Count, "A").End(xlUp).Row
t = 1

Set WB = ActiveWorkbook
WB.Sheets(1).UsedRange.Copy newWS.Cells(t, 2)

Next

Sub ListFolders(fldStart As Object, Mask As String)
    Dim fld As Object 'Folder
    For Each fld In fldStart.SubFolders
        'Debug.Print fld.Path & "\"
        ListFiles fld, Mask
        ListFolders fld, Mask
    Next
End Sub
 

Sub ListFiles(fld As Object, Mask As String)
Dim t As Long
    Dim fl As Object 'File
    For Each fl In fld.Files
        If fl.Name Like Mask Then
        t = Sheets(1).Cells(Rows.Count, "A").End(xlUp).Row + 1
            'Debug.Print fld.Path & "\" & fl.Name
            If Sheets(1).Cells(1, 1) = "" Then
            Sheets(1).Cells(1, 1) = fld.Path & "\" & fl.Name
            Else
            Sheets(1).Cells(t, 1) = fld.Path & "\" & fl.Name
            End If
        End If
    Next
End Sub

Can't access old emails in inbox

My inbox only seems to go back a few years, even though I pay the annual fee for upgraded storage.  I can't locate emails from deceased family members, most importantly my mother.  Has Outlook auto deleted or archived older messages? I've had this account since approx. 2006. Thanks. 

diane poremsky mvp say:
Hi Susan,
Do you know when you last saw the messages? What are the oldest messages you have in your mailbox?

If the messages were there less than 30 days ago, check the Deleted items folder for the missing messages as well as the recovery folder linked at the top of the Deleted Items folder.

Microsoft only purges the mailbox if you don't log in on a regular basis - it used to be 270 days and is now 1 year. If you have an active office 365 subscription, there is no log in requirement.

Common reasons old mail is missing: you used a POP3 client to read your email and it deleted the mail from the server as it downloaded it. Your account was hacked and the hacker deleted the mail. Or you deleted a long time ago.
