Sub FromCSVToExcel()
Set rng = Range("Raw Data")

Dim r As Long, c As Long, Lr As Long, Lc As Long

r = rng.Row
c = rng.Column
Lr = rng.Rows.Count

Lc = rng.Columns.Count + c - 1
Dim L As Long
Dim myWB As Workbook, WB As Workbook
Set myWB = ThisWorkbook
Dim ShName
ShName = ActiveSheet.Name
Application.ScreenUpdating = False
Range(Cells(r + 1, c), Cells(Lr, Lc)).ClearContents
Set WB = Workbooks.Open("C:\Folder CSV\my File.csv") '<<< change Path and file Name as needed
ActiveSheet.UsedRange.Copy
myWB.Sheets(ShName).Cells(rng.Row + 1, rng.Column).PasteSpecial xlValues
Application.CutCopyMode = False

ActiveWorkbook.Close False
L = myWB.Sheets(ShName).Cells(Rows.Count, c).End(xlUp).Row
Dim v
Dim x As Long, i As Long
For i = r + 1 To L
v = Split(Cells(i, c), ",")
For x = 0 To UBound(v)
Cells(i, x + c) = v(x)
Next x
Next i
myWB.Save
rng.EntireColumn.AutoFit
Application.ScreenUpdating = True
End Sub

It calculates in ~0.03 of a second but that timing will depend on how fast your PC is. It's an ARRAY formula so see below on how to enter it. It will probably line wrap so paste into the formula bar.

 

 

=COUNT(1/FREQUENCY(IF($A$1:$A$50000=D2,IF($B$1:$B$50000<>"",MATCH($B$1:$B$50000,$B$1:$B$50000,0))),ROW($B$1:$B$50000)-ROW($B$1)+1))

 

 

This is an array formula which must be entered by pressing CTRL+Shift+Enter
and not just Enter. If you do it correctly then Excel will put curly brackets
around the formula {}. You can't type these yourself. If you edit the formula
you must enter it again with CTRL+Shift+Enter.
