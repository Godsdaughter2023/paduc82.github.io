hansv mvp say:

When I run the following macro, linkText is correctly set to the text between the two instances of $SW Link$ (i.e. the actual text represented by (*), not something with *:

 

Sub Testing()
    Dim linkText As String
    Dim notFound As Boolean
    Selection.HomeKey Unit:=wdStory
    With Selection.Find
        .ClearFormatting
        .Replacement.ClearFormatting
        .Text = "($SW Link$)(*)($SW Link$)"
        .Replacement.Text = "\2"
        .Forward = True
        .Wrap = wdFindContinue
        .Format = False
        .MatchCase = True
        .MatchWholeWord = False
        .MatchAllWordForms = False
        .MatchSoundsLike = False
        .MatchWildcards = True
        If .Execute = True Then
            linkText = Replace(Selection.Text, "$SW Link$", "")
            MsgBox "Text found. " + linkText
        Else
            MsgBox "Text not found."
            notFound = True
        End If
    End With
End Sub
