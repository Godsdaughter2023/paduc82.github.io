Function NewSkills(rngOld As Range, rngNew As Range) As String
    Dim NS As Variant
    Dim i As Integer

    NS = Split(rngNew.Value, ",")
    For i = LBound(NS) To UBound(NS)
        If InStr(1, rngOld.Value, NS(i)) = 0 Then
            NewSkills = NewSkills & NS(i) & ", "
        End If
    Next i

If you have never heard of mailmerge events, I would suggest that you use either method d suggested by Peter or the method suggested by Graham.

Dim i As Long

        With Selection.Columns(1).Cells(i).Range

Dim i As Long
