Copy hyperlink from a hyperlinked word to all other occurrances of - Microsoft Community

audiovideoer say:

Copy hyperlink from a hyperlinked word to all other occurrances of that word in a document.

I have a list of hyperlinked words/numbers and would like to apply the hyperlinks to all occurrances of these words/numbers in a document. Finding and replacing them one by one would be very time consuming. It would help if I could keep the list of hyperlinked words in a separate Word document. The macro should go to that document and then copy a hyperlink to the related word in the working document.

pesach shelnitz say:

I've revised the macro according to your request. Open the document or documents into which you want the hyperlinks copied. Then run this revised version of the macro from the document containing the links to be copied.

Sub HyperlinkAll()
    Const arraySize = 10
    Const Error_StringNotValid = 5625
    Dim searchString As String
    Dim textArray As Variant
    Dim addressArray As Variant
    Dim i As Integer
    Dim j As Integer
    Dim total As Integer
   
    ReDim textArray(arraySize) As String
    ReDim addressArray(arraySize) As String

    searchString = InputBox( _
        "Type the hyperlink search string.", _
        "Hyperlink Search String")
    i = 0
    Selection.HomeKey wdStory
    With Selection.Find
        .Text = searchString
        .Forward = True
        .Wrap = wdFindStop
        On Error Resume Next
        .Execute
        If Err.Number = Error_StringNotValid Then
            MsgBox "The search string is not valid.", _
              vbCritical Or vbOKOnly, "Invalid String"
            Exit Sub
        End If
        On Error GoTo 0
        Selection.HomeKey wdStory
        Do While .Execute = True
            If Selection.Hyperlinks.Count > 0 Then
                i = i + 1
                If i > arraySize Then
                    MsgBox ("The array size is too small " & _
                        "for the number of hyperlinks found." & _
                        vbCrLf + "Increase the value of " & _
                        "ArraySize and try again.")
                    Exit Sub
                End If
                textArray(i) = Selection.Range.Text
                addressArray(i) = Selection.Hyperlinks(1).Address
            End If
            Selection.Collapse Direction:=wdCollapseEnd
        Loop
        total = i
    End With
    For j = 1 To Documents.Count
        Documents(j).Activate
        For i = 1 To total
            Selection.HomeKey wdStory
            With Selection.Find
                .Text = textArray(i)
                .Forward = True
                .Wrap = wdFindStop
                Do While .Execute = True
                    If Selection.Hyperlinks.Count = 0 Then
                        Selection.Hyperlinks.Add Anchor:=Selection.Range, _
                            Address:=addressArray(i)
                    End If
                    Selection.Collapse Direction:=wdCollapseEnd
                Loop
            End With
        Next
    Next
End Sub
