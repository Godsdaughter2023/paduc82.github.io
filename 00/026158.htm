andreas killer say:
Please never use SELECT, SELECTION, ACTIVECELL, it is slow and error prone. Always refer to the objects directly.

Use "To 3" instead of "To 2" if you don't want to have an empty row below the heading.

Andreas.

Sub Test()
  Dim i As Long
  Dim Where As Range
 
  'From the last used cell in column C up to the top
  For i = Range("C" & Rows.Count).End(xlUp).Row To 3 Step -1
    'Skip hidden and empty cells
    With Range("C" & i)
      If .Hidden Then GoTo Skip
      If IsEmpty(.Value) Then GoTo Skip
    End If
    With Range("C" & i - 1)
      If .Hidden Then GoTo Skip
      If IsEmpty(.Value) Then GoTo Skip
    End If
    'If the content is different
    If Range("C" & i) <> Range("C" & i - 1) Then
      'Insert a row
      Range("C" & i).EntireRow.Insert
    End If
Skip:
  Next
  'Find all empty cells
  Set Where = Range("C1", Range("C" & Rows.Count).End(xlUp)).SpecialCells(xlCellTypeBlanks)
  Where.Value = "whatever"
End Sub
