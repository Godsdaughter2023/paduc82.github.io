- Open the Control Panel, and click/tap on the User Accounts icon.
- Click/tap on the Manage another account link.
- If prompted by UAC, click/tap on Yes.
- Click/tap on a user you want to delete.
- Click/tap on the Delete the account link.
- Click/tap on Delete Files or Keep Files for what you want to do with the user account's personal files.
- Click/tap on Delete Account to confirm.
- When finished, you can close the Control Panel if you like

graham mayor say:

Try the following instead. You could incorporate the two functions in the main macro, but it saves my time to copy and paste them.

Option Explicit

Sub Macro5()

Dim strFileName As String
Dim shpTbox As Shape
Dim rngTbox As Range
    strFileName = "C:\Users\My Documents\Logo.jpg"
    Set shpTbox = ActiveDocument.Shapes.AddTextbox( _
                  Orientation:=msoTextOrientationHorizontal, _
                  Left:=450, Top:=20, Width:=100, Height:=40)
    Set rngTbox = shpTbox.TextFrame.TextRange
    With rngTbox
        .Bookmarks.Add Name:="bmkTbox"
        .Collapse wdCollapseEnd
    End With
    ImageToBM "bmkTbox", strFileName
End Sub

Private Sub ImageToBM(strBMName As String, strImagePath As String)
'Graham Mayor - www.gmayor.com
Dim oRng As Range
    If Not FileExists(strImagePath) Then GoTo lbl_Exit
    With ActiveDocument
        On Error GoTo lbl_Exit
        Set oRng = .Bookmarks(strBMName).Range
        oRng.InlineShapes.AddPicture _
                Filename:=strImagePath, LinkToFile:=False, _
                SaveWithDocument:=True
        oRng.End = oRng.End + 2
        oRng.Bookmarks.Add strBMName
    End With
lbl_Exit:
    Set oRng = Nothing
    Exit Sub
End Sub

Private Function FileExists(strFullName As String) As Boolean
'Graham Mayor - wwe.gmayor.com
'strFullName is the name with path of the file to check
Dim fso As Object
    Set fso = CreateObject("Scripting.FileSystemObject")
    If fso.FileExists(strFullName) Then
        FileExists = True
    Else
        FileExists = False
    End If
lbl_Exit:
    Set fso = Nothing
    Exit Function
End Function
