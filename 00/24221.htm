jay freedman say:
Sub demo()
    Dim n As String
    Dim rg As Range
   
    n = "Constructed from userform"
   
    ' If the document has more than one section
    ' this will have to be revised.
   
    With ActiveDocument.Sections(1)
        ' turn on the "different first page" option
        .PageSetup.DifferentFirstPageHeaderFooter = True
       
        ' insert the predefined Page X of Y autotext entry
        ' in the first page footer
        Set rg = .Footers(wdHeaderFooterFirstPage).Range
        ActiveDocument.AttachedTemplate.AutoTextEntries("Page X of Y").Insert _
            Where:=rg
        ' insert a tab character at the left margin to center the page number
        Set rg = .Footers(wdHeaderFooterFirstPage).Range
        rg.Collapse wdCollapseStart
        rg.Text = vbTab
        ' update the fields
        .Footers(wdHeaderFooterFirstPage).Range.Fields.Update
       
        ' copy the footer from the first page footer
        ' to the footer for the rest of the pages
        .Footers(wdHeaderFooterPrimary).Range.FormattedText = _
            .Footers(wdHeaderFooterFirstPage).Range.FormattedText
        ' remove the unwanted paragraph mark copied from the first page
        Set rg = .Footers(wdHeaderFooterPrimary).Range
        rg.Collapse wdCollapseEnd
        rg.MoveStart wdCharacter, -1
        rg.Delete
        ' add a tab character and the string "n"
        rg.Text = vbTab & n
        ' update the fields
        .Footers(wdHeaderFooterPrimary).Range.Fields.Update
    End With
   
End Sub

jay freedman say:
If you select some text and call up the built-in Word Count dialog box (from the button in the Proofing group on the Review tab, or by clicking the word count in the status bar), it includes the number of lines in the selection.

You can use this in VBA, without showing the dialog box on screen, by calling its .Execute method. The dialog object will then contain the line count in its .Lines member. Here's sample code:

Sub LineCount()
    Dim lines As Long
    With Dialogs(wdDialogToolsWordCount)
        .Execute
        lines = .Lines
    End With
   
    MsgBox "Selection has " & lines & " lines"
End Sub
