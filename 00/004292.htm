    =IF(OR(COUNTIF(A1:D1,'F")>0,COUNTIF(A1:D1,"I")>0),"F",IF(OR(COUNTIF(A1:D1,'P")>0,COUNTIF(A1:D1,"H")>0),"P",""))

    Hi - thanks for any help you can give me.

    If any of the cells A1 to D1 contain an "F", then return an "F" into E1.

    There are a few ways to go about this. First, you can sync with the OS X Contacts app and look for duplicates manually, which is a pain.

    Second, you can export your contact lists from other services, choose the one you want to sync with, and import them there, clearing duplicates manually as you go and only leaving the one service on.

    Your contacts are backed up in case you're not happy with the results, and you're done. Whole process should take less than 30 minutes, and you don't need to be there for most of it.

    Powerpoint VBA cropping syntax & calculation - Microsoft Community

    kergga say:

    Powerpoint VBA cropping syntax & calculation

    I am trying to create a macro so that I can select two shapes, and crop the shape/image that is further back to the same height/width as the top shape, keeping only the part of the background shape that was directly behind the upper shape. For example, if the top shape is a rectangle with no fill, this allows me to do a homemade 'intersect' and only keep the portion of the background image that was directly under the rectangle.

     

    The code below seems to be working, until I try to apply the values for cropping the background shape. Then I get RTE: The specififed value is out of range. I also tried dividing the Length, Height, Width, Top by 72 in case it was a pixel vs inch issue, but no joy. Any suggestions?

     

     

    Many thanks!

     

     

     

    Sub Resize_Selected_Shapes()

     

    Dim shp As Shape         'used to loop both shapes
    Dim SelShp As Shape    'used to store which one is the back shape to be cropped
    BackShp = 1000            'default to high number, every zorderposition will be less than this

     

    If ActiveWindow.Selection.Type = ppSelectionNone Then
        MsgBox "nothing selected", , "Please try again"
    Else
        If ActiveWindow.Selection.Type <> ppSelectionShapes Then
            MsgBox "Non-shapes are selected (slide, etc)", , "Cannot process selection"
        Else
            If ActiveWindow.Selection.ShapeRange.Count = 1 Then
                MsgBox "Please select two shapes", , "Not enough selections"
            ElseIf ActiveWindow.Selection.ShapeRange.Count > 2 Then
                MsgBox "Please select only two shapes", , "Too many selections"
            Else

                'we have two selected shapes:
                For Each shp In ActiveWindow.Selection.ShapeRange
                   
                    'Make sure that there is no aspect ratio problems
                    shp.LockAspectRatio = msoFalse
                   
                    'figure out the furthest back shape during the loop
                    If shp.ZOrderPosition < BackShp Then
                        'Set this as the back shape
                        Set SelShp = shp
                        BackShp = shp.ZOrderPosition

                        'also need to grab the data just in case the order of processing never triggers the ELSE loop to assign values
                        If SW = 0 Then
                            SW = shp.Width
                            SH = shp.Height
                            SL = shp.Left
                            ST = shp.Top
                        End If
                    Else
                        'set the measures for the front shape
                        SW = shp.Width
                        SH = shp.Height
                        SL = shp.Left
                        ST = shp.Top
                    End If
                Next shp
                   
                    'set the backshape crop
                    SelShp.PictureFormat.CropLeft = SL     '<<-- breaks here, with a SL value of 300 (or 4.3)
                    SelShp.PictureFormat.CropTop = ST
                    SelShp.PictureFormat.CropRight = SW - SL
                    SelShp.PictureFormat.CropBottom = SH - ST

            End If
        End If
    End If

    End Sub
