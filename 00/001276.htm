Sub SplitTable_newWBKs()

'Mar. 27, 2014

Dim myWB As Workbook, newWB As Workbook
Set myWB = ThisWorkbook
Dim ws As Worksheet
Set ws = myWB.Sheets("DATA")   '<<< change name
Dim r1 As Long, r2 As Long, r3 As Long
r1 = ws.Cells(Rows.Count, "A").End(xlUp).Row
Set myWB = ThisWorkbook
Dim myPath
myPath = myWB.Path & "\"
Dim asSht As Worksheet
Application.ScreenUpdating = False
Set asSht = Sheets.Add(before:=Sheets(1))
Range("A1") = "header"
Dim x As Long
For x = 2 To r1
myWB.Sheets(1).Cells(x, "A") = "'" & Split(ws.Cells(x, "A"), ".")(0)
Next
myWB.Sheets(1).Range("A1:A" & r1).Sort Key1:=myWB.Sheets(1).Range("A1"), Order1:=xlAscending, Header:=xlYes
For x = r1 To 3 Step -1
If myWB.Sheets(1).Cells(x, "A") = myWB.Sheets(1).Cells(x - 1, "A") Then myWB.Sheets(1).Cells(x, "A").Delete shift:=xlUp
Next
r2 = myWB.Sheets(1).Cells(Rows.Count, "A").End(xlUp).Row
For x = 2 To r2
Set newWB = Workbooks.Add
With ws
.AutoFilterMode = False
.Range("A1").CurrentRegion.AutoFilter field:=1, Criteria1:=myWB.Sheets(1).Cells(x, "A") & "*"
t = 2
For Each rr In .Range("A2:A" & r1).SpecialCells(xlCellTypeVisible)
myWB.Sheets(1).Cells(t, "B") = "'" & Split(rr, ".")(1)
t = t + 1
Next
.AutoFilterMode = False
End With
r3 = myWB.Sheets(1).Cells(Rows.Count, "B").End(xlUp).Row
Dim nnew As Worksheet
For Each rr In myWB.Sheets(1).Range("B:B").SpecialCells(xlCellTypeConstants) 'qqq
With ws
.Range("A1").CurrentRegion.AutoFilter field:=1, Criteria1:=myWB.Sheets(1).Cells(x, "A") & "." & rr.Value
Set nnew = newWB.Sheets.Add(after:=newWB.Sheets(newWB.Sheets.Count))
nnew.Name = rr.Value
.Range("A1").CurrentRegion.Copy Range("A1")
ActiveSheet.UsedRange.EntireColumn.AutoFit
.AutoFilterMode = False
End With
Next
myWB.Sheets(1).Range("B:B").ClearContents
Application.DisplayAlerts = False
Dim rng As Range
For Each sh In newWB.Sheets
Set rng = sh.Cells.Find(what:="*")
If rng Is Nothing Then sh.Delete
Next
newWB.SaveAs Filename:=myPath & myWB.Sheets(1).Cells(x, "A").Value & ".xlsx"
newWB.Close False
Next
myWB.Sheets(1).Delete
myWB.Save
Application.DisplayAlerts = True
Application.ScreenUpdating = True
MsgBox "done"
End Sub

Because we now want to fill the treeview when the form is loaded and also when an item is selected in the combo box, I split off the routine to fill the treeview:

 

Private Sub FillTreeView()
    Dim dbs As DAO.Database
    Dim rst1 As DAO.Recordset
    Dim rst2 As DAO.Recordset
    Dim rst3 As DAO.Recordset
    Dim nod1 As MSComctlLib.Node
    Dim nod2 As MSComctlLib.Node
    Dim strSQL As String
    Me.TreeView0.Nodes.Clear
    Set dbs = CurrentDb
    strSQL = "SELECT DISTINCT StartDate FROM tblClasses"
    If Not IsNull(Me.cboDate) Then
        strSQL = strSQL & " WHERE StartDate=#" & Format(Me.cboDate, "mm\/dd\/yyyy") & "#"
    End If
    Set rst1 = dbs.OpenRecordset(strSQL, dbOpenForwardOnly)
    Do While Not rst1.EOF
        Set nod1 = Me.TreeView0.Nodes.Add(Key:="D" & rst1!StartDate, _
            Text:="Date " & rst1!StartDate)
        Set rst2 = dbs.OpenRecordset("SELECT ClassID, ClassNumber FROM tblClasses WHERE StartDate=#" & _
            Format(rst1!StartDate, "mm\/dd\/yyyy") & "# ORDER BY ClassID", dbOpenForwardOnly)
        Do While Not rst2.EOF
            Set nod2 = Me.TreeView0.Nodes.Add(Relative:=nod1, Relationship:=tvwChild, _
                Key:="C" & rst2!ClassID, _
                Text:="Class " & rst2!ClassNumber)
            Set rst3 = dbs.OpenRecordset("SELECT StudentID, StudentFullName FROM tblStudents WHERE StudentClassID=" & _
                rst2!ClassID & " ORDER BY StudentFullName", dbOpenForwardOnly)
            Do While Not rst3.EOF
                Me.TreeView0.Nodes.Add Relative:=nod2, Relationship:=tvwChild, _
                    Key:="M" & rst3!StudentID, Text:=rst3!StudentFullName
              rst3.MoveNext
            Loop
            rst3.Close
            rst2.MoveNext
        Loop
        rst2.Close
        rst1.MoveNext
    Loop
    ' Optional - expand treeview
    For Each nod1 In Me.TreeView0.Nodes
        nod1.Expanded = True
    Next nod1
    Set rst3 = Nothing
    Set rst2 = Nothing
    rst1.Close
    Set rst1 = Nothing
    Set dbs = Nothing
End Sub

 

We can then call it where needed:

 

Private Sub cboDate_AfterUpdate()
    FillTreeView
End Sub

Private Sub Form_Load()
    FillTreeView
End Sub

 

where cboDate is the name of the combo box. With the above code, you don't have to use a query for the treeview - the filtering is done within the code.
