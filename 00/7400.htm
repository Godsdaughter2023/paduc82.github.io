jllatham say:

OK, I've revised the code and looked closely at the results and it appears to me to work as you desire.  Give the code below a test run.  Just replace the previous code with the code below.  I've made a couple of changes to improve performance, and set up MaxRows based on the last row used in column C on sheet [B].  I've kept the definitions of the various critical items as Const definitions so you can hopefully easily adapt the code in the future if the layout of either sheet changes.  Here's the new code:

Sub CopyDataFromAtoB()
  Const srcSheetName = "A" ' source sheet name
  Const destSheetName = "B" ' destination sheet name
  Const keyCell = "S4" ' on sheet [A]
  Const srcRange1Address = "P13:P17" ' on sheet [A]
  Const srcRange2Address = "O13:O17" ' on sheet [A]
  Const FirstDestRow = 11 ' start at row 11 on sheet [B]
  Const dest1Start = "J" ' on sheet [B]
  Const dest1End = "N" ' on sheet [B]
  Const dest2Start = "Q" ' on sheet [B]
  Const dest2End = "U" ' on sheet [B]
  Const CheckCol = "C" ' on sheet [B]

  Dim LoopCounter As Integer
  Dim srcSheet As Worksheet
  Dim srcRange1 As Range
  Dim srcRange2 As Range
  Dim destSheet As Worksheet
  Dim destRange As Range
  Dim destRow As Integer

  Dim MaxRows As Long
  Dim cellCount As Integer
  Dim anyCell As Range

  Set destSheet = ThisWorkbook.Worksheets(destSheetName)
  'calculate MaxRows based on last entry in column C of sheet [B]
  MaxRows = destSheet.Range(CheckCol & Rows.Count).End(xlUp).Row
  If MaxRows < FirstDestRow Then
    MsgBox "No data below row " & FirstDestRow - 1, vbOKOnly + vbInformation, _
     "No Data to Process"
    Set destSheet = Nothing
    Exit Sub
  End If
  MaxRows = MaxRows - FirstDestRow + 1 ' # times to go through the loop

  Set srcSheet = ThisWorkbook.Worksheets(srcSheetName)
  'these don't change, set them up outside of the
  'loop for improved performance.
  Set srcRange1 = srcSheet.Range(srcRange1Address)
  Set srcRange2 = srcSheet.Range(srcRange2Address)
  For LoopCounter = 1 To MaxRows ' 1 to # of rows with data in C on [B]
    'put numbers 1 through last row used into A!S4
    srcSheet.Range(keyCell) = LoopCounter
    'determine what row on sheet [B] to put the data
    destRow = srcSheet.Range(keyCell) + FirstDestRow - 1
    'P13:P17 to J#:N#
    Set destRange = destSheet.Range(dest1Start & destRow)
    cellCount = 0
    For Each anyCell In srcRange1
      destRange.Offset(0, cellCount).Value = anyCell.Value
      cellCount = cellCount + 1
    Next
    'O13:O17 to Q#:U#
    Set destRange = destSheet.Range(dest2Start & destRow)
    cellCount = 0
    For Each anyCell In srcRange2
      destRange.Offset(0, cellCount).Value = anyCell.Value
      cellCount = cellCount + 1
    Next
  Next ' end of LoopCounter loop
  'housekeeping
  Set srcRange1 = Nothing
  Set srcRange2 = Nothing
  Set destRange = Nothing
  Set srcSheet = Nothing
  Set destSheet = Nothing
  'inform user that work is complete
  MsgBox "Data Copying Completed.", vbOKOnly, "Task Done"
End Sub

stefan blom say:

Click the letter E; then right-click and try the Continue Numbering command on the context menu.

If that doesn't work, click in the "A" paragraph and press Ctrl+Shift+C (to copy the formatting of that paragraph). Then select the incorrectly numbered paragraphs and press Ctrl+Shift+V to paste the formatting from "A."
