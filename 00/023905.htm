rich007a say:

Ah, I see what you want now, you have ~125 different search terms to find on the sheet....

Taking MikeH's code, I put it in a loop to loop through the range A1:A125 to search the current selection for the contents of A1, then A2, then A3.... to A150.  Try this:

 

Sub Find_All_ManyTimes()
    Dim FindRange As Range, c As Range
    Dim c2 As Range, Cells2Find As Range
    Dim FirstAddress As String

    Set Cells2Find = Range("A1:A125") 'Adjust as required, terms to be found.
    For Each c2 In Cells2Find
        If c2.Value <> "" Then 'Skip blank cells
            With Selection
                Set c = .Find(c2.Value, LookIn:=xlValues, LookAt:=xlWhole)
                If Not c Is Nothing Then
                    FirstAddress = c.Address
                    Do
                        If FindRange Is Nothing Then
                            Set FindRange = c
                        Else
                            Set FindRange = Union(FindRange, c)
                        End If
                        Set c = .FindNext(c)
                    Loop While Not c Is Nothing And c.Address <> FirstAddress
                End If
            End With
            If Not FindRange Is Nothing Then
                FindRange.Interior.ColorIndex = 6
            End If
        End If
    Next c2End Sub

 

Hope that helps.
