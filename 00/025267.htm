HERE - this should work.  ONLY select cells in column B that need to be examined.

Sub FindMismatches()
'the sheet with the data to be examined
'must be selected when you run this macro
'
'SELECT the cells in COLUMN B ONLY that are
'to be examined and then run this macro

   Dim anyBCell As Range
   Dim testB As String
   Dim testC As String
  
  'remove any previous shading
  Application.ScreenUpdating = False
  For Each anyBCell In Selection
    testB = Trim(anyBCell)
    testC = Trim(anyBCell.Offset(0, 1))
    'clear any existing shading
    anyBCell.Interior.ColorIndex = xlNone
    anyBCell.Offset(0, 1).Interior.ColorIndex = xlNone
    If InStr(testC, " ") > 0 Then
      testC = Left(testC, InStr(testC, " ") - 1)
    End If
    If testB <> testC Then
      anyBCell.Interior.ColorIndex = 6
      anyBCell.Offset(0, 1).Interior.ColorIndex = 6
    End If
  Next
End Sub
