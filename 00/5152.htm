  For Each cl In rw.Cells

      temp = temp & cl.Value & ";"
    Else

Hi,

 

Get to the VB Editor by pressing ALT + F11.

Sub WriteToCSV()
Dim FileNumber As Long
Dim temp As String
Dim cl As Range
Dim rw As Range
    FileNumber = FreeFile    '
    'get a new file number
    FileNumber = FreeFile
    ' change path & file name as required
    Open "C:/Temp/TEST.csv" For Output As #FileNumber
    Print #FileNumber, temp

   'change the worksheet index by its real position or Name between quotes, eg Worksheets("Sheet1").
    For Each rw In Worksheets(1).Range("A1").CurrentRegion.Rows
        For Each cl In rw.Cells
            temp = temp & cl.Value & ";"
        Next cl
        Print #FileNumber, temp
        're=initialise string
        temp = ""
    Next rw
    Close #FileNumber
End Sub
