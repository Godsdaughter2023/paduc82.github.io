Sub ConvertOLEObjectsToPicture()
  Dim Item As InlineShape
  For Each Item In ActiveDocument.InlineShapes
    Select Case Item.Type
      Case wdInlineShapeEmbeddedOLEObject, wdInlineShapeLinkedOLEObject
        Item.Select
        With Selection

        End With
    End Select

a. user say:

How about this?

=IF(A2=B2,C2,"")

Or, if you are looking for a match anywhere in the other column:

=IFERROR(VLOOKUP(A2,B:C,2,FALSE),"")

If your excel is old enough that you don't have iferror the second code looks like this:

=if(iserror(vlookup(a2,B2:c2,2,false)),"",vlookup(a2,b2:c2,2,false))

If you have a large amount of data than index match (which you mention) may be less computer intensive.

=iferror(index(b2:c2,match(a2,b2:b2,0),2),"")

Right click your worksheet tab, view code and paste this code in on the right. Close VB editor and your done.



Private Sub Worksheet_Change(ByVal Target As Range)
If Not Intersect(Target, Range("F:F")) Is Nothing And _
IsNumeric(Target.Value) Then
Application.EnableEvents = False
If Target.Value > 1 Then
Target.Offset(1, -5).Resize(Target.Value - 1, 6).Insert
Target.Offset(, -5).Resize(Target.Value, 5).FillDown
End If
End If
Application.EnableEvents = True
End Sub

Private Sub Worksheet_Change(ByVal Target As Range)

 End Sub
