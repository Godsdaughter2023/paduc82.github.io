Firstly, some basic error handling would help.

Secondly, you are missing a bracket in the Documents.Add line

Function EEAF()
On Error GoTo Error_Handler

    Dim objWord As Object
    Dim objmergedoc As Object

On Error Resume Next
       Set objWord = GetObject(, "Word.Application")
        If Err.Number <> 0 Then
             Set objWord = CreateObject("Word.Application")
        End If

        Err.Clear
   
        Set objmergedoc = objWord.Documents.Add("\\Bcp\bcp\DB\Documents\Entitlement\Entitlement.dotm")
        objWord.Visible = True
        objmergedoc.MailMerge.OpenDataSource "G:\db\Employees\" & [Forms]![001_Start]![Matrix] & "\Entitlement.rtf", LinkToSource = True, ConfirmConversions = False 'Fails
        DoEvents
        objmergedoc.MailMerge.ViewMailMergeFieldCodes = False
        objmergedoc.Application.Run ("Entitlement")

Error_Handler_Exit:
    On Error Resume Next
    objmergedoc.Close True 'Save and close the doc
    Set objmergedoc = Nothing
    objWord.Quit    'close word
    Set objWord = Nothing
    Exit Function

Error_Handler:
    MsgBox "MS Access has generated the following error" & vbCrLf & vbCrLf & "Error Number: " & _
    Err.Number & vbCrLf & "Error Source: EEAF" & vbCrLf & "Error Description: " & _
    Err.Description, vbCritical, "An Error has Occured!"
    Resume Error_Handler_Exit
End Function

First, select Tools | Options...

In the View tab, make sure that the 'Worksheet Tabs ' check box is ticked, then click OK.
