andreas killer say:
As a time in Excel is a fraction of a day, you can convert a time with a factor of 24 to a number that represents a hour, and the formula becomes:

=IF(G4*24<12,IF(G4*24<6,0,0.5),1)/24
Means: If less then 6 hours then 0, if less then 12 hours then 0.5 hours otherwise 1 hour.

If you don't like to convert the values you have to use the TIME function (which does the same but in the other parts of the formula):

=IF(G4<TIME(12,0,0),IF(G4<TIME(6,0,0),0,TIME(0,30,0)),TIME(1,0,0))

Option Explicit

Sub AddAppointments()

  Dim myoutlook As Object ' Outlook.Application
  Dim r As Long
  Dim myapt As Object ' Outlook.AppointmentItem

  ' late bound constants
  Const olAppointmentItem = 1
  Const olBusy = 2
  Const olMeeting = 1

  ' Create the Outlook session
  Set myoutlook = CreateObject("Outlook.Application")

  ' Start at row 11
  r = 11

  Do Until Trim$(Cells(r, 1).Value) = ""
    ' Create the AppointmentItem
    Set myapt = myoutlook.CreateItem(olAppointmentItem)
    ' Set the appointment properties
    With myapt
      .Subject = Sheets("Alert").Range("D11")
      .Location = Sheets("Alert").Range("D11")
      .Start = Sheets("Alert").Range("F11")
      .End = Sheets("Alert").Range("G11")
      .Duration = Sheets("Alert").Range("H11")
     .Recipients.Add = Sheets("Alert").Range("C13") ''or   .Recipients.ResolveAll
      .MeetingStatus = olMeeting
      ' not necessary if recipients are email addresses
      ' myapt.Recipients.ResolveAll
      .AllDayEvent = Sheets("Alert").Range("I11")

      ' If Busy Status is not specified, default to 2 (Busy)
      If Len(Trim$(Cells(r, 5).Value)) = 0 Then
        .BusyStatus = olBusy
      Else
        .BusyStatus = Cells(r, 5).Value
      End If

      If Cells(r, 6).Value > 0 Then
        .ReminderSet = True
        .ReminderMinutesBeforeStart = 0
      Else
        .ReminderSet = False
      End If

      .Body = Sheets("Alert").Range("D11")
      .Save
      r = r + 11
      .Send
    End With
  Loop
End Sub
